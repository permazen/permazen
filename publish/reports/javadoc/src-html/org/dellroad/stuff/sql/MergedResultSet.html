038/**
039 * Merges multiple {@link ResultSet}s into a single {@link ResultSet}. The merge is performed in an online fashion
040 * and supports preserving sort order for sorted {@link ResultSet}s.
041 *
042 * <p>
043 * The given {@link ResultSet}s must have the same column geometry. If the {@link ResultSet}s are sorted, then in order
044 * to preserve that sorting the {@link ResultSet}s must be sorted on the same columns, and the sort columns and corresponding
045 * sort orderings must be provided to the appropriate constructor.
046 * </p>
047 *
048 * <p>
049 * This class provides a {@link ResultSet} of type {@link #TYPE_FORWARD_ONLY} and concurrency mode {@link #CONCUR_READ_ONLY}.
050 * Non-row specific information, such as is returned by {@link #getMetaData}, {@link #getStatement}, {@link #getFetchSize},
051 * {@link #getHoldability}, and {@link #findColumn findColumn()} is derived from the first {@link ResultSet} provided.
052 * </p>
053 */
054@SuppressWarnings("deprecation")
055public class MergedResultSet implements ResultSet {

114     * @param sortColumns the names of the columns on which the {@link ResultSet}s are sorted
115     * @param sortOrders the ordering corresponding to {@code sortColumns} (true for ascending); if {@code sortOrders}
116     *  is shorter than {@code sortColumns}, then ascending is assumed for the unspecified columns
117     * @throws IllegalArgumentException if any parameter is null
118     * @throws IllegalArgumentException if {@code resultSets} is empty or contains a null element
119     */
120    public MergedResultSet(List<ResultSet> resultSets, String[] sortColumns, boolean[] sortOrders) {

124    /**
125     * Constructor.
126     *
127     * @param resultSets the {@link ResultSet}s to be merged
128     * @param sortColumns the indicies (1-based) of the columns on which the {@link ResultSet}s are sorted
129     * @param sortOrders the ordering corresponding to {@code sortColumns} (true for ascending); if {@code sortOrders}
130     *  is shorter than {@code sortColumns}, then ascending is assumed for the unspecified columns
131     * @throws IllegalArgumentException if any parameter is null
132     * @throws IllegalArgumentException if {@code resultSets} is empty or contains a null element
133     */
134    public MergedResultSet(List<ResultSet> resultSets, final int[] sortColumns, boolean[] sortOrders) {
135        if (resultSets == null)
136            throw new IllegalArgumentException("null resultSets");
137        if (resultSets.isEmpty())
138            throw new IllegalArgumentException("empty resultSets");
139        for (ResultSet rs : resultSets) {
140            if (rs == null)
141                throw new IllegalArgumentException("null resultSet");
142        }
143        if (sortColumns == null)
144            throw new IllegalArgumentException("null sortColumns");
145        if (sortOrders == null)
146            throw new IllegalArgumentException("null sortOrders");
147        if (sortOrders.length < sortColumns.length) {
148            final boolean[] newSortOrders = new boolean[sortColumns.length];
149            Arrays.fill(newSortOrders, true);
150            System.arraycopy(sortOrders, 0, newSortOrders, 0, Math.min(sortOrders.length, newSortOrders.length));
151            sortOrders = newSortOrders;
152        }
153        this.resultSets = new ArrayList<ResultSet>(resultSets);
154        this.representative = this.resultSets.get(0);
155        final boolean[] sortOrders2 = sortOrders;
156        this.remaining = new PriorityQueue<ResultSet>(this.resultSets.size(), new Comparator<ResultSet>() {
157            @Override
158            @SuppressWarnings("unchecked")
159            public int compare(ResultSet rs1, ResultSet rs2) {
160                for (int i = 0; i < sortColumns.length; i++) {
161                    Object value1;
162                    Object value2;
163                    try {
164                        value1 = rs1.getObject(sortColumns[i]);
165                        value2 = rs2.getObject(sortColumns[i]);
166                    } catch (SQLException e) {
167                        throw new RuntimeException(e);
168                    }
169                    if (value1 == null && value2 != null)
170                        return -1;
171                    if (value1 != null && value2 == null)
172                        return 1;
173                    if (value1 == null && value2 == null)
174                        return 0;
175                    int diff = ((Comparable)value1).compareTo(value2);
176                    if (diff != 0)
177                        return sortOrders2[i] ? diff : -diff;
178                }
179                return 0;
180            }
181        });
182    }

184    // Convert column names into column indexes
185    private static int[] determineSortColumns(List<ResultSet> resultSets, String[] sortColumnNames) {
186        if (resultSets.isEmpty())
187            throw new IllegalArgumentException("empty resultSets");
188        final int[] sortColumnNums = new int[sortColumnNames.length];
189        for (int i = 0; i < sortColumnNums.length; i++) {
190            try {
191                sortColumnNums[i] = resultSets.get(0).findColumn(sortColumnNames[i]);
192            } catch (SQLException e) {
193                throw new IllegalArgumentException("column `" + sortColumnNames[i] + "' not found", e);
194            }
195        }
196        return sortColumnNums;
197    }

199    private ResultSet getCurrent() throws SQLException {
200        this.checkClosed();
201        if (this.remaining.isEmpty())
202            throw new SQLException("there is no current row");
203        return this.remaining.peek();
204    }
205
206    private void checkClosed() throws SQLException {
207        if (this.closed)
208            throw new SQLException("ResultSet is closed");
209    }

211    @Override
212    public boolean next() throws SQLException {
213        try {
214            if (this.currentRow == 0) {                 // we are before the first row
215
216                // Perform the initial load
217                for (ResultSet resultSet : this.resultSets) {
218                    if (resultSet.next())
219                        this.remaining.add(resultSet);
220                }
221            } else if (!this.remaining.isEmpty()) {   // we are somewhere in the middle
222
223                // Advance the ResultSet we just read from and re-insert it into the priority queue
224                final ResultSet resultSet = this.remaining.remove();
225                if (resultSet.next())
226                    this.remaining.add(resultSet);
227            }
228
229            // Are all ResultSets completely exhausted?
230            if (this.remaining.isEmpty()) {
231                this.currentRow = -1;               // indicates we are past the last row
232                return false;
233            }
234        } catch (RuntimeException e) {
235            if (e.getCause() instanceof SQLException)
236                throw (SQLException)e.getCause();
237            throw e;
238        }
239
240        // Done
241        this.currentRow++;
242        return true;
243    }

245    @Override
246    public void close() throws SQLException {
247        this.closed = true;
248        for (ResultSet rs : this.resultSets)
249            rs.close();
250        this.remaining.clear();
251    }
