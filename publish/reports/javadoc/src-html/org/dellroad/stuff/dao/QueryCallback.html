008package org.dellroad.stuff.dao;
009
010import javax.persistence.EntityManager;
011import javax.persistence.Query;
012
013/**
014 * Helper class for building and executing JPA queries.
015 *
016 * @param <R> query return type
017 */
018public abstract class QueryCallback<R> {
019
020    /**
021     * Perform the query and return its result.
022     *
023     * <p>
024     * The implementation in {@link QueryCallback} simply delegates to {@link #buildQuery} and then {@link #executeQuery}.
025     * Subclasses may override if desired.
026     *
027     * @param entityManager JPA {@link EntityManager} representing an open transaction
028     * @return result of query
029     */
030    public R query(EntityManager entityManager) {
031        return this.executeQuery(this.buildQuery(entityManager));
032    }
033
034    /**
035     * Build the query.
036     *
037     * @param entityManager JPA {@link EntityManager} representing an open transaction
038     * @return constructed JPA {@link Query} to perform
039     */
040    protected abstract Query buildQuery(EntityManager entityManager);
041
042    /**
043     * Execute the query.
044     *
045     * @param query the {@link Query} constructed by {@link #buildQuery}
046     * @return result of query
047     */
048    protected abstract R executeQuery(Query query);
049}
