008    package org.dellroad.stuff.dao;
009    
010    import java.util.List;
011    
012    import javax.persistence.EntityManager;
013    import javax.persistence.EntityManagerFactory;
014    import javax.persistence.FlushModeType;
015    import javax.persistence.TypedQuery;
016    import javax.persistence.criteria.CriteriaBuilder;
017    import javax.persistence.criteria.CriteriaQuery;
018    
019    import org.slf4j.Logger;
020    import org.slf4j.LoggerFactory;
021    import org.springframework.orm.jpa.JpaCallback;
022    import org.springframework.orm.jpa.JpaTemplate;
023    import org.springframework.orm.jpa.support.JpaDaoSupport;
024    import org.springframework.transaction.support.TransactionSynchronizationManager;
025    
026    /**
027     * Support superclass for JPA DAO implementations.
028     *
029     * @param <T> persistent instance type
030     */
031    public abstract class AbstractDAO<T> extends JpaDaoSupport implements DAO<T> {
032    
033        protected final Logger log = LoggerFactory.getLogger(getClass());
034    
035        /**
036         * Persistent instance type.
037         */
038        protected final Class<T> type;
039    
040        /**
041         * Constructor.
042         *
043         * @param type persistent instance type
044         */
045        protected AbstractDAO(Class<T> type) {
046            if (type == null)
047                throw new IllegalArgumentException("null type");
048            this.type = type;
049        }
050    
051        /**
052         * Constructor.
053         *
054         * @param type persistent instance type
055         * @param entityManagerFactory {@link EntityManagerFactory} from which to create the {@link JpaTemplate} used by this instance
056         */
057        protected AbstractDAO(Class<T> type, EntityManagerFactory entityManagerFactory) {
058            this(type);
059            this.setEntityManagerFactory(entityManagerFactory);
060        }
061    
062        /**
063         * Constructor.
064         *
065         * @param type persistent instance type
066         * @param jpaTemplate {@link JpaTemplate} to be used by this instance
067         */
068        protected AbstractDAO(Class<T> type, JpaTemplate jpaTemplate) {
069            this(type);
070            this.setJpaTemplate(jpaTemplate);
071        }
072    
073    // Access methods
074    
075        @Override
076        public T getById(long id) {
077            return this.getJpaTemplate().find(this.type, id);
078        }
079    
080        @Override
081        public List<T> getAll() {
082            return this.getBy(new DAOCriteriaListCallback() {
083                @Override
084                protected void configureQuery(CriteriaQuery<T> criteriaQuery, CriteriaBuilder criteriaBuilder) {
085                    // no criteria - we want them all
086                }
087            });
088        }
089    
090        @Override
091        public T getReference(long id) {
092            return this.getJpaTemplate().getReference(this.type, id);
093        }
094    
095        /**
096         * Find instances using a query string and query parameters.
097         */
098        protected List<T> find(final String queryString, final Object... params) {
099            return this.getBy(new DAOQueryListCallback() {
100                @Override
101                protected TypedQuery<T> buildQuery(EntityManager entityManager) {
102                    return AbstractDAO.this.buildQuery(entityManager, queryString, params);
103                }
104            });
105        }
106    
107        /**
108         * Find a unique instance using a query string and query parameters.
109         */
110        protected T findUnique(final String queryString, final Object... params) {
111            return this.getBy(new DAOQueryUniqueCallback() {
112                @Override
113                protected TypedQuery<T> buildQuery(EntityManager entityManager) {
114                    return AbstractDAO.this.buildQuery(entityManager, queryString, params);
115                }
116            });
117        }
118    
119        /**
120         * Get the unique instance matched by the given query callback.
121         */
122        protected <R> R getBy(QueryCallback<R> callback) {
123            return this.getJpaTemplate().execute(callback);
124        }
125    
126        /**
127         * Perform a bulk update.
128         */
129        protected int bulkUpdate(UpdateCallback callback) {
130            return this.getJpaTemplate().execute(callback);
131        }
132    
133    // Lifecycle methods
134    
135        @Override
136        public void save(T obj) {
137            this.getJpaTemplate().persist(obj);
138        }
139    
140        @Override
141        public void delete(T obj) {
142            this.getJpaTemplate().remove(obj);
143        }
144    
145        @Override
146        public T merge(T obj) {
147            return this.getJpaTemplate().merge(obj);
148        }
149    
150        @Override
151        public void refresh(T obj) {
152            this.getJpaTemplate().refresh(obj);
153        }
154    
155        @Override
156        public void detach(final Object obj) {
157            this.getJpaTemplate().execute(new JpaCallback<Void>() {
158                @Override
159                public Void doInJpa(EntityManager entityManager) {
160                    entityManager.detach(obj);
161                    return null;
162                }
163            });
164        }
165    
166    // Session methods
167    
168        @Override
169        public void flush() {
170            this.getJpaTemplate().flush();
171        }
172    
173        @Override
174        public void setFlushMode(final FlushModeType flushMode) {
175            this.getJpaTemplate().execute(new JpaCallback<Void>() {
176                @Override
177                public Void doInJpa(EntityManager entityManager) {
178                    entityManager.setFlushMode(flushMode);
179                    return null;
180                }
181            });
182        }
183    
184        @Override
185        public void clear() {
186            this.getJpaTemplate().execute(new JpaCallback<Void>() {
187                @Override
188                public Void doInJpa(EntityManager entityManager) {
189                    entityManager.clear();
190                    return null;
191                }
192            });
193        }
194    
195        @Override
196        public boolean isReadOnly() {
197            return TransactionSynchronizationManager.isCurrentTransactionReadOnly();
198        }
199    
200    // Type and cast methods
201    
202        /**
203         * Cast the given object to this instance's persistent instance type.
204         */
205        protected T cast(Object obj) {
206            return this.type.cast(obj);
207        }
208    
209        /**
210         * Cast the given list to a list of this instance's persistent instance type.
211         * Does not actually inspect the contents of the list.
212         */
213        @SuppressWarnings("unchecked")
214        protected List<T> castList(List<?> list) {
215            return (List<T>)list;
216        }
217    
218    // Helper methods
219    
220        private TypedQuery<T> buildQuery(EntityManager entityManager, String queryString, Object[] params) {
221            TypedQuery<T> query = entityManager.createQuery(queryString, this.type);
222            if (params != null) {
223                for (int i = 0; i < params.length; i++)
224                    query.setParameter(i + 1, params[i]);
225            }
226            return query;
227        }
228    
229    // Helper classes
230    
231        /**
232         * Convenience subclass of {@link QueryCallback} for use by DAO subclasses when returning lists of persistent instances.
233         */
234        protected abstract class DAOQueryListCallback extends TypedQueryCallback<T, List<T>> {
235    
236            @Override
237            protected final List<T> executeQuery(TypedQuery<T> query) {
238                return query.getResultList();
239            }
240        }
241    
242        /**
243         * Convenience subclass of {@link QueryCallback} for use by DAO subclasses when returning a single persistent instance.
244         */
245        protected abstract class DAOQueryUniqueCallback extends TypedQueryCallback<T, T> {
246    
247            @Override
248            protected final T executeQuery(TypedQuery<T> query) {
249                return query.getSingleResult();
250            }
251        }
252    
253        /**
254         * Convenience subclass of {@link CriteriaCallback} for use by DAO subclasses when returning lists of persistent instances.
255         */
256        protected abstract class DAOCriteriaListCallback extends CriteriaCallback<T, List<T>> {
257    
258            protected DAOCriteriaListCallback() {
259                super(AbstractDAO.this.type);
260            }
261    
262            @Override
263            protected final List<T> executeQuery(TypedQuery<T> query) {
264                return query.getResultList();
265            }
266        }
267    
268        /**
269         * Convenience subclass of {@link CriteriaCallback} for use by DAO subclasses when returning a single persistent instance.
270         */
271        protected abstract class DAOCriteriaUniqueCallback extends CriteriaCallback<T, T> {
272    
273            protected DAOCriteriaUniqueCallback() {
274                super(AbstractDAO.this.type);
275            }
276    
277            @Override
278            protected final T executeQuery(TypedQuery<T> query) {
279                return query.getSingleResult();
280            }
281        }
282    }
