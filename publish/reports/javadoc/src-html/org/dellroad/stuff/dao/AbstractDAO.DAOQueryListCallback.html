008package org.dellroad.stuff.dao;
009
010import java.util.List;
011
012import javax.persistence.EntityManager;
013import javax.persistence.FlushModeType;
014import javax.persistence.NoResultException;
015import javax.persistence.PersistenceContext;
016import javax.persistence.TypedQuery;
017import javax.persistence.criteria.CriteriaBuilder;
018import javax.persistence.criteria.CriteriaQuery;
019
020import org.slf4j.Logger;
021import org.slf4j.LoggerFactory;
022import org.springframework.beans.factory.InitializingBean;
023import org.springframework.dao.EmptyResultDataAccessException;
024import org.springframework.transaction.support.TransactionSynchronizationManager;
025
026/**
027 * Support superclass for JPA DAO implementations.
028 *
029 * @param <T> persistent instance type
030 * @see org.springframework.dao.annotation.PersistenceExceptionTranslationPostProcessor
031 */
032public abstract class AbstractDAO<T> implements DAO<T>, InitializingBean {
033
034    protected final Logger log = LoggerFactory.getLogger(getClass());
035
036    /**
037     * Persistent instance type.
038     */
039    protected final Class<T> type;
040
041    /**
042     * The configured {@link EntityManager}.
043     */
044    private EntityManager entityManager;
045
046    /**
047     * Constructor.
048     *
049     * @param type persistent instance type
050     */
051    protected AbstractDAO(Class<T> type) {
052        if (type == null)
053            throw new IllegalArgumentException("null type");
054        this.type = type;
055    }
056
057// Configuration and lifecycle methods
058
059    /**
060     * Get the configured {@link EntityManager}.
061     */
062    protected EntityManager getEntityManager() {
063        return this.entityManager;
064    }
065
066    /**
067     * Set the configured {@link EntityManager}.
068     *
069     * <p>
070     * This method has a {@link PersistenceContext @PersistenceContext} annotation and normally
071     * would be invoked automatically by Spring.
072     */
073    @PersistenceContext
074    public void setEntityManager(EntityManager entityManager) {
075        this.entityManager = entityManager;
076    }
077
078    @Override
079    public void afterPropertiesSet() {
080        if (this.getEntityManager() == null)
081            throw new IllegalArgumentException("no entityManager configured");
082    }
083
084// Meta-data methods
085
086    @Override
087    public Class<T> getType() {
088        return this.type;
089    }
090
091// Access methods
092
093    @Override
094    public T getById(long id) {
095        return this.getEntityManager().find(this.type, id);
096    }
097
098    @Override
099    public List<T> getAll() {
100        return this.getBy(new DAOCriteriaListCallback() {
101            @Override
102            protected void configureQuery(CriteriaQuery<T> criteriaQuery, CriteriaBuilder criteriaBuilder) {
103                criteriaQuery.from(this.type);
104            }
105        });
106    }
107
108    @Override
109    public T getReference(long id) {
110        return this.getEntityManager().getReference(this.type, id);
111    }
112
113    /**
114     * Find instances using a query string and query parameters.
115     */
116    protected List<T> find(final String queryString, final Object... params) {
117        return this.getBy(new DAOQueryListCallback() {
118            @Override
119            protected TypedQuery<T> buildQuery(EntityManager em) {
120                return AbstractDAO.this.buildQuery(em, queryString, params);
121            }
122        });
123    }
124
125    /**
126     * Find a unique instance using a query string and query parameters.
127     *
128     * @return unique instance found, or null if none was found
129     */
130    protected T findUnique(final String queryString, final Object... params) {
131        return this.getBy(new DAOQueryUniqueCallback() {
132            @Override
133            protected TypedQuery<T> buildQuery(EntityManager em) {
134                return AbstractDAO.this.buildQuery(em, queryString, params);
135            }
136        });
137    }
138
139    /**
140     * Search using a {@link QueryCallback}.
141     */
142    protected <R> R getBy(QueryCallback<R> callback) {
143        return callback.query(this.getEntityManager());
144    }
145
146    /**
147     * Perform a bulk update.
148     */
149    protected int bulkUpdate(UpdateCallback callback) {
150        return callback.query(this.getEntityManager());
151    }
152
153// Lifecycle methods
154
155    @Override
156    public void save(T obj) {
157        this.getEntityManager().persist(obj);
158    }
159
160    @Override
161    public void delete(T obj) {
162        this.getEntityManager().remove(obj);
163    }
164
165    @Override
166    public T merge(T obj) {
167        return this.getEntityManager().merge(obj);
168    }
169
170    @Override
171    public void refresh(T obj) {
172        this.getEntityManager().refresh(obj);
173    }
174
175    @Override
176    public void detach(Object obj) {
177        this.getEntityManager().detach(obj);
178    }
179
180// Session methods
181
182    @Override
183    public void flush() {
184        this.getEntityManager().flush();
185    }
186
187    @Override
188    public void setFlushMode(FlushModeType flushMode) {
189        this.getEntityManager().setFlushMode(flushMode);
190    }
191
192    @Override
193    public void clear() {
194        this.getEntityManager().clear();
195    }
196
197    @Override
198    public boolean isReadOnly() {
199        return TransactionSynchronizationManager.isCurrentTransactionReadOnly();
200    }
201
202    @Override
203    public boolean contains(T obj) {
204        return this.getEntityManager().contains(obj);
205    }
206
207// Type and cast methods
208
209    /**
210     * Cast the given object to this instance's persistent instance type.
211     */
212    protected T cast(Object obj) {
213        return this.type.cast(obj);
214    }
215
216    /**
217     * Cast the given list to a list of this instance's persistent instance type.
218     * Does not actually inspect the contents of the list.
219     */
220    @SuppressWarnings("unchecked")
221    protected List<T> castList(List<?> list) {
222        return (List<T>)list;
223    }
224
225// Helper methods
226
227    private TypedQuery<T> buildQuery(EntityManager em, String queryString, Object[] params) {
228        TypedQuery<T> query = em.createQuery(queryString, this.type);
229        if (params != null) {
230            for (int i = 0; i < params.length; i++)
231                query.setParameter(i + 1, params[i]);
232        }
233        return query;
234    }
235
236// Helper classes
237
238    /**
239     * Convenience subclass of {@link QueryCallback} for use by DAO subclasses when returning lists of persistent instances.
240     */
241    protected abstract class DAOQueryListCallback extends TypedQueryCallback<T, List<T>> {
242
243        @Override
244        protected final List<T> executeQuery(TypedQuery<T> query) {
245            return query.getResultList();
246        }
247    }
248
249    /**
250     * Convenience subclass of {@link QueryCallback} for use by DAO subclasses when returning a single persistent instance.
251     *
252     * <p>
253     * Returns null if instance is not found.
254     */
255    protected abstract class DAOQueryUniqueCallback extends TypedQueryCallback<T, T> {
256
257        @Override
258        protected final T executeQuery(TypedQuery<T> query) {
259            try {
260                return query.getSingleResult();
261            } catch (NoResultException e) {
262                return null;
263            } catch (EmptyResultDataAccessException e) {
264                return null;
265            }
266        }
267    }
268
269    /**
270     * Convenience subclass of {@link CriteriaCallback} for use by DAO subclasses when returning lists of persistent instances.
271     */
272    protected abstract class DAOCriteriaListCallback extends CriteriaCallback<T, List<T>> {
273
274        protected DAOCriteriaListCallback() {
275            super(AbstractDAO.this.type);
276        }
277
278        @Override
279        protected final List<T> executeQuery(TypedQuery<T> query) {
280            return query.getResultList();
281        }
282    }
283
284    /**
285     * Convenience subclass of {@link CriteriaCallback} for use by DAO subclasses when returning a single persistent instance.
286     *
287     * <p>
288     * Returns null if instance is not found.
289     */
290    protected abstract class DAOCriteriaUniqueCallback extends CriteriaCallback<T, T> {
291
292        protected DAOCriteriaUniqueCallback() {
293            super(AbstractDAO.this.type);
294        }
295
296        @Override
297        protected final T executeQuery(TypedQuery<T> query) {
298            try {
299                return query.getSingleResult();
300            } catch (NoResultException e) {
301                return null;
302            } catch (EmptyResultDataAccessException e) {
303                return null;
304            }
305        }
306    }
307}
