008package org.dellroad.stuff.dao;
009
010import javax.persistence.EntityManager;
011import javax.persistence.TypedQuery;
012import javax.persistence.criteria.CriteriaBuilder;
013import javax.persistence.criteria.CriteriaQuery;
014
015/**
016 * Specialization of {@link QueryCallback} for {@link CriteriaQuery} queries.
017 *
018 * @param <T> query result type, i.e., what the {@link CriteriaQuery} returns zero or more of
019 * @param <R> query Java return type, usually either {@code R} or {@code List<R>}
020 */
021public abstract class CriteriaCallback<T, R> extends TypedQueryCallback<T, R> {
022
023    protected final Class<T> type;
024
025    protected CriteriaCallback(Class<T> type) {
026        if (type == null)
027            throw new IllegalArgumentException("null type");
028        this.type = type;
029    }
030
031    /**
032     * Build the query by delegating to {@link #configureQuery configureQuery()} for configuration.
033     */
034    @Override
035    protected TypedQuery<T> buildQuery(EntityManager entityManager) {
036        final CriteriaBuilder criteriaBuilder = entityManager.getCriteriaBuilder();
037        final CriteriaQuery<T> criteriaQuery = criteriaBuilder.createQuery(this.type);
038        this.configureQuery(criteriaQuery, criteriaBuilder);
039        return entityManager.createQuery(criteriaQuery);
040    }
041
042    /**
043     * Configure the {@link CriteriaQuery}.
044     */
045    protected abstract void configureQuery(CriteriaQuery<T> criteriaQuery, CriteriaBuilder criteriaBuilder);
046}
