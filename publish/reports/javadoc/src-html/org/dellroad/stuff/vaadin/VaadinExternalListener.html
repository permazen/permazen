008package org.dellroad.stuff.vaadin;
009
010/**
011 * Support superclass customized for use by listeners that are part of a Vaadin application when listening
012 * to non-Vaadin ("external") event sources.
013 *
014 * <p>
015 * Listeners that are part of a Vaadin application should use this superclass if they are going to be registered
016 * with non-Vaadin event sources, and use the methods {@link #register()} and {@link #unregister()} to control listener
017 * registration. Subclasses implement {@link #register(Object) register(S)} and {@link #unregister(Object) register(S)}
018 * to perform the actual listener registration/unregister operations.
019 *
020 * <p>
021 * Use of this class will ensure two things:
022 * <ul>
023 *  <li>Events can be delivered {@linkplain ContextApplication#invoke in the proper Vaadin application context}; and</li>
024 *  <li>The listener is automatically unregistered from the external event source when the Vaadin application is closed;
025 *      this avoids a memory leak</li>
026 * </ul>
027 * </p>
028 *
029 * <p>
030 * Subclass listener methods should use {@link #handleEvent handleEvent()} to handle events.
031 * </p>
032 *
033 * <p>
034 * Note: when listening to event sources that are scoped to specific Vaadin application instances and already originate events
035 * within the proper Vaadin application context (i.e., event sources that are not external to the Vaadin application),
036 * then the use of this superclass is not necessary (however, it also doesn't hurt to use it anyway).
037 * </p>
038 *
039 * @param <S> The type of the event source
040 * @see ContextApplication#invoke
041 * @see VaadinApplicationScope
042 * @see VaadinApplicationListener
043 * @see SpringContextApplication
044 */
045public abstract class VaadinExternalListener<S> {
046
047    private final S eventSource;
048    private final ContextApplication application;
049    private final CloseListener closeListener = new CloseListener();
050
051    /**
052     * Convenience constructor. Equivalent to:
053     * <blockquote>
054     *  {@link #VaadinExternalListener(Object, ContextApplication) VaadinExternalListener(eventSource, ContextApplication.get())}
055     * </blockquote>
056     *
057     * @param eventSource the event source on which this listener will register
058     * @throws IllegalArgumentException if {@code eventSource} is null
059     * @throws IllegalStateException if there is no {@link ContextApplication} associated with the current thread
060     */
061    protected VaadinExternalListener(S eventSource) {
062        this(eventSource, ContextApplication.get());
063    }
064
065    /**
066     * Primary constructor.
067     *
068     * @param eventSource the event source on which this listener will register
069     * @param application the associated Vaadin application
070     * @throws IllegalArgumentException if either parameter is null
071     */
072    protected VaadinExternalListener(S eventSource, ContextApplication application) {
073        if (eventSource == null)
074            throw new IllegalArgumentException("null eventSource");
075        if (application == null)
076            throw new IllegalArgumentException("null application");
077        this.eventSource = eventSource;
078        this.application = application;
079    }
080
081    /**
082     * Register as a listener on configured event source.
083     *
084     * <p>
085     * This also registers a {@link ContextApplication.CloseListener} on the
086     * {@linkplain #getApplication configured Vaadin application} so that when the application
087     * closes we can unregister this instance from the event source to avoid a memory leak.
088     */
089    public void register() {
090        this.application.addListener(this.closeListener);
091        this.register(this.eventSource);
092    }
093
094    /**
095     * Un-register as a listener on configured event source.
096     *
097     * <p>
098     * This also unregisters the {@link ContextApplication.CloseListener} registered by {@link #register}.
099     */
100    public void unregister() {
101        this.application.removeListener(this.closeListener);
102        this.unregister(this.eventSource);
103    }
104
105    /**
106     * Get the {@link ContextApplication} with which this instance is associated.
107     */
108    public final ContextApplication getApplication() {
109        return this.application;
110    }
111
112    /**
113     * Get the event source with which this instance is (or was) registered as a listener.
114     */
115    public final S getEventSource() {
116        return this.eventSource;
117    }
118
119    /**
120     * Execute the given action using the {@link ContextApplication} with which this instance is associated.
121     * Subclass listener methods should handle events using this method to ensure they are handled
122     * {@linkplain ContextApplication#invoke in the proper Vaadin application context}.
123     *
124     * @param action action to perform
125     */
126    protected void handleEvent(Runnable action) {
127        this.getApplication().invoke(action);
128    }
129
130    /**
131     * Register as a listener on the given event source.
132     *
133     * <p>
134     * Subclass must implement this to perform the actual listener registration.
135     *
136     * @param eventSource event source, never null
137     */
138    protected abstract void register(S eventSource);
139
140    /**
141     * Register as a listener from the given event source.
142     *
143     * <p>
144     * Subclass must implement this to perform the actual listener registration.
145     *
146     * @param eventSource event source, never null
147     */
148    protected abstract void unregister(S eventSource);
149
150// Application close listener
151
152    private final class CloseListener implements ContextApplication.CloseListener {
153
154        @Override
155        public void applicationClosed(ContextApplication.CloseEvent closeEvent) {
156            VaadinExternalListener.this.unregister();
157        }
158    }
159}
