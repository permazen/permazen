008package org.dellroad.stuff.vaadin;
009
010import com.vaadin.ui.AbstractSelect;
011import com.vaadin.ui.ComboBox;
012
013/**
014 * {@link ComboBox} that chooses an {@link Enum} value.
015 *
016 * @see EnumContainer
017 */
018@SuppressWarnings("serial")
019public class EnumComboBox extends ComboBox {
020
021    /**
022     * Default constructor.
023     *
024     * <p>
025     * Caller must separately invoke {@link #setEnumDataSource}.
026     * </p>
027     */
028    public EnumComboBox() {
029        this(null);
030    }
031
032    /**
033     * Convenience constructor.
034     *
035     * <p>
036     * Equivalent to:
037     *  <blockquote><code>
038     *  EnumComboBox(enumClass, EnumContainer.TO_STRING_PROPERTY.getName(), false);
039     *  </code></blockquote>
040     * </p>
041     *
042     * @param enumClass enum type, or null to leave data source unset
043     */
044    public <T extends Enum<T>> EnumComboBox(Class<T> enumClass) {
045        this(enumClass, false);
046    }
047
048    /**
049     * Convenience constructor.
050     *
051     * <p>
052     * Equivalent to:
053     *  <blockquote><code>
054     *  EnumComboBox(enumClass, EnumContainer.TO_STRING_PROPERTY, allowNull);
055     *  </code></blockquote>
056     * </p>
057     *
058     * @param enumClass enum type, or null to leave data source unset
059     * @param allowNull true to allow a null selection, false otherwise
060     */
061    public <T extends Enum<T>> EnumComboBox(Class<T> enumClass, boolean allowNull) {
062        this(enumClass, EnumContainer.TO_STRING_PROPERTY, allowNull);
063    }
064
065    /**
066     * Convenience constructor.
067     *
068     * <p>
069     * Equivalent to:
070     *  <blockquote><code>
071     *  EnumComboBox(enumClass, displayPropertyName, false);
072     *  </code></blockquote>
073     * </p>
074     *
075     * @param enumClass enum type
076     * @param displayPropertyName container property to display in the combo box
077     */
078    public <T extends Enum<T>> EnumComboBox(Class<T> enumClass, String displayPropertyName) {
079        this(enumClass, displayPropertyName, false);
080    }
081
082    /**
083     * Convenience constructor.
084     *
085     * <p>
086     * Equivalent to:
087     *  <blockquote><code>
088     *  EnumComboBox(enumClass != null ? new EnumContainer&lt;T&gt;(enumClass) : null, displayPropertyName, allowNull);
089     *  </code></blockquote>
090     * </p>
091     *
092     * @param enumClass enum type, or null to leave data source unset
093     * @param displayPropertyName container property to display in the combo box
094     * @param allowNull true to allow a null selection, false otherwise
095     */
096    public <T extends Enum<T>> EnumComboBox(Class<T> enumClass, String displayPropertyName, boolean allowNull) {
097        this(enumClass != null ? new EnumContainer<T>(enumClass) : null, displayPropertyName, allowNull);
098    }
099
100    /**
101     * Primary constructor.
102     *
103     * <p>
104     * This instance is configured for item caption {@link AbstractSelect#ITEM_CAPTION_MODE_PROPERTY} mode, with
105     * {@code displayPropertyName} as the {@linkplain #setItemCaptionPropertyId item caption property}.
106     * </p>
107     *
108     * @param container container data source, or null to leave data source unset
109     * @param displayPropertyName container property to display in the combo box
110     * @param allowNull true to allow a null selection, false otherwise
111     */
112    public EnumComboBox(EnumContainer<?> container, String displayPropertyName, boolean allowNull) {
113        if (container != null)
114            this.setContainerDataSource(container);
115        this.setNewItemsAllowed(false);
116        this.setTextInputAllowed(false);
117        this.setFilteringMode(AbstractSelect.Filtering.FILTERINGMODE_OFF);
118        this.setItemCaptionMode(AbstractSelect.ITEM_CAPTION_MODE_PROPERTY);
119        this.setItemCaptionPropertyId(displayPropertyName);
120
121        // Set up whether null selection is allowed
122        this.setNullSelectionAllowed(allowNull);
123        if (!allowNull && !this.getContainerDataSource().getItemIds().isEmpty())
124            this.setValue(this.getContainerDataSource().getItemIds().iterator().next());
125    }
126
127    /**
128     * Set the {@link Enum} type whose instances serve as this instance's data source.
129     *
130     * @throws IllegalArgumentException if {@code enumClass} is null
131     */
132    public <T extends Enum<T>> void setEnumDataSource(Class<T> enumClass) {
133        this.setContainerDataSource(new EnumContainer<T>(enumClass));
134        if (!this.isNullSelectionAllowed() && !this.getContainerDataSource().getItemIds().isEmpty())
135            this.setValue(this.getContainerDataSource().getItemIds().iterator().next());
136    }
137}
