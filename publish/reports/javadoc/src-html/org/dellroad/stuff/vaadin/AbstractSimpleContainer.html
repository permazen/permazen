026/**
027 * Support superclass for simple read-only, in-memory {@link Container} implementations where each
028 * {@link Item} in the container is backed by a Java object.
029 *
030 * <p>
031 * This {@link Container}'s {@link Property}s are defined via {@link PropertyDef}s, and a {@link PropertyExtractor}
032 * is used to actually extract the property values from each underlying object (alternately, subclasses can override
033 * {@link #getPropertyValue getPropertyValue()}). However, the easist way to configure the container {@link Property}s
034 * is to pass a {@link ProvidesProperty &#64;ProvidesProperty}-annotated Java class to the {@link #AbstractSimpleContainer(Class)}
035 * constructor.
036 * </p>
037 *
038 * <p>
039 * Use {@link #load load()} to load or reload the container.
040 * </p>
041 *
042 * @param <I> the item ID type
043 * @param <T> the type of the Java objects that back each {@link Item} in the container
044 *
045 * @see SimpleItem
046 * @see SimpleProperty
047 */
048@SuppressWarnings("serial")
049public abstract class AbstractSimpleContainer<I, T> extends AbstractInMemoryContainer<I, String, SimpleItem<T>>
050  implements PropertyExtractor<T>, Container.Filterable, Container.SimpleFilterable, Container.Sortable {
051
052    private final HashMap<String, PropertyDef<?>> propertyMap = new HashMap<String, PropertyDef<?>>();
053    private PropertyExtractor<? super T> propertyExtractor;
054
055// Constructors
056
057    /**
058     * Constructor.
059     *
060     * <p>
061     * After using this constructor, subsequent invocations of {@link #setPropertyExtractor setPropertyExtractor()}
062     * and {@link #setProperties setProperties()} are required to define the properties of this container
063     * and how to extract them.
064     * </p>
065     */
066    protected AbstractSimpleContainer() {
067        this((PropertyExtractor<? super T>)null);
068    }
069
070    /**
071     * Constructor.
072     *
073     * <p>
074     * After using this constructor, a subsequent invocation of {@link #setProperties setProperties()} is required
075     * to define the properties of this container.
076     * </p>
077     *
078     * @param propertyExtractor used to extract properties from the underlying Java objects;
079     *  may be null but then container is not usable until one is configured via
080     * {@link #setPropertyExtractor setPropertyExtractor()}
081     */
082    protected AbstractSimpleContainer(PropertyExtractor<? super T> propertyExtractor) {
083        this(propertyExtractor, null);
084    }

086    /**
087     * Constructor.
088     *
089     * <p>
090     * After using this constructor, a subsequent invocation of {@link #setPropertyExtractor setPropertyExtractor()} is required
091     * to define how to extract the properties of this container; alternately, subclasses can override
092     * {@link #getPropertyValue getPropertyValue()}.
093     * </p>
094     *
095     * @param propertyDefs container property definitions; null is treated like the empty set
096     */
097    protected AbstractSimpleContainer(Collection<? extends PropertyDef<?>> propertyDefs) {
098        this(null, propertyDefs);
099    }
100
101    /**
102     * Constructor.
103     *
104     * @param propertyExtractor used to extract properties from the underlying Java objects;
105     *  may be null but then container is not usable until one is configured via
106     * {@link #setPropertyExtractor setPropertyExtractor()}
107     * @param propertyDefs container property definitions; null is treated like the empty set
108     */
109    protected AbstractSimpleContainer(PropertyExtractor<? super T> propertyExtractor,
110      Collection<? extends PropertyDef<?>> propertyDefs) {
111        this.setItemSorter(new SimpleItemSorter());
112        this.setPropertyExtractor(propertyExtractor);
113        this.setProperties(propertyDefs);
114    }
115
116    /**
117     * Constructor.
118     *
119     * <p>
120     * Properties will be determined by the {@link ProvidesProperty &#64;ProvidesProperty} and
121     * {@link ProvidesPropertySort &#64;ProvidesPropertySort} annotated methods in the given class.
122     * </p>
123     *
124     * @param type class to introspect for annotated methods
125     * @throws IllegalArgumentException if {@code type} is null
126     * @throws IllegalArgumentException if {@code type} has two {@link ProvidesProperty &#64;ProvidesProperty}
127     *  or {@link ProvidesPropertySort &#64;ProvidesPropertySort} annotated methods for the same property
128     * @throws IllegalArgumentException if a {@link ProvidesProperty &#64;ProvidesProperty}-annotated method with no
129     *  {@linkplain ProvidesProperty#value property name specified} has a name which cannot be interpreted as a bean
130     *  property "getter" method
131     * @see ProvidesProperty
132     * @see ProvidesPropertySort
133     * @see ProvidesPropertyScanner
134     */
135    @SuppressWarnings({ "unchecked", "rawtypes" })
136    protected AbstractSimpleContainer(Class<? super T> type) {
137        // Why the JLS forces this stupid cast:
138        //  http://stackoverflow.com/questions/4902723/why-cant-a-java-type-parameter-have-a-lower-bound
139        final ProvidesPropertyScanner<? super T> propertyReader
140          = (ProvidesPropertyScanner<? super T>)new ProvidesPropertyScanner(type);
141        this.setItemSorter(new SimpleItemSorter());
142        this.setPropertyExtractor(propertyReader.getPropertyExtractor());
143        this.setProperties(propertyReader.getPropertyDefs());
144    }

166    /**
167     * Read the value of the property defined by {@code propertyDef} from the given object.
168     *
169     * <p>
170     * The implementation in {@link AbstractSimpleContainer} just delegates to the {@linkplain #setPropertyExtractor configured}
171     * {@link PropertyExtractor}; subclasses may override to customize property extraction.
172     * </p>
173     *
174     * @param obj Java object
175     * @param propertyDef definition of which property to read
176     * @throws NullPointerException if either parameter is null
177     * @throws IllegalStateException if no {@link PropertyExtractor} is configured for this container
178     */
179    @Override
180    public <V> V getPropertyValue(T obj, PropertyDef<V> propertyDef) {
181        if (this.propertyExtractor == null)
182            throw new IllegalStateException("no PropertyExtractor is configured for this container");
183        return this.propertyExtractor.getPropertyValue(obj, propertyDef);
184    }
185
186    /**
187     * Change the configured properties of this container.
188     *
189     * @param propertyDefs container property definitions; null is treated like the empty set
190     * @throws IllegalArgumentException if {@code propertyDefs} contains a property with a duplicate name
191     */
192    public void setProperties(Collection<? extends PropertyDef<?>> propertyDefs) {
193        if (propertyDefs == null)
194            propertyDefs = Collections.<PropertyDef<?>>emptySet();
195        this.propertyMap.clear();
196        for (PropertyDef<?> propertyDef : propertyDefs) {
197            if (this.propertyMap.put(propertyDef.getName(), propertyDef) != null)
198                throw new IllegalArgumentException("duplicate property name `" + propertyDef.getName() + "'");
199        }
200        this.fireContainerPropertySetChange();
201    }
202
203    /**
204     * Add or replace a configured property of this container.
205     *
206     * @param propertyDef new container property definitions
207     * @throws IllegalArgumentException if {@code propertyDef} is null
208     */
209    public void setProperty(PropertyDef<?> propertyDef) {
210        if (propertyDef == null)
211            throw new IllegalArgumentException("null propertyDef");
212        this.propertyMap.put(propertyDef.getName(), propertyDef);
213        this.fireContainerPropertySetChange();
214    }

216    /**
217     * Change this container's contents.
218     *
219     * @param contents new container contents
220     * @throws IllegalArgumentException if {@code contents} or any item in {@code contents} is null
221     */
222    public void load(Iterable<? extends T> contents) {
223
224        // Sanity check
225        if (contents == null)
226            throw new IllegalArgumentException("null contents");
227
228        // Reset item IDs
229        this.resetItemIds();
230        this.internalRemoveAllItems();
231
232        // Bulk load and register items with id's 0, 1, 2, ...
233        int index = 0;
234        for (T obj : contents) {
235            if (obj == null)
236                throw new IllegalArgumentException("null item in contents at index " + index);
237            SimpleItem<T> item = new SimpleItem<T>(obj, this.propertyMap, this);
238            this.internalAddItemAtEnd(this.generateItemId(obj), item, false);
239            index++;
240        }
241
242        // Apply filters
243        this.filterAll();
244
245        // Notify subclass
246        this.afterReload();
247
248        // Fire event
249        this.fireItemSetChange();
250    }

252    /**
253     * Get the container item ID corresponding to the given underlying Java object which is wrapped by this container.
254     * Objects are tested for equality using {@link Object#equals Object.equals()}.
255     *
256     * <p>
257     * The implementation in {@link AbstractSimpleContainer} requires a linear search of the container.
258     * Some subclasses may provide a more efficient implementation.
259     * </p>
260     *
261     * <p>
262     * This method is not used by this class but is defined as a convenience for subclasses.
263     * </p>
264     *
265     * <p>
266     * Note: items that are filtered out will not be found.
267     *
268     * @param obj underlying container object
269     * @return item ID corresponding to {@code object}, or null if {@code object} is not found in this container
270     * @throws IllegalArgumentException if {@code object} is null
271     * @see #getItemIdForSame
272     */
273    public I getItemIdFor(T obj) {
274        if (obj == null)
275            throw new IllegalArgumentException("null object");
276        for (I itemId : this.getItemIds()) {
277            T candidate = this.getJavaObject(itemId);
278            if (obj.equals(candidate))
279                return itemId;
280        }
281        return null;
282    }

284    /**
285     * Get the container item ID corresponding to the given underlying Java object which is wrapped by this container.
286     * Objects are tested for equality using object equality, not {@link Object#equals Object.equals()}.
287     *
288     * <p>
289     * The implementation in {@link AbstractSimpleContainer} requires a linear search of the container.
290     * Some subclasses may provide a more efficient implementation.
291     * </p>
292     *
293     * <p>
294     * This method is not used by this class but is defined as a convenience for subclasses.
295     * </p>
296     *
297     * <p>
298     * Note: items that are filtered out will not be found.
299     *
300     * @param obj underlying container object
301     * @return item ID corresponding to {@code object}, or null if {@code object} is not found in this container
302     * @throws IllegalArgumentException if {@code object} is null
303     * @see #getItemIdFor
304     */
305    public I getItemIdForSame(T obj) {
306        if (obj == null)
307            throw new IllegalArgumentException("null object");
308        for (I itemId : this.getItemIds()) {
309            T candidate = this.getJavaObject(itemId);
310            if (obj == candidate)
311                return itemId;
312        }
313        return null;
314    }

427    @Override
428    public Collection<?> getSortableContainerPropertyIds() {
429        final ArrayList<String> propertyIds = new ArrayList<String>(this.propertyMap.size());
430        for (Map.Entry<String, PropertyDef<?>> entry : this.propertyMap.entrySet()) {
431            if (this.propertyExtractor instanceof SortingPropertyExtractor) {
432                final SortingPropertyExtractor<? super T> sortingPropertyExtractor
433                  = (SortingPropertyExtractor<? super T>)this.propertyExtractor;
434                if (sortingPropertyExtractor.canSort(entry.getValue())) {
435                    propertyIds.add(entry.getKey());
436                    continue;
437                }
438            }
439            if (entry.getValue().isSortable())
440                propertyIds.add(entry.getKey());
441        }
442        return propertyIds;
443    }

445// ItemSorter class

447    /**
448     * {@link ItemSorter} implementation used by {@link AbstractSimpleContainer}.
449     */
450    private class SimpleItemSorter extends DefaultItemSorter {
451
452        @Override
453        @SuppressWarnings("unchecked")
454        protected int compareProperty(Object propertyId, boolean ascending, Item item1, Item item2) {
455
456            // Get property definition
457            final PropertyDef<?> propertyDef = AbstractSimpleContainer.this.propertyMap.get(propertyId);
458            if (propertyDef == null)
459                return super.compareProperty(propertyId, ascending, item1, item2);
460
461            // Ask SortingPropertyExtractor if we have one
462            if (AbstractSimpleContainer.this.propertyExtractor instanceof SortingPropertyExtractor) {
463                final SortingPropertyExtractor<? super T> sortingPropertyExtractor
464                  = (SortingPropertyExtractor<? super T>)AbstractSimpleContainer.this.propertyExtractor;
465                if (sortingPropertyExtractor.canSort(propertyDef)) {
466                    final T obj1 = ((SimpleItem<T>)item1).getObject();
467                    final T obj2 = ((SimpleItem<T>)item2).getObject();
468                    final int diff = sortingPropertyExtractor.sort(propertyDef, obj1, obj2);
469                    return ascending ? diff : -diff;
470                }
471            }
472
473            // Ask property definition
474            if (propertyDef.isSortable()) {
475                final int diff = this.sort(propertyDef, item1, item2);
476                return ascending ? diff : -diff;
477            }
478
479            // Defer to superclass
480            return super.compareProperty(propertyId, ascending, item1, item2);
481        }
482
483        // This method exists only to allow the generic parameter <V> to be bound
484        private <V> int sort(PropertyDef<V> propertyDef, Item item1, Item item2) {
485            return propertyDef.sort(propertyDef.read(item1), propertyDef.read(item2));
486        }
487    }
