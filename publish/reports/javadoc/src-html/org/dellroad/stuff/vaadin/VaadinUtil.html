008package org.dellroad.stuff.vaadin;
009
010/**
011 * Miscellaneous utility methods.
012 */
013public final class VaadinUtil {
014
015    private VaadinUtil() {
016    }
017
018    /**
019     * Verify that we are running in the context of the given {@link ContextApplication} and that it is locked.
020     * This method can be used by any code that manipulates Vaadin state to assert that the proper Vaadin
021     * locking has been performed.
022     *
023     * @param application the {@link ContextApplication} we are supposed to be running with
024     * @throws IllegalArgumentException if {@code application} is null
025     * @throws IllegalStateException if there is no {@link ContextApplication} associated with the current thread
026     * @throws IllegalStateException if the {@link ContextApplication} associated with the current thread is not {@code application}
027     * @throws IllegalStateException if the {@link ContextApplication} associated with the current thread is not locked
028     */
029    public static void assertApplication(ContextApplication application) {
030        if (application == null)
031            throw new IllegalArgumentException("null application");
032        final ContextApplication currentApplication = ContextApplication.currentApplication();
033        if (currentApplication == null)
034            throw new IllegalStateException("there is no ContextApplication associated with the current thread");
035        if (currentApplication != application) {
036            throw new IllegalStateException("the ContextApplication associated with the current thread " + currentApplication
037              + " is not the same application as the given one " + application);
038        }
039        if (!Thread.holdsLock(application)) {
040            throw new IllegalStateException("the ContextApplication associated with the current thread " + currentApplication
041              + " is not locked by this thread");
042        }
043    }
044}
