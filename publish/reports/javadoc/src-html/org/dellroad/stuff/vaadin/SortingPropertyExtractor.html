010/**
011 * {@link PropertyExtractor} that also sorts properties, given two instances of the target Java type.
012 *
013 * @param <T> target object type for extraction
014 */
015public interface SortingPropertyExtractor<T> extends PropertyExtractor<T> {
016
017    /**
018     * Determine if the given property can be sorted by this instance.
019     * If this method returns false, then the property's sorting resorts to the default behavior,
020     * i.e., depending on whether the property value implements {@link Comparable}.
021     *
022     * @param propertyDef definition of property
023     * @return true if the property defined by {@code propertyDef} can be sorted by this instance
024     * @throws NullPointerException if {@code propertyDef} is null
025     */
026    boolean canSort(PropertyDef<?> propertyDef);
027
028    /**
029     * Sort two values based on the given property.
030     *
031     * @param obj1 first Java object
032     * @param obj2 second Java object
033     * @param propertyDef definition of property to sort on
034     * @return result of comparing {@code obj1}'s property value to {@code obj2}'s property value
035     * @throws NullPointerException if any parameter is null
036     * @throws UnsupportedOperationException if the property defined by {@code propertyDef} cannot be sorted by this instance
037     */
038    int sort(PropertyDef<?> propertyDef, T obj1, T obj2);
039}
