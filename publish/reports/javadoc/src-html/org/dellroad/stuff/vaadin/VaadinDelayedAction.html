008package org.dellroad.stuff.vaadin;
009
010import org.dellroad.stuff.spring.DelayedAction;
011import org.springframework.scheduling.TaskScheduler;
012
013/**
014 * {@link DelayedAction} for actions that are associated with a Vaadin appliction.
015 */
016public abstract class VaadinDelayedAction extends DelayedAction {
017
018    private final ContextApplication application;
019
020    /**
021     * Convenience constructor. Equivalent to:
022     *  <blockquote><code>
023     *  VaadinDelayedAction(taskScheduler, ContextApplication.get())
024     *  </code></blockquote>
025     */
026    public VaadinDelayedAction(TaskScheduler taskScheduler) {
027        this(taskScheduler, ContextApplication.get());
028    }
029
030    /**
031     * Primary constructor.
032     *
033     * @param taskScheduler scheduler object
034     * @param application Vaadin application with which the action is associated
035     * @throws IllegalArgumentException if {@code application} is null
036     */
037    public VaadinDelayedAction(TaskScheduler taskScheduler, ContextApplication application) {
038        super(application, taskScheduler);
039        if (application == null)
040            throw new IllegalArgumentException("null application");
041        this.application = application;
042    }
043
044    /**
045     * Invokes {@link #runInVaadin} in the context of the configured {@link ContextApplication}.
046     */
047    @Override
048    public final void run() {
049        this.application.invoke(new Runnable() {
050            @Override
051            public void run() {
052                VaadinDelayedAction.this.runInVaadin();
053            }
054        });
055    }
056
057    /**
058     * Perform the action. This method will be invoked in the context of the configured {@link ContextApplication}.
059     */
060    protected abstract void runInVaadin();
061}
