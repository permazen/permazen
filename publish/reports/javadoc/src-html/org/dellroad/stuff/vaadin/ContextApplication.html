034/**
035 * {@link Application} subclass that provides some basic infrastructure for Vaadin applications:
036 * <ul>
037 *  <li>Access to the currently running Vaadin application (via a {@link ContextApplication#get()})</li>
038 *  <li>A way to safely interact with a Vaadin application from a background thread (via {@link ContextApplication#invoke})</li>
039 *  <li>Support for Vaadin {@linkplain ContextApplication#addListener application close event notifications}</li>
040 *  <li>Displays any exceptions thrown in an overlay error window</li>
041 *  <li>A {@link Logger} to use</li>
042 * </ul>
043 *
044 * @since 1.0.134
045 */
046@SuppressWarnings("serial")
047public abstract class ContextApplication extends Application implements Executor, HttpServletRequestListener {
048
049    /**
050     * Default notification linger time for error notifications (in milliseconds): Value is {@value}ms.
051     */
052    public static final int DEFAULT_NOTIFICATION_DELAY = 30000;
053
054    private static final ThreadLocal<ContextApplication> CURRENT_APPLICATION = new ThreadLocal<ContextApplication>();
055    private static final ThreadLocal<HttpServletRequest> CURRENT_REQUEST = new ThreadLocal<HttpServletRequest>();
056    private static final ThreadLocal<HttpServletResponse> CURRENT_RESPONSE = new ThreadLocal<HttpServletResponse>();
057
058    protected final Logger log = LoggerFactory.getLogger(this.getClass());
059
060    private final HashSet<CloseListener> closeListeners = new HashSet<CloseListener>();

066    /**
067     * Initialize the application.
068     *
069     * <p>
070     * The implementation in {@link ContextApplication} delegates to {@link #initApplication}.
071     * </p>
072     */
073    @Override
074    public final synchronized void init() {             // synchronized to work around http://dev.vaadin.com/ticket/9946
075
076        // Set current application
077        CURRENT_APPLICATION.set(this);
078
079        // Create executor service
080        this.executorService = Executors.newSingleThreadExecutor();
081
082        // Initialize application
083        boolean initialized = false;
084        try {
085            this.initApplication();
086            initialized = true;
087        } finally {
088            if (!initialized)
089                this.shutdownExecutorService();
090        }
091    }

100    /**
101     * Handle an uncaugt exception thrown by a Vaadin HTTP request.
102     *
103     * <p>
104     * The implementation in {@link ContextApplication} logs the error and displays in on the
105     * user's screen via {@link #showError(String, Throwable)}.
106     */
107    @Override
108    public void terminalError(Terminal.ErrorEvent event) {
109
110        // Delegate to superclass
111        super.terminalError(event);
112
113        // Get exception; ignore client "hangups"
114        final Throwable t = event.getThrowable();
115        if (t instanceof SocketException)
116            return;
117
118        // Notify user and log it
119        this.showError("Internal Error", "" + t);
120        this.log.error("error within Vaadin operation", t);
121    }
122
123    /**
124     * Display an error message to the user.
125     */
126    public void showError(String title, String description) {
127
128        // Get window
129        Window window = this.getMainWindow();
130        if (window == null)
131            return;
132
133        // Show error
134        Window.Notification notification = new Window.Notification(title, description, Window.Notification.TYPE_ERROR_MESSAGE);
135        notification.setStyleName("warning");
136        notification.setDelayMsec(this.getNotificationDelay());
137        window.showNotification(notification);
138    }
139
140    /**
141     * Display an error message to the user caused by an exception.
142     */
143    public void showError(String title, Throwable t) {
144        for (int i = 0; i < 100 && t.getCause() != null; i++)
145            t = t.getCause();
146        this.showError(title, this.getErrorMessage(t));
147    }

168    /**
169     * Set this instance as the "current application" (if not set already) while invoking the given callback.
170     *
171     * <p>
172     * This method is useful for situations in which non-Vaadin threads need to call into Vaadin code
173     * that expects to successfully retrieve the current application via {@link #currentApplication}.
174     * </p>
175     *
176     * <p>
177     * This method also synchronizes on this {@link ContextApplication} instance as required by Vaadin for thread safety.
178     * </p>
179     *
180     * @param action action to perform
181     * @throws IllegalStateException if a different {@link ContextApplication} is already set as the current application
182     *  associated with the current thread
183     * @see #invokeLater invokeLater()
184     */
185    public void invoke(Runnable action) {
186        final ContextApplication previous = ContextApplication.CURRENT_APPLICATION.get();
187        if (previous != null && previous != this)
188            throw new IllegalStateException("there is already a current application for this thread");
189        ContextApplication.CURRENT_APPLICATION.set(this);
190        try {
191            synchronized (this) {
192                action.run();
193            }
194        } finally {
195            ContextApplication.CURRENT_APPLICATION.set(previous);
196        }
197    }
198
199    /**
200     * Set this instance as the "current application" and invoke the given callback from within another thread.
201     *
202     * <p>
203     * This method functions like {@link #invoke invoke()} except that {@code action} will be invoked from within
204     * a separate thread dedicated to this application instance. This is useful to reduce Vaadin application lock
205     * contention, by performing Vaadin-related actions in a separate, dedicated thread. Actions are executed
206     * in the order they are given to this method.
207     * </p>
208     *
209     * <p>
210     * The returned {@link Future}'s {@link Future#get get()} method will return null upon successful completion.
211     * This method itself always returns immediately.
212     * </p>
213     *
214     * @param action action to perform
215     * @return a {@link Future} representing the pending results of {@code action}
216     * @throws IllegalStateException if this instance is not initialized or has been closed
217     * @see #invoke invoke()
218     */
219    public Future<?> invokeLater(Runnable action) {
220        ExecutorService executor = this.executorService;
221        if (executor == null)
222            throw new IllegalStateException("application instance is either not initialized or already closed");
223        return executor.submit(new LaterRunnable(action));
224    }

227     * {@link Executor} interface method, equivalent to {@link #invoke invoke()}.
228     *
229     * @see #invoke invoke()
230     */
231    public void execute(Runnable action) {
232        this.invoke(action);
233    }

235    /**
236     * Handle the start of a request.
237     *
238     * <p>
239     * The implementation in {@link ContextApplication} delegates to {@link #doOnRequestStart}.
240     * </p>
241     */
242    @Override
243    public final void onRequestStart(HttpServletRequest request, HttpServletResponse response) {
244        ContextApplication.CURRENT_APPLICATION.set(this);
245        ContextApplication.CURRENT_REQUEST.set(request);
246        ContextApplication.CURRENT_RESPONSE.set(response);
247        this.doOnRequestStart(request, response);
248    }

250    /**
251     * Handle the end of a request.
252     *
253     * <p>
254     * The implementation in {@link ContextApplication} delegates to {@link #doOnRequestEnd}.
255     * </p>
256     */
257    @Override
258    public final void onRequestEnd(HttpServletRequest request, HttpServletResponse response) {
259        try {
260            this.doOnRequestEnd(request, response);
261        } finally {
262            ContextApplication.CURRENT_APPLICATION.remove();
263            ContextApplication.CURRENT_REQUEST.remove();
264            ContextApplication.CURRENT_RESPONSE.remove();
265        }
266    }

386    /**
387     * Close this instance.
388     *
389     * <p>
390     * The implementation in {@link ContextApplication} first delegates to the superclass and then
391     * notifies any registered {@link CloseListener}s.
392     * </p>
393     */
394    @Override
395    public void close() {
396
397        // Invoke superclass
398        super.close();
399
400        // Notify listeners
401        CloseEvent closeEvent = new CloseEvent(this);
402        for (CloseListener closeListener : this.getCloseListeners()) {
403            try {
404                closeListener.applicationClosed(closeEvent);
405            } catch (ThreadDeath t) {
406                throw t;
407            } catch (Throwable t) {
408                this.log.error("exception thrown by CloseListener " + closeListener, t);
409            }
410        }
411
412        // Shutdown ExecutorService
413        this.shutdownExecutorService();
414    }
415
416    private void shutdownExecutorService() {
417
418        // Already shutdown?
419        if (this.executorService == null)
420            return;
421
422        // Shut it down; waiting up to 1 second to finish
423        this.executorService.shutdown();
424        boolean terminated = false;
425        try {
426            terminated = this.executorService.awaitTermination(1, TimeUnit.SECONDS);
427        } catch (InterruptedException e) {
428            Thread.currentThread().interrupt();
429        }
430
431        // Log warnings if it didn't finish
432        if (!terminated) {
433            this.log.warn("forcibly terminating outstanding tasks for closed Vaadin application " + this);
434            List<Runnable> list = this.executorService.shutdownNow();
435            if (!list.isEmpty())
436                this.log.warn(list.size() + " outstanding task(s) remain for closed Vaadin application " + this + ": " + list);
437        }
438
439        // Done
440        this.executorService = null;
441    }

443    /**
444     * Add a {@link CloseListener} to be notified when this instance is closed.
445     */
446    public void addListener(CloseListener listener) {
447        synchronized (this.closeListeners) {
448            this.closeListeners.add(listener);
449        }
450    }
451
452    /**
453     * Remove a {@link CloseListener}.
454     */
455    public void removeListener(CloseListener listener) {
456        synchronized (this.closeListeners) {
457            this.closeListeners.remove(listener);
458        }
459    }
460
461    private HashSet<CloseListener> getCloseListeners() {
462        synchronized (this.closeListeners) {
463            return new HashSet<CloseListener>(this.closeListeners);
464        }
465    }

486            super(application);
487        }

497    // Used by invokeLater()
498    private class LaterRunnable implements Runnable {
499
500        private final Runnable action;
501
502        LaterRunnable(Runnable action) {
503            this.action = action;
504        }
505
506        @Override
507        public void run() {
508            try {
509                ContextApplication.this.invoke(this.action);
510            } catch (ThreadDeath t) {
511                throw t;
512            } catch (Throwable e) {
513                ContextApplication.this.log.error("exception thrown by invokeLater() action " + this.action, e);
514            }
515        }
516
517        @Override
518        public String toString() {
519            return this.action.toString();
520        }
521    }

525    private void readObject(ObjectInputStream in) throws IOException, ClassNotFoundException {
526        in.defaultReadObject();
527        if (in.readBoolean())
528            this.executorService = Executors.newSingleThreadExecutor();
529    }

531    private void writeObject(ObjectOutputStream out) throws IOException {
532        out.defaultWriteObject();
533        out.writeBoolean(this.executorService != null);
534    }
