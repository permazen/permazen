008package org.dellroad.stuff.servlet;
009
010import java.sql.Driver;
011import java.sql.DriverManager;
012import java.util.Enumeration;
013
014import javax.servlet.ServletContextEvent;
015
016import org.dellroad.stuff.spring.ThreadLocalContext;
017import org.slf4j.Logger;
018import org.slf4j.LoggerFactory;
019import org.springframework.web.util.IntrospectorCleanupListener;
020
021/**
022 * Clears references that can cause container memory leaks on Web container class unloading.
023 *
024 * @see <a href="http://opensource.atlassian.com/confluence/spring/pages/viewpage.action?pageId=2669">Spring Wiki Discussion of
025 *      ClassLoader Memory Leaks</a>
026 * @see <a href="http://bugs.mysql.com/bug.php?id=65909">MySQL Bug #65909</a>
027 * @since 1.0.49
028 */
029public class ContextCleanupListener extends IntrospectorCleanupListener {
030
031    protected final Logger log = LoggerFactory.getLogger(getClass());
032
033    /**
034     * Unregisters any JDBC drivers registered under the current class loader after
035     * first invoking the superclass version of this method.
036     *
037     * <p>
038     * Also resets the {@link ThreadLocalContext} singleton instance
039     * so Tomcat won't complain about the lingering ThreadLocal variables.
040     * </p>
041     */
042    @Override
043    public void contextDestroyed(ServletContextEvent event) {
044        try {
045            super.contextDestroyed(event);
046
047            // Get my class loader
048            final ClassLoader thisClassLoader = this.getClass().getClassLoader();
049
050            // Work around MySQL bug http://bugs.mysql.com/bug.php?id=65909
051            try {
052                Class.forName("com.mysql.jdbc.AbandonedConnectionCleanupThread", false,
053                  thisClassLoader).getMethod("shutdown").invoke(null);
054            } catch (Exception e) {
055                // ignore
056            }
057
058            // Unregister JDBC drivers
059            for (Enumeration<Driver> e = DriverManager.getDrivers(); e.hasMoreElements(); ) {
060                final Driver driver = e.nextElement();
061                if (driver.getClass().getClassLoader() == thisClassLoader)
062                    DriverManager.deregisterDriver(driver);
063            }
064
065            // Reset the ThreadLocalContext singleton instance to make Tomcat happy
066            ThreadLocalContext.setInstance(new ThreadLocalContext());
067        } catch (Throwable e) {
068            log.error("exception cleaning up servlet context", e);
069        }
070    }
071}
