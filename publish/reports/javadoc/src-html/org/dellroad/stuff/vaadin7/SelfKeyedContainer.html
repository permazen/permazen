008    package org.dellroad.stuff.vaadin7;
009    
010    import java.util.Collection;
011    
012    /**
013     * A {@link SimpleKeyedContainer} where the item IDs are the underlying container objects themselves.
014     *
015     * <p>
016     * Restriction: instances can't contain two objects that are {@link Object#equals equal()} to each other.
017     *
018     * @param <T> the type of the Java objects that back each {@link com.vaadin.data.Item} in the container
019     */
020    @SuppressWarnings("serial")
021    public class SelfKeyedContainer<T> extends SimpleKeyedContainer<T, T> {
022    
023        /**
024         * Constructor.
025         *
026         * <p>
027         * After using this constructor, subsequent invocations of {@link #setPropertyExtractor setPropertyExtractor()}
028         * and {@link #setProperties setProperties()} are required to define the properties of this container
029         * and how to extract them.
030         */
031        public SelfKeyedContainer() {
032        }
033    
034        /**
035         * Constructor.
036         *
037         * <p>
038         * After using this constructor, a subsequent invocation of {@link #setProperties setProperties()} is required
039         * to define the properties of this container.
040         *
041         * @param propertyExtractor used to extract properties from the underlying Java objects;
042         *  may be null but then container is not usable until one is configured via
043         * {@link #setPropertyExtractor setPropertyExtractor()}
044         */
045        public SelfKeyedContainer(PropertyExtractor<? super T> propertyExtractor) {
046            super(propertyExtractor);
047        }
048    
049        /**
050         * Primary constructor.
051         *
052         * @param propertyExtractor used to extract properties from the underlying Java objects;
053         *  may be null but then container is not usable until one is configured via
054         * {@link #setPropertyExtractor setPropertyExtractor()}
055         * @param propertyDefs container property definitions; null is treated like the empty set
056         */
057        public SelfKeyedContainer(PropertyExtractor<? super T> propertyExtractor, Collection<? extends PropertyDef<?>> propertyDefs) {
058            super(propertyExtractor, propertyDefs);
059        }
060    
061        /**
062         * Get the key to be used as item ID for the given object.
063         *
064         * <p>
065         * The implementation in {@link SelfKeyedContainer} always returns {@code obj}.
066         *
067         * @param obj underlying container object
068         * @return key for object
069         */
070        @Override
071        protected T getKeyFor(T obj) {
072            return obj;
073        }
074    }
