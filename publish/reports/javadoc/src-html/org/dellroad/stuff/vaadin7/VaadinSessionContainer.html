008package org.dellroad.stuff.vaadin7;
009
010import com.vaadin.server.SessionDestroyEvent;
011import com.vaadin.server.SessionDestroyListener;
012import com.vaadin.server.SessionInitEvent;
013import com.vaadin.server.SessionInitListener;
014import com.vaadin.server.VaadinService;
015import com.vaadin.server.VaadinSession;
016
017import java.util.ArrayList;
018import java.util.Collection;
019
020import org.slf4j.Logger;
021import org.slf4j.LoggerFactory;
022
023/**
024 * Container containing active {@link VaadinSession}s.
025 *
026 * <p>
027 * This class is useful when you need to display sessions in your GUI, for example, a table showing all logged-in users.
028 * Due to the fact that each {@link VaadinSession} has its own lock, building such a container without race conditions
029 * and deadlocks is somewhat tricky. This class performs the locking required during updates and provides thread-safe
030 * {@link #update} and {@link #reload} methods. Information about active {@link VaadinSession}s comes from the session
031 * tracking feature of the {@link SpringVaadinServlet}.
032 *
033 * <p>
034 * By subclassing this class and {@link VaadinSessionInfo}, additional session-related properties,
035 * such as logged-in user, can be added to the container by annotating the {@link VaadinSessionInfo}
036 * subclass with {@link ProvidesProperty &#64;ProvidesProperty} annotations.
037 * </p>
038 *
039 * <p>
040 * Note the following:
041 * <ul>
042 *  <li>This class only works with configurations where Vaadin sessions are stored in memory,
043 *      and the {@link SpringVaadinServlet} must be used and configured with init parameter
044 *      {@link SpringVaadinServlet#SESSION_TRACKING_PARAMETER} set to {@code true}.</li>
045 *  <li>The {@link #connect} and {@link #disconnect} methods must be invoked before and after (respectively)
046 *      this container is used; typically these would be invoked in the {@link com.vaadin.ui.Component#attach}
047 *      and {@link com.vaadin.ui.Component#detach} methods of a corresponding widget.
048 * </ul>
049 * </p>
050 *
051 * @param <T> the type of the Java objects that back each {@link com.vaadin.data.Item} in the container
052 */
053@SuppressWarnings("serial")
054public abstract class VaadinSessionContainer<T extends VaadinSessionInfo> extends SelfKeyedContainer<T> {
055
056    /**
057     * The {@link VaadinSession} that this container instance is associated with.
058     * Determined by whatever {@link VaadinSession} is associated with the current thread at construction time.
059     */
060    protected final VaadinSession session;
061
062    private final Logger log = LoggerFactory.getLogger(this.getClass());
063    private final SessionEventListener listener = new SessionEventListener();
064
065    /**
066     * Constructor.
067     *
068     * <p>
069     * Properties will be determined by the {@link ProvidesProperty &#64;ProvidesProperty} and
070     * {@link ProvidesPropertySort &#64;ProvidesPropertySort} annotated methods in the given class.
071     * </p>
072     *
073     * @param type class to introspect for annotated methods
074     * @throws IllegalArgumentException if {@code type} is null
075     * @throws IllegalArgumentException if {@code type} has two {@link ProvidesProperty &#64;ProvidesProperty}
076     *  or {@link ProvidesPropertySort &#64;ProvidesPropertySort} annotated methods for the same property
077     * @throws IllegalArgumentException if a {@link ProvidesProperty &#64;ProvidesProperty}-annotated method with no
078     *  {@linkplain ProvidesProperty#value property name specified} has a name which cannot be interpreted as a bean
079     *  property "getter" method
080     * @throws IllegalStateException if there is no {@link VaadinSession} associated with the current thread
081     * @throws IllegalStateException if there is no {@link SpringVaadinServlet} associated
082     *  with the current thread's {@link VaadinSession}
083     * @see ProvidesProperty
084     * @see ProvidesPropertySort
085     * @see ProvidesPropertyScanner
086     */
087    protected VaadinSessionContainer(Class<T> type) {
088        super(type);
089        this.session = VaadinUtil.getCurrentSession();
090        SpringVaadinServlet.getServlet(this.session);
091    }
092
093    /**
094     * Constructor.
095     *
096     * @param propertyExtractor used to extract properties from the underlying Java objects;
097     *  may be null but then container is not usable until one is configured via
098     * {@link #setPropertyExtractor setPropertyExtractor()}
099     * @param propertyDefs container property definitions; null is treated like the empty set
100     */
101    protected VaadinSessionContainer(PropertyExtractor<? super T> propertyExtractor,
102      Collection<? extends PropertyDef<?>> propertyDefs) {
103        super(propertyExtractor, propertyDefs);
104        this.session = VaadinUtil.getCurrentSession();
105        SpringVaadinServlet.getServlet(this.session);
106    }
107
108    /**
109     * Connect this container and start tracking sessions.
110     */
111    public void connect() {
112        this.listener.register();
113        this.reload();
114    }
115
116    /**
117     * Disconnect this container and stop tracking sessions.
118     */
119    public void disconnect() {
120        this.listener.unregister();
121    }
122
123    /**
124     * Asynchronously reload this container.
125     *
126     * <p>
127     * This method can be invoked from any thread. It creates a new thread to do the actual
128     * reloading via {@link #doReload} to avoid potential deadlocks.
129     * </p>
130     */
131    public void reload() {
132        new Thread("VaadinSessionContainer.reload()") {
133            @Override
134            public void run() {
135                try {
136                    VaadinSessionContainer.this.doReload();
137                } catch (ThreadDeath t) {
138                    throw t;
139                } catch (Throwable t) {
140                    VaadinSessionContainer.this.log.error("error reloading container " + this, t);
141                }
142            }
143        }.start();
144    }
145
146    /**
147     * Asynchronously update this container's items.
148     *
149     * <p>
150     * This method can be invoked from any thread. It creates a new thread to do the actual
151     * updating via {@link #doUpdate} to avoid potential deadlocks.
152     * </p>
153     */
154    public void update() {
155        new Thread("VaadinSessionContainer.update()") {
156            @Override
157            public void run() {
158                try {
159                    VaadinSessionContainer.this.doUpdate();
160                } catch (ThreadDeath t) {
161                    throw t;
162                } catch (Throwable t) {
163                    VaadinSessionContainer.this.log.error("error updating container " + this, t);
164                }
165            }
166        }.start();
167    }
168
169    /**
170     * Update each {@link VaadinSessionInfo} instance in this container.
171     * Using this method is more efficient than reloading the entire container.
172     *
173     * <p>
174     * This method handles the complicated locking required to avoid deadlocks: first, for each {@link VaadinSessionInfo},
175     * {@link VaadinSessionInfo#updateInformation} is invoked while the {@link VaadinSession}
176     * corresponding to <i>that {@link VaadinSessionInfo} instance</i> is locked, so that information from that session
177     * can be safely gathered; then, {@link VaadinSessionInfo#makeUpdatesVisible} is invoked while the {@link VaadinSession}
178     * associated with <i>this container</i> is locked, so item properties can be safely updated, etc.
179     * </p>
180     *
181     * <p>
182     * This method must <b>not</b> be invoked while any {@link VaadinSession} is locked.
183     * For example, it may be invoked by a regular timer (only while this container is {@linkplain #connect connected}).
184     * </p>
185     *
186     * @throws IllegalStateException if there is a locked {@link VaadinSession} associated with the current thread
187     */
188    protected void doUpdate() {
189
190        // Sanity check
191        if (VaadinSession.getCurrent() != null && VaadinSession.getCurrent().hasLock())
192            throw new IllegalStateException("inside locked VaadinSession");
193
194        // Snapshot the set of sessions in this container while holding the lock to this container's session
195        final ArrayList<T> sessionInfoList = new ArrayList<T>();
196        VaadinSessionContainer.this.session.accessSynchronously(new Runnable() {
197            @Override
198            public void run() {
199                sessionInfoList.addAll(VaadinSessionContainer.this.getItemIds());
200            }
201        });
202
203        // Update each session's information while holding that session's lock
204        for (T sessionInfo : sessionInfoList) {
205            final T sessionInfo2 = sessionInfo;
206            sessionInfo.getVaadinSession().accessSynchronously(new Runnable() {
207                @Override
208                public void run() {
209                    sessionInfo2.updateInformation();
210                }
211            });
212        }
213
214        // Now update this container with the newly gathered information while again holding the lock to this container's session
215        VaadinSessionContainer.this.session.accessSynchronously(new Runnable() {
216            @Override
217            public void run() {
218                for (T sessionInfo : sessionInfoList)
219                    sessionInfo.makeUpdatesVisible();
220            }
221        });
222    }
223
224    /**
225     * Reload this container. Reloads this container with {@link VaadinSessionInfo} instances for each {@link VaadinSession}
226     * (created via {@link #createVaadinSessionInfo}), and then invokes {@link #doUpdate}.
227     *
228     * <p>
229     * This method must <b>not</b> be invoked while any {@link VaadinSession} is locked.
230     * </p>
231     *
232     * @throws IllegalStateException if there is a locked {@link VaadinSession} associated with the current thread
233     */
234    protected void doReload() {
235
236        // Sanity check
237        if (VaadinSession.getCurrent() != null && VaadinSession.getCurrent().hasLock())
238            throw new IllegalStateException("inside locked VaadinSession");
239
240        // Create a VaadinSessionInfo object for each session, but doing so while that session is locked
241        final ArrayList<T> sessionInfoList = new ArrayList<T>();
242        for (VaadinSession otherSession : SpringVaadinServlet.getServlet(this.session).getSessions()) {
243            otherSession.accessSynchronously(new Runnable() {
244                @Override
245                public void run() {
246                    sessionInfoList.add(VaadinSessionContainer.this.createVaadinSessionInfo());
247                }
248            });
249        }
250
251        // Reload this container
252        this.session.access(new Runnable() {
253            @Override
254            public void run() {
255                VaadinSessionContainer.this.load(sessionInfoList);
256            }
257        });
258
259        // Update this container
260        this.doUpdate();
261    }
262
263    /**
264     * Create a {@link VaadinSessionInfo} backing object for the {@link VaadinSession} associated with the current thread.
265     * The {@link VaadinSession} will be already locked.
266     */
267    protected abstract T createVaadinSessionInfo();
268
269// SessionEventListener
270
271    private class SessionEventListener extends VaadinExternalListener<VaadinService>
272      implements SessionInitListener, SessionDestroyListener {
273
274        public SessionEventListener() {
275            super(VaadinUtil.getCurrentSession().getService());
276        }
277
278        @Override
279        public void sessionInit(SessionInitEvent event) {
280            VaadinSessionContainer.this.reload();
281        }
282
283        @Override
284        public void sessionDestroy(SessionDestroyEvent event) {
285            VaadinSessionContainer.this.reload();
286        }
287
288        @Override
289        protected void register(VaadinService vaadinService) {
290            vaadinService.addSessionInitListener(this);
291            vaadinService.addSessionDestroyListener(this);
292        }
293
294        @Override
295        protected void unregister(VaadinService vaadinService) {
296            vaadinService.removeSessionInitListener(this);
297            vaadinService.removeSessionDestroyListener(this);
298        }
299    }
300}
