049/**
050 * Automatically builds and binds fields for a Java bean annotated with {@link FieldBuilder} annotations.
051 * The various nested {@link FieldBuilder} annotation types annotate Java bean property "getter" methods and specify
052 * how the the bean properties of that class should be edited using {@link AbstractField}s. This allows all information
053 * about how to edit a Java model class to stay contained within that class.
054 *
055 * <p>
056 * This class supports two types of annotations: first,the {@link ProvidesField} annotation annotates a method that knows
057 * how to build an {@link AbstractField} suitable for editing the bean property specified by
058 * its {@link ProvidesField#value value()}. So {@link ProvidesField} is analgous to {@link ProvidesProperty}, except that
059 * it defines an editing field rather than a container property.
060 * </p>
061 *
062 * <p>
063 * The {@link FieldBuilder.AbstractField} hierarchy annotations are the other type of annotation. These annotations annotate
064 * a Java bean property getter method and specify how to configure an {@link AbstractField} to edit the bean property
065 * corresponding to the getter method.
066 * {@link FieldBuilder.AbstractField} is the top level annotation in a hierarchy of annotations that correspond to the
067 * {@link AbstractField} class hierarchy. {@link FieldBuilder.AbstractField} corresponds to {@link AbstractField},
068 * and its properties configure corresponding {@link AbstractField} properties.
069 * More specific annotations correspond to the various {@link AbstractField} subclasses,
070 * for example {@link ComboBox FieldBuilder.ComboBox} corresponds to {@link ComboBox}.
071 * When using more specific annotations, the "superclass" annotations configure the superclass properties.
072 * </p>
073 *
074 * <p>
075 * A simple example shows how these annotations are used:
076 * <blockquote><pre>
077 * // Use a 10x40 TextArea to edit the "description" property
078 * <b>&#64;FieldBuilder.TextArea(columns = 40, rows = 10)</b>
079 * <b>&#64;FieldBuilder.AbstractField(caption = "Description:")</b>
080 * public String getDescription() {
081 *     return this.description;
082 * }
083 *
084 * // Use my own custom field to edit the "foobar" property
085 * <b>&#64;FieldBuilder.ProvidesField("foobar")</b>
086 * private MyCustomField createFoobarField() {
087 *     ...
088 * }
089 * </pre></blockquote>
090 * </p>
091 *
092 * <p>
093 * A {@link FieldBuilder} instance will read these annotations and build the fields automatically. For example:
094 * <blockquote><pre>
095 * // Create fields based on FieldGroup.* annotations
096 * Person person = new Person("Joe Smith", 100);
097 * BeanFieldGroup&lt;Person&gt; fieldGroup = <b>FieldBuilder.buildFieldGroup(person)</b>;
098 *
099 * // Layout the fields in a form
100 * FormLayout layout = new FormLayout();
101 * for (Field&lt;?&gt; field : fieldGroup.getFields())
102 *     layout.addComponent(field);
103 * </pre></blockquote>
104 * </p>
105 *
106 * <p>
107 * For all annotations in the {@link FieldBuilder.AbstractField} hierarchy, leaving properties set to their default values
108 * results in the default behavior.
109 * </p>
110 *
111 * @see AbstractSelect
112 * @see AbstractTextField
113 * @see CheckBox
114 * @see ComboBox
115 * @see DateField
116 * @see ListSelect
117 * @see PasswordField
118 * @see TextArea
119 * @see TextField
120 */

123    /**
124     * Introspect for {@link FieldBuilder} annotations on property getter methods of the
125     * {@link BeanFieldGroup}'s data source, and then build and bind the corresponding fields.
126     *
127     * @param fieldGroup field group to configure
128     * @throws IllegalArgumentException if {@code fieldGroup} is null
129     * @throws IllegalArgumentException if {@code fieldGroup} does not yet
130     *  {@linkplain BeanFieldGroup#setItemDataSource(Object) have a data source}
131     */
132    public void buildAndBind(BeanFieldGroup<?> fieldGroup) {
133
134        // Sanity check
135        if (fieldGroup == null)
136            throw new IllegalArgumentException("null beanType");
137        final BeanItem<?> beanItem = fieldGroup.getItemDataSource();
138        if (beanItem == null)
139            throw new IllegalArgumentException("fieldGroup does not yet have a data source");
140
141        // Scan bean properties to build fields
142        for (Map.Entry<String, com.vaadin.ui.AbstractField<?>> entry : this.buildBeanPropertyFields(beanItem.getBean()).entrySet())
143            fieldGroup.bind(entry.getValue(), entry.getKey());
144    }

146    /**
147     * Introspect for {@link FieldBuilder} annotations on property getter methods and build
148     * a mapping from Java bean property name to a field that may be used to edit that property.
149     *
150     * @param bean Java bean
151     * @return mapping from bean property name to field
152     * @throws IllegalArgumentException if {@code bean} is null
153     * @throws IllegalArgumentException if invalid or conflicting annotations are encountered
154     */
155    public Map<String, com.vaadin.ui.AbstractField<?>> buildBeanPropertyFields(Object bean) {
156
157        // Sanity check
158        if (bean == null)
159            throw new IllegalArgumentException("null bean");
160
161        // Look for all bean property getter methods
162        BeanInfo beanInfo;
163        try {
164            beanInfo = Introspector.getBeanInfo(bean.getClass());
165        } catch (IntrospectionException e) {
166            throw new RuntimeException("unexpected exception", e);
167        }
168        final ArrayList<Method> getterList = new ArrayList<Method>();
169        final HashMap<String, Method> getterMap = new HashMap<String, Method>();
170        for (PropertyDescriptor propertyDescriptor : beanInfo.getPropertyDescriptors()) {
171            final Method method = propertyDescriptor.getReadMethod();
172            if (method != null && method.getReturnType() != void.class && method.getParameterTypes().length == 0)
173                getterMap.put(propertyDescriptor.getName(), method);
174        }
175
176        // Scan getters for FieldBuilder.* annotations other than FieldBuidler.ProvidesField
177        final HashMap<String, com.vaadin.ui.AbstractField<?>> map = new HashMap<String, com.vaadin.ui.AbstractField<?>>();
178        for (Map.Entry<String, Method> entry : getterMap.entrySet()) {
179            final String propertyName = entry.getKey();
180            final Method method = entry.getValue();
181
182            // Get annotations, if any
183            final List<AnnotationApplier<?, ?>> applierList = this.buildApplierList(method);
184            if (applierList.isEmpty())
185                continue;
186
187            // Build field
188            map.put(propertyName, this.buildField(applierList, "method " + method));
189        }
190
191        // Scan all methods for @FieldBuidler.ProvidesField annotations
192        final HashMap<String, Method> providerMap = new HashMap<String, Method>();
193        for (Method method : bean.getClass().getDeclaredMethods()) {
194            if (method.getReturnType() == void.class || method.getParameterTypes().length > 0)
195                continue;
196            this.buildProviderMap(providerMap, bean.getClass(), method.getName());
197        }
198
199        // Check for conflicts between @FieldBuidler.ProvidesField and other annotations and add fields to map
200        for (Map.Entry<String, Method> entry : providerMap.entrySet()) {
201            final String propertyName = entry.getKey();
202            final Method method = entry.getValue();
203
204            // Verify field is not already defined
205            if (map.containsKey(propertyName)) {
206                throw new IllegalArgumentException("conflicting annotations exist for property `" + propertyName + "': annotation @"
207                  + ProvidesField.class.getName() + " on method " + method
208                  + " cannot be combined with other @FieldBuilder.* annotation types");
209            }
210
211            // Invoke method to create field
212            com.vaadin.ui.AbstractField<?> field;
213            try {
214                method.setAccessible(true);
215            } catch (Exception e) {
216                // ignore
217            }
218            try {
219                field = (com.vaadin.ui.AbstractField)method.invoke(bean);
220            } catch (Exception e) {
221                throw new RuntimeException("error invoking @" + ProvidesField.class.getName()
222                  + " annotation on method " + method, e);
223            }
224
225            // Save field
226            map.put(propertyName, field);
227        }
228
229        // Done
230        return map;
231    }

233    // Used by buildBeanPropertyFields()
234    private void buildProviderMap(Map<String, Method> providerMap, Class<?> type, String methodName) {
235
236        // Terminate recursion
237        if (type == null)
238            return;
239
240        // Check the method in this class
241        do {
242
243            // Get method
244            Method method;
245            try {
246                method = type.getDeclaredMethod(methodName);
247            } catch (NoSuchMethodException e) {
248                break;
249            }
250
251            // Get annotation
252            final ProvidesField providesField = method.getAnnotation(ProvidesField.class);
253            if (providesField == null)
254                break;
255
256            // Validate method return type is compatible with AbstractField
257            if (!com.vaadin.ui.AbstractField.class.isAssignableFrom(method.getReturnType())) {
258                throw new IllegalArgumentException("invalid @" + ProvidesField.class.getName() + " annotation on method " + method
259                  + ": return type " + method.getReturnType().getName() + " is not a subtype of "
260                  + com.vaadin.ui.AbstractField.class.getName());
261            }
262
263            // Check for two methods declaring fields for the same property
264            final String propertyName = providesField.value();
265            final Method otherMethod = providerMap.get(propertyName);
266            if (otherMethod != null && !otherMethod.getName().equals(methodName)) {
267                throw new IllegalArgumentException("conflicting @" + ProvidesField.class.getName()
268                  + " annotations exist for property `" + propertyName + "': both method "
269                  + otherMethod + " and method " + method + " are specified");
270            }
271
272            // Save method
273            if (otherMethod == null)
274                providerMap.put(propertyName, method);
275        } while (false);
276
277        // Recurse on interfaces
278        for (Class<?> iface : type.getInterfaces())
279            this.buildProviderMap(providerMap, iface, methodName);
280
281        // Recurse on superclass
282        this.buildProviderMap(providerMap, type.getSuperclass(), methodName);
283    }

286     * Create a {@link BeanFieldGroup} using the given instance, introspect for {@link FieldBuilder} annotations
287     * on property getter methods of the given bean's class, and build and bind the corresponding fields, and return
288     * the result.
289     *
290     * @param bean model bean annotated with {@link FieldBuilder} annotations
291     * @throws IllegalArgumentException if {@code bean} is null
292     */
293    @SuppressWarnings("unchecked")
294    public static <T> BeanFieldGroup<T> buildFieldGroup(T bean) {
295
296        // Sanity check
297        if (bean == null)
298            throw new IllegalArgumentException("null bean");
299
300        // Create field group
301        final BeanFieldGroup<T> fieldGroup = new BeanFieldGroup<T>((Class<T>)bean.getClass());
302        fieldGroup.setItemDataSource(bean);
303        new FieldBuilder().buildAndBind(fieldGroup);
304        return fieldGroup;
305    }

307    /**
308     * Instantiate and configure an {@link AbstractField} according to the given scanned annotations.
309     */
310    protected com.vaadin.ui.AbstractField<?> buildField(Collection<AnnotationApplier<?, ?>> appliers, String description) {
311
312        // Get comparator that sorts by class hierarcy, narrower types first; note Collections.sort() is stable,
313        // so for any specific annotation type, that annotation on subtype appears before that annotation on supertype.
314        final Comparator<AnnotationApplier<?, ?>> comparator = new Comparator<AnnotationApplier<?, ?>>() {
315            @Override
316            public int compare(AnnotationApplier<?, ?> a1, AnnotationApplier<?, ?> a2) {
317                final Class<? extends com.vaadin.ui.AbstractField<?>> type1 = a1.getFieldType();
318                final Class<? extends com.vaadin.ui.AbstractField<?>> type2 = a2.getFieldType();
319                if (type1 == type2)
320                    return 0;
321                if (type1.isAssignableFrom(type2))
322                    return 1;
323                if (type2.isAssignableFrom(type1))
324                    return -1;
325                return 0;
326            }
327        };
328
329        // Sanity check for duplicates and conflicts
330        final ArrayList<AnnotationApplier<?, ?>> applierList = new ArrayList<AnnotationApplier<?, ?>>(appliers);
331        Collections.sort(applierList, comparator);
332        for (int i = 0; i < applierList.size() - 1; ) {
333            final AnnotationApplier<?, ?> a1 = applierList.get(i);
334            final AnnotationApplier<?, ?> a2 = applierList.get(i + 1);
335
336            // Let annotations on subclass override annotations on superclass
337            if (a1.getAnnotation().annotationType() == a2.getAnnotation().annotationType()) {
338                applierList.remove(i + 1);
339                continue;
340            }
341
342            // Check for conflicting annotation types (e.g., both FieldBuilder.TextField and FieldBuilder.DateField)
343            if (comparator.compare(a1, a2) == 0) {
344                throw new IllegalArgumentException("conflicting annotations of type "
345                  + a1.getAnnotation().annotationType().getName() + " and " + a2.getAnnotation().annotationType().getName()
346                  + " for " + description);
347            }
348            i++;
349        }
350
351        // Determine field type
352        Class<? extends com.vaadin.ui.AbstractField<?>> type = null;
353        AnnotationApplier<?, ?> typeApplier = null;
354        for (AnnotationApplier<?, ?> applier : applierList) {
355
356            // Pick up type() if specified
357            if (applier.getActualFieldType() == null)
358                continue;
359            if (type == null) {
360                type = applier.getActualFieldType();
361                typeApplier = applier;
362                continue;
363            }
364
365            // Verify the field type specified by a narrower annotation has compatible narrower field type
366            if (!applier.getActualFieldType().isAssignableFrom(type)) {
367                throw new IllegalArgumentException("conflicting field types specified by annotations of type "
368                  + typeApplier.getAnnotation().annotationType().getName() + " (type() = " + type.getName() + ") and "
369                  + applier.getAnnotation().annotationType().getName() + " (type() = " + applier.getActualFieldType().getName()
370                  + ") for " + description);
371            }
372        }
373        if (type == null)
374            throw new IllegalArgumentException("cannot determine field type; no type() specified for " + description);
375
376        // Instantiate field
377        final com.vaadin.ui.AbstractField<?> field = FieldBuilder.instantiate(type);
378
379        // Configure the field
380        for (AnnotationApplier<?, ?> applier : applierList)
381            this.apply(applier, field);
382
383        // Done
384        return field;
385    }

393    private static <T> T instantiate(Class<T> type) {
394        Constructor<T> constructor;
395        try {
396            constructor = type.getDeclaredConstructor();
397        } catch (Exception e) {
398            throw new RuntimeException("cannot instantiate " + type + " because no zero-arg constructor could be found", e);
399        }
400        try {
401            constructor.setAccessible(true);
402        } catch (Exception e) {
403            // ignore
404        }
405        try {
406            return constructor.newInstance();
407        } catch (Exception e) {
408            throw new RuntimeException("cannot instantiate " + type + " using its zero-arg constructor", e);
409        }
410    }

412    /**
413     * Find all relevant annotations on the given method as well as on any supertype methods it overrides.
414     * The method must be a getter method taking no arguments. Annotations are ordered so that annotations
415     * on a method in type X appear before annotations on an overridden method in type Y, a supertype of X.
416     *
417     * @throws IllegalArgumentException if {@code method} is null
418     * @throws IllegalArgumentException if {@code method} has parameters
419     */
420    protected List<AnnotationApplier<?, ?>> buildApplierList(Method method) {
421
422        // Sanity check
423        if (method == null)
424            throw new IllegalArgumentException("null method");
425        if (method.getParameterTypes().length > 0)
426            throw new IllegalArgumentException("method takes parameters");
427
428        // Recurse
429        final ArrayList<AnnotationApplier<?, ?>> list = new ArrayList<AnnotationApplier<?, ?>>();
430        this.buildApplierList(method.getDeclaringClass(), method.getName(), list);
431        return list;
432    }
433
434    private void buildApplierList(Class<?> type, String methodName, List<AnnotationApplier<?, ?>> list) {
435
436        // Terminate recursion
437        if (type == null)
438            return;
439
440        // Check class
441        Method method;
442        try {
443            method = type.getMethod(methodName);
444        } catch (NoSuchMethodException e) {
445            method = null;
446        }
447        if (method != null)
448            list.addAll(this.buildApplierList((AccessibleObject)method));
449
450        // Recurse on interfaces
451        for (Class<?> iface : type.getInterfaces())
452            this.buildApplierList(iface, methodName, list);
453
454        // Recurse on superclass
455        this.buildApplierList(type.getSuperclass(), methodName, list);
456    }

458    /**
459     * Find all relevant annotations declared directly on the given {@link AccessibleObject}.
460     *
461     * @throws IllegalArgumentException if {@code member} is null
462     */
463    protected List<AnnotationApplier<?, ?>> buildApplierList(AccessibleObject member) {
464
465        // Sanity check
466        if (member == null)
467            throw new IllegalArgumentException("null member");
468
469        // Build list
470        final ArrayList<AnnotationApplier<?, ?>> list = new ArrayList<AnnotationApplier<?, ?>>();
471        for (Annotation annotation : member.getDeclaredAnnotations()) {
472            final AnnotationApplier<? extends Annotation, ?> applier = this.getAnnotationApplier(annotation);
473            if (applier != null)
474                list.add(applier);
475        }
476        return list;
477    }

479    /**
480     * Get the {@link AnnotationApplier} that applies the given annotation.
481     *
482     * @return corresponding {@link AnnotationApplier}, or null if annotation is unknown
483     */
484    protected AnnotationApplier<?, ?> getAnnotationApplier(Annotation annotation) {
485        if (annotation instanceof FieldBuilder.AbstractField)
486            return new AbstractFieldApplier((FieldBuilder.AbstractField)annotation);
487        if (annotation instanceof FieldBuilder.AbstractSelect)
488            return new AbstractSelectApplier((FieldBuilder.AbstractSelect)annotation);
489        if (annotation instanceof FieldBuilder.CheckBox)
490            return new CheckBoxApplier((FieldBuilder.CheckBox)annotation);
491        if (annotation instanceof FieldBuilder.ComboBox)
492            return new ComboBoxApplier((FieldBuilder.ComboBox)annotation);
493        if (annotation instanceof FieldBuilder.ListSelect)
494            return new ListSelectApplier((FieldBuilder.ListSelect)annotation);
495        if (annotation instanceof FieldBuilder.DateField)
496            return new DateFieldApplier((FieldBuilder.DateField)annotation);
497        if (annotation instanceof FieldBuilder.AbstractTextField)
498            return new AbstractTextFieldApplier((FieldBuilder.AbstractTextField)annotation);
499        if (annotation instanceof FieldBuilder.TextField)
500            return new TextFieldApplier((FieldBuilder.TextField)annotation);
501        if (annotation instanceof FieldBuilder.TextArea)
502            return new TextAreaApplier((FieldBuilder.TextArea)annotation);
503        if (annotation instanceof FieldBuilder.PasswordField)
504            return new PasswordFieldApplier((FieldBuilder.PasswordField)annotation);
505        return null;
506    }

511     * Class that knows how to apply annotation properties to a corresponding field.
512     */
513    protected abstract static class AnnotationApplier<A extends Annotation, F extends com.vaadin.ui.AbstractField<?>> {
514
515        protected final A annotation;
516        protected final Class<F> fieldType;
517
518        protected AnnotationApplier(A annotation, Class<F> fieldType) {
519            if (annotation == null)
520                throw new IllegalArgumentException("null annotation");
521            if (fieldType == null)
522                throw new IllegalArgumentException("null fieldType");
523            this.annotation = annotation;
524            this.fieldType = fieldType;
525        }
526
527        public final A getAnnotation() {
528            return this.annotation;
529        }
530
531        public final Class<F> getFieldType() {
532            return this.fieldType;
533        }
534
535        public abstract Class<? extends F> getActualFieldType();
536
537        public abstract void applyTo(F field);
538    }

540    /**
541     * Applies properties from a {@link FieldBuilder.AbstractField} annotation to a {@link com.vaadin.ui.AbstractField}.
542     */
543    private static class AbstractFieldApplier
544      extends AnnotationApplier<FieldBuilder.AbstractField, com.vaadin.ui.AbstractField<?>> {
545
546        @SuppressWarnings("unchecked")
547        public AbstractFieldApplier(FieldBuilder.AbstractField annotation) {
548            super(annotation, (Class<com.vaadin.ui.AbstractField<?>>)(Object)com.vaadin.ui.AbstractField.class);
549        }
550
551        @Override
552        @SuppressWarnings("unchecked")
553        public Class<? extends com.vaadin.ui.AbstractField<?>> getActualFieldType() {
554            return (Class<com.vaadin.ui.AbstractField<?>>)(Object)(this.annotation.type() != com.vaadin.ui.AbstractField.class ?
555              this.annotation.type() : null);
556        }
557
558        @Override
559        @SuppressWarnings("unchecked")
560        public void applyTo(com.vaadin.ui.AbstractField<?> field) {
561            if (this.annotation.width().length() > 0)
562                field.setWidth(this.annotation.width());
563            if (this.annotation.height().length() > 0)
564                field.setHeight(this.annotation.height());
565            if (this.annotation.caption().length() > 0)
566                field.setCaption(this.annotation.caption());
567            if (this.annotation.description().length() > 0)
568                field.setDescription(this.annotation.description());
569            field.setEnabled(this.annotation.enabled());
570            field.setImmediate(this.annotation.immediate());
571            field.setReadOnly(this.annotation.readOnly());
572            field.setBuffered(this.annotation.buffered());
573            field.setInvalidAllowed(this.annotation.invalidAllowed());
574            field.setInvalidCommitted(this.annotation.invalidCommitted());
575            field.setValidationVisible(this.annotation.validationVisible());
576            field.setRequired(this.annotation.required());
577            field.setTabIndex(this.annotation.tabIndex());
578            if (this.annotation.converter() != Converter.class)
579                field.setConverter(FieldBuilder.instantiate(this.annotation.converter()));
580            for (Class<? extends Validator> validatorType : this.annotation.validators())
581                field.addValidator(FieldBuilder.instantiate(validatorType));
582            for (String styleName : this.annotation.styleNames())
583                field.addStyleName(styleName);
584            if (this.annotation.conversionError().length() > 0)
585                field.setConversionError(this.annotation.conversionError());
586            if (this.annotation.requiredError().length() > 0)
587                field.setRequiredError(this.annotation.requiredError());
588        }
589    }
