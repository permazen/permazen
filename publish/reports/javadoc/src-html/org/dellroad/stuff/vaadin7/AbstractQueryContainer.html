022/**
023 * Support superclass for read-only {@link Container} implementations where each {@link Item} in the container
024 * is backed by a Java object, and the Java objects are generated via {@linkplain #query a query} that returns
025 * a {@link QueryList} containing only some portion of the total list of backing objects at any one time.
026 *
027 * <p>
028 * This {@link Container}'s {@link Property}s are defined via {@link PropertyDef}s, and a {@link PropertyExtractor}
029 * is used to actually extract the property values from each underlying object (alternately, subclasses can override
030 * {@link #getPropertyValue getPropertyValue()}). However, the easist way to configure the container {@link Property}s
031 * is to pass a {@link ProvidesProperty &#64;ProvidesProperty}-annotated Java class to the {@link #AbstractQueryContainer(Class)}
032 * constructor.
033 * </p>
034 *
035 * <p>
036 * This {@link Container}'s item ID's are simply the indexes of the corresponding objects in the overall list. The {@link QueryList}
037 * interface is designed for scalability; at minimum, it is required to provide only the size of the total list
038 * (which may only be an estimate) and one backing object at a specified index.
039 *
040 * <p>
041 * This class will invoke {@link #query} as needed to (re)generate the {@link QueryList}; the {@link QueryList} is then cached.
042 * However, if any invocation of {@link QueryList#get} throws an {@link InvalidQueryListException}, then the cached
043 * list is discarded and {@link #query} is invoked again to regenerate it. In this way, the {@link QueryList} is
044 * allowed to decide, on demand, when it is invalid or incapable of providing a specific list member. For example,
045 * when using JPA, a list may be considered invalid if the current EntityManager session has changed.
046 * </p>
047 *
048 * <p>
049 * Note that the {@link QueryList} being invalid is an orthogonal concept from the contents of the list having changed.
050 * Invalid means "this list can no longer be used" while changed means "this list contains out-of-date information".
051 * Normally, the latter implies the former (but not vice-versa). The list becoming invalid does not in itself not cause
052 * any notifications to be sent, so no new query will be performed until e.g. the user interface explicitly requests
053 * more information.
054 * </p>
055 *
056 * <p>
057 * Therefore, if the list size or content changes, first invoke {@link #invalidate} to discard the cached {@link QueryList},
058 * and then {@link #fireItemSetChange} to notify listeners; for convenience, {@link #reload} will perform these two steps for
059 * you. The new size and content will be provided by the {@link QueryList} returned by the next invocation of {@link #query}.
060 * </p>
061 *
062 * <p>
063 * The subclass may forcibly invalidate the current {@link QueryList} via {@link #invalidate}; this merely discards it
064 * and will force a new invocation of {@link #query} on the next container access. In many situations, however,
065 * the use of {@link #invalidate} is never required.
066 * </p>
067 *
068 * <p>
069 * For scalability reasons the {@link QueryList} may actually only contain a portion of the list, throwing
070 * {@link InvalidQueryListException}s when other list members are accessed.  An "index hint" parameter provided to
071 * {@link #query} indicates which member of the container is of current interest. The returned {@link QueryList}
072 * is required to provide exception-free access only to the indicated member, so in the extreme case only a single
073 * list member could be kept. In practice, normally range of members near to the index hint would be kept;
074 * see for example {@link WindowQueryList}.
075 * </p>
076 *
077 * @param <T> the type of the Java objects that back each {@link Item} in the container
078 * @see QueryList
079 * @see SimpleQueryList
080 * @see WindowQueryList
081 * @see AbstractUnsizedContainer
082 */
083@SuppressWarnings("serial")
084public abstract class AbstractQueryContainer<T> extends AbstractContainer implements PropertyExtractor<T>,
085  Container.Ordered, Container.Indexed, Container.PropertySetChangeNotifier, Container.ItemSetChangeNotifier {

226     * @throws IllegalArgumentException if {@code propertyDefs} contains a property with a duplicate name
227     */
228    public void setProperties(Collection<? extends PropertyDef<?>> propertyDefs) {
229        if (propertyDefs == null)
230            propertyDefs = Collections.<PropertyDef<?>>emptySet();
231        this.propertyMap.clear();
232        for (PropertyDef<?> propertyDef : propertyDefs) {
233            if (this.propertyMap.put(propertyDef.getName(), propertyDef) != null)
234                throw new IllegalArgumentException("duplicate property name `" + propertyDef.getName() + "'");
235        }
236        this.fireContainerPropertySetChange();
237    }

252    /**
253     * Perform a query to (re)generate the list of Java objects backing this container.
254     *
255     * <p>
256     * The particular position in the list we are interested in is given as a hint by the {@code hint} parameter.
257     * That is, an invocation of {@link QueryList#get}{@code (hint)} is likely immediately after this method
258     * returns and if so it must complete without throwing an exception, unless {@code hint} is out of range.
259     * </p>
260     *
261     * <p>
262     * The {@code hint} can be used to implement a highly scalable query list containing external objects
263     * (such as from a database) where only a small "window" of objects is actually kept in memory at any one time.
264     * Of course, implementations are also free to ignore {@code hint}. However, the returned {@link QueryList}
265     * must at least tolerate one invocation of {@link QueryList#get get}{@code (hint)} without throwing an exception
266     * when {@code hint} is less that the {@link QueryList#size size()} of the returned {@link QueryList}.
267     * </p>
268     *
269     * @param hint index of the list element we are interested in
270     * @return list of Java objects backing this container
271     */
272    protected abstract QueryList<? extends T> query(long hint);

281    /**
282     * Get the Java backing object at the given index in the list.
283     *
284     * @return backing object, or null if {@code index} is out of range
285     */
286    protected T getJavaObject(int index) {
287        Exception exception;
288        long currentSize;
289        int attempt = 1;
290        while (true) {
291            currentSize = this.ensureList(index);
292            if (index < 0 || index >= currentSize)
293                return null;
294            try {
295                return this.queryList.get(index);
296            } catch (InvalidQueryListException e) {
297                this.invalidate();
298                if (attempt == 100) {           // avoid infinite loops
299                    exception = e;
300                    break;
301                }
302            } catch (IndexOutOfBoundsException e) {
303                exception = e;
304                break;
305            }
306            attempt++;
307        }
308
309        // The QueryList is behaving badly
310        throw new RuntimeException("query(" + index + ") returned a QueryList with size() = "
311          + currentSize + " but QueryList.get(" + index + ") failed (attempt #" + attempt + ")", exception);
312    }
313
314    /**
315     * Ensure we have a cached query list.
316     *
317     * @param hint index of the list element we are interested in, passed to {@link #query} if no query list is cached
318     * @return the size of the cached query list
319     */
320    protected long ensureList(int hint) {
321        if (this.queryList == null) {
322            this.queryList = this.query(hint);
323            final long oldTotalSize = this.totalSize;
324            final long newTotalSize = this.queryList.size();
325            this.totalSize = newTotalSize;
326            if (oldTotalSize != -1 && newTotalSize != oldTotalSize)
327                this.handleSizeChange();
328        }
329        return this.totalSize;
330    }

332    /**
333     * Invoked when a new {@link QueryList} has returned a changed {@link QueryList#size size()}
334     * for the underlying list.
335     *
336     * <p>
337     * Normally, this implies an item set change notification will be generated elsewhere and so
338     * no additional action needs to be taken by this method. However, some implementations may
339     * lack such a mechanism, for example, when the container's size is only ever calculated when
340     * {@link #query} is invoked. In such cases, this method may trigger a notification.
341     * </p>
342     *
343     * <p>
344     * Note: to avoid re-entrancy problems, this method should not send out any notifications itself;
345     * instead, it may schedule notifications to be delivered later, e.g., via {@link VaadinUtil#invokeLater}.
346     * </p>
347     *
348     * <p>
349     * The implementation in {@link AbstractQueryContainer} does nothing, assuming the notification is handled
350     * elsewhere. Subclasses may override if needed.
351     * </p>
352     */
353    protected void handleSizeChange() {
354    }

481    @Override

483        if (numberOfItems < 0)
484            throw new IllegalArgumentException("numberOfItems < 0");
485        final long size = this.ensureList(startIndex);
486        if (startIndex < 0 || startIndex > size)
487            throw new IndexOutOfBoundsException("startIndex=" + startIndex + " but size=" + size);
488        if (startIndex + numberOfItems > size)
489            numberOfItems = (int)(size - startIndex);
490        return new IntList(startIndex, numberOfItems);
491    }

620    private static class IntList extends AbstractList<Integer> {
621
622        private final int min;
623        private final int size;
624
625        public IntList(int size) {
626            this(0, size);
627        }
628
629        public IntList(int min, int size) {
630            if (size < 0)
631                throw new IllegalArgumentException("size < 0");
632            this.min = min;
633            this.size = size;
634        }
635
636        @Override
637        public int size() {
638            return this.size;
639        }
640
641        @Override
642        public Integer get(int index) {
643            if (index < 0 || index >= this.size)
644                throw new IndexOutOfBoundsException();
645            return this.min + index;
646        }
647    }
