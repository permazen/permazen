008    package org.dellroad.stuff.vaadin7;
009    
010    import com.vaadin.data.Container;
011    import com.vaadin.data.Item;
012    import com.vaadin.data.Property;
013    
014    /**
015     * Vaadin 7 version of {@link org.dellroad.stuff.vaadin.PropertyDef}.
016     *
017     * @param <T> property's value type
018     * @see org.dellroad.stuff.vaadin.PropertyDef
019     */
020    public class PropertyDef<T> extends org.dellroad.stuff.vaadin.PropertyDef<T> {
021    
022        /**
023         * Convenience contructor.
024         *
025         * <p>
026         * Equivalent to:
027         *  <blockquote><pre>
028         *  PropertyDef(name, type, null);
029         *  </pre></blockquote>
030         * </p>
031         */
032        public PropertyDef(String name, Class<T> type) {
033            super(name, type);
034        }
035    
036        /**
037         * Primary constructor.
038         *
039         * @param name property name; also serves as the property ID
040         * @param type property type
041         * @param defaultValue default value for the property; may be null
042         */
043        public PropertyDef(String name, Class<T> type, T defaultValue) {
044            super(name, type, defaultValue);
045        }
046    
047        @Override
048        public Property<T> get(Item item) {
049            return this.cast(item.getItemProperty(this.getPropertyId()));
050        }
051    
052        @Override
053        public Property<T> get(Container container, Object itemId) {
054            return this.cast(container.getContainerProperty(itemId, this.getPropertyId()));
055        }
056    
057        @SuppressWarnings("unchecked")
058        @Override
059        public Property<T> cast(Property/*<?>*/ property) {
060            return (Property<T>)super.cast(property);
061        }
062    }
