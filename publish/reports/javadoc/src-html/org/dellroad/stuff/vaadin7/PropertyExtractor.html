008package org.dellroad.stuff.vaadin7;
009
010/**
011 * Classes that can extract {@link com.vaadin.data.Property} values from Java objects.
012 *
013 * @param <T> target object type for extraction
014 */
015public interface PropertyExtractor<T> {
016
017    /**
018     * Read the value of the property defined by {@code propertyDef} from the given object.
019     *
020     * @param obj Java object
021     * @param propertyDef definition of which property to read
022     * @throws IllegalArgumentException if this instance does not recognize {@code propertyDef}
023     * @throws NullPointerException if either parameter is null
024     */
025    <V> V getPropertyValue(T obj, PropertyDef<V> propertyDef);
026}
