008package org.dellroad.stuff.vaadin7;
009
010import com.vaadin.server.VaadinSession;
011
012import org.springframework.context.ApplicationEvent;
013import org.springframework.context.ApplicationListener;
014import org.springframework.context.event.ApplicationEventMulticaster;
015
016/**
017 * A Spring {@link org.springframework.context.ApplicationListener} support superclass customized for use by
018 * listeners that are part of a Vaadin application when listening to non-Vaadin application event sources.
019 *
020 * <p>
021 * Listeners that are part of a Vaadin application should use this superclass if they are going to be registered
022 * with non-Vaadin event sources. This will ensure that events are delivered in the proper Vaadin application context
023 * and memory leaks are avoided when the session closes.
024 * </p>
025 *
026 * @param <E> The type of the event
027 * @see VaadinExternalListener
028 * @see VaadinUtil#invoke
029 * @see VaadinApplicationScope
030 * @see SpringVaadinSessionListener
031 */
032public abstract class VaadinApplicationListener<E extends ApplicationEvent>
033  extends VaadinExternalListener<ApplicationEventMulticaster> implements ApplicationListener<E> {
034
035    /**
036     * Convenience constructor. Equivalent to:
037     * <blockquote>
038     *  {@link #VaadinApplicationListener(ApplicationEventMulticaster, VaadinSession)
039     *      VaadinApplicationListener(multicaster, VaadinUtil.getCurrentSession())}
040     * </blockquote>
041     *
042     * @param multicaster the application event multicaster on which this listener will register
043     * @throws IllegalArgumentException if {@code multicaster} is null
044     * @throws IllegalArgumentException if there is no {@link VaadinSession} associated with the current thread
045     */
046    public VaadinApplicationListener(ApplicationEventMulticaster multicaster) {
047        this(multicaster, VaadinUtil.getCurrentSession());
048    }
049
050    /**
051     * Primary constructor.
052     *
053     * @param multicaster the application event multicaster on which this listener will register
054     * @param session the associated Vaadin application
055     * @throws IllegalArgumentException if either parameter is null
056     */
057    public VaadinApplicationListener(ApplicationEventMulticaster multicaster, VaadinSession session) {
058        super(multicaster, session);
059    }
060
061    @Override
062    protected void register(ApplicationEventMulticaster multicaster) {
063        multicaster.addApplicationListener(this);
064    }
065
066    @Override
067    protected void unregister(ApplicationEventMulticaster multicaster) {
068        multicaster.removeApplicationListener(this);
069    }
070
071    @Override
072    public final void onApplicationEvent(final E event) {
073        this.handleEvent(new Runnable() {
074            @Override
075            public void run() {
076                VaadinApplicationListener.this.onApplicationEventInternal(event);
077            }
078        });
079    }
080
081    /**
082     * Handle a listener event. When this method is invoked, it will already be within the context
083     * of the {@link VaadinSession} with which this listener is associated.
084     *
085     * @see VaadinUtil#invoke
086     * @see VaadinUtil#getCurrentSession
087     */
088    protected abstract void onApplicationEventInternal(E event);
089}
