008package org.dellroad.stuff.vaadin7;
009
010import com.vaadin.shared.ui.combobox.FilteringMode;
011import com.vaadin.ui.AbstractSelect.ItemCaptionMode;
012import com.vaadin.ui.ComboBox;
013
014/**
015 * {@link ComboBox} that chooses an {@link Enum} value.
016 *
017 * @see EnumContainer
018 */
019@SuppressWarnings("serial")
020public class EnumComboBox extends ComboBox {
021
022    /**
023     * Convenience constructor.
024     *
025     * <p>
026     * Equivalent to:
027     *  <blockquote><code>
028     *  EnumComboBox(new EnumContainer(type), EnumContainer.TO_STRING_PROPERTY.getName(), false);
029     *  </code></blockquote>
030     * </p>
031     *
032     * @param enumClass enum type
033     */
034    public <T extends Enum<T>> EnumComboBox(Class<T> enumClass) {
035        this(enumClass, false);
036    }
037
038    /**
039     * Convenience constructor.
040     *
041     * <p>
042     * Equivalent to:
043     *  <blockquote><code>
044     *  EnumComboBox(new EnumContainer(type), EnumContainer.TO_STRING_PROPERTY, allowNull);
045     *  </code></blockquote>
046     * </p>
047     *
048     * @param enumClass enum type
049     * @param allowNull true to allow a null selection, false otherwise
050     */
051    public <T extends Enum<T>> EnumComboBox(Class<T> enumClass, boolean allowNull) {
052        this(enumClass, EnumContainer.TO_STRING_PROPERTY, allowNull);
053    }
054
055    /**
056     * Convenience constructor.
057     *
058     * <p>
059     * Equivalent to:
060     *  <blockquote><code>
061     *  EnumComboBox(new EnumContainer(type), displayPropertyName, false);
062     *  </code></blockquote>
063     * </p>
064     *
065     * @param enumClass enum type
066     * @param displayPropertyName container property to display in the combo box
067     */
068    public <T extends Enum<T>> EnumComboBox(Class<T> enumClass, String displayPropertyName) {
069        this(enumClass, displayPropertyName, false);
070    }
071
072    /**
073     * Convenience constructor.
074     *
075     * <p>
076     * Equivalent to:
077     *  <blockquote><code>
078     *  EnumComboBox(new EnumContainer(type), displayPropertyName, allowNull);
079     *  </code></blockquote>
080     * </p>
081     *
082     * @param enumClass enum type
083     * @param displayPropertyName container property to display in the combo box
084     * @param allowNull true to allow a null selection, false otherwise
085     */
086    public <T extends Enum<T>> EnumComboBox(Class<T> enumClass, String displayPropertyName, boolean allowNull) {
087        this(new EnumContainer<T>(enumClass), displayPropertyName, allowNull);
088    }
089
090    /**
091     * Primary constructor.
092     *
093     * @param container container data source
094     * @param displayPropertyName container property to display in the combo box
095     * @param allowNull true to allow a null selection, false otherwise
096     */
097    public EnumComboBox(EnumContainer<?> container, String displayPropertyName, boolean allowNull) {
098        this.setContainerDataSource(container);
099        this.setNewItemsAllowed(false);
100        this.setTextInputAllowed(false);
101        this.setFilteringMode(FilteringMode.OFF);
102        this.setItemCaptionMode(ItemCaptionMode.PROPERTY);
103        this.setItemCaptionPropertyId(displayPropertyName);
104
105        // Set up whether null selection is allowed
106        this.setNullSelectionAllowed(allowNull);
107        if (!allowNull)
108            this.setValue(this.getContainerDataSource().getItemIds().iterator().next());
109    }
110}
