026/**
027 * Support superclass for simple read-only, in-memory {@link Container} implementations where each
028 * {@link Item} in the container is backed by a Java object.
029 *
030 * <p>
031 * This {@link Container}'s {@link Property}s are defined via {@link PropertyDef}s, and a {@link PropertyExtractor}
032 * is used to actually extract the property values from each underlying object (alternately, subclasses can override
033 * {@link #getPropertyValue getPropertyValue()}). However, the easist way to configure the container {@link Property}s
034 * is to pass a {@link ProvidesProperty &#64;ProvidesProperty}-annotated Java class to the {@link #AbstractSimpleContainer(Class)}
035 * constructor.
036 * </p>
037 *
038 * <p>
039 * Use {@link #load load()} to load or reload the container.
040 * </p>
041 *
042 * @param <I> the item ID type
043 * @param <T> the type of the Java objects that back each {@link Item} in the container
044 *
045 * @see SimpleItem
046 */
047@SuppressWarnings("serial")
048public abstract class AbstractSimpleContainer<I, T> extends AbstractInMemoryContainer<I, String, BackedItem<T>>
049  implements PropertyExtractor<T>, Container.Filterable, Container.SimpleFilterable, Container.Sortable, Connectable {
050
051    private final HashMap<String, PropertyDef<?>> propertyMap = new HashMap<>();
052    private final HashMap<Object, BackedItem<T>> itemMap = new HashMap<>();
053
054    private PropertyExtractor<? super T> propertyExtractor;
055

059     * Constructor.

102    /**
103     * Constructor.
104     *
105     * @param propertyExtractor used to extract properties from the underlying Java objects;
106     *  may be null but then container is not usable until one is configured via
107     * {@link #setPropertyExtractor setPropertyExtractor()}
108     * @param propertyDefs container property definitions; null is treated like the empty set
109     */
110    protected AbstractSimpleContainer(PropertyExtractor<? super T> propertyExtractor,
111      Collection<? extends PropertyDef<?>> propertyDefs) {
112        this.setItemSorter(new SimpleItemSorter());
113        this.setPropertyExtractor(propertyExtractor);
114        this.setProperties(propertyDefs);
115    }

117    /**
118     * Constructor.
119     *
120     * <p>
121     * Properties will be determined by the {@link ProvidesProperty &#64;ProvidesProperty} and
122     * {@link ProvidesPropertySort &#64;ProvidesPropertySort} annotated methods in the given class.
123     * </p>
124     *
125     * @param type class to introspect for annotated methods
126     * @throws IllegalArgumentException if {@code type} is null
127     * @throws IllegalArgumentException if {@code type} has two {@link ProvidesProperty &#64;ProvidesProperty}
128     *  or {@link ProvidesPropertySort &#64;ProvidesPropertySort} annotated methods for the same property
129     * @throws IllegalArgumentException if a {@link ProvidesProperty &#64;ProvidesProperty}-annotated method with no
130     *  {@linkplain ProvidesProperty#value property name specified} has a name which cannot be interpreted as a bean
131     *  property "getter" method
132     * @see ProvidesProperty
133     * @see ProvidesPropertySort
134     * @see ProvidesPropertyScanner
135     */
136    @SuppressWarnings({ "unchecked", "rawtypes" })
137    protected AbstractSimpleContainer(Class<? super T> type) {
138        // Why the JLS forces this stupid cast:
139        //  http://stackoverflow.com/questions/4902723/why-cant-a-java-type-parameter-have-a-lower-bound
140        final ProvidesPropertyScanner<? super T> propertyReader
141          = (ProvidesPropertyScanner<? super T>)new ProvidesPropertyScanner(type);
142        this.setItemSorter(new SimpleItemSorter());
143        this.setPropertyExtractor(propertyReader.getPropertyExtractor());
144        this.setProperties(propertyReader.getPropertyDefs());
145    }

167    /**
168     * Read the value of the property defined by {@code propertyDef} from the given object.
169     *
170     * <p>
171     * The implementation in {@link AbstractSimpleContainer} just delegates to the {@linkplain #setPropertyExtractor configured}
172     * {@link PropertyExtractor}; subclasses may override to customize property extraction.
173     * </p>
174     *
175     * @param obj Java object
176     * @param propertyDef definition of which property to read
177     * @throws NullPointerException if either parameter is null
178     * @throws IllegalStateException if no {@link PropertyExtractor} is configured for this container
179     */
180    @Override
181    public <V> V getPropertyValue(T obj, PropertyDef<V> propertyDef) {
182        if (this.propertyExtractor == null)
183            throw new IllegalStateException("no PropertyExtractor is configured for this container");
184        return this.propertyExtractor.getPropertyValue(obj, propertyDef);
185    }
186
187    /**
188     * Change the configured properties of this container.
189     *
190     * @param propertyDefs container property definitions; null is treated like the empty set
191     * @throws IllegalArgumentException if {@code propertyDefs} contains a property with a duplicate name
192     */
193    public void setProperties(Collection<? extends PropertyDef<?>> propertyDefs) {
194        if (propertyDefs == null)
195            propertyDefs = Collections.<PropertyDef<?>>emptySet();
196        this.propertyMap.clear();
197        for (PropertyDef<?> propertyDef : propertyDefs) {
198            if (this.propertyMap.put(propertyDef.getName(), propertyDef) != null)
199                throw new IllegalArgumentException("duplicate property name `" + propertyDef.getName() + "'");
200        }
201        this.fireContainerPropertySetChange();
202    }
203
204    /**
205     * Add or replace a configured property of this container.
206     *
207     * @param propertyDef new container property definitions
208     * @throws IllegalArgumentException if {@code propertyDef} is null
209     */
210    public void setProperty(PropertyDef<?> propertyDef) {
211        if (propertyDef == null)
212            throw new IllegalArgumentException("null propertyDef");
213        this.propertyMap.put(propertyDef.getName(), propertyDef);
214        this.fireContainerPropertySetChange();
215    }

217    /**
218     * Change this container's contents.
219     *
220     * @param contents new container contents
221     * @throws IllegalArgumentException if {@code contents} or any item in {@code contents} is null
222     */
223    public void load(Iterable<? extends T> contents) {
224
225        // Sanity check
226        if (contents == null)
227            throw new IllegalArgumentException("null contents");
228
229        // Reset item IDs
230        this.resetItemIds();
231        this.internalRemoveAllItems();
232
233        // Bulk load and register items with id's 0, 1, 2, ...
234        int index = 0;
235        for (T obj : contents) {
236            if (obj == null)
237                throw new IllegalArgumentException("null item in contents at index " + index);
238            final BackedItem<T> item = this.createBackedItem(obj, this.propertyMap.values(), this);
239            this.internalAddItemAtEnd(this.generateItemId(obj), item, false);
240            index++;
241        }
242
243        // Apply filters
244        this.filterAll();
245
246        // Notify subclass
247        this.afterReload();
248
249        // Fire event
250        this.fireItemSetChange();
251    }
252
253    @Override
254    protected void internalRemoveAllItems() {
255        super.internalRemoveAllItems();
256        this.itemMap.clear();
257    }

259    /**
260     * Get the container item ID corresponding to the given underlying Java object which is wrapped by this container.
261     * Objects are tested for equality using {@link Object#equals Object.equals()}.
262     *
263     * <p>
264     * The implementation in {@link AbstractSimpleContainer} requires a linear search of the container.
265     * Some subclasses may provide a more efficient implementation.
266     * </p>
267     *
268     * <p>
269     * This method is not used by this class but is defined as a convenience for subclasses.
270     * </p>
271     *
272     * <p>
273     * Note: items that are filtered out will not be found.
274     *
275     * @param obj underlying container object
276     * @return item ID corresponding to {@code object}, or null if {@code object} is not found in this container
277     * @throws IllegalArgumentException if {@code object} is null
278     * @see #getItemIdForSame
279     */
280    public I getItemIdFor(T obj) {
281        if (obj == null)
282            throw new IllegalArgumentException("null object");
283        for (I itemId : this.getItemIds()) {
284            T candidate = this.getJavaObject(itemId);
285            if (obj.equals(candidate))
286                return itemId;
287        }
288        return null;
289    }

291    /**
292     * Get the container item ID corresponding to the given underlying Java object which is wrapped by this container.
293     * Objects are tested for equality using object equality, not {@link Object#equals Object.equals()}.
294     *
295     * <p>
296     * The implementation in {@link AbstractSimpleContainer} requires a linear search of the container.
297     * Some subclasses may provide a more efficient implementation.
298     * </p>
299     *
300     * <p>
301     * This method is not used by this class but is defined as a convenience for subclasses.
302     * </p>
303     *
304     * <p>
305     * Note: items that are filtered out will not be found.
306     *
307     * @param obj underlying container object
308     * @return item ID corresponding to {@code object}, or null if {@code object} is not found in this container
309     * @throws IllegalArgumentException if {@code object} is null
310     * @see #getItemIdFor
311     */
312    public I getItemIdForSame(T obj) {
313        if (obj == null)
314            throw new IllegalArgumentException("null object");
315        for (I itemId : this.getItemIds()) {
316            T candidate = this.getJavaObject(itemId);
317            if (obj == candidate)
318                return itemId;
319        }
320        return null;
321    }

385    @Override
386    public BackedItem<T> getUnfilteredItem(Object itemId) {
387        final T obj = this.getJavaObject(itemId);
388        if (obj == null)
389            return null;
390        BackedItem<T> item = this.itemMap.get(itemId);
391        if (item == null) {
392            item = this.createBackedItem(obj, this.propertyMap.values(), this);
393            this.itemMap.put(itemId, item);
394        }
395        return item;
396    }

497    @Override
498    public Collection<?> getSortableContainerPropertyIds() {
499        final ArrayList<String> propertyIds = new ArrayList<>(this.propertyMap.size());
500        for (Map.Entry<String, PropertyDef<?>> entry : this.propertyMap.entrySet()) {
501            if (this.propertyExtractor instanceof SortingPropertyExtractor) {
502                final SortingPropertyExtractor<? super T> sortingPropertyExtractor
503                  = (SortingPropertyExtractor<? super T>)this.propertyExtractor;
504                if (sortingPropertyExtractor.canSort(entry.getValue())) {
505                    propertyIds.add(entry.getKey());
506                    continue;
507                }
508            }
509            if (entry.getValue().isSortable())
510                propertyIds.add(entry.getKey());
511        }
512        return propertyIds;
513    }

517    /**
518     * {@link ItemSorter} implementation used by {@link AbstractSimpleContainer}.
519     */
520    private class SimpleItemSorter extends DefaultItemSorter {
521
522        @Override
523        @SuppressWarnings("unchecked")
524        protected int compareProperty(Object propertyId, boolean ascending, Item item1, Item item2) {
525
526            // Get property definition
527            final PropertyDef<?> propertyDef = AbstractSimpleContainer.this.propertyMap.get(propertyId);
528            if (propertyDef == null)
529                return super.compareProperty(propertyId, ascending, item1, item2);
530
531            // Ask SortingPropertyExtractor if we have one
532            if (AbstractSimpleContainer.this.propertyExtractor instanceof SortingPropertyExtractor) {
533                final SortingPropertyExtractor<? super T> sortingPropertyExtractor
534                  = (SortingPropertyExtractor<? super T>)AbstractSimpleContainer.this.propertyExtractor;
535                if (sortingPropertyExtractor.canSort(propertyDef)) {
536                    final T obj1 = ((BackedItem<T>)item1).getObject();
537                    final T obj2 = ((BackedItem<T>)item2).getObject();
538                    final int diff = sortingPropertyExtractor.sort(propertyDef, obj1, obj2);
539                    return ascending ? diff : -diff;
540                }
541            }
542
543            // Ask property definition
544            if (propertyDef.isSortable()) {
545                final int diff = this.sort(propertyDef, item1, item2);
546                return ascending ? diff : -diff;
547            }
548
549            // Defer to superclass
550            return super.compareProperty(propertyId, ascending, item1, item2);
551        }
552
553        // This method exists only to allow the generic parameter <V> to be bound
554        private <V> int sort(PropertyDef<V> propertyDef, Item item1, Item item2) {
555            return propertyDef.sort(propertyDef.read(item1), propertyDef.read(item2));
556        }
557    }
