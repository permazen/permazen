008    package org.dellroad.stuff.vaadin7;
009    
010    import com.vaadin.server.SessionDestroyEvent;
011    import com.vaadin.server.SessionDestroyListener;
012    import com.vaadin.server.VaadinServiceSession;
013    
014    import java.util.HashMap;
015    
016    import org.springframework.beans.factory.ObjectFactory;
017    import org.springframework.beans.factory.config.BeanFactoryPostProcessor;
018    import org.springframework.beans.factory.config.ConfigurableListableBeanFactory;
019    import org.springframework.beans.factory.config.Scope;
020    
021    /**
022     * A Spring custom {@link Scope} for Vaadin applications.
023     *
024     * <p>
025     * This scopes beans to the lifetime of the {@link VaadinServiceSession} (formerly known as "Vaadin application").
026     * Spring {@linkplain org.springframework.beans.factory.DisposableBean#destroy destroy-methods}
027     * will be invoked when the {@link VaadinServiceSession} is closed.
028     * </p>
029     *
030     * <p>
031     * To enable this scope, simply add this bean to your application context as a singleton (it will register itself):
032     * <blockquote><pre>
033     *  &lt;!-- Enable the "vaadinApplication" custom scope --&gt;
034     *  &lt;bean class="org.dellroad.stuff.vaadin.VaadinApplicationScope"/&gt;
035     * </pre></blockquote>
036     * Then declare scoped beans normally using the scope name {@code "vaadinApplication"}.
037     * </p>
038     */
039    public class VaadinApplicationScope implements Scope, BeanFactoryPostProcessor, SessionDestroyListener {
040    
041        /**
042         * Key to the current {@link VaadinServiceSession} instance. For use by {@link #resolveContextualObject}.
043         */
044        public static final String VAADIN_SERVICE_SESSION_KEY = "vaadinServiceSession";
045    
046        /**
047         * The name of this scope (i.e., <code>{@value}</code>).
048         */
049        public static final String SCOPE_NAME = "vaadinApplication";
050    
051        private final HashMap<VaadinServiceSession, SessionBeanHolder> beanHolders
052          = new HashMap<VaadinServiceSession, SessionBeanHolder>();
053    
054    // BeanFactoryPostProcessor methods
055    
056        @Override
057        public void postProcessBeanFactory(ConfigurableListableBeanFactory beanFactory) {
058            beanFactory.registerScope(VaadinApplicationScope.SCOPE_NAME, this);
059        }
060    
061    // SessionDestroyListener
062    
063        @Override
064        public void sessionDestroy(SessionDestroyEvent event) {
065            final VaadinServiceSession session = event.getSession();
066            SessionBeanHolder beanHolder;
067            synchronized (this) {
068                beanHolder = this.beanHolders.remove(session);
069            }
070            if (beanHolder != null)
071                beanHolder.close();
072        }
073    
074    // Scope methods
075    
076        @Override
077        public synchronized Object get(String name, ObjectFactory<?> objectFactory) {
078            return this.getSessionBeanHolder(true).getBean(name, objectFactory);
079        }
080    
081        @Override
082        public synchronized Object remove(String name) {
083            SessionBeanHolder beanHolder = this.getSessionBeanHolder(false);
084            return beanHolder != null ? beanHolder.remove(name) : null;
085        }
086    
087        @Override
088        public synchronized void registerDestructionCallback(String name, Runnable callback) {
089            this.getSessionBeanHolder(true).registerDestructionCallback(name, callback);
090        }
091    
092        @Override
093        public String getConversationId() {
094            VaadinServiceSession session = VaadinServiceSession.getCurrent();
095            if (session == null)
096                return null;
097            return session.getClass().getName() + "@" + System.identityHashCode(session);
098        }
099    
100        @Override
101        public Object resolveContextualObject(String key) {
102            if (VAADIN_SERVICE_SESSION_KEY.equals(key))
103                return VaadinServiceSession.getCurrent();
104            return null;
105        }
106    
107    // Internal methods
108    
109        private synchronized SessionBeanHolder getSessionBeanHolder(boolean create) {
110            VaadinServiceSession session = VaadinUtil.getCurrentSession();
111            VaadinUtil.addSessionDestroyListener(session, this);
112            SessionBeanHolder beanHolder = this.beanHolders.get(session);
113            if (beanHolder == null && create) {
114                beanHolder = new SessionBeanHolder(session);
115                this.beanHolders.put(session, beanHolder);
116            }
117            return beanHolder;
118        }
119    
120    // Bean holder class corresponding to a single Application instance
121    
122        private static class SessionBeanHolder {
123    
124            private final HashMap<String, Object> beans = new HashMap<String, Object>();
125            private final HashMap<String, Runnable> destructionCallbacks = new HashMap<String, Runnable>();
126            private final VaadinServiceSession session;
127    
128            public SessionBeanHolder(VaadinServiceSession session) {
129                this.session = session;
130            }
131    
132            public Object getBean(String name, ObjectFactory<?> objectFactory) {
133                Object bean = this.beans.get(name);
134                if (bean == null) {
135                    bean = objectFactory.getObject();
136                    this.beans.put(name, bean);
137                }
138                return bean;
139            }
140    
141            public Object remove(String name) {
142                this.destructionCallbacks.remove(name);
143                return this.beans.remove(name);
144            }
145    
146            public void registerDestructionCallback(String name, Runnable callback) {
147                this.destructionCallbacks.put(name, callback);
148            }
149    
150            public void close() {
151                for (Runnable callback : this.destructionCallbacks.values())
152                    callback.run();
153                this.beans.clear();
154                this.destructionCallbacks.clear();
155            }
156    
157            public boolean isEmpty() {
158                return this.beans.isEmpty();
159            }
160        }
161    }
