010import java.lang.reflect.Method;
011import java.util.ArrayList;
012import java.util.Collections;
013import java.util.Comparator;
014import java.util.HashMap;
015import java.util.List;
016import java.util.Map;
017import java.util.Set;
018
019import org.dellroad.stuff.java.MethodAnnotationScanner;
020import org.dellroad.stuff.java.Primitive;
021
022/**
023 * Scans a Java class hierarchy for {@link ProvidesProperty &#64;ProvidesProperty} and
024 * {@link ProvidesPropertySort &#64;ProvidesPropertySort} annotated getter methods and creates
025 * a corresponding set of {@link PropertyDef} property definitions and a {@link SortingPropertyExtractor}
026 * that will extract the properties from instances of the given class and sort them accordingly.
027 *
028 * @param <T> Java class to be introspected
029 * @see ProvidesProperty &#64;ProvidesProperty
030 * @see ProvidesPropertySort &#64;ProvidesPropertySort
031 */
032public class ProvidesPropertyScanner<T> {
033
034    private final ArrayList<PropertyDef<?>> propertyList = new ArrayList<>();
035    private final SortingPropertyExtractor<T> propertyExtractor;
036
037    /**
038     * Constructor.
039     *
040     * @param type Java class to be introspected
041     * @throws IllegalArgumentException if {@code type} is null
042     * @throws IllegalArgumentException if an annotated method with no {@linkplain ProvidesProperty#value property name specified}
043     *  has a name which cannot be interpreted as a bean property "getter" method
044     * @throws IllegalArgumentException if {@code type} has two {@link ProvidesProperty &#64;ProvidesProperty}-annotated
045     *  fields or methods with the same {@linkplain ProvidesProperty#value property name}
046     */
047    public ProvidesPropertyScanner(Class<T> type) {
048
049        // Sanity check
050        if (type == null)
051            throw new IllegalArgumentException("null type");
052
053        // Scan for @ProvidesProperty and @ProvidesPropertySort annotations
054        final Set<MethodAnnotationScanner<T, ProvidesProperty>.MethodInfo> providesPropertyMethods
055          = new MethodAnnotationScanner<T, ProvidesProperty>(type, ProvidesProperty.class).findAnnotatedMethods();
056        final Set<MethodAnnotationScanner<T, ProvidesPropertySort>.MethodInfo> providesPropertySortMethods
057          = new MethodAnnotationScanner<T, ProvidesPropertySort>(type, ProvidesPropertySort.class).findAnnotatedMethods();
058
059        // Check for duplicate @ProvidesProperty names
060        final HashMap<String, MethodAnnotationScanner<T, ProvidesProperty>.MethodInfo> providesPropertyNameMap = new HashMap<>();
061        for (MethodAnnotationScanner<T, ProvidesProperty>.MethodInfo methodInfo : providesPropertyMethods) {
062            final String propertyName = this.getPropertyName(methodInfo);
063
064            // Check for name conflict
065            final MethodAnnotationScanner<T, ProvidesProperty>.MethodInfo previousInfo = providesPropertyNameMap.get(propertyName);
066            if (previousInfo == null) {
067                providesPropertyNameMap.put(propertyName, methodInfo);
068                continue;
069            }
070
071            // If there is a name conflict, the sub-type method declaration wins
072            switch (this.compareDeclaringClass(previousInfo.getMethod(), methodInfo.getMethod())) {
073            case 0:
074                throw new IllegalArgumentException("duplicate @" + ProvidesProperty.class.getSimpleName()
075                  + " declaration for property `" + propertyName + "' on method " + previousInfo.getMethod()
076                  + " and " + methodInfo.getMethod() + " declared in the same class");
077            case 1:
078                providesPropertyNameMap.put(propertyName, methodInfo);
079                break;
080            default:
081                break;
082            }
083        }
084
085        // Check for duplicate @ProvidesPropertySort names, etc.
086        final HashMap<String, MethodAnnotationScanner<T, ProvidesPropertySort>.MethodInfo> providesPropertySortNameMap
087          = new HashMap<>();
088        for (MethodAnnotationScanner<T, ProvidesPropertySort>.MethodInfo methodInfo : providesPropertySortMethods) {
089            final String propertyName = this.getSortPropertyName(methodInfo);
090
091            // Verify the method's return type (or wrapper type if primitive) implements Comparable or Comparator
092            Class<?> methodType = methodInfo.getMethod().getReturnType();
093            if (methodType.isPrimitive())
094                methodType = Primitive.get(methodType).getWrapperType();
095            if (!Comparable.class.isAssignableFrom(methodType) && !Comparator.class.isAssignableFrom(methodType)) {
096                throw new IllegalArgumentException("invalid @" + ProvidesPropertySort.class.getSimpleName()
097                  + " declaration for property `" + propertyName + "': method " + methodInfo.getMethod()
098                  + " return type " + methodType.getName() + " implements neither " + Comparable.class.getName()
099                  + " nor " + Comparator.class.getName());
100            }
101
102            // Check for name conflict
103            final MethodAnnotationScanner<T, ProvidesPropertySort>.MethodInfo previousInfo
104              = providesPropertySortNameMap.get(propertyName);
105            if (previousInfo == null) {
106                providesPropertySortNameMap.put(propertyName, methodInfo);
107                continue;
108            }
109
110            // If there is a name conflict, the sub-type method declaration wins
111            switch (this.compareDeclaringClass(previousInfo.getMethod(), methodInfo.getMethod())) {
112            case 0:
113                throw new IllegalArgumentException("duplicate @" + ProvidesPropertySort.class.getSimpleName()
114                  + " declaration for property `" + propertyName + "' on method " + previousInfo.getMethod()
115                  + " and " + methodInfo.getMethod() + " declared in the same class");
116            case 1:
117                providesPropertySortNameMap.put(propertyName, methodInfo);
118                break;
119            default:
120                break;
121            }
122        }
123
124        // Build PropertyDef list
125        for (Map.Entry<String, MethodAnnotationScanner<T, ProvidesProperty>.MethodInfo> e : providesPropertyNameMap.entrySet()) {
126            final String propertyName = e.getKey();
127            final MethodAnnotationScanner<T, ProvidesProperty>.MethodInfo methodInfo = e.getValue();
128
129            // Get property type
130            Class<?> propertyType = methodInfo.getMethod().getReturnType();
131
132            // Get property default value
133            Object defaultValue = null;
134            if (propertyType.isPrimitive()) {
135                final Primitive<?> primitiveType = Primitive.get(propertyType);
136                defaultValue = primitiveType.getDefaultValue();
137                propertyType = primitiveType.getWrapperType();
138            }
139
140            // Get sort property (if any)
141            final MethodAnnotationScanner<T, ProvidesPropertySort>.MethodInfo sortMethodInfo
142              = providesPropertySortNameMap.get(propertyName);
143
144            // Create property definition
145            this.propertyList.add(this.createAnnotationPropertyDef(
146              propertyName, propertyType, defaultValue, methodInfo, sortMethodInfo));
147        }
148
149        // Build PropertyExtractor
150        this.propertyExtractor = new SortingPropertyExtractor<T>() {
151
152            @Override
153            @SuppressWarnings("unchecked")
154            public <V> V getPropertyValue(T obj, PropertyDef<V> propertyDef) {
155                if (!(propertyDef instanceof AnnotationPropertyDef))
156                    throw new RuntimeException("unknown property " + propertyDef);
157                final AnnotationPropertyDef<V> annotationPropertyDef = (AnnotationPropertyDef<V>)propertyDef;
158                return propertyDef.getType().cast(annotationPropertyDef.getMethodInfo().invoke(obj));
159            }
160
161            @Override
162            @SuppressWarnings("unchecked")
163            public boolean canSort(PropertyDef<?> propertyDef) {
164                if (!(propertyDef instanceof AnnotationPropertyDef))
165                    return false;
166                final AnnotationPropertyDef<?> annotationPropertyDef = (AnnotationPropertyDef)propertyDef;
167                return annotationPropertyDef.getSortMethodInfo() != null;
168            }
169
170            @Override
171            @SuppressWarnings("unchecked")
172            public int sort(PropertyDef<?> propertyDef, T obj1, T obj2) {
173                if (!(propertyDef instanceof AnnotationPropertyDef))
174                    throw new UnsupportedOperationException("unknown property " + propertyDef);
175                final AnnotationPropertyDef<?> annotationPropertyDef = (AnnotationPropertyDef<?>)propertyDef;
176                return annotationPropertyDef.getComparator().compare(obj1, obj2);
177            }
178        };
179    }
180
181    /**
182     * Get the list of {@link PropertyDef}s generated from the annotated methods.
183     *
184     * <p>
185     * All of the properties in the returned list can be extracted from instances of this reader's configured
186     * class by the {@link PropertyExtractor} returned by {@link #getPropertyExtractor}.
187     * </p>
188     *
189     * @return unmodifiable list of properties
190     * @see #getPropertyExtractor
191     */
192    public List<PropertyDef<?>> getPropertyDefs() {
193        return Collections.unmodifiableList(this.propertyList);
194    }
195
196    /**
197     * Get the {@link PropertyExtractor} that extracts Vaadin {@link com.vaadin.data.Property} values
198     * from instances of the annotated class when given one of the {@link PropertyDef}s returned by
199     * {@link #getPropertyDefs}.
200     *
201     * @see #getPropertyDefs
202     */
203    public SortingPropertyExtractor<T> getPropertyExtractor() {
204        return this.propertyExtractor;
205    }
206
207    private String getPropertyName(MethodAnnotationScanner<T, ProvidesProperty>.MethodInfo methodInfo) {
208        return methodInfo.getAnnotation().value().length() > 0 ?
209          methodInfo.getAnnotation().value() : methodInfo.getMethodPropertyName();
210    }
211
212    private String getSortPropertyName(MethodAnnotationScanner<T, ProvidesPropertySort>.MethodInfo methodInfo) {
213        return methodInfo.getAnnotation().value().length() > 0 ?
214          methodInfo.getAnnotation().value() : methodInfo.getMethodPropertyName();
215    }
216
217    // This method exists solely to bind the generic type
218    private <V> AnnotationPropertyDef<V> createAnnotationPropertyDef(String propertyName, Class<V> propertyType,
219      Object defaultValue, MethodAnnotationScanner<T, ProvidesProperty>.MethodInfo methodInfo,
220      MethodAnnotationScanner<T, ProvidesPropertySort>.MethodInfo sortMethodInfo) {
221        return new AnnotationPropertyDef<V>(propertyName, propertyType,
222          propertyType.cast(defaultValue), methodInfo, sortMethodInfo);
223    }
224
225    // Compare two methods to determine which one has the declaring class that is a sub-type of the other's
226    private int compareDeclaringClass(Method method1, Method method2) {
227        final Class<?> class1 = method1.getDeclaringClass();
228        final Class<?> class2 = method2.getDeclaringClass();
229        if (class1 == class2)
230            return 0;
231        if (class1.isAssignableFrom(class2))
232            return 1;
233        if (class2.isAssignableFrom(class1))
234            return -1;
235        throw new RuntimeException("internal error: incomparable classes " + class1.getName() + " and " + class2.getName());
236    }
237
238    private class AnnotationPropertyDef<V> extends PropertyDef<V> {
239
240        private final MethodAnnotationScanner<T, ProvidesProperty>.MethodInfo methodInfo;
241        private final MethodAnnotationScanner<T, ProvidesPropertySort>.MethodInfo sortMethodInfo;
242
243        private Comparator<T> comparator;
244
245        AnnotationPropertyDef(String name, Class<V> type, V defaultValue,
246          MethodAnnotationScanner<T, ProvidesProperty>.MethodInfo methodInfo,
247          MethodAnnotationScanner<T, ProvidesPropertySort>.MethodInfo sortMethodInfo) {
248            super(name, type, defaultValue);
249            this.methodInfo = methodInfo;
250            this.sortMethodInfo = sortMethodInfo;
251        }
252
253        public MethodAnnotationScanner<T, ProvidesProperty>.MethodInfo getMethodInfo() {
254            return this.methodInfo;
255        }
256
257        public MethodAnnotationScanner<T, ProvidesPropertySort>.MethodInfo getSortMethodInfo() {
258            return this.sortMethodInfo;
259        }
260
261        public Comparator<T> getComparator() {
262
263            // Sanity check
264            if (this.sortMethodInfo == null)
265                throw new UnsupportedOperationException("can't sort property " + this);
266
267            // Already created?
268            if (this.comparator != null)
269                return this.comparator;
270
271            // Handle Comparator case
272            if (Comparator.class.isAssignableFrom(this.sortMethodInfo.getMethod().getReturnType())) {
273                this.comparator = new Comparator<T>() {
274
275                    private Comparator<T> comparator;
276
277                    @Override
278                    @SuppressWarnings("unchecked")
279                    public int compare(T obj1, T obj2) {
280                        if (this.comparator == null)
281                            this.comparator = (Comparator<T>)AnnotationPropertyDef.this.sortMethodInfo.invoke(obj1);
282                        return this.comparator.compare(obj1, obj2);
283                    }
284                };
285                return this.comparator;
286            }
287
288            // Handle Comparable case
289            this.comparator = new Comparator<T>() {
290                @Override
291                @SuppressWarnings("unchecked")
292                public int compare(T obj1, T obj2) {
293                    final Comparable<Object> value1 = (Comparable<Object>)AnnotationPropertyDef.this.sortMethodInfo.invoke(obj1);
294                    final Comparable<Object> value2 = (Comparable<Object>)AnnotationPropertyDef.this.sortMethodInfo.invoke(obj2);
295                    if (value1 == null && value2 != null)
296                        return -1;
297                    if (value1 != null && value2 == null)
298                        return 1;
299                    if (value1 == null && value2 == null)
300                        return 0;
301                    return value1.compareTo(value2);
302                }
303            };
304            return this.comparator;
305        }
306    }
