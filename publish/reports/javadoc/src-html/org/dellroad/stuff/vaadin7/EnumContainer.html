008package org.dellroad.stuff.vaadin7;
009
010import java.util.List;
011
012import org.dellroad.stuff.java.EnumUtil;
013
014/**
015 * Container backed by the instances of an {@link Enum} type.
016 *
017 * <p>
018 * Instances will auto-detect properties defined by {@link ProvidesProperty &#64;ProvidesProperty} annotations (if any),
019 * and will also pre-define the following properties:
020 *  <ul>
021 *  <li>{@link #NAME_PROPERTY "name"} - {@link String} property derived from {@link Enum#name Enum.name()}</li>
022 *  <li>{@link #VALUE_PROPERTY "value"} - type {@code T} property that returns the {@link Enum} instance value itself</li>
023 *  <li>{@link #ORDINAL_PROPERTY "ordinal"} - {@link Integer} property derived from {@link Enum#ordinal Enum.ordinal()}</li>
024 *  <li>{@link #TO_STRING_PROPERTY "toString"} - {@link String} property derived from {@link Enum#toString}</li>
025 *  </ul>
026 * </p>
027 *
028 * @param <T> enum type
029 * @see EnumComboBox
030 */
031@SuppressWarnings("serial")
032public class EnumContainer<T extends Enum<T>> extends SelfKeyedContainer<T> {
033
034    public static final PropertyDef<String> NAME_PROPERTY = new PropertyDef<String>("name", String.class);
035    @SuppressWarnings("rawtypes")
036    public static final PropertyDef<Enum> VALUE_PROPERTY = new PropertyDef<Enum>("value", Enum.class);
037    public static final PropertyDef<Integer> ORDINAL_PROPERTY = new PropertyDef<Integer>("ordinal", Integer.class);
038    public static final PropertyDef<String> TO_STRING_PROPERTY = new PropertyDef<String>("toString", String.class);
039
040    /**
041     * Constructor.
042     *
043     * @param type enum type
044     * @throws IllegalArgumentException if {@code type} is null
045     * @throws IllegalArgumentException if {@code type} has a {@link ProvidesProperty &#64;ProvidesProperty}-annotated method
046     *  with no {@linkplain ProvidesProperty#value property name specified} has a name which cannot be interpreted as a bean
047     *  property "getter" method
048     * @throws IllegalArgumentException if {@code type} has two {@link ProvidesProperty &#64;ProvidesProperty}-annotated
049     *  fields or methods with the same {@linkplain ProvidesProperty#value property name}
050     */
051    public EnumContainer(final Class<T> type) {
052        super(type);
053        this.setProperty(NAME_PROPERTY);
054        this.setProperty(VALUE_PROPERTY);
055        this.setProperty(ORDINAL_PROPERTY);
056        this.setProperty(TO_STRING_PROPERTY);
057        this.load(this.getExposedValues(type));
058    }
059
060    /**
061     * Get the {@link Enum} values to expose in this container in the desired order.
062     *
063     * <p>
064     * The implementation in {@link EnumContainer} returns all values in their natural order.
065     * Subclasses may override to filter and/or re-order.
066     * </p>
067     */
068    protected List<T> getExposedValues(Class<T> type) {
069        return EnumUtil.getValues(type);
070    }
071
072    @Override
073    public <V> V getPropertyValue(T value, PropertyDef<V> propertyDef) {
074        if (propertyDef.equals(VALUE_PROPERTY))
075            return propertyDef.getType().cast(value);
076        if (propertyDef.equals(VALUE_PROPERTY))
077            return propertyDef.getType().cast(value);
078        if (propertyDef.equals(ORDINAL_PROPERTY))
079            return propertyDef.getType().cast(value.ordinal());
080        if (propertyDef.equals(TO_STRING_PROPERTY))
081            return propertyDef.getType().cast(value.toString());
082        return super.getPropertyValue(value, propertyDef);
083    }
084}
