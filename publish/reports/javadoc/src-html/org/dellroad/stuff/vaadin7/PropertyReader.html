008package org.dellroad.stuff.vaadin7;
009
010import java.beans.Introspector;
011import java.lang.reflect.AccessibleObject;
012import java.lang.reflect.Field;
013import java.lang.reflect.InvocationTargetException;
014import java.lang.reflect.Method;
015import java.util.ArrayList;
016import java.util.Collections;
017import java.util.Iterator;
018import java.util.List;
019
020import org.dellroad.stuff.java.Primitive;
021
022/**
023 * Builds a {@link PropertyExtractor} and a list of {@link PropertyDef}s from a class having
024 * {@link ProvidesProperty &#64;ProvidesProperty}-annotated methods by introspecting the class
025 * and its superclasses and superinterfaces.
026 *
027 * <p>
028 * See {@link ProvidesProperty &#64;ProvidesProperty} for further details.
029 * </p>
030 *
031 * @param <T> Java class to be introspected
032 * @see ProvidesProperty &#64;ProvidesProperty
033 */
034public class PropertyReader<T> {
035
036    private final ArrayList<PropertyInfo<?>> propertyList = new ArrayList<PropertyInfo<?>>();
037
038    /**
039     * Constructor.
040     *
041     * @param type Java class to be introspected
042     * @throws IllegalArgumentException if {@code type} is null
043     * @throws IllegalArgumentException if an annotated method with no {@linkplain ProvidesProperty#value property name specified}
044     *  has a name which cannot be interpreted as a bean property "getter" method
045     * @throws IllegalArgumentException if {@code type} has two {@link ProvidesProperty &#64;ProvidesProperty}-annotated
046     *  fields or methods with the same {@linkplain ProvidesProperty#value property name}
047     */
048    public PropertyReader(Class<T> type) {
049
050        // Sanity check
051        if (type == null)
052            throw new IllegalArgumentException("null type");
053
054        // Introspect fields and methods
055        this.findProperties(type);
056
057        // Sort by name - note, sort is stable so subclass methods will appear prior to method(s) they override
058        Collections.sort(this.propertyList);
059
060        // Check for duplicate names, but allow method overrides to declare the same property
061        PropertyInfo<?> previousInfo = null;
062        for (Iterator<PropertyInfo<?>> i = this.propertyList.iterator(); i.hasNext(); ) {
063            final PropertyInfo<?> propertyInfo = i.next();
064
065            // Check for duplicate name
066            if (previousInfo != null && propertyInfo.getName().equals(previousInfo.getName())) {
067
068                // Allow method overrides to declare the same property name
069                final AccessibleObject previousMember = previousInfo.getMember();
070                final AccessibleObject propertyMember = propertyInfo.getMember();
071                if (previousMember instanceof Method && propertyMember instanceof Method) {
072                    i.remove();
073                    continue;
074                }
075
076                // Otherwise, there is a duplicate property name
077                throw new IllegalArgumentException("duplicate property name `" + propertyInfo.getName()
078                  + "' (on " + previousMember + " and " + propertyMember + ")");
079            }
080            previousInfo = propertyInfo;
081        }
082
083        // Trim array
084        this.propertyList.trimToSize();
085    }
086
087    private void findProperties(Class<?> klass) {
088
089        // Stop at null
090        if (klass == null)
091            return;
092
093        // Search fields
094        for (Field field : klass.getDeclaredFields()) {
095            final ProvidesProperty annotation = field.getAnnotation(ProvidesProperty.class);
096            if (annotation != null)
097                this.addPropertyInfo(field, annotation.value(), field.getType());
098        }
099
100        // Search methods
101        for (Method method : klass.getDeclaredMethods()) {
102            if (method.getReturnType() == void.class || method.getParameterTypes().length > 0)
103                continue;
104            final ProvidesProperty annotation = method.getAnnotation(ProvidesProperty.class);
105            if (annotation != null)
106                this.addPropertyInfo(method, annotation.value(), method.getReturnType());
107        }
108
109        // Recurse on interfaces
110        for (Class<?> iface : klass.getInterfaces())
111            this.findProperties(iface);
112
113        // Recurse on superclass
114        this.findProperties(klass.getSuperclass());
115    }
116
117    private void addPropertyInfo(AccessibleObject member, String name, Class<?> type) {
118
119        // Sanity check type
120        if (type == void.class)
121            throw new IllegalArgumentException("property `" + name + "' has void type on " + member);
122
123        // Get (non-primitive) type and default value
124        final Primitive primitiveType = Primitive.get(type);
125        Object defaultValue = null;
126        if (type.isPrimitive()) {
127            defaultValue = primitiveType.getDefaultValue();
128            type = primitiveType.getWrapperType();
129        }
130
131        // Get name, if not explicitly specified
132        if (name.length() == 0) {
133            if (member instanceof Field)
134                name = ((Field)member).getName();
135            else {
136                name = ((Method)member).getName();
137                if (name.startsWith("get") && name.length() > 3)
138                    name = Introspector.decapitalize(name.substring(3));
139                else if (name.startsWith("is") && name.length() > 2 && primitiveType == Primitive.BOOLEAN)
140                    name = Introspector.decapitalize(name.substring(2));
141                else
142                    throw new IllegalArgumentException("can't infer property name from non-getter method " + member);
143            }
144        }
145
146        // Add new property to list
147        this.propertyList.add(this.createPropertyInfo(member, name, type, defaultValue));
148    }
149
150    // This method exists solely to bind the generic type
151    private <V> PropertyInfo<V> createPropertyInfo(AccessibleObject member, String name, Class<V> type, Object defaultValue) {
152        return new PropertyInfo<V>(member, name, type, type.cast(defaultValue));
153    }
154
155    /**
156     * Get a {@link PropertyExtractor} that extracts Vaadin {@link com.vaadin.data.Property} values
157     * from instances of this reader's configured class.
158     */
159    public PropertyExtractor<T> getPropertyExtractor() {
160        return new PropertyExtractor<T>() {
161            @Override
162            public <V> V getPropertyValue(T obj, PropertyDef<V> propertyDef) {
163                for (PropertyInfo<?> propertyInfo : PropertyReader.this.propertyList) {
164                    if (propertyDef.equals(propertyInfo.getPropertyDef()))
165                        return propertyDef.getType().cast(propertyInfo.readValue(obj));
166                }
167                throw new RuntimeException("unknown property " + propertyDef);
168            }
169        };
170    }
171
172    /**
173     * Get the list of {@link PropertyDef}s derived from the {@link ProvidesProperty &#64;ProvidesProperty}-annotated
174     * fields and methods of this instance's configured class.
175     *
176     * <p>
177     * All of the properties in the returned list can be extracted from instances of this reader's configured
178     * class by the {@link PropertyExtractor} returned by {@link #getPropertyExtractor}.
179     * </p>
180     */
181    public List<PropertyDef<?>> getPropertyDefs() {
182        final ArrayList<PropertyDef<?>> result = new ArrayList<PropertyDef<?>>(this.propertyList.size());
183        for (PropertyInfo<?> propertyInfo : this.propertyList)
184            result.add(propertyInfo.getPropertyDef());
185        return result;
186    }
187
188    private class PropertyInfo<V> implements Comparable<PropertyInfo<?>> {
189
190        private final AccessibleObject member;
191        private final PropertyDef<V> propertyDef;
192
193        PropertyInfo(AccessibleObject member, String name, Class<V> type, V defaultValue) {
194            this.propertyDef = new PropertyDef<V>(name, type, defaultValue);
195            this.member = member;
196            try {
197                this.member.setAccessible(true);
198            } catch (SecurityException e) {
199                // ignore
200            }
201        }
202
203        String getName() {
204            return this.propertyDef.getName();
205        }
206
207        AccessibleObject getMember() {
208            return this.member;
209        }
210
211        PropertyDef<V> getPropertyDef() {
212            return this.propertyDef;
213        }
214
215        Object readValue(T obj) {
216            if (this.member instanceof Field) {
217                final Field field = (Field)this.member;
218                try {
219                    return field.get(obj);
220                } catch (IllegalAccessException e) {
221                    throw new RuntimeException(e);
222                }
223            } else {
224                final Method method = (Method)this.member;
225                try {
226                    return method.invoke(obj);
227                } catch (InvocationTargetException e) {
228                    throw new RuntimeException(e);
229                } catch (IllegalAccessException e) {
230                    throw new RuntimeException(e);
231                }
232            }
233        }
234
235        @Override
236        public int compareTo(PropertyInfo<?> that) {
237            return this.getName().compareTo(that.getName());
238        }
239    }
