008package org.dellroad.stuff.vaadin7;
009
010import com.vaadin.server.DeploymentConfiguration;
011import com.vaadin.server.ServiceException;
012import com.vaadin.server.VaadinRequest;
013import com.vaadin.server.VaadinServlet;
014import com.vaadin.server.VaadinServletService;
015import com.vaadin.server.VaadinSession;
016
017import java.util.ArrayList;
018import java.util.List;
019import java.util.Properties;
020import java.util.WeakHashMap;
021
022import javax.servlet.ServletConfig;
023import javax.servlet.ServletException;
024
025/**
026 * A {@link VaadinServlet} that associates and manages a Spring
027 * {@link org.springframework.web.context.ConfigurableWebApplicationContext} with each
028 * {@link com.vaadin.server.VaadinSession} (aka, "Vaadin application" in the old terminology).
029 *
030 * <p>
031 * The {@code vaadinContextConfigLocation} servlet parameter may be used to specify the Spring XML config
032 * file location(s). For example:
033 *
034 * <blockquote><pre>
035 * &lt;servlet&gt;
036 *     &lt;servlet-name&gt;My Vaadin App&lt;/servlet-name&gt;
037 *     &lt;servlet-class&gt;org.dellroad.stuff.vaadin7.SpringVaadinServlet&lt;/servlet-class&gt;
038 *     &lt;init-param&gt;
039 *         &lt;param-name&gt;UI&lt;/param-name&gt;
040 *         &lt;param-value&gt;com.example.MyApplicationUI&lt;/param-value&gt;
041 *     &lt;/init-param&gt;
042 *     &lt;init-param&gt;
043 *         &lt;param-name&gt;configLocation&lt;/param-name&gt;
044 *         &lt;param-value&gt;classpath:com/example/MyApplicationContext.xml&lt;/param-value&gt;
045 *     &lt;/init-param&gt;
046 * &lt;/servlet&gt;
047 * </pre></blockquote>
048 * </p>
049 *
050 * <p>
051 * The main function of this servlet is to create and register a {@link SpringVaadinSessionListener} as a listener on the
052 * {@link com.vaadin.server.VaadinService} associated with this servlet. The {@link SpringVaadinSessionListener} in turn detects
053 * the creation and destruction of Vaadin application instances (represented by {@link com.vaadin.server.VaadinSession}
054 * instances) and does the work of managing the associated Spring application contexts.
055 * </p>
056 *
057 * <p>
058 * Use of this servlet in place of the standard Vaadin servlet is required for the {@link VaadinConfigurable @VaadinConfigurable}
059 * annotation to work.
060 * </p>
061 *
062 * <p>
063 * Supported URL parameters:
064 * <div style="margin-left: 20px;">
065 * <table border="1" cellpadding="3" cellspacing="0">
066 * <tr bgcolor="#ccffcc">
067 *  <th align="left">Parameter Name</th>
068 *  <th align="left">Required?</th>
069 *  <th align="left">Description</th>
070 * </tr>
071 * <tr>
072 * <td>{@code applicationName}</td>
073 * <td align="center">No</td>
074 * <td>
075 *  Vaadin application name. Used for logging purposes and as the name of the XML application context file
076 *  when {@code configLocation} is not specified. If this parameter is not specified, the
077 *  name of the servlet is used.
078 * </td>
079 * </tr>
080 * <tr>
081 * <td>{@code configLocation}</td>
082 * <td align="center">No</td>
083 * <td>
084 *  Location of Spring application context XML file(s). Multiple locations are separated by whitespace.
085 *  If omitted, {@code /WEB-INF/ServletName.xml} is used, where {@code ServletName} is the name of the Vaadin
086 *  application (see {@code applicationName}).
087 * </td>
088 * </tr>
089 * <tr>
090 * <td>{@code listenerClass}</td>
091 * <td align="center">No</td>
092 * <td>
093 *  Specify the name of a custom class extending {@link SpringVaadinSessionListener} and having the same constructor arguments.
094 *  If omitted, {@link SpringVaadinSessionListener} is used.
095 * </td>
096 * </tr>
097 * <tr>
098 * <td>{@code sessionTracking}</td>
099 * <td align="center">No</td>
100 * <td>
101 *  Boolean value that configures whether the {@link SpringVaadinSessionListener} should track Vaadin sessions; default
102 *  {@code false}. If set to {@code true}, then {@link #getSessions} can be used to access all active sessions.
103 *  Session tracking should not be used unless sessions are normally kept in memory; e.g., don't use session tracking
104 *  when sessions are being serialized and persisted. See also {@link VaadinSessionContainer}.
105 * </td>
106 * </tr>
107 * <tr>
108 * <td>{@code maxSessions}</td>
109 * <td align="center">No</td>
110 * <td>
111 *  Configures a limit on the number of simultaneous Vaadin sessions that may exist at one time. Going over this
112 *  limit will result in a {@link com.vaadin.server.ServiceException} being thrown. A zero or negative number
113 *  means there is no limit (this is the default). Ignored unless {@value #SESSION_TRACKING_PARAMETER} is set to {@code true}.
114 * </td>
115 * </tr>
116 * </table>
117 * </div>
118 * </p>
119 *
120 * @see SpringVaadinSessionListener
121 * @see VaadinConfigurable
122 * @see VaadinApplication
123 */
124@SuppressWarnings("serial")
125public class SpringVaadinServlet extends VaadinServlet {
126
127    /**
128     * Servlet initialization parameter (<code>{@value #CONFIG_LOCATION_PARAMETER}</code>) used to specify
129     * the location(s) of the Spring application context XML file(s). Multiple XML files may be separated by whitespace.
130     * This parameter is optional.
131     */
132    public static final String CONFIG_LOCATION_PARAMETER = "configLocation";
133
134    /**
135     * Servlet initialization parameter (<code>{@value #LISTENER_CLASS_PARAMETER}</code>) used to specify
136     * the name of an custom subclass of {@link SpringVaadinSessionListener}.
137     * This parameter is optional.
138     */
139    public static final String LISTENER_CLASS_PARAMETER = "listenerClass";
140
141    /**
142     * Servlet initialization parameter (<code>{@value #APPLICATION_NAME_PARAMETER}</code>) used to specify
143     * the name the application.
144     * This parameter is optional.
145     */
146    public static final String APPLICATION_NAME_PARAMETER = "applicationName";
147
148    /**
149     * Servlet initialization parameter (<code>{@value #SESSION_TRACKING_PARAMETER}</code>) that enables
150     * tracking of all Vaadin session.
151     * This parameter is optional, and defaults to <code>false</code>.
152     */
153    public static final String SESSION_TRACKING_PARAMETER = "sessionTracking";
154
155    /**
156     * Servlet initialization parameter (<code>{@value #MAX_SESSIONS_PARAMETER}</code>) that configures the
157     * maximum number of simultaneous Vaadin sessions. Requires {@link #SESSION_TRACKING_PARAMETER} to be set to {@code true}.
158     * This parameter is optional, and defaults to zero, which means no limit.
159     */
160    public static final String MAX_SESSIONS_PARAMETER = "maxSessions";
161
162    // We use weak references to avoid leaks caused by exceptions in SessionInitListeners; see http://dev.vaadin.com/ticket/12915
163    private final WeakHashMap<VaadinSession, Void> liveSessions = new WeakHashMap<>();
164
165    private String servletName;
166
167    @Override
168    public void init(ServletConfig config) throws ServletException {
169        this.servletName = config.getServletName();
170        if (this.servletName == null)
171            throw new IllegalArgumentException("null servlet name");
172        super.init(config);
173    }
174
175    @Override
176    protected void servletInitialized() throws ServletException {
177
178        // Sanity check
179        if (this.servletName == null)
180            throw new IllegalArgumentException("servlet not initialized");
181
182        // Defer to superclass
183        super.servletInitialized();
184
185        // Get params
186        final VaadinServletService servletService = this.getService();
187        final Properties params = servletService.getDeploymentConfiguration().getInitParameters();
188        final String contextLocation = params.getProperty(CONFIG_LOCATION_PARAMETER);
189        final String listenerClassName = params.getProperty(LISTENER_CLASS_PARAMETER);
190        String applicationName = params.getProperty(APPLICATION_NAME_PARAMETER);
191        if (applicationName == null)
192            applicationName = this.servletName;
193
194        // Detect listener class to use
195        Class<? extends SpringVaadinSessionListener> listenerClass = SpringVaadinSessionListener.class;
196        if (listenerClassName != null) {
197            try {
198                listenerClass = Class.forName(listenerClassName, false, Thread.currentThread().getContextClassLoader())
199                  .asSubclass(SpringVaadinSessionListener.class);
200            } catch (RuntimeException e) {
201                throw e;
202            } catch (Exception e) {
203                throw new ServletException("error finding class " + listenerClassName, e);
204            }
205        }
206
207        // Create session listener
208        SpringVaadinSessionListener sessionListener;
209        try {
210            sessionListener = listenerClass.getConstructor(String.class, String.class)
211              .newInstance(applicationName, contextLocation);
212        } catch (RuntimeException e) {
213            throw e;
214        } catch (Exception e) {
215            throw new ServletException("error instantiating " + listenerClass, e);
216        }
217
218        // Register session listener
219        servletService.addSessionInitListener(sessionListener);
220        servletService.addSessionDestroyListener(sessionListener);
221    }
222
223    @Override
224    protected VaadinServletService createServletService(DeploymentConfiguration deploymentConfiguration) throws ServiceException {
225
226        // Get session tracking parameters
227        final Properties params = deploymentConfiguration.getInitParameters();
228        final boolean sessionTracking = Boolean.valueOf(params.getProperty(SESSION_TRACKING_PARAMETER));
229        int maxSessionsParam = 0;
230        try {
231            maxSessionsParam = Integer.parseInt(params.getProperty(MAX_SESSIONS_PARAMETER));
232        } catch (Exception e) {
233            // ignore
234        }
235        final int maxSessions = maxSessionsParam;
236
237        // If not tracking sessions, do the normal thing
238        if (!sessionTracking)
239            return super.createServletService(deploymentConfiguration);
240
241        // Return a VaadinServletService that tracks sessions
242        final VaadinServletService service = new VaadinServletService(this, deploymentConfiguration) {
243
244            @Override
245            protected VaadinSession createVaadinSession(VaadinRequest request) throws ServiceException {
246                final VaadinSession session;
247                synchronized (SpringVaadinServlet.this.liveSessions) {
248                    if (maxSessions > 0 && SpringVaadinServlet.this.liveSessions.size() >= maxSessions)
249                        throw new ServiceException("The maximum number of active sessions has been reached");
250                    session = super.createVaadinSession(request);
251                    SpringVaadinServlet.this.liveSessions.put(session, null);
252                }
253                return session;
254            }
255
256            @Override
257            public void fireSessionDestroy(VaadinSession session) {
258                synchronized (SpringVaadinServlet.this.liveSessions) {
259                    SpringVaadinServlet.this.liveSessions.remove(session);
260                }
261                super.fireSessionDestroy(session);
262            }
263        };
264        service.init();
265        return service;
266    }
267
268    /**
269     * Get all live {@link VaadinSession}s associated with this instance.
270     *
271     * @return live tracked sessions, or an empty collection if session tracking is not enabled
272     * @see VaadinSessionContainer
273     */
274    public List<VaadinSession> getSessions() {
275        synchronized (this.liveSessions) {
276            return new ArrayList<VaadinSession>(this.liveSessions.keySet());
277        }
278    }
279
280    /**
281     * Get the {@link SpringVaadinServlet} that is associated with the given {@link VaadinSession}.
282     *
283     * @throws IllegalStateException if the {@link VaadinServlet} associated with {@code session} is not a
284     *  {@link SpringVaadinServlet}
285     */
286    public static SpringVaadinServlet getServlet(VaadinSession session) {
287        if (session == null)
288            throw new IllegalArgumentException("null session");
289        if (!(session.getService() instanceof VaadinServletService))
290            throw new IllegalStateException("the VaadinService associated with the session is not a VaadinServletService instance");
291        final VaadinServletService service = (VaadinServletService)session.getService();
292        if (!(service.getServlet() instanceof SpringVaadinServlet))
293            throw new IllegalStateException("the VaadinServlet associated with the session is not a SpringVaadinServlet instance");
294        return (SpringVaadinServlet)service.getServlet();
295    }
296}
