008package org.dellroad.stuff.vaadin7;
009
010import com.vaadin.server.VaadinSession;
011
012/**
013 * Represents a session in the {@link VaadinSessionContainer}.
014 */
015public class VaadinSessionInfo {
016
017    private final VaadinSession session;
018
019    /**
020     * Constructor.
021     *
022     * @throws IllegalStateException if there is no current {@link VaadinSession}
023     */
024    public VaadinSessionInfo() {
025        this.session = VaadinUtil.getCurrentSession();
026    }
027
028    /**
029     * Get the associated {@link VaadinSession}.
030     */
031    public VaadinSession getVaadinSession() {
032        return this.session;
033    }
034
035    /**
036     * Determine if the associated session is also the current session.
037     */
038    public boolean isCurrentSession() {
039        return this.session == VaadinSession.getCurrent();
040    }
041}
