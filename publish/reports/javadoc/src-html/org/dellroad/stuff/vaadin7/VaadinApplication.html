008package org.dellroad.stuff.vaadin7;
009
010import com.vaadin.server.SessionDestroyListener;
011import com.vaadin.server.VaadinSession;
012
013import java.io.IOException;
014import java.io.ObjectInputStream;
015import java.io.Serializable;
016import java.util.concurrent.Future;
017
018import org.slf4j.Logger;
019import org.slf4j.LoggerFactory;
020
021/**
022 * A globally accessible "Vaadin application" singleton.
023 *
024 * <p>
025 * At most one instance is associated with each {@link VaadinSession} (attempting to create more than one will result in
026 * an exception). This singleton instance is then always accessible from any Vaadin thread via {@link #get()}.
027 * </p>
028 *
029 * <p>
030 * Although this class does not require Spring, the singleton instance can be declared as a bean in the Spring XML
031 * application context created by a {@link SpringVaadinSessionListener}, which allows other beans and widgets
032 * in the Vaadin application context to autowire it and have access to the methods provided here:
033 * <blockquote><pre>
034 *  &lt;bean class="org.dellroad.stuff.vaadin7.VaadinApplication"/&gt;
035 * </pre></blockquote>
036 * If you are in a clustered environment and are subclassing this class, {@code scope="session"} is recommended;
037 * without it, there will be one instance per server+session rather than one unique instance per session. Of course,
038 * whether that matters depends on the semantics of the subclass fields. In any case, be aware that no instance will
039 * exist in session until the application context is refreshed.
040 * </p>
041 *
042 * <p>
043 * Note: using {@code scope="session"} requires adding a {@code <listener>} clause registering Spring's
044 * {@link org.springframework.web.context.request.RequestContextListener} in your {@code web.xml}.
045 * </p>
046 *
047 * <p>
048 * If this class is subclassed, additional application-specific fields and methods can be supplied to the entire
049 * application via the same mechanism. Subclasses can override {@link #init} for any application-specific initialization.
050 * </p>
051 *
052 * <p>
053 * Note: the singleton instance of this class is stored in the Vaadin session and is {@link Serializable}. Therefore,
054 * in a clustered environment, subclasses must be prepared to handle (de)serialization. Also, avoid declaring fields
055 * that reference this class in non-session-scope beans; instead, use {@link #get VaadinApplication.get()}.
056 * </p>
057 *
058 * @see SpringVaadinServlet
059 * @see SpringVaadinSessionListener
060 * @see com.vaadin.server.VaadinService
061 */
062public class VaadinApplication implements Serializable {
063
064    private static final long serialVersionUID = -4202507689739224846L;
065    private static final Class<VaadinApplication> ATTRIBUTE_KEY = VaadinApplication.class;
066
067    protected final Logger log = LoggerFactory.getLogger(this.getClass());
068
069    private transient VaadinSession session;
070
071    /**
072     * Convenience constructor. Equivalent to:
073     * <blockquote></code>
074     *  {@link #VaadinApplication(VaadinSession) VaadinApplication}({@link VaadinUtil#getCurrentSession})
075     * </code></blockquote>
076     *
077     * @throws IllegalStateException if there is no {@link VaadinSession} associated with the current thread
078     * @throws IllegalStateException if there is already a {@link VaadinApplication} instance associated with the current session
079     */
080    public VaadinApplication() {
081        this(VaadinUtil.getCurrentSession());
082    }
083
084    /**
085     * Primary Constructor.
086     *
087     * @param session the session with which this instance should be associated
088     * @throws IllegalArgumentException if {@code session} is null
089     * @throws IllegalStateException if there is already a {@link VaadinApplication} instance associated with {@code session}
090     */
091    public VaadinApplication(VaadinSession session) {
092
093        // Get session
094        if (session == null)
095            throw new IllegalArgumentException("null session");
096        this.session = session;
097
098        // Check for already-existing instance
099        VaadinApplication vaadinApplication = this.session.getAttribute(ATTRIBUTE_KEY);
100        if (vaadinApplication != null) {
101            throw new IllegalStateException("there is already a VaadinApplication associated with VaadinSession "
102              + this.session + ": " + vaadinApplication + "; did you accidentally declare more than one instance of"
103              + " VaadinApplication in the Vaadin Spring XML application context?");
104        }
105
106        // Set session attribute
107        VaadinApplication.setAttribute(this.session, ATTRIBUTE_KEY, this);
108
109        // Delegate to subclass for further initialization
110        this.init();
111    }
112
113    /**
114     * Perform any further initialization at construction time.
115     *
116     * <p>
117     * The implementation in {@link VaadinApplication} does nothing. Subclasses may override as desired.
118     * </p>
119     */
120    protected void init() {
121    }
122
123    /**
124     * Get the {@link VaadinSession} associated with this instance.
125     *
126     * @return associated VaadinSession, never null
127     */
128    public VaadinSession getSession() {
129        return this.session;
130    }
131
132    /**
133     * Close the {@link VaadinSession} associated with this instance.
134     * After invoking this method, the caller would normally ensure that no further references to this
135     * instance remain so that it and the associated {@link VaadinSession} can be freed.
136     *
137     * <p>
138     * The implementation in {@link VaadinApplication} just delegates to {@link com.vaadin.server.VaadinSession#close}.
139     */
140    public void close() {
141        this.session.close();
142    }
143
144    /**
145     * Get the singleton {@link VaadinApplication} instance associated with the current {@link VaadinSession}.
146     *
147     * <p>
148     * This is a convenience method, equivalent to:
149     * <blockquote><code>
150     *  {@link #get(Class) VaadinApplication.get}({@link VaadinApplication VaadinApplication.class})
151     * </code></blockquote>
152     *
153     * @return singleton instance for the current Vaadin application, never null
154     * @throws IllegalStateException if there is no {@link VaadinSession} associated with the current thread
155     * @throws IllegalStateException if there is no {@link VaadinApplication} instance associated with the current session
156     */
157    public static VaadinApplication get() {
158        return VaadinApplication.get(VaadinApplication.class);
159    }
160
161    /**
162     * Get the singleton instance of the specified class associated with the current {@link VaadinSession}.
163     *
164     * <p>
165     * This is a convenience method, equivalent to:
166     * <blockquote><code>
167     *  {@link #get(Class) VaadinApplication.get}({@link VaadinUtil#getCurrentSession}, clazz)
168     * </code></blockquote>
169     *
170     * <p>
171     * Useful for subclasses of {@link VaadinApplication} that want to provide their own zero-parameter {@code get()} methods.
172     *
173     * @param clazz singleton instance type
174     * @return singleton instance of {@code clazz} in the session, never null
175     * @throws IllegalArgumentException if {@code clazz} is null
176     * @throws IllegalStateException if there is no {@link VaadinSession} associated with the current thread
177     * @throws IllegalStateException if there is no singleton of type {@code clazz} associated with the current session
178     */
179    public static <T extends VaadinApplication> T get(Class<T> clazz) {
180        return VaadinApplication.get(VaadinUtil.getCurrentSession(), clazz);
181    }
182
183    /**
184     * Get the singleton instance of the specified class associated with the given session.
185     *
186     * @param session Vaadin session
187     * @param clazz singleton instance type
188     * @return singleton instance of {@code clazz} in the session, never null
189     * @throws IllegalArgumentException if {@code session} is null
190     * @throws IllegalArgumentException if {@code clazz} is null
191     * @throws IllegalStateException if there is no singleton of type {@code clazz} associated with the {@code session}
192     */
193    public static <T extends VaadinApplication> T get(VaadinSession session, Class<T> clazz) {
194
195        // Sanity check
196        if (session == null)
197            throw new IllegalArgumentException("null session");
198        if (clazz == null)
199            throw new IllegalArgumentException("null clazz");
200
201        // Get the application
202        VaadinApplication vaadinApplication = session.getAttribute(ATTRIBUTE_KEY);
203        if (vaadinApplication == null) {
204            throw new IllegalStateException("there is no VaadinApplication associated with the current VaadinSession"
205              + "; did you declare an instance of VaadinApplication in the Vaadin Spring XML application context?"
206              + " And if declared scope=\"session\", did you register Spring's RequestContextListener in web.xml?");
207        }
208
209        // Check type
210        if (!clazz.isInstance(vaadinApplication)) {
211            throw new IllegalStateException("there is a VaadinApplication associated with the current VaadinSession"
212              + " but it is not an instance of " + clazz + "; instead it has type " + vaadinApplication.getClass().getName());
213        }
214
215        // Done
216        return clazz.cast(vaadinApplication);
217    }
218
219    /**
220     * Peform some action while holding the lock of the {@link VaadinSession} associated with this instance.
221     *
222     * <p>
223     * This is a convenience method that in turn invokes {@link VaadinUtil#invoke VaadinUtil.invoke()} using the
224     * {@link VaadinSession} associated with this instance.
225     * </p>
226     *
227     * @param action action to perform
228     * @throws IllegalArgumentException if {@code action} is null
229     * @see VaadinUtil#invoke
230     */
231    public void invoke(Runnable action) {
232        VaadinUtil.invoke(this.session, action);
233    }
234
235    /**
236     * Peform some action asynchronously while holding the lock of the {@link VaadinSession} associated with this instance.
237     *
238     * <p>
239     * This is a convenience method that in turn invokes {@link VaadinUtil#invokeLater VaadinUtil.invokeLater()} using the
240     * {@link VaadinSession} associated with this instance.
241     * </p>
242     *
243     * @param action action to perform
244     * @throws IllegalArgumentException if {@code action} is null
245     * @see VaadinUtil#invokeLater
246     */
247    public Future<Void> invokeLater(Runnable action) {
248        return VaadinUtil.invokeLater(this.session, action);
249    }
250
251    /**
252     * Register for a notification when the {@link VaadinSession} is closed, without creating a memory leak.
253     *
254     * <p>
255     * This is a convenience method that in turn invokes
256     * {@link VaadinUtil#addSessionDestroyListener VaadinUtil.addSessionDestroyListener()} using the
257     * {@link VaadinSession} associated with this instance.
258     *
259     * @throws IllegalArgumentException if {@code listener} is null
260     * @see VaadinUtil#addSessionDestroyListener
261     */
262    public void addSessionDestroyListener(SessionDestroyListener listener) {
263        VaadinUtil.addSessionDestroyListener(this.session, listener);
264    }
265
266    /**
267     * Remove a listener added via {@link #addSessionDestroyListener addSessionDestroyListener()}.
268     *
269     * <p>
270     * This is a convenience method that in turn invokes
271     * {@link VaadinUtil#removeSessionDestroyListener VaadinUtil.removeSessionDestroyListener()} using the
272     * {@link VaadinSession} associated with this instance.
273     *
274     * @throws IllegalArgumentException if {@code listener} is null
275     * @see VaadinUtil#removeSessionDestroyListener
276     */
277    public void removeSessionDestroyListener(SessionDestroyListener listener) {
278        VaadinUtil.removeSessionDestroyListener(this.session, listener);
279    }
280
281    // This method exists solely to bind the generic type
282    private static <T> void setAttribute(VaadinSession session, Class<T> clazz, Object value) {
283        session.setAttribute(clazz, clazz.cast(value));
284    }
285
286// Serialization
287
288    private void readObject(ObjectInputStream input) throws IOException, ClassNotFoundException {
289        input.defaultReadObject();
290        this.session = VaadinUtil.getCurrentSession();
291    }
292}
