008package org.dellroad.stuff.io;
009
010import java.io.IOException;
011import java.io.InputStream;
012import java.io.OutputStream;
013
014/**
015 * Interface for atomic storage and retrieval of an opaque byte array, with access being done
016 * via {@link InputStream} and {@link OutputStream} interfaces.
017 *
018 * <p>
019 * Conceptually, there is a single underlying byte array. Instances behave as if an atomic readable snapshot
020 * is made when {@link #getInputStream} is invoked, and the results of that snapshot are then available
021 * in the returned {@link InputStream}. Similarly, instances behave as if the underlying byte array is
022 * atomically updated when the output stream returned by {@link #getOutputStream} is successfully
023 * {@linkplain OutputStream#close closed}.
024 *
025 * <p>
026 * Instances are thread safe, and support multiple concurrent open input and output streams. In particular,
027 * two output streams may be {@linkplain OutputStream#close closed} at the same time by two different threads,
028 * and one will always win the race.
029 */
030public interface StreamRepository {
031
032    /**
033     * Get an input stream reading the current value of the underlying store.
034     *
035     * @throws IOException if an error occurs
036     */
037    InputStream getInputStream() throws IOException;
038
039    /**
040     * Get an output stream writing to the underlying store. The underlying store is not affected until the
041     * returned output stream is successfully {@linkplain OutputStream#close closed}, at which time it is
042     * atomically updated with the newly written content.
043     *
044     * <p>
045     * If the returned stread throws an {@link IOException} at any time, including during {@link OutputStream#close close()},
046     * then no update to the underlying storage occurs.
047     *
048     * @throws IOException if an error occurs
049     */
050    OutputStream getOutputStream() throws IOException;
051}
