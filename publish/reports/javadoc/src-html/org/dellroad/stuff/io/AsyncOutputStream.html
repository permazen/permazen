008package org.dellroad.stuff.io;
009
010import java.io.FilterOutputStream;
011import java.io.IOException;
012import java.io.OutputStream;
013import java.util.concurrent.atomic.AtomicInteger;
014
015import org.dellroad.stuff.java.CheckedExceptionWrapper;
016import org.dellroad.stuff.java.Predicate;
017import org.dellroad.stuff.java.TimedWait;
018import org.slf4j.Logger;
019import org.slf4j.LoggerFactory;

021/**
022 * An {@link OutputStream} that performs writes using a background thread, so that
023 * write, flush, and close operations never block.
024 * <p/>
025 *
026 * <p>
027 * If the underlying output stream throws an {@link IOException} during any operation,
028 * this instance will re-throw the exception for all subsequent operations.
029 * </p>
030 *
031 * <p>
032 * Instances use an internal buffer whose size is configured at construction time;
033 * if the buffer overflows, a {@link BufferOverflowException} is thrown. Alternately,
034 * if a buffer size of zero is configured, the internal buffer will expand automatically as needed
035 * (up to 2<super>31</super> bytes).
036 * However, this creates a memory leak if the underlying {@link OutputStream} blocks indefinitely.
037 * </p>
038 *
039 * <p>
040 * Instances of this class are thread safe, and moreover writes are atomic: if multiple threads are writing
041 * at the same time the bytes written in any single method invocation are written contiguously to the
042 * underlying output.
043 * </p>
044 */
045public class AsyncOutputStream extends FilterOutputStream {
046
047    private static final AtomicInteger COUNTER = new AtomicInteger();
048    private static final int MIN_BUFFER_SIZE = 20;
049    private static final int MAX_BUFFER_SIZE = Integer.MAX_VALUE - 32;
050
051    protected final Logger log = LoggerFactory.getLogger(getClass());
052
053    private byte[] buf;                 // output buffer
054    private int count;                  // number of bytes in output buffer ready to be written
055    private int flushMark = -1;         // buffer byte at which a flush is requested, or -1 if none
056    private Thread thread;              // async writer thread
057    private IOException exception;      // exception caught by async thread
058    private boolean closed;             // this instance has been close()'d
059    private boolean expand;             // whether to auto-expand buffer as needed
060
061    /**
062     * Convenience constructor for when an auto-expanding buffer is desired and a default thread name is to be used.
063     *
064     * @param out     underlying output stream
065     * @throws IllegalArgumentException if {@code out} is null
066     */
067    public AsyncOutputStream(OutputStream out) {
068        this(out, 0, AsyncOutputStream.class.getSimpleName() + "-" + AsyncOutputStream.COUNTER.incrementAndGet());
069    }
070
071    /**
072     * Convenience constructor for when an auto-expanding buffer is desired.
073     *
074     * @param out     underlying output stream
075     * @param name    name for this instance; used to create the name of the background thread
076     * @throws IllegalArgumentException if {@code out} or {@code name} is null
077     */
078    public AsyncOutputStream(OutputStream out, String name) {
079        this(out, 0, name);
080    }
081
082    /**
083     * Constructor.
084     *
085     * @param out     underlying output stream
086     * @param bufsize maximum number of bytes we can buffer, or zero for an auto-expanding buffer that has no fixed limit
087     * @param name    name for this instance; used to create the name of the background thread
088     * @throws IllegalArgumentException if {@code out} or {@code name} is null
089     * @throws IllegalArgumentException if {@code bufsize} is negative
090     */
091    public AsyncOutputStream(OutputStream out, int bufsize, String name) {
092        super(out);
093        if (out == null)
094            throw new IllegalArgumentException("null output");
095        if (name == null)
096            throw new IllegalArgumentException("null name");
097        if (bufsize < 0)
098            throw new IllegalArgumentException("invalid bufsize " + bufsize);
099        this.expand = bufsize == 0;
100        this.buf = new byte[Math.min(bufsize, MIN_BUFFER_SIZE)];
101
102        // Start worker thread
103        this.thread = new Thread(name) {
104            @Override
105            public void run() {
106                AsyncOutputStream.this.threadMain();
107            }
108        };
109        this.thread.setDaemon(true);
110        this.thread.start();
111        synchronized (this) { }
112    }
113
114    /**
115     * Write data.
116     *
117     * <p>
118     * This method will never block. To effect a normal blocking write, use {@link #waitForSpace} first.
119     * </p>
120     *
121     * @param b byte to write (lower 8 bits)
122     * @throws IOException             if an exception has been thrown by the underlying stream
123     * @throws IOException             if this instance has been closed
124     * @throws BufferOverflowException if the buffer does not have room for the new byte
125     */
126    @Override
127    public void write(int b) throws IOException {
128        this.write(new byte[] { (byte)b }, 0, 1);
129    }
130
131    /**
132     * Write data.
133     *
134     * <p>
135     * This method will never block. To effect a normal blocking write, invoke {@link #waitForSpace} first.
136     * </p>
137     *
138     * @param data bytes to write
139     * @param off  starting offset in buffer
140     * @param len  number of bytes to write
141     * @throws IOException              if an exception has been thrown by the underlying stream
142     * @throws IOException              if this instance has been closed
143     * @throws BufferOverflowException  if the buffer does not have room for the new data
144     * @throws IllegalArgumentException if {@code len} is negative
145     */
146    @Override
147    public synchronized void write(byte[] data, int off, int len) throws IOException {
148
149        // Check exception conditions
150        this.checkExceptions();
151        if (len < 0)
152            throw new IllegalArgumentException("len = " + len);
153        if (len == 0)
154            return;
155        if (this.count + len > this.buf.length) {
156            if (!this.expand) {
157                throw new BufferOverflowException(this.count + " + " + len + " = " + (this.count + len)
158                  + " byte(s) would exceed the " + this.buf.length + " byte buffer");
159            }
160            this.resizeBuffer(Math.max(this.count + len, this.buf.length * 2));
161
162            final byte[] newBuf = new byte[Math.max(this.count + len, this.buf.length * 2)];
163            System.arraycopy(this.buf, 0, newBuf, 0, this.count);
164            this.buf = newBuf;
165        }
166
167        // Add data to buffer
168        System.arraycopy(data, off, this.buf, this.count, len);
169        this.count += len;
170
171        // Wakeup writer thread
172        this.notifyAll();
173    }
174
175    /**
176     * Flush output. This method will cause the underlying stream to be flushed once all of the data written to this
177     * instance at the time this method is invoked has been written to it.
178     *
179     * <p>
180     * If additional data is written and then a second flush is requested before the first flush has actually occurred,
181     * the first flush will be canceled and only the second flush will be applied. Normally this is not a problem because
182     * the act of writing more data and then flushing forces earlier data to be flushed as well.
183     * </p>
184     *
185     * <p>
186     * This method will never block. To block until the underlying flush operation completes, invoke {@link #waitForIdle}.
187     * </p>
188     *
189     * @throws IOException if this instance has been closed
190     * @throws IOException if an exception has been detected on the underlying stream
191     * @throws IOException if the current thread is interrupted; the nested exception will an {@link InterruptedException}
192     */
193    @Override
194    public synchronized void flush() throws IOException {
195        this.checkExceptions();
196        this.flushMark = this.count;
197        this.notifyAll();                               // wake up writer thread
198    }
199
200    /**
201     * Close this instance. This will (eventually) close the underlying output stream.
202     *
203     * <p>
204     * If this instance has already been closed, nothing happens.
205     * </p>
206     *
207     * <p>
208     * This method will never block. To block until the underlying close operation completes, invoke {@link #waitForIdle}.
209     * </p>
210     *
211     * @throws IOException if an exception has been detected on the underlying stream
212     */
213    @Override
214    public synchronized void close() throws IOException {
215        if (this.closed)
216            return;
217        this.closed = true;
218        this.notifyAll();                               // wake up writer thread
219    }
220
221    /**
222     * Get the exception thrown by the underlying output stream, if any.
223     *
224     * @return thrown exception, or {@code null} if none has been thrown by the underlying stream
225     */
226    public synchronized IOException getException() {
227        return this.exception;
228    }
229
230    /**
231     * Get the capacity of this instance's output buffer.
232     *
233     * <p>
234     * If a (fixed) non-zero value was given at construction time, this will return that value.
235     * </p>
236     *
237     * @return current output buffer capacity
238     */
239    public synchronized int getBufferSize() {
240        return this.buf.length;
241    }
242
243    /**
244     * Get the number of free bytes remaining in the output buffer.
245     *
246     * @return current number of available bytes in the output buffer
247     * @throws IOException              if this instance is or has been closed
248     * @throws IOException              if an exception has been detected on the underlying stream
249     * @see #waitForSpace
250     */
251    public synchronized int availableBufferSpace() throws IOException {
252        this.checkExceptions();
253        return this.buf.length - this.count;
254    }
255
256    /**
257     * Determine if there is outstanding work still to be performed (writes, flushes, and/or close operations)
258     * by the background thread.
259     *
260     * @throws IOException              if this instance is or has been closed
261     * @throws IOException              if an exception has been detected on the underlying stream
262     * @see #waitForIdle
263     */
264    public synchronized boolean isWorkOutstanding() throws IOException {
265        this.checkExceptions();
266        return this.threadHasWork();
267    }
268
269    /**
270     * Wait for buffer space availability.
271     *
272     * <p>
273     * If a zero buffer size was configured at construction time, indicating an auto-expanding buffer,
274     * this will return immediately.
275     * </p>
276     *
277     * @param numBytes amount of buffer space required
278     * @param timeout  maximum time to wait in milliseconds, or zero for infinite
279     * @return true if space was found, false if time expired
280     * @throws IOException              if this instance is or has been closed
281     * @throws IOException              if an exception has been detected on the underlying stream
282     * @throws IllegalArgumentException if {@code numBytes} is greater than the configured buffer size
283     * @throws IllegalArgumentException if {@code timeout} is negative
284     * @throws InterruptedException     if the current thread is interrupted
285     * @see #availableBufferSpace
286     */
287    public boolean waitForSpace(final int numBytes, long timeout) throws IOException, InterruptedException {
288        if (this.expand)
289            return true;
290        if (numBytes > this.buf.length)
291            throw new IllegalArgumentException("numBytes (" + numBytes + ") > buffer size (" + this.buf.length + ")");
292        return this.waitForPredicate(timeout, new Predicate() {
293            @Override
294            public boolean test() {
295                return AsyncOutputStream.this.buf.length - AsyncOutputStream.this.count >= numBytes;
296            }
297        });
298    }
299
300    /**
301     * Wait for all outstanding work to complete.
302     *
303     * @param timeout maximum time to wait in milliseconds, or zero for infinite
304     * @return true for success, false if time expired
305     * @throws IOException              if this instance is or has been closed
306     * @throws IOException              if an exception has been detected on the underlying stream
307     * @throws IllegalArgumentException if {@code timeout} is negative
308     * @throws InterruptedException     if the current thread is interrupted
309     * @see #isWorkOutstanding
310     */
311    public synchronized boolean waitForIdle(long timeout) throws IOException, InterruptedException {
312        return this.waitForPredicate(timeout, new Predicate() {
313            @Override
314            public boolean test() {
315                return !AsyncOutputStream.this.threadHasWork();
316            }
317        });
318    }
319
320    /**
321     * Check for exceptions.
322     *
323     * @throws IOException if this instance has been closed
324     * @throws IOException if an exception has been detected on the underlying stream
325     */
326    private void checkExceptions() throws IOException {
327        if (this.closed)
328            throw new IOException("instance has been closed");
329        if (this.exception != null)
330            throw new IOException("exception from underlying output stream", this.exception);
331    }
332
333    /**
334     * Determine if there is outstanding work still to be performed (writes, flushes, and/or close operations)
335     * by the background thread.
336     */
337    private boolean threadHasWork() {
338        return this.count > 0 || this.flushMark != -1 || this.closed;
339    }
340
341    /**
342     * Resize internal buffer.
343     */
344    private /*synchronized*/ void resizeBuffer(int size) {
345        assert Thread.holdsLock(this);
346        assert this.count <= size;
347        size = Math.min(size, MAX_BUFFER_SIZE);
348        size = Math.max(size, MIN_BUFFER_SIZE);
349        final byte[] newBuf = new byte[size];
350        System.arraycopy(this.buf, 0, newBuf, 0, this.count);
351        this.buf = newBuf;
352    }
353
354    /**
355     * Writer thread main entry point.
356     */
357    private void threadMain() {
358        try {
359            this.runLoop();
360        } catch (Throwable t) {
361            synchronized (this) {
362                this.exception = t instanceof IOException ? (IOException)t : new IOException("caught unexpected exception", t);
363                this.notifyAll();                       // wake up sleepers in waitForSpace() and waitForIdle()
364            }
365        } finally {
366            synchronized (this) {
367                this.thread = null;
368                this.notifyAll();                       // wake up sleepers in waitForIdle()
369            }
370        }
371    }
372
373    /**
374     * Async writer thread main loop.
375     */
376    private void runLoop() throws IOException, InterruptedException {
377        while (true) {
378
379            // Wait for something to do
380            synchronized (this) {
381                while (!this.threadHasWork())
382                    this.wait();                        // will be woken up by write(), flush(), or close()
383            }
384
385            // Determine what needs to be done
386            byte[] currbuf;
387            int wlen;
388            boolean flush;
389            boolean close;
390            synchronized (this) {
391                currbuf = this.buf;
392                wlen = this.count;
393                flush = this.flushMark == 0;
394                close = this.closed;
395            }
396
397            // First priority: any data to write?
398            if (wlen > 0) {
399
400                // Write data
401                this.out.write(currbuf, 0, wlen);
402
403                // Shift data in buffer
404                synchronized (this) {
405                    System.arraycopy(this.buf, wlen, this.buf, 0, this.count - wlen);
406                    this.count -= wlen;
407                    if (this.flushMark != -1)
408                        this.flushMark = Math.max(0, this.flushMark - wlen);
409                    if (this.count <= this.buf.length >> 7)
410                        this.resizeBuffer(this.count);
411                    this.notifyAll();                   // wake up sleepers in waitForSpace() and waitForIdle()
412                }
413                continue;
414            }
415
416            // Second priority: is a flush required?
417            if (flush) {
418
419                // Flush output
420                this.out.flush();
421
422                // Update flush mark
423                synchronized (this) {
424                    if (this.flushMark == 0) {
425                        this.flushMark = -1;
426                        this.notifyAll();               // wake up sleepers in waitForIdle()
427                    }
428                }
429                continue;
430            }
431
432            // Third priority:  is a close required?
433            if (close) {
434                this.out.close();
435                break;
436            }
437        }
438    }

440    /**
441     * Wait for some condition to become true. Of course somebody has to wake us up when it becomes true.
442     */
443    private synchronized boolean waitForPredicate(long timeout, final Predicate predicate)
444      throws IOException, InterruptedException {
445        try {
446            return TimedWait.wait(this, timeout, new Predicate() {
447                @Override
448                public boolean test() {
449                    try {
450                        checkExceptions();
451                    } catch (IOException e) {
452                        throw new CheckedExceptionWrapper(e);
453                    }
454                    return predicate.test();
455                }
456            });
457        } catch (CheckedExceptionWrapper e) {
458            throw (IOException)e.getException();
459        }
460    }
