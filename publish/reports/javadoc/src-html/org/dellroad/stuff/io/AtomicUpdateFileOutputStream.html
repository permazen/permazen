008package org.dellroad.stuff.io;
009
010import java.io.File;
011import java.io.FileNotFoundException;
012import java.io.FileOutputStream;
013import java.io.IOException;
014
015/**
016 * A {@link FileOutputStream} that atomically updates the target file.
017 *
018 * <p>
019 * Instances write to a temporary file until {@link #close} is invoked, at which time the temporary file
020 * gets {@linkplain File#renameTo renamed} to the target file. This rename operation is atomic on most systems
021 * (e.g., all UNIX variants). The result is that the target file always exists, and if opened at any time,
022 * will contain either the previous content or the new content, but never a mix of the two.
023 *
024 * <p>
025 * An open instance can be thought of as representing an open transaction to rewrite the file.
026 * The "transaction" is committed via {@link #close}, or may be aborted via {@link #cancel} (which
027 * deletes the temporary file).
028 */
029public class AtomicUpdateFileOutputStream extends FileOutputStream {
030
031    private final File targetFile;
032
033    private File tempFile;
034    private long timestamp;
035
036    /**
037     * Constructor.
038     *
039     * @param targetFile the ultimate destination for the output when {@linkplain #close closed}.
040     * @param tempFile temporary file that accumulates output until {@linkplain #close close}.
041     * @throws FileNotFoundException if {@code tempFile} cannot be opened for any reason
042     * @throws SecurityException if a security manager prevents writing to {@code tempFile}
043     * @throws NullPointerException if either parameter is null
044     */
045    public AtomicUpdateFileOutputStream(File targetFile, File tempFile) throws FileNotFoundException {
046        super(tempFile);
047        this.tempFile = tempFile;
048        if (targetFile == null)
049            throw new NullPointerException("null targetFile");
050        this.targetFile = targetFile;
051    }
052
053    /**
054     * Convenience constructor.
055     *
056     * <p>
057     * This constructor uses a temporary file within the same directory as {@code targetFile}.
058     *
059     * @param targetFile the ultimate destination for the output when {@linkplain #close closed}.
060     * @throws FileNotFoundException if {@code tempFile} cannot be opened for any reason
061     * @throws IOException if a temporary file could not be created
062     * @throws SecurityException if a security manager prevents writing to {@code tempFile}
063     * @throws NullPointerException if either parameter is null
064     */
065    public AtomicUpdateFileOutputStream(File targetFile) throws IOException {
066        this(targetFile, File.createTempFile("atomicupdate", null, targetFile.getParentFile()));
067    }
068
069    /**
070     * Get the target file.
071     *
072     * @return target file, never null
073     */
074    public synchronized File getTargetFile() {
075        return this.targetFile;
076    }
077
078    /**
079     * Get the temporary file.
080     *
081     * <p>
082     * If this instance has already been {@linkplain #close closed} (either successfully or not)
083     * or {@linkplain #cancel canceled}, this will return null.
084     *
085     * @return temporary file, or null if {@link #close} or {@link #cancel} has already been invoked
086     */
087    public synchronized File getTempFile() {
088        return this.tempFile;
089    }
090
091    /**
092     * Cancel this instance. This "aborts" the open "transaction", and deletes the temporary file.
093     *
094     * <p>
095     * Does nothing if {@link #close} or {@link #cancel} has already been invoked.
096     */
097    public synchronized void cancel() {
098        if (this.tempFile != null) {
099
100            // Close output stream to release file descriptor
101            try {
102                super.close();
103            } catch (IOException e) {
104                // ignore
105            }
106
107            // Delete temporary file
108            this.tempFile.delete();
109            this.tempFile = null;
110        }
111    }
112
113    /**
114     * Close this instance. This "commits" the open "transaction".
115     *
116     * <p>
117     * If successful, the configured {@code tempFile} will be {@linkplain File#renameTo renamed}
118     * to the configured destination file {@code targetFile}. In any case, after this method returns
119     * (either normally or abnormally), the temporary file will be deleted.
120     *
121     * @throws IOException if {@link #close} or {@link #cancel} has already been invoked
122     */
123    @Override
124    public synchronized void close() throws IOException {
125
126        // Sanity check
127        if (this.tempFile == null)
128            throw new IOException("already closed or canceled");
129
130        // Close temporary file
131        super.close();
132
133        // Read updated modification time
134        final long newTimestamp = this.tempFile.lastModified();
135
136        // Rename file, or delete it if that fails
137        try {
138            if (!this.tempFile.renameTo(this.targetFile)) {
139                throw new IOException("error renaming temporary file `" + this.tempFile.getName()
140                  + "' to `" + this.targetFile.getName() + "'");
141            }
142            this.tempFile = null;
143        } finally {
144            if (this.tempFile != null)          // exception thrown, cancel transaction
145                this.cancel();
146        }
147
148        // Update target file timestamp
149        this.timestamp = newTimestamp;
150    }
151
152    /**
153     * Get the last modification timestamp of the target file as it was at the time it was updated by this instance.
154     *
155     * <p>
156     * This method only works after {@link #close} has been successfully invoked, otherwise it returns zero.
157     *
158     * @return target file modification time, or zero if {@link #close} has not been successfully invoked
159     */
160    public synchronized long getTimestamp() {
161        return this.timestamp;
162    }
163
164    /**
165     * Ensure the temporary file is deleted in cases where this instance never got successfully closed.
166     */
167    @Override
168    protected void finalize() throws IOException {
169        try {
170            if (this.tempFile != null)
171                this.cancel();
172        } finally {
173            super.finalize();
174        }
175    }
176}
