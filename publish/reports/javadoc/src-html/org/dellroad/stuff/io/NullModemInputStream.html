008    package org.dellroad.stuff.io;
009    
010    import java.io.IOException;
011    import java.io.OutputStream;
012    import java.io.PipedInputStream;
013    import java.io.PipedOutputStream;
014    
015    import org.slf4j.Logger;
016    import org.slf4j.LoggerFactory;
017    
018    /**
019     * Presents an {@link InputStream} interface given a callback that can write to an {@link OutputStream}.
020     * A separate thread is created to perform the actual writing.
021     */
022    public class NullModemInputStream extends PipedInputStream {
023    
024        protected final Logger log = LoggerFactory.getLogger(getClass());
025    
026        private final CallableWriter writer;
027        private final PipedOutputStream output;
028    
029        /**
030         * Constructor.
031         *
032         * @param writer    {@link OutputStream} writer callback
033         * @param name      name for this instance; used to create the name of the background thread
034         */
035        public NullModemInputStream(CallableWriter writer, String name) {
036            if (writer == null)
037                throw new IllegalArgumentException("null writer");
038            this.writer = writer;
039            try {
040                this.output = new PipedOutputStream(this);
041            } catch (IOException e) {
042                throw new RuntimeException("impossible case", e);
043            }
044            Thread thread = new Thread(name) {
045                @Override
046                public void run() {
047                    NullModemInputStream.this.threadMain();
048                }
049            };
050            thread.setDaemon(true);
051            thread.start();
052        }
053    
054        /**
055         * Writer thread main entry point.
056         */
057        private void threadMain() {
058            try {
059                this.writer.writeTo(this.output);
060                this.output.close();
061            } catch (IOException e) {
062                // ignore - reader will get another IOException because pipe is about to be broken
063            }
064        }
065    
066        /**
067         * Callback interface used by {@link NullModemInputStream}.
068         */
069        public interface CallableWriter {
070    
071            /**
072             * Write the output to the given output stream.
073             *
074             * <p>
075             * This method will be invoked asynchronously by a dedicated writer thread.
076             * </p>
077             *
078             * @throws IOException if an I/O error occurs
079             */
080            void writeTo(OutputStream output) throws IOException;
081        }
082    }
