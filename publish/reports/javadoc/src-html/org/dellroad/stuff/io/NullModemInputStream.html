008package org.dellroad.stuff.io;
009
010import java.io.FilterInputStream;
011import java.io.IOException;
012import java.io.PipedInputStream;
013import java.io.PipedOutputStream;
014
015/**
016 * Presents an {@link java.io.InputStream} interface given a {@link WriteCallback} that can write to an
017 * {@link java.io.OutputStream}. A separate thread is created to perform the actual writing.
018 *
019 * @since 1.0.74
020 */
021public class NullModemInputStream extends FilterInputStream {
022
023    private final PipedOutputStream output;
024
025    /**
026     * Constructor.
027     *
028     * <p>
029     * The {@code writer}'s {@link WriteCallback#writeTo writeTo()} method will be invoked (once)
030     * asynchronously in a dedicated writer thread. The {@link java.io.OutputStream} provided to it will
031     * relay the bytes that are then read from this instance.
032     * </p>
033     *
034     * @param writer    {@link java.io.OutputStream} writer callback
035     * @param name      name for this instance; used to create the name of the background thread
036     */
037    public NullModemInputStream(final WriteCallback writer, String name) {
038        super(new PipedInputStream());
039
040        // Sanity check
041        if (writer == null)
042            throw new IllegalArgumentException("null writer");
043
044        // Create other end of pipe
045        try {
046            this.output = new PipedOutputStream(this.getPipedInputStream());
047        } catch (IOException e) {
048            throw new RuntimeException("unexpected exception", e);
049        }
050
051        // Launch writer thread
052        Thread thread = new WriterThread(writer, this.output, name);
053        thread.setDaemon(true);
054        thread.start();
055    }
056
057    /**
058     * Get the wrapped stream cast as a {@link PipedInputStream}.
059     */
060    protected PipedInputStream getPipedInputStream() {
061        return (PipedInputStream)this.in;
062    }
063
064    /**
065     * Ensure input stream is closed when this instance is no longer referenced.
066     *
067     * <p>
068     * This ensures the writer thread wakes up (and exits, avoiding a memory leak) when an instance of this class
069     * is created but never read from.
070     */
071    @Override
072    protected void finalize() throws Throwable {
073        try {
074            try {
075                this.getPipedInputStream().close();
076            } catch (IOException e) {
077                // ignore
078            }
079        } finally {
080            super.finalize();
081        }
082    }
083
084    /**
085     * Writer thread. This is designed to not hold a reference to the {@link NullModemInputStream}.
086     */
087    private static class WriterThread extends Thread {
088
089        private final WriteCallback writer;
090        private final PipedOutputStream output;
091
092        WriterThread(WriteCallback writer, PipedOutputStream output, String name) {
093            super(name);
094            this.writer = writer;
095            this.output = output;
096        }
097
098        @Override
099        public void run() {
100            try {
101                this.writer.writeTo(this.output);
102            } catch (IOException e) {
103                // ignore - reader will get another IOException because pipe is about to be broken
104            } finally {
105                try {
106                    this.output.close();
107                } catch (IOException e) {
108                    // ignore
109                }
110            }
111        }
112    }
113}
