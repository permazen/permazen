008package org.dellroad.stuff.xml;
009
010import javax.xml.namespace.NamespaceContext;
011import javax.xml.stream.XMLStreamException;
012import javax.xml.stream.XMLStreamWriter;
013
014/**
015 * Support superclass for filtering {@link XMLStreamWriter} implementations.
016 * All methods delegate to the configured parent.
017 */
018public class StreamWriterDelegate implements XMLStreamWriter {
019
020    private XMLStreamWriter parent;
021
022    /**
023     * Default constructor. The parent must be configured via {@link #setParent}.
024     */
025    public StreamWriterDelegate() {
026    }
027
028    /**
029     * Constructor.
030     *
031     * @param parent underlying writer wrapped by this instance
032     */
033    public StreamWriterDelegate(XMLStreamWriter parent) {
034        this.parent = parent;
035    }
036
037    /**
038     * Get parent instance.
039     */
040    public XMLStreamWriter getParent() {
041        return this.parent;
042    }
043
044    /**
045     * Set parent instance.
046     */
047    public void setParent(XMLStreamWriter parent) {
048        this.parent = parent;
049    }
050
051    @Override
052    public void writeStartElement(String localName) throws XMLStreamException {
053        this.parent.writeStartElement(localName);
054    }
055
056    @Override
057    public void writeStartElement(String namespaceURI, String localName) throws XMLStreamException {
058        this.parent.writeStartElement(namespaceURI, localName);
059    }
060
061    @Override
062    public void writeStartElement(String prefix, String namespaceURI, String localName) throws XMLStreamException {
063        this.parent.writeStartElement(prefix, namespaceURI, localName);
064    }
065
066    @Override
067    public void writeEmptyElement(String namespaceURI, String localName) throws XMLStreamException {
068        this.parent.writeEmptyElement(namespaceURI, localName);
069    }
070
071    @Override
072    public void writeEmptyElement(String prefix, String namespaceURI, String localName) throws XMLStreamException {
073        this.parent.writeEmptyElement(prefix, namespaceURI, localName);
074    }
075
076    @Override
077    public void writeEmptyElement(String localName) throws XMLStreamException {
078        this.parent.writeEmptyElement(localName);
079    }
080
081    @Override
082    public void writeEndElement() throws XMLStreamException {
083        this.parent.writeEndElement();
084    }
085
086    @Override
087    public void writeEndDocument() throws XMLStreamException {
088        this.parent.writeEndDocument();
089    }
090
091    @Override
092    public void close() throws XMLStreamException {
093        this.parent.close();
094    }
095
096    @Override
097    public void flush() throws XMLStreamException {
098        this.parent.flush();
099    }
100
101    @Override
102    public void writeAttribute(String localName, String value) throws XMLStreamException {
103        this.parent.writeAttribute(localName, value);
104    }
105
106    @Override
107    public void writeAttribute(String prefix, String namespaceURI, String localName, String value) throws XMLStreamException {
108        this.parent.writeAttribute(prefix, namespaceURI, localName, value);
109    }
110
111    @Override
112    public void writeAttribute(String namespaceURI, String localName, String value) throws XMLStreamException {
113        this.parent.writeAttribute(namespaceURI, localName, value);
114    }
115
116    @Override
117    public void writeNamespace(String prefix, String namespaceURI) throws XMLStreamException {
118        this.parent.writeNamespace(prefix, namespaceURI);
119    }
120
121    @Override
122    public void writeDefaultNamespace(String namespaceURI) throws XMLStreamException {
123        this.parent.writeDefaultNamespace(namespaceURI);
124    }
125
126    @Override
127    public void writeComment(String data) throws XMLStreamException {
128        this.parent.writeComment(data);
129    }
130
131    @Override
132    public void writeProcessingInstruction(String target) throws XMLStreamException {
133        this.parent.writeProcessingInstruction(target);
134    }
135
136    @Override
137    public void writeProcessingInstruction(String target, String data) throws XMLStreamException {
138        this.parent.writeProcessingInstruction(target, data);
139    }
140
141    @Override
142    public void writeCData(String data) throws XMLStreamException {
143        this.parent.writeCData(data);
144    }
145
146    @Override
147    public void writeDTD(String dtd) throws XMLStreamException {
148        this.parent.writeDTD(dtd);
149    }
150
151    @Override
152    public void writeEntityRef(String name) throws XMLStreamException {
153        this.parent.writeEntityRef(name);
154    }
155
156    @Override
157    public void writeStartDocument() throws XMLStreamException {
158        this.parent.writeStartDocument();
159    }
160
161    @Override
162    public void writeStartDocument(String version) throws XMLStreamException {
163        this.parent.writeStartDocument(version);
164    }
165
166    @Override
167    public void writeStartDocument(String encoding, String version) throws XMLStreamException {
168        this.parent.writeStartDocument(encoding, version);
169    }
170
171    @Override
172    public void writeCharacters(String text) throws XMLStreamException {
173        this.parent.writeCharacters(text);
174    }
175
176    @Override
177    public void writeCharacters(char[] text, int start, int len) throws XMLStreamException {
178        this.parent.writeCharacters(text, start, len);
179    }
180
181    @Override
182    public String getPrefix(String uri) throws XMLStreamException {
183        return this.parent.getPrefix(uri);
184    }
185
186    @Override
187    public void setPrefix(String prefix, String uri) throws XMLStreamException {
188        this.parent.setPrefix(prefix, uri);
189    }
190
191    @Override
192    public void setDefaultNamespace(String uri) throws XMLStreamException {
193        this.parent.setDefaultNamespace(uri);
194    }
195
196    @Override
197    public void setNamespaceContext(NamespaceContext context) throws XMLStreamException {
198        this.parent.setNamespaceContext(context);
199    }
200
201    @Override
202    public NamespaceContext getNamespaceContext() {
203        return this.parent.getNamespaceContext();
204    }
205
206    @Override
207    public Object getProperty(String name) {
208        return this.parent.getProperty(name);
209    }
210}
