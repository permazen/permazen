030/**
031 * Outputs the result of an SQL query as XML.
032 *
033 * <p>
034 * For example, the query <code>SELECT ID, LAST_NAME AS LAST, FIRST_NAME AS FIRST FROM EMPLOYEE</code> might become:
035 * <pre>
036 *  &lt;result-set&gt;
037 *      &lt;query&gt;&lt;![CDATA[SELECT ID, LAST_NAME, FIRST_NAME FROM EMPLOYEE]]&gt;&lt;/query&gt;
038 *      &lt;columns&gt;
039 *          &lt;column index="1" name="ID" label="ID" precision="20" type="BIGINT" typeName="BIGINT" nullable="false"/&gt;
040 *          &lt;column index="2" name="LAST_NAME" label="LAST" precision="255" type="VARCHAR" typeName="VARCHAR" nullable="false"/&gt;
041 *          &lt;column index="3" name="FIRST_NAME" label="FIRST" precision="255" type="VARCHAR" typeName="VARCHAR" nullable="false"/&gt;
042 *      &lt;/columns&gt;
043 *      &lt;data&gt;
044 *          &lt;row&gt;
045 *              &lt;column index="1"&gt;1302&lt;/column&gt;
046 *              &lt;column index="2"&gt;Washington&lt;/column&gt;
047 *              &lt;column index="3"&gt;George&lt;/column&gt;
048 *          &lt;/row&gt;
049 *          &lt;row&gt;
050 *              &lt;column index="1"&gt;1303&lt;/column&gt;
051 *              &lt;column index="2"&gt;Lincoln&lt;/column&gt;
052 *              &lt;column index="3"&gt;Abraham&lt;/column&gt;
053 *          &lt;/row&gt;
054 *          ...
055 *      &lt;/data&gt;
056 *  &lt;/result-set&gt;
057 * </pre>
058 * </p>
059 *
060 * <p>
061 * If you turn on column name tags via {@link #setColumnNameTags setColumnNameTags()}, the output would look like this:
062 * <pre>
063 *  &lt;result-set&gt;
064 *      &lt;query&gt;&lt;![CDATA[SELECT ID, LAST_NAME, FIRST_NAME FROM EMPLOYEE]]&gt;&lt;/query&gt;
065 *      &lt;columns&gt;
066 *          &lt;column index="1" name="ID" label="ID" precision="20" type="BIGINT" typeName="BIGINT" nullable="false"/&gt;
067 *          &lt;column index="2" name="LAST_NAME" label="LAST" precision="255" type="VARCHAR" typeName="VARCHAR" nullable="false"/&gt;
068 *          &lt;column index="3" name="FIRST_NAME" label="FIRST" precision="255" type="VARCHAR" typeName="VARCHAR" nullable="false"/&gt;
069 *      &lt;/columns&gt;
070 *      &lt;data&gt;
071 *          &lt;row&gt;
072 *              &lt;ID index="1"&gt;1302&lt;/ID&gt;
073 *              &lt;LAST index="2"&gt;Washington&lt;/LAST&gt;
074 *              &lt;FIRST index="3"&gt;George&lt;/FIRST&gt;
075 *          &lt;/row&gt;
076 *          &lt;row&gt;
077 *              &lt;ID index="1"&gt;1303&lt;/ID&gt;
078 *              &lt;LAST index="2"&gt;Lincoln&lt;/LAST&gt;
079 *              &lt;FIRST index="3"&gt;Abraham&lt;/FIRST&gt;
080 *          &lt;/row&gt;
081 *          ...
082 *      &lt;/data&gt;
083 *  &lt;/result-set&gt;
084 * </pre>
085 * </p>
086 */
087public class XMLResultSetWriter {
088
089    /**
090     * Mapping from result set {@link Types} values to their corresponding names.
091     */
092    public static final Map<Integer, String> TYPE_NAMES;
093    static {
094        HashMap<Integer, String> map = new HashMap<Integer, String>();
095        for (Field field : Types.class.getDeclaredFields()) {
096            if (field.getType() != Integer.TYPE || field.getModifiers() != (Modifier.PUBLIC | Modifier.STATIC | Modifier.FINAL))
097                continue;
098            try {
099                map.put(field.getInt(null), field.getName());
100            } catch (IllegalAccessException e) {
101                throw new RuntimeException(e);
102            }
103        }
104        TYPE_NAMES = Collections.unmodifiableMap(map);
105    }
106
107    private static final String DATE_FORMAT = "yyyy-MM-dd";
108    private static final String TIMESTAMP_FORMAT = DATE_FORMAT + " HH:mm:ss";
109
110    private final XMLStreamWriter writer;
111    private final String indentSpace;
112
113    private State state = State.LAST_CLOSE;
114    private boolean columnNameTags;
115    private int nesting;
116
117    // Tracks state of the output
118    private enum State {
119        LAST_OPEN,
120        LAST_CLOSE,
121        LAST_TEXT
122    };
123
124    /**
125     * Constructor.
126     *
127     * @param writer where to write the XML
128     * @param indent number of spaces for indentation
129     * @throws IllegalArgumentException if {@code writer} is null
130     */
131    public XMLResultSetWriter(XMLStreamWriter writer, int indent) {
132        if (writer == null)
133            throw new IllegalArgumentException("null writer");
134        this.writer = writer;
135        char[] spaces = new char[indent];
136        Arrays.fill(spaces, ' ');
137        this.indentSpace = new String(spaces);
138    }

140    /**
141     * Set whether to use the SQL column labels as the XML element names for the data columns
142     * instead of <code>&lt;column&gt;</code>.
143     * Default is false.
144     */
145    public void setColumnNameTags(boolean columnNameTags) {
146        this.columnNameTags = columnNameTags;
147    }

149    /**
150     * Write the given {@link ResultSet} out within a <code>&lt;result-set&gt;</code> XML element.
151     *
152     * <p>
153     * This is a convenience method, equivalent to:
154     *  <blockquote><code>
155     *      write(null, resultSet)
156     *  </code></blockquote>
157     * element is included.
158     */
159    public void write(ResultSet resultSet) throws SQLException, XMLStreamException {
160        this.write(null, resultSet);
161    }
162
163    /**
164     * Write the given {@link ResultSet} out within a <code>&lt;result-set&gt;</code> XML element including the query.
165     * The given SQL query will be present as a <code>&lt;query&gt;</code> first child element,
166     * followed by a <code>&lt;columns&gt;</code> element containing column meta-data, and finally
167     * the <code>&lt;data&gt;</code> element containing each returned row.
168     *
169     * <p>
170     * This method just outputs an XML element. To output an entire XML document, you would do something like this:
171     * <pre>
172     *  XMLResultSetWriter xmlResultSetWriter = new XMLResultSetWriter(writer, 4);
173     *  writer.writeStartDocument("UTF-8", "1.0");
174     *  xmlResultSetWriter.write(query, resultSet);
175     *  writer.writeEndDocument();
176     * </pre>
177     * </p>
178     *
179     * <p>
180     * The result set is iterated in a forward direction only.
181     *
182     * @param query the SQL query, or null to omit
183     * @param resultSet result of the query
184     * @throws XMLStreamException if there is an error writing out the XML
185     * @throws SQLException if there is an error reading from {@code resultSet}
186     */
187    public void write(String query, ResultSet resultSet) throws SQLException, XMLStreamException {
188
189        // Start result set
190        this.openTag("result-set");
191
192        // Emit query
193        if (query != null) {
194            this.openTag("query");
195            this.emitCData(query);
196            this.closeTag();
197        }
198
199        // Emit column info
200        final ResultSetMetaData metaData = resultSet.getMetaData();
201        final int numColumns = metaData.getColumnCount();
202        String[] columnNames = new String[numColumns];
203        this.openTag("columns");
204        for (int col = 1; col <= numColumns; col++) {
205            this.openTag("column");
206            this.writer.writeAttribute("index", "" + col);
207            String name = metaData.getColumnName(col);
208            if (name != null)
209                this.writer.writeAttribute("name", name);
210            String label = metaData.getColumnLabel(col);
211            if (label != null)
212                this.writer.writeAttribute("label", label);
213            if (this.columnNameTags)
214                columnNames[col - 1] = label != null ? label : name != null ? name : null;
215            int precision = metaData.getPrecision(col);
216            if (precision != 0)
217                this.writer.writeAttribute("precision", "" + precision);
218            int scale = metaData.getScale(col);
219            if (scale != 0)
220                this.writer.writeAttribute("scale", "" + scale);
221            int type = metaData.getColumnType(col);
222            final String typeName = XMLResultSetWriter.TYPE_NAMES.get(type);
223            this.writer.writeAttribute("type", typeName != null ? typeName : "" + type);
224            this.writer.writeAttribute("typeName", metaData.getColumnTypeName(col));
225            String nullableDesc;
226            switch (metaData.isNullable(col)) {
227            case ResultSetMetaData.columnNullable:
228                nullableDesc = "true";
229                break;
230            case ResultSetMetaData.columnNoNulls:
231                nullableDesc = "false";
232                break;
233            default:
234                nullableDesc = "unknown";
235                break;
236            }
237            this.writer.writeAttribute("nullable", nullableDesc);
238            this.closeTag();
239        }
240        this.closeTag();
241
242        // Emit result data
243        this.openTag("data");
244        while (resultSet.next()) {
245            this.openTag("row");
246            for (int col = 1; col <= numColumns; col++)
247                this.writeDataColumn(resultSet, metaData.getColumnType(col), col, columnNames[col - 1]);
248            this.closeTag();
249        }
250        this.closeTag();
251
252        // End result set
253        this.closeTag();
254    }

256// Writing a data row
257
258    private void writeDataColumn(ResultSet resultSet, int columnType, int col, String columnName)
259      throws SQLException, XMLStreamException {
260        Object value = null;
261        switch (columnType) {
262        case Types.BIT:
263        case Types.BOOLEAN:
264        {
265            boolean b = resultSet.getBoolean(col);
266            if (!resultSet.wasNull())
267                value = "" + b;
268            break;
269        }
270        case Types.CLOB:
271        case Types.NCLOB:
272        case Types.LONGVARCHAR:
273        case Types.LONGNVARCHAR:
274        case Types.VARCHAR:
275        case Types.NVARCHAR:
276        case Types.CHAR:
277        case Types.NCHAR:
278        {
279            Reader reader = resultSet.getCharacterStream(col);
280            if (reader == null)
281                break;
282            int len;
283            char[] buf = new char[1024];
284            this.openColumnTag(col, columnName);
285            try {
286                while ((len = reader.read(buf)) != -1)
287                    this.emitText(buf, 0, len);
288                reader.close();
289            } catch (IOException e) {
290                throw (SQLException)new SQLException("error reading from CLOB").initCause(e);
291            }
292            this.closeTag();
293            return;
294        }
295        case Types.BIGINT:
296        case Types.DECIMAL:
297        case Types.DOUBLE:
298        case Types.FLOAT:
299        case Types.REAL:
300        case Types.NUMERIC:
301            value = resultSet.getBigDecimal(col);
302            break;
303        case Types.INTEGER:
304        case Types.TINYINT:
305        case Types.SMALLINT:
306            value = resultSet.getInt(col);
307            break;
308        case Types.DATE:
309        {
310            Date date = resultSet.getDate(col);
311            if (date != null)
312                value = new SimpleDateFormat(DATE_FORMAT).format(date);
313            break;
314        }
315        case Types.TIME:
316            value = resultSet.getTime(col);
317            break;
318        case Types.TIMESTAMP:
319        {
320            Timestamp timestamp = resultSet.getTimestamp(col);
321            if (timestamp != null)
322                value = new SimpleDateFormat(TIMESTAMP_FORMAT).format(timestamp);
323            break;
324        }
325        case Types.BLOB:
326        case Types.BINARY:
327        case Types.VARBINARY:
328        case Types.LONGVARBINARY:
329        {
330            InputStream input = resultSet.getBinaryStream(col);
331            if (input == null)
332                break;
333            this.openColumnTag(col, columnName);
334            try {
335                this.writeBinary(input);
336                input.close();
337            } catch (IOException e) {
338                throw (SQLException)new SQLException("error reading from BLOB").initCause(e);
339            }
340            this.closeTag();
341            return;
342        }
343        case Types.ROWID:
344        {
345            try {
346                value = resultSet.getRowId(col);
347            } catch (AbstractMethodError e) {
348                value = resultSet.getObject(col);
349            }
350            break;
351        }
352        // Unhandled cases...
353        case Types.ARRAY:
354        case Types.DATALINK:
355        case Types.DISTINCT:
356        case Types.JAVA_OBJECT:
357        case Types.NULL:
358        case Types.OTHER:
359        case Types.REF:
360        case Types.SQLXML:
361        case Types.STRUCT:
362        default:
363            value = resultSet.getObject(col);
364            break;
365        }
366
367        // Handle null by omitting the tag entirely
368        if (value == null || resultSet.wasNull())
369            return;
370
371        // Emit value
372        this.openColumnTag(col, columnName);
373        this.emitText(value.toString());
374        this.closeTag();
375    }
376
377    private void writeBinary(InputStream input) throws IOException, XMLStreamException {
378        int len;
379        byte[] buf = new byte[1024];
380        char[] ch = new char[2];
381        while ((len = input.read(buf)) != -1) {
382            for (int i = 0; i < len; i++) {
383                int val = buf[i] & 0xff;
384                ch[0] = Character.forDigit(val >> 4, 16);
385                ch[1] = Character.forDigit(val & 0xf, 16);
386                this.emitText(ch, 0, 2);
387            }
388        }
389    }

391// XML output formatting

393    private void openColumnTag(int col, String columnName) throws XMLStreamException {
394        this.openTag(columnName != null ? this.xmlify(columnName) : "column");
395        this.emitAttr("index", "" + col);
396    }
397
398    private String xmlify(String name) {
399        StringBuilder buf = new StringBuilder(name.length());
400        for (int i = 0; i < name.length(); i++) {
401            final char ch = name.charAt(i);
402            if (Character.isLetter(ch) || ch == '_' || ch == ':') {
403                buf.append(ch);
404                continue;
405            }
406            if (i > 0 && (Character.isDigit(ch) || ch == '.' || ch == '-')) {
407                buf.append(ch);
408                continue;
409            }
410            buf.append('_');
411        }
412        return buf.toString();
413    }
414
415    private void openTag(String name) throws XMLStreamException {
416        switch (this.state) {
417        case LAST_OPEN:
418            this.writer.writeCharacters("\n");
419            break;
420        case LAST_CLOSE:
421            break;
422        case LAST_TEXT:
423        default:
424            throw new RuntimeException("mixed element content not supported");
425        }
426        for (int i = 0; i < this.nesting; i++)
427            this.writer.writeCharacters(this.indentSpace);
428        this.writer.writeStartElement(name);
429        this.nesting++;
430        this.state = State.LAST_OPEN;
431    }
432
433    private void emitAttr(String name, String value) throws XMLStreamException {
434        this.writer.writeAttribute(name, value);
435    }
436
437    private void emitText(String content) throws XMLStreamException {
438        this.writer.writeCharacters(content);
439        this.state = State.LAST_TEXT;
440    }
441
442    private void emitText(char[] buf, int off, int len) throws XMLStreamException {
443        this.writer.writeCharacters(buf, off, len);
444        this.state = State.LAST_TEXT;
445    }
446
447    private void emitCData(String content) throws XMLStreamException {
448        this.writer.writeCData(content);
449        this.state = State.LAST_TEXT;
450    }
451
452    private void closeTag() throws XMLStreamException {
453        this.nesting--;
454        if (this.state == State.LAST_CLOSE) {
455            for (int i = 0; i < this.nesting; i++)
456                this.writer.writeCharacters(this.indentSpace);
457        }
458        this.writer.writeEndElement();
459        if (this.nesting > 0)
460            this.writer.writeCharacters("\n");
461        this.state = State.LAST_CLOSE;
462    }
