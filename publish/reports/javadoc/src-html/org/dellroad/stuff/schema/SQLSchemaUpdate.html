008    package org.dellroad.stuff.schema;
009    
010    import java.sql.Connection;
011    import java.sql.SQLException;
012    
013    /**
014     * A {@link SchemaUpdate} that performs its update using a {@link SQLDatabaseAction}.
015     */
016    public class SQLSchemaUpdate extends AbstractSchemaUpdate {
017    
018        private SQLDatabaseAction action;
019    
020        /**
021         * Configure the {@link SQLDatabaseAction}. This is a required property.
022         *
023         * @see SQLDatabaseAction
024         */
025        public void setSQLDatabaseAction(SQLDatabaseAction action) {
026            this.action = action;
027        }
028    
029        @Override
030        public void apply(Connection c) throws SQLException {
031            if (this.action == null)
032                throw new IllegalArgumentException("no SQLDatabaseAction configured");
033            this.action.apply(c);
034        }
035    }
