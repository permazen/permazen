008    package org.dellroad.stuff.schema;
009    
010    import java.sql.Connection;
011    import java.sql.SQLException;
012    import java.util.Comparator;
013    import java.util.HashMap;
014    import java.util.Map;
015    
016    import javax.sql.DataSource;
017    
018    import org.springframework.beans.factory.BeanFactory;
019    import org.springframework.beans.factory.BeanFactoryAware;
020    import org.springframework.beans.factory.InitializingBean;
021    import org.springframework.beans.factory.ListableBeanFactory;
022    import org.springframework.jdbc.BadSqlGrammarException;
023    import org.springframework.jdbc.core.ConnectionCallback;
024    import org.springframework.jdbc.core.JdbcTemplate;
025    
026    /**
027     * {@link SchemaUpdater} optimized for use with Spring.
028     * <ul>
029     * <li>{@link #databaseNeedsInitialization databaseNeedsInitialization()} and
030     *  {@link #applyAction(DataSource, DatabaseAction) applyAction()} are overridden to use Spring's {@link JdbcTemplate}
031     *  so Spring {@link org.springframework.dao.DataAccessException}s are thrown.</li>
032     * <li>{@link #databaseNeedsInitialization databaseNeedsInitialization()} is overridden to catch exceptions
033     *  and more precisely to filter out false positives.</li>
034     * <li>{@link #getOrderingTieBreaker} is overridden to break ties by ordering updates in the same order
035     *  as they are defined in the bean factory.</li>
036     * <li>This class implements {@link InitializingBean} and verifies all required properties are set.</li>
037     * <li>If no updates are {@linkplain #setUpdates explicitly configured}, then all {@link SchemaUpdate}s found
038     *  in the containing bean factory are automatically configured.
039     * </ul>
040     *
041     * <p>
042     * An example of how this class can be combined with custom XML to define an updater, all its updates,
043     * and a {@link SchemaUpdatingDataSource} that automatically updates the database schema:
044     * <blockquote><pre>
045     *  &lt;beans xmlns="http://www.springframework.org/schema/beans"
046     *    <b>xmlns:dellroad-stuff="http://dellroad-stuff.googlecode.com/schema/dellroad-stuff"</b>
047     *    xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
048     *    xmlns:p="http://www.springframework.org/schema/p"
049     *    xsi:schemaLocation="
050     *      http://www.springframework.org/schema/beans
051     *        http://www.springframework.org/schema/beans/spring-beans-3.0.xsd
052     *      <b>http://dellroad-stuff.googlecode.com/schema/dellroad-stuff
053     *        http://dellroad-stuff.googlecode.com/svn/wiki/schemas/dellroad-stuff-1.0.xsd</b>"&gt;
054     *
055     *     &lt;!-- DataSource that automatically updates the database schema --&gt;
056     *     <b>&lt;bean id="dataSource" class="org.dellroad.stuff.schema.SchemaUpdatingDataSource"
057     *       p:dataSource-ref="realDataSource" p:schemaUpdater-ref="schemaUpdater"/&gt;</b>
058     *
059     *     &lt;!-- Database updater; note use of dellroad-stuff's update table initialization for MySQL --&gt;
060     *     <b>&lt;bean id="schemaUpdater" class="org.dellroad.stuff.schema.SpringSchemaUpdater"&gt;
061     *         &lt;property name="databaseInitialization"&gt;
062     *             &lt;dellroad-stuff:sql resource="classpath:databaseInit.sql"/&gt;
063     *         &lt;/property&gt;
064     *         &lt;property name="updateTableInitialization"&gt;
065     *             &lt;dellroad-stuff:sql resource="classpath:org/dellroad/stuff/schema/updateTable-mysql.sql"/&gt;
066     *         &lt;/property&gt;
067     *     &lt;/bean&gt;</b>
068     *
069     *      &lt;!-- Schema update to add the 'phone' column to the 'User' table --&gt;
070     *      <b>&lt;dellroad-stuff:sql-update id="addPhone"&gt;ALTER TABLE User ADD phone VARCHAR(64)&lt;/dellroad-stuff:sql-update&gt;</b>
071     *
072     *      &lt;!-- Schema update to run some complicated external SQL script --&gt;
073     *      <b>&lt;dellroad-stuff:sql-update id="majorChanges" depends-on="addPhone" resource="classpath:majorChanges.sql"/&gt;</b>
074     *
075     *      &lt;!-- Multiple SQL commands that will be automatically separated into distinct updates --&gt;
076     *      <b>&lt;dellroad-stuff:sql-update id="renameColumn"&gt;
077     *          ALTER TABLE User ADD newName VARCHAR(64);
078     *          ALTER TABLE User SET newName = oldName;
079     *          ALTER TABLE User DROP oldName;
080     *      &lt;/dellroad-stuff:sql-update&gt;</b>
081     *
082     *      &lt;!-- add more schema updates over time as needed... --&gt;
083     *
084     *  &lt;/beans&gt;
085     * </pre></blockquote>
086     *
087     * <p>
088     * It is required that this updater and all of its schema updates are defined in the same {@link ListableBeanFactory}.
089     */
090    public class SpringSchemaUpdater extends SchemaUpdater implements BeanFactoryAware, InitializingBean {
091    
092        private ListableBeanFactory beanFactory;
093    
094        @Override
095        public void afterPropertiesSet() throws Exception {
096            if (this.beanFactory == null)
097                throw new Exception("no bean factory configured");
098            if (this.getDatabaseInitialization() == null)
099                throw new Exception("no database initialization configured");
100            if (this.getUpdateTableInitialization() == null)
101                throw new Exception("no update table initialization configured");
102            if (this.getUpdates() == null)
103                this.setUpdates(this.beanFactory.getBeansOfType(SchemaUpdate.class).values());
104        }
105    
106        @Override
107        public void setBeanFactory(BeanFactory beanFactory) {
108            if (!(beanFactory instanceof ListableBeanFactory))
109                throw new IllegalArgumentException("containing BeanFactory is not a ListableBeanFactory: " + beanFactory);
110            this.beanFactory = (ListableBeanFactory)beanFactory;
111        }
112    
113        /**
114         * Determine if the database needs initialization.
115         *
116         * <p>
117         * The implementation in {@link SpringSchemaUpdater} invokes <code>SELECT COUNT(*) FROM <i>UPDATETABLE</i></code>
118         * and checks for a {@link BadSqlGrammarException}.
119         *
120         * @throws org.springframework.dao.DataAccessException if there is any other problem
121         */
122        @Override
123        public boolean databaseNeedsInitialization(DataSource dataSource) {
124            try {
125                long numUpdates = new JdbcTemplate(dataSource).queryForLong("SELECT COUNT(*) FROM " + this.getUpdateTableName());
126                this.log.info("detected already initialized database, with " + numUpdates + " update(s) already applied");
127                return false;
128            } catch (BadSqlGrammarException e) {
129                log.warn("detected uninitialized database: update table `" + this.getUpdateTableName() + "' not found: " + e);
130                return true;
131            }
132        }
133    
134        /**
135         * Apply a {@link DatabaseAction} to a {@link DataSource}.
136         *
137         * <p>
138         * The implementation in {@link SpringSchemaUpdater} uses {@link JdbcTemplate} to apply the modification so that
139         * Spring {@link org.springframework.dao.DataAccessException}s are thrown in case of errors.
140         *
141         * @throws org.springframework.dao.DataAccessException if there is any problem
142         */
143        @Override
144        protected void applyAction(DataSource dataSource, final DatabaseAction action) {
145            new JdbcTemplate(dataSource).execute(new ConnectionCallback<Void>() {
146                @Override
147                public Void doInConnection(Connection c) throws SQLException {
148                    action.apply(c);
149                    return null;
150                }
151            });
152        }
153    
154        /**
155         * Get the preferred ordering of two updates that do not have any predecessor constraints
156         * (including implied indirect constraints) between them.
157         *
158         * <p>
159         * The {@link Comparator} returned by the implementation in {@link SpringSchemaUpdater} sorts
160         * updates in the same order that they appear in the {@link BeanFactory}.
161         */
162        @Override
163        protected Comparator<SchemaUpdate> getOrderingTieBreaker() {
164            String[] beanNames = this.beanFactory.getBeanDefinitionNames();
165            final Map<String, Integer> sort = new HashMap<String, Integer>(beanNames.length);
166            for (int i = 0; i < beanNames.length; i++)
167                sort.put(beanNames[i], i);
168            return new Comparator<SchemaUpdate>() {
169                @Override
170                public int compare(SchemaUpdate update1, SchemaUpdate update2) {
171                    String[] names = new String[] { update1.getName(), update2.getName() };
172                    Integer[] indexes = new Integer[] { sort.get(update1.getName()), sort.get(update2.getName()) };
173                    for (int i = 0; i < 2; i++) {
174                        if (indexes[i] == null) {
175                            throw new IllegalArgumentException("failed to find update `" + names[i]
176                              + "' in bean factory " + SpringSchemaUpdater.this.beanFactory);
177                        }
178                    }
179                    return indexes[0] - indexes[1];
180                };
181            };
182        }
183    }
