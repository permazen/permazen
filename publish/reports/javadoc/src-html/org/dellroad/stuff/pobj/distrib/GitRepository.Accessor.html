024/**
025 * Represents a {@code git(1)} repository and provides a few simple methods that wrap the command-line client
026 * and provide the functionality needed by {@link Synchronizer}.
027 * Only normal (i.e., non-bare) repositories are supported.
028 *
029 * <p>
030 * This class assumes the repository already has all branches and remotes already configured, and that no
031 * other external process is accessing the {@code git(1)} repository without our knowledge.
032 * </p>
033 *
034 * <p>
035 * Instances are thread safe and guarantees exclusive access to the working directory
036 * to each {@link Accessor} while it is in use.
037 * </p>
038 *
039 * <p>
040 * Requires git version at least 1.7.4.
041 * </p>
042 *
043 * @see <a href="http://git-scm.com/">Git web site</a>
044 */
045public class GitRepository extends AbstractBean {
046
047    /**
048     * Accepted git branch name pattern.
049     */
050    public static final Pattern BRANCH_NAME_PATTERN = Pattern.compile("[\\w][-:\\w]*");
051
052    /**
053     * Accepted git remote name pattern.
054     */
055    public static final Pattern REMOTE_NAME_PATTERN = Pattern.compile("[\\w][-:\\w]*");
056
057    /**
058     * SHA1 pattern.
059     */
060    public static final Pattern SHA1_PATTERN = Pattern.compile("[0-9a-f]{40}");
061
062    private final File dir;
063
064// Constructors
065
066    /**
067     * Constructor.
068     *
069     * @param dir repository filesystem location (i.e., the directory containing {@code .git} as a subdirectory)
070     * @throws IllegalArgumentException if {@code dir} is null
071     */
072    public GitRepository(File dir) {
073        if (dir == null)
074            throw new IllegalArgumentException("null dir");
075        this.dir = dir;
076    }
077
078// Lifecycle
079
080    @Override
081    public void afterPropertiesSet() throws Exception {
082        super.afterPropertiesSet();
083        final File gitDir = new File(this.dir, ".git");
084        if (!gitDir.exists() || !gitDir.isDirectory())
085            throw new Exception("`" + this.dir + "' is not a git repository");
086    }

088// Methods
089
090    /**
091     * Get the git working directory associated with this instance.
092     */
093    public File getDirectory() {
094        return this.dir;
095    }
096
097    /**
098     * Fetch data from all configured remotes to update our local copy of their information.
099     *
100     * @throws GitException if an error occurs
101     */
102    public synchronized void fetch() {
103        new GitCommand(this.dir, "fetch", "--all").run();
104    }
105
106    /**
107     * Fetch data from the named remotes to update our local copy of their information.
108     *
109     * @param remotes list of remotes
110     * @throws GitException if an error occurs
111     * @throws IllegalArgumentException if {@code remotes} or any element is null
112     * @throws IllegalArgumentException if an remote does not match {@link #REMOTE_NAME_PATTERN}
113     */
114    public synchronized void fetch(List<String> remotes) {
115
116        // Sanity check
117        if (remotes == null)
118            throw new IllegalArgumentException("null remotes");
119        for (String remote : remotes) {
120            if (remote == null)
121                throw new IllegalArgumentException("null remote");
122            if (!REMOTE_NAME_PATTERN.matcher(remote).matches())
123                throw new IllegalArgumentException("illegal remote name `" + remote + "'");
124        }
125
126        // Fetch
127        final ArrayList<String> params = new ArrayList<String>(remotes.size() + 2);
128        params.add("fetch");
129        params.add("--multiple");
130        params.addAll(remotes);
131        new GitCommand(this.dir, params).run();
132    }
133
134    /**
135     * Get the author date associated with the given commit.
136     *
137     * @param tree commit reference
138     * @return author date
139     * @throws GitException if git fails
140     * @throws IllegalArgumentException if {@code tree} is null
141     */
142    public synchronized Date getAuthorDate(String tree) {
143        try {
144            return new Date(Long.parseLong(new GitCommand(this.dir,
145              "log", "-1", "--format=format:%at", tree).runAndGetOutput(), 10) * 1000L);
146        } catch (NumberFormatException e) {
147            throw new GitException("error parsing git output", e);
148        }
149    }
150
151    /**
152     * Access the head of the specified branch or the named commit.
153     * Any changes made by the {@link Accessor} are discarded when this method completes.
154     *
155     * @param branch name of the branch or commit we want to access
156     * @param accessor callback interface to access the working directory contents
157     * @return commit name (SHA-1 hash) corresponding to what tree was accessed
158     * @throws GitException if {@code branch} is not a local branch or a commit ID
159     * @throws IllegalArgumentException if {@code branch} is null
160     * @throws IllegalArgumentException if {@code branch} does not match {@link #BRANCH_NAME_PATTERN}
161     * @throws IllegalArgumentException if {@code branch} is null
162     * @throws IllegalArgumentException if {@code accessor} is null
163     */
164    public synchronized String access(String branch, Accessor accessor) {
165
166        // Sanity check
167        if (branch == null)
168            throw new IllegalArgumentException("null branch");
169        if (!BRANCH_NAME_PATTERN.matcher(branch).matches())
170            throw new IllegalArgumentException("illegal branch name `" + branch + "'");
171        if (accessor == null)
172            throw new IllegalArgumentException("null accessor");
173
174        // Reset
175        this.reset();
176
177        // Checkout the commit
178        new GitCommand(this.dir, "checkout", "--force", branch).run();
179
180        // Get the commit ID that we have checked out
181        final String commit = this.getCurrentCommit();
182
183        // Grant access
184        try {
185            accessor.accessWorkingCopy(this.dir);
186        } finally {
187            this.reset();
188        }
189
190        // Done
191        return commit;
192    }
193
194    /**
195     * Commit changes onto the specified branch. The branch must already exist.
196     *
197     * <p>
198     * If after the {@code accessor} runs nothing has changed, then no commit is performed
199     * and the previous commit ID is returned.
200     * </p>
201     *
202     * @param branch name of the branch we want to commit onto
203     * @param accessor callback interface to update the working directory contents
204     * @param message commit message
205     * @return commit name (SHA-1 hash), or previous existing commit if no changes were made
206     * @throws GitException if {@code branch} is not an existing local branch
207     * @throws IllegalArgumentException if {@code branch} is null
208     * @throws IllegalArgumentException if {@code branch} does not match {@link #BRANCH_NAME_PATTERN}
209     * @throws IllegalArgumentException if {@code accessor} is null
210     * @throws IllegalArgumentException if {@code message} is null
211     */
212    public synchronized String commit(String branch, Accessor accessor, String message) {
213
214        // Sanity check
215        if (branch == null)
216            throw new IllegalArgumentException("null branch");
217        if (!BRANCH_NAME_PATTERN.matcher(branch).matches())
218            throw new IllegalArgumentException("illegal branch name `" + branch + "'");
219        if (accessor == null)
220            throw new IllegalArgumentException("null accessor");
221        if (message == null)
222            throw new IllegalArgumentException("null message");
223
224        // Get ref file and verify branch is really a local branch
225        final File refFile = this.getRepoFile("refs/heads/" + branch);
226        if (!refFile.exists())
227            throw new GitException("branch `" + branch + "' does not exist or is not a local branch");
228
229        // Reset
230        this.reset();
231
232        // Perform commit steps
233        try {
234
235            // Check out branch
236            new GitCommand(this.dir, "checkout", "--force", branch).run();
237
238            // Apply changes
239            accessor.accessWorkingCopy(this.dir);
240
241            // Attempt commit only if something has changed
242            if (new GitCommand(this.dir, "status", "--porcelain").runAndGetOutput().length() > 0) {
243
244                // Stage them into the index
245                new GitCommand(this.dir, "add", "--all").run();
246
247                // Commit them
248                new GitCommand(this.dir, "commit", "--message", message).run();
249            }
250        } finally {
251            this.reset();
252        }
253
254        // Done
255        return this.getCurrentCommit();
256    }

258    /**
259     * Ensure the specified local branch exists. If it does not, it will be created with an initial empty commit.
260     *
261     * @param branch name of the branch we want to ensure exists
262     * @param message commit message (only used if branch does not yet exist)
263     * @throws IllegalArgumentException if {@code branch} is null
264     * @throws IllegalArgumentException if {@code branch} does not match {@link #BRANCH_NAME_PATTERN}
265     * @throws IllegalArgumentException if {@code message} is null
266     */
267    public synchronized void ensureBranch(String branch, String message) {
268
269        // Sanity check
270        if (branch == null)
271            throw new IllegalArgumentException("null branch");
272        if (!BRANCH_NAME_PATTERN.matcher(branch).matches())
273            throw new IllegalArgumentException("illegal branch name `" + branch + "'");
274        if (message == null)
275            throw new IllegalArgumentException("null message");
276
277        // Does branch already exist?
278        final File refFile = this.getRepoFile("refs/heads/" + branch);
279        if (refFile.exists())
280            return;
281
282        // Create a commit with no parents and an empty tree
283        this.log.info("creating new local branch `" + branch + "' starting with an empty commit in directory `" + this.dir + "'");
284        new GitCommand(this.dir, "read-tree", "--empty").run();
285        final String tree = new GitCommand(this.dir, "write-tree").runAndGetOutput();
286        if (!SHA1_PATTERN.matcher(tree).matches())
287            throw new GitException("can't interpret output from `git write-tree': " + tree);
288        final String commit = new GitCommand(this.dir, "commit-tree", "-m", message, tree).runAndGetOutput();
289
290        // Create a branch starting there
291        new GitCommand(this.dir, "branch", branch, commit).run();
292    }

294    /**
295     * Compare two committed trees for equality.
296     *
297     * @param tree1 name of the first tree (branch name, commit, etc.)
298     * @param tree2 name of the second tree (branch name, commit, etc.)
299     * @return true if the trees are equal, otherwise false
300     * @throws GitException if {@code tree1} or {@code tree2} is not a valid branch or commit reference
301     * @throws IllegalArgumentException if {@code tree1} or {@code tree2} is null
302     */
303    public synchronized boolean equalTrees(String tree1, String tree2) {
304        final GitCommand diff = new GitCommand(this.dir, "diff", "--quiet", tree1, tree2);
305        final int exitValue = diff.run(true);
306        switch (exitValue) {
307        case 0:
308            return true;
309        case 1:
310            return false;
311        default:
312            throw new GitException("command `" + diff + "' in directory `" + this.dir + "' failed with exit value " + exitValue);
313        }
314    }

316    /**
317     * Merge and commit.
318     *
319     * <p>
320     * Trivial merges, i.e., when {@code other} is already an ancestor of {@code branch}, do not result in a new commit
321     * and the {@code accessor} is not run.
322     * </p>
323     *
324     * @param branch name of the branch we want to commit onto
325     * @param other name of the other commit or branch we want to merge into {@code branch}
326     * @param strategy merge strategy
327     * @param accessor callback interface to access the working directory contents after the merge; may be null
328     * @param message commit message
329     * @return commit name (SHA-1 hash)
330     * @throws GitException if {@code branch} is not a local branch
331     * @throws GitException if {@code other} is not a valid branch or commit reference
332     * @throws GitMergeConflictException if the merge fails with conflicts
333     * @throws IllegalArgumentException if {@code branch} or {@code other} is null
334     * @throws IllegalArgumentException if {@code branch} does not match {@link #BRANCH_NAME_PATTERN}
335     * @throws IllegalArgumentException if {@code message} is null
336     */
337    public synchronized String merge(String branch, String other, MergeStrategy strategy, Accessor accessor, String message) {
338
339        // Sanity check
340        if (branch == null)
341            throw new IllegalArgumentException("null branch");
342        if (!BRANCH_NAME_PATTERN.matcher(branch).matches())
343            throw new IllegalArgumentException("illegal branch name `" + branch + "'");
344        if (other == null)
345            throw new IllegalArgumentException("null other");
346        if (strategy == null)
347            throw new IllegalArgumentException("null strategy");
348        if (message == null)
349            throw new IllegalArgumentException("null message");
350
351        // Get ref file and verify branch is really a local branch
352        final File refFile = this.getRepoFile("refs/heads/" + branch);
353        if (!refFile.exists())
354            throw new GitException("branch `" + branch + "' does not exist or is not a local branch");
355
356        // Reset
357        this.reset();
358
359        // Perform merge steps
360        try {
361
362            // Check out branch
363            new GitCommand(this.dir, "checkout", "--force", branch).run();
364
365            // Perform the merge
366            strategy.merge(this.dir, other);
367
368            // Detect and ignore trivial merge
369            if (new GitCommand(this.dir, "status", "--porcelain").runAndGetOutput().length() > 0) {
370
371                // Grant access
372                if (accessor != null)
373                    accessor.accessWorkingCopy(this.dir);
374
375                // Commit changes
376                new GitCommand(this.dir, "commit", "--message", message).run();
377            }
378        } finally {
379
380            // Clean up
381            this.reset();
382        }
383
384        // Done
385        return this.getCurrentCommit();
386    }

388    /**
389     * Read file possibly containing a reference and follow it until we get a commit ID.
390     *
391     * @param filename filename relative to {@code .git}, e.g., {@code "HEAD"} or {@code "refs/heads/foobar"}
392     * @return SHA-1 commit ID
393     * @throws GitException if we can't figure it out
394     */
395    public String followReference(String filename) {
396        for (int i = 0; i < 10; i++) {
397
398            // Read file
399            final File file = this.getRepoFile(filename);
400            String line;
401            try {
402                line = this.readFirstLine(file);
403            } catch (IOException e) {
404                throw new GitException("error reading `" + file + "'", e);
405            }
406
407            // Is it a SHA-1?
408            if (SHA1_PATTERN.matcher(line).matches())
409                return line;
410
411            // Is it a reference?
412            final Matcher matcher = Pattern.compile("ref: (.*)$").matcher(line);
413            if (matcher.matches()) {
414                filename = matcher.group(1);
415                continue;
416            }
417
418            // Dunno
419            throw new GitException("can't interpret contents of `" + file + "': " + line);
420        }
421        throw new GitException("too many levels of git references");
422    }

424    /**
425     * Clean out and reset working directory.
426     */
427    protected void reset() {
428        new GitCommand(this.dir, "clean", "-xdf").run();                      // clean out working directory
429        new GitCommand(this.dir, "reset", "--hard").run();                    // discard uncommitted changes
430    }
431
432    /**
433     * Get the commit ID of what we have currently checked out (i.e., our HEAD).
434     *
435     * @throws GitException if we can't figure it out
436     */
437    protected String getCurrentCommit() {
438        return this.followReference("HEAD");
439    }
440
441    /**
442     * Read the first line of a file (trimmed) as a {@link String}.
443     */
444    protected String readFirstLine(File file) throws IOException {
445        final FileInputStream inputStream = new FileInputStream(file);
446        try {
447            final LineNumberReader reader = new LineNumberReader(new InputStreamReader(inputStream, Charset.forName("UTF-8")));
448            final String firstLine = reader.readLine();
449            if (firstLine == null)
450                throw new GitException("read empty content from `" + file + "'");
451            return firstLine.trim();
452        } finally {
453            try {
454                inputStream.close();
455            } catch (IOException e) {
456                // ignore
457            }
458        }
459    }

462     * Get the {@link File} corresponding to the given relative path under the {@code .git} directory.
463     *
464     * @param path relative path, e.g., {@code HEAD} or {@code refs/heads/foo}
465     */
466    public File getRepoFile(String path) {
467        return new File(new File(this.dir, ".git"), path);
468    }
