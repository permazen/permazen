008    package org.dellroad.stuff.pobj;
009    
010    /**
011     * Thrown by {@link AbstractConfiguredBean#getRequiredConfig} when the bean is not configured.
012     */
013    @SuppressWarnings("serial")
014    public class NotConfiguredException extends IllegalStateException {
015    
016        public NotConfiguredException() {
017        }
018    
019        public NotConfiguredException(Throwable t) {
020            super(t);
021        }
022    
023        public NotConfiguredException(String message) {
024            super(message);
025        }
026    
027        public NotConfiguredException(String message, Throwable t) {
028            super(message, t);
029        }
030    }
