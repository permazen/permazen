028/**
029 * Represents a {@code git(1)} repository and provides a few simple methods that wrap the command-line client
030 * and provide the functionality needed by {@link Synchronizer}.
031 * Only normal (i.e., non-bare) repositories are supported.
032 *
033 * <p>
034 * This class assumes the repository already has all branches and remotes already configured, and that no
035 * other external process is accessing the {@code git(1)} repository without our knowledge.
036 * </p>
037 *
038 * <p>
039 * Instances are thread safe and guarantees exclusive access to the working directory
040 * to each {@link Accessor} while it is in use.
041 * </p>
042 *
043 * @see <a href="http://git-scm.com/">Git web site</a>
044 */
045public class GitRepository extends AbstractBean {
046
047    /**
048     * Accepted git branch name pattern.
049     */
050    public static final Pattern BRANCH_NAME_PATTERN = Pattern.compile("[\\w][-:\\w]*");
051
052    /**
053     * Accepted git remote name pattern.
054     */
055    public static final Pattern REMOTE_NAME_PATTERN = Pattern.compile("[\\w][-:\\w]*");
056
057    /**
058     * SHA1 pattern.
059     */
060    public static final Pattern SHA1_PATTERN = Pattern.compile("[0-9a-f]{40}");
061
062    private final File dir;

066    /**
067     * Constructor.
068     *
069     * @param dir repository filesystem location (i.e., the directory containing {@code .git} as a subdirectory)
070     * @throws IllegalArgumentException if {@code dir} is null
071     */
072    public GitRepository(File dir) {
073        if (dir == null)
074            throw new IllegalArgumentException("null dir");
075        this.dir = dir;
076    }

082        super.afterPropertiesSet();
083        final File gitDir = new File(this.dir, ".git");
084        if (!gitDir.exists() || !gitDir.isDirectory())
085            throw new Exception("`" + this.dir + "' is not a git repository");
086    }

097    /**
098     * Fetch data from all configured remotes to update our local copy of their information.
099     *
100     * @throws GitException if an error occurs
101     */
102    public synchronized void fetch() {
103        new GitCommand("fetch", "--all").run();
104    }
105
106    /**
107     * Fetch data from the named remotes to update our local copy of their information.
108     *
109     * @param remotes list of remotes
110     * @throws GitException if an error occurs
111     * @throws IllegalArgumentException if {@code remotes} or any element is null
112     * @throws IllegalArgumentException if an remote does not match {@link #REMOTE_NAME_PATTERN}
113     */
114    public synchronized void fetch(List<String> remotes) {
115
116        // Sanity check
117        if (remotes == null)
118            throw new IllegalArgumentException("null remotes");
119        for (String remote : remotes) {
120            if (remote == null)
121                throw new IllegalArgumentException("null remote");
122            if (!REMOTE_NAME_PATTERN.matcher(remote).matches())
123                throw new IllegalArgumentException("illegal remote name `" + remote + "'");
124        }
125
126        // Fetch
127        final ArrayList<String> params = new ArrayList<String>(remotes.size() + 2);
128        params.add("fetch");
129        params.add("--multiple");
130        params.addAll(remotes);
131        new GitCommand(params).run();
132    }

134    /**
135     * Get the author date associated with the given commit.
136     *
137     * @param tree commit reference
138     * @return author date
139     * @throws GitException if git fails
140     * @throws IllegalArgumentException if {@code tree} is null
141     */
142    public synchronized Date getAuthorDate(String tree) {
143        try {
144            return new Date(Long.parseLong(new GitCommand("log", "-1", "--format=format:%at", tree).runAndGetOutput(), 10) * 1000L);
145        } catch (NumberFormatException e) {
146            throw new GitException("error parsing git output", e);
147        }
148    }

150    /**
151     * Access the head of the specified branch or the named commit.
152     * Any changes made by the {@link Accessor} are discarded when this method completes.
153     *
154     * @param branch name of the branch or commit we want to access
155     * @param accessor callback interface to access the working directory contents
156     * @return commit name (SHA-1 hash) corresponding to what tree was accessed
157     * @throws GitException if {@code branch} is not a local branch or a commit ID
158     * @throws IllegalArgumentException if {@code branch} is null
159     * @throws IllegalArgumentException if {@code branch} does not match {@link #BRANCH_NAME_PATTERN}
160     * @throws IllegalArgumentException if {@code branch} is null
161     * @throws IllegalArgumentException if {@code accessor} is null
162     */
163    public synchronized String access(String branch, Accessor accessor) {
164
165        // Sanity check
166        if (branch == null)
167            throw new IllegalArgumentException("null branch");
168        if (!BRANCH_NAME_PATTERN.matcher(branch).matches())
169            throw new IllegalArgumentException("illegal branch name `" + branch + "'");
170        if (accessor == null)
171            throw new IllegalArgumentException("null accessor");
172
173        // Reset
174        this.reset();
175
176        // Checkout the commit
177        new GitCommand("checkout", "--force", branch).run();
178
179        // Get the commit ID that we have checked out
180        final String commit = this.getCurrentCommit();
181
182        // Grant access
183        try {
184            accessor.accessWorkingCopy(this.dir);
185        } finally {
186            this.reset();
187        }
188
189        // Done
190        return commit;
191    }

193    /**
194     * Commit changes onto the specified branch. The branch must already exist.
195     *
196     * <p>
197     * If after the {@code accessor} runs nothing has changed, then no commit is performed
198     * and the previous commit ID is returned.
199     * </p>
200     *
201     * @param branch name of the branch we want to commit onto
202     * @param accessor callback interface to update the working directory contents
203     * @param message commit message
204     * @return commit name (SHA-1 hash), or previous existing commit if no changes were made
205     * @throws GitException if {@code branch} is not an existing local branch
206     * @throws IllegalArgumentException if {@code branch} is null
207     * @throws IllegalArgumentException if {@code branch} does not match {@link #BRANCH_NAME_PATTERN}
208     * @throws IllegalArgumentException if {@code accessor} is null
209     * @throws IllegalArgumentException if {@code message} is null
210     */
211    public synchronized String commit(String branch, Accessor accessor, String message) {
212
213        // Sanity check
214        if (branch == null)
215            throw new IllegalArgumentException("null branch");
216        if (!BRANCH_NAME_PATTERN.matcher(branch).matches())
217            throw new IllegalArgumentException("illegal branch name `" + branch + "'");
218        if (accessor == null)
219            throw new IllegalArgumentException("null accessor");
220        if (message == null)
221            throw new IllegalArgumentException("null message");
222
223        // Get ref file and verify branch is really a local branch
224        final File refFile = this.getRepoFile("refs/heads/" + branch);
225        if (!refFile.exists())
226            throw new GitException("branch `" + branch + "' does not exist or is not a local branch");
227
228        // Reset
229        this.reset();
230
231        // Perform commit steps
232        try {
233
234            // Check out branch
235            new GitCommand("checkout", "--force", branch).run();
236
237            // Apply changes
238            accessor.accessWorkingCopy(this.dir);
239
240            // Attempt commit only if something has changed
241            if (new GitCommand("status", "--porcelain").runAndGetOutput().length() > 0) {
242
243                // Stage them into the index
244                new GitCommand("add", "--all").run();
245
246                // Commit them
247                new GitCommand("commit", "--message", message).run();
248            }
249        } finally {
250            this.reset();
251        }
252
253        // Done
254        return this.getCurrentCommit();
255    }

257    /**
258     * Ensure the specified local branch exists. If it does not, it will be created with an initial empty commit.
259     *
260     * @param branch name of the branch we want to ensure exists
261     * @param message commit message (only used if branch does not yet exist)
262     * @throws IllegalArgumentException if {@code branch} is null
263     * @throws IllegalArgumentException if {@code branch} does not match {@link #BRANCH_NAME_PATTERN}
264     * @throws IllegalArgumentException if {@code message} is null
265     */
266    public synchronized void ensureBranch(String branch, String message) {
267
268        // Sanity check
269        if (branch == null)
270            throw new IllegalArgumentException("null branch");
271        if (!BRANCH_NAME_PATTERN.matcher(branch).matches())
272            throw new IllegalArgumentException("illegal branch name `" + branch + "'");
273        if (message == null)
274            throw new IllegalArgumentException("null message");
275
276        // Does branch already exist?
277        final File refFile = this.getRepoFile("refs/heads/" + branch);
278        if (refFile.exists())
279            return;
280
281        // Create a commit with no parents and an empty tree
282        this.log.info("creating new local branch `" + branch + "' starting with an empty commit in directory `" + this.dir + "'");
283        new GitCommand("read-tree", "--empty").run();
284        final String tree = new GitCommand("write-tree").runAndGetOutput();
285        if (!SHA1_PATTERN.matcher(tree).matches())
286            throw new GitException("can't interpret output from `git write-tree': " + tree);
287        final String commit = new GitCommand("commit-tree", "-m", message, tree).runAndGetOutput();
288
289        // Create a branch starting there
290        new GitCommand("branch", branch, commit).run();
291    }

293    /**
294     * Compare two committed trees for equality.
295     *
296     * @param tree1 name of the first tree (branch name, commit, etc.)
297     * @param tree2 name of the second tree (branch name, commit, etc.)
298     * @return true if the trees are equal, otherwise false
299     * @throws GitException if {@code tree1} or {@code tree2} is not a valid branch or commit reference
300     * @throws IllegalArgumentException if {@code tree1} or {@code tree2} is null
301     */
302    public synchronized boolean equalTrees(String tree1, String tree2) {
303        final GitCommand diff = new GitCommand("diff", "--quiet", tree1, tree2);
304        final int exitValue = diff.run(true);
305        switch (exitValue) {
306        case 0:
307            return true;
308        case 1:
309            return false;
310        default:
311            throw new GitException("command `" + diff + "' in directory `" + GitRepository.this.dir
312              + "' failed with exit value " + exitValue);
313        }
314    }

316    /**
317     * Merge and commit.
318     *
319     * <p>
320     * Trivial merges, i.e., when {@code other} is already an ancestor of {@code branch}, do not result in a new commit
321     * and the {@code accessor} is not run.
322     * </p>
323     *
324     * @param branch name of the branch we want to commit onto
325     * @param other name of the other commit or branch we want to merge into {@code branch}
326     * @param strategy merge strategy
327     * @param accessor callback interface to access the working directory contents after the merge; may be null
328     * @param message commit message
329     * @return commit name (SHA-1 hash)
330     * @throws GitException if {@code branch} is not a local branch
331     * @throws GitException if {@code other} is not a valid branch or commit reference
332     * @throws GitMergeConflictException if the merge fails with conflicts
333     * @throws IllegalArgumentException if {@code branch} or {@code other} is null
334     * @throws IllegalArgumentException if {@code branch} does not match {@link #BRANCH_NAME_PATTERN}
335     * @throws IllegalArgumentException if {@code message} is null
336     */
337    public synchronized String merge(String branch, String other, MergeStrategy strategy, Accessor accessor, String message) {
338
339        // Sanity check
340        if (branch == null)
341            throw new IllegalArgumentException("null branch");
342        if (!BRANCH_NAME_PATTERN.matcher(branch).matches())
343            throw new IllegalArgumentException("illegal branch name `" + branch + "'");
344        if (other == null)
345            throw new IllegalArgumentException("null other");
346        if (strategy == null)
347            throw new IllegalArgumentException("null strategy");
348        if (message == null)
349            throw new IllegalArgumentException("null message");
350
351        // Get ref file and verify branch is really a local branch
352        final File refFile = this.getRepoFile("refs/heads/" + branch);
353        if (!refFile.exists())
354            throw new GitException("branch `" + branch + "' does not exist or is not a local branch");
355
356        // Reset
357        this.reset();
358
359        // Perform merge steps
360        try {
361
362            // Check out branch
363            new GitCommand("checkout", "--force", branch).run();
364
365            // Attempt the merge
366            final GitCommand merge = new GitCommand("merge", "--no-commit", "--no-ff");
367            strategy.addOptions(merge.getArgs());
368            merge.getArgs().add("--");
369            merge.getArgs().add(other);
370            final int exitValue = merge.run(true);
371            if (exitValue != 0) {
372
373                // Detect other types of error (other than conflicts)
374                if (!this.getRepoFile("MERGE_HEAD").exists())
375                    throw new GitException("git merge failed: " + merge.getStandardError().trim());
376
377                // Abort the merge
378                new GitCommand("merge", "--abort").run();
379
380                // Bail out
381                throw new GitMergeConflictException("merge failed with conflict(s)");
382            }
383
384            // Detect and ignore trivial merge
385            if (new GitCommand("status", "--porcelain").runAndGetOutput().length() > 0) {
386
387                // Grant access
388                if (accessor != null)
389                    accessor.accessWorkingCopy(this.dir);
390
391                // Commit changes
392                new GitCommand("commit", "--message", message).run();
393            }
394        } finally {
395
396            // Clean up
397            this.reset();
398        }
399
400        // Done
401        return this.getCurrentCommit();
402    }

404    /**
405     * Read file possibly containing a reference and follow it until we get a commit ID.
406     *
407     * @param filename filename relative to {@code .git}, e.g., {@code "HEAD"} or {@code "refs/heads/foobar"}
408     * @return SHA-1 commit ID
409     * @throws GitException if we can't figure it out
410     */
411    public String followReference(String filename) {
412        for (int i = 0; i < 10; i++) {
413
414            // Read file
415            final File file = this.getRepoFile(filename);
416            String line;
417            try {
418                line = this.readFirstLine(file);
419            } catch (IOException e) {
420                throw new GitException("error reading `" + file + "'", e);
421            }
422
423            // Is it a SHA-1?
424            if (SHA1_PATTERN.matcher(line).matches())
425                return line;
426
427            // Is it a reference?
428            final Matcher matcher = Pattern.compile("ref: (.*)$").matcher(line);
429            if (matcher.matches()) {
430                filename = matcher.group(1);
431                continue;
432            }
433
434            // Dunno
435            throw new GitException("can't interpret contents of `" + file + "': " + line);
436        }
437        throw new GitException("too many levels of git references");
438    }

440    /**
441     * Clean out and reset working directory.
442     */
443    protected void reset() {
444        new GitCommand("clean", "-xdf").run();                      // clean out working directory
445        new GitCommand("reset", "--hard").run();                    // discard uncommitted changes
446    }

448    /**
449     * Get the commit ID of what we have currently checked out (i.e., our HEAD).
450     *
451     * @throws GitException if we can't figure it out
452     */
453    protected String getCurrentCommit() {
454        return this.followReference("HEAD");
455    }

457    /**
458     * Read the first line of a file (trimmed) as a {@link String}.
459     */
460    protected String readFirstLine(File file) throws IOException {
461        final FileInputStream inputStream = new FileInputStream(file);
462        try {
463            final LineNumberReader reader = new LineNumberReader(new InputStreamReader(inputStream, Charset.forName("UTF-8")));
464            final String firstLine = reader.readLine();
465            if (firstLine == null)
466                throw new GitException("read empty content from `" + file + "'");
467            return firstLine.trim();
468        } finally {
469            try {
470                inputStream.close();
471            } catch (IOException e) {
472                // ignore
473            }
474        }
475    }

504    /**
505     * A simple wrapper around the {@code git(1)} command.
506     */
507    public class GitCommand {
508
509        private final Logger log = LoggerFactory.getLogger(this.getClass());
510        private final List<String> args = new ArrayList<String>();
511
512        private ProcessRunner runner;
513
514        /**
515         * Constructor.
516         *
517         * @param params the command line arguments (not including the initial {@code "git"})
518         */
519        public GitCommand(String... params) {
520            this(Arrays.asList(params));
521        }
522
523        /**
524         * Constructor.
525         *
526         * @param params the command line arguments (not including the initial {@code "git"})
527         */
528        public GitCommand(List<String> params) {
529            if (params == null)
530                throw new IllegalArgumentException("null params");
531            this.args.add("git");
532            this.args.addAll(params);
533        }
534
535        /**
536         * Get command line arguments.
537         */
538        public List<String> getArgs() {
539            return this.args;
540        }
541
542        /**
543         * Run command. Equivalent to {@code run(false)}.
544         */
545        public int run() {
546            return this.run(false);
547        }
548
549        /**
550         * Run command.
551         *
552         * @param errorOk true if a non-zero exit value may be expected
553         * @return {@code git(1)} exit value; if {@code errorOk} is false this will always be zero
554         * @throws IllegalStateException if this command has already been run
555         * @throws RuntimeException if the current thread is interrupted
556         * @throws GitException if any {@code GIT_*} environment variables are set
557         * @throws GitException if {@code errorOk} is false and {@code git(1}} exits with a non-zero value
558         */
559        public int run(boolean errorOk) {
560
561            // Sanity check
562            if (this.runner != null)
563                throw new IllegalStateException("command already executed");
564
565            // Sanity check environment
566            final ArrayList<String> vars = new ArrayList<String>();
567            for (String var : System.getenv().keySet()) {
568                if (var.startsWith("GIT_"))
569                    vars.add(var);
570            }
571            if (!vars.isEmpty())
572                throw new GitException("need to unset GIT_* environment variables first: " + vars);
573
574            // Start process
575            Process process;
576            try {
577                process = Runtime.getRuntime().exec(args.toArray(new String[args.size()]), null, GitRepository.this.dir);
578            } catch (IOException e) {
579                this.log.debug("command `" + this + "' in directory `" + GitRepository.this.dir + "' failed: ", e);
580                throw new GitException("error invoking git(1)", e);
581            }
582            this.runner = new ProcessRunner(process);
583
584            // Let it finish
585            int exitValue;
586            try {
587                exitValue = this.runner.run();
588            } catch (InterruptedException e) {
589                throw new RuntimeException("unexpected exception", e);      // XXX
590            }
591
592            // Log command output
593            final StringBuilder buf = new StringBuilder();
594            final String[] stdout = this.getStandardOutput().trim().split("\n");
595            if (!(stdout.length == 1 && stdout[0].length() == 0)) {
596                for (String line : stdout)
597                    buf.append("\n  [stdout] ").append(line);
598            }
599            final String[] stderr = this.getStandardError().trim().split("\n");
600            if (!(stderr.length == 1 && stderr[0].length() == 0)) {
601                for (String line : stderr)
602                    buf.append("\n  [stderr] ").append(line);
603            }
604
605            // Check exit value
606            if (exitValue != 0) {
607                final String msg = "command `" + this + "' in directory `" + GitRepository.this.dir
608                  + "' failed with exit value " + exitValue;
609                if (errorOk)
610                    this.log.debug(msg);
611                else {
612                    this.log.error(msg);
613                    throw new GitException(msg);
614                }
615            } else {
616                this.log.debug("successfully executed command `" + this + "' in directory `"
617                  + GitRepository.this.dir + "'" + buf);
618            }
619
620            // Done
621            return exitValue;
622        }
623
624        /**
625         * Run command and return standard output, interpreted as a UTF-8 string, with leading and trailing whitespace trimmed.
626         *
627         * @throws IllegalStateException if this command has already been run
628         * @throws RuntimeException if the current thread is interrupted
629         * @throws GitException if any {@code GIT_*} environment variables are set
630         * @throws GitException if {@code git(1}} exits with a non-zero value
631         */
632        public String runAndGetOutput() {
633            this.run();
634            return this.getStandardOutput().trim();
635        }
636
637        /**
638         * Get the standard output from {@code git(1)}.
639         *
640         * @throws IllegalStateException if command has not yet completed
641         * @return command standard output interpreted as UTF-8
642         */
643        public String getStandardOutput() {
644            if (this.runner == null)
645                throw new IllegalStateException("command has not yet executed");
646            return new String(this.runner.getStandardOutput(), Charset.forName("UTF-8"));
647        }
648
649        /**
650         * Get the standard error from {@code git(1)}.
651         *
652         * @throws IllegalStateException if command has not yet completed
653         * @return command standard error interpreted as UTF-8
654         */
655        public String getStandardError() {
656            if (this.runner == null)
657                throw new IllegalStateException("command has not yet executed");
658            return new String(this.runner.getStandardError(), Charset.forName("UTF-8"));
659        }
660
661        @Override
662        public String toString() {
663            final StringBuilder buf = new StringBuilder();
664            for (String arg : this.args) {
665                if (buf.length() > 0)
666                    buf.append(' ');
667                buf.append(arg);
668            }
669            return buf.toString();
670        }
671    }
