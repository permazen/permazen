008package org.dellroad.stuff.pobj.vaadin;
009
010import com.vaadin.server.VaadinSession;
011
012import org.dellroad.stuff.pobj.PersistentObject;
013import org.dellroad.stuff.vaadin7.AbstractSimpleContainer;
014
015/**
016 * Facilitates populating a {@link AbstractSimpleContainer} with data from {@link PersistentObject}
017 * and automatically updating it when than {@link PersistentObject} changes.
018 *
019 * <p>
020 * The constructor requires a {@link Generator} to generate, from the {@link PersistentObject}
021 * root object, the Java objects that populate the {@link AbstractSimpleContainer}.
022 * </p>
023 *
024 * <p>
025 * <b>Note:</b> Instances of this class start out in a non-listening state. You must invoke {@link #connect connect()}
026 * to start listening to the {@link PersistentObject} and (re)load the container contents.
027 * You should invoke {@link #disconnect} whenever the container is no longer used to avoid a memory leak.
028 * Typically, {@link #connect connect()} and {@link #disconnect} are invoked from the {@link com.vaadin.ui.Component#attach}
029 * and {@link com.vaadin.ui.Component#detach} methods (respectively) of the GUI widget that is using the container.
030 * </p>
031 *
032 * @param <T> type of the persistent object
033 * @param <K> the type of the Java objects that back each {@link com.vaadin.data.Item} in the container
034 */
035@SuppressWarnings("serial")
036public class PersistentObjectContainerLoader<T, K> {
037
038    private final AbstractSimpleContainer<?, K> container;
039    private final Generator<T, K> generator;
040
041    private VaadinPersistentObjectListener<T> listener;
042
043    /**
044     * Constructor.
045     *
046     * @param container container to populate
047     * @param generator object that generates the Java objects that back the {@link com.vaadin.data.Item}s in the container
048     * @throws IllegalArgumentException if {@code container} is null
049     * @throws IllegalArgumentException if {@code generator} is null
050     */
051    public PersistentObjectContainerLoader(AbstractSimpleContainer<?, K> container, Generator<T, K> generator) {
052        if (container == null)
053            throw new IllegalArgumentException("null container");
054        if (generator == null)
055            throw new IllegalArgumentException("null generator");
056        this.container = container;
057        this.generator = generator;
058    }
059
060    /**
061     * Register this container as a listener on the persistent object and load the container.
062     * This method may only be invoked while the {@link VaadinSession} is locked by the current thread.
063     * If this method is invoked twice with no intervening {@link #disconnect}, this instance will
064     * automatically disconnect from the old {@link PersistentObject} before connecting to the new one.
065     *
066     * @throws IllegalArgumentException if {@code persistentObject} is null
067     * @throws IllegalStateException if there is no {@link VaadinSession} associated with the current thread
068     */
069    public void connect(PersistentObject<T> persistentObject) {
070
071        // Sanity check
072        this.assertSession();
073        if (persistentObject == null)
074            throw new IllegalArgumentException("null persistentObject");
075
076        // Connect
077        boolean needReload = false;
078        T root = null;
079        synchronized (this) {
080            if (this.listener != null)
081                this.listener.unregister();
082            this.listener = new VaadinPersistentObjectListener<T>(persistentObject) {
083                @Override
084                protected void handlePersistentObjectChange(T oldRoot, T newRoot, long version) {
085                    PersistentObjectContainerLoader.this.reload(newRoot);
086                }
087            };
088            this.listener.register();
089            root = persistentObject.getSharedRoot();
090            needReload = true;
091        }
092
093        // Load container if there is data
094        if (needReload)
095            this.reload(root);
096    }
097
098    /**
099     * Shutdown this container. This unregisters the container from the {@link PersistentObject} it was previously
100     * {@linkplain #connect connected} to. This method may be invoked from any thread and is idempotent.
101     * The associated container is not notified about any change.
102     *
103     * @throws IllegalStateException if there is no {@link VaadinSession} associated with the current thread
104     */
105    public synchronized void disconnect() {
106        this.assertSession();
107        if (this.listener != null) {
108            this.listener.unregister();
109            this.listener = null;
110        }
111    }
112
113    /**
114     * Reload the container associated with this instance using the given {@link PersistentObject} root.
115     *
116     * @throws IllegalStateException if there is no {@link VaadinSession} associated with the current thread
117     */
118    protected void reload(T root) {
119        this.assertSession();
120        this.container.load(this.generator.generateContainerObjects(root));
121    }
122
123    /**
124     * Verify there is a {@link VaadinSession} associated with the current thread.
125     *
126     * @throws IllegalStateException if there is no {@link VaadinSession} associated with the current thread
127     */
128    protected void assertSession() {
129        if (VaadinSession.getCurrent() == null)
130             throw new IllegalStateException("there is no VaadinSession associated with the current thread");
131    }
132
133// Generator
134
135    /**
136     * Interface used by {@link PersistentObjectContainerLoader} to derive {@link AbstractSimpleContainer} backing
137     * objects from a {@link PersistentObject} root.
138     *
139     * @param <T> type of the persistent object
140     * @param <K> the type of the Java objects that back each {@link com.vaadin.data.Item} in the {@link AbstractSimpleContainer}
141     */
142    public interface Generator<T, K> {
143
144        /**
145         * Generate the Java objects that will back the {@link AbstractSimpleContainer} associated with
146         * a {@link PersistentObjectContainerLoader} from the given {@link PersistentObject} root.
147         *
148         * <p>
149         * This method should not modify {@code root}.
150         * </p>
151         *
152         * <p>
153         * If the associated {@link PersistentObject} allows empty starts or stops, then {@code root} might be null.
154         * </p>
155         *
156         * @param root updated persistent object root, or null in the case of an empty start or stop
157         * @return container backing objects
158         */
159        Iterable<? extends K> generateContainerObjects(T root);
160    }
161}
