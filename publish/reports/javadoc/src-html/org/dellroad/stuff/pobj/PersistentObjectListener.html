008package org.dellroad.stuff.pobj;
009
010/**
011 * Listener interface for changes to a {@link PersistentObject}.
012 *
013 * <p>
014 * Listener notifications are delivered in a separate thread from the one that caused the root object to change.
015 *
016 * @param <T> type of the root persistent object
017 */
018public interface PersistentObjectListener<T> {
019
020    /**
021     * Handle notification of an updated root object.
022     */
023    void handleEvent(PersistentObjectEvent<T> event);
024}
