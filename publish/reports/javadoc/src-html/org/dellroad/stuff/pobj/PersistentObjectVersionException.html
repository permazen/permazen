008package org.dellroad.stuff.pobj;
009
010/**
011 * Optimistic locking exception thrown by {@link PersistentObject#setRoot PersistentObject.setRoot()}
012 * when the expected version number does not agree.
013 */
014@SuppressWarnings("serial")
015public class PersistentObjectVersionException extends PersistentObjectException {
016
017    private final long actualVersion;
018    private final long expectedVersion;
019
020    public PersistentObjectVersionException(long actualVersion, long expectedVersion) {
021        super("expected version " + expectedVersion + " but actual version was " + actualVersion);
022        this.actualVersion = actualVersion;
023        this.expectedVersion = expectedVersion;
024    }
025
026    /**
027     * Get the actual, unexpected version number.
028     */
029    public long getActualVersion() {
030        return this.actualVersion;
031    }
032
033    /**
034     * Get the version number that was expected.
035     */
036    public long getExpectedVersion() {
037        return this.expectedVersion;
038    }
039}
