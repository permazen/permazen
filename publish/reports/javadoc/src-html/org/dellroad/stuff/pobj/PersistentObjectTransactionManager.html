029/**
030 * {@link PersistentObject} implementation of Spring's
031 * {@link org.springframework.transaction.PlatformTransactionManager PlatformTransactionManager} interface,
032 * allowing methods annotated with Spring's {@link org.springframework.transaction.annotation.Transactional &#64;Transactional}
033 * annotation to perform transactions on {@link PersistentObject}s.
034 *
035 * <p>
036 * During a transaction, the transaction's root object is available via the method {@link #getRoot}, and may be replaced entirely
037 * via {@link #setRoot setRoot()}. When the transaction completes, the transaction's root object will be automatically
038 * written back to the {@link PersistentObject} via {@link PersistentObject#setRoot(Object, long) PersistentObject.setRoot()}
039 * (unless the transaction was read-only). During commit, the {@link PersistentObject} version number is verified, and
040 * if another update has occurred since the transaction was opened, a {@link PersistentObjectVersionException} is thrown
041 * (consider using {@link org.dellroad.stuff.spring.RetryTransaction &#64;RetryTransaction} for automatic retry in this case).
042 * </p>
043 *
044 * <p>
045 * Transactions are associated with threads; the {@link PersistentObjectTransactionManager} associated with the current thread's
046 * transaction is available via {@link #getCurrent()} (or {@link #getCurrent(String)} if there is more than one).
047 * </p>
048 *
049 * <p>
050 * The {@code persistentObject} property is required. The {@code readOnlySharedRoot} property is optional and configures whether,
051 * during read-only transactions only, {@link #getRoot} returns the {@linkplain PersistentObject#getSharedRoot shared root} object.
052 * In this mode, root object graph copies are avoided entirely for read-only transactions, but the application code must be
053 * careful to not modify the object graph returned by {@link #getRoot} at any time, either during or after the transaction.
054 * </p>
055 *
056 * <p>
057 * This class also provides an {@link XAResource} for use with JTA/XA tranasactions. Once elisted in a JTA transaction,
058 * the usage described above using {@link #getRoot} and {@link #setRoot setRoot()} is identical.
059 * </p>
060 *
061 * @param <T> type of the root object
062 * @see PersistentObject
063 * @see PersistentObject#getSharedRoot
064 */
065@SuppressWarnings("serial")
066public class PersistentObjectTransactionManager<T> extends AbstractPlatformTransactionManager
067  implements BeanNameAware, ResourceTransactionManager, InitializingBean {
068

156    /**
157     * Get the instance associated with an open transaction in the current thread and having the given bean name.
158     *
159     * @param beanName bean name assigned to the desired instance
160     * @throws IllegalArgumentException if {@code name} is null
161     * @throws IllegalStateException if the current thread is not running within a transaction managed by
162     *  a {@link PersistentObjectTransactionManager} assigned the given name
163     */
164    @SuppressWarnings("unchecked")
165    public static <T> PersistentObjectTransactionManager<T> getCurrent(String beanName) {
166        if (beanName == null)
167            throw new IllegalArgumentException("null beanName");
168        final HashMap<String, PersistentObjectTransactionManager<?>> managerMap
169          = PersistentObjectTransactionManager.getManagerMap(false);
170        final PersistentObjectTransactionManager<?> manager = managerMap != null ? managerMap.get(beanName) : null;
171        if (manager == null) {
172            throw new IllegalStateException("no PersistentObjectTransactionManager named `" + beanName
173              + "' has an open transaction in the current thread");
174        }
175        if (manager.logger.isTraceEnabled())
176            manager.logger.trace("PersistentObjectTransactionManager.getCurrent(\"" + beanName + "\"): returning " + manager);
177        return (PersistentObjectTransactionManager<T>)manager;
178    }
179
180    /**
181     * Get the (unique) instance associated with an open transaction in the current thread.
182     *
183     * <p>
184     * This is a convenience method for the common case where there is only one instance associated with the current thread.
185     * </p>
186     *
187     * @throws IllegalStateException if the current thread is not running within a
188     *  {@link PersistentObjectTransactionManager} transaction
189     * @throws IllegalStateException there is more than one {@link PersistentObjectTransactionManager} transaction
190     *  open in the current thread
191     */
192    @SuppressWarnings("unchecked")
193    public static <T> PersistentObjectTransactionManager<T> getCurrent() {
194        final HashMap<String, PersistentObjectTransactionManager<?>> managerMap
195          = PersistentObjectTransactionManager.getManagerMap(false);
196        if (managerMap == null || managerMap.isEmpty()) {
197            throw new IllegalStateException("there are no PersistentObjectTransactionManager transactions open"
198              + " in the current thread");
199        }
200        if (managerMap.size() > 1) {
201            throw new IllegalStateException("there are multiple PersistentObjectTransactionManager transactions open"
202              + " in the current thread: " + managerMap.keySet() + "; invoke getCurrent() with explicitly specified bean name");
203        }
204        final PersistentObjectTransactionManager<T> manager
205          = (PersistentObjectTransactionManager<T>)managerMap.values().iterator().next();
206        if (manager.logger.isTraceEnabled())
207            manager.logger.trace("PersistentObjectTransactionManager.getCurrent(): returning " + manager);
208        return manager;
209    }
210
211    private static HashMap<String, PersistentObjectTransactionManager<?>> getManagerMap(boolean create) {
212        HashMap<String, PersistentObjectTransactionManager<?>> managerMap = MANAGER_MAP.get();
213        if (managerMap == null && create) {
214            managerMap = new HashMap<String, PersistentObjectTransactionManager<?>>();
215            MANAGER_MAP.set(managerMap);
216        }
217        return managerMap;
218    }

222    /**
223     * Determine whether the transaction associated with the current thread is a read-only transaction.
224     *
225     * @return true if the transaction associated with the current thread is a read-only transaction
226     * @throws IllegalStateException if there is no transaction associated with the current thread
227     */
228    public boolean isReadOnly() {
229        final TxInfo<T> info = this.currentInfo.get();
230        if (info == null)
231            throw new IllegalStateException("there is no transaction associated with the current thread");
232        return info.isReadOnly();
233    }

238     * @return snapshot version of the transaction associated with the current thread.
239     * @throws IllegalStateException if there is no transaction associated with the current thread
240     */
241    public long getSnapshotVersion() {
242        final TxInfo<T> info = this.currentInfo.get();
243        if (info == null)
244            throw new IllegalStateException("there is no transaction associated with the current thread");
245        return info.getSnapshot().getVersion();
246    }

248    /**
249     * Get the root object graph for use in the transaction associated with the current thread.
250     *
251     * @return root object
252     * @throws IllegalStateException if there is no open transaction
253     */
254    public T getRoot() {
255        final TxInfo<T> info = this.currentInfo.get();
256        if (info == null)
257            throw new IllegalStateException("there is no transaction associated with the current thread");
258        return info.getSnapshot().getRoot();
259    }

261    /**
262     * Change the root object graph to be committed in the transaction associated with the current thread.
263     * Subsequent calls to {@link #getRoot getRoot} will return the new object.
264     *
265     * <p>
266     * This method may be invoked during read-only transactions (as before, the root will not actually be committed).
267     * </p>
268     *
269     * @param root new root object
270     * @throws IllegalStateException if there is no transaction associated with the current thread
271     * @throws IllegalArgumentException if {@code root} is null
272     */
273    public void setRoot(T root) {
274        if (root == null)
275            throw new IllegalArgumentException("null root");
276        final TxInfo<T> info = this.currentInfo.get();
277        if (info == null)
278            throw new IllegalStateException("there is no transaction associated with the current thread");
279        info.setSnapshot(this.persistentObject.new Snapshot(root, info.getSnapshot().getVersion()));
280    }

305    @Override
306    @SuppressWarnings("unchecked")
307    protected void doBegin(Object txObj, TransactionDefinition txDef) {
308
309        // Logging
310        if (this.logger.isTraceEnabled())
311            this.logger.trace("PersistentObjectTransactionManager.doBegin(): txObj=" + txObj + " txDef=" + txDef);
312
313        // Sanity check
314        if (this.currentInfo.get() != null)
315            throw new TransactionUsageException("there is already a transaction associated with the current thread");
316        if (this.isExistingTransaction(txObj))
317            throw new TransactionUsageException("there is already a transaction associated with the given transaction object");
318        final TxWrapper<T> tx = (TxWrapper<T>)txObj;
319
320        // Create transaction
321        final PersistentObject<T>.Snapshot snapshot;
322        try {
323            snapshot = txDef.isReadOnly() && this.readOnlySharedRoot ?
324              this.persistentObject.getSharedRootSnapshot() : this.persistentObject.getRootSnapshot();
325        } catch (PersistentObjectException e) {
326            throw new CannotCreateTransactionException("error creating new JSimpleDB transaction", e);
327        }
328
329        // Associate transaction with this thread for getCurrent()
330        final HashMap<String, PersistentObjectTransactionManager<?>> managerMap
331          = PersistentObjectTransactionManager.getManagerMap(true);
332        if (managerMap.containsKey(this.beanName)) {
333            throw new IllegalStateException("A PersistentObjectTransactionManager named `" + this.beanName
334              + "' already has an open transaction in the current thread; all bean names must be distinct");
335        }
336        managerMap.put(this.beanName, this);
337
338        // Associate new transaction with the current thread
339        final TxInfo<T> info = new TxInfo<T>(snapshot, txDef.isReadOnly());
340        this.currentInfo.set(info);
341
342        // Set transaction info into the Spring transaction object
343        tx.setInfo(info);
344    }

346    /**
347     * Suspend the current transaction.
348     */
349    @Override
350    @SuppressWarnings("unchecked")
351    protected Object doSuspend(Object txObj) {
352
353        // Logging
354        if (this.logger.isTraceEnabled())
355            this.logger.trace("PersistentObjectTransactionManager.doSuspend(): txObj=" + txObj);
356
357        // Sanity check
358        final TxWrapper<T> tx = (TxWrapper<T>)txObj;
359        final TxInfo<T> info = tx.getInfo();
360        if (info == null)
361            throw new TransactionUsageException("no PersistentObject transaction exists in the given transaction object");
362        if (info != this.currentInfo.get())
363            throw new TransactionUsageException("the provided transaction object contains the wrong PersistentObject transaction");
364
365        // Suspend it
366        if (this.logger.isTraceEnabled())
367            this.logger.trace("PersistentObjectTransactionManager.doSuspend(): suspending " + info.getSnapshot());
368        this.currentInfo.remove();
369        tx.setInfo(null);
370        PersistentObjectTransactionManager.getManagerMap(true).remove(this.beanName);
371
372        // Done
373        return info;
374    }

376    /**
377     * Resume a previously suspended transaction.
378     */
379    @Override
380    @SuppressWarnings("unchecked")
381    protected void doResume(Object txObj, Object suspendedResources) {
382
383        // Logging
384        if (this.logger.isTraceEnabled())
385            this.logger.trace("PersistentObjectTransactionManager.doResume(): txObj=" + txObj + " res=" + suspendedResources);
386
387        // Sanity check
388        if (this.currentInfo.get() != null)
389            throw new TransactionUsageException("there is already a transaction associated with the current thread");
390        if (this.isExistingTransaction(txObj))
391            throw new TransactionUsageException("there is already a transaction associated with the given transaction object");
392
393        // Get previously saved info
394        final TxWrapper<T> tx = (TxWrapper<T>)txObj;
395        final TxInfo<T> info = (TxInfo<T>)suspendedResources;
396
397        // Resume transaction
398        if (this.logger.isTraceEnabled())
399            this.logger.trace("PersistentObjectTransactionManager.doSuspend(): resuming " + info.getSnapshot());
400        final HashMap<String, PersistentObjectTransactionManager<?>> managerMap
401          = PersistentObjectTransactionManager.getManagerMap(true);
402        if (managerMap.containsKey(this.beanName)) {
403            throw new IllegalStateException("A PersistentObjectTransactionManager named `" + this.beanName
404              + "' already has an open transaction in the current thread; all bean names must be distinct");
405        }
406        managerMap.put(this.beanName, this);
407        this.currentInfo.set(info);
408        tx.setInfo(info);
409    }

411    @Override
412    @SuppressWarnings("unchecked")
413    protected void prepareForCommit(DefaultTransactionStatus status) {
414        final TxInfo<T> info = ((TxWrapper<T>)status.getTransaction()).getInfo();
415        if (this.logger.isTraceEnabled())
416            this.logger.trace("PersistentObjectTransactionManager.prepareForCommit(): status=" + status + " info=" + info);
417        if (info == null)
418            throw new NoTransactionException("no current transaction exists");
419    }
420
421    @Override
422    @SuppressWarnings("unchecked")
423    protected void doCommit(DefaultTransactionStatus status) {
424
425        // Logging
426        if (this.logger.isTraceEnabled())
427            this.logger.trace("PersistentObjectTransactionManager.doCommit(): status=" + status);
428
429        // Get transaction
430        final TxWrapper<T> tx = (TxWrapper<T>)status.getTransaction();
431        final TxInfo<T> info = tx.getInfo();
432        if (info == null)
433            throw new NoTransactionException("no current transaction exists");
434        final PersistentObject<T>.Snapshot snapshot = info.getSnapshot();
435
436        // Commit
437        try {
438            if (!info.isReadOnly()) {
439                if (this.logger.isTraceEnabled())
440                    this.logger.trace("PersistentObjectTransactionManager.doCommit(): committing " + info);
441                this.persistentObject.setRoot(snapshot.getRoot(), snapshot.getVersion());
442            } else
443                this.logger.trace("PersistentObjectTransactionManager.doCommit(): not committing read-only " + info);
444        } catch (PersistentObjectVersionException e) {
445            throw new OptimisticLockingFailureException(null, e);
446        } catch (PersistentObjectValidationException e) {
447            throw new DataIntegrityViolationException(null, e);
448        } catch (PersistentObjectException e) {
449            throw new TransactionSystemException("error committing transaction", e);
450        }
451    }

453    @Override

455    protected void doRollback(DefaultTransactionStatus status) {
456
457        // Logging
458        if (this.logger.isTraceEnabled())
459            this.logger.trace("PersistentObjectTransactionManager.doRollback(): status=" + status);
460
461        // Get transaction
462        final TxWrapper<T> tx = (TxWrapper<T>)status.getTransaction();
463        final TxInfo<T> info = tx.getInfo();
464        if (info == null)
465            throw new NoTransactionException("no current transaction exists");
466        final PersistentObject<T>.Snapshot snapshot = info.getSnapshot();
467
468        // Rollback
469        if (this.logger.isTraceEnabled())
470            this.logger.trace("PersistentObjectTransactionManager.doRollback(): rolling back " + info);
471        // no action required to rollback a PersistentObject transaction
472    }

494    @Override
495    @SuppressWarnings("unchecked")
496    protected void doCleanupAfterCompletion(Object txObj) {
497
498        // Logging
499        if (this.logger.isTraceEnabled())
500            this.logger.trace("PersistentObjectTransactionManager.doCleanupAfterCompletion(): txObj=" + txObj);
501
502        // Get transaction
503        final TxWrapper<T> tx = (TxWrapper<T>)txObj;
504
505        // Clean up
506        tx.setInfo(null);
507        this.currentInfo.remove();
508        final HashMap<String, PersistentObjectTransactionManager<?>> managerMap
509          = PersistentObjectTransactionManager.getManagerMap(false);
510        if (managerMap != null) {
511            managerMap.remove(this.beanName);
512            if (managerMap.isEmpty())
513                MANAGER_MAP.remove();
514        }
515    }
