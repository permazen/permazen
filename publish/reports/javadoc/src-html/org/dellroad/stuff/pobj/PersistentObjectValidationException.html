008package org.dellroad.stuff.pobj;
009
010import java.util.Set;
011
012import javax.validation.ConstraintViolation;
013
014/**
015 * Runtime exception thrown during {@link PersistentObject} operations.
016 */
017@SuppressWarnings("serial")
018public class PersistentObjectValidationException extends PersistentObjectException {
019
020    private final Set<ConstraintViolation<?>> violations;
021
022    /**
023     * Constructor.
024     *
025     * @param violations set of violations
026     * @throws IllegalArgumentException if {@code violations} is null
027     */
028    @SuppressWarnings("unchecked")
029    public <T> PersistentObjectValidationException(Set<ConstraintViolation<T>> violations) {
030        super(PersistentObjectValidationException.generateMessage(violations));
031        this.violations = (Set<ConstraintViolation<?>>)(Object)violations;
032    }
033
034    /**
035     * Get the set of constraint violations.
036     */
037    public Set<ConstraintViolation<?>> getViolations() {
038        return this.violations;
039    }
040
041    private static <T> String generateMessage(Set<ConstraintViolation<T>> violations) {
042        if (violations == null)
043            throw new IllegalArgumentException("null violations");
044        StringBuilder buf = new StringBuilder("object failed to validate with " + violations.size() + " violation(s):\n");
045        for (ConstraintViolation<T> violation : violations) {
046            buf.append("    [")
047              .append(violation.getPropertyPath())
048              .append("]: ")
049              .append(violation.getMessage())
050              .append('\n');
051        }
052        return buf.toString();
053    }
054}
