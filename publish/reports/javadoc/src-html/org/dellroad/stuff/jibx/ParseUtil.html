103    /**
104     * Deserialize an object by reference.
105     *
106     * <p>
107     * Invoke this method from your own custom deserializer to produce an result of the correct type.
108     *
109     * <p>
110     * The object must have been unmarshalled already and had its ID registered via {@link IdMapper#setId}.
111     *
112     * @throws IllegalArgumentException if {@code string} cannot be parsed
113     * @throws JiBXParseException if the reference has not been registered to any object yet (e.g., forward reference)
114     * @throws JiBXParseException if the referenced object is not an instance of {@code type}
115     * @see #serializeReference
116     * @see IdMapper
117     */
118    public static <T> T deserializeReference(String string, Class<T> type) throws JiBXParseException {
119        if (string == null)
120            return null;
121        long id;
122        try {
123            id = IdMapper.parseId(string);
124        } catch (IllegalArgumentException e) {
125            throw new JiBXParseException("invalid object reference", string, e);
126        }
127        Object obj = IdGenerator.get().getObject(id);
128        if (obj == null)
129            throw new JiBXParseException("unregistered object reference `" + string + "'; possible forward reference", string);
130        try {
131            return type.cast(obj);
132        } catch (ClassCastException e) {
133            throw new JiBXParseException("object reference `" + string + "' is assigned to an instance of "
134              + obj.getClass() + " which is not assignable to " + type, string);
135        }
136    }

139     * Serialize an object by reference.
140     *
141     * <p>
142     * The object must have been marshalled already and had its ID assigned via {@link IdMapper#getId}.
143     *
144     * @throws IllegalArgumentException if the object has not been registered yet (e.g., forward reference)
145     * @see #deserializeReference
146     * @see IdMapper
147     */
148    public static String serializeReference(Object obj) {
149        if (obj == null)
150            return null;
151        long id = IdGenerator.get().checkId(obj);
152        if (id == 0)
153            throw new IllegalArgumentException("unregistered object; possible forward reference to " + obj);
154        return IdMapper.formatId(id);
155    }

171    /**
172     * Derialize a millisecond-granularity time interval, e.g., "30s", "1d12h", "0.250s", etc.
173     *
174     * @see #serializeTimeInterval
175     */
176    public static long deserializeTimeInterval(String string) throws JiBXParseException {
177
178        // Apply regular expression
179        Matcher m = Pattern.compile(TIME_INTERVAL_PATTERN).matcher(string);
180        if (string.length() == 0 || !m.matches())
181            throw new JiBXParseException("invalid time interval", string);
182
183        // Parse regex groups
184        Object[][] groups = new Object[][] {
185            { 2, 24 * 60 * 60 * 1000 },
186            { 4,      60 * 60 * 1000 },
187            { 6,           60 * 1000 },
188            { 8,                1000 },
189        };
190        long value = 0;
191        for (int i = 0; i < groups.length; i++) {
192            String group = m.group((Integer)groups[i][0]);
193            if (group == null || group.length() == 0)
194                continue;
195            double num;
196            try {
197                num = Double.parseDouble(group);
198            } catch (NumberFormatException e) {
199                throw new JiBXParseException("invalid time interval", string);
200            }
201            value += Math.round(num * (Integer)groups[i][1]);
202            if (value < 0)
203                throw new JiBXParseException("time interval is too large", string);
204        }
205
206        // Done
207        return value;
208    }

210    /**
211     * Serialize a millisecond-granularity time interval, e.g., "30s", "1d12h", "0.250s", etc.
212     *
213     * @see #deserializeTimeInterval
214     */
215    public static String serializeTimeInterval(long value) {
216        if (value < 0)
217            throw new IllegalArgumentException("negative value");
218        StringBuilder b = new StringBuilder(32);
219        long days = value / (24 * 60 * 60 * 1000);
220        value = value % (24 * 60 * 60 * 1000);
221        if (days > 0)
222            b.append(days).append('d');
223        long hours = value / (60 * 60 * 1000);
224        value = value % (60 * 60 * 1000);
225        if (hours > 0)
226            b.append(hours).append('h');
227        long minutes = value / (60 * 1000);
228        value = value % (60 * 1000);
229        if (minutes > 0)
230            b.append(minutes).append('m');
231        long millis = value;
232        if (millis != 0 || b.length() == 0) {
233            if (millis >= 1000 && millis % 1000 == 0)
234                b.append(String.format("%ds", millis / 1000));
235            else
236                b.append(String.format("%.3fs", (double)millis / 1000.0));
237        }
238        return b.toString();
239    }

263    /**
264     * Deserialize a byte array using MAC address notation (colon-separated). Each byte must have two digits.
265     *
266     * @see #serializeByteArrayWithColons
267     */
268    public static byte[] deserializeByteArrayWithColons(String string) throws JiBXParseException {
269        if (string.length() == 0)
270            return new byte[0];
271        if (string.length() % 3 != 2)
272            throw new JiBXParseException("invalid byte array", string);
273        char[] nocolons = new char[((string.length() + 1) / 3) * 2];
274        int j = 0;
275        for (int i = 0; i < string.length(); i++) {
276            if (i % 3 == 2) {
277                if (string.charAt(i) != ':')
278                    throw new JiBXParseException("invalid byte array", string);
279                continue;
280            }
281            nocolons[j++] = string.charAt(i);
282        }
283        try {
284            return ByteArrayEncoder.decode(new String(nocolons));
285        } catch (IllegalArgumentException e) {
286            throw new JiBXParseException("invalid byte array", string, e);
287        }
288    }

290    /**
291     * Serialize a byte array using MAC address notation (colon-separated). Each byte will have two digits.
292     *
293     * @see #deserializeByteArrayWithColons
294     */
295    public static String serializeByteArrayWithColons(byte[] array) {
296        char[] colons = new char[array.length * 3 - 1];
297        String nocolons = ByteArrayEncoder.encode(array);
298        int j = 0;
299        for (int i = 0; i < nocolons.length(); i += 2) {
300            colons[j++] = nocolons.charAt(i);
301            colons[j++] = nocolons.charAt(i + 1);
302            if (j < colons.length)
303                colons[j++] = ':';
304        }
305        return new String(colons);
306    }

450     * JiBX {@link String} deserializer support method that verifies that the input string matches the
451     * given regular expression. This method can be invoked by custom deserializers that supply the
452     * regular expression to it.
453     *
454     * @throws NullPointerException if {@code string} of {@code regex} is null

460        if (!string.matches(regex))
461            throw new JiBXParseException("input does not match pattern \"" + regex + "\"", string);
462        return string;
463    }

465    /**
466     * Boolean parser that allows "yes" and "no" as well as the usual "true", "false", "0", "1".
467     * Comparisons are case-insensitive.
468     *
469     * @throws JiBXParseException if the value is not recognizable as a boolean
470     *
471     * @see #deserializeBooleanStrictly
472     */
473    public static boolean deserializeBoolean(String string) throws JiBXParseException {
474        for (String s : BOOLEAN_TRUES) {
475            if (string.equalsIgnoreCase(s))
476                return true;
477        }
478        for (String s : BOOLEAN_FALSES) {
479            if (string.equalsIgnoreCase(s))
480                return false;
481        }
482        throw new JiBXParseException("invalid Boolean value", string);
483    }

485    /**
486     * Deserialize a boolean strictly, only allowing the values {@code true} or {@code false}.
487     *
488     * @see #deserializeBoolean
489     */
490    public static boolean deserializeBooleanStrictly(String string) throws JiBXParseException {
491        if ("true".equals(string))
492            return true;
493        if ("false".equals(string))
494            return false;
495        throw new JiBXParseException("invalid boolean value; must be `true' or `false'", string);
496    }

498    /**
499     * Deserialize an array of integers separated by commas and/or whitespace.
500     *
501     * @see #serializeIntArray
502     */
503    public static int[] deserializeIntArray(String string) throws JiBXParseException {
504        String[] ints = string.trim().split("(\\s*,\\s*|\\s+)");
505        if (ints.length == 0 || ints[0].length() == 0)
506            return new int[0];
507        int[] array = new int[ints.length];
508        try {
509            for (int i = 0; i < array.length; i++)
510                array[i] = Integer.parseInt(ints[i]);
511        } catch (NumberFormatException e) {
512            throw new JiBXParseException("invalid integer list", string, e);
513        }
514        return array;
515    }

517    /**
518     * Serialize an array of integers. Example: "1, 2, 3".
519     *
520     * @see #deserializeIntArray
521     */
522    public static String serializeIntArray(int[] array) throws JiBXParseException {
523        StringBuilder buf = new StringBuilder(array.length * 2);
524        for (int i = 0; i < array.length; i++) {
525            if (i > 0)
526                buf.append(", ");
527            buf.append(array[i]);
528        }
529        return buf.toString();
530    }
