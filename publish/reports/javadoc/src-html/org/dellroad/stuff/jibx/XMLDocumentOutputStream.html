008package org.dellroad.stuff.jibx;
009
010import java.io.BufferedOutputStream;
011import java.io.IOException;
012import java.io.OutputStream;
013
014import org.dellroad.stuff.io.OutputStreamWriter;
015import org.jibx.runtime.JiBXException;
016
017/**
018 * {@link OutputStream} over which XML documents are passed. This class is a companion to {@link XMLDocumentInputStream}.
019 *
020 * <p>
021 * XML documents are created from Java objects via {@link JiBXUtil#writeObject(Object, OutputStream) JiBXUtil.writeObject()}.
022 * </p>
023 *
024 * <p>
025 * Instances of this class are thread-safe.
026 * </p>
027 *
028 * @param <T> XML document type
029 * @see XMLDocumentInputStream
030 */
031public class XMLDocumentOutputStream<T> {
032
033    private final OutputStreamWriter output;
034
035    /**
036     * Constructor.
037     *
038     * @param output data destination
039     */
040    public XMLDocumentOutputStream(OutputStream output) {
041        if (output == null)
042            throw new IllegalArgumentException("null output");
043        this.output = new OutputStreamWriter(new BufferedOutputStream(output));
044    }
045
046    /**
047     * Write the object encoded as XML to the underlying output stream.
048     * The underlying output stream is flushed.
049     */
050    public synchronized void write(T obj) throws IOException, JiBXException {
051        this.output.start();
052        JiBXUtil.writeObject(obj, this.output);
053        this.output.stop();
054    }
055
056    /**
057     * Close the underlying output stream.
058     */
059    public void close() throws IOException {
060        this.output.close();
061    }
062}
