008package org.dellroad.stuff.validation;
009
010import java.lang.annotation.Documented;
011import java.lang.annotation.ElementType;
012import java.lang.annotation.Retention;
013import java.lang.annotation.RetentionPolicy;
014import java.lang.annotation.Target;
015
016import javax.validation.Constraint;
017import javax.validation.Payload;
018import javax.validation.ReportAsSingleViolation;
019import javax.validation.constraints.Pattern;
020
021/**
022 * Validation constraint requiring a {@link String} to be a valid DNS hostname (or hostname component).
023 */
024@Documented
025@Constraint(validatedBy = {})
026@Target({ ElementType.METHOD, ElementType.FIELD, ElementType.ANNOTATION_TYPE })
027@Retention(RetentionPolicy.RUNTIME)
028@Pattern(regexp = "^(([\\p{Alnum}]([-\\p{Alnum}]*[\\p{Alnum}])?)\\.)*([\\p{Alpha}]([-\\p{Alnum}]*[\\p{Alnum}])?)$")
029@ReportAsSingleViolation
030public @interface Hostname {
031
032    String message() default "Invalid hostname or hostname component";
033
034    Class<?>[] groups() default {};
035
036    Class<? extends Payload>[] payload() default {};
037}
