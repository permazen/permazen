008package org.dellroad.stuff.validation;
009
010import java.util.Set;
011
012import javax.validation.ConstraintViolation;
013
014/**
015 * Validation utility methods.
016 */
017public final class ValidationUtil {
018
019    private ValidationUtil() {
020    }
021
022    /**
023     * Validate the given object.
024     *
025     * <p>
026     * This method simply creates a {@link ValidationContext} with the given root and invokes {@link ValidationContext#validate()}.
027     * </p>
028     */
029    public static <T> Set<ConstraintViolation<T>> validate(T obj) {
030        return new ValidationContext<T>(obj).validate();
031    }
032
033    /**
034     * Describe the validation errors in a friendly format.
035     *
036     * @param violations validation violations
037     * @return description of the validation errors
038     * @throws NullPointerException if {@code violations} is null
039     */
040    public static String describe(Set<? extends ConstraintViolation<?>> violations) {
041        if (violations.isEmpty())
042            return "  (no violations)";
043        StringBuilder buf = new StringBuilder(violations.size() * 32);
044        for (ConstraintViolation<?> violation : violations) {
045            String violationPath = violation.getPropertyPath().toString();
046            buf.append("  ");
047            if (violationPath.length() > 0)
048                buf.append("[").append(violationPath).append("]: ");
049            buf.append(violation.getMessage()).append('\n');
050        }
051        return buf.toString();
052    }
053}
