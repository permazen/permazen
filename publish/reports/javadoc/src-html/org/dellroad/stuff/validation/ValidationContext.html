008    package org.dellroad.stuff.validation;
009    
010    import java.util.Set;
011    import java.util.concurrent.Callable;
012    
013    import javax.validation.ConstraintViolation;
014    import javax.validation.Validation;
015    import javax.validation.Validator;
016    
017    import org.dellroad.stuff.java.ThreadLocalHolder;
018    
019    /**
020     * Provides additional context for {@link javax.validation.ConstraintValidator} implementations.
021     *
022     * <p>
023     * {@link ValidationContext} gives {@link javax.validation.ConstraintValidator} implementations access to the root object
024     * being validated. This breaks the usual principle of locality for validation (i.e., that validation of a specific bean
025     * proceeds unaware of that bean's parents) but it can make custom validators more convenient to implement.
026     * Subclasses are encouraged to provide additional application-specific information.
027     *
028     * <p>
029     * Validation must be performed via {@link #validate(Validator) validate()} for this class to work.
030     */
031    public class ValidationContext<T> {
032    
033        private static final ThreadLocalHolder<ValidationContext> CURRENT = new ThreadLocalHolder<ValidationContext>();
034    
035        private final T root;
036    
037        /**
038         * Construct a new validation context configured to validate the given root object.
039         *
040         * @param root root object to be validated
041         * @throws IllegalArgumentException if {@code root} is null
042         */
043        public ValidationContext(T root) {
044            if (root == null)
045                throw new IllegalArgumentException("null root");
046            this.root = root;
047        }
048    
049        /**
050         * Get the root object associated with this instance.
051         */
052        public final T getRoot() {
053            return this.root;
054        }
055    
056        /**
057         * Validate this instance's root object. This is a convenience method, equivalent to:
058         *  <blockquote>
059         *  <code>{@link #validate(Validator) validate}(Validation.buildDefaultValidatorFactory().getValidator())</code>
060         *  <blockquote>
061         *
062         * @throws IllegalStateException if this method is invoked re-entrantly
063         */
064        public Set<ConstraintViolation<T>> validate() {
065            return this.validate(Validation.buildDefaultValidatorFactory().getValidator());
066        }
067    
068        /**
069         * Validate this instance's root object using the given {@link Validator}, making the context
070         * available to the current thread during the validation process via {@link #getCurrentContext}.
071         *
072         * @throws IllegalStateException if this method is invoked re-entrantly
073         */
074        public Set<ConstraintViolation<T>> validate(final Validator validator) {
075    
076            // Sanity check
077            if (ValidationContext.CURRENT.get() != null)
078                throw new IllegalStateException("re-entrant invocation is not allowed");
079    
080            // Validate
081            try {
082                return ValidationContext.CURRENT.invoke(this, new Callable<Set<ConstraintViolation<T>>>() {
083                    @Override
084                    public Set<ConstraintViolation<T>> call() {
085                        return validator.validate(ValidationContext.this.root);
086                    }
087                });
088            } catch (RuntimeException e) {
089                throw e;
090            } catch (Exception e) {
091                throw new RuntimeException(e);
092            }
093        }
094    
095        /**
096         * Get the {@link ValidationContext} associated with the current thread, cast to the desired type.
097         * This method is only valid during invocations of {@link #validate(Validator) validate()}.
098         *
099         * @param type required type
100         * @return current {@link ValidationContext}
101         * @throws IllegalStateException if {@link #validate(Validator) validate()} is not currently executing
102         * @throws ClassCastException if the current {@link ValidationContext} is not of type {@code type}
103         */
104        public static <T extends ValidationContext<?>> T getCurrentContext(Class<T> type) {
105            ValidationContext<?> context = ValidationContext.CURRENT.get();
106            if (context == null)
107                throw new IllegalStateException("current thread is not executing validate()");
108            return type.cast(context);
109        }
110    
111        /**
112         * Convenience method to get the root object being validated by the current thread.
113         * This method is only valid during invocations of {@link #validate(Validator) validate()}.
114         * Subclasses may want to override to narrow the return type.
115         *
116         * @return current validation root object
117         * @throws IllegalStateException if {@link #validate(Validator) validate()} is not currently executing
118         */
119        public static Object getCurrentRoot() {
120            return ValidationContext.getCurrentContext(ValidationContext.class).getRoot();
121        }
122    }
