008package org.dellroad.stuff.spring;
009
010import java.sql.Connection;
011import java.sql.SQLException;
012
013import javax.persistence.EntityManager;
014
015import org.slf4j.Logger;
016import org.slf4j.LoggerFactory;
017import org.springframework.dao.DataAccessException;
018import org.springframework.dao.UncategorizedDataAccessException;
019import org.springframework.jdbc.datasource.DataSourceUtils;
020import org.springframework.orm.jpa.vendor.HibernateJpaDialect;
021import org.springframework.transaction.TransactionDefinition;
022
023/**
024 * Hibernate JPA adapter that adds supports for setting a non-default transaction isolation level
025 * and adds a workaround for <a href="https://jira.springsource.org/browse/SPR-10815">SPR-10815</a>.
026 *
027 * <p>
028 * The JPA specification, which is database technology agnostic, does not define a way to set SQL database isolation levels.
029 * This fixes the "Standard JPA does not support custom isolation levels - use a special JpaDialect" Spring exception
030 * that will otherwise occur.
031 *
032 * @see <a href="https://jira.springsource.org/browse/SPR-3812">SPR-3812</a>
033 * @see <a href="http://stackoverflow.com/questions/5234240/hibernatespringjpaisolation-does-not-work">Question on stackoverflow.com</a>
034 * @see <a href="https://jira.springsource.org/browse/SPR-10815">SPR-10815</a>
035 */
036@SuppressWarnings("serial")
037public class HibernateIsolationJpaDialect extends HibernateJpaDialect {
038
039    private final Logger log = LoggerFactory.getLogger(this.getClass());
040
041    @Override
042    public TransactionDataWrapper beginTransaction(EntityManager entityManager, TransactionDefinition definition)
043      throws SQLException {
044
045        // Set transaction timeout
046        if (definition.getTimeout() != TransactionDefinition.TIMEOUT_DEFAULT)
047            this.getSession(entityManager).getTransaction().setTimeout(definition.getTimeout());
048
049        // Set isolation level on the associated JDBC connection
050        Integer oldIsolation = null;
051        Connection connection = null;
052        if (definition.getIsolationLevel() != TransactionDefinition.ISOLATION_DEFAULT) {
053            connection = this.getJdbcConnection(entityManager, definition.isReadOnly()).getConnection();
054            if (this.log.isTraceEnabled())
055                this.log.trace("setting isolation level to " + definition.getIsolationLevel() + " on " + connection);
056            oldIsolation = DataSourceUtils.prepareConnectionForTransaction(connection, definition);
057        }
058
059        // Start transaction
060        entityManager.getTransaction().begin();
061
062        // Prepare transaction
063        return this.prepareTransaction(entityManager, definition, connection, oldIsolation);
064    }
065
066    /**
067     * Prepare transaction.
068     */
069    protected TransactionDataWrapper prepareTransaction(EntityManager entityManager, TransactionDefinition definition,
070      Connection connection, Integer oldIsolation) {
071
072        // Do superclass preparation
073        Object transactionData = this.prepareTransaction(entityManager, definition.isReadOnly(), definition.getName());
074
075        // Wrap result
076        return new TransactionDataWrapper(transactionData, connection, oldIsolation);
077    }
078
079    @Override
080    public void cleanupTransaction(Object obj) {
081
082        // Let superclass cleanup
083        final TransactionDataWrapper wrapper = (TransactionDataWrapper)obj;
084        super.cleanupTransaction(wrapper.getTransactionData());
085
086        // Restore isolation level on the JDBC connection
087        Connection connection = wrapper.getConnection();
088        if (connection != null) {
089            if (this.log.isTraceEnabled())
090                this.log.trace("restoring isolation level to " + wrapper.getIsolation() + " on " + connection);
091            DataSourceUtils.resetConnectionAfterTransaction(connection, wrapper.getIsolation());
092        }
093    }
094
095    /**
096     * Adds fix for SPR-10815.
097     *
098     * @see <a href="https://jira.springsource.org/browse/SPR-10815">SPR-10815</a>
099     */
100    @Override
101    public DataAccessException translateExceptionIfPossible(RuntimeException ex) {
102
103        // Handle case that Spring has already wrapped and hidden the real exception without recognizing it
104        for (Throwable t = ex; t != null; t = t.getCause()) {
105            if (t instanceof RuntimeException) {
106                final DataAccessException e = super.translateExceptionIfPossible((RuntimeException)t);
107                if (e != null && !(e instanceof UncategorizedDataAccessException))
108                    return e;
109            }
110        }
111
112        // Give up
113        return super.translateExceptionIfPossible(ex);
114    }
115
116    /**
117     * Wraps superclass transaction data and adds isolation level to restore.
118     */
119    protected class TransactionDataWrapper {
120
121        private final Object transactionData;
122        private final Connection connection;
123        private final Integer isolation;
124
125        public TransactionDataWrapper(Object transactionData, Connection connection, Integer isolation) {
126            this.transactionData = transactionData;
127            this.connection = connection;
128            this.isolation = isolation;
129        }
130
131        public Object getTransactionData() {
132            return this.transactionData;
133        }
134
135        public Connection getConnection() {
136            return this.connection;
137        }
138
139        public Integer getIsolation() {
140            return this.isolation;
141        }
142    }
143}
