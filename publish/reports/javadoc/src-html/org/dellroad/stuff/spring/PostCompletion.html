008    package org.dellroad.stuff.spring;
009    
010    import java.util.concurrent.Callable;
011    
012    /**
013     * Supports requesting the execution of post-completion callbacks when running within
014     * {@link PostCompletionSupport @PostCompletionSupport}-annotated methods.
015     *
016     * <p>
017     * See {@link PostCompletionSupport @PostCompletionSupport} for an example of how this class is normally used.
018     *
019     * @see PostCompletionSupport
020     */
021    public final class PostCompletion {
022    
023        private static final ThreadLocal<PostCompletionRegistry> CURRENT = new ThreadLocal<PostCompletionRegistry>();
024    
025        private PostCompletion() {
026        }
027    
028        /**
029         * Register a callback to be executed upon successful completion only.
030         *
031         * <p>
032         * Equivalent to:
033         *  <blockquote><pre>
034         *  {@link #execute(Runnable, boolean) execute}(action, false);
035         *  </pre></blockquote>
036         *
037         * @param action action to invoke
038         * @throws IllegalArgumentException if {@code action} is null
039         */
040        public static void execute(Runnable action) {
041            PostCompletion.execute(action, false);
042        }
043    
044        /**
045         * Register a post-completion callback.
046         *
047         * @param action action to invoke
048         * @param always true to execute the action even upon unsuccessful completion (i.e., exception thrown)
049         * @throws IllegalArgumentException if {@code action} is null
050         * @throws IllegalStateException if not running within a
051         *  {@link PostCompletionSupport @PostCompletionSupport}-annotated method.
052         */
053        public static void execute(Runnable action, boolean always) {
054            PostCompletion.get().add(action, always);
055        }
056    
057        /**
058         * Register a callback to be executed upon successful completion only.
059         *
060         * <p>
061         * Equivalent to:
062         *  <blockquote><pre>
063         *  {@link #execute(Callable, boolean) execute}(action, false);
064         *  </pre></blockquote>
065         *
066         * @param action action to invoke
067         * @throws IllegalArgumentException if {@code action} is null
068         */
069        public static void execute(Callable<?> action) {
070            PostCompletion.execute(action, false);
071        }
072    
073        /**
074         * Register a post-completion callback.
075         *
076         * @param action action to invoke
077         * @param always true to execute the action even upon unsuccessful completion (i.e., exception thrown)
078         * @throws IllegalArgumentException if {@code action} is null
079         * @throws IllegalStateException if not running within a
080         *  {@link PostCompletionSupport @PostCompletionSupport}-annotated method.
081         */
082        public static void execute(Callable<?> action, boolean always) {
083            PostCompletion.get().add(action, always);
084        }
085    
086        static PostCompletionRegistry get() {
087            PostCompletionRegistry current = CURRENT.get();
088            if (current == null) {
089                 throw new IllegalStateException("no PostCompletionRegistry instance is associated with the current thread;"
090                   + " are we executing within a @PostCompletionSupport-annotated method?");
091            }
092            return current;
093        }
094    
095        static void push() {
096            PostCompletionRegistry current = CURRENT.get();
097            if (current == null) {
098                current = new PostCompletionRegistry();
099                CURRENT.set(current);
100            }
101            current.ref();
102        }
103    
104        static boolean pop() {
105            PostCompletionRegistry current = PostCompletion.get();
106            if (!current.unref()) {
107                CURRENT.remove();
108                return false;
109            }
110            return true;
111        }
112    }
