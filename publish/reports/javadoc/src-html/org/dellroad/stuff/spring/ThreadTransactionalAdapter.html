008package org.dellroad.stuff.spring;
009
010import java.lang.annotation.Annotation;
011
012import org.springframework.transaction.annotation.Isolation;
013import org.springframework.transaction.annotation.Propagation;
014import org.springframework.transaction.annotation.Transactional;
015
016/**
017 * Adapter that converts a {@link ThreadTransactional @ThreadTransactional} annotation into a
018 * {@link Transactional @Transactional} one.
019 *
020 * <p>
021 * Used internally by {@link ThreadTransactionalAnnotationParser}.
022 *
023 * @see ThreadTransactional
024 * @see Transactional
025 */
026public class ThreadTransactionalAdapter implements Annotation, Transactional {
027
028    private final ThreadTransactional threadTransactional;
029
030    public ThreadTransactionalAdapter(ThreadTransactional threadTransactional) {
031        if (threadTransactional == null)
032            throw new IllegalArgumentException("null threadTransactional");
033        this.threadTransactional = threadTransactional;
034    }
035
036    @Override
037    public Class<? extends Annotation> annotationType() {
038        return Transactional.class;
039    }
040
041    @Override
042    public String value() {
043        return this.threadTransactional.value();
044    }
045
046    @Override
047    public Propagation propagation() {
048        return this.threadTransactional.propagation();
049    }
050
051    @Override
052    public Isolation isolation() {
053        return this.threadTransactional.isolation();
054    }
055
056    @Override
057    public int timeout() {
058        return this.threadTransactional.timeout();
059    }
060
061    @Override
062    public boolean readOnly() {
063        return this.threadTransactional.readOnly();
064    }
065
066    @Override
067    public Class<? extends Throwable>[] rollbackFor() {
068        return this.threadTransactional.rollbackFor();
069    }
070
071    @Override
072    public String[] rollbackForClassName() {
073        return this.threadTransactional.rollbackForClassName();
074    }
075
076    @Override
077    public Class<? extends Throwable>[] noRollbackFor() {
078        return this.threadTransactional.noRollbackFor();
079    }
080
081    @Override
082    public String[] noRollbackForClassName() {
083        return this.threadTransactional.noRollbackForClassName();
084    }
085
086    @Override
087    public boolean equals(Object obj) {
088        if (obj == null || obj.getClass() == this.getClass())
089            return false;
090        ThreadTransactionalAdapter that = (ThreadTransactionalAdapter)obj;
091        return this.threadTransactional.equals(that.threadTransactional);
092    }
093
094    @Override
095    public int hashCode() {
096        return this.threadTransactional.hashCode();
097    }
098}
