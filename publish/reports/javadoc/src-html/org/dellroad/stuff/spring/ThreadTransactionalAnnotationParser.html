008    package org.dellroad.stuff.spring;
009    
010    import java.lang.reflect.AnnotatedElement;
011    
012    import org.springframework.core.annotation.AnnotationUtils;
013    import org.springframework.transaction.annotation.SpringTransactionAnnotationParser;
014    import org.springframework.transaction.interceptor.TransactionAttribute;
015    
016    /**
017     * Strategy implementation for parsing the {@link ThreadTransactional @ThreadTransactional} annotation.
018     * Works just like {@link SpringTransactionAnnotationParser} but parses {@link ThreadTransactional @ThreadTransactional}
019     * annotations instead.
020     *
021     * @see ThreadTransactional
022     */
023    @SuppressWarnings("serial")
024    public class ThreadTransactionalAnnotationParser extends SpringTransactionAnnotationParser {
025    
026        @Override
027        public TransactionAttribute parseTransactionAnnotation(AnnotatedElement annotatedElement) {
028            ThreadTransactional threadTransactional = AnnotationUtils.getAnnotation(annotatedElement, ThreadTransactional.class);
029            if (threadTransactional == null)
030                return null;
031            return this.parseTransactionAnnotation(new ThreadTransactionalAdapter(threadTransactional));
032        }
033    }
