008package org.dellroad.stuff.java;
009
010import java.beans.Introspector;
011import java.lang.annotation.Annotation;
012import java.lang.reflect.InvocationTargetException;
013import java.lang.reflect.Method;
014import java.lang.reflect.Modifier;
015import java.util.Arrays;
016import java.util.HashSet;
017import java.util.Iterator;
018import java.util.LinkedHashMap;
019import java.util.Set;
020
021/**
022 * Scan a class hierarchy for annotated methods in an override-aware manner.
023 *
024 * <p>
025 * Instances find all methods annotated with a specific annotation in a given Java class or any of its superclasses
026 * and superinterfaces, while also being override-aware, i.e., filtering out annotations on overridden supertype methods
027 * when the overriding method also has the annotation. This operation is performed by {@link #findAnnotatedMethods}.
028 * </p>
029 *
030 * <p>
031 * Subclasses may validate the annotations, and control which annotated methods to include and/or reject, by overriding
032 * {@link #includeMethod includeMethod()}.
033 * </p>
034 *
035 * @param <T> Java type to be introspected
036 * @param <A> Java annotation type
037 */
038public class MethodAnnotationScanner<T, A extends Annotation> {
039
040    protected final Class<T> type;
041    protected final Class<A> annotationType;
042
043    /**
044     * Constructor.
045     *
046     * @param type Java class to be introspected
047     * @param annotationType Java annotation type to search for
048     * @throws IllegalArgumentException if either parameter is null
049     */
050    public MethodAnnotationScanner(Class<T> type, Class<A> annotationType) {
051
052        // Sanity check
053        if (type == null)
054            throw new IllegalArgumentException("null type");
055        if (annotationType == null)
056            throw new IllegalArgumentException("null annotationType");
057        this.type = type;
058        this.annotationType = annotationType;
059    }
060
061    /**
062     * Build set of annotated methods, with overridden annotated methods omitted.
063     *
064     * @return the set of all annotated methods, with overridden annotated methods removed
065     *  when the overriding method is also annotated
066     */
067    public Set<MethodInfo> findAnnotatedMethods() {
068
069        // Find all methods
070        final HashSet<MethodInfo> set = new HashSet<MethodInfo>();
071        this.findMethodInfos(this.type, set);
072
073        // Remove overridden and duplicate methods
074        final LinkedHashMap<String, HashSet<MethodInfo>> nameSetMap = new LinkedHashMap<String, HashSet<MethodInfo>>();
075    infoLoop:
076        for (MethodInfo methodInfo : set) {
077            final String name = methodInfo.getMethod().getName();
078            final Method method = methodInfo.getMethod();
079            HashSet<MethodInfo> nameSet = nameSetMap.get(name);
080            if (nameSet == null) {
081                nameSet = new HashSet<MethodInfo>();
082                nameSetMap.put(name, nameSet);
083                nameSet.add(methodInfo);
084                continue;
085            }
086            for (Iterator<MethodInfo> i = nameSet.iterator(); i.hasNext(); ) {
087                final Method otherMethod = i.next().getMethod();
088                if (MethodAnnotationScanner.overrides(method, otherMethod)) {
089                    i.remove();
090                    continue;
091                }
092                if (MethodAnnotationScanner.overrides(otherMethod, method))
093                    continue infoLoop;
094            }
095            nameSet.add(methodInfo);
096        }
097
098        // Return result
099        final HashSet<MethodInfo> result = new HashSet<MethodInfo>();
100        for (HashSet<MethodInfo> nameSet : nameSetMap.values())
101            result.addAll(set);
102        return result;
103    }
104
105    /**
106     * Scan the given type and all its supertypes for annotated methods and add them to the list.
107     *
108     * @param klass type to scan, possibly null
109     * @param set set to add methods to
110     */
111    protected void findMethodInfos(Class<?> klass, Set<MethodInfo> set) {
112
113        // Stop at null
114        if (klass == null)
115            return;
116
117        // Search methods
118        for (Method method : klass.getDeclaredMethods()) {
119            final A annotation = method.getAnnotation(this.annotationType);
120            if (annotation == null || !this.includeMethod(method, annotation))
121                continue;
122            final MethodInfo methodInfo = this.createMethodInfo(method, annotation);
123            if (methodInfo != null)
124                set.add(methodInfo);
125        }
126
127        // Recurse on interfaces
128        for (Class<?> iface : klass.getInterfaces())
129            this.findMethodInfos(iface, set);
130
131        // Recurse on superclass
132        this.findMethodInfos(klass.getSuperclass(), set);
133    }
134
135    /**
136     * Determine whether the annotated method should be included.
137     *
138     * <p>
139     * The implementation in {@link MethodAnnotationScanner} returns true if {@code method} takes zero parameters
140     * and returns anything other than void, otherwise false.
141     * </p>
142     *
143     * <p>
144     * Subclasses may apply different tests and optionally throw an exception if a method is improperly annotated.
145     * </p>
146     *
147     * @param method method to check
148     * @param annotation method's annotation
149     * @throws RuntimeException if method is erroneously annotated
150     * @return true to include method, false to ignore it
151     */
152    protected boolean includeMethod(Method method, A annotation) {
153        return method.getReturnType() != void.class && method.getParameterTypes().length == 0;
154    }
155
156    /**
157     * Determine if one method strictly overrides another.
158     *
159     * @param override possible overriding method (i.e., subclass method)
160     * @param original possible overriding method (i.e., superclass method)
161     * @return true if {@code override} overrides {@code original}, otherwise false;
162     *  if {@code override} equals {@code original}, false is returned
163     */
164    public static boolean overrides(Method override, Method original) {
165
166        // Check class hierarchy
167        if (!original.getDeclaringClass().isAssignableFrom(override.getDeclaringClass())
168          || original.getDeclaringClass() == override.getDeclaringClass())
169            return false;
170
171        // Compare names
172        if (!original.getName().equals(override.getName()))
173            return false;
174
175        // Check staticness
176        if ((original.getModifiers() & Modifier.STATIC) != 0 || (override.getModifiers() & Modifier.STATIC) != 0)
177            return false;
178
179        // Compare (raw) parameter types
180        if (!Arrays.equals(original.getParameterTypes(), override.getParameterTypes()))
181            return false;
182
183        // Done
184        return true;
185    }
186
187    /**
188     * Create a new {@link MethodInfo} instance corresponding to the given annotated method.
189     *
190     * <p>
191     * The implementation in {@link MethodAnnotationScanner} just instantiates a {@link MethodInfo} directly.
192     * </p>
193     *
194     * @param method the method
195     * @param annotation the annotation annotating the method
196     */
197    protected MethodInfo createMethodInfo(Method method, A annotation) {
198        return new MethodInfo(method, annotation);
199    }
200
201    /**
202     * Holds information about an annotated method detected by a {@link MethodAnnotationScanner}.
203     *
204     * @see MethodAnnotationScanner#findAnnotatedMethods
205     */
206    public class MethodInfo {
207
208        private final Method method;
209        private final A annotation;
210
211        public MethodInfo(Method method, A annotation) {
212            if (method == null)
213                throw new IllegalArgumentException("null method");
214            if (annotation == null)
215                throw new IllegalArgumentException("null annotation");
216            this.method = method;
217            this.annotation = annotation;
218            try {
219                this.method.setAccessible(true);
220            } catch (SecurityException e) {
221                // ignore
222            }
223        }
224
225        /**
226         * Get the annotated method.
227         *
228         * @return annotated method taking zero parameters and returning non-void
229         */
230        public Method getMethod() {
231            return this.method;
232        }
233
234        /**
235         * Get the method annotattion.
236         */
237        public A getAnnotation() {
238            return this.annotation;
239        }
240
241        /**
242         * Get the Java bean property name implied by this method's name, if any.
243         *
244         * @throws IllegalArgumentException if the method's name does not follow Java bean conventions
245         */
246        public String getMethodPropertyName() {
247            final String name = this.method.getName();
248            if (name.startsWith("get") && name.length() > 3)
249                return Introspector.decapitalize(name.substring(3));
250            if (name.startsWith("is") && name.length() > 2 && Primitive.get(method.getReturnType()) == Primitive.BOOLEAN)
251                return Introspector.decapitalize(name.substring(2));
252            throw new IllegalArgumentException("can't infer property name from non-Java bean method " + this.method);
253        }
254
255        /**
256         * Invoke the method and return the result. Any checked exception thrown is rethrown after being wrapped
257         * in a {@link RuntimeException}.
258         *
259         * @return result of invoking method
260         * @throws RuntimeException if invocation fails
261         */
262        public Object invoke(T obj, Object... params) {
263            try {
264                return this.method.invoke(obj, params);
265            } catch (InvocationTargetException e) {
266                if (e.getCause() instanceof RuntimeException)
267                    throw (RuntimeException)e.getCause();
268                if (e.getCause() instanceof Error)
269                    throw (Error)e.getCause();
270                throw new RuntimeException(e.getCause());
271            } catch (IllegalAccessException e) {
272                throw new RuntimeException(e);
273            }
274        }
275
276        @Override
277        public boolean equals(Object obj) {
278            if (obj == null || obj.getClass() != this.getClass())
279                return false;
280            final MethodAnnotationScanner<?, ?>.MethodInfo that = (MethodAnnotationScanner<?, ?>.MethodInfo)obj;
281            return this.method.equals(that.method) && this.annotation.equals(that.annotation);
282        }
283
284        @Override
285        public int hashCode() {
286            return this.method.hashCode() ^ this.annotation.hashCode();
287        }
288    }
289}
