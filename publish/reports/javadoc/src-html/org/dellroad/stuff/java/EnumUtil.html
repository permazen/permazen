008package org.dellroad.stuff.java;
009
010import java.util.Arrays;
011import java.util.Collections;
012import java.util.List;
013
014/**
015 * Utility methods for {@link Enum}s.
016 */
017public final class EnumUtil {
018
019    private EnumUtil() {
020    }
021
022    /**
023     * Get all instances of the given {@link Enum} class in a list in their natural ordering.
024     *
025     * @return unmodifiable list of enum values
026     */
027    @SuppressWarnings("unchecked")
028    public static <T extends Enum<?>> List<T> getValues(Class<T> enumClass) {
029
030        // Generate ClassCastException if type is not an enum type
031        enumClass.asSubclass(Enum.class);
032
033        // Get values
034        Object array;
035        try {
036            array = enumClass.getMethod("values").invoke(null);
037        } catch (Exception e) {
038            throw new RuntimeException("unexpected exception", e);
039        }
040        return Collections.unmodifiableList(Arrays.asList((T[])array));
041    }
042}
