010/**
011 * Visitor pattern interface for {@link Primitive}s.
012 *
013 * @param <R> switch method return type
014 */
015public interface PrimitiveSwitch<R> {
016
017    R caseVoid();
018
019    R caseBoolean();
020
021    R caseByte();
022
023    R caseCharacter();
024
025    R caseShort();
026
027    R caseInteger();
028
029    R caseFloat();
030
031    R caseLong();
032
033    R caseDouble();
034}
