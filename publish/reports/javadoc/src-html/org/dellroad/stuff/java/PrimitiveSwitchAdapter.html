008package org.dellroad.stuff.java;
009
010/**
011 * Adapter class for {@link PrimitiveSwitch} implementations.
012 *
013 * @param <R> switch method return type
014 * @since 1.0.65
015 */
016public class PrimitiveSwitchAdapter<R> implements PrimitiveSwitch<R> {
017
018    /**
019     * Handle the {@link Primitive#VOID} case.
020     * The implementation in {@link PrimitiveSwitchAdapter} delegates to {@link #caseDefault}.
021     */
022    @Override
023    public R caseVoid() {
024        return this.caseDefault();
025    }
026
027    /**
028     * Handle the {@link Primitive#BOOLEAN} case.
029     * The implementation in {@link PrimitiveSwitchAdapter} delegates to {@link #caseDefault}.
030     */
031    @Override
032    public R caseBoolean() {
033        return this.caseDefault();
034    }
035
036    /**
037     * Handle the {@link Primitive#BYTE} case.
038     * The implementation in {@link PrimitiveSwitchAdapter} delegates to {@link #caseNumber}.
039     */
040    @Override
041    public R caseByte() {
042        return this.caseNumber();
043    }
044
045    /**
046     * Handle the {@link Primitive#CHARACTER} case.
047     * The implementation in {@link PrimitiveSwitchAdapter} delegates to {@link #caseDefault}.
048     */
049    @Override
050    public R caseCharacter() {
051        return this.caseDefault();
052    }
053
054    /**
055     * Handle the {@link Primitive#SHORT} case.
056     * The implementation in {@link PrimitiveSwitchAdapter} delegates to {@link #caseNumber}.
057     */
058    @Override
059    public R caseShort() {
060        return this.caseNumber();
061    }
062
063    /**
064     * Handle the {@link Primitive#INTEGER} case.
065     * The implementation in {@link PrimitiveSwitchAdapter} delegates to {@link #caseNumber}.
066     */
067    @Override
068    public R caseInteger() {
069        return this.caseNumber();
070    }
071
072    /**
073     * Handle the {@link Primitive#FLOAT} case.
074     * The implementation in {@link PrimitiveSwitchAdapter} delegates to {@link #caseNumber}.
075     */
076    @Override
077    public R caseFloat() {
078        return this.caseNumber();
079    }
080
081    /**
082     * Handle the {@link Primitive#LONG} case.
083     * The implementation in {@link PrimitiveSwitchAdapter} delegates to {@link #caseNumber}.
084     */
085    @Override
086    public R caseLong() {
087        return this.caseNumber();
088    }
089
090    /**
091     * Handle the {@link Primitive#DOUBLE} case.
092     * The implementation in {@link PrimitiveSwitchAdapter} delegates to {@link #caseNumber}.
093     */
094    @Override
095    public R caseDouble() {
096        return this.caseNumber();
097    }
098
099    /**
100     * Internal roll-up method.
101     * The implementation in {@link PrimitiveSwitchAdapter} delegates to {@link #caseDefault}.
102     */
103    protected R caseNumber() {
104        return this.caseDefault();
105    }
106
107    /**
108     * Internal roll-up method.
109     * The implementation in {@link PrimitiveSwitchAdapter} returns {@code null}.
110     */
111    protected R caseDefault() {
112        return null;
113    }
114}
