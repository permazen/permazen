008package org.dellroad.stuff.java;
009
010/**
011 * Utility class for performing timed waits on objects.
012 *
013 * @see Object#wait
014 */
015public final class TimedWait {
016
017    private static final long MAX_WAIT_MILLISECONDS = 0x7fffffffffffffffL / 1000000L;
018
019    private TimedWait() {
020    }
021
022    /**
023     * Wait (using {@link Object#wait(long) Object.wait()}) up to a given time limit for some predicate to become true.
024     * This method correctly handles {@link Object#wait spurious wakeups}, restarting the wait loop as necessary.
025     *
026     * <p>
027     * This method assumes that {@code obj} will be notified whent the predicate becomes true and that the current thread
028     * is already synchronized on {@code obj}.
029     * </p>
030     *
031     * <p>
032     * This method uses {@link System#nanoTime} instead of {@link System#currentTimeMillis} and so is immune to
033     * adjustments in clock time.
034     * </p>
035     *
036     * @param obj       object to sleep on; must already be locked
037     * @param timeout   wait timeout in milliseconds, or zero for an infinite wait
038     * @param predicate predicate to test
039     * @return true if the predicate became true before the timeout, false if the timeout expired
040     * @throws IllegalArgumentException     if {@code timeout} is negative
041     * @throws IllegalMonitorStateException if {@code obj} is not already locked
042     * @throws InterruptedException         if the current thread is interrupted
043     */
044    public static boolean wait(Object obj, long timeout, final Predicate predicate) throws InterruptedException {
045
046        // Sanity check timeout value
047        if (timeout < 0)
048            throw new IllegalArgumentException("timeout < 0");
049
050        // Record start time (only if there's a timeout)
051        long startTime = timeout > 0 ? System.nanoTime() : 0;
052
053        // Loop waiting for predicate
054        while (!predicate.test()) {
055
056            // Did the timeout expire?
057            if (timeout < 0)
058                return false;
059
060            // Wait for remaining timeout to be woken up, but not more than 292 years :)
061            obj.wait(Math.min(timeout, MAX_WAIT_MILLISECONDS));
062
063            // If there's a timeout, subtract the time we just waited (rounding to the nearest millisecond)
064            if (timeout > 0) {
065                long stopTime = System.nanoTime();
066                timeout -= (stopTime - startTime + 500000L) / 1000000L;
067                if (timeout == 0)               // don't convert the last millisecond into an infinite wait
068                    timeout = -1;
069                startTime = stopTime;
070            }
071        }
072
073        // Predicate was true
074        return true;
075    }
076}
