008    package org.dellroad.stuff.java;
009    
010    import org.slf4j.LoggerFactory;
011    
012    /**
013     * Enumeration of possible actions to take when an error of some kind is detected.
014     */
015    public enum ErrorAction {
016    
017        /**
018         * Ignore the error: do nothing.
019         */
020        IGNORE {
021            @Override
022            public void execute(String message) {
023            }
024        },
025    
026        /**
027         * Log the error.
028         */
029        LOG {
030            @Override
031            public void execute(String message) {
032                LoggerFactory.getLogger(this.getClass()).error(message);
033            }
034        },
035    
036        /**
037         * Raise an assertion error (if assertions are enabled).
038         */
039        ASSERT {
040            @Override
041            public void execute(String message) {
042                assert false : message;
043            }
044        },
045    
046        /**
047         * Throw a {@link RuntimeException}.
048         */
049        EXCEPTION {
050            @Override
051            public void execute(String message) {
052                throw new RuntimeException(message);
053            }
054        };
055    
056        /**
057         * Take the action appropriate for this instance.
058         *
059         * @param message a message describing the error
060         * @throws AssertionError if this instance is {@link #ASSERT} and assertions are enabled
061         * @throws RuntimeException if this instance is {@link #EXCEPTION}
062         */
063        public abstract void execute(String message);
064    }
