008package org.dellroad.stuff.java;
009
010import java.beans.Introspector;
011import java.lang.annotation.Annotation;
012import java.lang.reflect.InvocationTargetException;
013import java.lang.reflect.Method;
014import java.util.ArrayList;
015import java.util.Collections;
016import java.util.LinkedHashMap;
017import java.util.List;
018
019/**
020 * Scans for annotated methods in a class hierarchy.
021 * Both superclasses and superinterfaces are introspected, and overrides are automatically detected.
022 * Only methods taking zero parameters and returning non-void are detected.
023 *
024 * @param <T> Java class to be introspected
025 * @param <A> Java annotation type
026 */
027public class MethodAnnotationScanner<T, A extends Annotation> {
028
029    protected final Class<T> type;
030    protected final Class<A> annotationType;
031
032    /**
033     * Constructor.
034     *
035     * @param type Java class to be introspected
036     * @param annotationType Java annotation type to search for
037     * @throws IllegalArgumentException if either parameter is null
038     */
039    public MethodAnnotationScanner(Class<T> type, Class<A> annotationType) {
040
041        // Sanity check
042        if (type == null)
043            throw new IllegalArgumentException("null type");
044        if (annotationType == null)
045            throw new IllegalArgumentException("null annotationType");
046        this.type = type;
047        this.annotationType = annotationType;
048    }
049
050    /**
051     * Build list of annotated methods, with overridden annotated methods omitted.
052     *
053     * @return list of annotated methods with any overridden annotated methods removed
054     */
055    public List<MethodInfo> buildMethodInfoList() {
056
057        // Find all methods
058        final ArrayList<MethodInfo> list = new ArrayList<MethodInfo>();
059        this.findMethodInfos(this.type, list);
060
061        // Sort them
062        Collections.sort(list);
063
064        // Remove overridden methods
065        final LinkedHashMap<String, MethodInfo> map = new LinkedHashMap<String, MethodInfo>();
066        for (MethodInfo methodInfo : list) {
067            if (map.containsKey(methodInfo.getMethod().getName()))
068                continue;
069            map.put(methodInfo.getMethod().getName(), methodInfo);
070        }
071
072        // Return result
073        return new ArrayList<MethodInfo>(map.values());
074    }
075
076    /**
077     * Scan the given type and all its supertypes for annotated methods and add them to the list
078     *
079     * @param klass type to scan, possibly null
080     * @param list list to append to
081     */
082    protected void findMethodInfos(Class<?> klass, List<MethodInfo> list) {
083
084        // Stop at null
085        if (klass == null)
086            return;
087
088        // Search methods
089        for (Method method : klass.getDeclaredMethods()) {
090            if (method.getReturnType() == void.class || method.getParameterTypes().length > 0)
091                continue;
092            final A annotation = method.getAnnotation(this.annotationType);
093            if (annotation != null) {
094                final MethodInfo methodInfo = this.createMethodInfo(method, annotation);
095                if (methodInfo != null)
096                    list.add(methodInfo);
097            }
098        }
099
100        // Recurse on interfaces
101        for (Class<?> iface : klass.getInterfaces())
102            this.findMethodInfos(iface, list);
103
104        // Recurse on superclass
105        this.findMethodInfos(klass.getSuperclass(), list);
106    }
107
108    /**
109     * Create a new {@link MethodInfo} instance corresponding to the given annotated method.
110     *
111     * <p>
112     * The implementation in {@link MethodAnnotationScanner} just instantiates a {@link MethodInfo} directly.
113     * </p>
114     *
115     * @param method the method
116     * @param annotation the annotation annotating the method
117     */
118    protected MethodInfo createMethodInfo(Method method, A annotation) {
119        return new MethodInfo(method, annotation);
120    }
121
122    /**
123     * Holds information about an annotated method detected by a {@link MethodAnnotationScanner}.
124     */
125    public class MethodInfo implements Comparable<MethodInfo> {
126
127        private final Method method;
128        private final A annotation;
129
130        public MethodInfo(Method method, A annotation) {
131            this.method = method;
132            this.annotation = annotation;
133            try {
134                this.method.setAccessible(true);
135            } catch (SecurityException e) {
136                // ignore
137            }
138        }
139
140        /**
141         * Get the annotated method.
142         *
143         * @return annotated method taking zero parameters and returning non-void
144         */
145        public Method getMethod() {
146            return this.method;
147        }
148
149        /**
150         * Get the method annotattion.
151         */
152        public A getAnnotation() {
153            return this.annotation;
154        }
155
156        /**
157         * Get the Java bean property name implied by this method's name, if any.
158         *
159         * @throws IllegalArgumentException if the method's name does not follow Java bean conventions
160         */
161        public String getMethodPropertyName() {
162            final String name = this.method.getName();
163            if (name.startsWith("get") && name.length() > 3)
164                return Introspector.decapitalize(name.substring(3));
165            if (name.startsWith("is") && name.length() > 2 && Primitive.get(method.getReturnType()) == Primitive.BOOLEAN)
166                return Introspector.decapitalize(name.substring(2));
167            throw new IllegalArgumentException("can't infer property name from non-Java bean method " + this.method);
168        }
169
170        /**
171         * Invoke the method to read its value.
172         *
173         * @return result of invoking method
174         */
175        public Object readValue(T obj) {
176            try {
177                return this.method.invoke(obj);
178            } catch (InvocationTargetException e) {
179                throw new RuntimeException(e);
180            } catch (IllegalAccessException e) {
181                throw new RuntimeException(e);
182            }
183        }
184
185        /**
186         * Sorts instances first by declaring class (subtypes before supertypes), then by method name.
187         */
188        @Override
189        public int compareTo(MethodInfo that) {
190            final Class<?> thisClass = this.method.getDeclaringClass();
191            final Class<?> thatClass = that.method.getDeclaringClass();
192            if (thisClass != thatClass) {
193                if (thatClass.isAssignableFrom(thisClass))
194                    return -1;
195                if (thisClass.isAssignableFrom(thatClass))
196                    return 1;
197            }
198            return this.method.getName().compareTo(that.method.getName());
199        }
200    }
201}
