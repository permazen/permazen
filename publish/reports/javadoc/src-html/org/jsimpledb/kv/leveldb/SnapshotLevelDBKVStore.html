008package org.jsimpledb.kv.leveldb;
009
010import org.iq80.leveldb.DB;
011import org.iq80.leveldb.ReadOptions;
012import org.iq80.leveldb.Snapshot;
013import org.jsimpledb.kv.CloseableKVStore;
014import org.slf4j.Logger;
015import org.slf4j.LoggerFactory;
016
017/**
018 * Read-only {@link LevelDBKVStore} based on a LevelDB {@link Snapshot}.
019 *
020 * <p>
021 * This class ensures that the configured {@link Snapshot} is closed when this instance is closed.
022 *
023 * <p>
024 * All mutation operations throw {@link UnsupportedOperationException}.
025 */
026public class SnapshotLevelDBKVStore extends LevelDBKVStore implements CloseableKVStore {
027
028    private final Logger log = LoggerFactory.getLogger(this.getClass());
029    private final Snapshot snapshot;
030    private boolean closed;
031
032    /**
033     * Constructor.
034     *
035     * @param db LevelDB database to snapshot
036     * @param verifyChecksums whether to verify checksums on reads
037     */
038    public SnapshotLevelDBKVStore(DB db, boolean verifyChecksums) {
039        this(db, db.getSnapshot(), verifyChecksums);
040    }
041
042    private SnapshotLevelDBKVStore(DB db, Snapshot snapshot, boolean verifyChecksums) {
043        super(db, new ReadOptions().snapshot(snapshot).verifyChecksums(verifyChecksums), null);
044        this.snapshot = snapshot;
045    }
046
047    @Override
048    public synchronized void close() {
049        super.close();
050        if (this.closed)
051            return;
052        this.closed = true;
053        try {
054            this.snapshot.close();
055        } catch (Throwable e) {
056            this.log.error("caught exception closing LevelDB snapshot (ignoring)", e);
057        }
058    }
059
060// KVStore
061
062    @Override
063    public void put(byte[] key, byte[] value) {
064        throw new UnsupportedOperationException("KVStore is read-only");
065    }
066
067    @Override
068    public void remove(byte[] key) {
069        throw new UnsupportedOperationException("KVStore is read-only");
070    }
071
072    @Override
073    public void removeRange(byte[] minKey, byte[] maxKey) {
074        throw new UnsupportedOperationException("KVStore is read-only");
075    }
076
077    @Override
078    public void adjustCounter(byte[] key, long amount) {
079        throw new UnsupportedOperationException("KVStore is read-only");
080    }
081}
