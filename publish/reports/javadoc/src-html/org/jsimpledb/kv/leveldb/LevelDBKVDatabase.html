029/**
030 * {@link org.jsimpledb.kv.KVDatabase} implementation based on LevelDB and providing concurrent transactions
031 * and linearizable ACID semantics.
032 *
033 * <p>
034 * A {@linkplain #setDirectory database directory} is the only required configuration property.
035 * Instances may be stopped and (re)started multiple times.
036 * </p>
037 *
038 * <p>
039 * The LevelDB Java implementation is configurable by specifying the {@link DBFactory} implementation.
040 * </p>
041 *
042 * @see <a href="https://github.com/dain/leveldb">leveldb</a>
043 * @see <a href="https://github.com/fusesource/leveldbjni">leveldbjni</a>
044 */
045public class LevelDBKVDatabase extends SnapshotKVDatabase {

048     * Class name for the {@link DBFactory} provided by <a href="https://github.com/fusesource/leveldbjni">leveldbjni</a>.
049     */
050    public static final String LEVELDBJNI_CLASS_NAME = "org.fusesource.leveldbjni.JniDBFactory";
051
052    /**
053     * Class name for the {@link DBFactory} provided by <a href="https://github.com/dain/leveldb">leveldb</a>.
054     */
055    public static final String LEVELDB_CLASS_NAME = "org.iq80.leveldb.impl.Iq80DBFactory";
056
057    /**
058     * The name of a system property that can be set to override the default {@link DBFactory} logic.
059     * Set to the name of a class that implements {@link DBFactory} and has a zero-arg constructor.
060     */
061    public static final String DB_FACTORY_PROPERTY = LevelDBKVDatabase.class.getName() + ".db_factory";

104    private static DBFactory getDefaultDBFactory() {
105
106        // Get class names to try
107        final ArrayList<String> classNames = new ArrayList<>(3);
108        final String configuredFactoryClass = System.getProperty(DB_FACTORY_PROPERTY, null);
109        if (configuredFactoryClass != null)
110            classNames.add(configuredFactoryClass);
111        classNames.add(LEVELDBJNI_CLASS_NAME);
112        classNames.add(LEVELDB_CLASS_NAME);
113
114        // Find class
115        final ClassLoader loader = Thread.currentThread().getContextClassLoader();
116        final Logger log = LoggerFactory.getLogger(LevelDBKVDatabase.class);
117        for (String className : classNames) {
118            try {
119                return Class.forName(className, false, loader).asSubclass(DBFactory.class).newInstance();
120            } catch (Exception e) {
121                if (log.isDebugEnabled())
122                    log.debug("can't load factory class `" + className + "': " + e);
123                continue;
124            }
125        }
126        throw new RuntimeException("no " + DBFactory.class.getName() + " implementation found; tried: " + classNames);
127    }

129// Lifecycle
130
131    /**
132     * Start this instance. This method must be called prior to creating any transactions.
133     *
134     * <p>
135     * This method is idempotent.
136     * </p>
137     *
138     * @throws IllegalStateException if this instance is not properly configured
139     */
140    @PostConstruct
141    public synchronized void start() {
142
143        // Already started?
144        if (this.db != null)
145            return;
146        this.log.info("starting " + this);
147
148        // Check configuration
149        if (this.directory == null)
150            throw new IllegalStateException("no directory configured");
151
152        // Open database
153        if (this.log.isDebugEnabled())
154            this.log.debug("opening " + this + " LevelDB database");
155        try {
156            this.db = this.factory.open(this.directory, this.options);
157        } catch (IOException e) {
158            throw new RuntimeException("LevelDB database startup failed", e);
159        }
160
161        // Configure KVStore
162        this.setKVStore(new LevelDBKVStore(this.db, new ReadOptions().verifyChecksums(this.options.verifyChecksums()), null));
163
164        // Add shutdown hook so we don't leak native resources
165        if (this.shutdownHookRegistered.compareAndSet(false, true)) {
166            Runtime.getRuntime().addShutdownHook(new Thread() {
167                public void run() {
168                    LevelDBKVDatabase.this.stop();
169                }
170            });
171        }
172    }
173
174    /**
175     * Stop this instance. Does nothing if not {@linkplain #start started} or already stopped.
176     */
177    @PreDestroy
178    public void stop() {
179
180        // Set stopping flag to prevent new transactions from being created
181        synchronized (this) {
182            if (this.db == null || this.stopping)
183                return;
184            this.log.info("stopping " + this);
185            this.stopping = true;
186        }
187
188        // Close any remaining open transactions
189        this.closeTransactions();
190
191        // Finish up
192        synchronized (this) {
193
194            // Sanity check
195            assert this.db != null;
196
197            // Close KVStore
198            ((LevelDBKVStore)this.getKVStore()).close();
199            this.setKVStore(null);
200
201            // Shut down LevelDB database
202            try {
203                if (this.log.isDebugEnabled())
204                    this.log.info("closing " + this + " LevelDB database");
205                this.db.close();
206            } catch (Throwable e) {
207                this.log.error("caught exception closing database during shutdown (ignoring)", e);
208            }
209
210            // Reset state
211            this.db = null;
212            this.stopping = false;
213        }
214    }

363// KVDatabase

365    /**
366     * Create a new transaction.
367     *
368     * @throws IllegalStateException if this instance is not {@linkplain #start started}
369     */
370    @Override
371    public synchronized LevelDBKVTransaction createTransaction() {
372
373        // Sanity check
374        if (this.db == null)
375            throw new IllegalStateException("not started");
376        if (this.stopping)
377            throw new IllegalStateException("stop in progress");
378
379        // OK
380        return (LevelDBKVTransaction)super.createTransaction();
381    }

383// SnapshotKVDatabase

385    @Override
386    protected LevelDBKVTransaction createSnapshotKVTransaction(SnapshotVersion versionInfo) {
387        return new LevelDBKVTransaction(this, versionInfo);
388    }
389
390    @Override
391    protected RuntimeException wrapException(SnapshotKVTransaction tx, RuntimeException e) {
392        if (e instanceof DBException)
393            return new KVTransactionException(tx, "LevelDB error", e);
394        return e;
395    }
