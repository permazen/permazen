019/**
020 * Manager of read/write locks on {@code byte[]} key ranges that ensures isolation and serialization while allowing concurrent
021 * access by multiple threads to a single underlying {@code byte[]} key/value store.
022 *
023 * <p>
024 * This implementation is straightforward: read locks can overlap, but write locks may not, and all locks owned
025 * by the same owner remain in force until all are {@linkplain #release released} at the same time.
026 * </p>
027 *
028 * <p>
029 * Instances are configured with a monitor object which is used for all internal locking and inter-thread wait/notify
030 * handshaking (by default, this instance). A user-supplied monitor object may be provided via the constructor.
031 * </p>
032 *
033 * <p>
034 * Two timeout values are supported:
035 * <ul>
036 *  <li>The wait timeout (specified as a parameter to {@link #lock lock()}) limits how long a thread will wait
037 *      on a lock held by another thread before giving up</li>
038 *  <li>The {@linkplain #getHoldTimeout hold timeout} limits how long a thread may hold
039 *      on to a contested lock before being forced to release all its locks; after that, the
040 *      next call to {@link #lock lock} or {@link #release release} will fail</li>
041 * </ul>
042 *
043 * <p>
044 * Note that if the hold timeout is set to zero (unlimited), then an application bug that leaks locks will result
045 * in those locks never being released.
046 * </p>
047 */
048public class LockManager {
049
050    private static final long TEN_YEARS_MILLIS = 10L * 365L * 24L * 60L * 60L * 1000L;
051
052    private final Object lockObject;
053
054    // Contains each owner's lock time, or null if hold timeout has expired.
055    // Invariant: if owner has any locks, then owner is a key in this map and value is not null.
056    // In the case owner's hold timeout expired and another owner forced it to release all of its locks, the expired
057    // owner will own no locks but still exist in this map (with a null value), until its next lock() or release().
058    private final HashMap<LockOwner, Long> lockTimes = new HashMap<>();
059
060    private final TreeSet<Lock> locksByMin = new TreeSet<>(Lock.MIN_COMPARATOR);            // locks ordered by minimum
061    private final TreeSet<Lock> locksByMax = new TreeSet<>(Lock.MAX_COMPARATOR);            // locks ordered by maximum
062    private final long nanoBasis = System.nanoTime();
063
064    private long holdTimeout;
065
066    /**
067     * Convenience constructor. Equivalent to <code>LockManager(null)</code>.
068     */
069    public LockManager() {
070        this(null);
071    }
072
073    /**
074     * Primary constructor.
075     *
076     * @param lockObject Java object used to synchronize field access and inter-thread wait/notify handshake,
077     *  or null to use this instance
078     */
079    public LockManager(Object lockObject) {
080        this.lockObject = lockObject != null ? lockObject : this;
081    }
082
083    /**
084     * Get the hold timeout configured for this instance.
085     *
086     * <p>
087     * The hold timeout limits how long a thread may hold on to a contested lock before being forced to release
088     * all of its locks; after that, the next call to {@link #lock lock} or {@link #release release} will fail.
089     * </p>
090     *
091     * @return hold timeout in milliseconds
092     */
093    public long getHoldTimeout() {
094        synchronized (this.lockObject) {
095            return this.holdTimeout;
096        }
097    }
098
099    /**
100     * Set the hold timeout for this instance. Default is zero (unlimited).
101     *
102     * @param holdTimeout how long a thread may hold a contestested lock before {@link LockResult#HOLD_TIMEOUT_EXPIRED}
103     *  will be returned by {@link #lock lock()} or {@link #release release()} in milliseconds, or zero for unlimited
104     * @throws IllegalArgumentException if {@code holdTimeout} is negative
105     */
106    public void setHoldTimeout(long holdTimeout) {
107        synchronized (this.lockObject) {
108            if (holdTimeout < 0)
109                throw new IllegalArgumentException("holdTimeout < 0");
110            this.holdTimeout = Math.min(holdTimeout, TEN_YEARS_MILLIS);             // limit to 10 years to avoid overflow
111        }
112    }
113
114    /**
115     * Acquire a lock on behalf of the specified owner.
116     *
117     * <p>
118     * This method will block for up to {@code waitTimeout} milliseconds if the lock is held by
119     * another thread, after which point {@link LockResult#WAIT_TIMEOUT_EXPIRED} is returned.
120     * The configured locking object will be used for inter-thread wait/notify handshaking.
121     * </p>
122     *
123     * <p>
124     * If {@code owner} already holds one or more locks, but the {@linkplain #getHoldTimeout hold timeout} has expired,
125     * then {@link LockResult#HOLD_TIMEOUT_EXPIRED} is returned and all of the other locks are will have already been
126     * automatically released.
127     * </p>
128     *
129     * <p>
130     * Once a lock is successfully acquired, it stays acquired until all locks are released together via {@link #release release()}.
131     * </p>
132     *
133     * @param owner lock owner
134     * @param minKey minimum key (inclusive); must not be null
135     * @param maxKey maximum key (exclusive), or null for no maximum
136     * @param write true for a write lock, false for a read lock
137     * @param waitTimeout how long to wait before returning {@link LockResult#WAIT_TIMEOUT_EXPIRED}
138     *  in milliseconds, or zero for unlimited
139     * @return a {@link LockResult}
140     * @throws InterruptedException if the current thread is interrupted while waiting for the lock
141     * @throws IllegalArgumentException if {@code owner}, {@code minKey}, or {@code range} is null
142     * @throws IllegalArgumentException if {@code minKey > maxKey}
143     * @throws IllegalArgumentException if {@code waitTimeout} is negative
144     */
145    public LockResult lock(LockOwner owner, byte[] minKey, byte[] maxKey, boolean write, long waitTimeout)
146      throws InterruptedException {
147        synchronized (this.lockObject) {
148
149            // Sanity check
150            if (owner == null)
151                throw new IllegalArgumentException("null owner");
152            if (waitTimeout < 0)
153                throw new IllegalArgumentException("waitTimeout < 0");
154            waitTimeout = Math.min(waitTimeout, TEN_YEARS_MILLIS);                  // limit to 10 years to avoid overflow
155
156            // Check hold timeout
157            final long lockerRemaining = this.checkHoldTimeout(owner);
158            if (lockerRemaining == -1)
159                return LockResult.HOLD_TIMEOUT_EXPIRED;
160
161            // Create lock
162            Lock lock = new Lock(owner, minKey, maxKey, write);
163
164            // Wait for lockability, until the first one of:
165            //  - Wait timeout
166            //  - Locker's hold timeout
167            //  - Lock owner's hold timeout
168            final LockChecker lockChecker = new LockChecker(lock);
169            if (!lockChecker.test()) {
170                long timeToWait = Math.min(waitTimeout, lockChecker.getTimeRemaining());
171                if (lockerRemaining != 0)
172                    timeToWait = Math.min(timeToWait, lockerRemaining);
173                if (!TimedWait.wait(this.lockObject, timeToWait, lockChecker))
174                    return LockResult.WAIT_TIMEOUT_EXPIRED;
175            }
176
177            // Check hold timeout again
178            if (this.checkHoldTimeout(owner) == -1)
179                return LockResult.HOLD_TIMEOUT_EXPIRED;
180
181            // Merge the lock with other locks it can merge with, removing those locks in the process
182            for (Lock that : lockChecker.getMergers()) {
183                final Lock mergedLock = lock.mergeWith(that);
184                if (mergedLock != null) {
185                    this.locksByMin.remove(that);
186                    this.locksByMax.remove(that);
187                    owner.locks.remove(that);
188                    lock = mergedLock;
189                }
190            }
191
192            // Add lock
193            this.locksByMin.add(lock);
194            this.locksByMax.add(lock);
195            owner.locks.add(lock);
196
197            // Set hold timeout (if not already set)
198            if (!this.lockTimes.containsKey(owner)) {
199                final long currentTime = System.nanoTime() - this.nanoBasis;
200                this.lockTimes.put(owner, currentTime);
201            } else
202                assert this.lockTimes.get(owner) != null;
203
204            // Done
205            return LockResult.SUCCESS;
206        }
207    }
208
209    /**
210     * Release all locks held by the specified owner.
211     *
212     * <p>
213     * If the owner's {@linkplain #getHoldTimeout hold timeout} has already expired, then all locks will have
214     * already been released and false is returned.
215     * </p>
216     *
217     * <p>
218     * Does nothing (and returns true) if {@code owner} does not own any locks.
219     * </p>
220     *
221     * @param owner lock owner
222     * @return true if successful, false if {@code owner}'s hold timeout expired
223     * @throws IllegalArgumentException if {@code owner} is null
224     */
225    public boolean release(LockOwner owner) {
226        synchronized (this.lockObject) {
227
228            // Sanity check
229            if (owner == null)
230                throw new IllegalArgumentException("null owner");
231
232            // Check if hold timeout has alread expired; in any case, remove lock time
233            if (this.lockTimes.containsKey(owner)) {
234                final Long lockTime = this.lockTimes.remove(owner);
235                if (lockTime == null)
236                    return false;
237            }
238
239            // Release all locks
240            this.doRelease(owner);
241
242            // Done
243            return true;
244        }
245    }
246
247    /**
248     * Check whether the {@linkplain #getHoldTimeout hold timeout} has expired for the given lock owner
249     * and if not return the amount of time remaining.
250     *
251     * <p>
252     * If the owner's hold timeout has expired, then {@code -1} is returned and any locks previously held by {@code owner}
253     * will have been automatically released.
254     * </p>
255     *
256     * @param owner lock owner
257     * @return milliseconds until {@code owner}'s hold timeout expires, zero if {@code owner} has no hold timeout
258     *  (e.g., nothing is locked or hold timeout disabled), or -1 if {@code owner}'s hold timeout has expired
259     * @throws IllegalArgumentException if {@code owner} is null
260     */
261    public long checkHoldTimeout(LockOwner owner) {
262        synchronized (this.lockObject) {
263            if (this.holdTimeout == 0)
264                return 0;
265            if (!this.lockTimes.containsKey(owner))
266                return 0;
267            final Long lockTime = this.lockTimes.get(owner);
268            if (lockTime == null)
269                return -1;
270            final long currentTime = System.nanoTime() - this.nanoBasis;
271            final long holdDeadline = lockTime + this.holdTimeout * 1000000L;
272            final long remaining = holdDeadline - currentTime;
273            if (remaining <= 0) {
274                this.lockTimes.put(owner, null);
275                this.doRelease(owner);
276                return -1;
277            }
278            return (remaining + 999999L) / 1000000L;
279        }
280    }
281
282    // Release all locks held by owner. Assumes synchronized already on this.lockObject.
283    private void doRelease(LockOwner owner) {
284        for (Lock lock : owner.locks) {
285            this.locksByMin.remove(lock);
286            this.locksByMax.remove(lock);
287        }
288        owner.locks.clear();
289        this.lockObject.notifyAll();
290    }
291
292    // Check whether we can lock, and fill the list of mergers and return zero if so. If not, return time remaining.
293    // Assumes synchronized already on this.lockObject.
294    private long checkLock(Lock lock, List<Lock> mergers) {
295
296        // Get lock's min & max
297        final byte[] lockMin = lock.getMin();
298        final byte[] lockMax = lock.getMax();
299
300    startOver:
301        while (true) {
302
303            // Get locks whose min is < lockMax
304            final NavigableSet<Lock> lhs = lockMax == null ? this.locksByMin :
305              this.locksByMin.headSet(Lock.getMinKey(lockMax, false), false);
306
307            // Get locks whose max is > lockMin
308            final NavigableSet<Lock> rhs = lockMin == null ? this.locksByMax : this.locksByMax.tailSet(
309              Lock.getMaxKey(ByteUtil.getNextKey(lockMin), false), true);
310
311            // Find overlapping locks and check for conflicts
312            final HashSet<Lock> overlaps = new HashSet<>();
313            for (Lock other : lhs) {
314
315                // Does this lock overlap?
316                if (!rhs.contains(other))
317                    continue;
318
319                // Do this lock & other lock conflict?
320                if (lock.conflictsWith(other)) {
321
322                    // See if other lock's owner's hold timeout has expired
323                    assert this.lockTimes.containsKey(other.owner);
324                    final long remaining = this.checkHoldTimeout(other.owner);
325                    if (remaining == -1)
326                        continue startOver;
327
328                    // Return time remaining until conflicting owner's hold timeout
329                    return Math.max(remaining, 1);
330                }
331
332                // Add overlap
333                overlaps.add(other);
334            }
335
336            //System.out.println(Thread.currentThread().getName() + ": LockChecker: BEFORE: lock = " + lock + "\n"
337            //+ "  lockByMin = " + this.locksByMin + "\n"
338            //+ "  lockByMax = " + this.locksByMax + "\n"
339            //+ "   overlaps = " + overlaps + "\n");
340
341            // Find overlaps we can merge with
342            for (Lock other : overlaps) {
343                if (lock.mergeWith(other) != null)
344                    mergers.add(other);
345            }
346
347            //System.out.println(Thread.currentThread().getName() + ": LockChecker: AFTER: lock = " + lock + "\n"
348            //+ "  lockByMin = " + this.locksByMin + "\n"
349            //+ "  lockByMax = " + this.locksByMax + "\n"
350            //+ "    mergers = " + mergers + "\n");
351
352            // Done
353            return 0;
354        }
355    }

359    /**
360     * Possible return values from {@link LockManager#lock LockManager.lock()}.
361     */
362    public enum LockResult {
363
364        /**
365         * The lock was successfully acquired.
366         */
367        SUCCESS,
368
369        /**
370         * The timeout expired while waiting to acquire the lock.
371         */
372        WAIT_TIMEOUT_EXPIRED,
373
374        /**
375         * The owner's hold timeout expired.
376         */
377        HOLD_TIMEOUT_EXPIRED;
378    }
379
380// LockChecker predicate
381
382    private class LockChecker implements Predicate {
383
384        private final Lock lock;
385        private final ArrayList<Lock> mergers = new ArrayList<>();
386
387        private long timeRemaining;
388
389        LockChecker(Lock lock) {
390            this.lock = lock;
391        }
392
393        public List<Lock> getMergers() {
394            return this.mergers;
395        }
396
397        public long getTimeRemaining() {
398            return this.timeRemaining;
399        }
400
401        @Override
402        public boolean test() {
403
404            // Reset state
405            this.mergers.clear();
406
407            // See if we can lock
408            this.timeRemaining = LockManager.this.checkLock(this.lock, this.mergers);
409            return this.timeRemaining == 0;
410        }
411    }
