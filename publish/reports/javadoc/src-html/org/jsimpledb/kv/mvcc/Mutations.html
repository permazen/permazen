006package org.jsimpledb.kv.mvcc;
007
008import java.util.Map;
009
010import org.jsimpledb.kv.KeyRange;
011
012/**
013 * Represents a set of mutations that can be applied to a {@link org.jsimpledb.kv.KVStore}.
014 *
015 * <p>
016 * Each mutation is either a key/value put, the removal of a key range (possibly containing only a single key),
017 * or a counter adjustment. Mutations are expected to be applied in the order: removes, puts, adjusts.
018 */
019public interface Mutations {
020
021    /**
022     * Get the key ranges removals contained by this instance.
023     *
024     * @return key ranges removed
025     */
026    Iterable<KeyRange> getRemoveRanges();
027
028    /**
029     * Get the written key/value pairs contained by this instance.
030     *
031     * <p>
032     * The caller must not modify any of the returned {@code byte[]} arrays.
033     * </p>
034     *
035     * @return mapping from key to corresponding value
036     */
037    Iterable<Map.Entry<byte[], byte[]>> getPutPairs();
038
039    /**
040     * Get the counter adjustments contained by this instance.
041     *
042     * <p>
043     * The caller must not modify any of the returned {@code byte[]} arrays.
044     * </p>
045     *
046     * @return mapping from key to corresponding counter adjustment
047     */
048    Iterable<Map.Entry<byte[], Long>> getAdjustPairs();
049}
