006package org.jsimpledb.kv.mvcc;
007
008import java.io.Closeable;
009import java.util.concurrent.atomic.AtomicLong;
010
011import org.jsimpledb.kv.KVStore;
012import org.jsimpledb.kv.KVTransaction;
013import org.jsimpledb.kv.StaleTransactionException;
014import org.jsimpledb.kv.TransactionTimeoutException;
015import org.jsimpledb.kv.util.ForwardingKVStore;
016import org.slf4j.Logger;
017import org.slf4j.LoggerFactory;
018
019/**
020 * {@link SnapshotKVDatabase} transaction.
021 */
022public class SnapshotKVTransaction extends ForwardingKVStore implements KVTransaction, Closeable {
023
024// Note: locking order: (1) SnapshotKVTransaction, (2) SnapshotKVDatabase
025
026    private static final AtomicLong COUNTER = new AtomicLong();
027
028    private final Logger log = LoggerFactory.getLogger(this.getClass());
029    private final long uniqueId = COUNTER.incrementAndGet();
030    private final long startTime;
031    private final SnapshotKVDatabase kvdb;
032    private final SnapshotVersion versionInfo;
033    private final MutableView mutableView;
034
035    private boolean closed;
036    private long timeout;
037
038    /**
039     * Constructor.
040     *
041     * @param kvdb the associated database
042     * @param versionInfo the associated MVCC version
043     */
044    protected SnapshotKVTransaction(SnapshotKVDatabase kvdb, SnapshotVersion versionInfo) {
045        this.kvdb = kvdb;
046        this.versionInfo = versionInfo;
047        this.startTime = System.nanoTime();
048        this.mutableView = new MutableView(versionInfo.getSnapshot());
049    }
050
051// Accessors
052
053    /**
054     * Get the {@link MutableView} associated with this instance.
055     *
056     * @return associated access and mutation state
057     */
058    public MutableView getMutableView() {
059        return this.mutableView;
060    }
061
062    /**
063     * Get the MVCC version associated with this instance.
064     *
065     * @return associated MVCC version
066     */
067    public SnapshotVersion getSnapshotVersion() {
068        return this.versionInfo;
069    }
070
071// ForwardingKVStore
072
073    /**
074     * Get the underlying {@link KVStore}.
075     *
076     * <p>
077     * The implementation in {@link SnapshotKVTransaction} returns the {@link MutableView} associated with this instance.
078     *
079     * @return the underlying {@link KVStore}
080     */
081    @Override
082    protected KVStore delegate() {
083        this.checkState();
084        return this.mutableView;
085    }
086
087// KVTransaction
088
089    @Override
090    public SnapshotKVDatabase getKVDatabase() {
091        return this.kvdb;
092    }
093
094    /**
095     * Set the transaction timeout.
096     *
097     * <p>
098     * {@link SnapshotKVTransaction}s do not perform any locking while the transaction is open. Therefore, the configured
099     * value is used instead as a timeout on the overall transaction duration. If the transaction is kept open for longer
100     * than {@code timeout} milliseconds, a {@link TransactionTimeoutException} will be thrown.
101     * </p>
102     *
103     * @param timeout transaction timeout in milliseconds, or zero for unlimited
104     * @throws IllegalArgumentException if {@code timeout} is negative
105     */
106    @Override
107    public synchronized void setTimeout(long timeout) {
108        if (timeout < 0)
109            throw new IllegalArgumentException("timeout < 0");
110        this.timeout = timeout;
111    }
112
113    @Override
114    public synchronized void commit() {
115        this.checkState();
116        this.closed = true;
117        this.kvdb.commit(this);
118    }
119
120    @Override
121    public synchronized void rollback() {
122        if (this.closed)
123            return;
124        this.closed = true;
125        this.kvdb.rollback(this);
126    }
127
128// Closeable
129
130    /**
131     * Close this instance.
132     *
133     * <p>
134     * Equivalent to invoking {@link #rollback}.
135     */
136    @Override
137    public void close() {
138        this.rollback();
139    }
140
141// Object
142
143    @Override
144    public String toString() {
145        return this.getClass().getSimpleName()
146          + "[id=" + this.uniqueId
147          + ",vers=" + this.versionInfo.getVersion()
148          + (this.closed ? ",closed" : "")
149          + "]";
150    }
151
152    @Override
153    protected void finalize() throws Throwable {
154        try {
155            if (!this.closed)
156               this.log.warn(this + " leaked without commit() or rollback()");
157            this.close();
158        } finally {
159            super.finalize();
160        }
161    }
162
163// Internal methods
164
165    private void checkState() {
166        if (this.closed)
167            throw this.kvdb.logException(new StaleTransactionException(this));
168        if (this.timeout == 0)
169            return;
170        final long time = (System.nanoTime() - this.startTime) / 1000000L;
171        if (time >= this.timeout) {
172            throw this.kvdb.logException(new TransactionTimeoutException(this,
173              "transaction has timed out after " + time + "ms > limit of " + this.timeout + "ms"));
174        }
175    }
176}
