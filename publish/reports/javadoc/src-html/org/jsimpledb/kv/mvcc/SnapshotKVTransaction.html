006package org.jsimpledb.kv.mvcc;
007
008import com.google.common.base.Preconditions;
009import com.google.common.util.concurrent.ListenableFuture;
010
011import java.io.Closeable;
012import java.util.concurrent.atomic.AtomicLong;
013
014import org.jsimpledb.kv.CloseableKVStore;
015import org.jsimpledb.kv.KVStore;
016import org.jsimpledb.kv.KVTransaction;
017import org.jsimpledb.kv.StaleTransactionException;
018import org.jsimpledb.kv.TransactionTimeoutException;
019import org.jsimpledb.kv.util.CloseableForwardingKVStore;
020import org.jsimpledb.kv.util.ForwardingKVStore;
021import org.slf4j.Logger;
022import org.slf4j.LoggerFactory;
023
024/**
025 * {@link SnapshotKVDatabase} transaction.
026 */
027public class SnapshotKVTransaction extends ForwardingKVStore implements KVTransaction, Closeable {
028
029// Note: locking order: (1) SnapshotKVTransaction, (2) SnapshotKVDatabase
030
031    private static final AtomicLong COUNTER = new AtomicLong();
032
033    private final Logger log = LoggerFactory.getLogger(this.getClass());
034    private final long uniqueId = COUNTER.incrementAndGet();
035    private final long startTime;
036    private final SnapshotKVDatabase kvdb;
037    private final SnapshotVersion versionInfo;
038    private final MutableView mutableView;
039
040    private boolean closed;
041    private long timeout;
042
043    /**
044     * Constructor.
045     *
046     * @param kvdb the associated database
047     * @param versionInfo the associated MVCC version
048     */
049    protected SnapshotKVTransaction(SnapshotKVDatabase kvdb, SnapshotVersion versionInfo) {
050        this.kvdb = kvdb;
051        this.versionInfo = versionInfo;
052        this.startTime = System.nanoTime();
053        this.mutableView = new MutableView(versionInfo.getSnapshotRefs().getKVStore());
054    }
055
056// Accessors
057
058    /**
059     * Get the {@link MutableView} associated with this instance.
060     *
061     * @return associated access and mutation state
062     */
063    public MutableView getMutableView() {
064        return this.mutableView;
065    }
066
067    /**
068     * Get the MVCC version associated with this instance.
069     *
070     * @return associated MVCC version
071     */
072    public SnapshotVersion getSnapshotVersion() {
073        return this.versionInfo;
074    }
075
076// ForwardingKVStore
077
078    /**
079     * Get the underlying {@link KVStore}.
080     *
081     * <p>
082     * The implementation in {@link SnapshotKVTransaction} returns the {@link MutableView} associated with this instance.
083     *
084     * @return the underlying {@link KVStore}
085     */
086    @Override
087    protected KVStore delegate() {
088        this.checkState();
089        return this.mutableView;
090    }
091
092// KVTransaction
093
094    @Override
095    public SnapshotKVDatabase getKVDatabase() {
096        return this.kvdb;
097    }
098
099    /**
100     * Set the transaction timeout.
101     *
102     * <p>
103     * {@link SnapshotKVTransaction}s do not perform any locking while the transaction is open. Therefore, the configured
104     * value is used instead as a timeout on the overall transaction duration. If the transaction is kept open for longer
105     * than {@code timeout} milliseconds, a {@link TransactionTimeoutException} will be thrown.
106     * </p>
107     *
108     * @param timeout transaction timeout in milliseconds, or zero for unlimited
109     * @throws IllegalArgumentException if {@code timeout} is negative
110     */
111    @Override
112    public synchronized void setTimeout(long timeout) {
113        Preconditions.checkArgument(timeout >= 0, "timeout < 0");
114        this.timeout = timeout;
115    }
116
117    @Override
118    public ListenableFuture<Void> watchKey(byte[] key) {
119        return this.kvdb.watchKey(key);
120    }
121
122    @Override
123    public synchronized void commit() {
124        this.checkState();
125        this.closed = true;
126        this.kvdb.commit(this);
127    }
128
129    @Override
130    public synchronized void rollback() {
131        if (this.closed)
132            return;
133        this.closed = true;
134        this.kvdb.rollback(this);
135    }
136
137    @Override
138    public CloseableKVStore mutableSnapshot() {
139        final SnapshotRefs snapshotRefs;
140        synchronized (this) {
141            this.checkState();
142            snapshotRefs = this.versionInfo.getSnapshotRefs();
143            snapshotRefs.ref();
144        }
145        final Writes writes;
146        synchronized (this.mutableView) {
147            writes = this.mutableView.getWrites().clone();
148        }
149        final MutableView snapshotView = new MutableView(snapshotRefs.getKVStore(), null, writes);
150        return new CloseableForwardingKVStore(snapshotView, snapshotRefs.getUnrefCloseable());
151    }
152
153// Closeable
154
155    /**
156     * Close this instance.
157     *
158     * <p>
159     * Equivalent to invoking {@link #rollback}.
160     */
161    @Override
162    public void close() {
163        this.rollback();
164    }
165
166// Object
167
168    @Override
169    public String toString() {
170        return this.getClass().getSimpleName()
171          + "[id=" + this.uniqueId
172          + ",vers=" + this.versionInfo.getVersion()
173          + (this.closed ? ",closed" : "")
174          + "]";
175    }
176
177    @Override
178    protected void finalize() throws Throwable {
179        try {
180            if (!this.closed)
181               this.log.warn(this + " leaked without commit() or rollback()");
182            this.close();
183        } finally {
184            super.finalize();
185        }
186    }
187
188// Internal methods
189
190    private void checkState() {
191        if (this.closed)
192            throw this.kvdb.logException(new StaleTransactionException(this));
193        if (this.timeout == 0)
194            return;
195        final long time = (System.nanoTime() - this.startTime) / 1000000L;
196        if (time >= this.timeout) {
197            throw this.kvdb.logException(new TransactionTimeoutException(this,
198              "transaction has timed out after " + time + "ms > limit of " + this.timeout + "ms"));
199        }
200    }
201}
