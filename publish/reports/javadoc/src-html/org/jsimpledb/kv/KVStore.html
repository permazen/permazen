006package org.jsimpledb.kv;
007
008import java.util.Iterator;
009
010/**
011 * General API into a key/value store where the keys are sorted lexicographically as unsigned bytes.
012 *
013 * <p>
014 * Implementations are not required to support accessing keys that start with {@code 0xff}.
015 * </p>
016 *
017 * <p><b>Lock-free Counters</b></p>
018 *
019 * <p>
020 * Implementations are encouraged to include support for encoding a 64 bit counter in a key/value pair such that the counter
021 * can be efficiently {@linkplain #adjustCounter adjusted} by concurrent transactions without conflict.
022 * In practice this means no locking is required to increment or decrement the counter by some amount, as long as
023 * it's not necessary to actually directly read or write the counter value in the same transaction.
024 * Whether counter adjustments are actually lock-free is implementation dependent, however, the counter methods
025 * {@link #encodeCounter encodeCounter()}, {@link #decodeCounter decodeCounter()}, and {@link #adjustCounter adjustCounter()}
026 * must function correctly as specified in all cases.
027 * </p>
028 *
029 * <p>
030 * How counters are encoded is specific to the implementation. Clients needing to read or write counter values directly
031 * should use {@link #decodeCounter decodeCounter()} and {@link #encodeCounter encodeCounter()}, respectively.
032 * Counters are removed using the normal methods (i.e., {@link #remove remove()} and {@link #removeRange removeRange()}).
033 * </p>
034 */
035public interface KVStore {
036
037    /**
038     * Get the value associated with the given key, if any.
039     *
040     * @param key key
041     * @return value associated with key, or null if not found
042     * @throws IllegalArgumentException if {@code key} starts with {@code 0xff} and such keys are not supported
043     * @throws StaleTransactionException if an underlying transaction is no longer usable
044     * @throws RetryTransactionException if an underlying transaction must be retried and is no longer usable
045     * @throws NullPointerException if {@code key} is null
046     */
047    byte[] get(byte[] key);
048
049    /**
050     * Get the key/value pair having the smallest key greater than or equal to the given minimum, if any.
051     *
052     * <p>
053     * If keys starting with {@code 0xff} are not supported by this instance, and {@code minKey} starts with {@code 0xff},
054     * then this method returns null.
055     * </p>
056     *
057     * @param minKey minimum key (inclusive), or null for no minimum (get the smallest key)
058     * @return smallest key/value pair with {@code key >= minKey}, or null if none exists
059     * @throws StaleTransactionException if an underlying transaction is no longer usable
060     * @throws RetryTransactionException if an underlying transaction must be retried and is no longer usable
061     */
062    KVPair getAtLeast(byte[] minKey);
063
064    /**
065     * Get the key/value pair having the largest key strictly less than the given maximum, if any.
066     *
067     * <p>
068     * If keys starting with {@code 0xff} are not supported by this instance, and {@code maxKey} starts with {@code 0xff},
069     * then this method behaves as if {@code maxKey} were null.
070     * </p>
071     *
072     * @param maxKey maximum key (exclusive), or null for no maximum (get the largest key)
073     * @return largest key/value pair with {@code key < maxKey}, or null if none exists
074     * @throws StaleTransactionException if an underlying transaction is no longer usable
075     * @throws RetryTransactionException if an underlying transaction must be retried and is no longer usable
076     */
077    KVPair getAtMost(byte[] maxKey);
078
079    /**
080     * Iterate the key/value pairs in the specified range. The returned {@link Iterator}'s {@link Iterator#remove remove()}
081     * must be supported and should have the same effect as invoking {@link #remove remove()} on the corresponding key.
082     *
083     * <p>
084     * If keys starting with {@code 0xff} are not supported by this instance, and {@code minKey} starts with {@code 0xff},
085     * then this method returns an empty iteration.
086     * </p>
087     *
088     * <p>
089     * If keys starting with {@code 0xff} are not supported by this instance, and {@code maxKey} starts with {@code 0xff},
090     * then this method behaves as if {@code maxKey} were null.
091     * </p>
092     *
093     * <p>
094     * The returned {@link Iterator} must not throw {@link java.util.ConcurrentModificationException};
095     * however, whether or not a "live" {@link Iterator} reflects any modifications made after its creation is
096     * implementation dependent. Implementations that do make post-creation updates visible in the {@link Iterator},
097     * even if the update occurs after some delay, must preserve the order in which the modifications actually occurred.
098     * </p>
099     *
100     * <p>
101     * The returned {@link Iterator} is not guaranteed to be thread safe.
102     * </p>
103     *
104     * @param minKey minimum key (inclusive), or null for no minimum (start at the smallest key)
105     * @param maxKey maximum key (exclusive), or null for no maximum (end at the largest key)
106     * @param reverse true to return key/value pairs in reverse order (i.e., keys descending)
107     * @return iteration of key/value pairs in the range {@code minKey} (inclusive) to {@code maxKey} (exclusive)
108     * @throws IllegalArgumentException if {@code minKey > maxKey}
109     * @throws StaleTransactionException if an underlying transaction is no longer usable
110     * @throws RetryTransactionException if an underlying transaction must be retried and is no longer usable
111     */
112    Iterator<KVPair> getRange(byte[] minKey, byte[] maxKey, boolean reverse);
113
114    /**
115     * Set the value associated with the given key.
116     *
117     * @param key key
118     * @param value value
119     * @throws IllegalArgumentException if {@code key} starts with {@code 0xff} and such keys are not supported
120     * @throws StaleTransactionException if an underlying transaction is no longer usable
121     * @throws RetryTransactionException if an underlying transaction must be retried and is no longer usable
122     * @throws NullPointerException if {@code key} or {@code value} is null
123     */
124    void put(byte[] key, byte[] value);
125
126    /**
127     * Remove the key/value pair with the given key, if it exists.
128     *
129     * @param key key
130     * @throws IllegalArgumentException if {@code key} starts with {@code 0xff} and such keys are not supported
131     * @throws StaleTransactionException if an underlying transaction is no longer usable
132     * @throws RetryTransactionException if an underlying transaction must be retried and is no longer usable
133     * @throws NullPointerException if {@code key} is null
134     */
135    void remove(byte[] key);
136
137    /**
138     * Remove all key/value pairs whose keys are in a given range.
139     *
140     * <p>
141     * The {@code minKey} must be less than or equal to {@code maxKey}; if they equal (and not null)
142     * then nothing happens; if they are both null then all entries are deleted.
143     * </p>
144     *
145     * <p>
146     * If keys starting with {@code 0xff} are not supported by this instance, then:
147     * <ul>
148     *  <li>If {@code minKey} starts with {@code 0xff}, then no change occurs</li>
149     *  <li>If {@code maxKey} starts with {@code 0xff}, then this method behaves as if {@code maxKey} were null</li>
150     * </ul>
151     *
152     * @param minKey minimum key (inclusive), or null for no minimum
153     * @param maxKey maximum key (exclusive), or null for no maximum
154     * @throws IllegalArgumentException if {@code minKey > maxKey}
155     * @throws StaleTransactionException if an underlying transaction is no longer usable
156     * @throws RetryTransactionException if an underlying transaction must be retried and is no longer usable
157     */
158    void removeRange(byte[] minKey, byte[] maxKey);
159
160    /**
161     * Encode a counter value into a {@code byte[]} value suitable for use with {@link #decodeCounter decodeCounter()}
162     * and/or {@link #adjustCounter adjustCounter()}.
163     *
164     * @param value desired counter value
165     * @return encoded counter value
166     * @throws StaleTransactionException if an underlying transaction is no longer usable
167     * @throws RetryTransactionException if an underlying transaction must be retried and is no longer usable
168     */
169    byte[] encodeCounter(long value);
170
171    /**
172     * Decode a counter value previously encoded by {@link #encodeCounter encodeCounter()}.
173     *
174     * @param value encoded counter value
175     * @return decoded counter value
176     * @throws IllegalArgumentException if {@code value} is not a valid counter value
177     * @throws StaleTransactionException if an underlying transaction is no longer usable
178     * @throws RetryTransactionException if an underlying transaction must be retried and is no longer usable
179     * @throws NullPointerException if {@code value} is null
180     */
181    long decodeCounter(byte[] value);
182
183    /**
184     * Adjust the counter at the given key by the given amount.
185     *
186     * <p>
187     * Ideally this operation should behave in a lock-free manner, so that concurrent transactions can invoke it without
188     * conflict. However, when lock-free behavior occurs (if at all) depends on the implementation.
189     * </p>
190     *
191     * <p>
192     * If there is no value associated with {@code key}, or {@code key}'s value is not a valid counter encoding as
193     * would be acceptable to {@link #decodeCounter decodeCounter()}, then how this operation affects {@code key}'s
194     * value is undefined.
195     * </p>
196     *
197     * @param key key
198     * @param amount amount to adjust counter value by
199     * @throws StaleTransactionException if an underlying transaction is no longer usable
200     * @throws RetryTransactionException if an underlying transaction must be retried and is no longer usable
201     * @throws NullPointerException if {@code key} is null
202     */
203    void adjustCounter(byte[] key, long amount);
204}
