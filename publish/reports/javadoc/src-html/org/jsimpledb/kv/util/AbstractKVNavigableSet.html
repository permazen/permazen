006package org.jsimpledb.kv.util;
007
008import java.util.Iterator;
009import java.util.NavigableSet;
010
011import org.jsimpledb.kv.KVPair;
012import org.jsimpledb.kv.KVStore;
013import org.jsimpledb.kv.KeyFilter;
014import org.jsimpledb.kv.KeyFilterUtil;
015import org.jsimpledb.kv.KeyRange;
016import org.jsimpledb.util.AbstractNavigableSet;
017import org.jsimpledb.util.Bounds;
018import org.jsimpledb.util.ByteReader;
019import org.jsimpledb.util.ByteUtil;
020import org.jsimpledb.util.ByteWriter;

022/**
023 * {@link java.util.NavigableSet} support superclass for sets backed by elements encoded as {@code byte[]}
024 * keys in a {@link KVStore} and whose sort order is consistent with their {@code byte[]} key encoding.
025 * There must be an equivalence between elements and {@code byte[]} key encodings (i.e., there must be
026 * only one valid encoding per set element).
027 *
028 * <p><b>Subclass Methods</b></p>
029 *
030 * <p>
031 * Subclasses must implement the {@linkplain #encode(ByteWriter, Object) encode()} and {@linkplain #decode decode()}
032 * methods to convert elements to/from {@code byte[]} keys (associated values are ignored), and
033 * {@link #createSubSet(boolean, KeyRange, KeyFilter, Bounds) createSubSet()}
034 * to allow creating reversed and restricted range sub-sets.
035 * </p>
036 *
037 * <p>
038 * Subclasses must also implement {@link #comparator comparator()}, and the resulting sort order must be consistent with
039 * the sort order of the encoded {@code byte[]} keys (possibly {@link #reversed}).
040 * </p>
041 *
042 * <p>
043 * This class provides a read-only implementation; for a mutable implementation, subclasses should also implement
044 * {@link #add add()} (if appropriate), {@link #remove remove()}, and {@link #clear}; note, these methods must verify
045 * the key {@link #isVisible isVisible()} before making any changes.
046 * </p>
047 *
048 * <p>
049 * Additional subclass notes:
050 * <ul>
051 *  <li>{@link #iterator} returns an {@link Iterator} whose {@link Iterator#remove Iterator.remove()} method delegates
052 *      to this instance's {@link #remove remove()} method.
053 * </ul>
054 *
055 * <p><b>Prefix Mode</b></p>
056 *
057 * <p>
058 * Instances support a "prefix mode" where the {@code byte[]} keys may have arbitrary trailing garbage, which is ignored,
059 * and so by definition no key can be a prefix of any other key. The length of the prefix is determined implicitly by the
060 * number of bytes produced by {@link #encode encode()} or consumed by {@link #decode decode()}.
061 * When not in prefix mode, {@link #decode decode()} <b>must</b> consume the entire key to preserve correct semantics.
062 * </p>
063 *
064 * <p><b>Key Restrictions</b></p>
065 *
066 * <p>
067 * Instances are configured with an (optional) {@link KeyRange}; when {@linkplain #bounds range restriction} is in
068 * effect, this key range corresponds to the bounds.
069 * </p>
070 *
071 * <p>
072 * Instances also support filtering visible values using a {@link KeyFilter}; see {@link #filterKeys filterKeys()}.
073 * To be {@linkplain #isVisible} in the set, keys must both be in the {@link KeyRange} and pass the {@link KeyFilter}.
074 * </p>
075 *
076 * <p><b>Concurrent Modifications</b></p>
077 *
078 * <p>
079 * This implementation never throws {@link java.util.ConcurrentModificationException}; instead, iterators always
080 * see the most up-to-date state of the associated {@link KVStore}.
081 * </p>
082 *
083 * @see AbstractKVNavigableMap
084 * @param <E> element type
085 */
086public abstract class AbstractKVNavigableSet<E> extends AbstractNavigableSet<E> {
087
088    /**
089     * The underlying {@link KVStore}.
090     */
091    protected final KVStore kv;
092
093    /**
094     * Whether we are in "prefix" mode.
095     */
096    protected final boolean prefixMode;
097
098    /**
099     * Whether the ordering of this instance is reversed.
100     */
101    protected final boolean reversed;
102
103    /**
104     * Key range, or null for the entire range.
105     */
106    protected final KeyRange keyRange;
107
108    /**
109     * Key filter, or null if all keys in the key range should be visible.
110     */
111    protected final KeyFilter keyFilter;
112
113// Constructors
114
115    /**
116     * Convenience constructor for when there are no range restrictions.
117     *
118     * @param kv underlying {@link KVStore}
119     * @param prefixMode whether to allow keys to have trailing garbage
120     * @throws IllegalArgumentException if {@code kv} is null
121     */
122    protected AbstractKVNavigableSet(KVStore kv, boolean prefixMode) {
123        this(kv, prefixMode, (KeyRange)null);
124    }
125
126    /**
127     * Convenience constructor for when the range of visible {@link KVStore} keys is all keys sharing a given {@code byte[]} prefix.
128     *
129     * @param kv underlying {@link KVStore}
130     * @param prefixMode whether to allow keys to have trailing garbage
131     * @param prefix prefix defining minimum and maximum keys
132     * @throws IllegalArgumentException if {@code kv} is null
133     * @throws IllegalArgumentException if {@code prefix} is null or empty
134     */
135    protected AbstractKVNavigableSet(KVStore kv, boolean prefixMode, byte[] prefix) {
136        this(kv, prefixMode, KeyRange.forPrefix(prefix));
137    }
138
139    /**
140     * Primary constructor.
141     *
142     * @param kv underlying {@link KVStore}
143     * @param prefixMode whether to allow keys to have trailing garbage
144     * @param keyRange key range restriction, or null for none
145     * @throws IllegalArgumentException if {@code kv} is null
146     */
147    protected AbstractKVNavigableSet(KVStore kv, boolean prefixMode, KeyRange keyRange) {
148        this(kv, prefixMode, false, keyRange, null, new Bounds<E>());
149    }
150
151    /**
152     * Internal constructor. Used for creating sub-sets and reversed views.
153     *
154     * <p>
155     * Note: if {@code bounds} are set, then {@code keyRange} must exclude all keys outside of those bounds.
156     * </p>
157     *
158     * @param kv underlying {@link KVStore}
159     * @param prefixMode whether to allow keys to have trailing garbage
160     * @param reversed whether ordering is reversed (implies {@code bounds} are also inverted, but <i>not</i> {@code keyRange})
161     * @param keyRange key range restriction, or null for none
162     * @param keyFilter key filter, or null for none
163     * @param bounds range restriction
164     * @throws IllegalArgumentException if {@code kv} or {@code bounds} is null
165     */
166    protected AbstractKVNavigableSet(KVStore kv, boolean prefixMode, boolean reversed,
167      KeyRange keyRange, KeyFilter keyFilter, Bounds<E> bounds) {
168        super(bounds);
169        if (kv == null)
170            throw new IllegalArgumentException("null kv");
171        this.kv = kv;
172        this.prefixMode = prefixMode;
173        this.reversed = reversed;
174        this.keyRange = keyRange;
175        this.keyFilter = keyFilter;
176    }
177
178    @Override
179    @SuppressWarnings("unchecked")
180    public boolean contains(Object obj) {
181
182        // Encode key and check visibility
183        final byte[] key = this.encodeVisible(obj, false);
184        if (key == null)
185            return false;
186
187        // Find key, or some longer key with the same prefix in prefix mode
188        final KVPair pair;
189        if (this.prefixMode) {
190            pair = this.kv.getAtLeast(key);
191            return pair != null && ByteUtil.isPrefixOf(key, pair.getKey());
192        } else
193            return this.kv.get(key) != null;
194    }
195
196    @Override
197    public Iterator<E> iterator() {
198        return new AbstractKVIterator<E>(this.kv, this.prefixMode, this.reversed, this.keyRange, this.keyFilter) {
199
200            @Override
201            protected E decodePair(KVPair pair, ByteReader keyReader) {
202                return AbstractKVNavigableSet.this.decode(keyReader);
203            }
204
205            @Override
206            protected void doRemove(E value, KVPair pair) {
207                AbstractKVNavigableSet.this.remove(value);
208            }
209        };
210    }
211
212    /**
213     * Create a view of this instance with additional filtering applied to the underlying {@code byte[]} encoded keys.
214     * Any set element for which the corresponding key does not pass {@code keyFilter} will be effectively hidden from view.
215     *
216     * <p>
217     * The restrictions of the given {@link KeyFilter} will be added to any current {@link KeyFilter} restrictions on this instance.
218     * The {@link #bounds} associated with this instance will not change.
219     * </p>
220     *
221     * @param keyFilter additional key filtering to apply
222     * @return filtered view of this instance
223     * @throws IllegalArgumentException if {@code keyFilter} is null
224     */
225    public NavigableSet<E> filterKeys(KeyFilter keyFilter) {
226        if (keyFilter == null)
227            throw new IllegalArgumentException("null keyFilter");
228        if (this.keyFilter != null)
229            keyFilter = KeyFilterUtil.intersection(keyFilter, this.keyFilter);
230        return this.createSubSet(this.reversed, this.keyRange, keyFilter, this.bounds);
231    }
232
233    @Override
234    protected boolean isWithinLowerBound(E elem) {
235        if (!super.isWithinLowerBound(elem))
236            return false;
237        if (this.keyRange == null)
238            return true;
239        final ByteWriter writer = new ByteWriter();
240        this.encode(writer, elem);
241        return KeyRange.compare(writer.getBytes(), this.keyRange.getMin()) >= 0;
242    }
243
244    @Override
245    protected boolean isWithinUpperBound(E elem) {
246        if (!super.isWithinUpperBound(elem))
247            return false;
248        if (this.keyRange == null)
249            return true;
250        final ByteWriter writer = new ByteWriter();
251        this.encode(writer, elem);
252        return KeyRange.compare(writer.getBytes(), this.keyRange.getMax()) < 0;
253    }
254
255    @Override
256    protected final NavigableSet<E> createSubSet(boolean reverse, Bounds<E> newBounds) {
257
258        // Determine the direction of the new sub-set
259        final boolean newReversed = this.reversed ^ reverse;
260
261        // Determine new min and max keys
262        final KeyRange newKeyRange = this.buildKeyRange(newReversed ? newBounds.reverse() : newBounds);
263
264        // Create subset
265        return this.createSubSet(newReversed, newKeyRange, this.keyFilter, newBounds);
266    }
267
268    /**
269     * Create a (possibly reversed) view of this instance with (possibly) tighter lower and/or upper bounds and
270     * the given {@link KeyFilter}, if any.
271     * The bounds are consistent with the reversed ordering (i.e., reversed if {@code reverse} is true)
272     * and have already been range-checked against this instance's bounds.
273     *
274     * @param newReversed whether the new set's ordering should be reversed (implies {@code newBounds} are also inverted,
275     *  but <i>not</i> {@code keyRange}); note: means "absolutely" reversed, not relative to this instance
276     * @param newKeyRange new key range, or null for none; will be consistent with {@code bounds}, if any
277     * @param newKeyFilter new key filter, or null for none
278     * @param newBounds new bounds
279     * @return restricted and/or filtered view of this instance
280     * @throws IllegalArgumentException if {@code newBounds} is null
281     */
282    protected abstract NavigableSet<E> createSubSet(boolean newReversed,
283      KeyRange newKeyRange, KeyFilter newKeyFilter, Bounds<E> newBounds);
284
285    /**
286     * Encode the given object into a {@code byte[]} key.
287     * Note that this method must throw {@link IllegalArgumentException}, not {@link ClassCastException}
288     * or {@code NullPointerException}, if {@code obj} does not have the correct type or is an illegal null value.
289     *
290     * @param writer output for encoded {@code byte[]} key corresponding to {@code obj}
291     * @param obj set element
292     * @throws IllegalArgumentException if {@code obj} is not of the required Java type supported by this set
293     * @throws IllegalArgumentException if {@code obj} is null and this set does not support null elements
294     */
295    protected abstract void encode(ByteWriter writer, Object obj);
296
297    /**
298     * Decode an element from a {@code byte[]} key.
299     *
300     * <p>
301     * If not in prefix mode, all of {@code reader} must be consumed; otherwise, the consumed portion
302     * is the prefix and any following keys with the same prefix are ignored.
303     * </p>
304     *
305     * @param reader input for encoded bytes
306     * @return decoded set element
307     */
308    protected abstract E decode(ByteReader reader);
309
310    /**
311     * Determine if the given {@code byte[]} key is visible in this set according to the configured
312     * {@link KeyRange} and/or {@link KeyFilter}, if any.
313     *
314     * @param key key to test
315     * @return true if key is visible
316     * @throws IllegalArgumentException if {@code key} is null
317     * @see #filterKeys filterKeys()
318     */
319    protected boolean isVisible(byte[] key) {
320        return (this.keyRange == null || this.keyRange.contains(key))
321          && (this.keyFilter == null || this.keyFilter.contains(key));
322    }
323
324    /**
325     * Encode the given object, if possible, and verify corresponding {@code byte[]} key is visible,
326     * otherwise return null or throw an exception.
327     * Delegates to {@link #encode(ByteWriter, Object)} to attempt the actual encoding.
328     *
329     * @param obj object to encode, possibly null
330     * @param fail whether, if {@code obj} can't be encoded, to throw an exception (true) or return null (false)
331     * @return encoed key for {@code obj}, or null if {@code fail} is false and {@code obj} has the wrong type or is out of bounds
332     * @throws IllegalArgumentException if {@code fail} is true and {@code obj} has the wrong type
333     * @throws IllegalArgumentException if {@code fail} is true and the resulting key is not {@linkplain #isVisible visible}
334     */
335    protected byte[] encodeVisible(Object obj, boolean fail) {
336        final ByteWriter writer = new ByteWriter();
337        try {
338            this.encode(writer, obj);
339        } catch (IllegalArgumentException e) {
340            if (!fail)
341                return null;
342            throw e;
343        }
344        final byte[] key = writer.getBytes();
345        if (this.keyRange != null && !this.keyRange.contains(key)) {
346            if (fail)
347                throw new IllegalArgumentException("value is out of bounds: " + obj);
348            return null;
349        }
350        if (this.keyFilter != null && !this.keyFilter.contains(key)) {
351            if (fail)
352                throw new IllegalArgumentException("value is filtered out: " + obj);
353            return null;
354        }
355        return key;
356    }
357
358    /**
359     * Derive a new {@link KeyRange} from (possibly) new element bounds. The given bounds must <i>not</i> ever be reversed.
360     */
361    private KeyRange buildKeyRange(Bounds<E> bounds) {
362        final byte[] minKey = this.keyRange != null ? this.keyRange.getMin() : null;
363        final byte[] maxKey = this.keyRange != null ? this.keyRange.getMax() : null;
364        byte[] newMinKey;
365        byte[] newMaxKey;
366        switch (bounds.getLowerBoundType()) {
367        case NONE:
368            newMinKey = minKey;
369            break;
370        default:
371            final ByteWriter writer = new ByteWriter();
372            this.encode(writer, bounds.getLowerBound());
373            newMinKey = writer.getBytes();
374            if (!bounds.getLowerBoundType().isInclusive())
375                newMinKey = this.prefixMode ? ByteUtil.getKeyAfterPrefix(newMinKey) : ByteUtil.getNextKey(newMinKey);
376            if (minKey != null)
377                newMinKey = ByteUtil.max(newMinKey, minKey);
378            if (maxKey != null)
379                newMinKey = ByteUtil.min(newMinKey, maxKey);
380            break;
381        }
382        switch (bounds.getUpperBoundType()) {
383        case NONE:
384            newMaxKey = maxKey;
385            break;
386        default:
387            final ByteWriter writer = new ByteWriter();
388            this.encode(writer, bounds.getUpperBound());
389            newMaxKey = writer.getBytes();
390            if (bounds.getUpperBoundType().isInclusive())
391                newMaxKey = this.prefixMode ? ByteUtil.getKeyAfterPrefix(newMaxKey) : ByteUtil.getNextKey(newMaxKey);
392            if (maxKey != null)
393                newMaxKey = ByteUtil.min(newMaxKey, maxKey);
394            if (minKey != null)
395                newMaxKey = ByteUtil.max(newMaxKey, minKey);
396            break;
397        }
398        return new KeyRange(newMinKey != null ? newMinKey : ByteUtil.EMPTY, newMaxKey);
399    }
