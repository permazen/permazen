006package org.jsimpledb.kv.util;
007
008import java.util.Iterator;
009
010import org.jsimpledb.kv.KVPair;
011import org.jsimpledb.kv.KVPairIterator;
012import org.jsimpledb.kv.KVStore;
013import org.jsimpledb.kv.KeyFilter;
014import org.jsimpledb.kv.KeyRange;
015import org.jsimpledb.util.ByteReader;
016import org.jsimpledb.util.ByteUtil;
017import org.slf4j.LoggerFactory;
018
019/**
020 * {@link Iterator} implementation whose values derive from key/value {@code byte[]} pairs in a {@link KVStore}.
021 * Instances support either forward or reverse iteration.
022 *
023 * <p><b>Subclass Methods</b></p>
024 *
025 * <p>
026 * Subclasses must implement {@linkplain #decodePair decodePair()} to convert key/value pairs into iteration elements.
027 * </p>
028 *
029 * <p>
030 * This class provides a read-only implementation; for a mutable implementation, subclasses should also implement
031 * {@link #doRemove doRemove()}.
032 * </p>
033 *
034 * <p><b>Prefix Mode</b></p>
035 *
036 * <p>
037 * Instances support "prefix mode" where the {@code byte[]} keys may have arbitrary trailing garbage, which is ignored,
038 * and so by definition no key can be a prefix of any other key. The length of the prefix is determined implicitly by the
039 * number of key bytes consumed by {@link #decodePair decodePair()}.
040 * When not in prefix mode, {@link #decodePair decodePair()} <b>must</b> consume the entire key to preserve correct semantics.
041 * </p>
042 *
043 * <p><b>Key Restrictions</b></p>
044 *
045 * <p>
046 * Instances are configured with an (optional) {@link KeyRange} that restricts the iteration to the specified range.
047 * </p>
048 *
049 * <p>
050 * Instances also support filtering visible values using a {@link KeyFilter}.
051 * To be visible in the iteration, keys must both be within the {@link KeyRange} and pass the {@link KeyFilter}.
052 * </p>
053 *
054 * <p><b>Concurrent Modification</b></p>
055 *
056 * <p>
057 * Instances of this class are thread safe.
058 * </p>
059 *
060 * <p>
061 * Internally, when not in prefix mode and no {@link KeyFilter} is configured, this instance will rely on the
062 * iteration from {@link KVStore#getRange KVStore.getRange()}; otherwise, it will use a {@link KVPairIterator}.
063 * Therefore, in the former case, whether this iteration always reflects the current state of the underlying
064 * {@link KVStore} depends on the behavior of {@link KVStore#getRange KVStore.getRange()}.
065 * </p>
066 *
067 * @see AbstractKVNavigableMap
068 * @see AbstractKVNavigableSet
069 * @param <E> iteration element type
070 */
071public abstract class AbstractKVIterator<E> implements java.util.Iterator<E> {
072
073    /**
074     * The underlying {@link KVStore}.
075     */
076    protected final KVStore kv;
077
078    /**
079     * Whether we are in "prefix" mode.
080     */
081    protected final boolean prefixMode;
082
083    /**
084     * Whether this instance is iterating in the reverse direction.
085     */
086    protected final boolean reversed;
087
088    // Iteration state
089    private final Iterator<KVPair> pairIterator;
090    private KVPair removePair;
091    private E removeValue;
092
093// Constructors
094
095    /**
096     * Convenience constructor for when there are no range restrictions.
097     *
098     * @param kv underlying {@link KVStore}
099     * @param prefixMode whether to allow keys to have trailing garbage
100     * @param reversed whether to iterate in the reverse direction
101     */
102    protected AbstractKVIterator(KVStore kv, boolean prefixMode, boolean reversed) {
103        this(kv, prefixMode, reversed, null, null);
104    }
105
106    /**
107     * Convenience constructor for when the range of visible {@link KVStore} keys is all keys sharing a given {@code byte[]} prefix.
108     *
109     * @param kv underlying {@link KVStore}
110     * @param prefixMode whether to allow keys to have trailing garbage
111     * @param reversed whether to iterate in the reverse direction
112     * @param prefix prefix defining minimum and maximum keys
113     * @throws IllegalArgumentException if {@code prefix} is null or empty
114     */
115    protected AbstractKVIterator(KVStore kv, boolean prefixMode, boolean reversed, byte[] prefix) {
116        this(kv, prefixMode, reversed, KeyRange.forPrefix(prefix), null);
117    }
118
119    /**
120     * Primary constructor.
121     *
122     * @param kv underlying {@link KVStore}
123     * @param prefixMode whether to allow keys to have trailing garbage
124     * @param reversed whether to iterate in the reverse direction
125     * @param keyRange key range restriction, or null for none
126     * @param keyFilter key filter, or null for none
127     * @throws IllegalArgumentException if {@code kv} is null
128     */
129    protected AbstractKVIterator(KVStore kv, boolean prefixMode, boolean reversed, KeyRange keyRange, KeyFilter keyFilter) {
130        if (kv == null)
131            throw new IllegalArgumentException("null kv");
132        this.kv = kv;
133        this.prefixMode = prefixMode;
134        this.reversed = reversed;
135
136        // If possible use a straight KVStore iterator which is more efficient than a KVPairIterator
137        if (!this.prefixMode && keyFilter == null) {
138            final byte[] minKey = keyRange != null ? keyRange.getMin() : null;
139            final byte[] maxKey = keyRange != null ? keyRange.getMax() : null;
140            this.pairIterator = this.kv.getRange(minKey, maxKey, this.reversed);
141        } else
142            this.pairIterator = new KVPairIterator(this.kv, keyRange, keyFilter, this.reversed);
143    }
144
145// Iterator
146
147    @Override
148    public boolean hasNext() {
149        return this.pairIterator.hasNext();
150    }
151
152    @Override
153    public synchronized E next() {
154
155        // Get next key/value pair
156        final KVPair pair = this.pairIterator.next();
157
158        // Decode key/value pair
159        final ByteReader keyReader = new ByteReader(pair.getKey());
160        final E value = this.decodePair(pair, keyReader);
161        if (!this.prefixMode && keyReader.remain() > 0) {
162            LoggerFactory.getLogger(this.getClass()).error(this.getClass().getName() + "@"
163              + Integer.toHexString(System.identityHashCode(this)) + ": " + keyReader.remain() + " undecoded bytes remain in key "
164              + ByteUtil.toString(pair.getKey()) + ", value " + ByteUtil.toString(pair.getValue()) + " -> " + value);
165        }
166
167        // In prefix mode, skip over any additional keys having the same prefix as what we just decoded
168        if (this.prefixMode) {
169            final KVPairIterator kvPairIterator = (KVPairIterator)this.pairIterator;
170            final byte[] prefix = keyReader.getBytes(0, keyReader.getOffset());
171            kvPairIterator.setNextTarget(this.reversed ? prefix : ByteUtil.getKeyAfterPrefix(prefix));
172        }
173
174        // Done
175        this.removePair = pair;
176        this.removeValue = value;
177        return value;
178    }
179
180    @Override
181    public void remove() {
182        final E removeValueCopy;
183        final KVPair removePairCopy;
184        synchronized (this) {
185            if ((removePairCopy = this.removePair) == null)
186                throw new IllegalStateException();
187            this.removePair = null;
188            removeValueCopy = this.removeValue;
189        }
190        this.doRemove(removeValueCopy, removePairCopy);
191    }
192
193// Subclass methods
194
195    /**
196     * Decode an iteration element from a key/value pair.
197     *
198     * <p>
199     * If not in prefix mode, all of {@code keyReader} must be consumed; otherwise, the consumed portion
200     * is the prefix and any following keys with the same prefix are ignored.
201     * </p>
202     *
203     * @param pair key/value pair
204     * @param keyReader key input
205     * @return decoded iteration element
206     */
207    protected abstract E decodePair(KVPair pair, ByteReader keyReader);
208
209    /**
210     * Remove the previously iterated value.
211     *
212     * <p>
213     * The implementation in {@link AbstractKVIterator} always throws {@link UnsupportedOperationException}.
214     * Subclasses should override to make the iterator mutable.
215     * </p>
216     *
217     * @param value most recent value returned by {@link #next}
218     * @param pair the key/value pair corresponding to {@code value}
219     */
220    protected void doRemove(E value, KVPair pair) {
221        throw new UnsupportedOperationException();
222    }
223}
