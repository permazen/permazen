006package org.jsimpledb.kv.util;
007
008import org.jsimpledb.kv.KVTransaction;
009
010/**
011 * {@link KVTransaction} view of all keys having a common {@code byte[]} prefix in a containing {@link KVTransaction}.
012 *
013 * <p>
014 * Instances are normally created indirectly from {@link PrefixKVDatabase} instances via {@link PrefixKVDatabase#createTransaction}.
015 * Instances may also be created directly from an existing {@link KVTransaction}; in that case,
016 * {@link #setTimeout setTimeout()}, {@link #commit}, and {@link #rollback} forward to the containing transaction,
017 * while {@link #getKVDatabase} throws {@link UnsupportedOperationException}.
018 * </p>
019 */
020public class PrefixKVTransaction extends PrefixKVStore implements KVTransaction {
021
022    private final KVTransaction tx;
023    private final PrefixKVDatabase db;
024
025// Constructors
026
027    /**
028     * Constructor that wraps an existing {@link KVTransaction}. There will be no associated {@link PrefixKVDatabase}.
029     *
030     * @param tx the containing {@link KVTransaction}
031     * @param keyPrefix prefix for all keys
032     * @throws IllegalArgumentException if {@code tx} or {@code keyPrefix} is null
033     */
034    public PrefixKVTransaction(KVTransaction tx, byte[] keyPrefix) {
035        this(tx, keyPrefix, null);
036    }
037
038    /**
039     * Constructor for when there is an associated {@link PrefixKVDatabase}.
040     *
041     * @param db the containing {@link PrefixKVDatabase}
042     * @throws NullPointerException if {@code db} is null
043     */
044    PrefixKVTransaction(PrefixKVDatabase db) {
045        this(db.getContainingKVDatabase().createTransaction(), db.getKeyPrefix(), db);
046    }
047
048    private PrefixKVTransaction(KVTransaction tx, byte[] keyPrefix, PrefixKVDatabase db) {
049        super(keyPrefix);
050        if (tx == null)
051            throw new IllegalArgumentException("null tx");
052        this.tx = tx;
053        this.db = db;
054    }
055
056// PrefixKVStore
057
058    @Override
059    protected KVTransaction delegate() {
060        return this.tx;
061    }
062
063// KVTransaction
064
065    /**
066     * Get the {@link PrefixKVDatabase} associated with this instance.
067     *
068     * @throws UnsupportedOperationException if this instance was created directly from a containing {@link KVTransaction}
069     */
070    @Override
071    public PrefixKVDatabase getKVDatabase() {
072        if (this.db == null)
073            throw new UnsupportedOperationException("instance was not created from a PrefixKVDatabase");
074        return this.db;
075    }
076
077    @Override
078    public void setTimeout(long timeout) {
079        this.delegate().setTimeout(timeout);
080    }
081
082    @Override
083    public void commit() {
084        this.delegate().commit();
085    }
086
087    @Override
088    public void rollback() {
089        this.delegate().rollback();
090    }
091}
