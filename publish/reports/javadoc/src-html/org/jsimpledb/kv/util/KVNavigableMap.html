006package org.jsimpledb.kv.util;
007
008import java.util.Collections;
009import java.util.Comparator;
010import java.util.NavigableMap;
011
012import org.jsimpledb.kv.KVPair;
013import org.jsimpledb.kv.KVStore;
014import org.jsimpledb.kv.KeyFilter;
015import org.jsimpledb.kv.KeyRange;
016import org.jsimpledb.util.Bounds;
017import org.jsimpledb.util.ByteReader;
018import org.jsimpledb.util.ByteUtil;
019import org.jsimpledb.util.ByteWriter;
020
021/**
022 * A {@link java.util.NavigableMap} view of the keys and values in a {@link KVStore}.
023 *
024 * Instances are mutable, with these exceptions:
025 * <ul>
026 *  <li>{@link #clear} is not supported when a {@link KeyFilter} is configured</li>
027 * </ul>
028 */
029@SuppressWarnings("serial")
030public class KVNavigableMap extends AbstractKVNavigableMap<byte[], byte[]> {
031
032// Constructors
033
034    /**
035     * Constructor for when there are no range restrictions.
036     *
037     * @param kv underlying {@link KVStore}
038     */
039    public KVNavigableMap(KVStore kv) {
040        this(kv, false, null, null);
041    }
042
043    /**
044     * Constructor for when the range of visible {@link KVStore} keys is all keys sharing a given {@code byte[]} prefix.
045     *
046     * @param kv underlying {@link KVStore}
047     * @param prefix prefix defining minimum and maximum keys
048     * @throws NullPointerException if {@code prefix} is null
049     */
050    public KVNavigableMap(KVStore kv, byte[] prefix) {
051        this(kv, false, KeyRange.forPrefix(prefix), null);
052    }
053
054    /**
055     * Primary constructor.
056     *
057     * @param kv underlying {@link KVStore}
058     * @param reversed whether ordering is reversed
059     * @param keyRange key range restriction, or null for none
060     * @param keyFilter key filter, or null for none
061     * @throws IllegalArgumentException if {@code kv} is null
062     */
063    protected KVNavigableMap(KVStore kv, boolean reversed, KeyRange keyRange, KeyFilter keyFilter) {
064        this(kv, reversed, keyRange, keyFilter, KVNavigableSet.createBounds(keyRange));
065    }
066
067    /**
068     * Internal constructor. Used for creating sub-maps and reversed views.
069     *
070     * @param kv underlying {@link KVStore}
071     * @param reversed whether ordering is reversed (implies {@code bounds} are also inverted)
072     * @param keyRange key range restriction, or null for none
073     * @param keyFilter key filter, or null for none
074     * @param bounds range restriction
075     * @throws IllegalArgumentException if {@code kv} or {@code bounds} is null
076     */
077    private KVNavigableMap(KVStore kv, boolean reversed, KeyRange keyRange, KeyFilter keyFilter, Bounds<byte[]> bounds) {
078        super(kv, false, reversed, keyRange, keyFilter, bounds);
079    }
080
081// Methods
082
083    @Override
084    public Comparator<byte[]> comparator() {
085        return this.reversed ? Collections.reverseOrder(ByteUtil.COMPARATOR) : ByteUtil.COMPARATOR;
086    }
087
088    @Override
089    public byte[] put(byte[] key, byte[] value) {
090        if (!this.isVisible(key))
091            throw new IllegalArgumentException("key is out of range or filtered out");
092        final byte[] previousValue = this.kv.get(key);
093        this.kv.put(key, value);
094        return previousValue;
095    }
096
097    @Override
098    public byte[] remove(Object obj) {
099        if (!(obj instanceof byte[]))
100            return null;
101        final byte[] key = (byte[])obj;
102        if (!this.isVisible(key))
103            return null;
104        final byte[] previousValue = this.kv.get(key);
105        if (previousValue != null)
106            this.kv.remove(key);
107        return previousValue;
108    }
109
110    @Override
111    public void clear() {
112        this.navigableKeySet().clear();
113    }
114
115    @Override
116    public boolean containsKey(Object obj) {
117        if (!(obj instanceof byte[]))
118            return false;
119        final byte[] key = (byte[])obj;
120        if (!this.isVisible(key))
121            return false;
122        return this.kv.get(key) != null;
123    }
124
125    @Override
126    protected void encodeKey(ByteWriter writer, Object obj) {
127        if (!(obj instanceof byte[]))
128            throw new IllegalArgumentException("key is not a byte[]");
129        writer.write((byte[])obj);
130    }
131
132    @Override
133    protected byte[] decodeKey(ByteReader reader) {
134        return reader.getBytes();
135    }
136
137    @Override
138    protected byte[] decodeValue(KVPair pair) {
139        return pair.getValue();
140    }
141
142    @Override
143    protected NavigableMap<byte[], byte[]> createSubMap(boolean newReversed,
144      KeyRange newKeyRange, KeyFilter newKeyFilter, Bounds<byte[]> newBounds) {
145        return new KVNavigableMap(this.kv, newReversed, newKeyRange, newKeyFilter, newBounds);
146    }
147}
