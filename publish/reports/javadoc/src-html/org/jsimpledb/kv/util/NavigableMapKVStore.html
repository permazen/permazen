006package org.jsimpledb.kv.util;
007
008import com.google.common.base.Function;
009import com.google.common.collect.Iterators;
010
011import java.util.Iterator;
012import java.util.Map;
013import java.util.NavigableMap;
014import java.util.concurrent.ConcurrentSkipListMap;
015
016import org.jsimpledb.kv.AbstractKVStore;
017import org.jsimpledb.kv.KVPair;
018import org.jsimpledb.util.ByteUtil;
019
020/**
021 * Provides a {@link org.jsimpledb.kv.KVStore} view of an underlying {@link NavigableMap NavigableMap&lt;byte[], byte[]&gt;}
022 * whose keys are sorted lexicographically as unsigned bytes.
023 */
024public class NavigableMapKVStore extends AbstractKVStore {
025
026    private final NavigableMap<byte[], byte[]> map;
027
028    /**
029     * Convenience constructor. Uses an internally constructed {@link ConcurrentSkipListMap}.
030     *
031     * <p>
032     * Equivalent to:
033     * <blockquote><code>
034     * NavigableMapKVStore(new ConcurrentSkipListMap&lt;byte[], byte[]&gt;(ByteUtil.COMPARATOR)
035     * </code></blockquote>
036     *
037     * @see org.jsimpledb.util.ByteUtil#COMPARATOR
038     */
039    public NavigableMapKVStore() {
040        this(new ConcurrentSkipListMap<byte[], byte[]>(ByteUtil.COMPARATOR));
041    }
042
043    /**
044     * Primary constructor.
045     *
046     * <p>
047     * The underlying map <b>must</b> sort keys lexicographically as unsigned bytes; otherwise, behavior is undefined.
048     * </p>
049     *
050     * <p>
051     * The underlying map's iterator <b>must not</b> throw {@link java.util.ConcurrentModificationException}. For example,
052     * {@link java.util.TreeMap} does not satisfy this constraint.
053     * </p>
054     *
055     * @param map underlying map
056     * @throws IllegalArgumentException if {@code map} is null
057     * @see org.jsimpledb.util.ByteUtil#COMPARATOR
058     */
059    public NavigableMapKVStore(NavigableMap<byte[], byte[]> map) {
060        if (map == null)
061            throw new IllegalArgumentException("null map");
062        this.map = map;
063    }
064
065    /**
066     * Get the underlying {@link NavigableMap} used by this instance.
067     *
068     * @return the underlying map
069     */
070    public NavigableMap<byte[], byte[]> getNavigableMap() {
071        return this.map;
072    }
073
074// KVStore
075
076    @Override
077    public byte[] get(byte[] key) {
078        if (key == null)
079            throw new NullPointerException("null key");
080        return this.map.get(key);
081    }
082
083    @Override
084    public KVPair getAtLeast(byte[] minKey) {
085        final Map.Entry<byte[], byte[]> entry = minKey != null ? this.map.ceilingEntry(minKey) : this.map.firstEntry();
086        return entry != null ? new KVPair(entry) : null;
087    }
088
089    @Override
090    public KVPair getAtMost(byte[] maxKey) {
091        final Map.Entry<byte[], byte[]> entry = maxKey != null ? this.map.lowerEntry(maxKey) : this.map.lastEntry();
092        return entry != null ? new KVPair(entry) : null;
093    }
094
095    @Override
096    public Iterator<KVPair> getRange(byte[] minKey, byte[] maxKey, boolean reverse) {
097        NavigableMap<byte[], byte[]> rangeMap = this.map;
098        if (minKey != null && maxKey != null)
099            rangeMap = rangeMap.subMap(minKey, true, maxKey, false);
100        else if (minKey != null)
101            rangeMap = rangeMap.tailMap(minKey, true);
102        else if (maxKey != null)
103            rangeMap = rangeMap.headMap(maxKey, false);
104        if (reverse)
105            rangeMap = rangeMap.descendingMap();
106        return Iterators.transform(rangeMap.entrySet().iterator(), new Function<Map.Entry<byte[], byte[]>, KVPair>() {
107            @Override
108            public KVPair apply(Map.Entry<byte[], byte[]> entry) {
109                return new KVPair(entry.getKey(), entry.getValue());
110            }
111        });
112    }
113
114    @Override
115    public void put(byte[] key, byte[] value) {
116        if (key == null)
117            throw new NullPointerException("null key");
118        if (value == null)
119            throw new NullPointerException("null value");
120        this.map.put(key, value);
121    }
122
123    @Override
124    public void remove(byte[] key) {
125        if (key == null)
126            throw new NullPointerException("null key");
127        this.map.remove(key);
128    }
129
130    @Override
131    public void removeRange(byte[] minKey, byte[] maxKey) {
132        if (minKey == null && maxKey == null)
133            this.map.clear();
134        else if (minKey == null)
135            this.map.headMap(maxKey).clear();
136        else if (maxKey == null)
137            this.map.tailMap(minKey).clear();
138        else
139            this.map.subMap(minKey, maxKey).clear();
140    }
141}
