028/**
029 * {@link java.util.NavigableMap} support superclass for maps backed by keys and values encoded as {@code byte[]}
030 * arrays in a {@link KVStore}, and whose key sort order is consistent with the {@code byte[]} key encoding.
031 * There must be an equivalence between map keys and {@code byte[]} key encodings (i.e., there must be
032 * only one valid encoding per map key).
033 *
034 * <p><b>Subclass Methods</b></p>
035 *
036 * <p>
037 * Subclasses must implement the {@linkplain #encodeKey encodeKey()}, {@linkplain #decodeKey decodeKey()},
038 * and {@linkplain #decodeValue decodeValue()} methods to convert keys and value to/from {@link KVStore} keys and values,
039 * and {@link #createSubMap(boolean, KeyRange, KeyFilter, Bounds) createSubMap()}
040 * to allow creating reversed and restricted range sub-maps.
041 *
042 * <p>
043 * Subclasses must also implement {@link #comparator comparator()}, and the resulting sort order must be consistent with
044 * the sort order of the encoded {@code byte[]} keys (possibly {@link #reversed}).
045 * </p>
046 *
047 * <p>
048 * This class provides a read-only implementation; for a mutable implementation, subclasses should also implement
049 * {@link #put put()}, {@link #remove remove()}, and {@link #clear}; note, these methods must verify the key
050 * {@link #isVisible isVisible()} before making any changes.
051 * </p>
052 *
053 * <p>
054 * Additional subclass notes:
055 * <ul>
056 *  <li>{@link #navigableKeySet} returns a {@link Set} for which:
057 *  <ul>
058 *      <li>The {@link Set#add add()} method is not supported.</li>
059 *      <li>The {@link Set#remove remove()} method and the {@link Iterator#remove remove()} method of the
060 *          associated {@link Iterator} delegate to this instance's {@link #remove remove()} method.</li>
061 *      <li>The {@link Set#clear clear()} method delegates to this instance's {@link #clear} method.</li>
062 *  </ul>
063 *  <li>{@link #entrySet} returns a {@link Set} for which:
064 *  <ul>
065 *      <li>The {@link Set#add add()} method is not supported.</li>
066 *      <li>The {@link Set#remove remove()} method and the {@link Iterator#remove remove()} method of the
067 *          associated {@link Iterator} delegate to this instance's {@link #remove remove()} method (but
068 *          only when the {@link java.util.Map.Entry} is still contained in the map).
069 *      <li>The {@link java.util.Map.Entry} elements' {@link java.util.Map.Entry#setValue setValue()} method delegates to
070 *          this instance's {@link #put put()} method (but only when the {@link java.util.Map.Entry} is still contained in the map).
071 *      <li>The {@link Set#clear clear()} method delegates to this instance's {@link #clear} method.</li>
072 *  </ul>
073 * </ul>
074 *
075 * <p><b>Prefix Mode</b></p>
076 *
077 * <p>
078 * Instances support "prefix mode" where the {@code byte[]} keys may have arbitrary trailing garbage, which is ignored,
079 * and so by definition no key can be a prefix of any other key. The length of the prefix is determined implicitly by the
080 * number of bytes produced by {@link #encodeKey encodeKey()} or consumed by {@link #decodeKey decodeKey()}.
081 * When not in prefix mode, {@link #decodeKey decodeKey()} <b>must</b> consume the entire key to preserve correct semantics.
082 * </p>
083 *
084 * <p><b>Key Restrictions</b></p>
085 *
086 * <p>
087 * Instances are configured with an (optional) {@link KeyRange}; when {@linkplain #bounds range restriction} is in
088 * effect, this key range corresponds to the bounds.
089 * </p>
090 *
091 * <p>
092 * Instances also support filtering visible keys using a {@link KeyFilter}; see {@link #filterKeys filterKeys()}.
093 * To be {@linkplain #isVisible} in the map, keys must both be in the {@link KeyRange} and pass the {@link KeyFilter}.
094 * </p>
095 *
096 * <p><b>Concurrent Modifications</b></p>
097 *
098 * <p>
099 * This implementation never throws {@link java.util.ConcurrentModificationException}; instead, iterators always
100 * see the most up-to-date state of the associated {@link KVStore}.
101 * </p>
102 *
103 * @see AbstractKVNavigableSet
104 * @param <K> key type
105 * @param <V> value type
106 */
107@SuppressWarnings("serial")
108public abstract class AbstractKVNavigableMap<K, V> extends AbstractNavigableMap<K, V> {
109
110    /**
111     * The underlying {@link KVStore}.
112     */
113    protected final KVStore kv;
114
115    /**
116     * Whether we are in "prefix" mode.
117     */

119
120    /**
121     * Whether the ordering of this instance is reversed.
122     */

124
125    /**
126     * Key range, or null for the entire range.

135// Constructors
136
137    /**
138     * Convenience constructor for when there are no range restrictions.
139     *
140     * @param kv underlying {@link KVStore}
141     * @param prefixMode whether to allow keys to have trailing garbage
142     * @throws IllegalArgumentException if {@code kv} is null
143     */

145        this(kv, prefixMode, (KeyRange)null);
146    }

148    /**
149     * Convenience constructor for when the range of visible {@link KVStore} keys is all keys sharing a given {@code byte[]} prefix.
150     *
151     * @param kv underlying {@link KVStore}
152     * @param prefixMode whether to allow keys to have trailing garbage
153     * @param prefix prefix defining minimum and maximum keys
154     * @throws IllegalArgumentException if {@code kv} is null
155     * @throws IllegalArgumentException if {@code prefix} is null or empty
156     */
157    protected AbstractKVNavigableMap(KVStore kv, boolean prefixMode, byte[] prefix) {
158        this(kv, prefixMode, KeyRange.forPrefix(prefix));
159    }
160
161    /**
162     * Primary constructor.
163     *
164     * @param kv underlying {@link KVStore}
165     * @param prefixMode whether to allow keys to have trailing garbage
166     * @param keyRange key range restriction, or null for none
167     * @throws IllegalArgumentException if {@code kv} is null
168     */
169    protected AbstractKVNavigableMap(KVStore kv, boolean prefixMode, KeyRange keyRange) {
170        this(kv, prefixMode, false, keyRange, null, new Bounds<K>());
171    }
172
173    /**
174     * Internal constructor. Used for creating sub-maps and reversed views.
175     *
176     * <p>
177     * Note: if {@code bounds} are set, then {@code keyRange} must exclude all keys outside of those bounds.
178     * </p>
179     *
180     * @param kv underlying {@link KVStore}
181     * @param prefixMode whether to allow keys to have trailing garbage
182     * @param reversed whether ordering is reversed (implies {@code bounds} are also inverted, but <i>not</i> {@code keyRange})
183     * @param keyRange key range restriction, or null for none
184     * @param keyFilter key filter, or null for none
185     * @param bounds range restriction
186     * @throws IllegalArgumentException if {@code kv} or {@code bounds} is null
187     */
188    protected AbstractKVNavigableMap(KVStore kv, boolean prefixMode, boolean reversed,
189      KeyRange keyRange, KeyFilter keyFilter, Bounds<K> bounds) {
190        super(bounds);
191        if (kv == null)
192            throw new IllegalArgumentException("null kv");
193        this.kv = kv;
194        this.prefixMode = prefixMode;
195        this.reversed = reversed;
196        this.keyRange = keyRange;
197        this.keyFilter = keyFilter;
198    }
199
200    @Override
201    public V get(Object obj) {
202
203        // Encode key and check visibility
204        final byte[] key = this.encodeVisibleKey(obj, false);
205        if (key == null)
206            return null;
207
208        // Find key, or some longer key with the same prefix in prefix mode
209        final KVPair pair;
210        if (this.prefixMode) {
211            pair = this.kv.getAtLeast(key);
212            if (pair == null || !ByteUtil.isPrefixOf(key, pair.getKey()))
213                return null;
214        } else {
215            final byte[] value = this.kv.get(key);
216            if (value == null)
217                return null;
218            pair = new KVPair(key, value);
219        }
220
221        // Decode value
222        return this.decodeValue(pair);
223    }
224
225    @Override
226    public Set<Map.Entry<K, V>> entrySet() {
227        return this.new EntrySet();
228    }
229
230    @Override
231    public NavigableSet<K> navigableKeySet() {
232        return new KeySet();
233    }

235    /**
236     * Create a view of this instance with additional filtering applied to the underlying {@code byte[]} keys.
237     * Any map entry for which the corresponding key does not pass {@code keyFilter} will be effectively hidden from view.
238     *
239     * <p>
240     * The restrictions of the given {@link KeyFilter} will be added to any current {@link KeyFilter} restrictions on this instance.
241     * The {@link #bounds} associated with this instance will not change.
242     * </p>
243     *
244     * @param keyFilter additional key filtering to apply
245     * @return filtered view of this instance
246     * @throws IllegalArgumentException if {@code keyFilter} is null
247     */
248    public NavigableMap<K, V> filterKeys(KeyFilter keyFilter) {
249        if (keyFilter == null)
250            throw new IllegalArgumentException("null keyFilter");
251        if (this.keyFilter != null)
252            keyFilter = KeyFilterUtil.intersection(keyFilter, this.keyFilter);
253        return this.createSubMap(this.reversed, this.keyRange, keyFilter, this.bounds);
254    }
255
256    @Override
257    protected boolean isWithinLowerBound(K key) {
258        if (!super.isWithinLowerBound(key))
259            return false;
260        if (this.keyRange == null)
261            return true;
262        final ByteWriter writer = new ByteWriter();
263        this.encodeKey(writer, key);
264        return KeyRange.compare(writer.getBytes(), this.keyRange.getMin()) >= 0;
265    }
266
267    @Override
268    protected boolean isWithinUpperBound(K key) {
269        if (!super.isWithinUpperBound(key))
270            return false;
271        if (this.keyRange == null)
272            return true;
273        final ByteWriter writer = new ByteWriter();
274        this.encodeKey(writer, key);
275        return KeyRange.compare(writer.getBytes(), this.keyRange.getMax()) < 0;
276    }
277
278    @Override
279    protected final NavigableMap<K, V> createSubMap(boolean reverse, Bounds<K> newBounds) {
280
281        // Determine the direction of the new sub-map
282        final boolean newReversed = this.reversed ^ reverse;
283
284        // Determine new min and max keys
285        final KeyRange newKeyRange = this.buildKeyRange(newReversed ? newBounds.reverse() : newBounds);
286
287        // Create submap
288        return this.createSubMap(newReversed, newKeyRange, this.keyFilter, newBounds);
289    }
290
291    /**
292     * Create a (possibly reversed) view of this instance with (possibly) tighter lower and/or upper bounds and
293     * the given {@link KeyFilter}, if any.
294     * The bounds are consistent with the reversed ordering (i.e., reversed if {@code reverse} is true)
295     * and have already been range-checked against this instance's bounds.
296     *
297     * @param newReversed whether the new map's ordering should be reversed (implies {@code newBounds} are also inverted,
298     *  but <i>not</i> {@code keyRange}); note: means "absolutely" reversed, not relative to this instance
299     * @param newKeyRange new key range, or null for none; will be consistent with {@code bounds}, if any
300     * @param newKeyFilter new key filter, or null for none
301     * @param newBounds new bounds
302     * @return restricted and/or filtered view of this instance
303     * @throws IllegalArgumentException if {@code newBounds} is null
304     */
305    protected abstract NavigableMap<K, V> createSubMap(boolean newReversed,
306      KeyRange newKeyRange, KeyFilter newKeyFilter, Bounds<K> newBounds);
307
308    /**
309     * Encode the given key object into a {@code byte[]} key.
310     * Note that this method must throw {@link IllegalArgumentException}, not {@link ClassCastException}
311     * or {@code NullPointerException}, if {@code obj} does not have the correct type or is an illegal null value.
312     *
313     * @param writer output for encoded {@code byte[]} key corresponding to {@code obj}
314     * @param obj map key object
315     * @throws IllegalArgumentException if {@code obj} is not of the required Java type supported by this set
316     * @throws IllegalArgumentException if {@code obj} is null and this set does not support null elements
317     */
318    protected abstract void encodeKey(ByteWriter writer, Object obj);
319
320    /**
321     * Decode a key object from an encoded {@code byte[]} key.
322     *
323     * <p>
324     * If not in prefix mode, all of {@code reader} must be consumed; otherwise, the consumed portion
325     * is the prefix and any following keys with the same prefix are ignored.
326     * </p>
327     *
328     * @param reader input for encoded bytes
329     * @return decoded map key
330     */
331    protected abstract K decodeKey(ByteReader reader);
332
333    /**
334     * Decode a value object from an encoded {@code byte[]} key/value pair.
335     *
336     * @param pair key/value pair
337     * @return decoded map value
338     */
339    protected abstract V decodeValue(KVPair pair);
340
341    /**
342     * Determine if the given {@code byte[]} key is visible in this map according to the configured
343     * {@link KeyRange} and/or {@link KeyFilter}, if any.
344     *
345     * @param key key to test
346     * @return true if key is visible
347     * @throws IllegalArgumentException if {@code key} is null
348     * @see #filterKeys filterKeys()
349     */
350    protected boolean isVisible(byte[] key) {
351        return (this.keyRange == null || this.keyRange.contains(key))
352          && (this.keyFilter == null || this.keyFilter.contains(key));
353    }
354
355    /**
356     * Encode the given key object, if possible, and verify the corresponding {@code byte[]} key is visible,
357     * otherwise return null or throw an exception.
358     * Delegates to {@link #encodeKey(ByteWriter, Object)} to attempt the actual encoding.
359     *
360     * @param obj key object to encode, possibly null
361     * @param fail whether, if {@code obj} can't be encoded, to throw an exception (true) or return null (false)
362     * @return encoed key for {@code obj}, or null if {@code fail} is false and {@code obj} has the wrong type or is out of bounds
363     * @throws IllegalArgumentException if {@code fail} is true and {@code obj} has the wrong type
364     * @throws IllegalArgumentException if {@code fail} is true and the resulting key is not {@linkplain #isVisible visible}
365     */
366    protected byte[] encodeVisibleKey(Object obj, boolean fail) {
367        final ByteWriter writer = new ByteWriter();
368        try {
369            this.encodeKey(writer, obj);
370        } catch (IllegalArgumentException e) {
371            if (!fail)
372                return null;
373            throw e;
374        }
375        final byte[] key = writer.getBytes();
376        if (this.keyRange != null && !this.keyRange.contains(key)) {
377            if (fail)
378                throw new IllegalArgumentException("key is out of bounds: " + obj);
379            return null;
380        }
381        if (this.keyFilter != null && !this.keyFilter.contains(key)) {
382            if (fail)
383                throw new IllegalArgumentException("key is filtered out: " + obj);
384            return null;
385        }
386        return key;
387    }
388
389    /**
390     * Derive a new {@link KeyRange} from (possibly) new element bounds. The given bounds must <i>not</i> ever be reversed.
391     */
392    private KeyRange buildKeyRange(Bounds<K> bounds) {
393        final byte[] minKey = this.keyRange != null ? this.keyRange.getMin() : null;
394        final byte[] maxKey = this.keyRange != null ? this.keyRange.getMax() : null;
395        byte[] newMinKey;
396        byte[] newMaxKey;
397        switch (bounds.getLowerBoundType()) {
398        case NONE:
399            newMinKey = minKey;
400            break;
401        default:
402            final ByteWriter writer = new ByteWriter();
403            this.encodeKey(writer, bounds.getLowerBound());
404            newMinKey = writer.getBytes();
405            if (!bounds.getLowerBoundType().isInclusive())
406                newMinKey = this.prefixMode ? ByteUtil.getKeyAfterPrefix(newMinKey) : ByteUtil.getNextKey(newMinKey);
407            if (minKey != null)
408                newMinKey = ByteUtil.max(newMinKey, minKey);
409            if (maxKey != null)
410                newMinKey = ByteUtil.min(newMinKey, maxKey);
411            break;
412        }
413        switch (bounds.getUpperBoundType()) {
414        case NONE:
415            newMaxKey = maxKey;
416            break;
417        default:
418            final ByteWriter writer = new ByteWriter();
419            this.encodeKey(writer, bounds.getUpperBound());
420            newMaxKey = writer.getBytes();
421            if (bounds.getUpperBoundType().isInclusive())
422                newMaxKey = this.prefixMode ? ByteUtil.getKeyAfterPrefix(newMaxKey) : ByteUtil.getNextKey(newMaxKey);
423            if (maxKey != null)
424                newMaxKey = ByteUtil.min(newMaxKey, maxKey);
425            if (minKey != null)
426                newMaxKey = ByteUtil.max(newMaxKey, minKey);
427            break;
428        }
429        return new KeyRange(newMinKey != null ? newMinKey : ByteUtil.EMPTY, newMaxKey);
430    }

437            super(AbstractKVNavigableMap.this.kv, AbstractKVNavigableMap.this.prefixMode, AbstractKVNavigableMap.this.reversed,

477    private class EntrySet extends AbstractIterationSet<Map.Entry<K, V>> {
478
479        @Override
480        public Iterator<Map.Entry<K, V>> iterator() {
481            return new AbstractKVIterator<Map.Entry<K, V>>(AbstractKVNavigableMap.this.kv, AbstractKVNavigableMap.this.prefixMode,
482              AbstractKVNavigableMap.this.reversed, AbstractKVNavigableMap.this.keyRange, AbstractKVNavigableMap.this.keyFilter) {
483
484                @Override
485                protected Map.Entry<K, V> decodePair(KVPair pair, ByteReader keyReader) {
486                    final K key = AbstractKVNavigableMap.this.decodeKey(keyReader);
487                    final V value = AbstractKVNavigableMap.this.decodeValue(pair);
488                    return new MapEntry(key, value);
489                }
490
491                @Override
492                protected void doRemove(Map.Entry<K, V> entry, KVPair pair) {
493                    AbstractKVNavigableMap.this.entrySet().remove(entry);
494                }
495            };
496        }
497
498        @Override
499        @SuppressWarnings("unchecked")
500        public boolean contains(Object obj) {
501
502            // Check type
503            if (!(obj instanceof Map.Entry))
504                return false;
505            final Map.Entry<?, ?> entry = (Map.Entry<?, ?>)obj;
506
507            // Find key
508            if (!AbstractKVNavigableMap.this.containsKey(entry.getKey()))
509                return false;
510
511            // Compare key/value pair
512            final K key = (K)entry.getKey();
513            final V value = AbstractKVNavigableMap.this.get(entry.getKey());
514            return new MapEntry(key, value).equals(entry);
515        }
516
517        @Override
518        @SuppressWarnings("unchecked")
519        public boolean remove(Object obj) {
520
521            // Check type
522            if (!(obj instanceof Map.Entry))
523                return false;
524            final Map.Entry<?, ?> entry = (Map.Entry<?, ?>)obj;
525
526            // Find key
527            if (!AbstractKVNavigableMap.this.containsKey(entry.getKey()))
528                return false;
529
530            // Compare key/value pair and remove entry (if contained)
531            final K key = (K)entry.getKey();
532            final V value = AbstractKVNavigableMap.this.get(entry.getKey());
533            if (new MapEntry(key, value).equals(entry)) {
534                AbstractKVNavigableMap.this.remove(key);
535                return true;
536            }
537
538            // Not found
539            return false;
540        }
541
542        @Override
543        public void clear() {
544            AbstractKVNavigableMap.this.clear();
545        }
546    }
