028/**
029 * {@link java.util.NavigableMap} support superclass for maps backed by keys and values encoded as {@code byte[]}
030 * keys and values in a {@link KVStore}, and whose key sort order is consistent with the {@code byte[]} key encoding.
031 *
032 * <p>
033 * Instances are configured with (optional) minimum and maximum keys; when {@linkplain #bounds range restriction} is in
034 * effect, these minimum and maximum keys must correspond to the bounds. Subclasses must implement the
035 * {@linkplain #encodeKey encodeKey()}, {@linkplain #decodeKey decodeKey()},
036 * and {@linkplain #decodeValue decodeValue()}, methods to convert keys and value to/from {@link KVStore} keys and values, and
037 * {@link #createSubMap(boolean, byte[], byte[], Bounds) createSubMap()} to allow creating reversed and restricted range sub-map.
038 *
039 * <p>
040 * Subclasses must also implement {@link #comparator comparator()}, and the resulting sort order must be consistent with
041 * the sort order of the encoded {@code byte[]} keys (possibly {@link #reversed}).
042 * </p>
043 *
044 * <p>
045 * Instances support "prefix mode" where the {@code byte[]} keys may have arbitrary trailing garbage, which is ignored,
046 * and so by definition no key can be a prefix of any other key. The length of the prefix is determined implicitly by the
047 * number of bytes produced by {@link #encodeKey encodeKey()} or consumed by {@link #decodeKey decodeKey()}.
048 * </p>
049 *
050 * <p>
051 * This class provides a read-only implementation; for a mutable implementation, subclasses should also implement
052 * {@link #put put()}, {@link #remove remove()}, and {@link #clear}; note, these methods must be aware
053 * of any {@linkplain AbstractNavigableMap#bounds range restrictions}.
054 * </p>
055 *
056 * <p>
057 * Notes on returned collection classes:
058 * <ul>
059 *  <li>{@link #navigableKeySet} returns a {@link Set} for which:
060 *  <ul>
061 *      <li>The {@link Set#add add()} method is not supported.</li>
062 *      <li>The {@link Set#remove remove()} method and the {@link Iterator#remove remove()} method of the
063 *          associated {@link Iterator} delegate to this instance's {@link #remove remove()} method.</li>
064 *      <li>The {@link Set#clear clear()} method delegates to this instance's {@link #clear} method.</li>
065 *  </ul>
066 *  <li>{@link #entrySet} returns a {@link Set} for which:
067 *  <ul>
068 *      <li>The {@link Set#add add()} method is not supported.</li>
069 *      <li>The {@link Set#remove remove()} method and the {@link Iterator#remove remove()} method of the
070 *          associated {@link Iterator} delegate to this instance's {@link #remove remove()} method (but
071 *          only when the {@link java.util.Map.Entry} is still contained in the map).
072 *      <li>The {@link java.util.Map.Entry} elements' {@link java.util.Map.Entry#setValue setValue()} method delegates to
073 *          this instance's {@link #put put()} method (but only when the {@link java.util.Map.Entry} is still contained in the map).
074 *      <li>The {@link Set#clear clear()} method delegates to this instance's {@link #clear} method.</li>
075 *  </ul>
076 * </ul>
077 * </p>
078 *
079 * <p>
080 * This implementation never throws {@link java.util.ConcurrentModificationException}; instead, iterators always
081 * see the most up-to-date state of the associated {@link KVStore}.
082 * </p>
083 *
084 * @see AbstractKVNavigableSet
085 * @param <K> key type
086 * @param <V> value type
087 */
088@SuppressWarnings("serial")
089public abstract class AbstractKVNavigableMap<K, V> extends AbstractNavigableMap<K, V> {
090
091    /**
092     * The underlying {@link KVStore}.
093     */
094    protected final KVStore kv;
095
096    /**
097     * Whether we are in "prefix" mode.
098     */
099    protected final boolean prefixMode;
100
101    /**
102     * Whether the ordering of this instance is reversed (i.e., from {@link #maxKey} down to {@link #minKey}).
103     */
104    protected final boolean reversed;
105
106    /**
107     * Minimum visible key (inclusive), or null for no minimum. Always less than or equal to {@link #maxKey},
108     * even if this instance is {@link #reversed}. Corresponds to the current {@link #bounds}, if any.
109     */
110    protected final byte[] minKey;
111
112    /**
113     * Maximum visible key (exclusive), or null for no maximum. Always greater than or equal to {@link #minKey},
114     * even if this instance is {@link #reversed}. Corresponds to the current {@link #bounds}, if any.
115     */
116    protected final byte[] maxKey;
117
118// Constructors
119
120    /**
121     * Convenience constructor for when there are no range restrictions.
122     *
123     * @param kv underlying {@link KVStore}
124     * @param prefixMode whether to allow keys to have trailing garbage
125     * @throws IllegalArgumentException if {@code kv} is null
126     */
127    public AbstractKVNavigableMap(KVStore kv, boolean prefixMode) {
128        this(kv, prefixMode, null, null);
129    }
130
131    /**
132     * Convenience constructor for when the range of visible {@link KVStore} keys is all keys sharing a given {@code byte[]} prefix.
133     *
134     * @param kv underlying {@link KVStore}
135     * @param prefixMode whether to allow keys to have trailing garbage
136     * @param prefix prefix defining minimum and maximum keys
137     * @throws NullPointerException if {@code prefix} is null
138     * @throws IllegalArgumentException if {@code kv} is null
139     */
140    public AbstractKVNavigableMap(KVStore kv, boolean prefixMode, byte[] prefix) {
141        this(kv, prefixMode, prefix, ByteUtil.getKeyAfterPrefix(prefix));
142    }
143
144    /**
145     * Primary constructor.
146     *
147     * @param kv underlying {@link KVStore}
148     * @param prefixMode whether to allow keys to have trailing garbage
149     * @param minKey minimum visible key (inclusive), or null for none
150     * @param maxKey maximum visible key (exclusive), or null for none
151     * @throws IllegalArgumentException if {@code minKey} and {@code maxKey} are both not null but {@code minKey > maxKey}
152     * @throws IllegalArgumentException if {@code kv} is null
153     */
154    public AbstractKVNavigableMap(KVStore kv, boolean prefixMode, byte[] minKey, byte[] maxKey) {
155        this(kv, prefixMode, false, minKey, maxKey, new Bounds<K>());
156    }
157
158    /**
159     * Internal constructor. Used for creating sub-maps and reversed views.
160     *
161     * @param kv underlying {@link KVStore}
162     * @param prefixMode whether to allow keys to have trailing garbage
163     * @param reversed whether ordering is reversed (implies {@code bounds} are also inverted,
164     *  but <i>not</i> {@code minKey} and {@code maxKey}); note: means "absolutely" reversed, not relative to this instance
165     * @param minKey minimum visible key (inclusive), or null for none; corresponds to {@code bounds}, if any
166     * @param maxKey maximum visible key (exclusive), or null for none; corresponds to {@code bounds}, if any
167     * @param bounds range restriction
168     * @throws IllegalArgumentException if {@code kv} or {@code bounds} is null
169     * @throws IllegalArgumentException if {@code minKey} and {@code maxKey} are both not null but {@code minKey > maxKey}
170     */
171    protected AbstractKVNavigableMap(KVStore kv, boolean prefixMode,
172      boolean reversed, byte[] minKey, byte[] maxKey, Bounds<K> bounds) {
173        super(bounds);
174        if (kv == null)
175            throw new IllegalArgumentException("null kv");
176        this.kv = kv;
177        this.prefixMode = prefixMode;
178        this.reversed = reversed;
179        this.minKey = minKey != null ? minKey.clone() : null;
180        this.maxKey = minKey != null ? maxKey.clone() : null;
181        if (this.minKey != null && this.maxKey != null && ByteUtil.compare(this.minKey, this.maxKey) > 0)
182            throw new IllegalArgumentException("minKey > maxKey");
183    }
184
185    @Override
186    public V get(Object obj) {
187
188        // Encode key
189        final byte[] key = this.encodeKey(obj, false);
190        if (key == null)
191            return null;
192
193        // Find key, or some longer key with the same prefix in prefix mode
194        final KVPair pair;
195        if (this.prefixMode) {
196            pair = this.kv.getAtLeast(key);
197            if (pair == null || !ByteUtil.isPrefixOf(key, pair.getKey()))
198                return null;
199        } else {
200            final byte[] value = this.kv.get(key);
201            if (value == null)
202                return null;
203            pair = new KVPair(key, value);
204        }
205
206        // Decode value
207        return this.decodeValue(pair);
208    }
209
210    @Override
211    public Set<Map.Entry<K, V>> entrySet() {
212        return this.new EntrySet();
213    }
214
215    @Override
216    public NavigableSet<K> navigableKeySet() {
217        return this.new KeySet(this.reversed, this.minKey, this.maxKey, this.bounds);
218    }
219
220    @Override
221    protected final NavigableMap<K, V> createSubMap(boolean reverse, Bounds<K> newBounds) {
222
223        // Determine the direction of the new sub-map
224        final boolean newReversed = this.reversed ^ reverse;
225
226        // Determine new min and max keys
227        final byte[][] newMinMax = this.buildMinMax(newReversed ? newBounds.reverse() : newBounds);
228
229        // Create submap
230        return this.createSubMap(newReversed, newMinMax[0], newMinMax[1], newBounds);
231    }
232
233    /**
234     * Create a (possibly reversed) view of this instance with (possibly) tighter lower and/or upper bounds.
235     * The bounds are consistent with the reversed ordering (i.e., reversed if {@code reverse} is true)
236     * and have already been range-checked against this instance's bounds.
237     *
238     * @param newReversed whether the new map's ordering should be reversed (implies {@code newBounds} are also inverted,
239     *  but <i>not</i> {@code newMinKey} and {@code newMaxKey}); note: means "absolutely" reversed, not relative to this instance
240     * @param newMinKey new minimum visible key (inclusive), or null for none; corresponds to {@code bounds}, if any
241     * @param newMaxKey new maximum visible key (exclusive), or null for none; corresponds to {@code bounds}, if any
242     * @param newBounds new bounds
243     * @throws IllegalArgumentException if {@code newBounds} is null
244     */
245    protected abstract NavigableMap<K, V> createSubMap(boolean newReversed,
246      byte[] newMinKey, byte[] newMaxKey, Bounds<K> newBounds);
247
248    /**
249     * Encode the given key object into a {@code byte[]} key.
250     * Note that this method must throw {@link IllegalArgumentException}, not {@link ClassCastException}
251     * or {@code NullPointerException}, if {@code obj} does not have the correct type or is an illegal null value.
252     *
253     * @param writer output for encoded {@code byte[]} key corresponding to {@code obj}
254     * @param obj map key object
255     * @throws IllegalArgumentException if {@code obj} is not of the required Java type supported by this set
256     * @throws IllegalArgumentException if {@code obj} is null and this set does not support null elements
257     */
258    protected abstract void encodeKey(ByteWriter writer, Object obj);
259
260    /**
261     * Decode a key object from an encoded {@code byte[]} key.
262     *
263     * @param reader input for encoded bytes
264     * @return decoded map key
265     */
266    protected abstract K decodeKey(ByteReader reader);
267
268    /**
269     * Decode a value object from an encoded {@code byte[]} key/value pair.
270     *
271     * @param pair key/value pair
272     * @return decoded map value
273     */
274    protected abstract V decodeValue(KVPair pair);
275
276    /**
277     * Determine if the given key is in this set's key range (i.e., between {@link #minKey} and {@link #maxKey}).
278     */
279    protected boolean inRange(byte[] key) {
280        if (this.minKey != null && ByteUtil.compare(key, this.minKey) < 0)
281            return false;
282        if (this.maxKey != null && ByteUtil.compare(key, this.maxKey) >= 0)
283            return false;
284        return true;
285    }
286
287    /**
288     * Encode the given key object, if possible, otherwise return null or throw an exception.
289     * Delegates to {@link #encodeKey(ByteWriter, Object)} to attempt the actual encoding.
290     *
291     * @param obj key object to encode, possibly null
292     * @param fail whether, if {@code obj} can't be encoded, to throw an exception (true) or return null (false)
293     * @return encoed key for {@code obj}, or null if {@code fail} is false and {@code obj} has the wrong type or is out of bounds
294     * @throws IllegalArgumentException if {@code fail} is true and {@code obj} has the wrong type
295     * @throws IllegalArgumentException if {@code fail} is true and {@code obj} is out of bounds
296     */
297    protected byte[] encodeKey(Object obj, boolean fail) {
298        final ByteWriter writer = new ByteWriter();
299        try {
300            this.encodeKey(writer, obj);
301        } catch (IllegalArgumentException e) {
302            if (!fail)
303                return null;
304            throw e;
305        }
306        final byte[] key = writer.getBytes();
307        if (!this.inRange(key)) {
308            if (fail)
309                throw new IllegalArgumentException("value is out of bounds: " + obj);
310            return null;
311        }
312        return key;
313    }
314
315    /**
316     * Derive new min and max keys from (possibly) new element bounds. The given bounds must <i>not</i> ever be reversed.
317     */
318    private byte[][] buildMinMax(Bounds<K> bounds) {
319        final byte[][] result = new byte[2][];
320        switch (bounds.getLowerBoundType()) {
321        case NONE:
322            result[0] = this.minKey;
323            break;
324        default:
325            final ByteWriter writer = new ByteWriter();
326            this.encodeKey(writer, bounds.getLowerBound());
327            result[0] = writer.getBytes();
328            if (!bounds.getLowerBoundType().isInclusive())
329                result[0] = ByteUtil.getNextKey(result[0]);
330            break;
331        }
332        switch (bounds.getUpperBoundType()) {
333        case NONE:
334            result[1] = this.maxKey;
335            break;
336        default:
337            final ByteWriter writer = new ByteWriter();
338            this.encodeKey(writer, bounds.getUpperBound());
339            result[1] = writer.getBytes();
340            if (bounds.getUpperBoundType().isInclusive())
341                result[1] = ByteUtil.getNextKey(result[1]);
342            break;
343        }
344        return result;
345    }

350
351        KeySet(boolean reversed, byte[] minKey, byte[] maxKey, Bounds<K> bounds) {

357            return AbstractKVNavigableMap.this.comparator();
358        }
359
360        @Override
361        public boolean remove(Object obj) {
362            final boolean existed = AbstractKVNavigableMap.this.containsKey(obj);
363            AbstractKVNavigableMap.this.remove(obj);
364            return existed;
365        }
366
367        @Override
368        public void clear() {
369            AbstractKVNavigableMap.this.clear();
370        }
371
372        @Override
373        protected void encode(ByteWriter writer, Object obj) {
374            AbstractKVNavigableMap.this.encodeKey(writer, obj);
375        }
376
377        @Override
378        protected K decode(ByteReader reader) {
379            return AbstractKVNavigableMap.this.decodeKey(reader);
380        }
381
382        @Override
383        protected NavigableSet<K> createSubSet(boolean newReversed, byte[] newMinKey, byte[] newMaxKey, Bounds<K> newBounds) {
384            return AbstractKVNavigableMap.this.createSubMap(newReversed, newMinKey, newMaxKey, newBounds).navigableKeySet();
385        }

390    private class EntrySet extends AbstractIterationSet<Map.Entry<K, V>> {
391
392        @Override
393        public EntrySetIterator iterator() {
394            return AbstractKVNavigableMap.this.new EntrySetIterator();
395        }
396
397        @Override
398        @SuppressWarnings("unchecked")
399        public boolean contains(Object obj) {
400
401            // Check type
402            if (!(obj instanceof Map.Entry))
403                return false;
404            final Map.Entry<?, ?> entry = (Map.Entry<?, ?>)obj;
405
406            // Find key
407            if (!AbstractKVNavigableMap.this.containsKey(entry.getKey()))
408                return false;
409
410            // Compare key/value pair
411            final K key = (K)entry.getKey();
412            final V value = AbstractKVNavigableMap.this.get(entry.getKey());
413            return new MapEntry(key, value).equals(entry);
414        }
415
416        @Override
417        @SuppressWarnings("unchecked")
418        public boolean remove(Object obj) {
419
420            // Check type
421            if (!(obj instanceof Map.Entry))
422                return false;
423            final Map.Entry<?, ?> entry = (Map.Entry<?, ?>)obj;
424
425            // Find key
426            if (!AbstractKVNavigableMap.this.containsKey(entry.getKey()))
427                return false;
428
429            // Compare key/value pair and remove entry (if contained)
430            final K key = (K)entry.getKey();
431            final V value = AbstractKVNavigableMap.this.get(entry.getKey());
432            if (new MapEntry(key, value).equals(entry)) {
433                AbstractKVNavigableMap.this.remove(key);
434                return true;
435            }
436
437            // Not found
438            return false;
439        }
440
441        @Override
442        public void clear() {
443            AbstractKVNavigableMap.this.clear();
444        }
445    }

447// EntrySetIterator
448
449    private class EntrySetIterator implements Iterator<Map.Entry<K, V>> {
450
451        private final Iterator<KVPair> pairIterator;
452        private Map.Entry<K, V> removeEntry;
453
454        public EntrySetIterator() {
455            if (AbstractKVNavigableMap.this.prefixMode) {
456                this.pairIterator = new KVPairIterator(AbstractKVNavigableMap.this.kv,
457                  AbstractKVNavigableMap.this.minKey, AbstractKVNavigableMap.this.maxKey, AbstractKVNavigableMap.this.reversed);
458            } else {
459                this.pairIterator = AbstractKVNavigableMap.this.kv.getRange(
460                  AbstractKVNavigableMap.this.minKey, AbstractKVNavigableMap.this.maxKey, AbstractKVNavigableMap.this.reversed);
461            }
462        }
463
464        @Override
465        public boolean hasNext() {
466            return this.pairIterator.hasNext();
467        }
468
469        @Override
470        public Map.Entry<K, V> next() {
471            final KVPair pair = this.pairIterator.next();
472
473            // Decode key
474            final ByteReader reader = new ByteReader(pair.getKey());
475            final K key = AbstractKVNavigableMap.this.decodeKey(reader);
476
477            // Decode value
478            final V value = AbstractKVNavigableMap.this.decodeValue(pair);
479
480            // In prefix mode, skip over any additional keys having the same prefix as what we just decoded
481            if (AbstractKVNavigableMap.this.prefixMode) {
482                final KVPairIterator kvPairIterator = (KVPairIterator)this.pairIterator;
483                final byte[] prefix = reader.getBytes(0, reader.getOffset());
484                kvPairIterator.setNextTarget(kvPairIterator.isReverse() ? prefix : ByteUtil.getKeyAfterPrefix(prefix));
485            }
486
487            // Save elment for possible remove()
488            this.removeEntry = new MapEntry(key, value);
489            return removeEntry;
490        }
491
492        @Override
493        public void remove() {
494            if (this.removeEntry == null)
495                throw new IllegalStateException();
496            AbstractKVNavigableMap.this.entrySet().remove(this.removeEntry);
497            this.removeEntry = null;
498        }
499    }

504
505        MapEntry(K key, V value) {
506            super(key, value);
507        }

511            AbstractKVNavigableMap.this.put(this.getKey(), value);
