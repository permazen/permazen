006package org.jsimpledb.kv.util;
007
008import java.util.Collections;
009import java.util.Comparator;
010import java.util.NavigableSet;
011
012import org.jsimpledb.kv.KVStore;
013import org.jsimpledb.kv.KeyFilter;
014import org.jsimpledb.kv.KeyRange;
015import org.jsimpledb.util.Bounds;
016import org.jsimpledb.util.ByteReader;
017import org.jsimpledb.util.ByteUtil;
018import org.jsimpledb.util.ByteWriter;
019
020/**
021 * A {@link java.util.NavigableSet} view of the keys in a {@link KVStore}.
022 *
023 * <p>
024 * Instances are mutable, with these exceptions:
025 * <ul>
026 *  <li>Adding new elements via {@link #add add()} is not supported</li>
027 *  <li>{@link #clear} is not supported when a {@link KeyFilter} is configured</li>
028 * </ul>
029 */
030@SuppressWarnings("serial")
031public class KVNavigableSet extends AbstractKVNavigableSet<byte[]> {
032
033// Constructors
034
035    /**
036     * Constructor for when there are no range restrictions.
037     *
038     * @param kv underlying {@link KVStore}
039     */
040    public KVNavigableSet(KVStore kv) {
041        this(kv, false, null, null);
042    }
043
044    /**
045     * Constructor for when the range of visible {@link KVStore} keys is all keys sharing a given {@code byte[]} prefix.
046     *
047     * @param kv underlying {@link KVStore}
048     * @param prefix prefix defining minimum and maximum keys
049     * @throws NullPointerException if {@code prefix} is null
050     */
051    public KVNavigableSet(KVStore kv, byte[] prefix) {
052        this(kv, false, KeyRange.forPrefix(prefix), null);
053    }
054
055    /**
056     * Primary constructor.
057     *
058     * @param kv underlying {@link KVStore}
059     * @param reversed whether ordering is reversed
060     * @param keyRange key range restriction, or null for none
061     * @param keyFilter key filter, or null for none
062     * @throws IllegalArgumentException if {@code kv} is null
063     */
064    protected KVNavigableSet(KVStore kv, boolean reversed, KeyRange keyRange, KeyFilter keyFilter) {
065        this(kv, reversed, keyRange, keyFilter, KVNavigableSet.createBounds(keyRange));
066    }
067
068    /**
069     * Internal constructor. Used for creating sub-sets and reversed views.
070     *
071     * @param kv underlying {@link KVStore}
072     * @param reversed whether ordering is reversed (implies {@code bounds} are also inverted)
073     * @param keyRange key range restriction, or null for none
074     * @param keyFilter key filter, or null for none
075     * @param bounds range restriction
076     * @throws IllegalArgumentException if {@code kv} or {@code bounds} is null
077     */
078    private KVNavigableSet(KVStore kv, boolean reversed, KeyRange keyRange, KeyFilter keyFilter, Bounds<byte[]> bounds) {
079        super(kv, false, reversed, keyRange, keyFilter, bounds);
080    }
081
082// Methods
083
084    @Override
085    public Comparator<byte[]> comparator() {
086        return this.reversed ? Collections.reverseOrder(ByteUtil.COMPARATOR) : ByteUtil.COMPARATOR;
087    }
088
089    @Override
090    public boolean remove(Object obj) {
091        if (!(obj instanceof byte[]))
092            return false;
093        final byte[] key = (byte[])obj;
094        if (!this.isVisible(key))
095            return false;
096        final byte[] value = this.kv.get(key);
097        if (value == null)
098            return false;
099        this.kv.remove(key);
100        return true;
101    }
102
103    @Override
104    public void clear() {
105        if (this.keyFilter != null)
106            throw new UnsupportedOperationException("clear() not supported when KeyFilter configured");
107        final byte[] minKey = this.keyRange != null ? this.keyRange.getMin() : null;
108        final byte[] maxKey = this.keyRange != null ? this.keyRange.getMax() : null;
109        this.kv.removeRange(minKey, maxKey);
110        return;
111    }
112
113    @Override
114    protected void encode(ByteWriter writer, Object obj) {
115        if (!(obj instanceof byte[]))
116            throw new IllegalArgumentException("value is not a byte[]");
117        writer.write((byte[])obj);
118    }
119
120    @Override
121    protected byte[] decode(ByteReader reader) {
122        return reader.getBytes();
123    }
124
125    @Override
126    protected NavigableSet<byte[]> createSubSet(boolean newReversed,
127      KeyRange newKeyRange, KeyFilter newKeyFilter, Bounds<byte[]> newBounds) {
128        return new KVNavigableSet(this.kv, newReversed, newKeyRange, newKeyFilter, newBounds);
129    }
130
131    static Bounds<byte[]> createBounds(KeyRange keyRange) {
132        return keyRange != null ? new Bounds<byte[]>(keyRange.getMin(), keyRange.getMax()) : new Bounds<byte[]>();
133    }
134}
