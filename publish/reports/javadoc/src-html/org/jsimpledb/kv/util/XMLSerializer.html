006package org.jsimpledb.kv.util;
007
008import java.io.InputStream;
009import java.io.OutputStream;
010import java.io.Writer;
011import java.util.Iterator;
012
013import javax.xml.namespace.QName;
014import javax.xml.stream.XMLInputFactory;
015import javax.xml.stream.XMLOutputFactory;
016import javax.xml.stream.XMLStreamException;
017import javax.xml.stream.XMLStreamReader;
018import javax.xml.stream.XMLStreamWriter;
019
020import org.dellroad.stuff.xml.IndentXMLStreamWriter;
021import org.jsimpledb.kv.KVPair;
022import org.jsimpledb.kv.KVStore;
023import org.jsimpledb.util.AbstractXMLStreaming;
024import org.jsimpledb.util.ByteUtil;
025
026/**
027 * Utility methods for serializing and deserializing the contents of a {@link org.jsimpledb.kv.KVStore} to/from XML.
028 *
029 * <p>
030 * The XML has a simple format; empty values may be omitted:
031 *  <pre>
032 *  &lt;xml version="1.0" encoding="UTF-8"?&gt;
033 *
034 *  &lt;entries&gt;
035 *      &lt;entry&gt;
036 *          &lt;key&gt;013f7b&lt;/key&gt;
037 *          &lt;value&gt;5502&lt;/value&gt;
038 *      &lt;/entry&gt;
039 *      &lt;entry&gt;
040 *          &lt;key&gt;ee7698&lt;/key&gt;
041 *      &lt;/entry&gt;
042 *      ...
043 *  &lt;/entries&gt;
044 *  </pre>
045 */
046public class XMLSerializer extends AbstractXMLStreaming {
047
048    public static final QName ENTRIES_TAG = new QName("entries");
049    public static final QName ENTRY_TAG = new QName("entry");
050    public static final QName KEY_TAG = new QName("key");
051    public static final QName VALUE_TAG = new QName("value");
052
053    private final KVStore kv;
054
055    /**
056     * Constructor.
057     *
058     * @param kv key/value store on which to operate
059     * @throws IllegalArgumentException if {@code kv} is null
060     */
061    public XMLSerializer(KVStore kv) {
062        if (kv == null)
063            throw new IllegalArgumentException("null kv");
064        this.kv = kv;
065    }
066
067    /**
068     * Import key/value pairs into the {@link KVStore} associated with this instance from the given XML input.
069     *
070     * <p>
071     * The {@code input} is not closed by this method.
072     * </p>
073     *
074     * @param input XML input
075     * @throws XMLStreamException if an error occurs
076     * @throws IllegalArgumentException if {@code input} is null
077     */
078    public void read(InputStream input) throws XMLStreamException {
079        if (input == null)
080            throw new IllegalArgumentException("null input");
081        this.read(XMLInputFactory.newFactory().createXMLStreamReader(input));
082    }
083
084    /**
085     * Import key/value pairs into the {@link KVStore} associated with this instance from the given XML input.
086     * This method expects to see an opening {@code <entries>} as the next event (not counting whitespace, comments, etc.),
087     * which is then consumed up through the closing {@code </entries>} event. The {@code reader} is not closed by this method.
088     * Therefore, this tag could be part of a larger XML document.
089     *
090     * @param reader XML reader
091     * @throws XMLStreamException if an error occurs
092     * @throws IllegalArgumentException if {@code reader} is null
093     */
094    public void read(XMLStreamReader reader) throws XMLStreamException {
095        if (reader == null)
096            throw new IllegalArgumentException("null reader");
097        this.expect(reader, false, ENTRIES_TAG);
098        while (this.expect(reader, true, ENTRY_TAG)) {
099            this.expect(reader, false, KEY_TAG);
100            byte[] key;
101            try {
102                key = ByteUtil.parse(reader.getElementText());
103            } catch (IllegalArgumentException e) {
104                throw new XMLStreamException("invalid hexadecimal key", reader.getLocation(), e);
105            }
106            if (!this.expect(reader, true, VALUE_TAG)) {
107                this.kv.put(key, ByteUtil.EMPTY);
108                continue;                           // closing </entry> tag alread read
109            }
110            byte[] value;
111            try {
112                value = ByteUtil.parse(reader.getElementText());
113            } catch (IllegalArgumentException e) {
114                throw new XMLStreamException("invalid hexadecimal value", reader.getLocation(), e);
115            }
116            this.kv.put(key, value);
117            this.expectClose(reader);               // read closing </entry> tag
118        }
119    }
120
121    /**
122     * Export all key/value pairs from the {@link KVStore} associated with this instance to the given output.
123     *
124     * <p>
125     * The {@code output} is not closed by this method.
126     * </p>
127     *
128     * @param output XML output; will not be closed by this method
129     * @param indent true to indent output, false for all on one line
130     * @throws XMLStreamException if an error occurs
131     * @throws IllegalArgumentException if {@code output} is null
132     */
133    public void write(OutputStream output, boolean indent) throws XMLStreamException {
134        if (output == null)
135            throw new IllegalArgumentException("null output");
136        XMLStreamWriter xmlWriter = XMLOutputFactory.newInstance().createXMLStreamWriter(output, "UTF-8");
137        if (indent)
138            xmlWriter = new IndentXMLStreamWriter(xmlWriter);
139        xmlWriter.writeStartDocument("UTF-8", "1.0");
140        this.write(xmlWriter, null, null);
141    }
142
143    /**
144     * Export all key/value pairs from the {@link KVStore} associated with this instance to the given writer.
145     *
146     * <p>
147     * The {@code writer} is not closed by this method.
148     * </p>
149     *
150     * @param writer XML output; will not be closed by this method
151     * @param indent true to indent output, false for all on one line
152     * @throws XMLStreamException if an error occurs
153     * @throws IllegalArgumentException if {@code writer} is null
154     */
155    public void write(Writer writer, boolean indent) throws XMLStreamException {
156        if (writer == null)
157            throw new IllegalArgumentException("null writer");
158        XMLStreamWriter xmlWriter = XMLOutputFactory.newInstance().createXMLStreamWriter(writer);
159        if (indent)
160            xmlWriter = new IndentXMLStreamWriter(xmlWriter);
161        xmlWriter.writeStartDocument("1.0");
162        this.write(xmlWriter, null, null);
163    }
164
165    /**
166     * Export a range of key/value pairs from the {@link KVStore} associated with this instance to the given XML output.
167     *
168     * <p>
169     * This method writes a start element as its first action, allowing the output to be embedded into a larger XML document.
170     * Callers not embedding the output may with to precede invocation of this method with a call to
171     * {@link XMLStreamWriter#writeStartDocument writer.writeStartDocument()}.
172     * </p>
173     *
174     * <p>
175     * The {@code writer} is not closed by this method.
176     * </p>
177     *
178     * @param writer XML writer; will not be closed by this method
179     * @param minKey minimum key (inclusive), or null for none
180     * @param maxKey maximum key (exclusive), or null for none
181     * @throws XMLStreamException if an error occurs
182     * @throws IllegalArgumentException if {@code writer} is null
183     */
184    public void write(XMLStreamWriter writer, byte[] minKey, byte[] maxKey) throws XMLStreamException {
185        if (writer == null)
186            throw new IllegalArgumentException("null writer");
187        writer.setDefaultNamespace(ENTRIES_TAG.getNamespaceURI());
188        writer.writeStartElement(ENTRIES_TAG.getNamespaceURI(), ENTRIES_TAG.getLocalPart());
189        for (Iterator<KVPair> i = this.kv.getRange(minKey, maxKey, false); i.hasNext(); ) {
190            writer.writeStartElement(ENTRY_TAG.getNamespaceURI(), ENTRY_TAG.getLocalPart());
191            final KVPair pair = i.next();
192            this.writeElement(writer, KEY_TAG, pair.getKey());
193            final byte[] value = pair.getValue();
194            if (value.length > 0)
195                this.writeElement(writer, VALUE_TAG, value);
196            writer.writeEndElement();
197        }
198        writer.writeEndElement();
199        writer.flush();
200    }
201
202// Internal methods
203
204    private void writeElement(XMLStreamWriter writer, QName element, byte[] value) throws XMLStreamException {
205        this.writeElement(writer, element, ByteUtil.toString(value));
206    }
207}
