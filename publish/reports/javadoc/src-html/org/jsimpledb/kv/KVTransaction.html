006package org.jsimpledb.kv;
007
008/**
009 * {@link KVDatabase} transaction API.
010 *
011 * <p>
012 * Provides a transactional view into a {@link KVStore}.
013 * </p>
014 *
015 * <p>
016 * Instances may throw {@link KVTransactionException} during any operation if the transaction cannot be continued.
017 * In particular, {@link StaleTransactionException} is thrown by a transaction that is no longer open, and
018 * {@link RetryTransactionException} is thrown when the transaction should be retried due to a transient
019 * problem (such as a write conflict with another transaction). When {@link RetryTransactionException} is thrown by
020 * {@link #commit}, the transaction may have actually been committed. Therefore, transactions should be written to be idempotent.
021 * When any {@link KVTransactionException} is thrown, the transaction must still support invoking {@link #rollback},
022 * but all other operations may throw {@link StaleTransactionException}.
023 * </p>
024 *
025 * <p>
026 * If an instance throws a {@link KVTransactionException}, the transaction should be implicitly rolled back.
027 * </p>
028 *
029 * <p>
030 * Implementations must throw {@link StaleTransactionException} if {@link #commit} or {@link #rollback} has already
031 * been invoked, or if the {@link KVTransaction} instance is no longer usable for some other reason. In particular,
032 * implementations should throw {@link TransactionTimeoutException} if an operation is attempted on a transaction
033 * that has been held open past some maximum allowed time limit.
034 * </p>
035 *
036 * <p>
037 * Implementations are responsible for ensuring modifications to {@code byte[]} arrays after method
038 * invocations do no harm. This usually means {@code byte[]} array parameters and return values must be copied.
039 * </p>
040 *
041 * <p>
042 * Accessing keys that start with {@code 0xff} is not supported and will result in {@link IllegalArgumentException}.
043 * </p>
044 */
045public interface KVTransaction extends KVStore {
046
047    /**
048     * Get the {@link KVDatabase} with which this instance is associated.
049     *
050     * @return associated database
051     */
052    KVDatabase getKVDatabase();
053
054    /**
055     * Change the timeout for this transaction from its default value (optional operation).
056     *
057     * @param timeout transaction timeout in milliseconds, or zero for unlimited
058     * @throws UnsupportedOperationException if this transaction does not support timeouts
059     * @throws IllegalArgumentException if {@code timeout} is negative
060     * @throws StaleTransactionException if this transaction is no longer usable
061     */
062    void setTimeout(long timeout);
063
064    /**
065     * Commit this transaction.
066     *
067     * <p>
068     * Note that if this method throws a {@link RetryTransactionException},
069     * the transaction was either successfully committed or rolled back. In either case,
070     * this instance is no longer usable.
071     * </p>
072     *
073     * @throws StaleTransactionException if this transaction is no longer usable
074     * @throws RetryTransactionException if this transaction must be retried and is no longer usable
075     */
076    void commit();
077
078    /**
079     * Cancel this transaction.
080     *
081     * <p>
082     * After this method returns, this instance is no longer usable.
083     *
084     * <p>
085     * This method may be invoked at any time, even after a previous invocation of
086     * {@link #commit} or {@link #rollback}, in which case the invocation will be ignored.
087     */
088    void rollback();
089}
