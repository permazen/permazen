006package org.jsimpledb.kv;
007
008import java.util.ArrayList;
009import java.util.Arrays;
010
011import org.jsimpledb.util.ByteUtil;
012
013/**
014 * Utility methods for dealing with {@link KeyFilter}s.
015 *
016 * @see KeyFilter
017 */
018public final class KeyFilterUtil {
019
020    private KeyFilterUtil() {
021    }
022
023    /**
024     * Create a {@link KeyFilter} that represents the union of the given instances.
025     *
026     * @param keyFilters instances to unify
027     * @return the union of the given {@code keyFilters}
028     * @throws IllegalArgumentException if {@code keyFilters} is empty
029     * @throws IllegalArgumentException if {@code keyFilters} or any element in {@code keyFilters} is null
030     */
031    public static KeyFilter union(KeyFilter... keyFilters) {
032        if (keyFilters == null)
033            throw new IllegalArgumentException("null keyFilters");
034        switch (keyFilters.length) {
035        case 0:
036            throw new IllegalArgumentException("empty keyFilters");
037        case 1:
038            return keyFilters[0];
039        default:
040            break;
041        }
042
043        // Optimize when all are KeyRanges instances
044        boolean allKeyRanges = true;
045        for (KeyFilter keyFilter : keyFilters) {
046            if (keyFilter == null)
047                throw new IllegalArgumentException("null keyFilter");
048            if (!(keyFilter instanceof KeyRanges))
049                allKeyRanges = false;
050        }
051        if (allKeyRanges) {
052            KeyRanges union = (KeyRanges)keyFilters[0];
053            for (int i = 1; i < keyFilters.length; i++)
054                union = union.union((KeyRanges)keyFilters[i]);
055            return union;
056        }
057
058        // Can't optimize
059        return new UnionKeyFilter(keyFilters);
060    }
061
062    /**
063     * Create a {@link KeyFilter} that represents the intersection of the given instances.
064     *
065     * @param keyFilters instances to intersect
066     * @return the intersection of the given {@code keyFilters}
067     * @throws IllegalArgumentException if {@code keyFilters} is empty
068     * @throws IllegalArgumentException if {@code keyFilters} or any element in {@code keyFilters} is null
069     */
070    public static KeyFilter intersection(KeyFilter... keyFilters) {
071        if (keyFilters == null)
072            throw new IllegalArgumentException("null keyFilters");
073        switch (keyFilters.length) {
074        case 0:
075            throw new IllegalArgumentException("empty keyFilters");
076        case 1:
077            return keyFilters[0];
078        default:
079            break;
080        }
081
082        // Optimize when all are KeyRanges instances
083        boolean allKeyRanges = true;
084        for (KeyFilter keyFilter : keyFilters) {
085            if (keyFilter == null)
086                throw new IllegalArgumentException("null keyFilter");
087            if (!(keyFilter instanceof KeyRanges))
088                allKeyRanges = false;
089        }
090        if (allKeyRanges) {
091            KeyRanges intersection = (KeyRanges)keyFilters[0];
092            for (int i = 1; i < keyFilters.length; i++)
093                intersection = intersection.intersection((KeyRanges)keyFilters[i]);
094            return intersection;
095        }
096
097        // Can't optimize
098        return new IntersectionKeyFilter(keyFilters);
099    }
100
101    /**
102     * Determine if the given instance is empty, i.e., contains no keys.
103     *
104     * @param keyFilter instance to check
105     * @return true if filter is empty
106     * @throws IllegalArgumentException if {@code keyFilter} is null
107     */
108    public static boolean isEmpty(KeyFilter keyFilter) {
109        if (keyFilter == null)
110            throw new IllegalArgumentException("null keyFilter");
111        return keyFilter.seekHigher(ByteUtil.EMPTY) == null;
112    }
113
114    private static byte[] seek(KeyFilter[] keyFilters, byte[] key, boolean seekHigher, boolean preferHigher) {
115        if (key == null)
116            throw new IllegalArgumentException("null key");
117        assert keyFilters.length > 0;
118        final boolean preferNull = seekHigher == preferHigher;
119        byte[] best = null;
120        for (int i = 0; i < keyFilters.length; i++) {
121            final KeyFilter keyFilter = keyFilters[i];
122            final byte[] next = seekHigher ? keyFilter.seekHigher(key) : keyFilter.seekLower(key);
123            if (i == 0)
124                best = next;
125            if (next == null) {
126                if (preferNull)
127                    return null;
128                continue;
129            }
130            assert next.length != 0 || key.length == 0;
131            if (i > 0 && best != null) {
132                final int diff = (!seekHigher && next.length == 0) ? 1 : ByteUtil.compare(next, best);
133                if (preferHigher ? diff < 0 : diff > 0)
134                    continue;
135            } else
136                assert i == 0 || !preferNull;
137            best = next;
138        }
139        return best;
140    }
141
142// UnionKeyFilter
143
144    private static class UnionKeyFilter implements KeyFilter {
145
146        private final KeyFilter[] keyFilters;
147
148        UnionKeyFilter(KeyFilter[] keyFilters) {
149            assert keyFilters.length >= 2;
150            final ArrayList<KeyFilter> list = new ArrayList<>();
151            for (KeyFilter keyFilter : keyFilters) {
152                if (keyFilter instanceof UnionKeyFilter)
153                    list.addAll(Arrays.asList(((UnionKeyFilter)keyFilter).keyFilters));
154                else
155                    list.add(keyFilter);
156            }
157            this.keyFilters = list.toArray(new KeyFilter[list.size()]);
158        }
159
160        @Override
161        public boolean contains(byte[] key) {
162            for (KeyFilter keyFilter : this.keyFilters) {
163                if (keyFilter.contains(key))
164                    return true;
165            }
166            return false;
167        }
168
169        @Override
170        public byte[] seekHigher(byte[] key) {
171            return KeyFilterUtil.seek(this.keyFilters, key, true, false);
172        }
173
174        @Override
175        public byte[] seekLower(byte[] key) {
176            return KeyFilterUtil.seek(this.keyFilters, key, false, true);
177        }
178    }
179
180// IntersectionKeyFilter

182    private static class IntersectionKeyFilter implements KeyFilter {
183
184        private final KeyFilter[] keyFilters;
185
186        IntersectionKeyFilter(KeyFilter[] keyFilters) {
187            assert keyFilters.length >= 2;
188            final ArrayList<KeyFilter> list = new ArrayList<>();
189            for (KeyFilter keyFilter : keyFilters) {
190                if (keyFilter instanceof IntersectionKeyFilter)
191                    list.addAll(Arrays.asList(((IntersectionKeyFilter)keyFilter).keyFilters));
192                else
193                    list.add(keyFilter);
194            }
195            this.keyFilters = list.toArray(new KeyFilter[list.size()]);
196        }
197
198        @Override
199        public boolean contains(byte[] key) {
200            for (KeyFilter keyFilter : this.keyFilters) {
201                if (!keyFilter.contains(key))
202                    return false;
203            }
204            return true;
205        }
206
207        @Override
208        public byte[] seekHigher(byte[] key) {
209            return KeyFilterUtil.seek(this.keyFilters, key, true, true);
210        }
211
212        @Override
213        public byte[] seekLower(byte[] key) {
214            return KeyFilterUtil.seek(this.keyFilters, key, false, false);
215        }
216    }
