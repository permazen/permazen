006package org.jsimpledb.kv;
007
008import java.util.Arrays;
009import java.util.Map;
010
011import org.jsimpledb.util.ByteUtil;
012
013/**
014 * A key/value pair.
015 *
016 * Note: the internal byte arrays are not copied; therefore, values passed to the constructor
017 * or returned from the accessor methods must not be modified if instances are to remain immutable.
018 */
019public class KVPair {
020
021    private final byte[] key;
022    private final byte[] value;
023
024    /**
025     * Constructor. The given arrays are copied.
026     *
027     * @param key key
028     * @param value value
029     * @throws IllegalArgumentException if {@code key} or {@code value} is null
030     */
031    public KVPair(byte[] key, byte[] value) {
032        if (key == null)
033            throw new IllegalArgumentException("null key");
034        if (value == null)
035            throw new IllegalArgumentException("null value");
036        this.key = key;
037        this.value = value;
038    }
039
040    /**
041     * Constructor. The given key and value arrays are copied.
042     *
043     * @param entry map entry
044     * @throws IllegalArgumentException if {@code entry} or its key or value is null
045     */
046    public KVPair(Map.Entry<byte[], byte[]> entry) {
047        if (entry == null)
048            throw new IllegalArgumentException("null entry");
049        this.key = entry.getKey();
050        this.value = entry.getValue();
051        if (this.key == null)
052            throw new IllegalArgumentException("null key");
053        if (this.value == null)
054            throw new IllegalArgumentException("null value");
055    }
056
057    /**
058     * Get the key.
059     *
060     * @return the key
061     */
062    public byte[] getKey() {
063        return this.key;
064    }
065
066    /**
067     * Get the value.
068     *
069     * @return the value
070     */
071    public byte[] getValue() {
072        return this.value;
073    }
074
075    @Override
076    public String toString() {
077        return "KVPair[" + ByteUtil.toString(this.key) + ":" + ByteUtil.toString(this.value) + "]";
078    }
079
080    @Override
081    public boolean equals(Object obj) {
082        if (obj == this)
083            return true;
084        if (obj == null || obj.getClass() != this.getClass())
085            return false;
086        final KVPair that = (KVPair)obj;
087        return Arrays.equals(this.key, that.key) && Arrays.equals(this.value, that.value);
088    }
089
090    @Override
091    public int hashCode() {
092        return Arrays.hashCode(this.key) ^ Arrays.hashCode(this.value);
093    }
094}
