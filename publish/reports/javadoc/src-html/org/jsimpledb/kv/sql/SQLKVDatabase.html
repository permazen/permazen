186    /**
187     * Create a new transaction.
188     *
189     * <p>
190     * The implementation in {@link SQLKVDatabase} invokes {@link #createTransactionConnection createTransactionConnection()}
191     * to get a {@link Connection} for the new transaction, then invokes these methods in order:
192     *  <ol>
193     *  <li>{@link #preBeginTransaction preBeginTransaction()}</li>
194     *  <li>{@link #beginTransaction beginTransaction()}</li>
195     *  <li>{@link #postBeginTransaction postBeginTransaction()}</li>
196     *  <li>{@link #createSQLKVTransaction createSQLKVTransaction()}</li>
197     *  </ol>
198     * and returns the result.
199     *
200     * @throws KVDatabaseException if an unexpected error occurs
201     * @throws IllegalStateException if no {@link DataSource} is {@linkplain #setDataSource configured}
202     */
203    @Override
204    public SQLKVTransaction createTransaction() {
205        if (this.dataSource == null)
206            throw new IllegalStateException("no DataSource configured");
207        try {
208            final Connection connection = this.createTransactionConnection();
209            this.preBeginTransaction(connection);
210            this.beginTransaction(connection);
211            this.postBeginTransaction(connection);
212            return this.createSQLKVTransaction(connection);
213        } catch (SQLException e) {
214            throw new KVDatabaseException(this, e);
215        }
216    }

224     * </p>
225     *
226     * @return new transaction {@link Connection}
227     * @throws SQLException if an error occurs
228     */

295    /**
296     * Create an SQL statement that reads the value column associated with key <code>&#63;1</code>.
297     *
298     * @return SQL query statement
299     */
300    public String createGetStatement() {
301        return "SELECT " + this.quote(this.valueColumnName) + " FROM "
302          + this.quote(this.tableName) + " WHERE " + this.quote(this.keyColumnName) + " = ?";
303    }

305    /**
306     * Create an SQL statement that reads the key and value columns (in that order) associated
307     * with the smallest key greater than or equal to <code>&#63;1</code>, if any.
308     *
309     * @param reverse true to return rows in descending key order, false to return rows in ascending key order
310     * @return SQL query statement
311     */
312    public String createGetAtLeastStatement(boolean reverse) {
313        return "SELECT " + this.quote(this.keyColumnName) + ", " + this.quote(this.valueColumnName)
314          + " FROM " + this.quote(this.tableName) + " WHERE " + this.quote(this.keyColumnName)
315          + " >= ? ORDER BY " + this.quote(this.keyColumnName) + (reverse ? " DESC" : " ASC");
316    }

323     * @return SQL query statement

328          + " < ? ORDER BY " + this.quote(this.keyColumnName) + (reverse ? " DESC" : " ASC");

450    /**
451     * Wrap the given {@link SQLException} in the appropriate {@link KVTransactionException}.
452     *
453     * @param transaction the {@link SQLKVTransaction} in which the exception occured
454     * @param e SQL exception
455     * @return appropriate {@link KVTransactionException} with chained exception {@code e}
456     * @throws NullPointerException if {@code e} is null
457     */
458    public KVTransactionException wrapException(SQLKVTransaction transaction, SQLException e) {
459        if (e instanceof SQLTimeoutException)
460            return new TransactionTimeoutException(transaction, e);
461        if (e instanceof SQLRecoverableException)
462            return new RetryTransactionException(transaction, e);
463        if (e instanceof SQLTransientException)
464            return new RetryTransactionException(transaction, e);
465        return new KVTransactionException(transaction, e);
466    }
