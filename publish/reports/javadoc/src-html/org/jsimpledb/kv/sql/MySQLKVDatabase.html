008package org.jsimpledb.kv.sql;
009
010import java.sql.Connection;
011import java.sql.SQLException;
012import java.sql.Statement;
013
014/**
015 * MySQL variant of {@link SQLKVDatabase}.
016 *
017 * <p>
018 * An idempotent script for initializing a MySQL database is available at
019 * {@code classpath:org/jsimpledb/kv/sql/createTable-mysql.sql}
020 * (see also {@link org.dellroad.stuff.schema.UpdatingDataSource}).
021 * </p>
022 *
023 */
024public class MySQLKVDatabase extends SQLKVDatabase {
025
026    @Override
027    protected Connection createConnection() throws SQLException {
028        final Connection connection = super.createConnection();
029        final Statement statement = connection.createStatement();
030        statement.execute("SET SESSION sql_mode = 'TRADITIONAL'");      // force error if key or value is too long
031        return connection;
032    }
033
034    @Override
035    public String createGetAtMostStatement() {
036        return super.createGetAtMostStatement() + " LIMIT 1";
037    }
038
039    @Override
040    public String createGetFirstStatement() {
041        return super.createGetFirstStatement() + " LIMIT 1";
042    }
043
044    @Override
045    public String createGetAtLeastStatement() {
046        return super.createGetAtLeastStatement() + " LIMIT 1";
047    }
048
049    @Override
050    public String createGetLastStatement() {
051        return super.createGetLastStatement() + " LIMIT 1";
052    }
053
054    /**
055     * Encloses the given {@code name} in backticks.
056     */
057    @Override
058    public String quote(String name) {
059        return "`" + name + "`";
060    }
061}
