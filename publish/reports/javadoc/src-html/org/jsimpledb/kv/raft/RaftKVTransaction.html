088    /**
089     * Configure whether to reduce the isolation guarantees for this transaction from linearizable consistency
090     * to read-only, "stale snapshot" consistency (technically, serializable consistency).
091     *
092     * <p>
093     * If so configured, this transaction guarantees a consistent, read-only view of the database as it existed
094     * at some point in the "recent past". The view is not guaranteed to be up-to-date; it's only guaranteed
095     * to be as up-to-date as is known to this node when the transaction was opened. For example, it will be
096     * at least as up to date as the most recently committed normal (linearizable) transaction on this node.
097     * However, in general (for example, if stuck in a network partition minority), the snapshot could be arbitrarily
098     * far in the past.
099     *
100     * <p>
101     * This setting may be modified freely during a transaction; it only determines behavior at {@link #commit} time.
102     * If set:
103     *  <ul>
104     *  <li>The transaction will be read-only: modifications will be allowed (and reflected in subsequent reads),
105     *      but they will be discarded on {@link #commit}.</li>
106     *  <li>The {@link #commit} operation will be faster, and possibly not require any network traffic at all.</li>
107     *  </ul>
108     *
109     * @param readOnlySnapshot true for snapshot isolation, false for normal linearizable ACID semantics
110     * @see <a href="https://aphyr.com/posts/313-strong-consistency-models">Strong consistency models</a>
111     */
112    public void setReadOnlySnapshot(boolean readOnlySnapshot) {
113        this.readOnlySnapshot = readOnlySnapshot;
114    }
115
116    /**
117     * Determine whether this instance is configured for read-only, snapshot isolation.
118     *
119     * @return true if this instance is configured for snapshot isolation
120     * @see #setReadOnlySnapshot
121     */
122    public boolean isReadOnlySnapshot() {
123        return this.readOnlySnapshot;
124    }

126// Configuration Stuff

128    /**
129     * Include a cluster configuration change when this transaction is committed.
130     *
131     * <p>
132     * The change will have been applied once this transaction is successfully committed.
133     *
134     * <p>
135     * Raft supports configuration changes that add or remove one node at a time to/from the cluster.
136     * If this method is invoked more than once in a single transaction, all but the last invocation are ignored.
137     *
138     * <p>
139     * Initially, nodes are <i>unconfigured</i>. An unconfigured node becomes configured in one of two ways:
140     * <ul>
141     *  <li>By receiving a message from a leader of some existing cluster, in which case the node joins that cluster
142     *      based on the provided configuration; or</li>
143     *  <li>By this method being invoked with {@code identity} equal to this node's identity and a non-null {@code address},
144     *      which creates a new cluster and adds this node to it.</li>
145     * </ul>
146     *
147     * <p>
148     * Therefore, this method must be used to intialize a new cluster.
149     *
150     * @param identity the identity of the node to add or remove
151     * @param address the network address of the node if adding, or null if removing
152     * @throws IllegalArgumentException if {@code identity} is null
153     */
154    public void configChange(String identity, String address) {
155
156        // Sanity check
157        Preconditions.checkArgument(identity != null, "null identity");
158        if (!this.state.equals(TxState.EXECUTING))
159            throw new StaleTransactionException(this);
160
161        // Set config change
162        this.configChange = new String[] { identity, address };
163    }

172    }

181    /**
182     * Set the commit timeout for this instance.
183     *
184     * <p>
185     * {@link RaftKVTransaction}s do not block while the transaction is open; the configured value is used
186     * as a timeout for the {@link #commit} operation only. If {@link #commit} takes longer than {@code timeout}
187     * milliseconds, a {@link org.jsimpledb.kv.RetryTransactionException} is thrown.
188     *
189     * <p>
190     * The default value for all transactions is configured by {@link RaftKVDatabase#setCommitTimeout}.
191     *
192     * @param timeout transaction commit timeout in milliseconds, or zero for unlimited
193     * @throws IllegalArgumentException if {@code timeout} is negative
194     */
195    @Override
196    public void setTimeout(long timeout) {
197        if (timeout < 0)
198            throw new IllegalArgumentException("timeout < 0");
199        this.timeout = (int)Math.min(timeout, Integer.MAX_VALUE);
200    }

281          + "[txId=" + this.txId
282          + ",state=" + this.state

290    @Override
291    protected void finalize() throws Throwable {
292        try {
293            if (!TxState.CLOSED.equals(this.state))
294               this.log.warn(this + " leaked without commit() or rollback()");
295            this.close();
296        } finally {
297            super.finalize();
298        }
299    }

301// CommitFuture
302
303    static class CommitFuture extends FutureTask<Void> {
304
305        private final ExceptionCallable callable;
306
307        public CommitFuture() {
308            this(new ExceptionCallable());
309        }
310
311        CommitFuture(ExceptionCallable callable) {
312            super(callable);
313            this.callable = callable;
314        }
315
316        public void succeed() {
317            Preconditions.checkState(!this.isDone(), "already done");
318            this.run();
319        }
320
321        public void fail(Exception e) {
322            Preconditions.checkState(!this.isDone(), "already done");
323            this.callable.setException(e);
324            this.run();
325        }
326    }
327
328    private static class ExceptionCallable implements Callable<Void> {
329
330        private volatile Exception e;
331
332        public void setException(Exception e) {
333            this.e = e;
334        }
335
336        @Override
337        public Void call() throws Exception {
338            if (this.e != null)
339                throw this.e;
340            return null;
341        }
342    }

344// Debug/Sanity Checking
345
346    void checkStateOpen(long currentTerm, long lastIndex, long raftCommitIndex) {
347        assert !this.commitFuture.isCancelled();
348        assert this.commitFuture.isDone() == this.state.compareTo(TxState.COMPLETED) >= 0;
349        assert this.baseTerm <= currentTerm;
350        switch (this.state) {
351        case EXECUTING:
352        case COMMIT_READY:
353            assert this.commitTerm == 0;
354            assert this.commitIndex == 0;
355            break;
356        case COMMIT_WAITING:
357            assert this.commitTerm >= this.baseTerm;
358            assert this.commitTerm <= currentTerm;
359            assert this.commitIndex >= this.baseIndex;                                      // equal implies a read-only tx
360            assert this.commitIndex > this.baseIndex || this.view.getWrites().isEmpty();
361            break;
362        case COMPLETED:
363            assert this.commitFuture.isDone();
364            assert this.commitTerm == 0 || this.commitTerm >= this.baseTerm;
365            assert this.commitIndex == 0 || this.commitIndex >= this.baseIndex;
366            assert this.commitTerm <= currentTerm;
367            break;
368        case CLOSED:
369        default:
370            assert false;
371            break;
372        }
373    }
