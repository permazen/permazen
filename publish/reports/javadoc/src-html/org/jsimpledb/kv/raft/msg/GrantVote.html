006package org.jsimpledb.kv.raft.msg;
007
008import java.nio.ByteBuffer;
009
010/**
011 * Sent from a follower to a candidate to grant a vote during an election.
012 */
013public class GrantVote extends Message {
014
015// Constructors
016
017    /**
018     * Constructor.
019     *
020     * @param clusterId cluster ID
021     * @param senderId sending node identity
022     * @param recipientId identity of recipient
023     * @param term current term
024     */
025    public GrantVote(int clusterId, String senderId, String recipientId, long term) {
026        super(Message.GRANT_VOTE_TYPE, clusterId, senderId, recipientId, term);
027    }
028
029    GrantVote(ByteBuffer buf) {
030        super(Message.GRANT_VOTE_TYPE, buf);
031    }
032
033// Message
034
035    @Override
036    public void visit(MessageSwitch handler) {
037        handler.caseGrantVote(this);
038    }
039
040// Object
041
042    @Override
043    public String toString() {
044        return this.getClass().getSimpleName()
045          + "[\"" + this.getSenderId() + "\"->\"" + this.getRecipientId() + "\""
046          + ",clusterId=" + String.format("%08x", this.getClusterId())
047          + ",term=" + this.getTerm()
048          + "]";
049    }
050}
