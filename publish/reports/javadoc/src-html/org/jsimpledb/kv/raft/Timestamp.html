006package org.jsimpledb.kv.raft;
007
008import com.google.common.base.Preconditions;
009
010/**
011 * Represents a relative timestamp in milliseconds. Stored as an unsigned integer that cycles after 2<sup>32</sup>
012 * milliseconds (about 48 days). Two values to be compared must have been generated within 24 days of each other
013 * to be ordered correctly.
014 *
015 * <p>
016 * This class uses {@link System#nanoTime}, not {@link System#currentTimeMillis}, and so is immune to changes in the system clock.
017 *
018 * <p>
019 * Instances are immutable.
020 */
021public class Timestamp implements Comparable<Timestamp> {
022
023    // Make timestamps start at zero to facilitate debugging
024    private static final int TIME_BASE = Timestamp.milliTime();
025
026    private final int millis;
027
028    /**
029     * Construtor returning the current time.
030     */
031    public Timestamp() {
032        this(Timestamp.now());
033    }
034
035    /**
036     * Constructor.
037     *
038     * @param millis relative milliseconds value from {@link #getMillis}
039     */
040    public Timestamp(int millis) {
041        this.millis = millis;
042    }
043
044    /**
045     * Get the relative milliseconds value contained by this instance.
046     *
047     * @return relative millisecond value
048     */
049    public int getMillis() {
050        return this.millis;
051    }
052
053    /**
054     * Get the number of milliseconds this instance is offset from the current time.
055     *
056     * @return relative millisecond offset
057     */
058    public int offsetFromNow() {
059        return this.millis - Timestamp.now();
060    }
061
062    /**
063     * Get the number of milliseconds this instance is offset from the given instance.
064     *
065     * @param base base timestamp
066     * @return relative millisecond offset
067     * @throws IllegalArgumentException if {@code base} is null
068     */
069    public int offsetFrom(Timestamp base) {
070        Preconditions.checkArgument(base != null, "null base");
071        return this.millis - base.millis;
072    }
073
074    /**
075     * Return this timestamp offset by the given amount.
076     *
077     * @param offset offset in milliseconds
078     * @return adjusted timestamp
079     */
080    public Timestamp offset(int offset) {
081        return new Timestamp(this.millis + offset);
082    }
083
084    /**
085     * Determine whether this timestamp is in the past or the future.
086     *
087     * @return true if this timestamp is in the past
088     */
089    public boolean hasOccurred() {
090        return Timestamp.now() - this.millis >= 0;
091    }
092
093    /**
094     * Get a {@link Timestamp} in the distant past. The actual returned value is about 12 days ago.
095     *
096     * @return timestamp value in the distant past
097     */
098    public static Timestamp distantPast() {
099        return new Timestamp(Timestamp.now() - 0x40000000);
100    }
101
102// Internal methods
103
104    private static int now() {
105        return Timestamp.milliTime() - TIME_BASE;
106    }
107
108    private static int milliTime() {
109        return (int)(System.nanoTime() / 1000000L);
110    }
111
112// Object
113
114    @Override
115    public boolean equals(Object obj) {
116        if (obj == null || obj.getClass() != this.getClass())
117            return false;
118        final Timestamp that = (Timestamp)obj;
119        return this.millis == that.millis;
120    }
121
122    @Override
123    public int hashCode() {
124        return this.millis;
125    }
126
127    @Override
128    public String toString() {
129        final long value = this.millis & 0xFFFFFFFFL;
130        return String.format("%05d.%03d", value / 1000, value % 1000);
131    }
132
133// Comparable
134
135    @Override
136    public int compareTo(Timestamp that) {
137        Preconditions.checkArgument(this.millis != (that.millis ^ 0x80000000));
138        final int diff = this.millis - that.millis;
139        return diff < 0 ? -1 : diff > 0 ? 1 : 0;
140    }
141}
