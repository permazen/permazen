032/**
033 * Common superclass for the three roles played by a Raft node:
034 * {@linkplain LeaderRole leader}, {@linkplain FollowerRole follower}, and {@linkplain CandidateRole candidate}.
035 */
036public abstract class Role {
037
038    final Logger log;
039    final RaftKVDatabase raft;
040    final Service checkReadyTransactionsService = new Service(this, "check ready transactions") {
041        @Override
042        public void run() {
043            Role.this.checkReadyTransactions();
044        }
045    };
046    final Service checkWaitingTransactionsService = new Service(this, "check waiting transactions") {
047        @Override
048        public void run() {
049            Role.this.checkWaitingTransactions();
050        }
051    };
052    final Service applyCommittedLogEntriesService = new Service(this, "apply committed logs") {
053        @Override
054        public void run() {
055            Role.this.applyCommittedLogEntries();
056        }
057    };
058    final Service triggerKeyWatchesService = new Service(this, "trigger key watches") {
059        @Override
060        public void run() {
061            Role.this.triggerKeyWatches();
062        }
063    };

065// Constructors
066
067    Role(RaftKVDatabase raft) {
068        this.raft = raft;
069        this.log = this.raft.log;
070        assert Thread.holdsLock(this.raft);
071    }

078     * @return associated database

080    public RaftKVDatabase getKVDatabase() {
081        return this.raft;

084// Lifecycle

086    void setup() {
087        assert Thread.holdsLock(this.raft);
088        this.raft.requestService(this.checkReadyTransactionsService);
089        this.raft.requestService(this.checkWaitingTransactionsService);
090        this.raft.requestService(this.applyCommittedLogEntriesService);
091    }
092
093    void shutdown() {
094        assert Thread.holdsLock(this.raft);
095        for (RaftKVTransaction tx : this.raft.openTransactions.values())
096            this.cleanupForTransaction(tx);
097    }

103    /**
104     * Check transactions in the {@link TxState#COMMIT_READY} state to see if we can advance them.
105     */
106    void checkReadyTransactions() {
107        for (RaftKVTransaction tx : new ArrayList<RaftKVTransaction>(this.raft.openTransactions.values()))
108            new CheckReadyTransactionService(this, tx).run();

111    /**
112     * Check transactions in the {@link TxState#COMMIT_WAITING} state to see if they are committed yet.
113     * We invoke this service method whenever our {@code commitIndex} advances.
114     */
115    void checkWaitingTransactions() {
116        for (RaftKVTransaction tx : new ArrayList<RaftKVTransaction>(this.raft.openTransactions.values()))
117            new CheckWaitingTransactionService(this, tx).run();
118    }

120    /**
121     * Apply committed but unapplied log entries to the state machine.
122     * We invoke this service method whenever log entries are added or our {@code commitIndex} advances.
123     */
124    void applyCommittedLogEntries() {
125
126        // Apply committed log entries to the state machine
127        while (this.raft.lastAppliedIndex < this.raft.commitIndex) {
128
129            // Grab the first unwritten log entry
130            final LogEntry logEntry = this.raft.raftLog.get(0);
131            assert logEntry.getIndex() == this.raft.lastAppliedIndex + 1;
132
133            // Check with subclass
134            if (!this.mayApplyLogEntry(logEntry))
135                break;
136
137            // Get the current config as of the log entry we're about to apply
138            final HashMap<String, String> logEntryConfig = new HashMap<>(this.raft.lastAppliedConfig);
139            logEntry.applyConfigChange(logEntryConfig);
140
141            // Prepare combined Mutations containing prefixed log entry changes plus my own
142            final Writes logWrites = logEntry.getWrites();
143            final Writes myWrites = new Writes();
144            myWrites.getPuts().put(RaftKVDatabase.LAST_APPLIED_TERM_KEY, LongEncoder.encode(logEntry.getTerm()));
145            myWrites.getPuts().put(RaftKVDatabase.LAST_APPLIED_INDEX_KEY, LongEncoder.encode(logEntry.getIndex()));
146            myWrites.getPuts().put(RaftKVDatabase.LAST_APPLIED_CONFIG_KEY, this.raft.encodeConfig(logEntryConfig));
147            final byte[] stateMachinePrefix = this.raft.getStateMachinePrefix();
148            final Mutations mutations = new Mutations() {
149
150                @Override
151                public Iterable<KeyRange> getRemoveRanges() {
152                    return Iterables.transform(logWrites.getRemoveRanges(), new PrefixKeyRangeFunction(stateMachinePrefix));
153                }
154
155                @Override
156                public Iterable<Map.Entry<byte[], byte[]>> getPutPairs() {
157                    return Iterables.concat(
158                      Iterables.transform(logWrites.getPutPairs(), new PrefixPutFunction(stateMachinePrefix)),
159                      myWrites.getPutPairs());
160                }
161
162                @Override
163                public Iterable<Map.Entry<byte[], Long>> getAdjustPairs() {
164                    return Iterables.transform(logWrites.getAdjustPairs(), new PrefixAdjustFunction(stateMachinePrefix));
165                }
166            };
167
168            // Apply updates to the key/value store (durably)
169            if (this.log.isDebugEnabled())
170                this.debug("applying committed log entry " + logEntry + " to key/value store");
171            try {
172                this.raft.kv.mutate(mutations, true);
173            } catch (Exception e) {
174                if (e instanceof RuntimeException && e.getCause() instanceof IOException)
175                    e = (IOException)e.getCause();
176                this.error("error applying log entry " + logEntry + " to key/value store", e);
177                break;
178            }
179
180            // Update in-memory state
181            this.raft.lastAppliedTerm = logEntry.getTerm();
182            assert logEntry.getIndex() == this.raft.lastAppliedIndex + 1;
183            this.raft.lastAppliedIndex = logEntry.getIndex();
184            logEntry.applyConfigChange(this.raft.lastAppliedConfig);
185            assert this.raft.currentConfig.equals(this.raft.buildCurrentConfig());
186
187            // Delete the log entry
188            this.raft.raftLog.remove(0);
189            if (!logEntry.getFile().delete())
190                this.error("failed to delete log file " + logEntry.getFile());
191        }
192    }

194    /**
195     * Determine whether the given log entry may be applied to the state machine.
196     */
197    boolean mayApplyLogEntry(LogEntry logEntry) {
198        return true;
199    }

201    /**
202     * Trigger any key watches for changes in log entries committed since the last time we checked.
203     *
204     * <p>
205     * This should be invoked:
206     * <ul>
207     *  <li>After advancing the commitIndex</li>
208     *  <li>After resetting the state machine</li>
209     *  <li>After installing a snapshot</li>
210     * </ul>
211     */
212    void triggerKeyWatches() {
213
214        // Sanity check
215        assert this.raft.commitIndex >= this.raft.lastAppliedIndex;
216        assert this.raft.commitIndex <= this.raft.lastAppliedIndex + this.raft.raftLog.size();
217        assert this.raft.keyWatchIndex <= this.raft.commitIndex;
218
219        // If we have recevied a snapshot install, we may not be able to tell which keys have changed since last notification;
220        // in that case, trigger all key watches; otherwise, trigger the keys affected by newly committed log entries
221        if (this.raft.keyWatchIndex < this.raft.lastAppliedIndex) {
222            this.raft.keyWatchTracker.triggerAll();
223            this.raft.keyWatchIndex = this.raft.commitIndex;
224        } else {
225            while (this.raft.keyWatchIndex < this.raft.commitIndex)
226                this.raft.keyWatchTracker.trigger(this.raft.getLogEntryAtIndex(++this.raft.keyWatchIndex).getWrites());
227        }
228    }

232    /**
233     * Check a transaction that is ready to be committed (in the {@link TxState#COMMIT_READY} state).
234     *
235     * <p>
236     * This should be invoked:
237     * <ul>
238     *  <li>After changing roles</li>
239     *  <li>After a transaction has entered the {@link TxState#COMMIT_READY} state</li>
240     *  <li>After the leader is newly known (in {@link FollowerRole})</li>
241     *  <li>After the leader's output queue goes from non-empty to empty (in {@link FollowerRole})</li>
242     *  <li>After the leader's {@code commitIndex} has advanced, in case a config change transaction
243     *      is waiting on a previous config change transaction (in {@link LeaderRole})</li>
244     * </ul>
245     *
246     * @param tx the transaction
247     * @throws KVTransactionException if an error occurs
248     */
249    void checkReadyTransaction(RaftKVTransaction tx) {
250
251        // Get transaction mutations
252        final Writes writes = tx.getMutableView().getWrites();
253        final String[] configChange = tx.getConfigChange();
254
255        // Determine whether transaction is truly read-only
256        final boolean readOnly = tx.isReadOnly() || (writes.isEmpty() && configChange == null);
257
258        // No need to talk to leader for read-only transactions unless LINEARIZABLE
259        if (readOnly) {
260            switch (tx.getConsistency()) {
261            case UNCOMMITTED:
262                if (this.log.isTraceEnabled())
263                    this.trace("trivial commit for read-only, UNCOMMITTED " + tx);
264                this.raft.succeed(tx);
265                return;
266            case EVENTUAL:
267                this.advanceReadyTransaction(tx, tx.getBaseTerm(), tx.getBaseIndex());
268                return;
269            default:
270                break;
271            }
272        }
273
274        // Requires leader communication - let subclass handle it
275        this.checkReadyLeaderTransaction(tx, readOnly);
276    }

278    /**
279     * Check a transaction that is ready to be committed (in the {@link TxState#COMMIT_READY} state)
280     * and requires communication with the leader.
281     *
282     * @param tx the transaction
283     * @param readOnly if transaction is read-only
284     * @throws KVTransactionException if an error occurs
285     */
286    abstract void checkReadyLeaderTransaction(RaftKVTransaction tx, boolean readOnly);

288    /**
289     * Advance a transaction from the {@link TxState#COMMIT_READY} state to the {@link TxState#COMMIT_WAITING} state.
290     *
291     * @param tx the transaction
292     */
293    void advanceReadyTransaction(RaftKVTransaction tx, long commitTerm, long commitIndex) {
294
295        // Sanity check
296        assert tx.getState().equals(TxState.COMMIT_READY);
297
298        // Set commit term & index and update state
299        if (this.log.isDebugEnabled())
300            this.debug("advancing " + tx + " to " + TxState.COMMIT_WAITING + " with commit " + commitIndex + "t" + commitTerm);
301        tx.setCommitTerm(commitTerm);
302        tx.setCommitIndex(commitIndex);
303        tx.setState(TxState.COMMIT_WAITING);
304
305        // Discard information we no longer need
306        tx.getMutableView().disableReadTracking();
307
308        // Check this transaction to see if it can be committed
309        this.raft.requestService(this.checkWaitingTransactionsService);
310    }
311
312    /**
313     * Check a transaction waiting for its log entry to be committed (in the {@link TxState#COMMIT_WAITING} state).
314     *
315     * <p>
316     * This should be invoked:
317     * <ul>
318     *  <li>After changing roles</li>
319     *  <li>After a transaction has entered the {@link TxState#COMMIT_WAITING} state</li>
320     *  <li>After advancing my {@code commitIndex} (as leader or follower)</li>
321     *  <li>After receiving an updated {@linkplain AppendResponse#getLeaderLeaseTimeout leader lease timeout}
322     *      (in {@link FollowerRole})</li>
323     * </ul>
324     *
325     * @param tx the transaction
326     * @throws KVTransactionException if an error occurs
327     */
328    void checkWaitingTransaction(RaftKVTransaction tx) {
329
330        // Handle the case the transaction's committed log index has already been applied to the state machine
331        final long commitIndex = tx.getCommitIndex();
332        if (commitIndex < this.raft.lastAppliedIndex) {
333
334            // This can happen if we lose contact and by the time we're back the log entry has
335            // already been applied to the state machine on some leader and that leader sent
336            // use an InstallSnapshot message. We don't know whether it actually got committed
337            // or not, so the transaction must be retried.
338            throw new RetryTransactionException(tx, "committed log entry was missed");
339        }
340
341        // Has the transaction's log entry been received and committed yet?
342        if (commitIndex > this.raft.commitIndex)
343            return;
344
345        // Verify the term of the committed log entry; if not what we expect, the log entry was overwritten by a new leader
346        final long commitTerm = tx.getCommitTerm();
347        if (this.raft.getLogTermAtIndex(commitIndex) != commitTerm)
348            throw new RetryTransactionException(tx, "leader was deposed during commit and transaction's log entry overwritten");
349
350        // Check with subclass
351        if (!this.mayCommit(tx))
352            return;
353
354        // Transaction is officially committed now
355        if (this.log.isTraceEnabled())
356            this.trace("commit successful for " + tx + " (commit index " + this.raft.commitIndex + " >= " + commitIndex + ")");
357        this.raft.succeed(tx);
358    }
