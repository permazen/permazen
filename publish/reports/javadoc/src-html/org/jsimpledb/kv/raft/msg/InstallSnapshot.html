006package org.jsimpledb.kv.raft.msg;
007
008import com.google.common.base.Preconditions;
009
010import java.nio.ByteBuffer;
011import java.util.HashMap;
012import java.util.Map;
013
014import org.jsimpledb.util.LongEncoder;
015
016/**
017 * Sent from leader to follower to with a chunk of key/value pairs that will wholesale replace the follower's key/value store.
018 */
019public class InstallSnapshot extends Message {
020
021    private final long snapshotTerm;
022    private final long snapshotIndex;
023    private final Map<String, String> snapshotConfig;
024    private final long pairIndex;
025    private final boolean lastChunk;
026    private final ByteBuffer data;
027
028// Constructors
029
030    /**
031     * Constructor.
032     *
033     * @param clusterId cluster ID
034     * @param senderId identity of sender
035     * @param recipientId identity of recipient
036     * @param term sender's current term
037     * @param snapshotTerm term of the last log entry in the snapshot
038     * @param snapshotIndex index of the last log entry in the snapshot
039     * @param snapshotConfig cluster config of the last log entry in the snapshot (first {@code pairIndex} only)
040     * @param pairIndex index of the first key/value pair in this chunk
041     * @param lastChunk true if this is the last chunk in the snapshot
042     * @param data encoded key/value pairs
043     */
044    public InstallSnapshot(int clusterId, String senderId, String recipientId, long term, long snapshotTerm,
045      long snapshotIndex, long pairIndex, Map<String, String> snapshotConfig, boolean lastChunk, ByteBuffer data) {
046        super(Message.INSTALL_SNAPSHOT_TYPE, clusterId, senderId, recipientId, term);
047        this.snapshotTerm = snapshotTerm;
048        this.snapshotIndex = snapshotIndex;
049        this.pairIndex = pairIndex;
050        this.snapshotConfig = snapshotConfig;
051        this.lastChunk = lastChunk;
052        this.data = data;
053        this.checkArguments();
054    }
055
056    InstallSnapshot(ByteBuffer buf) {
057        super(Message.INSTALL_SNAPSHOT_TYPE, buf);
058        this.snapshotTerm = LongEncoder.read(buf);
059        this.snapshotIndex = LongEncoder.read(buf);
060        this.pairIndex = LongEncoder.read(buf);
061        this.snapshotConfig = this.pairIndex == 0 ? InstallSnapshot.getSnapshotConfig(buf) : null;
062        this.lastChunk = Message.getBoolean(buf);
063        this.data = Message.getByteBuffer(buf);
064        this.checkArguments();
065    }
066
067    @Override
068    void checkArguments() {
069        super.checkArguments();
070        Preconditions.checkArgument(this.snapshotTerm > 0);
071        Preconditions.checkArgument(this.snapshotIndex > 0);
072        Preconditions.checkArgument(this.pairIndex >= 0);
073        Preconditions.checkArgument((this.pairIndex == 0) == (this.snapshotConfig != null));
074        Preconditions.checkArgument(this.data != null);
075    }
076
077// Properties
078
079    public long getSnapshotTerm() {
080        return this.snapshotTerm;
081    }
082
083    public long getSnapshotIndex() {
084        return this.snapshotIndex;
085    }
086
087    public Map<String, String> getSnapshotConfig() {
088        return this.snapshotConfig;
089    }
090
091    public long getPairIndex() {
092        return this.pairIndex;
093    }
094
095    public boolean isLastChunk() {
096        return this.lastChunk;
097    }
098
099    public ByteBuffer getData() {
100        return this.data.asReadOnlyBuffer();
101    }
102
103// Message
104
105    @Override
106    public boolean isLeaderMessage() {
107        return true;
108    }
109
110    @Override
111    public void visit(MessageSwitch handler) {
112        handler.caseInstallSnapshot(this);
113    }
114
115    @Override
116    public void writeTo(ByteBuffer dest) {
117        super.writeTo(dest);
118        LongEncoder.write(dest, this.snapshotTerm);
119        LongEncoder.write(dest, this.snapshotIndex);
120        LongEncoder.write(dest, this.pairIndex);
121        if (this.pairIndex == 0)
122            InstallSnapshot.putSnapshotConfig(dest, this.snapshotConfig);
123        Message.putBoolean(dest, this.lastChunk);
124        Message.putByteBuffer(dest, this.data);
125    }
126
127    @Override
128    protected int calculateSize() {
129        return super.calculateSize()
130          + LongEncoder.encodeLength(this.snapshotTerm)
131          + LongEncoder.encodeLength(this.snapshotIndex)
132          + LongEncoder.encodeLength(this.pairIndex)
133          + (this.pairIndex == 0 ? InstallSnapshot.calculateSize(this.snapshotConfig) : 0)
134          + 1
135          + Message.calculateSize(this.data);
136    }
137
138    private static Map<String, String> getSnapshotConfig(ByteBuffer buf) {
139        final int count = (int)LongEncoder.read(buf);
140        final HashMap<String, String> config = new HashMap<>(count);
141        for (int i = 0; i < count; i++)
142            config.put(Message.getString(buf), Message.getString(buf));
143        return config;
144    }
145
146    private static void putSnapshotConfig(ByteBuffer dest, Map<String, String> config) {
147        final int count = config.size();
148        LongEncoder.write(dest, count);
149        for (Map.Entry<String, String> entry : config.entrySet()) {
150            Message.putString(dest, entry.getKey());
151            Message.putString(dest, entry.getValue());
152        }
153    }
154
155    private static int calculateSize(Map<String, String> config) {
156        int total = LongEncoder.encodeLength(config.size());
157        for (Map.Entry<String, String> entry : config.entrySet())
158            total += Message.calculateSize(entry.getKey()) + Message.calculateSize(entry.getValue());
159        return total;
160    }
161
162// Object
163
164    @Override
165    public String toString() {
166        return this.getClass().getSimpleName()
167          + "[\"" + this.getSenderId() + "\"->\"" + this.getRecipientId() + "\""
168          + ",clusterId=" + String.format("%08x", this.getClusterId())
169          + ",term=" + this.getTerm()
170          + ",snapshotTerm=" + this.snapshotTerm
171          + ",snapshotIndex=" + this.snapshotIndex
172          + ",pairIndex=" + this.pairIndex
173          + (this.snapshotConfig != null ? ",snapshotConfig=" + this.snapshotConfig : "")
174          + ",lastChunk=" + this.lastChunk
175          + ",data=" + this.describe(this.data)
176          + "]";
177    }
178}
