026/**
027 * Support superclass for Raft messages.
028 */
029public abstract class Message {
030
031    // Codes identifying serialized message type
032    static final byte APPEND_REQUEST_TYPE = 1;
033    static final byte APPEND_RESPONSE_TYPE = 2;
034    static final byte COMMIT_REQUEST_TYPE = 3;
035    static final byte COMMIT_RESPONSE_TYPE = 4;
036    static final byte GRANT_VOTE_TYPE = 5;
037    static final byte INSTALL_SNAPSHOT_TYPE = 6;
038    static final byte REQUEST_VOTE_TYPE = 7;
039
040    // Serialization version number
041    private static final byte VERSION_1 = 1;
042
043    // Minimum buffer size to use a direct buffer
044    private static final int MIN_DIRECT_BUFFER_SIZE = 128;
045
046    private final byte type;
047    private final int clusterId;
048    private final String senderId;
049    private final String recipientId;
050    private final long term;
051
052    protected Message(byte type, int clusterId, String senderId, String recipientId, long term) {
053        this.type = type;
054        this.clusterId = clusterId;
055        this.senderId = senderId;
056        this.recipientId = recipientId;
057        this.term = term;
058    }
059
060    protected Message(byte type, ByteBuffer buf) {
061        this.type = type;
062        this.clusterId = buf.getInt();
063        this.senderId = Message.getString(buf);
064        this.recipientId = Message.getString(buf);
065        this.term = LongEncoder.read(buf);
066    }
067
068    void checkArguments() {
069        Preconditions.checkArgument(this.type >= APPEND_REQUEST_TYPE && this.type <= REQUEST_VOTE_TYPE);
070        Preconditions.checkArgument(this.clusterId != 0);
071        Preconditions.checkArgument(this.senderId != null);
072        Preconditions.checkArgument(this.recipientId != null);
073        Preconditions.checkArgument(this.term > 0);
074    }

132// (De)serialization methods

134    /**
135     * Decode a message from the given input.
136     *
137     * <p>
138     * Note that data is not necessarily copied out of {@code buf}, so the returned instance may become invalid
139     * if the data in {@code buf} gets overwritten.
140     *
141     * @param buf source for encoded message
142     * @return decoded message
143     * @throws java.nio.BufferUnderflowException if there is not enough data
144     * @throws IllegalArgumentException if encoded message is bogus
145     * @throws IllegalArgumentException if there is trailing garbage
146     */
147    public static Message decode(ByteBuffer buf) {
148
149        // Check encoding format version
150        final byte version = buf.get();
151        switch (version) {
152        case Message.VERSION_1:
153            break;
154        default:
155            throw new IllegalArgumentException("unrecognized message format version " + version);
156        }
157
158        // Read type and decode message
159        final Message msg;
160        final byte type = buf.get();
161        switch (type) {
162        case APPEND_REQUEST_TYPE:
163            msg = new AppendRequest(buf);
164            break;
165        case APPEND_RESPONSE_TYPE:
166            msg = new AppendResponse(buf);
167            break;
168        case COMMIT_REQUEST_TYPE:
169            msg = new CommitRequest(buf);
170            break;
171        case COMMIT_RESPONSE_TYPE:
172            msg = new CommitResponse(buf);
173            break;
174        case GRANT_VOTE_TYPE:
175            msg = new GrantVote(buf);
176            break;
177        case INSTALL_SNAPSHOT_TYPE:
178            msg = new InstallSnapshot(buf);
179            break;
180        case REQUEST_VOTE_TYPE:
181            msg = new RequestVote(buf);
182            break;
183        default:
184            throw new IllegalArgumentException("invalid message type " + type);
185        }
186
187        // Check for trailing garbage
188        if (buf.hasRemaining())
189            throw new IllegalArgumentException("buffer contains " + buf.remaining() + " bytes of extra garbage after " + msg);
190
191        // Done
192        return msg;
193    }
194
195    /**
196     * Serialize this instance.
197     *
198     * @return encoded message
199     */
200    public ByteBuffer encode() {
201        final int size = this.calculateSize();
202        final ByteBuffer buf = size >= MIN_DIRECT_BUFFER_SIZE ? ByteBuffer.allocateDirect(size) : ByteBuffer.allocate(size);
203        this.writeTo(buf);
204        if (buf.hasRemaining())
205            throw new RuntimeException("internal error: " + buf.remaining() + " remaining bytes in buffer from " + this);
206        return (ByteBuffer)buf.flip();
207    }
208
209    /**
210     * Serialize this instance into the given buffer.
211     *
212     * @param buf destination for encoded data
213     * @throws java.nio.BufferOverflowException if data overflows {@code buf}
214     */
215    public void writeTo(ByteBuffer buf) {
216        buf.put(Message.VERSION_1);
217        buf.put(this.type);
218        buf.putInt(this.clusterId);
219        Message.putString(buf, this.senderId);
220        Message.putString(buf, this.recipientId);
221        LongEncoder.write(buf, this.term);
222    }
223
224    /**
225     * Calculate an upper bound on the number of bytes required by {@link #writeTo writeTo()}.
226     *
227     * @return an upper bound on the number of encoded bytes
228     */
229    protected int calculateSize() {
230        return 1
231          + 1
232          + 4
233          + Message.calculateSize(this.senderId)
234          + Message.calculateSize(this.recipientId)
235          + LongEncoder.encodeLength(this.term);
236    }

245    /**
246     * Serialize a {@link ByteBuffer} into the buffer.
247     *
248     * @param dest destination for encoded data
249     * @param buf data to encode
250     * @throws ReadOnlyBufferException if {@code dest} is read only
251     * @throws BufferOverflowException if {@code dest} overflows
252     * @throws IllegalArgumentException if {@code buf} has more than 2<sup>^31</sup> bytes remaining
253     * @throws IllegalArgumentException if either parameter is null
254     */
255    protected static void putByteBuffer(ByteBuffer dest, ByteBuffer buf) {
256        Preconditions.checkArgument(dest != null, "null dest");
257        Preconditions.checkArgument(buf != null, "null buf");
258        UnsignedIntEncoder.write(dest, buf.remaining());
259        dest.put(buf.asReadOnlyBuffer());
260    }
261
262    /**
263     * Deserialize a {@link ByteBuffer} previously serialized by {@link #putByteBuffer putByteBuffer()} from the buffer.
264     *
265     * @param buf source for encoded data
266     * @return decoded data
267     * @throws BufferUnderflowException if {@code buf} underflows
268     * @throws IllegalArgumentException if input is bogus
269     * @throws IllegalArgumentException if {@code buf} is null
270     */
271    protected static ByteBuffer getByteBuffer(ByteBuffer buf) {
272        Preconditions.checkArgument(buf != null, "null buf");
273        final int numBytes = UnsignedIntEncoder.read(buf);
274        if (numBytes > buf.remaining())
275            throw new IllegalArgumentException("bogus buffer length " + numBytes + " > " + buf.remaining());
276        final ByteBuffer result = (ByteBuffer)buf.slice().limit(numBytes);
277        buf.position(buf.position() + numBytes);
278        return result;
279    }
280
281    protected static int calculateSize(ByteBuffer buf) {
282        return UnsignedIntEncoder.encodeLength(buf.remaining()) + buf.remaining();
283    }
284
285    /**
286     * Serialize a {@link String} into the buffer.
287     *
288     * @param dest destination for encoded data
289     * @param string string to encode
290     * @throws ReadOnlyBufferException if {@code dest} is read only
291     * @throws BufferOverflowException if {@code dest} overflows
292     * @throws IllegalArgumentException if either parameter is null
293     */
294    protected static void putString(ByteBuffer dest, String string) {
295        Preconditions.checkArgument(dest != null, "null dest");
296        Preconditions.checkArgument(string != null, "null string");
297        for (int i = 0; i < string.length(); i++) {
298            final int c = string.charAt(i);
299            if (c >= 0x0001 && c <= 0x007f)
300                dest.put((byte)c);
301            else if (c < 0x0800) {
302                dest.put((byte)(0xc0 | ((c >> 6) & 0x1f)));
303                dest.put((byte)(0x80 | ((c >> 0) & 0x3f)));
304            } else {
305                dest.put((byte)(0xe0 | ((c >> 12) & 0x0f)));
306                dest.put((byte)(0x80 | ((c >>  6) & 0x3f)));
307                dest.put((byte)(0x80 | ((c >>  0) & 0x3f)));
308            }
309        }
310        dest.put((byte)0);
311    }
312
313    /**
314     * Deserialize a {@link String} previously serialized by {@link #putString putString()} from the buffer.
315     *
316     * @param buf source for encoded data
317     * @return decoded string, never null
318     * @throws BufferUnderflowException if {@code buf} underflows
319     * @throws IllegalArgumentException if input is bogus
320     */
321    protected static String getString(ByteBuffer buf) {
322        Preconditions.checkArgument(buf != null, "null buf");
323        final StringWriter writer = new StringWriter();
324        while (true) {
325            final int b1 = buf.get() & 0xff;
326            if (b1 == 0)
327                break;
328            if ((b1 & 0x80) == 0)
329                writer.append((char)b1);
330            else if ((b1 & 0xe0) == 0xc0) {
331                final int b2 = buf.get() & 0xff;
332                if ((b2 & 0xc0) != 0x80)
333                    throw new IllegalArgumentException("invalid UTF-8 sequence: " + b1 + " " + b2);
334                writer.append((char)((b1 & 0x1f) << 6 | (b2 & 0x3f)));
335            } else if ((b1 & 0xf0) == 0xe0) {
336                final int b2 = buf.get() & 0xff;
337                final int b3 = buf.get() & 0xff;
338                if ((b2 & 0xc0) != 0x80 || (b3 & 0xc0) != 0x80)
339                    throw new IllegalArgumentException("invalid UTF-8 sequence: " + b1 + " " + b2 + " " + b3);
340                writer.append((char)(((b1 & 0x0f) << 12) | ((b2 & 0x3f) << 6) | (b3 & 0x3f)));
341            } else
342                throw new IllegalArgumentException("invalid UTF-8 sequence: " + b1);
343        }
344        return writer.toString();
345    }
346
347    protected static int calculateSize(String string) {
348        Preconditions.checkArgument(string != null, "null string");
349        int total = 1;
350        for (int i = 0; i < string.length(); i++) {
351            final int ch = string.charAt(i);
352            total += (ch != 0x0000 && ch < 0x0080) ? 1 : ch < 0x0800 ? 2 : 3;
353        }
354        return total;
355    }

357    /**
358     * Serialize a boolean value into the buffer.
359     *
360     * @param dest destination for encoded data
361     * @param value value to encode
362     * @throws ReadOnlyBufferException if {@code dest} is read only
363     * @throws BufferOverflowException if {@code dest} overflows
364     * @throws IllegalArgumentException if {@code dest} is null
365     */
366    protected static void putBoolean(ByteBuffer dest, boolean value) {
367        Preconditions.checkArgument(dest != null, "null dest");
368        dest.put(value ? (byte)1 : (byte)0);
369    }

372     * Deserialize a boolean value previously serialized by {@link #putBoolean putBoolean()} from the buffer.
373     *
374     * @param buf source for encoded data
375     * @return decoded value
376     * @throws BufferUnderflowException if {@code buf} underflows
377     * @throws IllegalArgumentException if input is bogus
378     */
379    protected static boolean getBoolean(ByteBuffer buf) {
380        Preconditions.checkArgument(buf != null, "null buf");
381        switch (buf.get()) {
382        case (byte)0:
383            return false;
384        case (byte)1:
385            return true;
386        default:
387            throw new IllegalArgumentException("read invalid boolean value");
388        }
389    }

423    /**
424     * Serialize a {@link Reads} into the buffer.
425     *
426     * @param dest destination for encoded data
427     * @param reads value to encode
428     * @throws ReadOnlyBufferException if {@code dest} is read only
429     * @throws BufferOverflowException if {@code dest} overflows
430     * @throws IllegalArgumentException if either parameter is null
431     */
432    protected static void putReads(ByteBuffer dest, Reads reads) {
433        Preconditions.checkArgument(dest != null, "null dest");
434        Preconditions.checkArgument(reads != null, "null reads");
435        final ByteBufferOutputStream output = new ByteBufferOutputStream(dest);
436        try {
437            reads.serialize(output);
438            output.flush();
439        } catch (IOException e) {
440            if (e.getCause() instanceof BufferOverflowException)
441                throw (BufferOverflowException)e.getCause();
442            if (e.getCause() instanceof ReadOnlyBufferException)
443                throw (ReadOnlyBufferException)e.getCause();
444            throw new RuntimeException("unexpected exception", e);
445        }
446    }

511// Debugging

513    String describe(ByteBuffer buf) {
514        if (buf == null)
515            return null;
516        final int size = buf.remaining();
517        if (size <= 32) {
518            final byte[] data = new byte[size];
519            buf.asReadOnlyBuffer().get(data);
520            return ByteUtil.toString(data);
521        }
522        return size + " bytes";
523    }
