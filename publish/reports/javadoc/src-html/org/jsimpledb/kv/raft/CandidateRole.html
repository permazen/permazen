006package org.jsimpledb.kv.raft;
007
008import java.util.HashSet;
009
010import org.jsimpledb.kv.raft.msg.AppendRequest;
011import org.jsimpledb.kv.raft.msg.CommitResponse;
012import org.jsimpledb.kv.raft.msg.GrantVote;
013import org.jsimpledb.kv.raft.msg.InstallSnapshot;
014import org.jsimpledb.kv.raft.msg.RequestVote;
015
016/**
017 * Raft candidate role.
018 */
019public class CandidateRole extends NonLeaderRole {
020
021    private final HashSet<String> votes = new HashSet<>();
022    private final Service checkElectionResultService = new Service(this, "check election result") {
023        @Override
024        public void run() {
025            CandidateRole.this.checkElectionResult();
026        }
027    };
028
029// Constructors
030
031    CandidateRole(RaftKVDatabase raft) {
032        super(raft, true);
033    }
034
035// Status
036
037    /**
038     * Get the number of votes required to win the election.
039     *
040     * @return required votes
041     */
042    public int getVotesRequired() {
043        synchronized (this.raft) {
044            return this.raft.currentConfig.size() / 2 + 1;
045        }
046    }
047
048    /**
049     * Get the number of votes received so far. Includes this node's vote.
050     *
051     * @return received votes
052     */
053    public int getVotesReceived() {
054        synchronized (this.raft) {
055            return this.votes.size() + (this.raft.isClusterMember() ? 1 : 0);
056        }
057    }
058
059// Lifecycle
060
061    @Override
062    void setup() {
063        super.setup();
064
065        // Increment term
066        if (!this.raft.advanceTerm(this.raft.currentTerm + 1))
067            return;
068
069        // Request votes from other peers
070        final HashSet<String> voters = new HashSet<>(this.raft.currentConfig.keySet());
071        voters.remove(this.raft.identity);
072        if (this.log.isDebugEnabled())
073            this.debug("entering candidate role in term " + this.raft.currentTerm + "; requesting votes from " + voters);
074        for (String voter : voters) {
075            this.raft.sendMessage(new RequestVote(this.raft.clusterId, this.raft.identity, voter,
076              this.raft.currentTerm, this.raft.getLastLogTerm(), this.raft.getLastLogIndex()));
077        }
078
079        // Check election result - needed in case we are the only node in the cluster
080        this.raft.requestService(this.checkElectionResultService);
081    }
082
083// Service
084
085    @Override
086    void outputQueueEmpty(String address) {
087        // nothing to do
088    }
089
090    @Override
091    void handleElectionTimeout() {
092        this.raft.changeRole(new CandidateRole(this.raft));
093    }
094
095    private void checkElectionResult() {
096
097        // Tally votes
098        final int allVotes = this.raft.currentConfig.size();
099        final int numVotes = this.getVotesReceived();
100        final int votesRequired = this.getVotesRequired();
101        if (this.log.isDebugEnabled())
102            this.debug("current election tally: " + numVotes + "/" + allVotes + " with " + votesRequired + " required to win");
103
104        // Did we win?
105        if (numVotes >= votesRequired) {
106            if (this.log.isDebugEnabled())
107                this.debug("won the election for term " + this.raft.currentTerm + "; becoming leader");
108            this.raft.changeRole(new LeaderRole(this.raft));
109        }
110    }
111
112// Transactions
113
114    @Override
115    void checkReadyLeaderTransaction(RaftKVTransaction tx, boolean readOnly) {
116
117        // Sanity check
118        assert Thread.holdsLock(this.raft);
119        assert tx.getState().equals(TxState.COMMIT_READY);
120
121        // We can't do anything because we don't have a leader yet
122    }
123
124// MessageSwitch
125
126    @Override
127    void caseAppendRequest(AppendRequest msg) {
128        if (this.log.isDebugEnabled())
129            this.debug("rec'd " + msg + " in " + this + "; reverting to follower");
130        this.raft.changeRole(new FollowerRole(this.raft, msg.getSenderId(), this.raft.returnAddress));
131        this.raft.receiveMessage(this.raft.returnAddress, msg);
132    }
133
134// MessageSwitch
135
136    @Override
137    void caseCommitResponse(CommitResponse msg) {
138        this.failUnexpectedMessage(msg);                    // we could not have ever sent a CommitRequest in this term
139    }
140
141    @Override
142    void caseInstallSnapshot(InstallSnapshot msg) {
143        this.failUnexpectedMessage(msg);                    // we could not have ever sent an AppendResponse in this term
144    }
145
146    @Override
147    void caseRequestVote(RequestVote msg) {
148
149        // Ignore - we are also a candidate and have already voted for ourself
150        if (this.log.isDebugEnabled())
151            this.debug("ignoring " + msg + " rec'd while in " + this);
152    }
153
154    @Override
155    void caseGrantVote(GrantVote msg) {
156
157        // Record vote
158        this.votes.add(msg.getSenderId());
159        if (this.log.isDebugEnabled())
160            this.debug("rec'd election vote from \"" + msg.getSenderId() + "\" in term " + this.raft.currentTerm);
161
162        // Check election result
163        this.raft.requestService(this.checkElectionResultService);
164    }
165
166// Object
167
168    @Override
169    public String toString() {
170        return this.toStringPrefix()
171          + ",votes=" + this.votes
172          + "]";
173    }
174
175// Debug
176
177    @Override
178    boolean checkState() {
179        if (!super.checkState())
180            return false;
181        assert this.electionTimer.isRunning();
182        assert this.raft.isClusterMember();
183        return true;
184    }
185}
