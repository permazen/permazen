006package org.jsimpledb.kv.raft.msg;
007
008import java.nio.ByteBuffer;
009
010import org.jsimpledb.kv.raft.Timestamp;
011
012/**
013 * Sent from hermits to other nodes when trying to establish communication with a majority.
014 */
015public class PingRequest extends AbstractPingMessage {
016
017// Constructors
018
019    /**
020     * Constructor.
021     *
022     * @param timestamp request timestamp
023     */
024    public PingRequest(int clusterId, String senderId, String recipientId, long term, Timestamp timestamp) {
025        super(Message.PING_REQUEST_TYPE, clusterId, senderId, recipientId, term, timestamp);
026        this.checkArguments();
027    }
028
029    PingRequest(ByteBuffer buf) {
030        super(Message.PING_REQUEST_TYPE, buf);
031        this.checkArguments();
032    }
033
034// Message
035
036    @Override
037    public void visit(MessageSwitch handler) {
038        handler.casePingRequest(this);
039    }
040}
