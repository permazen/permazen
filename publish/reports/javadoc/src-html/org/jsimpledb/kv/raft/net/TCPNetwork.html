027/**
028 * {@link Network} implementation based on TCP sockets.
029 *
030 * <p>
031 * Remote peers have the {@link String} form <code>IP-Address[:port]</code>. If the port is omitted,
032 * {@link #DEFAULT_TCP_PORT} is assumed.

190    @Override
191    public synchronized void start(Handler handler) throws IOException {
192        super.start();
193        boolean successful = false;
194        try {
195            if (this.handler != null)
196                return;
197            if (this.log.isDebugEnabled())
198                this.log.debug("starting " + this + " listening on " + this.listenAddress);
199            this.serverSocketChannel = ServerSocketChannel.open();
200            this.configureServerSocketChannel(this.serverSocketChannel);
201            this.serverSocketChannel.bind(this.listenAddress);
202            this.selectionKey = this.createSelectionKey(this.serverSocketChannel, new IOHandler() {
203                @Override
204                public void serviceIO(SelectionKey key) throws IOException {
205                    if (key.isAcceptable())
206                        TCPNetwork.this.handleAccept();
207                }
208                @Override
209                public void close(Throwable cause) {
210                    TCPNetwork.this.log.error("stopping " + this + " due to exception", cause);
211                    TCPNetwork.this.stop();
212                }
213            });
214            this.selectForAccept(true);
215            this.executor = Executors.newSingleThreadExecutor();
216            this.handler = handler;
217            successful = true;
218        } finally {
219            if (!successful)
220                this.stop();
221        }
222    }
223
224    @Override
225    public void stop() {
226        super.stop();
227        synchronized (this) {
228            if (this.handler == null)
229                return;
230            if (this.log.isDebugEnabled())
231                this.log.debug("stopping " + this);
232            if (this.serverSocketChannel != null) {
233                try {
234                    this.serverSocketChannel.close();
235                } catch (Exception e) {
236                    // ignore
237                }
238                this.serverSocketChannel = null;
239            }
240            if (this.executor != null) {
241                this.executor.shutdownNow();
242                try {
243                    this.executor.awaitTermination(1000, TimeUnit.MILLISECONDS);
244                } catch (InterruptedException e) {
245                    Thread.currentThread().interrupt();
246                }
247                this.executor = null;
248            }
249            this.selectionKey = null;
250            this.handler = null;
251        }
252    }

256    @Override
257    public synchronized boolean send(String peer, ByteBuffer msg) {
258
259        // Sanity check
260        Preconditions.checkArgument(peer != null, "null peer");
261
262        // Get/create connection
263        Connection connection = this.connectionMap.get(peer);
264        if (connection == null) {
265
266            // Create connection
267            try {
268                connection = this.createConnection(peer);
269            } catch (IOException e) {
270                this.log.info(this + " unable to send message to `" + peer + "': " + e.getMessage());
271                return false;
272            }
273
274            // Record connection
275            this.connectionMap.put(peer, connection);
276        }
277
278        // Send message
279        return connection.output(msg);
280    }

284    /**
285     * Parse out the address part of an address that has an optional colon plus TCP port number suffix.
286     *
287     * @param address address of the form {@code ipaddr} or {@code ipaddr:port}
288     * @return the IP address part
289     */
290    public static String parseAddressPart(String address) {
291        return (String)TCPNetwork.parseAddress(address, 0)[0];
292    }
293
294    /**
295     * Parse out the port part of an address that has an optional colon plus TCP port number suffix.
296     *
297     * @param address address of the form {@code ipaddr} or {@code ipaddr:port}
298     * @param defaultPort default port if none specified in {@code address}
299     * @return the port part, or {@code defaultPort} if there is no explicit port
300     */
301    public static int parsePortPart(String address, int defaultPort) {
302        return (Integer)TCPNetwork.parseAddress(address, defaultPort)[1];
303    }
304
305    private static Object[] parseAddress(String string, int defaultPort) {
306        final int colon = string.lastIndexOf(':');
307        if (colon == -1)
308            return new Object[] { string, defaultPort };
309        try {
310            final int port = Integer.parseInt(string.substring(colon + 1), 10);
311            if (port < 1 || port > 65535)
312                return new Object[] { string, defaultPort };
313            return new Object[] { string.substring(0, colon), port };
314        } catch (Exception e) {
315            return new Object[] { string, defaultPort };
316        }
317    }

353// I/O Ready Conditions

355    // Invoked when we have a new incoming connection
356    private void handleAccept() throws IOException {
357
358        // Sanity check
359        assert this.isServiceThread();
360
361        // Check connection size limit
362        if (this.connectionMap.size() >= this.maxConnections) {
363            this.log.warn("too many network connections (" + this.connectionMap.size() + " >= "
364              + this.maxConnections + "), not accepting any more (for now)");
365            this.selectForAccept(false);
366            return;
367        }
368
369        // Accept connection
370        SocketChannel socketChannel = this.serverSocketChannel.accept();
371        if (socketChannel == null)
372            return;
373        this.log.info("accepted incoming connection from " + socketChannel.getRemoteAddress());
374        socketChannel
375          .setOption(StandardSocketOptions.SO_KEEPALIVE, true)
376          .setOption(StandardSocketOptions.TCP_NODELAY, true);
377        this.configureSocketChannel(socketChannel);
378
379        // Create peer
380        final InetSocketAddress remote = (InetSocketAddress)socketChannel.socket().getRemoteSocketAddress();
381        final String peer = remote.getHostString() + (remote.getPort() != DEFAULT_TCP_PORT ? ":" + remote.getPort() : "");
382
383        // Are we already connected to this peer? If so (deterministically) choose which connection wins
384        Connection connection = this.connectionMap.get(peer);
385        if (connection != null) {
386
387            // Compare the socket addresses of the initiator side of each connection
388            final SocketAddress oldAddr = (InetSocketAddress)connection.getSocketChannel().socket().getLocalSocketAddress();
389            final SocketAddress newAddr = (InetSocketAddress)socketChannel.getRemoteAddress();
390            final String oldDesc = oldAddr.toString().replaceAll("^[^/]*/", "");            // strip off hostname part, if any
391            final String newDesc = newAddr.toString().replaceAll("^[^/]*/", "");            // strip off hostname part, if any
392            final int diff = newDesc.compareTo(oldDesc);
393            this.log.info("connection mid-air collision: old: " + oldDesc + ", new: " + newDesc
394              + ", winner: " + (diff < 0 ? "new" : diff > 0 ? "old" : "neither (?)"));
395
396            // Close the loser(s)
397            if (diff >= 0) {
398                this.log.info("rejecting incoming connection from " + socketChannel.getRemoteAddress() + " as duplicate");
399                socketChannel.close();
400                socketChannel = null;
401            }
402            if (diff <= 0) {
403                this.log.info("closing existing duplicate connection to " + socketChannel.getRemoteAddress());
404                this.connectionMap.remove(peer);
405                connection.close(new IOException("duplicate connection"));
406                connection = null;
407            }
408        }
409
410        // Create connection from new socket if needed
411        if (connection == null && socketChannel != null) {
412            connection = new Connection(this, peer, socketChannel);
413            this.connectionMap.put(peer, connection);
414            this.handleOutputQueueEmpty(connection);
415        }
416    }
417
418    // Enable/disable incoming connections
419    private void selectForAccept(boolean enabled) throws IOException {
420        if (this.selectionKey == null)
421            return;
422        if (enabled && (this.selectionKey.interestOps() & SelectionKey.OP_ACCEPT) == 0) {
423            this.selectFor(this.selectionKey, SelectionKey.OP_ACCEPT, true);
424            if (this.log.isDebugEnabled())
425                this.log.debug(this + " started listening for incoming connections");
426        } else if (!enabled && (this.selectionKey.interestOps() & SelectionKey.OP_ACCEPT) != 0) {
427            this.selectFor(this.selectionKey, SelectionKey.OP_ACCEPT, false);
428            if (this.log.isDebugEnabled())
429                this.log.debug(this + " stopped listening for incoming connections");
430        }
431    }

435    // Invoked when a message arrives on a connection
436    void handleMessage(final Connection connection, final ByteBuffer msg) {
437        assert Thread.holdsLock(this);
438        assert this.isServiceThread();
439        this.executor.submit(new Runnable() {
440            @Override
441            public void run() {
442                try {
443                    TCPNetwork.this.handler.handle(connection.getPeer(), msg);
444                } catch (Throwable t) {
445                    TCPNetwork.this.log.error("exception in callback", t);
446                }
447            }
448        });
449    }
450
451    // Invoked a connection's output queue goes empty
452    void handleOutputQueueEmpty(final Connection connection) {
453        assert Thread.holdsLock(this);
454        assert this.isServiceThread();
455        this.executor.submit(new Runnable() {
456            @Override
457            public void run() {
458                try {
459                    TCPNetwork.this.handler.outputQueueEmpty(connection.getPeer());
460                } catch (Throwable t) {
461                    TCPNetwork.this.log.error("exception in callback", t);
462                }
463            }
464        });
465    }

467    // Invoked when a connection closes
468    void handleConnectionClosed(Connection connection) {
469        assert Thread.holdsLock(this);
470        assert this.isServiceThread();
471        if (this.log.isDebugEnabled())
472            this.log.debug(this + " handling closed connection " + connection);
473        this.connectionMap.remove(connection.getPeer());
474        this.handleOutputQueueEmpty(connection);
475        this.wakeup();
476    }

480    // Create a new connection to the specified peer
481    private synchronized Connection createConnection(String peer) throws IOException {
482
483        // Create new one
484        final SocketChannel socketChannel = SocketChannel.open()
485          .setOption(StandardSocketOptions.SO_KEEPALIVE, true)
486          .setOption(StandardSocketOptions.TCP_NODELAY, true);
487        this.configureSocketChannel(socketChannel);
488        socketChannel.configureBlocking(false);
489        if (this.log.isDebugEnabled())
490            this.log.debug(this + " looking up peer address `" + peer + "'");
491
492        // Resolve peer name into a socket address
493        InetSocketAddress socketAddress = null;
494        try {
495            socketAddress = new InetSocketAddress(
496              TCPNetwork.parseAddressPart(peer), TCPNetwork.parsePortPart(peer, DEFAULT_TCP_PORT));
497        } catch (IllegalArgumentException e) {
498            if (this.log.isTraceEnabled())
499                this.log.trace(this + " peer address `" + peer + "' is invalid", e);
500        }
501        if (socketAddress == null || socketAddress.isUnresolved())
502            throw new IOException("invalid or unresolvable peer address `" + peer + "'");
503
504        // Initiate connection to peer
505        if (this.log.isDebugEnabled()) {
506            this.log.debug(this + ": resolved peer address `" + peer + "' to " + socketAddress.getAddress()
507              + "; now initiating connection");
508        }
509        socketChannel.connect(socketAddress);
510
511        // Create new connection
512        return new Connection(this, peer, socketChannel);
513    }

517    @Override
518    protected void serviceHousekeeping() {
519
520        // Perform connection housekeeping
521        for (Connection connection : new ArrayList<Connection>(this.connectionMap.values())) {
522            try {
523                connection.performHousekeeping();
524            } catch (IOException e) {
525                if (this.log.isDebugEnabled())
526                    this.log.debug("I/O error from " + connection, e);
527                connection.close(e);
528            } catch (Throwable t) {
529                this.log.error("error performing housekeeping for " + connection, t);
530                connection.close(t);
531            }
532        }
533
534        // Perform my own housekeeping
535        try {
536            this.selectForAccept(this.connectionMap.size() < this.maxConnections);
537        } catch (IOException e) {
538            throw new RuntimeException("unexpected exception", e);
539        }
540    }

545            connection.close(null);
