031/**
032 * A fixed set of {@link KeyRange} instances that can be treated as a unified whole, in particular as a {@link KeyFilter}.
033 *
034 * <p>
035 * Instances are immutable.
036 * </p>
037 *
038 * @see KeyRange
039 */
040public class KeyRanges implements Iterable<KeyRange>, KeyFilter, SizeEstimating {
041
042    /**
043     * The empty instance containing zero ranges.
044     */
045    public static final KeyRanges EMPTY = new KeyRanges(Collections.<KeyRange>emptyList());

047    /**
048     * The "full" instance containing a single {@link KeyRange} that contains all keys.
049     */
050    public static final KeyRanges FULL = new KeyRanges(Arrays.asList(KeyRange.FULL));

058    /**
059     * Constructor.
060     *
061     * <p>
062     * Creates an instance that contains all keys contained by any of the {@link KeyRange}s in {@code ranges}.
063     * The given {@code ranges} may be adjacent, overlap, and/or be listed in any order; this constructor
064     * will normalize them.
065     * </p>
066     *
067     * @param ranges individual key ranges
068     * @throws IllegalArgumentException if {@code ranges} or any {@link KeyRange} therein is null
069     */
070    public KeyRanges(Iterable<? extends KeyRange> ranges) {
071        if (ranges == null)
072            throw new IllegalArgumentException("null ranges");
073        this.ranges = KeyRanges.minimize(Lists.<KeyRange>newArrayList(ranges));
074    }

203    /**
204     * Create the inverse of this instance. The inverse contains all keys not contained by this instance.
205     *
206     * @return the inverse of this instance
207     */
208    public KeyRanges inverse() {
209        final ArrayList<KeyRange> list = new ArrayList<>(this.ranges.size() + 1);
210        if (this.ranges.isEmpty())
211            return KeyRanges.FULL;
212        final KeyRange first = this.ranges.get(0);
213        int i = 0;
214        byte[] lastMax;
215        if (first.getMin().length == 0) {
216            if ((lastMax = first.getMax()) == null) {
217                assert this.ranges.size() == 1;
218                return KeyRanges.EMPTY;
219            }
220            i++;
221        } else
222            lastMax = ByteUtil.EMPTY;
223        while (true) {
224            if (i == this.ranges.size()) {
225                list.add(new KeyRange(lastMax, null));
226                break;
227            }
228            final KeyRange next = this.ranges.get(i++);
229            list.add(new KeyRange(lastMax, next.getMin()));
230            if ((lastMax = next.getMax()) == null) {
231                assert i == this.ranges.size();
232                break;
233            }
234        }
235        return new KeyRanges(list);
236    }

238    /**
239     * Determine whether this instance contains the given {@link KeyRanges}, i.e., all keys contained by
240     * the given {@link KeyRanges} are also contained by this instance.
241     *
242     * @param ranges other instance to test
243     * @return true if this instance contains {@code ranges}, otherwise false
244     * @throws IllegalArgumentException if {@code ranges} is null
245     */
246    public boolean contains(KeyRanges ranges) {
247        if (ranges == null)
248            throw new IllegalArgumentException("null ranges");
249        return ranges.equals(this.intersection(ranges));
250    }

252    /**
253     * Determine whether this instance contains the given {@link KeyRange}, i.e., all keys contained by
254     * the given {@link KeyRange} are also contained by this instance.
255     *
256     * @param range key range to test
257     * @return true if this instance contains {@code range}, otherwise false
258     * @throws IllegalArgumentException if {@code range} is null
259     */
260    public boolean contains(KeyRange range) {
261        if (range == null)
262            throw new IllegalArgumentException("null range");
263        final KeyRange[] pair = this.findKey(range.getMin());
264        if (pair[0] != pair[1])
265            return false;
266        final KeyRange container = pair[0];
267        if (container == null)
268            return false;
269        return container.contains(range);
270    }

272    /**
273     * Find the contiguous {@link KeyRange}(s) within this instance containing, or adjacent to, the given key.
274     *
275     * <p>
276     * This method returns an array of length two: if this instance contains {@code key} then both elements are the
277     * same Java object, namely, the {@link KeyRange} that contains {@code key}; otherwise, the first element is
278     * the nearest {@link KeyRange} to the left of {@code key}, or null if none exists, and the second element is
279     * the {@link KeyRange} to the right of {@code key}, or null if none exists. Note if this instance is empty
280     * then <code>{ null, null }</code> is returned.
281     * </p>
282     *
283     * @param key key to find
284     * @return array with the containing {@link KeyRange} or nearest neighbor to the left (or null) and
285     *  the containing {@link KeyRange} or nearest neighbor to the right (or null)
286     * @throws IllegalArgumentException if {@code key} is null
287     */
288    public KeyRange[] findKey(byte[] key) {
289
290        // Sanity check
291        if (key == null)
292            throw new IllegalArgumentException("null key");
293
294        // Optimization: assume previous success is likely to repeat
295        final KeyRange temp = this.lastContainingKeyRange;
296        if (temp != null) {
297            if (temp.contains(key))
298                return new KeyRange[] { temp, temp };
299            this.lastContainingKeyRange = null;
300        }
301
302        // Search for matching range
303        final int i = ~Collections.binarySearch(this.ranges, new KeyRange(key, key), KeyRange.SORT_BY_MIN);
304        assert i >= 0;                                                  // this.ranges should never contain new KeyRange(key, key)
305        KeyRange left = null;
306        if (i > 0) {
307            if ((left = this.ranges.get(i - 1)).contains(key)) {
308                this.lastContainingKeyRange = left;
309                return new KeyRange[] { left, left };
310            }
311        }
312        KeyRange right = null;
313        if (i < this.ranges.size()) {
314            if ((right = this.ranges.get(i)).contains(key)) {
315                this.lastContainingKeyRange = right;
316                return new KeyRange[] { right, right };
317            }
318        }
319
320        // Not contained
321        return new KeyRange[] { left, right };
322    }

324    /**
325     * Return a new {@link KeyRanges} instance equal to this instance with all keys in the given {@link KeyRange} added.
326     *
327     * @param range range to add
328     * @return this instance with {@code range} added
329     * @throws IllegalArgumentException if {@code range} is null
330     */
331    public KeyRanges add(KeyRange range) {
332        if (range == null)
333            throw new IllegalArgumentException("null range");
334        if (range.isEmpty())
335            return this;
336        final ArrayList<KeyRange> list = new ArrayList<>(this.ranges.size() + 1);
337        list.addAll(this.ranges);
338        list.add(range);
339        return new KeyRanges(list);
340    }

342    /**
343     * Return a new {@link KeyRanges} instance equal to this instance with all keys in the given {@link KeyRange} removed.
344     *
345     * @param range range to remove
346     * @return this instance with {@code range} removed
347     * @throws IllegalArgumentException if {@code range} is null
348     */
349    public KeyRanges remove(KeyRange range) {
350        if (range == null)
351            throw new IllegalArgumentException("null range");
352        if (range.isEmpty())
353            return this;
354        return this.intersection(new KeyRanges(range).inverse());
355    }

357    /**
358     * Create an instance that represents the union of this and the provided instance(s).
359     *
360     * @param others other instances
361     * @return the union of this instance and {@code others}
362     * @throws IllegalArgumentException if {@code others} or any element in {@code others} is null
363     */
364    public KeyRanges union(KeyRanges... others) {
365        if (others == null)
366            throw new IllegalArgumentException("null others");
367        if (others.length == 0)
368            return this;
369        final ArrayList<KeyRange> list = new ArrayList<>(this.ranges.size() + others.length);
370        list.addAll(this.ranges);
371        for (KeyRanges other : others) {
372            if (other == null)
373                throw new IllegalArgumentException("null other");
374            list.addAll(other.ranges);
375        }
376        return new KeyRanges(list);
377    }

379    /**
380     * Create an instance that represents the intersection of this and the provided instance(s).
381     *
382     * @param others other instances
383     * @return the intersection of this instance and {@code others}
384     * @throws IllegalArgumentException if {@code others} or any element in {@code others} is null
385     */
386    public KeyRanges intersection(KeyRanges... others) {
387        if (others == null)
388            throw new IllegalArgumentException("null others");
389        if (others.length == 0)
390            return this;
391        final ArrayList<KeyRange> list = new ArrayList<>(this.ranges.size() + others.length);
392        list.addAll(this.inverse().ranges);
393        for (KeyRanges other : others) {
394            if (other == null)
395                throw new IllegalArgumentException("null other");
396            list.addAll(other.inverse().ranges);
397        }
398        return new KeyRanges(list).inverse();
399    }

422    /**
423     * Serialize this instance.
424     *
425     * @param out output
426     * @throws IOException if an error occurs
427     * @throws IllegalArgumentException if {@code out} is null
428     */
429    public void serialize(OutputStream out) throws IOException {
430        UnsignedIntEncoder.write(out, this.ranges.size());
431        byte[] prev = null;
432        for (KeyRange range : this.ranges) {
433            final byte[] min = range.getMin();
434            final byte[] max = range.getMax();
435            assert max != null || range == this.ranges.get(this.ranges.size() - 1);
436            KeyListEncoder.write(out, min, prev);
437            KeyListEncoder.write(out, max != null ? max : min, min);            // map final [min, null) to [min, min]
438            prev = max;
439        }
440    }

442    /**
443     * Calculate the number of bytes required to serialize this instance via {@link #serialize serialize()}.
444     *
445     * @return number of serialized bytes
446     */
447    public long serializedLength() {
448        long total = UnsignedIntEncoder.encodeLength(this.ranges.size());
449        byte[] prev = null;
450        for (KeyRange range : this.ranges) {
451            final byte[] min = range.getMin();
452            final byte[] max = range.getMax();
453            total += KeyListEncoder.writeLength(min, prev);
454            total += KeyListEncoder.writeLength(max != null ? max : min, min);
455            prev = max;
456        }
457        return total;
458    }

460    /**
461     * Deserialize an instance created by {@link #serialize serialize()}.
462     *
463     * @param input input stream containing data from {@link #serialize serialize()}
464     * @return deserialized instance
465     * @throws IOException if an I/O error occurs
466     * @throws java.io.EOFException if the input ends unexpectedly
467     * @throws IllegalArgumentException if {@code input} is null
468     */
469    public static KeyRanges deserialize(InputStream input) throws IOException {
470        if (input == null)
471            throw new IllegalArgumentException("null input");
472        final int count = UnsignedIntEncoder.read(input);
473        final ArrayList<KeyRange> rangeList = new ArrayList<>(count);
474        byte[] prev = null;
475        for (int i = 0; i < count; i++) {
476            final byte[] min = KeyListEncoder.read(input, prev);
477            final byte[] max = KeyListEncoder.read(input, min);
478            rangeList.add(new KeyRange(min, Arrays.equals(min, max) ? null : max));
479            prev = max;
480        }
481        return new KeyRanges(rangeList);
482    }

484    /**
485     * Deserialize an instance created by {@link #serialize serialize()} in the form of an
486     * iterator of the individual {@link KeyRange}s.
487     *
488     * <p>
489     * If an {@link IOException} is thrown while reading, the returned {@link Iterator}
490     * will throw a {@link RuntimeException} wrapping it. If invalid data is encountered,
491     * the returned {@link Iterator} will throw an {@link IllegalArgumentException}.
492     *
493     * @param input input stream containing data from {@link #serialize serialize()}
494     * @return deserialized iteration of {@link KeyRange}s
495     * @throws IllegalArgumentException if {@code input} is null
496     */
497    public static Iterator<KeyRange> deserializeIterator(final InputStream input) {
498        Preconditions.checkArgument(input != null, "null input");
499        return new UnmodifiableIterator<KeyRange>() {
500
501            private int remain = -1;
502            private byte[] prev;
503
504            @Override
505            public boolean hasNext() {
506                try {
507                    this.init();
508                } catch (IOException e) {
509                    throw new RuntimeException(e);
510                }
511                return this.remain > 0;
512            }
513
514            @Override
515            public KeyRange next() {
516                if (this.remain == 0)
517                    throw new NoSuchElementException();
518                final byte[] min;
519                final byte[] max;
520                try {
521                    this.init();
522                    min = KeyListEncoder.read(input, this.prev);
523                    max = KeyListEncoder.read(input, min);
524                } catch (IOException e) {
525                    throw new RuntimeException(e);
526                }
527                final KeyRange range = new KeyRange(min, Arrays.equals(min, max) ? null : max);
528                this.prev = max;
529                this.remain--;
530                return range;
531            }
532
533            private void init() throws IOException {
534                if (this.remain == -1)
535                    this.remain = UnsignedIntEncoder.read(input);
536            }
537        };
538    }

542    @Override
543    public boolean contains(byte[] key) {
544        final KeyRange[] pair = this.findKey(key);
545        return pair[0] == pair[1] && pair[0] != null;
546    }

548    @Override
549    public byte[] seekHigher(byte[] key) {
550        final KeyRange[] pair = this.findKey(key);
551        if (pair[0] == pair[1])
552            return pair[0] != null ? key : null;
553        return pair[1] != null ? pair[1].getMin() : null;
554    }
555
556    @Override
557    public byte[] seekLower(byte[] key) {
558        if (key == null)
559            throw new IllegalArgumentException("null key");
560        if (key.length == 0) {
561            if (this.ranges.isEmpty())
562                return null;
563            final byte[] lastMax = this.ranges.get(this.ranges.size() - 1).getMax();
564            return lastMax != null ? lastMax : ByteUtil.EMPTY;
565        }
566        final KeyRange[] pair = this.findKey(key);
567        if (pair[0] == pair[1])
568            return pair[0] != null ? key : null;
569        return pair[0] != null ? pair[0].getMax() : null;
570    }

607    // Return a "minimal" list with these properties:
608    //  - Sorted according to KeyRange.SORT_BY_MIN
609    //  - No overlapping ranges
610    //  - Adjacent ranges consolidated into a single range
611    private static ArrayList<KeyRange> minimize(List<KeyRange> ranges) {
612
613        // Create array
614        final ArrayList<KeyRange> sortedRanges = new ArrayList<>(ranges);
615
616        // Remove empty ranges
617        for (Iterator<KeyRange> i = sortedRanges.iterator(); i.hasNext(); ) {
618            if (i.next().isEmpty())
619                i.remove();
620        }
621
622        // Sort remaining ranges by min, then max
623        Collections.sort(sortedRanges, KeyRange.SORT_BY_MIN);
624
625        // Consolidate
626        final ArrayList<KeyRange> list = new ArrayList<>(ranges.size());
627        KeyRange prev = null;
628        for (KeyRange range : sortedRanges) {
629
630            // Sanity check range
631            if (range == null)
632                throw new IllegalArgumentException("null range");
633
634            // Handle first in list
635            if (prev == null) {
636                prev = range;
637                continue;
638            }
639
640            // Compare to previous range
641            final int diff1 = KeyRange.compare(range.getMin(), prev.getMin());
642            assert diff1 >= 0;
643            if (diff1 == 0) {                           // range contains prev -> discard prev
644                assert range.contains(prev);
645                prev = range;
646                continue;
647            }
648            final int diff2 = KeyRange.compare(range.getMin(), prev.getMax());
649            if (diff2 <= 0) {                           // prev and range overlap -> take their union
650                final byte[] max = KeyRange.compare(range.getMax(), prev.getMax()) > 0 ? range.getMax() : prev.getMax();
651                prev = new KeyRange(prev.getMin(), max);
652                continue;
653            }
654
655            // OK add it
656            list.add(prev);                             // prev and range don't overlap -> accept prev
657            prev = range;
658            continue;
659        }
660        if (prev != null)
661            list.add(prev);
662        list.trimToSize();
663        return list;
664    }
