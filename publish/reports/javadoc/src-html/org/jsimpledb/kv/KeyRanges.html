029/**
030 * A fixed set of {@link KeyRange} instances that can be treated as a unified whole, in particular as a {@link KeyFilter}.
031 *
032 * <p>
033 * Instances are immutable.
034 * </p>
035 *
036 * @see KeyRange
037 */
038public class KeyRanges implements Iterable<KeyRange>, KeyFilter, SizeEstimating {
039
040    /**
041     * The empty instance containing zero ranges.
042     */
043    public static final KeyRanges EMPTY = new KeyRanges(Collections.<KeyRange>emptyList());
044
045    /**
046     * The "full" instance containing a single {@link KeyRange} that contains all keys.
047     */
048    public static final KeyRanges FULL = new KeyRanges(Arrays.asList(KeyRange.FULL));

056    /**
057     * Constructor.
058     *
059     * <p>
060     * Creates an instance that contains all keys contained by any of the {@link KeyRange}s in {@code ranges}.
061     * The given {@code ranges} may be adjacent, overlap, and/or be listed in any order; this constructor
062     * will normalize them.
063     * </p>
064     *
065     * @param ranges individual key ranges
066     * @throws IllegalArgumentException if {@code ranges} or any {@link KeyRange} therein is null
067     */
068    public KeyRanges(Iterable<? extends KeyRange> ranges) {
069        Preconditions.checkArgument(ranges != null, "null ranges");
070        this.ranges = KeyRanges.minimize(Lists.<KeyRange>newArrayList(ranges));
071    }

185     * Create a new instance from this one, with each {@link KeyRange} prefixed by the given byte sequence.

192        return new KeyRanges(Lists.transform(this.ranges, new Function<KeyRange, KeyRange>() {

195                return keyRange.prefixedBy(prefix);
196            }

200    /**
201     * Create the inverse of this instance. The inverse contains all keys not contained by this instance.
202     *
203     * @return the inverse of this instance
204     */
205    public KeyRanges inverse() {
206        final ArrayList<KeyRange> list = new ArrayList<>(this.ranges.size() + 1);
207        if (this.ranges.isEmpty())
208            return KeyRanges.FULL;
209        final KeyRange first = this.ranges.get(0);
210        int i = 0;
211        byte[] lastMax;
212        if (first.getMin().length == 0) {
213            if ((lastMax = first.getMax()) == null) {
214                assert this.ranges.size() == 1;
215                return KeyRanges.EMPTY;
216            }
217            i++;
218        } else
219            lastMax = ByteUtil.EMPTY;
220        while (true) {
221            if (i == this.ranges.size()) {
222                list.add(new KeyRange(lastMax, null));
223                break;
224            }
225            final KeyRange next = this.ranges.get(i++);
226            list.add(new KeyRange(lastMax, next.getMin()));
227            if ((lastMax = next.getMax()) == null) {
228                assert i == this.ranges.size();
229                break;
230            }
231        }
232        return new KeyRanges(list);
233    }

235    /**
236     * Determine whether this instance contains the given {@link KeyRanges}, i.e., all keys contained by
237     * the given {@link KeyRanges} are also contained by this instance.
238     *
239     * @param ranges other instance to test
240     * @return true if this instance contains {@code ranges}, otherwise false
241     * @throws IllegalArgumentException if {@code ranges} is null
242     */
243    public boolean contains(KeyRanges ranges) {
244        Preconditions.checkArgument(ranges != null, "null ranges");
245        return ranges.equals(this.intersection(ranges));
246    }

248    /**
249     * Determine whether this instance contains the given {@link KeyRange}, i.e., all keys contained by
250     * the given {@link KeyRange} are also contained by this instance.
251     *
252     * @param range key range to test
253     * @return true if this instance contains {@code range}, otherwise false
254     * @throws IllegalArgumentException if {@code range} is null
255     */
256    public boolean contains(KeyRange range) {
257        Preconditions.checkArgument(range != null, "null range");
258        final KeyRange[] pair = this.findKey(range.getMin());
259        if (pair[0] != pair[1])
260            return false;
261        final KeyRange container = pair[0];
262        if (container == null)
263            return false;
264        return container.contains(range);
265    }

267    /**
268     * Find the contiguous {@link KeyRange}(s) within this instance containing, or adjacent to, the given key.
269     *
270     * <p>
271     * This method returns an array of length two: if this instance contains {@code key} then both elements are the
272     * same Java object, namely, the {@link KeyRange} that contains {@code key}; otherwise, the first element is
273     * the nearest {@link KeyRange} to the left of {@code key}, or null if none exists, and the second element is
274     * the {@link KeyRange} to the right of {@code key}, or null if none exists. Note if this instance is empty
275     * then <code>{ null, null }</code> is returned.
276     * </p>
277     *
278     * @param key key to find
279     * @return array with the containing {@link KeyRange} or nearest neighbor to the left (or null) and
280     *  the containing {@link KeyRange} or nearest neighbor to the right (or null)
281     * @throws IllegalArgumentException if {@code key} is null
282     */
283    public KeyRange[] findKey(byte[] key) {
284
285        // Sanity check
286        Preconditions.checkArgument(key != null, "null key");
287
288        // Optimization: assume previous success is likely to repeat
289        final KeyRange temp = this.lastContainingKeyRange;
290        if (temp != null) {
291            if (temp.contains(key))
292                return new KeyRange[] { temp, temp };
293            this.lastContainingKeyRange = null;
294        }
295
296        // Search for matching range
297        final int i = ~Collections.binarySearch(this.ranges, new KeyRange(key, key), KeyRange.SORT_BY_MIN);
298        assert i >= 0;                                                  // this.ranges should never contain new KeyRange(key, key)
299        KeyRange left = null;
300        if (i > 0) {
301            if ((left = this.ranges.get(i - 1)).contains(key)) {
302                this.lastContainingKeyRange = left;
303                return new KeyRange[] { left, left };
304            }
305        }
306        KeyRange right = null;
307        if (i < this.ranges.size()) {
308            if ((right = this.ranges.get(i)).contains(key)) {
309                this.lastContainingKeyRange = right;
310                return new KeyRange[] { right, right };
311            }
312        }
313
314        // Not contained
315        return new KeyRange[] { left, right };
316    }

318    /**
319     * Return a new {@link KeyRanges} instance equal to this instance with all keys in the given {@link KeyRange} added.
320     *
321     * @param range range to add
322     * @return this instance with {@code range} added
323     * @throws IllegalArgumentException if {@code range} is null
324     */
325    public KeyRanges add(KeyRange range) {
326        Preconditions.checkArgument(range != null, "null range");
327        if (range.isEmpty())
328            return this;
329        final ArrayList<KeyRange> list = new ArrayList<>(this.ranges.size() + 1);
330        list.addAll(this.ranges);
331        list.add(range);
332        return new KeyRanges(list);
333    }

335    /**
336     * Return a new {@link KeyRanges} instance equal to this instance with all keys in the given {@link KeyRange} removed.
337     *
338     * @param range range to remove
339     * @return this instance with {@code range} removed
340     * @throws IllegalArgumentException if {@code range} is null
341     */
342    public KeyRanges remove(KeyRange range) {
343        Preconditions.checkArgument(range != null, "null range");
344        if (range.isEmpty())
345            return this;
346        return this.intersection(new KeyRanges(range).inverse());
347    }

349    /**
350     * Create an instance that represents the union of this and the provided instance(s).
351     *
352     * @param others other instances
353     * @return the union of this instance and {@code others}
354     * @throws IllegalArgumentException if {@code others} or any element in {@code others} is null
355     */
356    public KeyRanges union(KeyRanges... others) {
357        Preconditions.checkArgument(others != null, "null others");
358        if (others.length == 0)
359            return this;
360        final ArrayList<KeyRange> list = new ArrayList<>(this.ranges.size() + others.length);
361        list.addAll(this.ranges);
362        for (KeyRanges other : others) {
363            Preconditions.checkArgument(other != null, "null other");
364            list.addAll(other.ranges);
365        }
366        return new KeyRanges(list);
367    }

369    /**
370     * Create an instance that represents the intersection of this and the provided instance(s).
371     *
372     * @param others other instances
373     * @return the intersection of this instance and {@code others}
374     * @throws IllegalArgumentException if {@code others} or any element in {@code others} is null
375     */
376    public KeyRanges intersection(KeyRanges... others) {
377        Preconditions.checkArgument(others != null, "null others");
378        if (others.length == 0)
379            return this;
380        final ArrayList<KeyRange> list = new ArrayList<>(this.ranges.size() + others.length);
381        list.addAll(this.inverse().ranges);
382        for (KeyRanges other : others) {
383            Preconditions.checkArgument(other != null, "null other");
384            list.addAll(other.inverse().ranges);
385        }
386        return new KeyRanges(list).inverse();
387    }

410    /**
411     * Serialize this instance.
412     *
413     * @param out output
414     * @throws IOException if an error occurs
415     * @throws IllegalArgumentException if {@code out} is null
416     */
417    public void serialize(OutputStream out) throws IOException {
418        UnsignedIntEncoder.write(out, this.ranges.size());
419        byte[] prev = null;
420        for (KeyRange range : this.ranges) {
421            final byte[] min = range.getMin();
422            final byte[] max = range.getMax();
423            assert max != null || range == this.ranges.get(this.ranges.size() - 1);
424            KeyListEncoder.write(out, min, prev);
425            KeyListEncoder.write(out, max != null ? max : min, min);            // map final [min, null) to [min, min]
426            prev = max;
427        }
428    }

430    /**
431     * Calculate the number of bytes required to serialize this instance via {@link #serialize serialize()}.
432     *
433     * @return number of serialized bytes
434     */
435    public long serializedLength() {
436        long total = UnsignedIntEncoder.encodeLength(this.ranges.size());
437        byte[] prev = null;
438        for (KeyRange range : this.ranges) {
439            final byte[] min = range.getMin();
440            final byte[] max = range.getMax();
441            total += KeyListEncoder.writeLength(min, prev);
442            total += KeyListEncoder.writeLength(max != null ? max : min, min);
443            prev = max;
444        }
445        return total;
446    }

448    /**
449     * Deserialize an instance created by {@link #serialize serialize()}.
450     *
451     * @param input input stream containing data from {@link #serialize serialize()}
452     * @return deserialized instance
453     * @throws IOException if an I/O error occurs
454     * @throws java.io.EOFException if the input ends unexpectedly
455     * @throws IllegalArgumentException if {@code input} is null
456     */
457    public static KeyRanges deserialize(InputStream input) throws IOException {
458        Preconditions.checkArgument(input != null, "null input");
459        final int count = UnsignedIntEncoder.read(input);
460        final ArrayList<KeyRange> rangeList = new ArrayList<>(count);
461        byte[] prev = null;
462        for (int i = 0; i < count; i++) {
463            final byte[] min = KeyListEncoder.read(input, prev);
464            final byte[] max = KeyListEncoder.read(input, min);
465            rangeList.add(new KeyRange(min, Arrays.equals(min, max) ? null : max));
466            prev = max;
467        }
468        return new KeyRanges(rangeList);
469    }

471    /**
472     * Deserialize an instance created by {@link #serialize serialize()} in the form of an
473     * iterator of the individual {@link KeyRange}s.
474     *
475     * <p>
476     * If an {@link IOException} is thrown while reading, the returned {@link Iterator}
477     * will throw a {@link RuntimeException} wrapping it. If invalid data is encountered,
478     * the returned {@link Iterator} will throw an {@link IllegalArgumentException}.
479     *
480     * @param input input stream containing data from {@link #serialize serialize()}
481     * @return deserialized iteration of {@link KeyRange}s
482     * @throws IllegalArgumentException if {@code input} is null
483     */
484    public static Iterator<KeyRange> deserializeIterator(final InputStream input) {
485        Preconditions.checkArgument(input != null, "null input");
486        return new UnmodifiableIterator<KeyRange>() {
487
488            private int remain = -1;
489            private byte[] prev;
490
491            @Override
492            public boolean hasNext() {
493                try {
494                    this.init();
495                } catch (IOException e) {
496                    throw new RuntimeException(e);
497                }
498                return this.remain > 0;
499            }
500
501            @Override
502            public KeyRange next() {
503                if (this.remain == 0)
504                    throw new NoSuchElementException();
505                final byte[] min;
506                final byte[] max;
507                try {
508                    this.init();
509                    min = KeyListEncoder.read(input, this.prev);
510                    max = KeyListEncoder.read(input, min);
511                } catch (IOException e) {
512                    throw new RuntimeException(e);
513                }
514                final KeyRange range = new KeyRange(min, Arrays.equals(min, max) ? null : max);
515                this.prev = max;
516                this.remain--;
517                return range;
518            }
519
520            private void init() throws IOException {
521                if (this.remain == -1)
522                    this.remain = UnsignedIntEncoder.read(input);
523            }
524        };
525    }

529    @Override
530    public boolean contains(byte[] key) {
531        final KeyRange[] pair = this.findKey(key);
532        return pair[0] == pair[1] && pair[0] != null;
533    }

535    @Override
536    public byte[] seekHigher(byte[] key) {
537        final KeyRange[] pair = this.findKey(key);
538        if (pair[0] == pair[1])
539            return pair[0] != null ? key : null;
540        return pair[1] != null ? pair[1].getMin() : null;
541    }
542
543    @Override
544    public byte[] seekLower(byte[] key) {
545        Preconditions.checkArgument(key != null, "null key");
546        if (key.length == 0) {
547            if (this.ranges.isEmpty())
548                return null;
549            final byte[] lastMax = this.ranges.get(this.ranges.size() - 1).getMax();
550            return lastMax != null ? lastMax : ByteUtil.EMPTY;
551        }
552        final KeyRange[] pair = this.findKey(key);
553        if (pair[0] == pair[1])
554            return pair[0] != null ? key : null;
555        return pair[0] != null ? pair[0].getMax() : null;
556    }

593    // Return a "minimal" list with these properties:
594    //  - Sorted according to KeyRange.SORT_BY_MIN
595    //  - No overlapping ranges
596    //  - Adjacent ranges consolidated into a single range
597    private static ArrayList<KeyRange> minimize(List<KeyRange> ranges) {
598
599        // Create array
600        final ArrayList<KeyRange> sortedRanges = new ArrayList<>(ranges);
601
602        // Remove empty ranges
603        for (Iterator<KeyRange> i = sortedRanges.iterator(); i.hasNext(); ) {
604            if (i.next().isEmpty())
605                i.remove();
606        }
607
608        // Sort remaining ranges by min, then max
609        Collections.sort(sortedRanges, KeyRange.SORT_BY_MIN);
610
611        // Consolidate
612        final ArrayList<KeyRange> list = new ArrayList<>(ranges.size());
613        KeyRange prev = null;
614        for (KeyRange range : sortedRanges) {
615
616            // Sanity check range
617            Preconditions.checkArgument(range != null, "null range");
618
619            // Handle first in list
620            if (prev == null) {
621                prev = range;
622                continue;
623            }
624
625            // Compare to previous range
626            final int diff1 = KeyRange.compare(range.getMin(), prev.getMin());
627            assert diff1 >= 0;
628            if (diff1 == 0) {                           // range contains prev -> discard prev
629                assert range.contains(prev);
630                prev = range;
631                continue;
632            }
633            final int diff2 = KeyRange.compare(range.getMin(), prev.getMax());
634            if (diff2 <= 0) {                           // prev and range overlap -> take their union
635                final byte[] max = KeyRange.compare(range.getMax(), prev.getMax()) > 0 ? range.getMax() : prev.getMax();
636                prev = new KeyRange(prev.getMin(), max);
637                continue;
638            }
639
640            // OK add it
641            list.add(prev);                             // prev and range don't overlap -> accept prev
642            prev = range;
643            continue;
644        }
645        if (prev != null)
646            list.add(prev);
647        list.trimToSize();
648        return list;
649    }
