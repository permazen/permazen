059/**
060 * {@link AtomicKVStore} based on {@link ArrayKVStore} with background compaction.
061 *
062 * <p>
063 * This implementation is designed to maximize the speed of reads and minimize the amount of memory overhead per key/value pair.
064 * It is optimized for relatively infrequent writes.
065 *
066 * <p>
067 * Instances periodically compact outstanding changes into new arrays in a background thread. A compaction is scheduled whenever:
068 *  <ul>
069 *  <li>The size of uncompacted changes exceeds the {@linkplain #setCompactLowWater compaction space low-water mark}</li>
070 *  <li>The oldest uncompacted modification is older than the {@linkplain #setCompactMaxDelay compaction maximum delay}</li>
071 *  <li>{@link #scheduleCompaction} is invoked</li>
072 *  </ul>
073 *
074 * <p>
075 * There is also a {@linkplain #setCompactHighWater high water mark for the size of uncompacted changes}: when this value
076 * is exceeded, new write attempts will block until the current compaction cycle completes. This prevents the unbounded growth
077 * of uncompacted changes when there is extremely high write volume.
078 *
079 * <p>
080 * "Hot" backups created in parallel with normal operation are supported via {@link #hotCopy hotCopy()}.
081 * Hard links are used (when available) to make this operation fast.
082 *
083 * <p>
084 * The {@linkplain #setDirectory database directory} is a required configuration property.
085 *
086 * <p>
087 * Key and value data must not exceed 2GB (each separately).
088 *
089 * <p>
090 * Instances may be stopped and (re)started multiple times.
091 */
092public class AtomicArrayKVStore extends AbstractKVStore implements AtomicKVStore {

264    @Override
265    @PostConstruct
266    public void start() {
267        boolean success = false;
268        this.writeLock.lock();
269        try {
270
271            // Already started?
272            if (this.kvstore != null) {
273                success = true;
274                return;
275            }
276            this.log.info("starting " + this);
277
278            // Sanity check
279            assert this.compaction == null;
280            assert this.scheduledExecutorService == null;
281            assert !this.createdExecutorService;
282            assert this.generation == 0;
283            assert this.generationFile == null;
284            assert this.lockFile == null;
285            assert this.lockFileChannel == null;
286            assert this.indxFile == null;
287            assert this.keysFile == null;
288            assert this.valsFile == null;
289            assert this.modsFile == null;
290            assert this.modsFileOutput == null;
291            assert this.directoryChannel == null;
292            assert this.modsFileLength == 0;
293            assert this.modsFileSyncPoint == 0;
294            assert this.indx == null;
295            assert this.keys == null;
296            assert this.vals == null;
297            assert this.kvstore == null;
298            assert this.mods == null;
299            assert this.firstModTimestamp == 0;
300
301            // Check configuration
302            Preconditions.checkState(this.directory != null, "no directory configured");
303
304            // Create executor if needed
305            this.createdExecutorService = this.scheduledExecutorService == null;
306            if (this.createdExecutorService) {
307                this.scheduledExecutorService = Executors.newSingleThreadScheduledExecutor(new ThreadFactory() {
308                    @Override
309                    public Thread newThread(Runnable action) {
310                        final Thread thread = new Thread(action);
311                        thread.setName("Compactor for " + AtomicArrayKVStore.this);
312                        return thread;
313                    }
314                });
315            }
316
317            // Create directory if needed
318            boolean initializeFiles = false;
319            if (!this.directory.exists()) {
320                if (!this.directory.mkdirs())
321                    throw new ArrayKVException("failed to create directory `" + this.directory + "'");
322            }
323            if (!this.directory.isDirectory())
324                throw new ArrayKVException("file `" + this.directory + "' is not a directory");
325
326            // Get directory channel we can fsync()
327            this.directoryChannel = FileChannel.open(this.directory.toPath());
328
329            // Open and lock the lock file
330            this.lockFile = new File(this.directory, LOCK_FILE_NAME);
331            if (!this.lockFile.exists()) {
332                this.lockFileChannel = FileChannel.open(this.lockFile.toPath(),
333                  StandardOpenOption.WRITE, StandardOpenOption.CREATE_NEW);
334            } else {
335                this.lockFileChannel = FileChannel.open(this.lockFile.toPath(),
336                  StandardOpenOption.WRITE, StandardOpenOption.TRUNCATE_EXISTING);
337            }
338            FileLock fileLock = null;
339            try {
340                fileLock = this.lockFileChannel.tryLock();
341            } catch (OverlappingFileLockException e) {
342                // too bad
343            }
344            if (fileLock == null)
345                throw new ArrayKVException("database is already locked by another process or thread");
346
347            // If no generation file exists, initialize generation zero
348            this.generationFile = new File(this.directory, GENERATION_FILE_NAME);
349            if (!this.generationFile.exists()) {
350
351                // Create empty index, keys, and values files
352                try (
353                  final FileOutputStream indxOutput = new FileOutputStream(new File(this.directory, INDX_FILE_NAME_BASE + 0));
354                  final FileOutputStream keysOutput = new FileOutputStream(new File(this.directory, KEYS_FILE_NAME_BASE + 0));
355                  final FileOutputStream valsOutput = new FileOutputStream(new File(this.directory, VALS_FILE_NAME_BASE + 0));
356                  final ArrayKVWriter arrayWriter = new ArrayKVWriter(indxOutput, keysOutput, valsOutput)) {
357                    arrayWriter.flush();                                                        // avoid compiler warning
358                    valsOutput.getChannel().force(false);
359                    keysOutput.getChannel().force(false);
360                    indxOutput.getChannel().force(false);
361                }
362
363                // Create generation file
364                try (FileOutputStream output = new FileOutputStream(this.generationFile)) {
365                    new PrintStream(output, true).println(0);
366                    output.getChannel().force(false);
367                }
368            }
369
370            // Read current generation number
371            try {
372                this.generation = Long.parseLong(
373                  new String(Files.readAllBytes(this.generationFile.toPath()), StandardCharsets.UTF_8).trim(), 10);
374                if (this.generation < 0)
375                    throw new ArrayKVException("read negative generation number from " + this.generationFile);
376            } catch (Exception e) {
377                throw new ArrayKVException("error reading generation file", e);
378            }
379
380            // Set corresponding filenames
381            this.indxFile = new File(this.directory, INDX_FILE_NAME_BASE + this.generation);
382            this.keysFile = new File(this.directory, KEYS_FILE_NAME_BASE + this.generation);
383            this.valsFile = new File(this.directory, VALS_FILE_NAME_BASE + this.generation);
384            this.modsFile = new File(this.directory, MODS_FILE_NAME_BASE + this.generation);
385
386            // Scan directory for unexpected files
387            final List<File> expectedFiles = Arrays.asList(this.lockFile, this.generationFile,
388              this.indxFile, this.keysFile, this.valsFile, this.modsFile);
389            for (File file : this.directory.listFiles()) {
390                if (!expectedFiles.contains(file))
391                    this.log.warn("ignoring unexpected file " + file.getName() + " in my database directory");
392            }
393
394            // Create buffers that wrap the index, keys, and values files
395            try (FileInputStream input = new FileInputStream(this.indxFile)) {
396                this.indx = AtomicArrayKVStore.getBuffer(this.indxFile, input.getChannel());
397            }
398            try (FileInputStream input = new FileInputStream(this.keysFile)) {
399                this.keys = AtomicArrayKVStore.getBuffer(this.keysFile, input.getChannel());
400            }
401            try (FileInputStream input = new FileInputStream(this.valsFile)) {
402                this.vals = AtomicArrayKVStore.getBuffer(this.valsFile, input.getChannel());
403            }
404
405            // Set up underlying k/v store and uncompacted modifications
406            this.kvstore = new ArrayKVStore(this.indx, this.keys, this.vals);
407            this.mods = new MutableView(this.kvstore, null, new Writes());
408
409            // Setup modifications file
410            this.modsFileOutput = new FileOutputStream(this.modsFile, true);
411            this.modsFileLength = this.modsFileOutput.getChannel().size();
412            this.modsFileSyncPoint = this.modsFileLength;
413
414            // Read and apply pre-existing uncompacted modifications from modifications file
415            if (this.modsFileLength > 0) {
416                this.log.info("reading " + this.modsFileLength + " bytes of uncompacted modifications from " + this.modsFile);
417                try (FileInputStream input = new FileInputStream(this.modsFile)) {
418                    while (input.available() > 0) {
419                        final Writes writes;
420                        try {
421                            writes = Writes.deserialize(input);
422                        } catch (Exception e) {
423                            break;                                                      // probably a partial write
424                        }
425                        writes.applyTo(this.mods);
426                    }
427                }
428                this.firstModTimestamp = System.nanoTime() | 1;                     // avoid zero value which is special
429            }
430
431            // Schedule compaction if necessary
432            this.scheduleCompactionIfNecessary();
433
434            // Done
435            success = true;
436        } catch (IOException e) {
437            throw new ArrayKVException("startup failed", e);
438        } finally {
439            try {
440                if (!success)
441                    this.cleanup();
442            } finally {
443                this.writeLock.unlock();
444            }
445        }
446    }

448    @Override
449    @PreDestroy
450    public void stop() {
451        this.writeLock.lock();
452        try {
453
454            // Check state
455            if (this.kvstore == null)
456                return;
457            this.log.info("stopping " + this);
458
459            // Cleanup
460            this.cleanup();
461        } finally {
462            this.writeLock.unlock();
463        }
464    }

466    private void cleanup() {
467
468        // Should hold write lock now
469        assert this.lock.isWriteLockedByCurrentThread();
470
471        // Cancel compaction if possible, otherwise wait for it to complete
472        if (this.compaction != null && !this.compaction.cancel()) {
473
474            // Wait for compaction to complete
475            this.log.debug("waiting for in-progress compaction to complete before shutdown");
476            this.compaction.waitForCompletion();
477            this.log.debug("compaction completed, proceeding with shutdown");
478
479            // Check whether another thread invoked stop() while we were asleep
480            if (this.kvstore == null)
481                return;
482        }
483
484        // Wait for any in-progress hot copies to complete
485        if (this.hotCopiesInProgress > 0) {
486            this.log.debug("waiting for " + this.hotCopiesInProgress + " hot copies to complete before shutdown");
487            boolean interrupted = false;
488            do {
489
490                // Wait for hot copies to complete
491                try {
492                    this.hotCopyFinishedCondition.await();
493                } catch (InterruptedException e) {
494                    this.log.warn("thread interrupted while waiting for "
495                      + this.hotCopiesInProgress + " hot copies to complete (ignoring)", e);
496                    interrupted = true;
497                }
498
499                // Check whether another thread invoked stop() while we were asleep
500                if (this.kvstore == null)
501                    return;
502            } while (this.hotCopiesInProgress > 0);
503            if (interrupted)
504                Thread.currentThread().interrupt();
505            this.log.debug("hot copies completed, proceeding with shutdown");
506        }
507
508        // Shut down executor - only if we created it
509        if (this.createdExecutorService) {
510            this.scheduledExecutorService.shutdownNow();
511            this.scheduledExecutorService = null;
512            this.createdExecutorService = false;
513        }
514
515        // Close files
516        for (Closeable closeable : new Closeable[] { this.modsFileOutput, this.directoryChannel, this.lockFileChannel }) {
517            if (closeable != null) {
518                try {
519                    closeable.close();
520                } catch (IOException e) {
521                    // ignore
522                }
523            }
524        }
525
526        // Reset state
527        this.generation = 0;
528        this.generationFile = null;
529        this.lockFile = null;
530        this.lockFileChannel = null;
531        this.indxFile = null;
532        this.keysFile = null;
533        this.valsFile = null;
534        this.modsFile = null;
535        this.modsFileOutput = null;
536        this.directoryChannel = null;
537        this.modsFileLength = 0;
538        this.modsFileSyncPoint = 0;
539        this.indx = null;
540        this.keys = null;
541        this.vals = null;
542        this.kvstore = null;
543        this.mods = null;
544        this.firstModTimestamp = 0;
545    }

645    @Override
646    public CloseableKVStore snapshot() {
647        this.readLock.lock();
648        try {
649            Preconditions.checkState(this.kvstore != null, "closed");
650
651            // Get outstanding modifications; if we are compacting, there are two sets of modifications
652            final Writes writes1;
653            if (this.mods.getKVStore() instanceof MutableView) {
654                final MutableView oldMods = (MutableView)this.mods.getKVStore();
655                assert oldMods.getKVStore() == this.kvstore;
656                writes1 = oldMods.getWrites();
657            } else
658                writes1 = null;
659            final Writes writes2 = this.mods.getWrites();
660
661            // Clone (non-empty) writes to construct snapshot
662            KVStore snapshot = this.kvstore;
663            if (writes1 != null && !writes1.isEmpty())
664                snapshot = new MutableView(snapshot, null, writes1.clone());
665            if (writes2 != null && !writes2.isEmpty())
666                snapshot = new MutableView(snapshot, null, writes2.clone());
667
668            // Done
669            return new CloseableForwardingKVStore(snapshot);
670        } finally {
671            this.readLock.unlock();
672        }
673    }

675    @Override
676    public void mutate(Mutations mutations, final boolean sync) {
677        Preconditions.checkArgument(mutations != null, "null mutations");
678        this.writeLock.lock();
679        try {
680
681            // Verify we are started
682            Preconditions.checkState(this.kvstore != null, "not started");
683
684            // Block until compaction completes if size of outstanding modifications exceeds high water mark
685            while (this.compaction != null && this.modsFileLength > this.compactHighWater) {
686
687                // If scheduled compaction is in the future, reschedule to be immediate
688                if (this.compaction.getDelay() > 0)
689                    this.scheduleCompaction(0);
690
691                // Wait for it to complete
692                this.log.debug("reached compaction high-water mark; waiting for in-progress compaction to complete"
693                  + " before applying mutation(s)");
694                this.compaction.waitForCompletion();
695                this.log.debug("compaction completed, proceeding with application of mutation(s)");
696
697                // Verify we are still open after giving up lock
698                if (this.kvstore == null)
699                    throw new ArrayKVException("k/v store was closed while waiting for compaction to complete");
700            }
701
702            // Get mutations as a Writes object
703            final Writes writes;
704            if (mutations instanceof Writes)
705                writes = (Writes)mutations;
706            else {
707                writes = new Writes();
708                writes.setRemoves(new KeyRanges(mutations.getRemoveRanges()));
709                for (Map.Entry<byte[], byte[]> entry : mutations.getPutPairs())
710                    writes.getPuts().put(entry.getKey(), entry.getValue());
711                for (Map.Entry<byte[], Long> entry : mutations.getAdjustPairs())
712                    writes.getAdjusts().put(entry.getKey(), entry.getValue());
713            }
714
715            // Check for the trivial case where there are zero modifications
716            if (writes.isEmpty())
717                return;
718
719            // Append mutations to uncompacted mods file
720            try {
721                writes.serialize(this.modsFileOutput);
722                this.modsFileOutput.flush();
723            } catch (IOException e) {
724                try {
725                    this.modsFileOutput.getChannel().truncate(this.modsFileLength);              // undo append
726                } catch (IOException e2) {
727                    this.log.error("error truncating log file (ignoring)", e2);
728                }
729                throw new ArrayKVException("error appending to " + this.modsFile, e);
730            }
731            final long newModsFileLength;
732            try {
733                newModsFileLength = this.modsFileOutput.getChannel().size();
734            } catch (IOException e) {
735                throw new ArrayKVException("error getting length of " + this.modsFile, e);
736            }
737            if (this.log.isDebugEnabled()) {
738                this.log.debug("appended " + (newModsFileLength - this.modsFileLength) + " bytes to "
739                  + this.modsFile + " (new length " + newModsFileLength + ")");
740            }
741            this.modsFileLength = newModsFileLength;
742
743            // Apply removes
744            for (KeyRange range : mutations.getRemoveRanges()) {
745                final byte[] min = range.getMin();
746                final byte[] max = range.getMax();
747                this.mods.removeRange(min, max);
748            }
749
750            // Apply puts
751            for (Map.Entry<byte[], byte[]> entry : mutations.getPutPairs()) {
752                final byte[] key = entry.getKey();
753                final byte[] val = entry.getValue();
754                this.mods.put(key, val);
755            }
756
757            // Apply counter adjustments
758            for (Map.Entry<byte[], Long> entry : mutations.getAdjustPairs()) {
759                final byte[] key = entry.getKey();
760                final long adjust = entry.getValue();
761                this.mods.adjustCounter(key, adjust);
762            }
763
764            // Set first uncompacted modification timestamp, if not already set
765            if (this.firstModTimestamp == 0)
766                this.firstModTimestamp = System.nanoTime() | 1;                 // avoid zero value which is special
767
768            // Schedule compaction if necessary
769            this.scheduleCompactionIfNecessary();
770
771            // If we're not syncing, we're done
772            if (!sync)
773                return;
774
775            // Update sync point, so compaction knows to also sync these mods when it copies them
776            this.modsFileSyncPoint = this.modsFileLength;
777
778            // Downgrade lock
779            this.readLock.lock();
780        } finally {
781            this.writeLock.unlock();
782        }
783
784        // Sync the mods file while holding only the read lock
785        try {
786            this.modsFileOutput.getChannel().force(false);
787        } catch (IOException e) {
788            this.log.error("error syncing log file (ignoring)", e);
789        } finally {
790            this.readLock.unlock();
791        }
792    }

796    /**
797     * Create a "hot" copy of this instance in the specified destination directory.
798     *
799     * <p>
800     * The {@code target} directory will be created if it does not exist; otherwise, it must be empty.
801     * Files are copied using {@linkplain Files#createLink hard links} when possible.
802     *
803     * <p>
804     * The hot copy operation proceeds in parallel with normal database activity, with the exception
805     * that a compaction operation must wait until there are no concurrent hot copies to complete.
806     *
807     * <p>
808     * The {@code target} directory is not fsync()'d by this method; the caller
809     * must perform that action to ensure durability if required.
810     *
811     * @param target destination directory
812     * @throws IOException if an I/O error occurs
813     * @throws IllegalArgumentException if {@code target} exists and is not a directory or is non-empty
814     * @throws IllegalArgumentException if {@code target} is null
815     */
816    public void hotCopy(File target) throws IOException {
817
818        // Sanity check
819        Preconditions.checkArgument(target != null, "null target");
820        final Path dir = target.toPath();
821
822        // Create/verify directory
823        if (!Files.exists(dir))
824            Files.createDirectories(dir);
825        if (!Files.isDirectory(dir))
826            throw new IllegalArgumentException("target `" + dir + "' is not a directory");
827        try (DirectoryStream<Path> stream = Files.newDirectoryStream(dir)) {
828            for (Path path : stream)
829                throw new IllegalArgumentException("target `" + dir + "' is not empty");
830        }
831
832        // Increment hot copy counter - this prevents compaction from removing files while we're copying them
833        this.writeLock.lock();
834        try {
835
836            // Sanity check
837            Preconditions.checkState(this.kvstore != null, "not started");
838
839            // Bump counter
840            this.hotCopiesInProgress++;
841        } finally {
842            this.writeLock.unlock();
843        }
844        try {
845
846            // Logit
847            this.log.debug("started hot copy into " + target);
848
849            // Copy index, keys, and values files using hard links (if possible) as these files are read-only
850            final ArrayList<File> regularCopyFiles = new ArrayList<>(5);
851            for (File file : new File[] { this.indxFile, this.keysFile, this.valsFile }) {
852                try {
853                    Files.createLink(dir.resolve(file.getName()), file.toPath());
854                } catch (IOException | UnsupportedOperationException e) {
855                    regularCopyFiles.add(file);                      // fall back to normal copy
856                }
857            }
858
859            // Copy remaining files without using hard links
860            regularCopyFiles.add(this.modsFile);                     // it's ok if we copy a partial write
861            regularCopyFiles.add(this.generationFile);               // copy this one last
862            for (File file : regularCopyFiles)
863                Files.copy(file.toPath(), dir.resolve(file.getName()));
864        } finally {
865            this.writeLock.lock();
866            try {
867
868                // Sanity check
869                assert this.hotCopiesInProgress > 0;
870                assert this.kvstore != null;
871
872                // Logit
873                this.log.debug("completed hot copy into " + target);
874
875                // Decrement counter
876                this.hotCopiesInProgress--;
877
878                // Wakeup waiters
879                this.hotCopyFinishedCondition.signalAll();
880            } finally {
881                this.writeLock.unlock();
882            }
883        }
884    }

888    /**
889     * Schedule a new compaction cycle, unless there is one already scheduled or running, or there are no
890     * outstanding uncompacted modifications.
891     *
892     * @return a future for the completion of the next compaction cycle, or null if there are no uncompacted modifications
893     * @throws IllegalStateException if this instance is not started
894     */
895    public Future<?> scheduleCompaction() {
896        this.writeLock.lock();
897        try {
898
899            // Sanity check
900            Preconditions.checkState(this.kvstore != null, "not started");
901
902            // Is there anything to compact?
903            if (this.modsFileLength == 0)
904                return null;
905
906            // Schedule an immediate compaction
907            return this.scheduleCompaction(0);
908        } finally {
909            this.writeLock.unlock();
910        }
911    }
912
913    private Future<?> scheduleCompaction(long millis) {
914
915        // Should hold write lock now
916        assert this.lock.isWriteLockedByCurrentThread();
917        assert this.modsFileLength > 0;
918
919        // (Re)schedule compaction if necessary
920        if (this.compaction == null || (this.compaction.getDelay() > millis && this.compaction.cancel())) {
921            assert this.compaction == null;
922            this.compaction = new Compaction(millis);
923        }
924
925        // Done
926        return new ForwardingFuture.SimpleForwardingFuture<Void>(this.compaction.getFuture()) {
927            @Override
928            public boolean cancel(boolean mayInterrupt) {               // disallow this operation
929                return false;
930            }
931        };
932    }
933
934    private void scheduleCompactionIfNecessary() {
935
936        // Should hold write lock now
937        assert this.lock.isWriteLockedByCurrentThread();
938
939        // Is there anything to do?
940        if (this.modsFileLength == 0)
941            return;
942
943        // Check space low-water mark
944        if (this.modsFileLength > this.compactLowWater) {
945            this.scheduleCompaction(0);
946            return;
947        }
948
949        // Check maximum compaction delay
950        if (this.firstModTimestamp != 0) {
951            final long firstModAgeMillis = (System.nanoTime() - this.firstModTimestamp) / 1000000L;     // convert ns -> ms
952            this.scheduleCompaction(firstModAgeMillis);
953            return;
954        }
955    }

957    @SuppressWarnings("fallthrough")
958    private void compact(final Compaction compaction) throws IOException {
959
960        // Sanity check
961        assert compaction != null;
962
963        // Handle cancel() race condition
964        this.writeLock.lock();
965        try {
966
967            // Were we canceled?
968            if (compaction != this.compaction)
969                return;
970
971            // Set now running - this prevents future cancel()'s
972            assert !compaction.isRunning();
973            assert !compaction.isCompleted();
974            compaction.setRunning();
975        } finally {
976            this.writeLock.unlock();
977        }
978
979        // Start compaction
980        final long compactionStartTime;
981        try {
982
983            // Snapshot (and wrap) pending modifications, and mark log file position
984            final Writes writesToCompact;
985            final long previousModsFileLength;
986            final long previousModsFileSyncPoint;
987            this.writeLock.lock();
988            try {
989
990                // Sanity checks
991                assert this.kvstore != null;
992                assert this.modsFileLength > 0;
993
994                // Mark start time and get uncompacted modifications
995                compactionStartTime = System.nanoTime();
996                writesToCompact = this.mods.getWrites();
997
998                // It's possible the uncompacted modifications are a no-op; if so, no compaction is necessary
999                if (writesToCompact.isEmpty()) {
1000
1001                    // Wait for any in-progress hot copies to complete
1002                    while (this.hotCopiesInProgress > 0) {
1003                        this.log.debug("waiting for " + this.hotCopiesInProgress
1004                          + " hot copies to complete before completing (trivial) compaction");
1005                        try {
1006                            this.hotCopyFinishedCondition.await();
1007                        } catch (InterruptedException e) {
1008                            throw new ArrayKVException("thread was interrupted while waiting for "
1009                              + this.hotCopiesInProgress + " hot copies to complete", e);
1010                        }
1011                        this.log.debug("hot copies completed, proceeding with completion of (trivial) compaction");
1012                    }
1013                    assert this.kvstore != null;
1014                    assert this.modsFileLength > 0;
1015
1016                    // Discard outstanding mods
1017                    this.modsFileOutput.getChannel().truncate(0);
1018                    this.modsFileLength = 0;
1019                    this.modsFileSyncPoint = 0;
1020                    this.modsFileOutput.getChannel().force(false);
1021                    return;
1022                }
1023
1024                // Allow new modifications to be added by other threads while we are compacting the old modifications
1025                this.mods = new MutableView(this.mods, null, new Writes());
1026                previousModsFileLength = this.modsFileLength;
1027                previousModsFileSyncPoint = this.modsFileSyncPoint;
1028            } finally {
1029                this.writeLock.unlock();
1030            }
1031            if (this.log.isDebugEnabled()) {
1032                this.log.debug("starting compaction for generation " + this.generation + " -> " + (this.generation + 1)
1033                  + " with mods file length " + previousModsFileLength);
1034            }
1035
1036            // Create the next generation
1037            final long newGeneration = this.generation + 1;
1038            final File newIndxFile = new File(this.directory, INDX_FILE_NAME_BASE + newGeneration);
1039            final File newKeysFile = new File(this.directory, KEYS_FILE_NAME_BASE + newGeneration);
1040            final File newValsFile = new File(this.directory, VALS_FILE_NAME_BASE + newGeneration);
1041            final File newModsFile = new File(this.directory, MODS_FILE_NAME_BASE + newGeneration);
1042            ByteBuffer newIndx = null;
1043            ByteBuffer newKeys = null;
1044            ByteBuffer newVals = null;
1045            FileOutputStream newModsFileOutput = null;
1046            boolean success = false;
1047            try {
1048
1049                // Merge existing compacted key/value data with uncompacted modifications
1050                try (
1051                  final FileOutputStream indxOutput = new FileOutputStream(newIndxFile);
1052                  final FileOutputStream keysOutput = new FileOutputStream(newKeysFile);
1053                  final FileOutputStream valsOutput = new FileOutputStream(newValsFile);
1054                  final ArrayKVWriter arrayWriter = new ArrayKVWriter(indxOutput, keysOutput, valsOutput)) {
1055
1056                    // Write out merged key/value pairs
1057                    arrayWriter.writeMerged(this.kvstore, this.kvstore.getRange(null, null, false), writesToCompact);
1058
1059                    // Sync file data
1060                    arrayWriter.flush();
1061                    valsOutput.getChannel().force(false);
1062                    keysOutput.getChannel().force(false);
1063                    indxOutput.getChannel().force(false);
1064                }
1065                assert newIndxFile.exists();
1066                assert newKeysFile.exists();
1067                assert newValsFile.exists();
1068
1069                // Create byte buffers from new files
1070                try (FileInputStream input = new FileInputStream(newIndxFile)) {
1071                    newIndx = AtomicArrayKVStore.getBuffer(newIndxFile, input.getChannel());
1072                }
1073                try (FileInputStream input = new FileInputStream(newKeysFile)) {
1074                    newKeys = AtomicArrayKVStore.getBuffer(newKeysFile, input.getChannel());
1075                }
1076                try (FileInputStream input = new FileInputStream(newValsFile)) {
1077                    newVals = AtomicArrayKVStore.getBuffer(newValsFile, input.getChannel());
1078                }
1079
1080                // Create new, empty mods file
1081                newModsFile.delete();                                           // shouldn't exist, but just in case...
1082                newModsFileOutput = new FileOutputStream(newModsFile, true);
1083                assert newModsFile.exists();
1084
1085                // Sync directory
1086                this.directoryChannel.force(false);
1087
1088                // We're done creating files
1089                success = true;
1090            } finally {
1091
1092                // Finish up
1093                this.writeLock.lock();
1094                try {
1095                    if (success) {
1096
1097                        // Initialize for wrap-up
1098                        success = false;
1099
1100                        // Size up additional mods
1101                        final long additionalModsLength = this.modsFileLength - previousModsFileLength;
1102
1103                        // Logit
1104                        if (this.log.isDebugEnabled()) {
1105                            final float duration = (System.nanoTime() - compactionStartTime) / 1000000000f;
1106                            this.log.debug("compaction for generation " + this.generation + " -> " + (this.generation + 1)
1107                              + " finishing up with " + additionalModsLength + " bytes of new modifications after "
1108                              + String.format("%.4f", duration) + " seconds");
1109                        }
1110
1111                        // Wait for any in-progress hot copies to complete
1112                        while (this.hotCopiesInProgress > 0) {
1113                            this.log.debug("waiting for " + this.hotCopiesInProgress
1114                              + " hot copies to complete before completing compaction");
1115                            try {
1116                                this.hotCopyFinishedCondition.await();
1117                            } catch (InterruptedException e) {
1118                                throw new ArrayKVException("thread was interrupted while waiting for "
1119                                  + this.hotCopiesInProgress + " hot copies to complete", e);
1120                            }
1121                            assert this.compaction == compaction;
1122                            assert this.kvstore != null;
1123                            this.log.debug("hot copies completed, proceeding with completion of compaction");
1124                        }
1125
1126                        // Apply any changes that were made while we were unlocked and writing files
1127                        long newModsFileLength = 0;
1128                        long newModsFileSyncPoint = 0;
1129                        if (additionalModsLength > 0) {
1130                            try (
1131                              final FileChannel modsFileChannel = FileChannel.open(this.modsFile.toPath());
1132                              final FileChannel newModsFileChannel = FileChannel.open(newModsFile.toPath(),
1133                               StandardOpenOption.WRITE, StandardOpenOption.APPEND)) {
1134
1135                                // Append new data in old mods file to new mods file
1136                                while (newModsFileLength < additionalModsLength) {
1137                                    final long transferred = modsFileChannel.transferTo(
1138                                      previousModsFileLength + newModsFileLength, additionalModsLength - newModsFileLength,
1139                                      newModsFileChannel);
1140                                    assert transferred > 0;
1141                                    newModsFileLength += transferred;
1142                                }
1143
1144                                // If any of that new data was fsync()'d, we must fsync() the copy of it in the new mods file
1145                                if (this.modsFileSyncPoint > previousModsFileSyncPoint) {
1146                                    newModsFileChannel.force(false);
1147                                    newModsFileSyncPoint = newModsFileLength;
1148                                }
1149                            }
1150                        }
1151
1152                        // Atomically update new generation file contents
1153                        final AtomicUpdateFileOutputStream genOutput = new AtomicUpdateFileOutputStream(this.generationFile);
1154                        boolean genSuccess = false;
1155                        try {
1156                            new PrintStream(genOutput, true).println(newGeneration);
1157                            genOutput.getChannel().force(false);
1158                            genSuccess = true;
1159                        } finally {
1160                            if (genSuccess)
1161                                genOutput.close();
1162                            else
1163                                genOutput.cancel();
1164                        }
1165
1166                        // Declare success
1167                        success = true;
1168
1169                        // Remember old info so we can clean it up
1170                        final File oldIndxFile = this.indxFile;
1171                        final File oldKeysFile = this.keysFile;
1172                        final File oldValsFile = this.valsFile;
1173                        final File oldModsFile = this.modsFile;
1174                        final FileOutputStream oldModsFileOutput = this.modsFileOutput;
1175
1176                        // Change to the new generation
1177                        this.generation = newGeneration;
1178                        this.indx = newIndx;
1179                        this.keys = newKeys;
1180                        this.vals = newVals;
1181                        this.indxFile = newIndxFile;
1182                        this.keysFile = newKeysFile;
1183                        this.valsFile = newValsFile;
1184                        this.modsFile = newModsFile;
1185                        this.modsFileOutput = newModsFileOutput;
1186                        this.modsFileLength = newModsFileLength;
1187                        this.modsFileSyncPoint = newModsFileSyncPoint;
1188                        this.kvstore = new ArrayKVStore(this.indx, this.keys, this.vals);
1189                        this.mods = new MutableView(this.kvstore, null, this.mods.getWrites());
1190
1191                        // Sync directory prior to deleting files
1192                        try {
1193                            this.directoryChannel.force(false);
1194                        } catch (IOException e) {
1195                            this.log.error("error syncing directory " + this.directory + " (ignoring)", e);
1196                        }
1197
1198                        // Delete old files
1199                        oldIndxFile.delete();
1200                        oldKeysFile.delete();
1201                        oldValsFile.delete();
1202                        oldModsFile.delete();
1203
1204                        // Close old mods file output stream
1205                        try {
1206                            oldModsFileOutput.close();
1207                        } catch (IOException e) {
1208                            // ignore
1209                        }
1210                    }
1211                } finally {
1212                    try {
1213
1214                        // Logit
1215                        if (this.log.isDebugEnabled()) {
1216                            final float duration = (System.nanoTime() - compactionStartTime) / 1000000000f;
1217                            this.log.debug("compaction for generation " + (newGeneration - 1) + " -> " + newGeneration
1218                              + (success ? " succeeded" : " failed") + " in " + String.format("%.4f", duration) + " seconds");
1219                        }
1220
1221                        // Cleanup/undo on failure
1222                        if (!success) {
1223
1224                            // Put back the old uncompacted modifications, and merge any new mods into them
1225                            final Writes writesDuringCompaction = this.mods.getWrites();
1226                            this.mods = new MutableView(this.kvstore, null, writesToCompact);
1227                            writesDuringCompaction.applyTo(this.mods);
1228
1229                            // Delete the files we were creating
1230                            newIndxFile.delete();
1231                            newKeysFile.delete();
1232                            newValsFile.delete();
1233                            if (newModsFileOutput != null) {
1234                                newModsFile.delete();
1235                                try {
1236                                    newModsFileOutput.close();
1237                                } catch (Exception e) {
1238                                    // ignore
1239                                }
1240                            }
1241                        }
1242                    } finally {
1243                        this.writeLock.unlock();
1244                    }
1245                }
1246            }
1247        } finally {
1248
1249            // Update state
1250            this.writeLock.lock();
1251            try {
1252                assert this.kvstore != null;
1253                assert this.compaction == compaction;
1254                compaction.setCompleted();
1255                this.compaction = null;
1256            } finally {
1257                this.writeLock.unlock();
1258            }
1259        }
1260    }

1264    private class Compaction implements Runnable {
1265
1266        private final Condition completedCondition = AtomicArrayKVStore.this.writeLock.newCondition();
1267        private final ScheduledFuture<Void> future;
1268
1269        private boolean running;
1270        private boolean completed;
1271
1272        @SuppressWarnings("unchecked")
1273        Compaction(long millis) {
1274
1275            // Sanity check
1276            assert AtomicArrayKVStore.this.lock.isWriteLockedByCurrentThread();
1277            Preconditions.checkState(AtomicArrayKVStore.this.compaction == null, "compaction already exists");
1278
1279            // Schedule
1280            this.future = (ScheduledFuture<Void>)AtomicArrayKVStore.this.scheduledExecutorService.schedule(this,
1281              millis, TimeUnit.MILLISECONDS);
1282        }
1283
1284        /**
1285         * Cancel this compaction.
1286         *
1287         * @return true if canceled, false if already running
1288         */
1289        public boolean cancel() {
1290
1291            // Sanity check
1292            assert AtomicArrayKVStore.this.lock.isWriteLockedByCurrentThread();
1293            Preconditions.checkState(this.future != null, "not scheduled");
1294
1295            // Already canceled?
1296            if (AtomicArrayKVStore.this.compaction != this)
1297                return false;
1298
1299            // Already running?
1300            if (this.running)
1301                return false;
1302
1303            // Attempt to cancel scheduled task
1304            assert this.future != null;
1305            this.future.cancel(false);
1306
1307            // Make this task do nothing even if it runs anyway
1308            AtomicArrayKVStore.this.compaction = null;
1309            return true;
1310        }
1311
1312        /**
1313         * Get delay until start.
1314         *
1315         * @return delay in milliseconds
1316         */
1317        public long getDelay() {
1318
1319            // Sanity check
1320            assert AtomicArrayKVStore.this.lock.isWriteLockedByCurrentThread();
1321            Preconditions.checkState(this.future != null, "not scheduled");
1322
1323            // Get delay
1324            return Math.max(0, this.future.getDelay(TimeUnit.MILLISECONDS));
1325        }
1326
1327        /**
1328         * Wait for this compaction to complete.
1329         */
1330        public void waitForCompletion() {
1331
1332            // Sanity check
1333            assert AtomicArrayKVStore.this.lock.isWriteLockedByCurrentThread();
1334            Preconditions.checkState(this.running, "not running");
1335
1336            // Wait for completion
1337            boolean interrupted = false;
1338            while (!this.completed) {
1339                try {
1340                    this.completedCondition.await();
1341                } catch (InterruptedException e) {
1342                    AtomicArrayKVStore.this.log.warn(
1343                      "thread was interrupted while waiting for compaction to complete (ignoring)", e);
1344                    interrupted = true;
1345                }
1346            }
1347            if (interrupted)
1348                Thread.currentThread().interrupt();
1349        }
1350
