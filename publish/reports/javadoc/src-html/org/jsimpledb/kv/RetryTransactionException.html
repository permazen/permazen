006package org.jsimpledb.kv;
007
008/**
009 * Exception thrown when an attempt to access a {@link KVTransaction} results
010 * in a conflict requiring that the transaction be restarted.
011 */
012@SuppressWarnings("serial")
013public class RetryTransactionException extends KVTransactionException {
014
015    private static final String DEFAULT_MESSAGE = "transaction must be retried";
016
017    public RetryTransactionException(KVTransaction kvt) {
018        super(kvt, DEFAULT_MESSAGE);
019    }
020
021    public RetryTransactionException(KVTransaction kvt, Throwable cause) {
022        super(kvt, DEFAULT_MESSAGE, cause);
023    }
024
025    public RetryTransactionException(KVTransaction kvt, String message) {
026        super(kvt, message);
027    }
028
029    public RetryTransactionException(KVTransaction kvt, String message, Throwable cause) {
030        super(kvt, message, cause);
031    }
032}
