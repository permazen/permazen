006package org.jsimpledb.kv;
007
008import java.util.Arrays;
009import java.util.Iterator;
010
011import org.jsimpledb.util.ByteReader;
012import org.jsimpledb.util.ByteUtil;
013import org.jsimpledb.util.ByteWriter;
014
015/**
016 * Support superclass for {@link KVStore} implementations.
017 *
018 * <p>
019 * This class provides a partial implementation via the following methods:
020 * <ul>
021 *  <li>A {@link #get get()} implementation based on {@link #getAtLeast getAtLeast()}</li>
022 *  <li>{@link #getAtLeast getAtLeast()} and {@link #getAtMost getAtMost()} implementations based on
023 *      {@link #getRange getRange()}.</li>
024 *  <li>A {@link #remove remove()} implementation that delegates to {@link #removeRange removeRange()}.</li>
025 *  <li>A {@link #removeRange removeRange()} implementation that delegates to {@link #getRange getRange()},
026 *      iterating through the range of keys and removing them one-by-one via {@link Iterator#remove}.</li>
027 *  <li>{@link #encodeCounter encodeCounter()}, {@link #decodeCounter encodeCounter()}, and
028 *      {@link #adjustCounter adjustCounter()} implementations using normal reads and writes
029 *      of values in big-endian encoding (does not provide any lock-free behavior).</li>
030 *  <li>A {@link #put put()} implementation throwing {@link UnsupportedOperationException}</li>
031 * </ul>
032 *
033 * <p>
034 * Therefore, a read-only {@link KVStore} implementation is possible simply by implementing {@link #getRange}.
035 * </p>
036 */
037public abstract class AbstractKVStore implements KVStore {
038
039    protected AbstractKVStore() {
040    }
041
042    @Override
043    public byte[] get(byte[] key) {
044        final KVPair pair = this.getAtLeast(key);
045        return pair != null && Arrays.equals(pair.getKey(), key) ? pair.getValue() : null;
046    }
047
048    @Override
049    public KVPair getAtLeast(byte[] minKey) {
050        final Iterator<KVPair> i = this.getRange(minKey, null, false);
051        try {
052            return i.hasNext() ? i.next() : null;
053        } finally {
054            this.closeIfPossible(i);
055        }
056    }
057
058    @Override
059    public KVPair getAtMost(byte[] maxKey) {
060        final Iterator<KVPair> i = this.getRange(null, maxKey, true);
061        try {
062            return i.hasNext() ? i.next() : null;
063        } finally {
064            this.closeIfPossible(i);
065        }
066    }
067
068    @Override
069    public void put(byte[] key, byte[] value) {
070        throw new UnsupportedOperationException();
071    }
072
073    @Override
074    public void remove(byte[] key) {
075        this.removeRange(key, ByteUtil.getNextKey(key));
076    }
077
078    @Override
079    public void removeRange(byte[] minKey, byte[] maxKey) {
080        final Iterator<KVPair> i = this.getRange(minKey, maxKey, false);
081        try {
082            while (i.hasNext()) {
083                i.next();
084                i.remove();
085            }
086        } finally {
087            this.closeIfPossible(i);
088        }
089    }
090
091    @Override
092    public byte[] encodeCounter(long value) {
093        final ByteWriter writer = new ByteWriter(8);
094        ByteUtil.writeLong(writer, value);
095        return writer.getBytes();
096    }
097
098    @Override
099    public long decodeCounter(byte[] value) {
100        if (value.length != 8)
101            throw new IllegalArgumentException("invalid encoded counter value: length = " + value.length + " != 8");
102        return ByteUtil.readLong(new ByteReader(value));
103    }
104
105    @Override
106    public void adjustCounter(byte[] key, long amount) {
107        if (key == null)
108            throw new NullPointerException("null key");
109        final byte[] previous = this.get(key);
110        if (previous == null)
111            return;
112        this.put(key, this.encodeCounter(this.decodeCounter(previous) + amount));
113    }
114
115    private void closeIfPossible(Iterator<KVPair> i) {
116        if (i instanceof AutoCloseable) {
117            try {
118                ((AutoCloseable)i).close();
119            } catch (Exception e) {
120                // ignore
121            }
122        }
123    }
124}
