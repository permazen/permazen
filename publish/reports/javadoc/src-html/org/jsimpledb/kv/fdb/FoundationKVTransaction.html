007

009import com.foundationdb.KeyValue;
010import com.foundationdb.MutationType;
011import com.foundationdb.Range;
012import com.foundationdb.ReadTransaction;
013import com.foundationdb.Transaction;
014import com.foundationdb.async.AsyncIterator;
015import com.google.common.base.Function;
016import com.google.common.collect.Iterators;
017import com.google.common.primitives.Bytes;
018
019import java.util.Iterator;
020
021import org.jsimpledb.kv.KVPair;
022import org.jsimpledb.kv.KVTransaction;
023import org.jsimpledb.kv.KVTransactionException;
024import org.jsimpledb.kv.RetryTransactionException;
025import org.jsimpledb.kv.StaleTransactionException;
026import org.jsimpledb.kv.TransactionTimeoutException;
027import org.jsimpledb.util.ByteReader;

031/**
032 * FoundationDB transaction.
033 */
034public class FoundationKVTransaction implements KVTransaction {
035
036    private static final byte[] MIN_KEY = ByteUtil.EMPTY;                   // minimum possible key (inclusive)
037    private static final byte[] MAX_KEY = new byte[] { (byte)0xff };        // maximum possible key (exclusive)
038
039    private final FoundationKVDatabase store;
040    private final Transaction tx;
041    private final byte[] keyPrefix;
042
043    private volatile boolean stale;
044    private volatile boolean canceled;
045
046    /**
047     * Constructor.
048     */
049    FoundationKVTransaction(FoundationKVDatabase store, byte[] keyPrefix) {
050        if (store == null)
051            throw new IllegalArgumentException("null store");
052        this.store = store;
053        this.tx = this.store.getDatabase().createTransaction();
054        this.keyPrefix = keyPrefix;
055    }
056
057// KVTransaction
058
059    @Override
060    public FoundationKVDatabase getKVDatabase() {
061        return this.store;
062    }
063
064    /**
065     * Get the underlying {@link Transaction} associated with this instance.
066     *
067     * @return the associated transaction
068     */
069    public Transaction getTransaction() {
070        return this.tx;
071    }
072
073    @Override
074    public void setTimeout(long timeout) {
075        if (timeout < 0)
076            throw new IllegalArgumentException("timeout < 0");
077        this.tx.options().setTimeout(timeout);
078    }
079
080    @Override
081    public byte[] get(byte[] key) {
082        if (this.stale)
083            throw new StaleTransactionException(this);
084        if (key.length > 0 && key[0] == (byte)0xff)
085            throw new IllegalArgumentException("key starts with 0xff");
086        try {
087            return this.tx.get(this.addPrefix(key)).get();
088        } catch (FDBException e) {
089            throw this.wrapException(e);
090        }
091    }
092
093    @Override
094    public KVPair getAtLeast(byte[] minKey) {
095        if (this.stale)
096            throw new StaleTransactionException(this);
097        if (minKey != null && minKey.length > 0 && minKey[0] == (byte)0xff)
098            return null;
099        return this.getFirstInRange(minKey, null, false);
100    }
101
102    @Override
103    public KVPair getAtMost(byte[] maxKey) {
104        if (this.stale)
105            throw new StaleTransactionException(this);
106        if (maxKey != null && maxKey.length > 0 && maxKey[0] == (byte)0xff)
107            maxKey = null;
108        return this.getFirstInRange(null, maxKey, true);
109    }
110
111    @Override
112    public Iterator<KVPair> getRange(byte[] minKey, byte[] maxKey, boolean reverse) {
113        if (this.stale)
114            throw new StaleTransactionException(this);
115        if (minKey != null && minKey.length > 0 && minKey[0] == (byte)0xff)
116            minKey = MAX_KEY;
117        if (maxKey != null && maxKey.length > 0 && maxKey[0] == (byte)0xff)
118            maxKey = null;
119        if (minKey != null && maxKey != null && ByteUtil.compare(minKey, maxKey) > 0)
120            throw new IllegalArgumentException("minKey > maxKey");
121        try {
122            return Iterators.transform(this.tx.getRange(this.addPrefix(minKey, maxKey),
123              ReadTransaction.ROW_LIMIT_UNLIMITED, reverse).iterator(), new Function<KeyValue, KVPair>() {
124                @Override
125                public KVPair apply(KeyValue kv) {
126                    return new KVPair(FoundationKVTransaction.this.removePrefix(kv.getKey()), kv.getValue());
127                }
128            });
129        } catch (FDBException e) {
130            throw this.wrapException(e);
131        }
132    }
133
134    private KVPair getFirstInRange(byte[] minKey, byte[] maxKey, boolean reverse) {
135        try {
136            final AsyncIterator<KeyValue> i = this.tx.getRange(this.addPrefix(minKey, maxKey),
137              ReadTransaction.ROW_LIMIT_UNLIMITED, reverse).iterator();
138            if (!i.hasNext())
139                return null;
140            final KeyValue kv = i.next();
141            return new KVPair(this.removePrefix(kv.getKey()), kv.getValue());
142        } catch (FDBException e) {
143            throw this.wrapException(e);
144        }
145    }
146
147    @Override
148    public void put(byte[] key, byte[] value) {
149        if (this.stale)
150            throw new StaleTransactionException(this);
151        if (key.length > 0 && key[0] == (byte)0xff)
152            throw new IllegalArgumentException("key starts with 0xff");
153        try {
154            this.tx.set(this.addPrefix(key), value);
155        } catch (FDBException e) {
156            throw this.wrapException(e);
157        }
158    }
159
160    @Override
161    public void remove(byte[] key) {
162        if (this.stale)
163            throw new StaleTransactionException(this);
164        if (key.length > 0 && key[0] == (byte)0xff)
165            throw new IllegalArgumentException("key starts with 0xff");
166        try {
167            this.tx.clear(this.addPrefix(key));
168        } catch (FDBException e) {
169            throw this.wrapException(e);
170        }
171    }
172
173    @Override
174    public void removeRange(byte[] minKey, byte[] maxKey) {
175        if (this.stale)
176            throw new StaleTransactionException(this);
177        if (minKey != null && minKey.length > 0 && minKey[0] == (byte)0xff)
178            return;
179        if (maxKey != null && maxKey.length > 0 && maxKey[0] == (byte)0xff)
180            maxKey = null;
181        if (minKey != null && maxKey != null && ByteUtil.compare(minKey, maxKey) > 0)
182            throw new IllegalArgumentException("minKey > maxKey");
183        try {
184            this.tx.clear(this.addPrefix(minKey, maxKey));
185        } catch (FDBException e) {
186            throw this.wrapException(e);
187        }
188    }
189
190    @Override
191    public void commit() {
192        if (this.stale)
193            throw new StaleTransactionException(this);
194        this.stale = true;
195        try {
196            this.tx.commit().get();
197        } catch (FDBException e) {
198            throw this.wrapException(e);
199        } finally {
200            this.cancel();
201        }
202    }
203
204    @Override
205    public void rollback() {
206        if (this.stale)
207            return;
208        this.stale = true;
209        this.cancel();
210    }
211
212    private void cancel() {
213        if (this.canceled)
214            return;
215        this.canceled = true;
216        try {
217            this.tx.cancel();
218        } catch (FDBException e) {
219            // ignore
220        }
221    }
222
223    @Override
224    public byte[] encodeCounter(long value) {
225        final ByteWriter writer = new ByteWriter(8);
226        ByteUtil.writeLong(writer, value);
227        final byte[] bytes = writer.getBytes();
228        this.reverse(bytes);
229        return bytes;
230    }
231
232    @Override
233    public long decodeCounter(byte[] bytes) {
234        if (this.stale)
235            throw new StaleTransactionException(this);
236        if (bytes.length != 8)
237            throw new IllegalArgumentException("invalid encoded counter value: length = " + bytes.length + " != 8");
238        bytes = bytes.clone();
239        this.reverse(bytes);
240        return ByteUtil.readLong(new ByteReader(bytes));
241    }
242
243    @Override
244    public void adjustCounter(byte[] key, long amount) {
245        if (this.stale)
246            throw new StaleTransactionException(this);
247        this.tx.mutate(MutationType.ADD, this.addPrefix(key), this.encodeCounter(amount));
248    }
249
250    private void reverse(byte[] bytes) {
251        int i = 0;
252        int j = bytes.length - 1;
253        while (i < j) {
254            final byte temp = bytes[i];
255            bytes[i] = bytes[j];
256            bytes[j] = temp;
257            i++;
258            j--;
259        }
260    }
261
262// Other methods
263
264    /**
265     * Wrap the given {@link FDBException} in the appropriate {@link KVTransactionException}.
266     *
267     * @param e FoundationDB exception
268     * @return appropriate {@link KVTransactionException} with chained exception {@code e}
269     * @throws NullPointerException if {@code e} is null
270     */
271    public KVTransactionException wrapException(FDBException e) {
272        try {
273            this.cancel();
274        } catch (KVTransactionException e2) {
275            // ignore
276        }
277        switch (e.getCode()) {
278        case ErrorCodes.TRANSACTION_TIMED_OUT:
279        case ErrorCodes.PAST_VERSION:
280            return new TransactionTimeoutException(this, e);
281        case ErrorCodes.NOT_COMMITTED:
282        case ErrorCodes.COMMIT_UNKNOWN_RESULT:
283            return new RetryTransactionException(this, e);
284        default:
285            return new KVTransactionException(this, e);
286        }
287    }
288
289// Key prefixing
290
291    private byte[] addPrefix(byte[] key) {
292        return this.keyPrefix != null ? Bytes.concat(this.keyPrefix, key) : key;
293    }
294
295    private Range addPrefix(byte[] minKey, byte[] maxKey) {
296        return new Range(this.addPrefix(minKey != null ? minKey : MIN_KEY), this.addPrefix(maxKey != null ? maxKey : MAX_KEY));
297    }
298
299    private byte[] removePrefix(byte[] key) {
300        if (this.keyPrefix == null)
301            return key;
302        if (!ByteUtil.isPrefixOf(this.keyPrefix, key)) {
303            throw new IllegalArgumentException("read key " + ByteUtil.toString(key) + " not having "
304              + ByteUtil.toString(this.keyPrefix) + " as a prefix");
305        }
306        final byte[] stripped = new byte[key.length - this.keyPrefix.length];
307        System.arraycopy(key, this.keyPrefix.length, stripped, 0, stripped.length);
308        return stripped;
309    }
