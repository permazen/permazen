006package org.jsimpledb.kv.fdb;
007
008import com.foundationdb.Database;
009import com.foundationdb.FDB;
010import com.foundationdb.FDBException;
011import com.foundationdb.NetworkOptions;
012
013import java.util.concurrent.Executor;
014
015import javax.annotation.PostConstruct;
016import javax.annotation.PreDestroy;
017
018import org.jsimpledb.kv.KVDatabase;
019import org.jsimpledb.kv.KVDatabaseException;
020
021/**
022 * FoundationDB {@link KVDatabase} implementation.
023 *
024 * <p>
025 * Allows specifying a {@linkplain #setKeyPrefix key prefix} for all keys, allowing multiple independent databases.
026 * </p>
027 */
028public class FoundationKVDatabase implements KVDatabase {
029
030    /**
031     * The API version used by this class.
032     */
033    public static final int API_VERSION = 300;
034
035    private final FDB fdb = FDB.selectAPIVersion(API_VERSION);
036    private final NetworkOptions options = this.fdb.options();
037
038    private String clusterFilePath;
039    private byte[] databaseName = new byte[] { (byte)'D', (byte)'B' };
040    private byte[] keyPrefix;
041    private Executor executor;
042
043    private Database database;
044
045    /**
046     * Constructor.
047     *
048     * @throws FDBException if {@link #API_VERSION} is not supported
049     */
050    public FoundationKVDatabase() {
051    }
052
053    /**
054     * Get the {@link NetworkOptions} associated with this instance.
055     * Options must be configured prior to {@link #start}.
056     *
057     * @return network options
058     */
059    public NetworkOptions getNetworkOptions() {
060        return this.options;
061    }
062
063    /**
064     * Configure the {@link Executor} used for the FoundationDB networking event loop.
065     *
066     * <p>
067     * By default, the default thread pool is used to execute the FoundationDB network.
068     * </p>
069     *
070     * @param executor executor for networking activity
071     * @see FDB#startNetwork(Executor) FDB.startNetwork()
072     */
073    public void setExecutor(Executor executor) {
074        this.executor = executor;
075    }
076
077    /**
078     * Configure the cluster file path. Default is null, which results in the default fdb.cluster file being used.
079     *
080     * @param clusterFilePath cluster file pathname
081     */
082    public void setClusterFilePath(String clusterFilePath) {
083        this.clusterFilePath = clusterFilePath;
084    }
085
086    /**
087     * Configure the database name. Currently the default value ({@code "DB".getBytes()}) is the only valid value.
088     *
089     * @param databaseName database name
090     */
091    public void setDatabaseName(byte[] databaseName) {
092        this.databaseName = databaseName;
093    }
094
095    /**
096     * Get the key prefix for all keys.
097     *
098     * @return key prefix, or null if there is none configured
099     */
100    public byte[] getKeyPrefix() {
101        return this.keyPrefix.clone();
102    }
103
104    /**
105     * Configure a prefix for all keys. The prefix will be added/removed automatically with all access.
106     * The default prefix is null, which is equivalent to an empty prefix.
107     *
108     * <p>
109     * The key prefix may not be changed after this instance has {@linkplain #start started}.
110     * </p>
111     *
112     * @param keyPrefix new prefix, or null for none
113     * @throws IllegalArgumentException if {@code keyPrefix} starts with {@code 0xff}
114     * @throws IllegalStateException if this instance has already been {@linkplain #start started}
115     */
116    public void setKeyPrefix(byte[] keyPrefix) {
117        if (this.database != null)
118            throw new IllegalStateException("already started");
119        if (keyPrefix != null && keyPrefix.length > 0 && keyPrefix[0] == (byte)0xff)
120            throw new IllegalArgumentException("prefix starts with 0xff");
121        this.keyPrefix = keyPrefix != null && keyPrefix.length > 0 ? keyPrefix.clone() : null;
122    }
123
124    /**
125     * Get the underlying {@link Database} associated with this instance.
126     *
127     * @return the associated {@link Database}
128     * @throws IllegalStateException if this instance has not yet been {@linkplain #start started}
129     */
130    public Database getDatabase() {
131        if (this.database == null)
132            throw new IllegalStateException("not started");
133        return this.database;
134    }
135
136    /**
137     * Start this instance. This method must be called prior to creating any transactions.
138     *
139     * @throws IllegalStateException if this instance has already been {@linkplain #start started}
140     * @throws IllegalStateException if this instance has already been {@linkplain #stop stopped}
141     */
142    @PostConstruct
143    public synchronized void start() {
144        if (this.database != null)
145            throw new IllegalStateException("already started");
146        this.database = this.fdb.open(this.clusterFilePath, this.databaseName);
147        if (this.executor != null)
148            this.fdb.startNetwork(this.executor);
149        else
150            this.fdb.startNetwork();
151    }
152
153    /**
154     * Stop this instance. Invokes {@link FDB#stopNetwork}. Does nothing if not {@linkplain #start started}.
155     *
156     * @throws FDBException if an error occurs
157     */
158    @PreDestroy
159    public synchronized void stop() {
160        if (this.database == null)
161            return;
162        this.fdb.stopNetwork();
163    }
164
165    /**
166     * Create a new transaction.
167     *
168     * @throws IllegalStateException if this instance has not yet been {@linkplain #start started}
169     * @throws IllegalStateException if this instance has already been {@linkplain #stop stopped}
170     * @throws KVDatabaseException if an unexpected error occurs
171     */
172    @Override
173    public FoundationKVTransaction createTransaction() {
174        if (this.database == null)
175            throw new IllegalStateException("not started");
176        try {
177            return new FoundationKVTransaction(this, this.keyPrefix);
178        } catch (FDBException e) {
179            throw new KVDatabaseException(this, e);
180        }
181    }
182}
