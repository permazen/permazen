014import javax.annotation.PostConstruct;
015import javax.annotation.PreDestroy;

018import org.jsimpledb.kv.mvcc.AtomicKVStore;
019import org.jsimpledb.kv.mvcc.Mutations;

032/**
033 * An {@link AtomicKVStore} view of a RocksDB database.
034 *
035 * <p>
036 * A {@linkplain #setDirectory database directory} is the only required configuration property.
037 * Instances may be stopped and (re)started multiple times.
038 */
039public class RocksDBAtomicKVStore extends ForwardingKVStore implements AtomicKVStore {
040
041    private final Logger log = LoggerFactory.getLogger(this.getClass());
042    private final AtomicBoolean shutdownHookRegistered = new AtomicBoolean();
043
044    // Configuration
045    private Options options;
046    private File directory;
047
048    // "Runtime" state
049    private RocksDBKVStore kv;
050    private RocksDB db;
051
052// Constructors
053
054    /**
055     * Constructor.
056     */
057    public RocksDBAtomicKVStore() {
058        this.setOptions(new Options().setCreateIfMissing(true));
059    }
060
061// Accessors
062
063    /**
064     * Get the filesystem directory containing the database.
065     *
066     * @return database directory
067     */
068    public synchronized File getDirectory() {
069        return this.directory;
070    }
071
072    /**
073     * Configure the filesystem directory containing the database. Required property.
074     *
075     * @param directory database directory
076     * @throws IllegalStateException if this instance is already {@link #start}ed
077     */
078    public synchronized void setDirectory(File directory) {
079        Preconditions.checkState(this.db == null, "already started");
080        this.directory = directory;
081    }
082
083    /**
084     * Get the underlying {@link RocksDB} associated with this instance.
085     *
086     * @return the associated {@link RocksDB}
087     * @throws IllegalStateException if this instance is not {@link #start}ed
088     */
089    public synchronized RocksDB getDB() {
090        Preconditions.checkState(this.db != null, "not started");
091        return this.db;
092    }
093
094// Options
095
096    /**
097     * Get the {@link Options} this instance will use when opening the database at startup.
098     *
099     * @return database options
100     */
101    public synchronized Options getOptions() {
102        return this.options;
103    }
104
105    /**
106     * Set the {@link Options} this instance will use when opening the database at startup.
107     * Overwrites any previous options configuration(s).
108     *
109     * @param options database options
110     * @throws IllegalArgumentException if {@code options} is null
111     * @throws IllegalStateException if this instance is already {@link #start}ed
112     */
113    public synchronized void setOptions(Options options) {
114        Preconditions.checkArgument(options != null, "null options");
115        Preconditions.checkState(this.db == null, "already started");
116        this.options = options;
117        this.options.setMergeOperatorName("uint64add");
118        this.setLogger(this.options);
119    }
120
121    private void setLogger(Options options) {
122        Preconditions.checkArgument(options != null);
123        options.setLogger(new org.rocksdb.Logger(options) {
124            @Override
125            protected void log(InfoLogLevel level, String message) {
126                switch (level) {
127                case DEBUG_LEVEL:
128                    RocksDBAtomicKVStore.this.log.trace("[RocksDB] " + message);
129                    break;
130                case INFO_LEVEL:
131                    RocksDBAtomicKVStore.this.log.info("[RocksDB] " + message);
132                    break;
133                case WARN_LEVEL:
134                    RocksDBAtomicKVStore.this.log.warn("[RocksDB] " + message);
135                    break;
136                case ERROR_LEVEL:
137                case FATAL_LEVEL:
138                default:
139                    RocksDBAtomicKVStore.this.log.error("[RocksDB] " + message);
140                    break;
141                }
142            }
143        });
144    }
145
146// Lifecycle
147
148    @Override
149    @PostConstruct
150    public synchronized void start() {
151
152        // Already started?
153        if (this.db != null)
154            return;
155        this.log.info("starting " + this);
156
157        // Check configuration
158        Preconditions.checkState(this.directory != null, "no directory configured");
159
160        // Create directory if needed
161        if (!this.directory.exists()) {
162            if (!this.options.createIfMissing())
163                throw new RuntimeException("directory `" + this.directory + "' does not exist");
164            if (!this.directory.mkdirs())
165                throw new RuntimeException("failed to create directory `" + this.directory + "'");
166        }
167        if (!this.directory.isDirectory())
168            throw new RuntimeException("file `" + this.directory + "' is not a directory");
169
170        // Open database
171        if (this.log.isDebugEnabled())
172            this.log.debug("opening " + this + " RocksDB database");
173        try {
174            this.db = RocksDB.open(this.options, this.directory.toString());
175        } catch (RocksDBException e) {
176            throw new RuntimeException("RocksDB database startup failed", e);
177        }
178
179        // Create k/v store view
180        this.kv = new RocksDBKVStore(this.db);
181
182        // Add shutdown hook so we don't leak native resources
183        if (this.shutdownHookRegistered.compareAndSet(false, true)) {
184            Runtime.getRuntime().addShutdownHook(new Thread() {
185                public void run() {
186                    RocksDBAtomicKVStore.this.stop();
187                }
188            });
189        }
190    }
191
192    @Override
193    @PreDestroy
194    public synchronized void stop() {
195
196        // Check state
197        if (this.db == null)
198            return;
199        this.log.info("stopping " + this);
200
201        // Close k/v store view
202        this.kv.close();
203        this.kv = null;
204
205        // Shut down RocksDB database
206        try {
207            if (this.log.isDebugEnabled())
208                this.log.info("closing " + this + " RocksDB database");
209            this.db.close();
210        } catch (Throwable e) {
211            this.log.error("caught exception closing database during shutdown (ignoring)", e);
212        }
213        this.db = null;
214    }
215
216// ForwardingKVStore
217
218    @Override
219    protected RocksDBKVStore delegate() {
220        Preconditions.checkState(this.db != null && this.kv != null, "closed");
221        return this.kv;
222    }
223
224// AtomicKVStore
225
226    @Override
227    public synchronized SnapshotRocksDBKVStore snapshot() {
228        Preconditions.checkState(this.db != null, "closed");
229        return new SnapshotRocksDBKVStore(this.db);
230    }
231
232    @Override
233    public synchronized void mutate(Mutations mutations, boolean sync) {
234        Preconditions.checkArgument(mutations != null, "null mutations");
235        Preconditions.checkState(this.db != null, "closed");
236
237        // Apply mutations in a batch
238        final WriteBatch batch = new WriteBatch();
239        try {
240
241            // Apply removes
242            final ReadOptions iteratorOptions = new ReadOptions().setFillCache(false);
243            try {
244                for (KeyRange range : mutations.getRemoveRanges()) {
245                    final byte[] min = range.getMin();
246                    final byte[] max = range.getMax();
247                    if (min != null && max != null && ByteUtil.isConsecutive(min, max))
248                        batch.remove(min);
249                    else {
250                        try (RocksDBKVStore.Iterator i = this.kv.createIterator(iteratorOptions, min, max, false)) {
251                            while (i.hasNext())
252                                batch.remove(i.next().getKey());
253                        }
254                    }
255                }
256            } finally {
257                iteratorOptions.dispose();
258            }
259
260            // Apply puts
261            for (Map.Entry<byte[], byte[]> entry : mutations.getPutPairs())
262                batch.put(entry.getKey(), entry.getValue());
263
264            // Apply counter adjustments
265            for (Map.Entry<byte[], Long> entry : mutations.getAdjustPairs())
266                batch.merge(entry.getKey(), this.kv.encodeCounter(entry.getValue()));
267
268            // Write the batch
269            final WriteOptions writeOptions = new WriteOptions().setSync(sync);
270            try {
271                this.db.write(writeOptions, batch);
272            } finally {
273                writeOptions.dispose();
274            }
275        } catch (RocksDBException e) {
276            throw new RuntimeException("error applying changes to RocksDB", e);
277        } finally {
278            batch.dispose();
279        }
280    }
281
282// Object
283
284    /**
285     * Finalize this instance. Invokes {@link #stop} to close any unclosed iterators.
286     */
287    @Override
288    protected void finalize() throws Throwable {
289        try {
290            if (this.db != null)
291               this.log.warn(this + " leaked without invoking stop()");
292            this.stop();
293        } finally {
294            super.finalize();
295        }
296    }
297
298    @Override
299    public String toString() {
300        return this.getClass().getSimpleName()
301          + "[dir=" + this.directory
302          + ",kv=" + this.kv
303          + "]";
304    }
305}
