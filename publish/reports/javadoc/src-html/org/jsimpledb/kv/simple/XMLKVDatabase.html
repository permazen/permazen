006package org.jsimpledb.kv.simple;
007
008import java.io.BufferedInputStream;
009import java.io.File;
010import java.io.FileInputStream;
011import java.io.FileNotFoundException;
012import java.io.IOException;
013import java.io.InputStream;
014import java.io.OutputStream;
015
016import javax.xml.stream.XMLStreamException;
017
018import org.dellroad.stuff.io.AtomicUpdateFileOutputStream;
019import org.dellroad.stuff.io.FileStreamRepository;
020import org.dellroad.stuff.io.StreamRepository;
021import org.jsimpledb.kv.KVDatabaseException;
022import org.jsimpledb.kv.RetryTransactionException;
023import org.jsimpledb.kv.util.XMLSerializer;
024
025/**
026 * Simple persistent {@link org.jsimpledb.kv.KVDatabase} backed by an XML file stored in a {@link StreamRepository}.
027 * The data is kept in memory, and the XML file is rewritten in its entirety after each successful commit.
028 * In normal usage, the XML file is stored in a regular {@link File} using a {@link FileStreamRepository}, which
029 * guarantees (via the use of {@link AtomicUpdateFileOutputStream}) that a partially written XML file can never exist.
030 *
031 * <p>
032 * If a {@link FileNotFoundException} is caught when trying to read the XML file, we assume that the underlying file has
033 * not yet been created and the database will initially be empty. Alternately, you can configure a file containing
034 * default initial content via {@link #setInitialContentFile setInitialContentFile()}, or override {@link #getInitialContent}
035 * to create the initial content more dynamically.
036 * </p>
037 *
038 * <p>
039 * When a {@link FileStreamRepository} is used, instances support "out-of-band" updates of the XML file. In that case,
040 * each time a transaction is accessed the modification timestamp of the XML file is examined. If the XML file has been
041 * updated by some external process since the time the transaction was created, the database will be reloaded from
042 * the XML file and the transaction will fail with a {@link RetryTransactionException}.
043 * </p>
044 *
045 * <p>
046 * Note that two different processes modifying the XML file at the same time is not without race conditions: e.g., it's possible
047 * for an external process to update the XML file just as a transaction associated with this instance is being committed
048 * and written to the file, which will result in overwriting the external process' changes.
049 * </p>
050 *
051 * @see XMLSerializer
052 * @see AtomicUpdateFileOutputStream
053 * @see org.jsimpledb.spring.SpringXMLKVDatabase
054 */
055public class XMLKVDatabase extends SimpleKVDatabase {
056
057    private final StreamRepository repository;
058    private final XMLSerializer serializer;
059    private final File file;
060
061    private int generation;
062    private long timestamp;
063    private File initialContentFile;
064
065// Constructors
066
067    /**
068     * Normal constructor. Uses a {@link FileStreamRepository} backed by the specified file.
069     *
070     * @param file persistent XML file
071     * @throws IllegalArgumentException if {@code file} is null
072     */
073    public XMLKVDatabase(File file) {
074        this(new FileStreamRepository(file));
075    }
076
077    /**
078     * Constructor allowing storage in any user-supplied {@link StreamRepository}.
079     *
080     * @param repository XML file storage
081     * @throws IllegalArgumentException if {@code file} is null
082     */
083    public XMLKVDatabase(StreamRepository repository) {
084        if (repository == null)
085            throw new IllegalArgumentException("null repository");
086        this.repository = repository;
087        this.serializer = new XMLSerializer(this.kv);
088        this.file = repository instanceof FileStreamRepository ? ((FileStreamRepository)repository).getFile() : null;
089        this.reload();
090    }
091
092    /**
093     * Get the initial content for an uninitialized database. This method is invoked when, on the first load,
094     * the backing XML file is not found. It should return a stream that reads initial content for the database,
095     * if any, otherwise null.
096     *
097     * <p>
098     * The implementation in {@link XMLKVDatabase} opens and returns the {@link File} configured by
099     * {@link #setInitialContentFile setInitialContentFile()}, if any, otherwise null.
100     * </p>
101     *
102     * @return default initial XML database content, or null for none
103     * @throws IOException if an error occurs accessing the initial content file
104     */
105    protected InputStream getInitialContent() throws IOException {
106        return this.initialContentFile != null ? new FileInputStream(this.initialContentFile) : null;
107    }
108
109    /**
110     * Configure the {@link File} containing default initial content for an uninitialized database. This method is invoked
111     * by {@link #getInitialContent} when, on the first load, the backing XML file is not found.
112     *
113     * @param initialContentFile file containing default initial XML database content, or null for none
114     */
115    public void setInitialContentFile(File initialContentFile) {
116        this.initialContentFile = initialContentFile;
117    }
118
119    @Override
120    public synchronized XMLKVTransaction createTransaction() {
121        this.checkForOutOfBandUpdate();
122        return new XMLKVTransaction(this, this.getWaitTimeout(), this.generation);
123    }
124
125    /**
126     * Forcibly reload this database by re-reading the XML file.
127     *
128     * <p>
129     * Any transactions that are in-progress when this method is called immediately become unusable.
130     * </p>
131     */
132    public synchronized void reload() {
133        this.readXML();
134    }
135
136    /**
137     * Get the generation number associated with the XML file.
138     * The generation number is incremented every time the database is wholesale updated by reading the file into memory,
139     * e.g., by invoking {@link #reload}.
140     *
141     * @return XML file generation number
142     * @see XMLKVTransaction#getGeneration
143     */
144    public synchronized int getGeneration() {
145        return this.generation;
146    }
147
148    /**
149     * Check the XML file's timestamp and reload it if it has been modified since the most recent
150     * read or write by this instance.
151     *
152     * @return true if file was updated and re-read, otherwise false
153     */
154    public synchronized boolean checkForOutOfBandUpdate() {
155        if (this.file == null)
156            return false;
157        final long fileTime = this.file.lastModified();
158        if (fileTime == 0)
159            return false;
160        if (this.timestamp != 0) {
161            if (fileTime <= this.timestamp)
162                return false;
163            this.log.info("detected out-of-band update of XMLKVDatabase file `" + this.file + "'; reloading");
164        }
165        this.readXML();
166        return true;
167    }
168
169    @Override
170    protected void checkState(SimpleKVTransaction tx) {
171        this.checkForOutOfBandUpdate();
172        final int txGeneration = ((XMLKVTransaction)tx).getGeneration();
173        if (txGeneration != this.generation) {
174            throw new RetryTransactionException(tx, "XML file changed since transaction started (generation number changed from "
175              + txGeneration + " to " + this.generation + ")");
176        }
177    }
178
179    @Override
180    protected void postCommit(SimpleKVTransaction tx, boolean successful) {
181
182        // If something weird happened, reload from storage
183        if (!successful) {
184            this.readXML();
185            return;
186        }
187
188        // Persist data to file
189        this.writeXML();
190    }
191
192    protected synchronized void readXML() {
193
194        // Clear all existing keys
195        this.kv.removeRange(null, null);
196
197        // Snapshot file's current modification timestamp
198        final long newTimestamp = this.file != null ? this.file.lastModified() : 0;
199
200        // Open file input
201        InputStream input;
202        try {
203            input = this.repository.getInputStream();
204        } catch (FileNotFoundException e) {
205
206            // If this is not the first load, file must have mysteriously disappeared
207            if (this.generation != 0)
208                throw new KVDatabaseException(this, "error reading XML content: file not longer available", e);
209
210            // Get default initial content instead, if any
211            try {
212                input = this.getInitialContent();
213            } catch (IOException e2) {
214                throw new KVDatabaseException(this, "error opening initial XML content", e2);
215            }
216            final String desc = this.file != null ? "file `" + this.file + "'" : "database file";
217            if (input == null)
218                this.log.info(desc + " not found and no initial content is configured; creating new, empty database");
219            else
220                this.log.info(desc + " not found; applying default initial content");
221        } catch (IOException e) {
222            throw new KVDatabaseException(this, "error opening XML content", e);
223        }
224
225        // Read XML
226        if (input != null) {
227            try {
228                this.serializer.read(new BufferedInputStream(input));
229            } catch (XMLStreamException e) {
230                throw new KVDatabaseException(this, "error reading XML content", e);
231            } finally {
232                try {
233                    input.close();
234                } catch (IOException e) {
235                    // ignore
236                }
237            }
238        }
239
240        // Update timestamp and generation number
241        if (newTimestamp != 0)
242            this.timestamp = newTimestamp;
243        this.generation++;
244    }
245
246    protected synchronized void writeXML() {
247        boolean successful = false;
248        try {
249            final OutputStream output = this.repository.getOutputStream();
250            try {
251                this.serializer.write(output, true);
252                if (output instanceof AtomicUpdateFileOutputStream)
253                    ((AtomicUpdateFileOutputStream)output).getFD().sync();
254                output.close();
255                if (this.file != null)
256                    this.timestamp = this.file.lastModified();
257                successful = true;
258            } finally {
259                if (!successful && output instanceof AtomicUpdateFileOutputStream)
260                    ((AtomicUpdateFileOutputStream)output).cancel();
261            }
262        } catch (IOException e) {
263            throw new KVDatabaseException(this, "error writing XML content", e);
264        } catch (XMLStreamException e) {
265            throw new KVDatabaseException(this, "error writing XML content", e);
266        }
267    }
268}
