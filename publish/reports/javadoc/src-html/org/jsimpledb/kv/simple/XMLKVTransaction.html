006package org.jsimpledb.kv.simple;
007
008/**
009 * Transaction associated with a {@link XMLKVDatabase}.
010 */
011public class XMLKVTransaction extends SimpleKVTransaction {
012
013    private final int generation;
014
015    XMLKVTransaction(XMLKVDatabase database, long waitTimeout, int generation) {
016        super(database, waitTimeout);
017        this.generation = generation;
018    }
019
020    /**
021     * Get the generation number associated with this instance.
022     *
023     * @return the generation number on which this transaction is based
024     * @see XMLKVDatabase#getGeneration
025     */
026    public int getGeneration() {
027        return this.generation;
028    }
029}
