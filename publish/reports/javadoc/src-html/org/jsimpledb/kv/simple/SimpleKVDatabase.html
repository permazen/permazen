025/**
026 * Simple implementation of the {@link KVDatabase} interface that provides a concurrent, transactional view
027 * of an underlying {@link KVStore} with strong ACID semantics (<b>D</b>urability must be provided by the {@link KVStore}).
028 *
029 * <p>
030 * The ACID semantics, as well as {@linkplain #getWaitTimeout wait timeouts} and {@linkplain #getHoldTimeout hold timeouts},
031 * are provided by the {@link LockManager} class. If the wait timeout is exceeded, a {@link RetryTransactionException}
032 * is thrown. If the hold timeout is exceeded, a {@link TransactionTimeoutException} is thrown.
033 * </p>
034 *
035 * <p>
036 * Instances wrap an underlying {@link KVStore} which provides persistence and from which committed data is read and written.
037 * During a transaction, all mutations are recorded internally; if/when the transaction is committed, those mutations are
038 * applied to the underlying {@link KVStore} all at once, and this operation is bracketed by calls to
039 * {@link #preCommit preCommit()} and {@link #postCommit postCommit()}.
040 * </p>
041 *
042 * @see LockManager
043 */
044public class SimpleKVDatabase implements KVDatabase {
045

215     * Verify that the given transaction is still usable.
216     *
217     * <p>
218     * This method is invoked at the start of the {@link KVStore} data access and {@link SimpleKVTransaction#commit commit()}

221     * Note: transaction state is also protected by this instance's lock.
222     * </p>

226     * </p>
227     *
228     * @param tx the transaction being accessed
229     * @throws StaleTransactionException if this instance is no longer usable

231     * @throws TransactionTimeoutException if the transaction has timed out

236    private void checkUsable(SimpleKVTransaction tx) {
237        if (tx.stale)
238            throw new StaleTransactionException(tx);
239        if (this.lockManager.checkHoldTimeout(tx.lockOwner) == -1) {
240            this.rollback(tx);
241            throw new TransactionTimeoutException(tx,
242              "transaction taking too long: hold timeout of " + this.lockManager.getHoldTimeout() + "ms has expired");
243        }
244    }

248    synchronized byte[] get(SimpleKVTransaction tx, byte[] key) {
249
250        // Sanity check
251        if (key.length > 0 && key[0] == (byte)0xff)
252            throw new IllegalArgumentException("key starts with 0xff");
253        this.checkUsable(tx);
254        this.checkState(tx);
255
256        // Check transaction mutations
257        final Mutation mutation = tx.findMutation(key);
258        if (mutation != null)
259            return mutation instanceof Put ? ((Put)mutation).getValue() : null;
260
261        // Read from underlying store
262        this.getLock(tx, key, ByteUtil.getNextKey(key), false);
263        return this.kv.get(key);
264    }

266    synchronized KVPair getAtLeast(SimpleKVTransaction tx, byte[] minKey) {
267
268        // Realize minKey
269        if (minKey == null)
270            minKey = ByteUtil.EMPTY;
271
272        // Sanity check
273        this.checkUsable(tx);
274        this.checkState(tx);
275
276        // Save original min key for locking purposes
277        final byte[] originalMinKey = minKey;
278
279        // Look for a mutation starting before minKey but containing it
280        if (minKey.length > 0) {
281            final Mutation overlap = tx.findMutation(minKey);
282            if (overlap != null) {
283                if (overlap instanceof Put) {
284                    final Put put = (Put)overlap;
285                    assert Arrays.equals(put.getKey(), minKey);
286                    return new KVPair(put.getKey(), put.getValue());
287                }
288                assert overlap instanceof Del;
289                final byte[] max = overlap.getMax();
290                if (max == null)
291                    return null;
292                minKey = max;
293            }
294        }
295
296        // Find whichever is first: a transaction Put, or an underlying store entry not covered by a transaction Delete
297        SortedSet<Mutation> mutations = tx.mutations;
298        while (true) {
299
300            // Get the next mutation and kvstore entry >= minKey (if they exist)
301            if (minKey != null)
302                mutations = mutations.tailSet(Mutation.key(minKey));
303            final Mutation mutation = !mutations.isEmpty() ? mutations.first() : null;
304            final KVPair entry = this.kv.getAtLeast(minKey);
305
306            // Handle the case where neither is found
307            if (mutation == null && entry == null) {
308                this.getLock(tx, originalMinKey, null, false);
309                return null;
310            }
311
312            // Check for whether mutation or kvstore wins (i.e., which is first)
313            if (mutation != null && (entry == null || mutation.compareTo(entry.getKey()) <= 0)) {
314                if (mutation instanceof Del) {
315                    if ((minKey = mutation.getMax()) == null) {
316                        this.getLock(tx, originalMinKey, null, false);
317                        return null;
318                    }
319                    continue;
320                }
321                final Put put = (Put)mutation;
322                this.getLock(tx, originalMinKey, ByteUtil.getNextKey(put.getKey()), false);
323                return new KVPair(put.getKey(), put.getValue());
324            } else {
325                this.getLock(tx, originalMinKey, ByteUtil.getNextKey(entry.getKey()), false);
326                return entry;
327            }
328        }
329    }

331    synchronized KVPair getAtMost(SimpleKVTransaction tx, byte[] maxKey) {
332
333        // Sanity check
334        this.checkUsable(tx);
335        this.checkState(tx);
336
337        // Save original max key for locking purposes
338        final byte[] originalMaxKey = maxKey;
339
340        // Find whichever is first: a transaction addition, or an underlying store entry not covered by a transaction deletion
341        SortedSet<Mutation> mutations = tx.mutations;
342        while (true) {
343
344            // Get the next mutation and kvstore entry < maxKey (if they exist)
345            if (maxKey != null)
346                mutations = mutations.headSet(Mutation.key(maxKey));
347            final Mutation mutation = !mutations.isEmpty() ? mutations.last() : null;
348            final KVPair entry = this.kv.getAtMost(maxKey);
349
350            // Handle the case where neither is found
351            if (mutation == null && entry == null) {
352                this.getLock(tx, ByteUtil.EMPTY, originalMaxKey, false);
353                return null;
354            }
355
356            // Check for whether mutation or kvstore wins (i.e., which is first)
357            if (mutation != null && (entry == null || mutation.compareTo(entry.getKey()) >= 0)) {
358                if (mutation instanceof Del) {
359                    if ((maxKey = mutation.getMin()) == null) {
360                        this.getLock(tx, ByteUtil.EMPTY, originalMaxKey, false);
361                        return null;
362                    }
363                    continue;
364                }
365                final Put put = (Put)mutation;
366                this.getLock(tx, put.getKey(), originalMaxKey, false);
367                return new KVPair(put.getKey(), put.getValue());
368            } else {
369                this.getLock(tx, entry.getKey(), originalMaxKey, false);
370                return entry;
371            }
372        }
373    }

375    synchronized void put(SimpleKVTransaction tx, byte[] key, byte[] value) {
376
377        // Sanity check
378        if (value == null)
379            throw new NullPointerException();
380        if (key.length > 0 && key[0] == (byte)0xff)
381            throw new IllegalArgumentException("key starts with 0xff");
382        this.checkUsable(tx);
383        this.checkState(tx);
384        final byte[] keyNext = ByteUtil.getNextKey(key);
385
386        // Check transaction mutations
387        final Mutation mutation = tx.findMutation(key);
388        if (mutation instanceof Put) {
389            assert Arrays.equals(((Put)mutation).getKey(), key);
390
391            // Replace Put with new Put
392            tx.mutations.remove(mutation);
393            tx.mutations.add(new Put(key, value));
394        } else if (mutation instanceof Del) {
395
396            // Split [Del] -> [Del*, Put, Del*]  *if needed
397            final Del del = (Del)mutation;
398            final byte[] delMin = del.getMin();
399            final byte[] delMax = del.getMax();
400            tx.mutations.remove(del);
401            if (KeyRange.compare(delMin, key) < 0)
402                tx.mutations.add(new Del(delMin, key));
403            if (KeyRange.compare(keyNext, delMax) < 0)
404                tx.mutations.add(new Del(keyNext, delMax));
405            tx.mutations.add(new Put(key, value));
406        } else {
407
408            // Add write lock and new tx mutation
409            this.getLock(tx, key, keyNext, true);
410            tx.mutations.add(new Put(key, value));
411        }
412    }

414    synchronized void remove(SimpleKVTransaction tx, byte[] key) {
415
416        // Sanity check
417        if (key.length > 0 && key[0] == (byte)0xff)
418            throw new IllegalArgumentException("key starts with 0xff");
419        this.checkUsable(tx);
420        this.checkState(tx);
421        final byte[] keyNext = ByteUtil.getNextKey(key);
422
423        // Check transaction mutations
424        final Mutation mutation = tx.findMutation(key);
425        if (mutation instanceof Put) {
426            assert Arrays.equals(((Put)mutation).getKey(), key);
427
428            // Replace Put with Del
429            tx.mutations.remove(mutation);
430            tx.mutations.add(new Del(key));
431        } else if (mutation == null) {
432
433            // Add write lock and new tx mutation
434            this.getLock(tx, key, keyNext, true);
435            tx.mutations.add(new Del(key));
436        }
437    }

439    synchronized void removeRange(SimpleKVTransaction tx, byte[] minKey, byte[] maxKey) {
440
441        // Realize minKey
442        if (minKey == null)
443            minKey = ByteUtil.EMPTY;
444
445        // Sanity check
446        int diff = KeyRange.compare(minKey, maxKey);
447        if (diff > 0)
448            throw new IllegalArgumentException("minKey > maxKey");
449        this.checkUsable(tx);
450        this.checkState(tx);
451        if (diff == 0)                                                          // range is empty
452            return;
453        final byte[] originalMinKey = minKey;
454        final byte[] originalMaxKey = maxKey;
455
456        // Deal with partial overlap at the left end of the range
457        if (minKey.length > 0) {
458            final Mutation leftMutation = tx.findMutation(minKey);
459            if (leftMutation instanceof Put) {
460                assert Arrays.equals(((Put)leftMutation).getKey(), minKey);
461                tx.mutations.remove(leftMutation);                                          // overwritten by this change
462            } else if (leftMutation instanceof Del) {
463                final Del del = (Del)leftMutation;
464                tx.mutations.remove(del);                                                   // will merge into this change
465                minKey = del.getMin();                                                      // guaranteed to be <= minKey
466                if (KeyRange.compare(del.getMax(), maxKey) > 0)                             // get higher of the two maxKeys
467                    maxKey = del.getMax();
468            }
469        }
470
471        // Deal with partial overlap at the right end of the range
472        if (maxKey != null) {
473            Mutation rightMutation = null;
474            try {
475                rightMutation = minKey != null ?
476                  tx.mutations.subSet(Mutation.key(minKey), Mutation.key(maxKey)).last() :
477                  tx.mutations.headSet(Mutation.key(maxKey)).last();
478            } catch (NoSuchElementException e) {
479                // ignore
480            }
481            if (rightMutation instanceof Put)
482                tx.mutations.remove(rightMutation);                                         // overwritten by this change
483            else if (rightMutation instanceof Del) {
484                final Del del = (Del)rightMutation;
485                tx.mutations.remove(del);                                                   // will merge into this change
486                if (KeyRange.compare(del.getMax(), maxKey) > 0)                             // get higher of the two maxKeys
487                    maxKey = del.getMax();
488            }
489        }
490
491        // Remove all mutations in the middle
492        if (originalMinKey == null && originalMaxKey == null)
493            tx.mutations.clear();
494        else if (originalMinKey == null)
495            tx.mutations.headSet(Mutation.key(originalMaxKey)).clear();
496        else if (originalMaxKey == null)
497            tx.mutations.tailSet(Mutation.key(originalMinKey)).clear();
498        else
499            tx.mutations.subSet(Mutation.key(originalMinKey), Mutation.key(originalMaxKey)).clear();
500
501        // Add write lock and new tx mutation
502        this.getLock(tx, minKey, maxKey, true);
503        tx.mutations.add(new Del(minKey, maxKey));
504    }

506    synchronized void commit(SimpleKVTransaction tx) {
507
508        // Prevent use after commit() or rollback() invoked
509        if (tx.stale)
510            throw new StaleTransactionException(tx);
511        tx.stale = true;
512
513        // Release all locks
514        if (!this.lockManager.release(tx.lockOwner)) {
515            throw new TransactionTimeoutException(tx,
516              "transaction taking too long: hold timeout of " + this.lockManager.getHoldTimeout() + "ms has expired");
517        }
518
519        // Check subclass state
520        this.checkState(tx);
521
522        // If there are no mutations, there's no need to write anything
523        if (tx.mutations.isEmpty())
524            return;
525
526        // Commit mutations
527        this.preCommit(tx);
528        boolean successful = false;
529        try {
530            for (Mutation mutation : tx.mutations)
531                mutation.apply(this.kv);
532            successful = true;
533        } finally {
534            tx.mutations.clear();
535            this.postCommit(tx, successful);
536        }
537    }
538
539    synchronized void rollback(SimpleKVTransaction tx) {
540
541        // Prevent use after commit() or rollback() invoked
542        if (tx.stale)
543            return;
544        tx.stale = true;
545
546        // Release all locks
547        this.lockManager.release(tx.lockOwner);
548    }

552    private /*synchronized*/ void getLock(SimpleKVTransaction tx, byte[] minKey, byte[] maxKey, boolean write) {
553
554        // Attempt to get the lock
555        LockManager.LockResult lockResult;
556        try {
557            lockResult = this.lockManager.lock(tx.lockOwner, minKey, maxKey, write, tx.waitTimeout);
558        } catch (InterruptedException e) {
559            this.rollback(tx);
560            Thread.currentThread().interrupt();
561            throw new RetryTransactionException(tx, "transaction interrupted while waiting to acquire lock", e);
562        }
563
564        // Check result
565        switch (lockResult) {
566        case SUCCESS:
567            break;
568        case WAIT_TIMEOUT_EXPIRED:
569            this.rollback(tx);
570            throw new RetryTransactionException(tx, "could not acquire lock after " + tx.waitTimeout + "ms");
571        case HOLD_TIMEOUT_EXPIRED:
572            this.rollback(tx);
573            throw new TransactionTimeoutException(tx,
574              "transaction taking too long: hold timeout of " + this.lockManager.getHoldTimeout() + "ms has expired");
575        default:
576            throw new RuntimeException("internal error");
577        }
578    }
