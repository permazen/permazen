006package org.jsimpledb.spring;
007
008import java.io.IOException;
009
010import javax.servlet.FilterChain;
011import javax.servlet.ServletException;
012import javax.servlet.http.HttpServletRequest;
013import javax.servlet.http.HttpServletResponse;
014
015import org.jsimpledb.JSimpleDB;
016import org.jsimpledb.JTransaction;
017import org.jsimpledb.ValidationMode;
018import org.springframework.transaction.interceptor.DefaultTransactionAttribute;
019import org.springframework.transaction.interceptor.TransactionAttribute;
020import org.springframework.web.context.WebApplicationContext;
021import org.springframework.web.context.support.WebApplicationContextUtils;
022import org.springframework.web.filter.OncePerRequestFilter;
023
024/**
025 * A servlet {@link javax.servlet.Filter} binds a {@link JSimpleDB} {@link JTransaction} to the current thread
026 * for the entire processing of the request. Intended for the "Open Session in View" pattern, i.e. to allow
027 * for lazy loading in web views despite the original transactions already being completed.
028 *
029 * <p>
030 * This filter makes {@link JTransaction}s available via the current thread, which will be autodetected by
031 * a {@link JSimpleDBTransactionManager}.
032 * </p>
033 *
034 * <p>
035 * Looks up the {@link JSimpleDB} in Spring's root web application context. Supports a {@code "JSimpleDBBeanName"}
036 * filter init-param in web.xml; the default bean name is {@code "jsimpledb"}. Also supports setting the following
037 * filter init-params:
038 * <ul>
039 *  <li>{@code "transactionAttributes"} - configures {@link TransactionAttribute}s; value should be
040 *      a string compatible with {@link org.springframework.transaction.interceptor.TransactionAttributeEditor}.</li>
041 *  <li>{@code "allowNewSchema"} - whether creation of a new schema version in the database is allowed.
042 *      (see {@link JSimpleDB#createTransaction JSimpleDB.createTransaction()}). Default false.</li>
043 *  <li>{@code "validationMode"} - validation mode for the transaction
044 *      (see {@link JSimpleDB#createTransaction JSimpleDB.createTransaction()}). Default {@link ValidationMode#AUTOMATIC}.</li>
045 * </ul>
046 */
047public class OpenTransactionInViewFilter extends OncePerRequestFilter {
048
049    /**
050     * The default name of the {@link JSimpleDB} bean: <code>{@value}</code>.
051     *
052     * @see #JSIMPLEDB_BEAN_NAME_PARAMETER
053     */
054    public static final String DEFAULT_JSIMPLEDB_BEAN_NAME = "jsimpledb";
055
056    /**
057     * Filter init parameter that specifies the name of the {@link JSimpleDB} bean: <code>{@value}</code>.
058     *
059     * @see #DEFAULT_JSIMPLEDB_BEAN_NAME
060     */
061    public static final String JSIMPLEDB_BEAN_NAME_PARAMETER = "JSimpleDBBeanName";
062
063    /**
064     * Filter init parameter that specifies transaction attributes: <code>{@value}</code>.
065     */
066    public static final String JSIMPLEDB_TRANSACTION_ATTRIBUTE_PARAMETER = "transactionAttributes";
067
068    /**
069     * Filter init parameter that specifies whether creation of a new schema version in the database is allowed.
070     * Default false.
071     */
072    public static final String JSIMPLEDB_ALLOW_NEW_SCHEMA_PARAMETER = "allowNewSchema";
073
074    /**
075     * Filter init parameter that specifies the validation mode for the transaction.
076     * Default {@link ValidationMode#AUTOMATIC}.
077     */
078    public static final String JSIMPLEDB_VALIDATION_MODE_PARAMETER = "validationMode";
079
080    private String jsimpledbBeanName = DEFAULT_JSIMPLEDB_BEAN_NAME;
081    private TransactionAttribute transactionAttributes;
082    private boolean allowNewSchema;
083    private ValidationMode validationMode = ValidationMode.AUTOMATIC;
084
085    private volatile JSimpleDB jdb;
086
087    /**
088     * Get the name of the {@link JSimpleDB} bean to find in the Spring root application context.
089     *
090     * @return bean name
091     * @see #JSIMPLEDB_BEAN_NAME_PARAMETER
092     */
093    public String getJSimpleDBBeanName() {
094        return this.jsimpledbBeanName;
095    }
096
097    /**
098     * Set the name of the {@link JSimpleDB} bean to find in the Spring root application context.
099     * Default is {@link #DEFAULT_JSIMPLEDB_BEAN_NAME}.
100     *
101     * @param jsimpledbBeanName {@link JSimpleDB} bean name
102     * @see #JSIMPLEDB_BEAN_NAME_PARAMETER
103     */
104    public void setJSimpleDBBeanName(String jsimpledbBeanName) {
105        this.jsimpledbBeanName = jsimpledbBeanName;
106    }
107
108    /**
109     * Get the transaction attributes.
110     *
111     * @return transaction attributes
112     * @see #JSIMPLEDB_TRANSACTION_ATTRIBUTE_PARAMETER
113     */
114    public TransactionAttribute getTransactionAttributes() {
115        return this.transactionAttributes;
116    }
117
118    /**
119     * Set the transaction attributes.
120     *
121     * @param transactionAttributes transaction attributes
122     * @see #JSIMPLEDB_TRANSACTION_ATTRIBUTE_PARAMETER
123     */
124    public void setTransactionAttributes(TransactionAttribute transactionAttributes) {
125        this.transactionAttributes = transactionAttributes;
126    }
127
128    /**
129     * Get whether new scheme creation is allowed.
130     *
131     * @return whether to allow recording new schema versions
132     * @see #JSIMPLEDB_ALLOW_NEW_SCHEMA_PARAMETER
133     */
134    public boolean isAllowNewSchema() {
135        return this.allowNewSchema;
136    }
137
138    /**
139     * Set whether new scheme creation is allowed.
140     *
141     * @param allowNewSchema whether to allow recording new schema versions
142     * @see #JSIMPLEDB_ALLOW_NEW_SCHEMA_PARAMETER
143     */
144    public void setAllowNewSchema(boolean allowNewSchema) {
145        this.allowNewSchema = allowNewSchema;
146    }
147
148    /**
149     * Get transaction validation mode.
150     *
151     * @return validation mode for transactions
152     * @see #JSIMPLEDB_VALIDATION_MODE_PARAMETER
153     */
154    public ValidationMode getValidationMode() {
155        return this.validationMode;
156    }
157
158    /**
159     * Set transaction validation mode.
160     *
161     * @param validationMode validation mode for transactions
162     * @see #JSIMPLEDB_VALIDATION_MODE_PARAMETER
163     */
164    public void setValidationMode(ValidationMode validationMode) {
165        if (validationMode == null)
166            validationMode = ValidationMode.AUTOMATIC;
167        this.validationMode = validationMode;
168    }
169
170    /**
171     * Look up the {@link JSimpleDB} that this filter should use.
172     *
173     * @return the JSimpleDB to use
174     * @see #getJSimpleDBBeanName
175     */
176    protected JSimpleDB lookupJSimpleDB() {
177        if (this.jdb == null) {
178            final WebApplicationContext wac = WebApplicationContextUtils.getRequiredWebApplicationContext(this.getServletContext());
179            this.jdb = wac.getBean(this.getJSimpleDBBeanName(), JSimpleDB.class);
180        }
181        return this.jdb;
182    }
183
184    @Override
185    protected void doFilterInternal(final HttpServletRequest request, final HttpServletResponse response,
186      final FilterChain filterChain) throws ServletException, IOException {
187
188        // Sanity check
189        try {
190            JTransaction.getCurrent();
191            throw new IllegalStateException("a JTransaction is already associated with the current thread");
192        } catch (IllegalStateException e) {
193            // expected
194        }
195
196        // Get transaction attributes
197        TransactionAttribute attr = this.transactionAttributes;
198        if (attr == null)
199            attr = new DefaultTransactionAttribute();
200
201        // Create transaction
202        final JTransaction jtx = this.lookupJSimpleDB().createTransaction(this.allowNewSchema, this.validationMode);
203
204        // Configure it
205        if (attr.isReadOnly())
206            jtx.getTransaction().setReadOnly(true);
207        final int timeout = attr.getTimeout();
208        if (timeout != TransactionAttribute.TIMEOUT_DEFAULT) {
209            try {
210                jtx.getTransaction().setTimeout(timeout * 1000L);
211            } catch (UnsupportedOperationException e) {
212                // ignore
213            }
214        }
215
216        // Proceed with transaction
217        boolean success = false;
218        try {
219            JTransaction.setCurrent(jtx);
220            filterChain.doFilter(request, response);
221            success = true;
222        } catch (Throwable t) {
223            if (attr.rollbackOn(t) || jtx.getTransaction().isRollbackOnly())
224                jtx.rollback();
225            else
226                jtx.commit();
227            if (t instanceof Error)
228                throw (Error)t;
229            if (t instanceof RuntimeException)
230                throw (Error)t;
231            if (t instanceof ServletException)
232                throw (ServletException)t;
233            if (t instanceof IOException)
234                throw (IOException)t;
235            throw new RuntimeException(t);
236        } finally {
237            JTransaction.setCurrent(null);
238            if (success) {
239                if (jtx.getTransaction().isRollbackOnly())
240                    jtx.rollback();
241                else
242                    jtx.commit();
243            }
244        }
245    }
246}
