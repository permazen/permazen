006package org.jsimpledb.spring;
007
008import org.jsimpledb.core.DeletedObjectException;
009import org.jsimpledb.core.InvalidSchemaException;
010import org.jsimpledb.core.ReadOnlyTransactionException;
011import org.jsimpledb.core.ReferencedObjectException;
012import org.jsimpledb.core.RollbackOnlyTransactionException;
013import org.jsimpledb.core.SchemaMismatchException;
014import org.jsimpledb.core.StaleTransactionException;
015import org.jsimpledb.kv.RetryTransactionException;
016import org.jsimpledb.kv.TransactionTimeoutException;
017import org.springframework.dao.ConcurrencyFailureException;
018import org.springframework.dao.DataAccessException;
019import org.springframework.dao.DataIntegrityViolationException;
020import org.springframework.dao.EmptyResultDataAccessException;
021import org.springframework.dao.InvalidDataAccessApiUsageException;
022import org.springframework.dao.InvalidDataAccessResourceUsageException;
023import org.springframework.dao.QueryTimeoutException;
024import org.springframework.dao.support.PersistenceExceptionTranslator;
025
026/**
027 * JSimpleDB implementation of Spring's {@link PersistenceExceptionTranslator} interface.
028 *
029 * @see org.jsimpledb.spring
030 */
031@SuppressWarnings("serial")
032public class JSimpleDBExceptionTranslator implements PersistenceExceptionTranslator {
033
034    @Override
035    public DataAccessException translateExceptionIfPossible(RuntimeException e0) {
036        if (e0 instanceof DeletedObjectException) {
037            final DeletedObjectException e = (DeletedObjectException)e0;
038            return new EmptyResultDataAccessException("object " + e.getId() + " not found", 1, e);
039        }
040        if (e0 instanceof InvalidSchemaException || e0 instanceof ReadOnlyTransactionException)
041            return new InvalidDataAccessResourceUsageException(null, e0);
042        if (e0 instanceof ReferencedObjectException)
043            return new DataIntegrityViolationException(null, e0);
044        if (e0 instanceof RollbackOnlyTransactionException)
045            return new InvalidDataAccessApiUsageException(null, e0);
046        if (e0 instanceof SchemaMismatchException)
047            return new DataIntegrityViolationException(null, e0);
048        if (e0 instanceof StaleTransactionException || e0 instanceof org.jsimpledb.kv.StaleTransactionException)
049            return new InvalidDataAccessApiUsageException(null, e0);
050        if (e0 instanceof RetryTransactionException)
051            return new ConcurrencyFailureException(null, e0);
052        if (e0 instanceof TransactionTimeoutException)
053            return new QueryTimeoutException(null, e0);
054        return null;
055    }
056}
