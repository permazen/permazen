006package org.jsimpledb.spring;
007
008import org.jsimpledb.annotation.JSimpleClass;
009import org.springframework.beans.factory.annotation.AnnotatedBeanDefinition;
010import org.springframework.core.env.Environment;
011import org.springframework.core.type.ClassMetadata;
012
013/**
014 * Scans the classpath for types annotated as {@link JSimpleClass &#64;JSimpleClass}.
015 */
016public class JSimpleDBClassScanner extends AnnotatedClassScanner {
017
018    /**
019     * Constructor.
020     */
021    public JSimpleDBClassScanner() {
022        super(JSimpleClass.class);
023    }
024
025    /**
026     * Constructor.
027     *
028     * @param useDefaultFilters whether to register the default filters for {@link JSimpleClass &#64;JSimpleClass} type annotations
029     * @param environment environment to use
030     */
031    public JSimpleDBClassScanner(boolean useDefaultFilters, Environment environment) {
032        super(useDefaultFilters, environment, JSimpleClass.class);
033    }
034
035    /**
036     * Determine if the given bean definition is a possible search candidate.
037     *
038     * <p>
039     * This method is overridden in {@link JSimpleDBClassScanner} to allow abstract classes.
040     * </p>
041     */
042    @Override
043    protected boolean isCandidateComponent(AnnotatedBeanDefinition beanDefinition) {
044        final ClassMetadata metadata = beanDefinition.getMetadata();
045        return !metadata.isInterface() && metadata.isIndependent();
046    }
047}
