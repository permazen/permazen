006package org.jsimpledb.spring;
007
008import org.springframework.beans.factory.xml.NamespaceHandlerSupport;
009
010/**
011 * Spring {@link org.springframework.beans.factory.xml.NamespaceHandler NamespaceHandler}
012 * for the <code>jsimpledb</code> XML namespace.
013 *
014 * <p>
015 * This class adds support for the <code>&lt;jsimpledb:jsimpledb&gt;</code>, <code>&lt;jsimpledb:scan-classes&gt;</code> and
016 * <code>&lt;jsimpledb:scan-field-types&gt;</code> XML tags.
017 * </p>
018 *
019 * @see org.jsimpledb.spring
020 */
021public class JSimpleDBNamespaceHandler extends NamespaceHandlerSupport {
022
023    public static final String JSIMPLEDB_NAMESPACE_URI = "http://jsimpledb.googlecode.com/schema/jsimpledb";
024
025    public static final String JSIMPLEDB_TAG = "jsimpledb";
026    public static final String SCAN_CLASSES_TAG = "scan-classes";
027    public static final String SCAN_FIELD_TYPES_TAG = "scan-field-types";
028
029    @Override
030    public void init() {
031        this.registerBeanDefinitionParser(JSIMPLEDB_TAG, new JSimpleDBBeanDefinitionParser());
032        this.registerBeanDefinitionParser(SCAN_CLASSES_TAG, new ScanClassesBeanDefinitionParser());
033        this.registerBeanDefinitionParser(SCAN_FIELD_TYPES_TAG, new ScanFieldTypesBeanDefinitionParser());
034    }
035}
