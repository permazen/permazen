006package org.jsimpledb.spring;
007
008import java.lang.annotation.Annotation;
009import java.util.ArrayList;
010
011import org.springframework.beans.factory.config.BeanDefinition;
012import org.springframework.context.annotation.ClassPathScanningCandidateComponentProvider;
013import org.springframework.core.env.Environment;
014import org.springframework.core.io.DefaultResourceLoader;
015import org.springframework.core.type.filter.AnnotationTypeFilter;
016
017/**
018 * Scans the classpath for classes having one or more configured type annotations.
019 */
020public class AnnotatedClassScanner extends ClassPathScanningCandidateComponentProvider {
021
022    // This stupid hack is required because of JLS stupidity that doesn't allow initializing final fields prior to super()
023    private static final ThreadLocal<ArrayList<Class<? extends Annotation>>> INIT_HACK = new ThreadLocal<>();
024
025    private final ArrayList<Class<? extends Annotation>> annotationTypes;
026
027    /**
028     * Constructor. Enables use of default filters and uses a {@link org.springframework.core.env.StandardEnvironment}.
029     *
030     * @param annotationTypes type annotations to search for
031     * @throws IllegalArgumentException if {@code annotationTypes} is null, empty, or contains any non-annotation types
032     */
033    public AnnotatedClassScanner(Class<?>... annotationTypes) {
034        super(AnnotatedClassScanner.initHack(true, annotationTypes));
035        this.setResourceLoader(new DefaultResourceLoader());
036        this.annotationTypes = INIT_HACK.get();
037        INIT_HACK.remove();
038    }
039
040    /**
041     * Constructor.
042     *
043     * @param useDefaultFilters whether to register the default filters for the specified {@code annotationTypes}
044     * @param environment environment to use
045     * @param annotationTypes type annotations to search for
046     * @throws IllegalArgumentException if {@code annotationTypes} is null, empty, or contains any non-annotation types
047     */
048    public AnnotatedClassScanner(boolean useDefaultFilters, Environment environment, Class<?>... annotationTypes) {
049        super(AnnotatedClassScanner.initHack(useDefaultFilters, annotationTypes), environment);
050        this.annotationTypes = INIT_HACK.get();
051        INIT_HACK.remove();
052    }
053
054    private static boolean initHack(boolean rtn, Class<?>... annotationTypes) {
055        if (annotationTypes == null || annotationTypes.length == 0)
056            throw new IllegalArgumentException("null/empty annotationTypes");
057        final ArrayList<Class<? extends Annotation>> list = new ArrayList<>();
058        for (Class<?> cl : annotationTypes) {
059            try {
060                list.add(cl.asSubclass(Annotation.class));
061            } catch (ClassCastException e) {
062                throw new IllegalArgumentException(cl + " is not an annotation type");
063            }
064        }
065        INIT_HACK.set(list);
066        return rtn;
067    }
068
069    /**
070     * Locate annotated classes on the classpath. Does not actually load the classes.
071     *
072     * @param basePackages one or more base pacakge names
073     * @return list of names of classes having any of the configured type annotations
074     * @throws IllegalArgumentException if {@code basePackages} is empty
075     */
076    public ArrayList<String> scanForClasses(String... basePackages) {
077        if (basePackages == null || basePackages.length == 0)
078            throw new IllegalArgumentException("at least one base package name is required");
079        final ArrayList<String> nameList = new ArrayList<>();
080        for (String basePackage : basePackages) {
081            for (BeanDefinition candidate : this.findCandidateComponents(basePackage))
082                nameList.add(candidate.getBeanClassName());
083        }
084        return nameList;
085    }
086
087    /**
088     * Register default include/exclude filters.
089     *
090     * <p>
091     * Overridden in {@link AnnotatedClassScanner} to set the default filters to be ones that match classes
092     * annotated with any of the annotation types provided to the constructor.
093     * </p>
094     */
095    @Override
096    protected void registerDefaultFilters() {
097        for (Class<? extends Annotation> annotationType : this.annotationTypes != null ? this.annotationTypes : INIT_HACK.get())
098            this.addIncludeFilter(new AnnotationTypeFilter(annotationType));
099    }
100}
