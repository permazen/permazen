009import com.google.common.collect.Iterables;

011import java.util.NoSuchElementException;

013import org.jsimpledb.JClass;
014import org.jsimpledb.JField;

023/**
024 * Parsing utility routines.
025 */
026public final class ParseUtil {
027
028    private ParseUtil() {
029    }
030
031    /**
032     * Truncate a string with ellipsis if necessary.
033     *
034     * @param string original string
035     * @param len maximum length beyond which to truncate
036     * @return truncated string
037     */
038    public static String truncate(String string, int len) {
039        if (len < 4)
040            throw new IllegalArgumentException("len = " + len + " < 4");
041        if (string.length() <= len)
042            return string;
043        return string.substring(0, len - 3) + "...";
044    }
045
046    /**
047     * Generate completions based on a set of possibilities and the provided input prefix.
048     *
049     * @param choices possible choices
050     * @param prefix prefix input so far
051     * @return possible completions
052     */
053    public static Iterable<String> complete(Iterable<String> choices, String prefix) {
054        return Iterables.transform(Iterables.filter(choices, new PrefixPredicate(prefix)), new StripPrefixFunction(prefix));
055    }
056
057    /**
058     * Locate the {@link JField} with the given name in the specified object.
059     *
060     * @param session current session
061     * @param id object ID
062     * @param name field name
063     * @return the specified field
064     * @throws IllegalArgumentException if object does not exist
065     * @throws IllegalArgumentException if object's type does not exist in schema
066     * @throws IllegalArgumentException if field is not found
067     * @throws IllegalArgumentException if any parameter is null
068     */
069    public static JField resolveJField(ParseSession session, ObjId id, String name) {
070
071        // Sanity check
072        if (session == null)
073            throw new IllegalArgumentException("null session");
074        if (id == null)
075            throw new IllegalArgumentException("null id");
076        if (name == null)
077            throw new IllegalArgumentException("null name");
078        if (!session.hasJSimpleDB())
079            throw new IllegalArgumentException("this session has no JSimpleDB");
080
081        // Get object type
082        final ObjInfo info = ObjInfo.getObjInfo(session, id);
083        if (info == null)
084            throw new IllegalArgumentException("error accessing field `" + name + "': object " + id + " does not exist");
085        final ObjType objType = info.getObjType();
086
087        // Find JClass
088        final JClass<?> jclass;
089        try {
090            jclass = session.getJSimpleDB().getJClass(objType.getStorageId());
091        } catch (UnknownTypeException e) {
092            throw new IllegalArgumentException("error accessing field `" + name + "': " + e.getMessage(), e);
093        }
094
095        // Find JField
096        final JField jfield = jclass.getJFieldsByName().get(name);
097        if (jfield == null)
098            throw new IllegalArgumentException("error accessing field `" + name + "': there is no such field in " + objType);
099        return jfield;
100    }
101
102    /**
103     * Locate the field with the given name in the specified object.
104     *
105     * @param session current session
106     * @param id object ID
107     * @param name field name
108     * @return the specified field
109     * @throws IllegalArgumentException if object does not exist
110     * @throws IllegalArgumentException if field is not found
111     * @throws IllegalArgumentException if any parameter is null
112     */
113    public static Field<?> resolveField(ParseSession session, ObjId id, String name) {
114
115        // Sanity check
116        if (session == null)
117            throw new IllegalArgumentException("null session");
118        if (id == null)
119            throw new IllegalArgumentException("null id");
120        if (name == null)
121            throw new IllegalArgumentException("null name");
122
123        // Get object type
124        final ObjInfo info = ObjInfo.getObjInfo(session, id);
125        if (info == null)
126            throw new IllegalArgumentException("error accessing field `" + name + "': object " + id + " does not exist");
127        final ObjType objType = info.getObjType();
128
129        // Find the field
130        try {
131            return Iterables.find(objType.getFields().values(), new HasNamePredicate(name));
132        } catch (NoSuchElementException e) {
133            throw new IllegalArgumentException("error accessing field `" + name + "': there is no such field in " + objType);
134        }
135    }
136
137// HasNamePredicate
138
139    /**
140     * Predicate matching {@link SchemaItem}s by name.
141     */
142    public static class HasNamePredicate implements Predicate<SchemaItem> {
143
144        private final String name;
145
146        /**
147         * Constructor.
148         *
149         * @param name item name
150         */
151        public HasNamePredicate(String name) {
152            this.name = name;
153        }
154
155        @Override
156        public boolean apply(SchemaItem item) {
157            return item.getName().equals(this.name);
158        }
159    }
160}
