006package org.jsimpledb.parse.util;
007
008import com.google.common.base.Predicate;
009
010/**
011 * A {@link Predicate} that tests whether given objects are instances of a certain type.
012 */
013public class InstancePredicate implements Predicate<Object> {
014
015    private final Class<?> type;
016
017    public InstancePredicate(Class<?> type) {
018        if (type == null)
019            throw new IllegalArgumentException("null type");
020        this.type = type;
021    }
022
023    @Override
024    public boolean apply(Object obj) {
025        return obj != null && this.type.isInstance(obj);
026    }
027}
