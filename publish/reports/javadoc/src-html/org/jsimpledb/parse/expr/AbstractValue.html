050    @Override
051    public int checkIntegral(ParseSession session, String operation) {
052        final Object value = this.checkNotNull(session, operation);
053        if (value instanceof Character)
054            return (int)(Character)value;
055        if (!(value instanceof Byte) && !(value instanceof Short) && !(value instanceof Integer))
056            throw new EvalException("invalid " + operation + " operation on " + AbstractValue.describeType(value, "non-integral"));
057        return ((Number)value).intValue();
058    }

060    @Override
061    public <T> T checkType(ParseSession session, String operation, Class<T> type) {
062        if (type == null)
063            throw new IllegalArgumentException("null type");
064        final Object value = this.checkNotNull(session, operation);
065        if (!type.isInstance(value)) {
066            throw new EvalException("invalid " + operation + " operation on "
067              + AbstractValue.describeType(value, "non-" + type.getSimpleName()));
068        }
069        return type.cast(value);
070    }

072    @Override
073    public Value xxcrement(ParseSession session, String operation, boolean increment) {
074        final int amount = increment ? 1 : -1;
075        final LValue thisLValue = this.asLValue(operation);
076        final Object value = this.get(session);
077        Object num = value instanceof Character ? value : AbstractValue.checkNumeric(session, value, operation);
078        if (num instanceof Byte)
079            num = (byte)((Byte)num + (byte)amount);
080        else if (num instanceof Character)
081            num = (char)((Character)num + (char)amount);
082        else if (num instanceof Short)
083            num = (short)((Short)num + (short)amount);
084        else if (num instanceof Integer)
085            num = (Integer)num + amount;
086        else if (num instanceof Float)
087            num = (Float)num + (float)amount;
088        else if (num instanceof Long)
089            num = (Long)num + (long)amount;
090        else if (num instanceof Double)
091            num = (Double)num + (double)amount;
092        else if (num instanceof BigInteger)
093            num = ((BigInteger)num).add(increment ? BigInteger.ONE : BigInteger.ONE.negate());
094        else if (num instanceof BigDecimal)
095            num = ((BigDecimal)num).add(increment ? BigDecimal.ONE : BigDecimal.ONE.negate());
096        else
097            throw new EvalException("invalid " + operation + " operation on " + AbstractValue.describeType(num));
098        final Value result = new ConstValue(num);
099        thisLValue.set(session, result);
100        return result;
101    }

225    public Value subtract(ParseSession session, Value that) {
226
227        // Handle NavigableSet with equal comparators XXX might not have compatible elements
228        final Object thisValue = this.checkNotNull(session, "subtract");
229        final Object thatValue = that.checkNotNull(session, "subtract");
230        if (thisValue instanceof NavigableSet
231          && thatValue instanceof NavigableSet
232          && (((NavigableSet<?>)thisValue).comparator() != null ?
233           ((NavigableSet<?>)thisValue).comparator().equals(((NavigableSet<?>)thatValue).comparator()) :
234           ((NavigableSet<?>)thatValue).comparator() == null))

237        // Handle Set

241        // Handle numeric
242        final Number lnum = AbstractValue.promoteNumeric(session, thisValue, "subtract", thatValue);
243        final Number rnum = AbstractValue.promoteNumeric(session, thatValue, "subtract", thisValue);
244        if (lnum instanceof BigDecimal)

314    @Override
315    @SuppressWarnings("unchecked")
316    public Value and(ParseSession session, Value that) {
317
318        // Handle boolean
319        final Object thisValue = this.checkNotNull(session, "`and'");
320        final Object thatValue = that.checkNotNull(session, "`and'");
321        if (thisValue instanceof Boolean && thatValue instanceof Boolean)
322            return new ConstValue((Boolean)thisValue & (Boolean)thatValue);
323
324        // Handle NavigableSet with equal comparators XXX might not have compatible elements
325        if (thisValue instanceof NavigableSet
326          && thatValue instanceof NavigableSet
327          && (((NavigableSet<?>)thisValue).comparator() != null ?
328           ((NavigableSet<?>)thisValue).comparator().equals(((NavigableSet<?>)thatValue).comparator()) :
329           ((NavigableSet<?>)thatValue).comparator() == null))
330            return new ConstValue(NavigableSets.intersection((NavigableSet<Object>)thisValue, (NavigableSet<Object>)thatValue));
331
332        // Handle Set
333        if (thisValue instanceof Set && thatValue instanceof Set)
334            return new ConstValue(Sets.intersection((Set<Object>)thisValue, (Set<Object>)thatValue));
335
336        // Handle numeric
337        final Number lnum = AbstractValue.promoteNumeric(session, thisValue, "`and'", thatValue);
338        final Number rnum = AbstractValue.promoteNumeric(session, thatValue, "`and'", thisValue);
339        if (!(lnum instanceof Integer) && !(lnum instanceof Long) && !(lnum instanceof BigInteger))
340            throw new EvalException("invalid `and' operation on " + AbstractValue.describeType(lnum));
341        if (lnum instanceof BigInteger)
342            return new ConstValue(((BigInteger)lnum).and((BigInteger)rnum));
343        if (lnum instanceof Long)
344            return new ConstValue(((Long)lnum).longValue() & ((Long)rnum).longValue());
345        return new ConstValue(((Integer)lnum).intValue() & ((Integer)rnum).intValue());
346    }

386        // Handle boolean
387        final Object thisValue = this.checkNotNull(session, "exclusive `or'");

389        if (thisValue instanceof Boolean && thatValue instanceof Boolean)

418    @Override
419    @SuppressWarnings("unchecked")
420    public Value compare(ParseSession session, Value that, int mask) {
421
422        // Try numeric comparison
423        final Object thisValue = this.checkNotNull(session, "comparison");
424        final Object thatValue = that.checkNotNull(session, "comparison");
425        while (thisValue instanceof Number && thatValue instanceof Number) {
426            final Number lnum = AbstractValue.promoteNumeric(session, thisValue, "comparison", thatValue);
427            final Number rnum = AbstractValue.promoteNumeric(session, thatValue, "comparison", thisValue);
428            int result;
429            if (lnum instanceof BigDecimal)
430                result = ((BigDecimal)lnum).compareTo((BigDecimal)rnum);
431            else if (lnum instanceof Double)
432                result = Double.compare((Double)lnum, (Double)rnum);
433            else if (lnum instanceof Float)
434                result = Float.compare((Float)lnum, (Float)rnum);
435            else if (lnum instanceof BigInteger)
436                result = ((BigInteger)lnum).compareTo((BigInteger)rnum);
437            else if (lnum instanceof Long)
438                result = Long.compare((Long)lnum, (Long)rnum);
439            else if (lnum instanceof Integer)
440                result = Integer.compare((Integer)lnum, (Integer)rnum);
441            else
442                break;
443            result = result < 0 ? LT : result > 0 ? GT : EQ;
444            return new ConstValue((result & mask) != 0);
445        }
446
447        // Try natural ordering using Comparable
448        while (thisValue instanceof Comparable && thatValue instanceof Comparable) {
449            int result;
450            try {
451                result = ((Comparable<Object>)thisValue).compareTo(thatValue);
452            } catch (ClassCastException e) {
453                break;
454            }
455            result = result < 0 ? LT : result > 0 ? GT : EQ;
456            return new ConstValue((result & mask) != 0);
457        }
458
459        // We don't know how to order these
460        throw new EvalException("invalid comparison operation between "
461          + AbstractValue.describeType(thisValue) + " and " + AbstractValue.describeType(thatValue));
462    }

473    /**
474     * Promote the given object to the widest numeric type among itself and the provided objects, with integer as a lower bound.
475     *
476     * @param session current session
477     * @param obj Java value
478     * @param operation description of operation for error messages
479     * @param args zero or more other Java values
480     * @throws IllegalArgumentException if any argument is not numeric
481     */
482    static Number promoteNumeric(final ParseSession session, final Object obj, final String operation, Object... args) {
483
484        // Promote any byte, char, or short up to int
485        Number num = obj instanceof Character ? (int)(Character)obj : AbstractValue.checkNumeric(session, obj, operation);
486        if (num instanceof Byte || num instanceof Short)
487            num = num.intValue();
488
489        // Promote to widest type of any argument
490        if (args.length > 0) {
491            final ArrayList<Number> nums = new ArrayList<Number>(1 + args.length);
492            nums.add(num);
493            nums.addAll(Lists.transform(Arrays.asList(args), new Function<Object, Number>() {
494                @Override
495                public Number apply(Object value) {
496                    return AbstractValue.promoteNumeric(session, value, operation);
497                }
498            }));
499            if (Iterables.find(nums, new InstancePredicate(BigDecimal.class), null) != null)
500                num = AbstractValue.toBigDecimal(num);
501            else if (Iterables.find(nums, new InstancePredicate(Double.class), null) != null)
502                num = num.doubleValue();
503            else if (Iterables.find(nums, new InstancePredicate(Float.class), null) != null)
504                num = num.floatValue();
505            else if (Iterables.find(nums, new InstancePredicate(BigInteger.class), null) != null)
506                num = AbstractValue.toBigInteger(num);
507            else if (Iterables.find(nums, new InstancePredicate(Long.class), null) != null)
508                num = num.longValue();
509            else
510                num = num.intValue();
511        }
512
513        // Done
514        return num;
515    }

531    /**
532     * Convert a {@link Number} to {@link BigDecimal}.
533     *
534     * @param num number
535     */
536    static BigDecimal toBigDecimal(Number num) {
537        if (num instanceof BigDecimal)
538            return (BigDecimal)num;
539        if (num instanceof BigInteger)
540            return new BigDecimal((BigInteger)num);
541        if (num instanceof Double || num instanceof Float)
542            return new BigDecimal(num.doubleValue());
543        if (num instanceof Long || num instanceof Integer || num instanceof Short || num instanceof Byte)
544            return new BigDecimal(num.longValue());
545        throw new EvalException("can't convert " + AbstractValue.describeType(num) + " to BigDecimal");
546    }

557    static String describeType(Object obj, String... prefix) {
558        if (obj == null)
559            return "null value";
560        return (prefix.length > 0 ? prefix[0] + " " : "") + "value of type " + obj.getClass().getName();
