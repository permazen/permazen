006package org.jsimpledb.parse.expr;
007
008import org.jsimpledb.parse.ParseSession;
009
010/**
011 * Identifier.
012 */
013public class IdentNode extends AbstractNamed implements Node {
014
015    /**
016     * Constructor.
017     *
018     * @param name variable name
019     * @throws IllegalArgumentException if name is null
020     * @throws IllegalArgumentException if name is not a valid Java identifier
021     */
022    public IdentNode(String name) {
023        super(name);
024    }
025
026// Node
027
028    @Override
029    public Value evaluate(ParseSession session) {
030        throw new RuntimeException("internal error: can't evaluate ident `" + this.name + "'");
031    }
032}
