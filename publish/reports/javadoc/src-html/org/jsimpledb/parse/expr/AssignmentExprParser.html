006package org.jsimpledb.parse.expr;
007
008import org.jsimpledb.parse.ParseSession;
009
010/**
011 * Parses Java assignment expressions of the form {@code x = y}, {@code x += y}, etc.
012 */
013public class AssignmentExprParser extends BinaryExprParser {
014
015    public static final AssignmentExprParser INSTANCE = new AssignmentExprParser();
016
017    public AssignmentExprParser() {
018        super(ConditionalParser.INSTANCE, false, Op.EQUALS, Op.PLUS_EQUALS, Op.MINUS_EQUALS, Op.MULTIPLY_EQUALS,
019          Op.DIVIDE_EQUALS, Op.MODULO_EQUALS, Op.AND_EQUALS, Op.XOR_EQUALS, Op.OR_EQUALS, Op.LSHIFT_EQUALS,
020          Op.RSHIFT_EQUALS, Op.URSHIFT_EQUALS);
021    }
022
023    @Override
024    protected Node createNode(final Op op, final Node lhs, final Node rhs) {
025        return new Node() {
026            @Override
027            public Value evaluate(ParseSession session) {
028
029                // Get left-hand value, which must be an LValue
030                final LValue lvalue = lhs.evaluate(session).asLValue("assignment");
031
032                // Calculate new value
033                Value value = rhs.evaluate(session);
034                if (op != Op.EQUALS)
035                    value = Op.forSymbol(op.getSymbol().replaceAll("=", "")).apply(session, lvalue, value);
036
037                // Assign new value to lvalue
038                lvalue.set(session, value);
039
040                // Done
041                return value;
042            }
043        };
044    }
045}
