006package org.jsimpledb.parse.expr;
007
008import java.lang.reflect.InvocationTargetException;
009import java.lang.reflect.Method;
010
011import org.jsimpledb.parse.ParseSession;
012
013/**
014 * {@link Value} that reflects a mutable bean property in some object.
015 */
016public class MutableBeanPropertyValue extends BeanPropertyValue implements LValue {
017
018    protected final Method setter;
019
020    /**
021     * Constructor.
022     *
023     * @param bean bean object
024     * @param name property name
025     * @param getter getter method
026     * @param setter setter method
027     * @throws IllegalArgumentException if any parameter is null
028     */
029    public MutableBeanPropertyValue(Object bean, String name, Method getter, Method setter) {
030        super(bean, name, getter);
031        if (setter == null)
032            throw new IllegalArgumentException("null setter");
033        this.setter = setter;
034    }
035
036    @Override
037    public void set(ParseSession session, Value value) {
038        final Object obj = value.get(session);
039        try {
040            this.setter.invoke(this.bean, obj);
041        } catch (Exception e) {
042            final Throwable t = e instanceof InvocationTargetException ?
043              ((InvocationTargetException)e).getTargetException() : e;
044            throw new EvalException("error writing property `" + this.name + "' from object of type "
045              + this.bean.getClass().getName() + ": " + t, t);
046        }
047    }
048}
