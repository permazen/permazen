006package org.jsimpledb.parse.expr;

008import java.math.BigInteger;
009import java.util.TreeSet;
010import java.util.regex.Matcher;
011import java.util.regex.Pattern;
012
013import org.dellroad.stuff.java.Primitive;
014import org.dellroad.stuff.string.StringEncoder;
015import org.jsimpledb.JTransaction;
016import org.jsimpledb.core.FieldType;
017import org.jsimpledb.core.ObjId;
018import org.jsimpledb.parse.ObjIdParser;
019import org.jsimpledb.parse.ParseContext;
020import org.jsimpledb.parse.ParseException;
021import org.jsimpledb.parse.ParseSession;
022import org.jsimpledb.parse.ParseUtil;
023import org.jsimpledb.parse.Parser;
024import org.jsimpledb.parse.SpaceParser;

026/**
027 * The lowest parse level. Parses parenthesized expressions, literals, and identifiers.
028 *
029 * <p>
030 * Includes these special extensions:
031 * <ul>
032 *  <li>Variable references, e.g., {@code $foo}</li>
033 *  <li>Object ID literals, e.g., <code>@fc21bf6d8930a215</code></li>
034 *  <li>Any {@link FieldType} value in {@linkplain FieldType#fromParseableString parseable string form}
035 *      preceded by the {@link FieldType} name in curly braces, e.g., <code>{java.util.Date}2015-01-23T07:19:42</code></li>
036 * </ul>
037 */
038public class AtomParser implements Parser<Node> {
039
040    public static final AtomParser INSTANCE = new AtomParser();
041
042    private final SpaceParser spaceParser = new SpaceParser();
043    private final TreeSet<String> identifierCompletions;
044
045    public AtomParser() {
046        this(null);
047    }
048
049    /**
050     * Constructor.
051     *
052     * @param identifierCompletions set of valid identifiers tab completions, or null to allow any identifiers
053     */
054    public AtomParser(Iterable<String> identifierCompletions) {
055        if (identifierCompletions != null) {
056            this.identifierCompletions = new TreeSet<String>();
057            for (String identifierCompletion : identifierCompletions)
058                this.identifierCompletions.add(identifierCompletion);
059        } else
060            this.identifierCompletions = null;
061    }
062
063    @Override
064    public Node parse(ParseSession session, ParseContext ctx, boolean complete) {
065        final int mark = ctx.getIndex();
066
067        // Check for parenthesized expression
068        if (ctx.tryLiteral("(")) {
069            this.spaceParser.parse(ctx, complete);
070            final Node node = AssignmentExprParser.INSTANCE.parse(session, ctx, complete);
071            this.spaceParser.parse(ctx, complete);
072            if (!ctx.tryLiteral(")"))
073                throw new ParseException(ctx).addCompletion(") ");
074            return node;
075        }
076        final int start = ctx.getIndex();
077
078        // Try to match null
079        if (this.tryWord(ctx, "null") != null)
080            return new LiteralNode(null);
081
082        // Try to match boolean
083        final String booleanMatch = this.tryWord(ctx, "false|true");
084        if (booleanMatch != null)
085            return new LiteralNode(Boolean.valueOf(booleanMatch));
086
087        // Try to match int or long literal
088        try {
089            final int radix;
090            Matcher matcher;
091            if ((matcher = ctx.tryPattern("(?si)(\\+|-|)(?:0x|#)(\\p{XDigit}+)(L)?")) != null)
092                radix = 16;
093            else if ((matcher = ctx.tryPattern("(?si)(\\+|-|)0b([01]+)(L)?")) != null)
094                radix = 2;
095            else if ((matcher = ctx.tryPattern("(?si)(\\+|-|)(0[0-7]*)(L)?")) != null)
096                radix = 8;
097            else if ((matcher = ctx.tryPattern("(?si)(\\+|-|)([1-9][0-9]*)(L)?")) != null)
098                radix = 10;
099            else
100                throw new IllegalArgumentException("no pattern matched");
101            final String digits = matcher.group(1) + matcher.group(2);
102            final boolean isLong = matcher.group(3) != null;
103            if (ctx.tryPattern("[.\\p{javaJavaIdentifierPart}]") != null)
104                throw new IllegalArgumentException("followed by floating stuff");
105            final BigInteger big = new BigInteger(digits, radix);
106            if (big.bitLength() > (isLong ? 64 : 32))
107                throw new IllegalArgumentException("bit length = " + big.bitLength());
108            final Number value = isLong ? (Number)big.longValue() : (Number)big.intValue();
109            return new LiteralNode(value);
110        } catch (IllegalArgumentException e) {
111            ctx.setIndex(start);
112        }
113
114        // Try to match float or double literal
115        final String floatMatch = this.tryWord(ctx, Primitive.FLOAT.getParsePattern() + "[fF]");
116        if (floatMatch != null) {
117            try {
118                return new LiteralNode(Primitive.FLOAT.parseValue(floatMatch.substring(0, floatMatch.length() - 1)));
119            } catch (IllegalArgumentException e) {
120                ctx.setIndex(start);
121            }
122        }
123        final String doubleMatch = this.tryFollow(ctx, Primitive.DOUBLE.getParsePattern(), "[^.\\p{javaJavaIdentifierPart}]");
124        if (doubleMatch != null) {
125            try {
126                return new LiteralNode(Primitive.DOUBLE.parseValue(doubleMatch));
127            } catch (IllegalArgumentException e) {
128                ctx.setIndex(start);
129            }
130        }
131
132        // Try to match a char literal
133        String match = this.tryWord(ctx,
134          StringEncoder.ENQUOTE_PATTERN.toString().replaceAll("\"", "'").replaceAll("\\*", ""));    // kludge
135        if (match != null) {
136            match = match.substring(1, match.length() - 1);
137            if (match.length() > 0 && match.charAt(0) == '\'') {
138                ctx.setIndex(start);
139                throw new ParseException(ctx, "invalid character: contains unescaped single quote");
140            }
141            match = StringEncoder.decode(match.replaceAll(Pattern.quote("\\'"), Matcher.quoteReplacement("'")));
142            if (match.length() != 1) {
143                ctx.setIndex(start);
144                throw new ParseException(ctx, "invalid character: quotes must contain exactly one character");
145            }
146            return new LiteralNode(match.charAt(0));
147        }
148
149        // Try to match a string literal
150        final Matcher stringMatch = ctx.tryPattern(StringEncoder.ENQUOTE_PATTERN);
151        if (stringMatch != null)
152            return new LiteralNode(new String(StringEncoder.dequote(stringMatch.group())));
153
154        // Try to type from type registry within curly braces
155        int fieldTypeStart = ctx.getIndex();
156        final Matcher braceMatch = ctx.tryPattern("\\{(" + FieldType.NAME_PATTERN + ")\\}");
157        if (braceMatch != null) {
158            final String fieldTypeName = braceMatch.group(1);
159            final FieldType<?> fieldType = session.getDatabase().getFieldTypeRegistry().getFieldType(fieldTypeName);
160            if (fieldType == null) {
161                ctx.setIndex(fieldTypeStart);
162                throw new ParseException(ctx, "unknown simple field type `" + fieldTypeName + "'");
163            }
164            fieldTypeStart = ctx.getIndex();
165            try {
166                return new LiteralNode(fieldType.fromParseableString(ctx));
167            } catch (IllegalArgumentException e) {
168                ctx.setIndex(fieldTypeStart);
169                throw new ParseException(ctx, "invalid value for type `" + fieldTypeName + "'");
170            }
171        }
172
173        // Try to match variable
174        if (ctx.tryLiteral("$")) {
175            final Matcher varMatcher = ctx.tryPattern(AbstractNamed.NAME_PATTERN);
176            if (varMatcher == null)
177                throw new ParseException(ctx).addCompletions(session.getVars().keySet());
178            final String name = varMatcher.group();
179            if (ctx.isEOF() && complete)
180                throw new ParseException(ctx).addCompletions(ParseUtil.complete(session.getVars().keySet(), name));
181            return new VarNode(name);
182        }
183
184        // Try to match object literal
185        if ((session.getMode().hasCoreAPI() || session.getMode().hasJSimpleDB()) && ctx.tryLiteral("@")) {
186            final ObjId id = new ObjIdParser().parse(session, ctx, complete);
187            return !session.getMode().hasJSimpleDB() ? new LiteralNode(id) : new Node() {
188                @Override
189                public Value evaluate(ParseSession session) {
190                    return new ConstValue(JTransaction.getCurrent().getJObject(id));
191                }
192            };
193        }
194
195        // Try to match identifier; support tab-completion of configured identifiers, if any
196        final Matcher identMatcher = ctx.tryPattern(IdentNode.NAME_PATTERN);
197        if (identMatcher != null) {
198            final String name = identMatcher.group();
199            if (complete && ctx.isEOF() && this.identifierCompletions != null && !name.equals("new"))
200                throw new ParseException(ctx).addCompletions(ParseUtil.complete(this.identifierCompletions, name));
201            return new IdentNode(name);
202        }
203        if (complete && ctx.isEOF() && this.identifierCompletions != null)
204            throw new ParseException(ctx).addCompletions(this.identifierCompletions);
205
206        // No match
207        throw new ParseException(ctx);
208    }
209
210// Match a pattern not followed by a identifier letter
211
212    private String tryWord(ParseContext ctx, Pattern pattern) {
213        return this.tryWord(ctx, pattern.toString());
214    }
215
216    private String tryWord(ParseContext ctx, String pattern) {
217        return this.tryFollow(ctx, pattern, "[^\\p{javaJavaIdentifierPart}]");
218    }
219
220    private String tryFollow(ParseContext ctx, Pattern pattern, String follow) {
221        return this.tryFollow(ctx, pattern.toString(), follow);
222    }
223
224    private String tryFollow(ParseContext ctx, String pattern, String follow) {
225        final Matcher matcher = Pattern.compile("(" + pattern + ")(" + follow + "(?s:.*))?").matcher(ctx.getInput());
226        if (!matcher.matches())
227            return null;
228        final String match = matcher.group(1);
229        ctx.setIndex(ctx.getIndex() + match.length());
230        return match;
231    }
232}
