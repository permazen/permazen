006package org.jsimpledb.parse.expr;
007
008import java.lang.reflect.Field;
009
010import org.jsimpledb.core.ObjId;
011import org.jsimpledb.parse.ParseSession;
012
013/**
014 * {@link Value} that reflects a core API {@link org.jsimpledb.core.Field} in some database object.
015 *
016 * @see JFieldValue
017 */
018public class FieldValue extends AbstractValue {
019
020    protected final ObjId id;
021    protected final org.jsimpledb.core.Field<?> field;
022
023    /**
024     * Constructor.
025     *
026     * @param id object ID
027     * @param field database field
028     * @throws IllegalArgumentException if {@code id} is null
029     * @throws IllegalArgumentException if {@code field} is null
030     */
031    public FieldValue(ObjId id, org.jsimpledb.core.Field<?> field) {
032        if (id == null)
033            throw new IllegalArgumentException("null id");
034        if (field == null)
035            throw new IllegalArgumentException("null field");
036        this.id = id;
037        this.field = field;
038    }
039
040    @Override
041    public Object get(ParseSession session) {
042        try {
043            return this.field.getValue(session.getTransaction(), this.id);
044        } catch (Exception e) {
045            throw new EvalException("error reading field `" + this.field.getName() + "' from object " + this.id + ": "
046              + (e.getMessage() != null ? e.getMessage() : e));
047        }
048    }
049}
