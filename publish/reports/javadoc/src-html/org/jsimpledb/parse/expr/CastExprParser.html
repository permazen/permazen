006package org.jsimpledb.parse.expr;
007
008import java.util.regex.Matcher;
009
010import org.dellroad.stuff.java.Primitive;
011import org.dellroad.stuff.java.PrimitiveSwitch;
012import org.jsimpledb.parse.ParseContext;
013import org.jsimpledb.parse.ParseSession;
014import org.jsimpledb.parse.Parser;
015import org.jsimpledb.parse.SpaceParser;
016
017/**
018 * Parses type cast expressions.
019 */
020public class CastExprParser implements Parser<Node> {
021
022    public static final CastExprParser INSTANCE = new CastExprParser();
023
024    private final SpaceParser spaceParser = new SpaceParser();
025
026    @Override
027    public Node parse(ParseSession session, ParseContext ctx, boolean complete) {
028
029        // Try cast
030        final int start = ctx.getIndex();
031        final Matcher castMatcher = ctx.tryPattern("\\((" + IdentNode.NAME_PATTERN
032          + "(\\s*\\.\\s*" + IdentNode.NAME_PATTERN + ")*)\\s*\\)\\s*");
033        if (castMatcher != null) {
034            final String className = castMatcher.group(1).replaceAll("\\s", "");
035            if (className.equals("null")) {
036                ctx.setIndex(start);
037                return UnaryExprParser.INSTANCE.parse(session, ctx, complete);
038            }
039            final Node target = this.parse(session, ctx, complete);             // associates right-to-left
040            ctx.skipWhitespace();
041            return new Node() {
042                @Override
043                public Value evaluate(final ParseSession session) {
044
045                    // Evaluate target
046                    final Object obj = target.evaluate(session).get(session);
047
048                    // Handle primitive cast, e.g. "(int)foo"
049                    final Primitive<?> primitive = Primitive.forName(className);
050                    if (primitive != null && primitive != Primitive.VOID) {
051                        if (obj == null)
052                            throw new EvalException("invalid cast of null value to " + className);
053                        if (primitive == Primitive.BOOLEAN)
054                            return new ConstValue(new ConstValue(obj).checkBoolean(session, "cast to " + className));
055                        final Number num = obj instanceof Character ?
056                          (int)(Character)obj : new ConstValue(obj).checkNumeric(session, "cast to " + className);
057                        return new ConstValue(primitive.visit(new PrimitiveSwitch<Object>() {
058                            @Override
059                            public Object caseVoid() {
060                                throw new RuntimeException("internal error");
061                            }
062                            @Override
063                            public Object caseBoolean() {
064                                throw new RuntimeException("internal error");
065                            }
066                            @Override
067                            public Object caseCharacter() {
068                                throw new RuntimeException("internal error");
069                            }
070                            @Override
071                            public Object caseByte() {
072                                return num.byteValue();
073                            }
074                            @Override
075                            public Object caseShort() {
076                                return num.shortValue();
077                            }
078                            @Override
079                            public Object caseInteger() {
080                                return num.intValue();
081                            }
082                            @Override
083                            public Object caseFloat() {
084                                return num.floatValue();
085                            }
086                            @Override
087                            public Object caseLong() {
088                                return num.longValue();
089                            }
090                            @Override
091                            public Object caseDouble() {
092                                return num.doubleValue();
093                            }
094                        }));
095                    }
096
097                    // Handle regular cast
098                    final Class<?> cl = session.resolveClass(className);
099                    if (cl == null)
100                        throw new EvalException("unknown class `" + className + "'");     // TODO: tab-completions
101                    if (obj != null && !cl.isInstance(obj))
102                        throw new EvalException("can't cast object of type " + obj.getClass().getName() + " to " + className);
103                    return new ConstValue(obj);
104                }
105            };
106        }
107
108        // Parse unary
109        return UnaryExprParser.INSTANCE.parse(session, ctx, complete);
110    }
111}
