006package org.jsimpledb.parse.expr;
007
008import java.lang.reflect.Field;
009import java.lang.reflect.Modifier;
010
011import org.jsimpledb.parse.ParseSession;
012
013/**
014 * {@link Value} that reflects a instance Java field in some object.
015 */
016public class ObjectFieldValue extends AbstractFieldValue {
017
018    protected final Object object;
019
020    /**
021     * Constructor.
022     *
023     * @param object object containing field
024     * @param field field to access
025     * @throws IllegalArgumentException if {@code object} is null
026     * @throws IllegalArgumentException if {@code field} is null
027     * @throws IllegalArgumentException if {@code field} is static
028     */
029    public ObjectFieldValue(Object object, Field field) {
030        super(field);
031        if (object == null)
032            throw new IllegalArgumentException("null object");
033        if ((field.getModifiers() & Modifier.STATIC) != 0)
034            throw new IllegalArgumentException("field is static");
035        this.object = object;
036    }
037
038    @Override
039    public Object get(ParseSession session) {
040        try {
041            return this.field.get(this.object);
042        } catch (Exception e) {
043            throw new EvalException("error reading field `" + this.field.getName() + "' in object of type "
044              + this.object.getClass().getName() + ": " + e, e);
045        }
046    }
047
048    @Override
049    public void set(ParseSession session, Value value) {
050        final Object obj = value.get(session);
051        try {
052            this.field.set(this.object, obj);
053        } catch (Exception e) {
054            throw new EvalException("error writing field `" + this.field.getName() + "' in object of type "
055              + this.object.getClass().getName() + ": " + e, e);
056        }
057    }
058}
