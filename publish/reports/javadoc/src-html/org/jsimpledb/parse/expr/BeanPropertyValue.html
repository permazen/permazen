006package org.jsimpledb.parse.expr;
007
008import java.lang.reflect.InvocationTargetException;
009import java.lang.reflect.Method;
010
011import org.jsimpledb.parse.ParseSession;
012
013/**
014 * {@link Value} that reflects a bean property in some object.
015 *
016 * <p>
017 * For mutable bean properties, use {@link MutableBeanPropertyValue}, which is also an {@link LValue}.
018 * </p>
019 */
020public class BeanPropertyValue extends AbstractValue {
021
022    protected final Object bean;
023    protected final String name;
024    protected final Method getter;
025
026    /**
027     * Constructor.
028     *
029     * @param bean bean object
030     * @param name property name
031     * @param getter getter method
032     * @throws IllegalArgumentException if any parameter is null
033     */
034    public BeanPropertyValue(Object bean, String name, Method getter) {
035        if (bean == null)
036            throw new IllegalArgumentException("null bean");
037        if (name == null)
038            throw new IllegalArgumentException("null name");
039        if (getter == null)
040            throw new IllegalArgumentException("null getter");
041        this.bean = bean;
042        this.name = name;
043        this.getter = getter;
044    }
045
046    @Override
047    public Object get(ParseSession session) {
048        try {
049            return getter.invoke(this.bean);
050        } catch (Exception e) {
051            final Throwable t = e instanceof InvocationTargetException ?
052              ((InvocationTargetException)e).getTargetException() : e;
053            throw new EvalException("error reading property `" + this.name + "' from object of type "
054              + this.bean.getClass().getName() + ": " + t, t);
055        }
056    }
057}
