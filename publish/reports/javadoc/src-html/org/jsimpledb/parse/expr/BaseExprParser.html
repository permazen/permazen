043/**
044 * Parses basic Java expressions such as {@link Class} literals, {@code new} expressions, auto-increment expressions,
045 * array access, field access, method invocation, etc.
046 *
047 * <p>
048 * Includes these extensions:
049 * <ul>
050 *  <li>Access to Java bean methods via "property syntax", e.g., {@code foo.name = "fred"} means {@code foo.setName("fred")}</li>
051 *  <li>Access database object fields via "property syntax" given an object ID, e.g., <code>@fc21bf6d8930a215.name</code></li>
052 *  <li>Array syntax for {@link Map} and {@link List} value access, e.g., {@code mymap[key] = value} means
053 *      {@code mymap.put(key, value)}, and {@code mylist[12] = "abc"} means {@code mylist.set(12, "abc")}</li>
054 * </ul>
055 */
056public class BaseExprParser implements Parser<Node> {
057
058    public static final BaseExprParser INSTANCE = new BaseExprParser();
059
060    private final SpaceParser spaceParser = new SpaceParser();
061
062    @Override
063    public Node parse(ParseSession session, ParseContext ctx, boolean complete) {
064
065        // Parse initial atom
066        Node node = new AtomParser(
067          Iterables.transform(session.getFunctions().keySet(), new AddSuffixFunction("("))).parse(session, ctx, complete);
068
069        // Handle new
070        if (node instanceof IdentNode && ((IdentNode)node).getName().equals("new")) {
071
072            // Get class name (or array type base), which must be a sequence of identifiers connected via dots
073            new SpaceParser(true).parse(ctx, complete);
074            final Matcher firstMatcher = ctx.tryPattern(IdentNode.NAME_PATTERN);
075            if (firstMatcher == null)
076                throw new ParseException(ctx, "expected class name");
077            String className = firstMatcher.group();
078            while (true) {
079                final Matcher matcher = ctx.tryPattern("\\s*\\.\\s*(" + IdentNode.NAME_PATTERN + ")");
080                if (matcher == null)
081                    break;
082                className += "." + matcher.group(1);
083            }
084
085            // Get array dimensions, if any
086            final ArrayList<Node> dims = new ArrayList<>();
087            boolean sawNull = false;
088            while (true) {
089                ctx.skipWhitespace();
090                if (!ctx.tryLiteral("["))
091                    break;
092                ctx.skipWhitespace();
093                if (ctx.tryLiteral("]")) {
094                    dims.add(null);
095                    sawNull = true;
096                    continue;
097                }
098                if (!sawNull) {
099                    dims.add(ExprParser.INSTANCE.parse(session, ctx, complete));
100                    ctx.skipWhitespace();
101                }
102                if (!ctx.tryLiteral("]"))
103                    throw new ParseException(ctx, "expected `]'").addCompletion("]");
104            }
105
106            // Resolve (base) class name
107            final Class<?> baseClass;
108            final Primitive<?> primitive;
109            if (!dims.isEmpty() && (primitive = Primitive.forName(className)) != null) {
110                if (primitive == Primitive.VOID)
111                    throw new ParseException(ctx, "illegal instantiation of void array");
112                baseClass = primitive.getType();
113            } else {
114                baseClass = session.resolveClass(className);
115                if (baseClass == null)
116                    throw new ParseException(ctx, "unknown class `" + className + "'");     // TODO: tab-completions
117            }
118
119            // Handle non-array
120            if (dims.isEmpty()) {
121
122                // Parse parameters
123                if (!ctx.tryLiteral("("))
124                    throw new ParseException(ctx, "expected `('").addCompletion("(");
125                ctx.skipWhitespace();
126                final List<Node> paramNodes = BaseExprParser.parseParams(session, ctx, complete);
127
128                // Return constructor invocation node
129                node = new Node() {
130                    @Override
131                    public Value evaluate(final ParseSession session) {
132                        if (baseClass.isInterface())
133                            throw new EvalException("invalid instantiation of " + baseClass);
134                        final Object[] params = Lists.transform(paramNodes, new com.google.common.base.Function<Node, Object>() {
135                            @Override
136                            public Object apply(Node param) {
137                                return param.evaluate(session).get(session);
138                            }
139                        }).toArray();
140                        for (Constructor<?> constructor : baseClass.getConstructors()) {
141                            final Class<?>[] ptypes = constructor.getParameterTypes();
142                            if (ptypes.length != params.length)
143                                continue;
144                            try {
145                                return new ConstValue(constructor.newInstance(params));
146                            } catch (IllegalArgumentException e) {
147                                continue;                               // wrong method, a parameter type didn't match
148                            } catch (Exception e) {
149                                final Throwable t = e instanceof InvocationTargetException ?
150                                  ((InvocationTargetException)e).getTargetException() : e;
151                                throw new EvalException("error invoking constructor `" + baseClass.getName() + "()': " + t, t);
152                            }
153                        }
154                        throw new EvalException("no compatible constructor found in " + baseClass);
155                    }
156                };
157            } else {                                        // handle array
158
159                // Check number of dimensions
160                if (dims.size() > 255)
161                    throw new ParseException(ctx, "too many array dimensions (" + dims.size() + " > 255)");
162
163                // Array literal must be present if and only if no dimensions are given
164                final List<?> literal = dims.get(0) == null ?
165                  this.parseArrayLiteral(session, ctx, complete, this.getArrayClass(baseClass, dims.size() - 1)) : null;
166
167                // Return array instantiation invocation node
168                node = new Node() {
169                    @Override
170                    public Value evaluate(final ParseSession session) {
171                        final Class<?> elemType = BaseExprParser.this.getArrayClass(baseClass, dims.size() - 1);
172                        return new ConstValue(literal != null ?
173                          this.createLiteral(session, elemType, literal) : this.createEmpty(session, elemType, dims));
174                    }
175
176                    private Object createEmpty(ParseSession session, Class<?> elemType, List<Node> dims) {
177                        final int length = dims.get(0).evaluate(session).checkIntegral(session, "array creation");
178                        final Object array = Array.newInstance(elemType, length);
179                        final List<Node> remainingDims = dims.subList(1, dims.size());
180                        if (!remainingDims.isEmpty() && remainingDims.get(0) != null) {
181                            for (int i = 0; i < length; i++)
182                                Array.set(array, i, this.createEmpty(session, elemType.getComponentType(), remainingDims));
183                        }
184                        return array;
185                    }
186
187                    private Object createLiteral(ParseSession session, Class<?> elemType, List<?> values) {
188                        final int length = values.size();
189                        final Object array = Array.newInstance(elemType, length);
190                        for (int i = 0; i < length; i++) {
191                            if (elemType.isArray()) {
192                                Array.set(array, i, this.createLiteral(session,
193                                  elemType.getComponentType(), (List<?>)values.get(i)));
194                            } else {
195                                try {
196                                    Array.set(array, i, ((Node)values.get(i)).evaluate(session).get(session));
197                                } catch (Exception e) {
198                                    throw new EvalException("error setting array value: " + e, e);
199                                }
200                            }
201                        }
202                        return array;
203                    }
204                };
205            }
206        }
207
208        // Repeatedly parse operators (this gives left-to-right association)
209        while (true) {
210
211            // Parse operator, if any (one of `[', `.', `(', `++', or `--')
212            final Matcher opMatcher = ctx.tryPattern("\\s*(\\[|\\.|\\(|\\+{2}|-{2})");
213            if (opMatcher == null)
214                return node;
215            final String opsym = opMatcher.group(1);
216            final int mark = ctx.getIndex();
217
218            // Handle operators
219            switch (opsym) {
220            case "(":
221            {
222                // Atom must be an identifier, the name of a global function being invoked
223                if (!(node instanceof IdentNode))
224                    throw new ParseException(ctx);
225                final String functionName = ((IdentNode)node).getName();
226                final AbstractFunction function = session.getFunctions().get(functionName);
227                if (function == null) {
228                    throw new ParseException(ctx, "unknown function `" + functionName + "()'")
229                      .addCompletions(ParseUtil.complete(session.getFunctions().keySet(), functionName));
230                }
231
232                // Parse function parameters
233                ctx.skipWhitespace();
234                final Object params = function.parseParams(session, ctx, complete);
235
236                // Return node that applies the function to the parameters
237                node = new Node() {
238                    @Override
239                    public Value evaluate(ParseSession session) {
240                        return function.apply(session, params);
241                    }
242                };
243                break;
244            }
245            case ".":
246            {
247                // Parse next atom - it must be an identifier, either a method or property name
248                ctx.skipWhitespace();
249                final Node memberNode = AtomParser.INSTANCE.parse(session, ctx, complete);
250                if (!(memberNode instanceof IdentNode)) {
251                    ctx.setIndex(mark);
252                    throw new ParseException(ctx);
253                }
254                String member = ((IdentNode)memberNode).getName();
255
256                // If first atom was an identifier, this must be a class name followed by a field or method name
257                Class<?> cl = null;
258                if (node instanceof IdentNode) {
259
260                    // Keep parsing identifiers as long as we can; after each identifier, try to resolve a class name
261                    final TreeMap<Integer, Class<?>> classes = new TreeMap<>();
262                    String idents = ((IdentNode)node).getName();
263                    while (true) {
264                        classes.put(ctx.getIndex(), session.resolveClass(idents));
265                        final Matcher matcher = ctx.tryPattern("\\s*\\.\\s*(" + IdentNode.NAME_PATTERN + ")");
266                        if (matcher == null)
267                            break;
268                        idents += "." + member;
269                        member = matcher.group(1);
270                    }
271
272                    // Use the longest class name resolved, if any
273                    for (Map.Entry<Integer, Class<?>> entry : classes.descendingMap().entrySet()) {
274                        if (entry.getValue() != null) {
275                            cl = entry.getValue();
276                            ctx.setIndex(entry.getKey());
277                            break;
278                        }
279                    }
280                    if (cl == null)
281                        throw new ParseException(ctx, "unknown class `" + idents + "'");        // TODO: tab-completions
282                }
283
284                // Handle property access
285                if (ctx.tryPattern("\\s*\\(") == null) {                                        // not a method call
286                    final String propertyName = member;
287                    final Node target = node;
288
289                    // Handle "class" when preceded by a class name
290                    if (cl != null && propertyName.equals("class")) {
291                        node = new LiteralNode(cl);
292                        break;
293                    }
294
295                    // Handle static fields
296                    if (cl != null) {
297
298                        // Get static field
299                        final Field field;
300                        try {
301                            field = cl.getField(propertyName);
302                        } catch (NoSuchFieldException e) {
303                            throw new ParseException(ctx, "class `" + cl.getName() + "' has no field named `"
304                              + propertyName + "'", e);
305                        }
306                        if ((field.getModifiers() & Modifier.STATIC) == 0) {
307                            throw new ParseException(ctx, "field `" + propertyName + "' in class `" + cl.getName()
308                              + "' is not a static field");
309                        }
310
311                        // Return node accessing it
312                        node = new ConstNode(new StaticFieldValue(field));
313                        break;
314                    }
315
316                    // Must be object property access
317                    node = new Node() {
318                        @Override
319                        public Value evaluate(ParseSession session) {
320                            return BaseExprParser.this.evaluateProperty(session, target.evaluate(session), propertyName);
321                        }
322                    };
323                    break;
324                }
325
326                // Handle method call
327                node = cl != null ?
328                  new MethodInvokeNode(cl, member, BaseExprParser.parseParams(session, ctx, complete)) :
329                  new MethodInvokeNode(node, member, BaseExprParser.parseParams(session, ctx, complete));
330                break;
331            }
332            case "[":
333            {
334                this.spaceParser.parse(ctx, complete);
335                final Node index = AssignmentExprParser.INSTANCE.parse(session, ctx, complete);
336                this.spaceParser.parse(ctx, complete);
337                if (!ctx.tryLiteral("]"))
338                    throw new ParseException(ctx).addCompletion("] ");
339                final Node target = node;
340                node = new Node() {
341                    @Override
342                    public Value evaluate(ParseSession session) {
343                        return Op.ARRAY_ACCESS.apply(session, target.evaluate(session), index.evaluate(session));
344                    }
345                };
346                break;
347            }
348            case "++":
349                node = this.createPostcrementNode("increment", node, true);
350                break;
351            case "--":
352                node = this.createPostcrementNode("decrement", node, false);
353                break;
354            default:
355                throw new RuntimeException("internal error: " + opsym);
356            }
357        }
358    }

360    // Parse array literal
361    private List<?> parseArrayLiteral(ParseSession session, ParseContext ctx, boolean complete, Class<?> elemType) {
362        final ArrayList<Object> list = new ArrayList<>();
363        this.spaceParser.parse(ctx, complete);
364        if (!ctx.tryLiteral("{"))
365            throw new ParseException(ctx).addCompletion("{ ");
366        this.spaceParser.parse(ctx, complete);
367        if (ctx.tryLiteral("}"))
368            return list;
369        while (true) {
370            list.add(elemType.isArray() ?
371              this.parseArrayLiteral(session, ctx, complete, elemType.getComponentType()) :
372              ExprParser.INSTANCE.parse(session, ctx, complete));
373            ctx.skipWhitespace();
374            if (ctx.tryLiteral("}"))
375                break;
376            if (!ctx.tryLiteral(","))
377                throw new ParseException(ctx, "expected `,'").addCompletion(", ");
378            this.spaceParser.parse(ctx, complete);
379        }
380        return list;
381    }

383    // Get the array class with the given base type and dimensions
384    private Class<?> getArrayClass(Class<?> base, int dimensions) {
385        if (dimensions == 0)
386            return base;
387        final String suffix = base.isArray() ? base.getName() :
388          base.isPrimitive() ? "" + Primitive.get(base).getLetter() : "L" + base.getName() + ";";
389        final StringBuilder buf = new StringBuilder(dimensions + suffix.length());
390        while (dimensions-- > 0)
391            buf.append('[');
392        buf.append(suffix);
393        final String className = buf.toString();
394        try {
395            return Class.forName(className, false, Thread.currentThread().getContextClassLoader());
396        } catch (Exception e) {
397            throw new EvalException("can't load array class `" + className + "'");
398        }
399    }

401    // Parse method parameters; we assume opening `(' has just been parsed
402    static List<Node> parseParams(ParseSession session, ParseContext ctx, boolean complete) {
403        final ArrayList<Node> params = new ArrayList<Node>();
404        final SpaceParser spaceParser = new SpaceParser();
405        spaceParser.parse(ctx, complete);
406        if (ctx.tryLiteral(")"))
407            return params;
408        while (true) {
409            params.add(ExprParser.INSTANCE.parse(session, ctx, complete));
410            spaceParser.parse(ctx, complete);
411            if (ctx.tryLiteral(")"))
412                break;
413            if (!ctx.tryLiteral(","))
414                throw new ParseException(ctx, "expected `,'").addCompletion(", ");
415            spaceParser.parse(ctx, complete);
416        }
417        return params;
418    }

420    private Value evaluateProperty(ParseSession session, Value value, final String name) {
421
422        // Evaluate target
423        final Object target = value.checkNotNull(session, "property `" + name + "' access");
424        final Class<?> cl = target.getClass();
425
426        // Handle properties of database objects (i.e., database fields)
427        if (session.hasJSimpleDB() && target instanceof JObject) {
428
429            // Get object and ID
430            final JObject jobj = (JObject)target;
431            final ObjId id = jobj.getObjId();
432
433            // Resolve JField
434            JField jfield0;
435            try {
436                jfield0 = ParseUtil.resolveJField(session, id, name);
437            } catch (IllegalArgumentException e) {
438                jfield0 = null;
439            }
440            final JField jfield = jfield0;
441
442            // Return value reflecting the field if the field was found
443            if (jfield instanceof JSimpleField)
444                return new JSimpleFieldValue(jobj, (JSimpleField)jfield);
445            else if (jfield != null)
446                return new JFieldValue(jobj, jfield);
447        } else if (!session.hasJSimpleDB() && target instanceof ObjId) {
448            final ObjId id = (ObjId)target;
449
450            // Resolve field
451            org.jsimpledb.core.Field<?> field0;
452            try {
453                field0 = ParseUtil.resolveField(session, id, name);
454            } catch (IllegalArgumentException e) {
455                field0 = null;
456            }
457            final org.jsimpledb.core.Field<?> field = field0;
458
459            // Return value reflecting the field if the field was found
460            if (field instanceof SimpleField)
461                return new SimpleFieldValue(id, (SimpleField<?>)field);
462            else if (field != null)
463                return new FieldValue(id, field);
464        }
465
466        // Try bean property accessed via bean methods
467        final BeanInfo beanInfo;
468        try {
469            beanInfo = Introspector.getBeanInfo(cl);
470        } catch (IntrospectionException e) {
471            throw new EvalException("error introspecting class `" + cl.getName() + "': " + e, e);
472        }
473        for (PropertyDescriptor propertyDescriptor : beanInfo.getPropertyDescriptors()) {
474            if (propertyDescriptor instanceof IndexedPropertyDescriptor)
475                continue;
476            if (!propertyDescriptor.getName().equals(name))
477                continue;
478            final Method getter = this.makeAccessible(propertyDescriptor.getReadMethod());
479            final Method setter = this.makeAccessible(propertyDescriptor.getWriteMethod());
480            if (getter != null && setter != null)
481                return new MutableBeanPropertyValue(target, propertyDescriptor.getName(), getter, setter);
482            else if (getter != null)
483                return new BeanPropertyValue(target, propertyDescriptor.getName(), getter);
484        }
485
486        // Try instance field
487        /*final*/ Field javaField;
488        try {
489            javaField = cl.getField(name);
490        } catch (NoSuchFieldException e) {
491            javaField = null;
492        }
493        if (javaField != null)
494            return new ObjectFieldValue(target, javaField);
495
496        // Try array.length
497        if (target.getClass().isArray() && name.equals("length"))
498            return new ConstValue(Array.getLength(target));
499
500        // Not found
501        throw new EvalException("property `" + name + "' not found in " + cl);
502    }

504    // Workaround the problem where non-public class C implements public method M of interface I.
505    // In that case, invoking method C.M results in IllegalAccessException; instead, you have to invoke I.M.
506    private Method makeAccessible(Method method) {
507        if (method == null)
508            return null;
509        Class<?> cl = method.getDeclaringClass();
510        if ((cl.getModifiers() & Modifier.PUBLIC) != 0)
511            return method;
512        do {
513            for (Class<?> iface : cl.getInterfaces()) {
514                try {
515                    return iface.getMethod(method.getName(), method.getParameterTypes());
516                } catch (NoSuchMethodException e) {
517                    continue;
518                }
519            }
520        } while ((cl = cl.getSuperclass()) != null);
521        return null;
522    }

524    private Node createPostcrementNode(final String operation, final Node node, final boolean increment) {
525        return new Node() {
526            @Override
527            public Value evaluate(ParseSession session) {
528                return node.evaluate(session).xxcrement(session, "post-" + operation, increment);
529            }
530        };
531    }
