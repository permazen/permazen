006package org.jsimpledb.parse.expr;
007
008import org.jsimpledb.parse.ParseSession;
009
010/**
011 * The product of a parse operation, which is capable of producing a {@link Value} when evaluated within a transaction.
012 */
013public interface Node {
014
015    /**
016     * Evaluate this node. There will be a transaction open.
017     *
018     * @param session parse session
019     * @return result of node evaluation
020     */
021    Value evaluate(ParseSession session);
022}
