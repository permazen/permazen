006package org.jsimpledb.parse;
007
008import java.util.ArrayList;
009import java.util.Arrays;
010import java.util.List;
011
012@SuppressWarnings("serial")
013public class ParseException extends RuntimeException {
014
015    private final ArrayList<String> completions = new ArrayList<>();
016    private final ParseContext ctx;
017
018    public ParseException(ParseContext ctx) {
019        this(ctx, null);
020    }
021
022    public ParseException(ParseContext ctx, String message) {
023        this(ctx, message, null);
024    }
025
026    public ParseException(ParseContext ctx, String message, Throwable cause) {
027        super((message != null ? message : "parse error") + " at "
028          + (ctx.isEOF() ? "end of input" : "`" + ParseUtil.truncate(ctx.getInput(), 20) + "'"), cause);
029        this.ctx = ctx;
030    }
031
032    public ParseContext getParseContext() {
033        return this.ctx;
034    }
035
036    public List<String> getCompletions() {
037        return this.completions;
038    }
039
040    public ParseException addCompletion(String completion) {
041        return this.addCompletions(Arrays.asList(completion));
042    }
043
044    public ParseException addCompletions(String... completions) {
045        return this.addCompletions(Arrays.asList(completions));
046    }
047
048    public ParseException addCompletions(Iterable<String> completions) {
049        for (String completion : completions)
050            this.completions.add(completion);
051        return this;
052    }
053}
