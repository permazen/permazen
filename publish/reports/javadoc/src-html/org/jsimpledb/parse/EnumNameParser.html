006package org.jsimpledb.parse;
007
008import com.google.common.base.Function;
009import com.google.common.collect.Lists;
010
011import java.util.regex.Matcher;
012
013import org.dellroad.stuff.java.EnumUtil;
014
015public class EnumNameParser<T extends Enum<T>> implements Parser<T> {
016
017    private final Class<T> type;
018    private final boolean lowerCase;
019    private final Function<T, String> nameFunction = new Function<T, String>() {
020        @Override
021        public String apply(T value) {
022            String name = value.name();
023            if (EnumNameParser.this.lowerCase)
024                name = name.toLowerCase();
025            return name;
026        }
027    };
028
029    public EnumNameParser(Class<T> type) {
030        this(type, true);
031    }
032
033    public EnumNameParser(Class<T> type, boolean lowerCase) {
034        if (type == null)
035            throw new IllegalArgumentException("null type");
036        this.type = type;
037        this.lowerCase = lowerCase;
038    }
039
040    @Override
041    public T parse(ParseSession session, ParseContext ctx, boolean complete) {
042
043        // Get name
044        final Matcher matcher = ctx.tryPattern("[^\\s;]*");
045        if (matcher == null && ctx.isEOF())
046            throw new ParseException(ctx).addCompletions(Lists.transform(EnumUtil.getValues(this.type), this.nameFunction));
047        final String name = matcher.group();
048
049        // Match name
050        for (T value : EnumUtil.getValues(this.type)) {
051            final String valueName = this.nameFunction.apply(value);
052            if (name.equals(valueName))
053                return value;
054        }
055
056        // Not found
057        throw new ParseException(ctx, "unknown value `" + name + "'").addCompletions(
058            ParseUtil.complete(Lists.transform(EnumUtil.getValues(this.type), this.nameFunction), name));
059    }
060}
