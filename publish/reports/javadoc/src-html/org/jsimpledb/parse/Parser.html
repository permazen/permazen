006package org.jsimpledb.parse;
007
008/**
009 * Generic parsing interface.
010 *
011 * @param <T> parsed value type
012 */
013public interface Parser<T> {
014
015    /**
016     * Parse text from the given parse context.
017     *
018     * <p>
019     * Generally speaking, this method may assume that any whitespace allowed before the item
020     * being parsed has already been skipped over (that's a matter for the containing parser).
021     * </p>
022     *
023     * @param session parse session
024     * @param ctx input to parse
025     * @param complete false if parse is "for real", true if only for tab completion calculation
026     * @return parsed value
027     * @throws ParseException if parse fails, or if {@code complete} is true and there are valid completions
028     */
029    T parse(ParseSession session, ParseContext ctx, boolean complete);
030}
