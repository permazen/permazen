107    /**
108     * Register the standard built-in functions such as {@code all()}, {@code foreach()}, etc.
109     */
110    public void registerStandardFunctions() {
111
112        // We don't use AnnotatedClassScanner here to avoid having a dependency on the spring classes
113        this.registerFunction(AllFunction.class);
114        this.registerFunction(ConcatFunction.class);
115        this.registerFunction(CountFunction.class);
116        this.registerFunction(CreateFunction.class);
117        this.registerFunction(FilterFunction.class);
118        this.registerFunction(ForEachFunction.class);
119        if (this.hasJSimpleDB())
120            this.registerFunction(InvertFunction.class);
121        this.registerFunction(LimitFunction.class);
122        this.registerFunction(ListFunction.class);
123        this.registerFunction(QueryCompositeIndexFunction.class);
124        this.registerFunction(QueryIndexFunction.class);
125        this.registerFunction(QueryListElementIndexFunction.class);
126        this.registerFunction(QueryMapValueIndexFunction.class);
127        this.registerFunction(QueryVersionFunction.class);
128        this.registerFunction(TransformFunction.class);
129        this.registerFunction(UpgradeFunction.class);
130        this.registerFunction(VersionFunction.class);
131    }
132
133    /**
134     * Create an instance of the specified class and register it as an {@link AbstractFunction}.
135     * as appropriate. The class must have a public constructor taking either a single {@link ParseSession} parameter
136     * or no parameters; they will be tried in that order.
137     *
138     * @param cl function class
139     * @throws IllegalArgumentException if {@code cl} has no suitable constructor
140     * @throws IllegalArgumentException if {@code cl} instantiation fails
141     * @throws IllegalArgumentException if {@code cl} does not subclass {@link AbstractFunction}
142     */
143    public void registerFunction(Class<?> cl) {
144        if (!AbstractFunction.class.isAssignableFrom(cl))
145            throw new IllegalArgumentException(cl + " does not subclass " + AbstractFunction.class.getName());
146        final AbstractFunction function = this.instantiate(cl.asSubclass(AbstractFunction.class));
147        this.functions.put(function.getName(), function);
148    }
149
150    /**
151     * Instantiate an instance of the given class.
152     * The class must have a public constructor taking either a single {@link ParseSession} parameter
153     * or no parameters; they will be tried in that order.
154     */
155    private <T> T instantiate(Class<T> cl) {
156        Throwable failure;
157        try {
158            return cl.getConstructor(ParseSession.class).newInstance(this);
159        } catch (NoSuchMethodException e) {
160            try {
161                return cl.getConstructor().newInstance();
162            } catch (NoSuchMethodException e2) {
163                throw new IllegalArgumentException("no suitable constructor found in class " + cl.getName());
164            } catch (Exception e2) {
165                failure = e2;
166            }
167        } catch (Exception e) {
168            failure = e;
169        }
170        if (failure instanceof InvocationTargetException)
171            failure = failure.getCause();
172        throw new IllegalArgumentException("unable to instantiate class " + cl.getName() + ": " + failure, failure);
173    }

175// Class name resolution
176
177    /**
178     * Resolve a class name against this instance's currently configured class imports.
179     *
180     * @param name class name
181     * @return resolved class, or null if not found
182     */
183    public Class<?> resolveClass(final String name) {
184        final int firstDot = name.indexOf('.');
185        final String firstPart = firstDot != -1 ? name.substring(0, firstDot - 1) : name;
186        final ArrayList<String> packages = new ArrayList<>(this.imports.size() + 1);
187        packages.add(null);
188        packages.addAll(this.imports);
189        for (String pkg : packages) {
190
191            // Get absolute class name
192            String className;
193            if (pkg == null)
194                className = name;
195            else if (pkg.endsWith(".*"))
196                className = pkg.substring(0, pkg.length() - 1) + name;
197            else {
198                if (!firstPart.equals(pkg.substring(pkg.lastIndexOf('.') + 1, pkg.length() - 2)))
199                    continue;
200                className = pkg.substring(0, pkg.length() - 2 - firstPart.length()) + name;
201            }
202
203            // Try package vs. nested classes
204            while (true) {
205                try {
206                    return Class.forName(className, false, Thread.currentThread().getContextClassLoader());
207                } catch (ClassNotFoundException e) {
208                    // not found
209                }
210                final int lastDot = className.lastIndexOf('.');
211                if (lastDot == -1)
212                    break;
213                className = className.substring(0, lastDot) + "$" + className.substring(lastDot + 1);
214            }
215        }
216        return null;
217    }
218
219    /**
220     * Relativize the given class's name, so that it is as short as possible given the configured imports.
221     * For example, for class {@link String} this will return {@code String}, but for class {@link ArrayList}
222     * this will return {@code java.util.ArrayList} unless {@code java.util.*} has been imported.
223     *
224     * @param klass class whose name to relativize
225     * @return relativized class name
226     * @throws IllegalArgumentException if {@code klass} is null
227     */
228    public String relativizeClassName(Class<?> klass) {
229        if (klass == null)
230            throw new IllegalArgumentException("null klass");
231        final String name = klass.getName();
232        for (int pos = name.lastIndexOf('.'); pos > 0; pos = name.lastIndexOf('.', pos - 1)) {
233            final String shortName = name.substring(pos + 1);
234            if (this.resolveClass(shortName) == klass)
235                return shortName;
236        }
237        return klass.getName();
238    }

240// Action
241
242    /**
243     * Perform the given action. This is a convenience method, equivalent to: {@code perform(null, action)}
244     *
245     * @param action action to perform
246     * @return true if {@code action} completed successfully, false if the transaction could not be created
247     *  or {@code action} threw an exception
248     * @throws IllegalArgumentException if {@code action} is null
249     */
250    public boolean perform(Action action) {
251        return this.perform(null, action);
252    }
253
254    /**
255     * Perform the given action within the given existing transaction, if any, otherwise within a new transaction.
256     * If {@code tx} is not null, it will used and left open when this method returns. Otherwise,
257     * if there is already an open transaction associated with this instance, it will be used;
258     * otherwise, a new transaction is created for the duration of {@code action} and then committed.
259     *
260     * <p>
261     * If {@code tx} is not null and there is already an open transaction associated with this instance and they
262     * are not the same transaction, an {@link IllegalStateException} is thrown.
263     * </p>
264     *
265     * @param tx transaction in which to perform the action, or null to create a new one (if necessary)
266     * @param action action to perform
267     * @return true if {@code action} completed successfully, false if the transaction could not be created
268     *  or {@code action} threw an exception
269     * @throws IllegalStateException if {@code tx} conflict with the already an open transaction associated with this instance
270     * @throws IllegalArgumentException if {@code action} is null
271     */
272    public boolean perform(Transaction tx, final Action action) {
273        return this.perform(tx, new Session.Action() {
274            @Override
275            public void run(Session session) throws Exception {
276                action.run((ParseSession)session);
277            }
278        });
279    }
280
281    /**
282     * Callback interface used by {@link ParseSession#perform ParseSession.perform()}.
283     */
284    public interface Action {
285
286        /**
287         * Perform some action using the given {@link ParseSession} while a transaction is open.
288         *
289         * @param session session with open transaction
290         * @throws Exception if an error occurs
291         */
292        void run(ParseSession session) throws Exception;
293    }
