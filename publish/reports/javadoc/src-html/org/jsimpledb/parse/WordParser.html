006package org.jsimpledb.parse;
007
008import java.util.Collection;
009import java.util.TreeSet;
010import java.util.regex.Matcher;
011
012/**
013 * Parses a word (one or more non-whitespace characters).
014 */
015public class WordParser implements Parser<String> {
016
017    private final Collection<String> words;
018    private final String description;
019
020    /**
021     * Constructor for when there's a fixed set of possibilities.
022     *
023     * @param words words to look for
024     * @param description what to call words (e.g., "command")
025     * @throws IllegalArgumentException if {@code words} is null
026     * @throws IllegalArgumentException if {@code description} is null
027     */
028    public WordParser(Collection<String> words, String description) {
029        if (words == null)
030            throw new IllegalArgumentException("null words");
031        if (description == null)
032            throw new IllegalArgumentException("null description");
033        this.words = new TreeSet<>(words);
034        this.description = description;
035    }
036
037    /**
038     * Constructor when any word is acceptable.
039     *
040     * @throws IllegalArgumentException if {@code words} is null
041     * @param description what to call words (e.g., "command")
042     * @throws IllegalArgumentException if {@code description} is null
043     */
044    public WordParser(String description) {
045        if (description == null)
046            throw new IllegalArgumentException("null description");
047        this.words = null;
048        this.description = description;
049    }
050
051    /**
052     * Get the set of valid words, if there is such a set.
053     *
054     * <p>
055     * The implementation in {@link WordParser} returns the collection provided to the constructor, if any.
056     * </p>
057     *
058     * @return collection of valid words, or null to not place any restriction
059     */
060    protected Collection<String> getWords() {
061        return this.words;
062    }
063
064    @Override
065    public String parse(ParseSession session, ParseContext ctx, boolean complete) {
066
067        // Get word
068        final Matcher matcher = ctx.tryPattern("[^\\s;]*");
069        if (matcher == null)
070            throw new ParseException(ctx);
071        final String word = matcher.group();
072
073        // Check word
074        final Collection<String> validWords = this.getWords();
075        if (validWords != null) {
076            final TreeSet<String> sortedWords = new TreeSet<>(validWords);
077            if (!sortedWords.contains(word)) {
078                throw new ParseException(ctx, "unknown " + this.description + " `" + word + "'")
079                  .addCompletions(ParseUtil.complete(sortedWords, word));
080            }
081        } else if (word.length() == 0)
082            throw new ParseException(ctx, "missing " + this.description);
083
084        // Done
085        return word;
086    }
087}
