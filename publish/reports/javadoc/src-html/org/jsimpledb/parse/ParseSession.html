120    /**
121     * Register the standard built-in functions such as {@code all()}, {@code foreach()}, etc.
122     */
123    public void registerStandardFunctions() {
124
125        // We don't use AnnotatedClassScanner here to avoid having a dependency on the spring classes
126        final Class<?>[] functionClasses = new Class<?>[] {
127            AllFunction.class,
128            ConcatFunction.class,
129            CountFunction.class,
130            CreateFunction.class,
131            FilterFunction.class,
132            ForEachFunction.class,
133            InvertFunction.class,
134            LimitFunction.class,
135            ListFunction.class,
136            QueryCompositeIndexFunction.class,
137            QueryIndexFunction.class,
138            QueryListElementIndexFunction.class,
139            QueryMapValueIndexFunction.class,
140            QueryVersionFunction.class,
141            TransformFunction.class,
142            UpgradeFunction.class,
143            VersionFunction.class,
144        };
145        for (Class<?> cl : functionClasses)
146            this.registerFunction(cl);
147    }
148
149    /**
150     * Create an instance of the specified class and register it as an {@link AbstractFunction}.
151     * as appropriate. The class must have a public constructor taking either a single {@link ParseSession} parameter
152     * or no parameters; they will be tried in that order.
153     *
154     * @param cl function class
155     * @return the newly instantiated function
156     * @throws IllegalArgumentException if {@code cl} has no suitable constructor
157     * @throws IllegalArgumentException if {@code cl} instantiation fails
158     * @throws IllegalArgumentException if {@code cl} does not subclass {@link AbstractFunction}
159     */
160    public AbstractFunction registerFunction(Class<?> cl) {
161        if (!AbstractFunction.class.isAssignableFrom(cl))
162            throw new IllegalArgumentException(cl + " does not subclass " + AbstractFunction.class.getName());
163        final AbstractFunction function = this.instantiate(cl.asSubclass(AbstractFunction.class));
164        try {
165            function.getSessionModes();
166        } catch (UnsupportedOperationException e) {
167            throw new IllegalArgumentException(cl + " does not know it's supported session modes", e);
168        }
169        this.functions.put(function.getName(), function);
170        return function;
171    }
172
173    /**
174     * Instantiate an instance of the given class.
175     * The class must have a public constructor taking either a single {@link ParseSession} parameter
176     * or no parameters; they will be tried in that order.
177     */
178    private <T> T instantiate(Class<T> cl) {
179        Throwable failure;
180        try {
181            return cl.getConstructor(ParseSession.class).newInstance(this);
182        } catch (NoSuchMethodException e) {
183            try {
184                return cl.getConstructor().newInstance();
185            } catch (NoSuchMethodException e2) {
186                throw new IllegalArgumentException("no suitable constructor found in class " + cl.getName());
187            } catch (Exception e2) {
188                failure = e2;
189            }
190        } catch (Exception e) {
191            failure = e;
192        }
193        if (failure instanceof InvocationTargetException)
194            failure = failure.getCause();
195        throw new IllegalArgumentException("unable to instantiate class " + cl.getName() + ": " + failure, failure);
196    }

198// Class name resolution
199
200    /**
201     * Resolve a class name against this instance's currently configured class imports.
202     *
203     * @param name class name
204     * @return resolved class, or null if not found
205     */
206    public Class<?> resolveClass(final String name) {
207        final int firstDot = name.indexOf('.');
208        final String firstPart = firstDot != -1 ? name.substring(0, firstDot - 1) : name;
209        final ArrayList<String> packages = new ArrayList<>(this.imports.size() + 1);
210        packages.add(null);
211        packages.addAll(this.imports);
212        for (String pkg : packages) {
213
214            // Get absolute class name
215            String className;
216            if (pkg == null)
217                className = name;
218            else if (pkg.endsWith(".*"))
219                className = pkg.substring(0, pkg.length() - 1) + name;
220            else {
221                if (!firstPart.equals(pkg.substring(pkg.lastIndexOf('.') + 1, pkg.length() - 2)))
222                    continue;
223                className = pkg.substring(0, pkg.length() - 2 - firstPart.length()) + name;
224            }
225
226            // Try package vs. nested classes
227            while (true) {
228                try {
229                    return Class.forName(className, false, Thread.currentThread().getContextClassLoader());
230                } catch (ClassNotFoundException e) {
231                    // not found
232                }
233                final int lastDot = className.lastIndexOf('.');
234                if (lastDot == -1)
235                    break;
236                className = className.substring(0, lastDot) + "$" + className.substring(lastDot + 1);
237            }
238        }
239        return null;
240    }

242    /**
243     * Relativize the given class's name, so that it is as short as possible given the configured imports.
244     * For example, for class {@link String} this will return {@code String}, but for class {@link ArrayList}
245     * this will return {@code java.util.ArrayList} unless {@code java.util.*} has been imported.
246     *
247     * @param klass class whose name to relativize
248     * @return relativized class name
249     * @throws IllegalArgumentException if {@code klass} is null
250     */
251    public String relativizeClassName(Class<?> klass) {
252        Preconditions.checkArgument(klass != null, "null klass");
253        final String name = klass.getName();
254        for (int pos = name.lastIndexOf('.'); pos > 0; pos = name.lastIndexOf('.', pos - 1)) {
255            final String shortName = name.substring(pos + 1);
256            if (this.resolveClass(shortName) == klass)
257                return shortName;
258        }
259        return klass.getName();
260    }

264    /**
265     * Perform the given action in the context of this session.
266     *
267     * <p>
268     * This is a {@link ParseSession}-specific overload of
269     * {@link Session#performSessionAction Session.performSessionAction()}; see that method for details.
270     *
271     * @param action action to perform, possibly within a transaction
272     * @return true if {@code action} completed successfully, false if a transaction could not be created
273     *  or {@code action} threw an exception
274     * @throws IllegalArgumentException if {@code action} is null
275     */
276    public boolean performParseSessionAction(final Action action) {
277        return this.performSessionAction(this.wrap(action));
278    }

280    /**
281     * Associate the current {@link org.jsimpledb.JTransaction} with this instance, if not already associated,
282     * while performing the given action.
283     *
284     * <p>
285     * This is a {@link ParseSession}-specific overload of
286     * {@link Session#performSessionActionWithCurrentTransaction Session.performSessionActionWithCurrentTransaction()};
287     * see that method for details.
288     *
289     * @param action action to perform
290     * @return true if {@code action} completed successfully, false if {@code action} threw an exception
291     * @throws IllegalStateException if there is a different open transaction already associated with this instance
292     * @throws IllegalStateException if this instance is not in mode {@link org.jsimpledb.SessionMode#JSIMPLEDB}
293     * @throws IllegalArgumentException if {@code action} is null
294     */
295    public boolean performParseSessionActionWithCurrentTransaction(final Action action) {
296        return this.performSessionActionWithCurrentTransaction(this.wrap(action));
297    }
298
299    private Session.Action wrap(final Action action) {
300        return action instanceof TransactionalAction ?
301          new Session.TransactionalAction() {
302            @Override
303            public void run(Session session) throws Exception {
304                action.run((ParseSession)session);
305            }
306          } :
307          new Session.Action() {
308            @Override
309            public void run(Session session) throws Exception {
310                action.run((ParseSession)session);
311            }
312          };
313    }

315    /**
316     * Callback interface used by {@link ParseSession#performParseSessionAction ParseSession.performParseSessionAction()}
317     * and {@link ParseSession#performParseSessionActionWithCurrentTransaction
318     *  ParseSession.performParseSessionActionWithCurrentTransaction()}.
319     */
320    public interface Action {
321
322        /**
323         * Perform some action using the given {@link ParseSession} while a transaction is open.
324         *
325         * @param session session with open transaction
326         * @throws Exception if an error occurs
327         */
328        void run(ParseSession session) throws Exception;
329    }
