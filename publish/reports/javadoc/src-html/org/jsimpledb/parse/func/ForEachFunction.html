006package org.jsimpledb.parse.func;
007
008import java.util.Map;
009
010import org.jsimpledb.parse.ParseSession;
011import org.jsimpledb.parse.expr.ConstValue;
012import org.jsimpledb.parse.expr.EvalException;
013import org.jsimpledb.parse.expr.Value;
014
015@Function
016public class ForEachFunction extends ApplyExprFunction {
017
018    public ForEachFunction() {
019        super("foreach");
020    }
021
022    @Override
023    public String getHelpSummary() {
024        return "evaluates an expression for each item in a collection";
025    }
026
027    @Override
028    public String getUsage() {
029        return "foreach(items, variable, expression)";
030    }
031
032    @Override
033    public String getHelpDetail() {
034        return "Iterates over an Iterable, for each item assigning the item to the specified variable and evaluating"
035          + " the specified expression. Maps are also supported, in which case the map's entrySet() is iterated.";
036    }
037
038    @Override
039    protected Value apply(ParseSession session, ParamInfo params) {
040
041        // Evaluate items
042        Object items = params.getItems().evaluate(session).checkNotNull(session, "foreach()");
043
044        // Iterate over items and evaluate expression
045        if (items instanceof Map)
046            items = ((Map<?, ?>)items).entrySet();
047        if (!(items instanceof Iterable))
048            throw new EvalException("invalid foreach() operation over non-Iterable object of type " + items.getClass().getName());
049        for (Object item : ((Iterable<?>)items))
050            this.evaluate(session, params.getVariable(), new ConstValue(item), params.getExpr());
051
052        // Done
053        return Value.NO_VALUE;
054    }
055}
