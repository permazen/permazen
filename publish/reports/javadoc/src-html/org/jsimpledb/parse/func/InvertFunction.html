006package org.jsimpledb.parse.func;
007
008import com.google.common.collect.Iterables;
009
010import java.util.regex.Matcher;
011
012import org.jsimpledb.JClass;
013import org.jsimpledb.JObject;
014import org.jsimpledb.JTransaction;
015import org.jsimpledb.core.UnknownTypeException;
016import org.jsimpledb.parse.ParseContext;
017import org.jsimpledb.parse.ParseException;
018import org.jsimpledb.parse.ParseSession;
019import org.jsimpledb.parse.expr.AbstractValue;
020import org.jsimpledb.parse.expr.EvalException;
021import org.jsimpledb.parse.expr.ExprParser;
022import org.jsimpledb.parse.expr.IdentNode;
023import org.jsimpledb.parse.expr.Node;
024import org.jsimpledb.parse.expr.Value;
025import org.jsimpledb.parse.util.ParseCastFunction;
026
027@Function(worksInCoreAPIMode = false)
028public class InvertFunction extends AbstractFunction {
029
030    public InvertFunction() {
031        super("invert");
032    }
033
034    @Override
035    public String getHelpSummary() {
036        return "invert a path of references";
037    }
038
039    @Override
040    public String getUsage() {
041        return "invert(type.fields..., objects)";
042    }
043
044    @Override
045    public String getHelpDetail() {
046        return "The invert() function inverts a reference path. The first parameter is the reference path, in the form"
047          + " of type name followed by a dot-separated list of one or more reference fields. The second parameter is a"
048          + " collection of target objects. The return value is the set of all objects that refer to any of the target"
049          + " objects through the path of references. This function does not work when running in core API mode.";
050    }
051
052    @Override
053    public InvertInfo parseParams(final ParseSession session, final ParseContext ctx, final boolean complete) {
054
055        // Sanity check
056        if (!session.hasJSimpleDB())
057            throw new ParseException(ctx, "invert() does not work in core API mode");
058
059        // Parse entire reference path
060        final int mark = ctx.getIndex();
061        final Matcher matcher = ctx.tryPattern(IdentNode.NAME_PATTERN + "\\s*(\\.\\s*" + IdentNode.NAME_PATTERN + "\\s*)*");
062        if (matcher == null)
063            throw new ParseException(ctx, "invalid reference path");
064        final String path = matcher.group().replaceAll("\\s+", "");
065        final int firstDot = path.indexOf('.');
066        if (firstDot == -1) {
067            ctx.setIndex(mark);
068            throw new ParseException(ctx, "invalid reference path");
069        }
070        final String typeName = path.substring(0, firstDot);
071        final String fieldPath = path.substring(firstDot + 1);
072
073        // Find start type
074        final JClass<?> jclass;
075        try {
076            jclass = session.getJSimpleDB().getJClass(session.getNameIndex().getSchemaObjectType(typeName).getStorageId());
077        } catch (UnknownTypeException e) {
078            ctx.setIndex(mark);
079            throw new ParseException(ctx, "invalid type `" + typeName + "': " + e.getMessage(), e);
080        }
081
082        // Verify reference path is valid (although we don't verify the last field in the path is a reference field here)
083        try {
084            session.getJSimpleDB().parseReferencePath(jclass.getType(), fieldPath);
085        } catch (IllegalArgumentException e) {
086            ctx.setIndex(mark);
087            throw new ParseException(ctx, "invalid path `" + fieldPath + "': " + e.getMessage(), e);
088        }
089
090        // Parase expression
091        ctx.skipWhitespace();
092        if (!ctx.tryLiteral(","))
093            throw new ParseException(ctx, "expected `,'").addCompletion(", ");
094        this.spaceParser.parse(ctx, complete);
095        final Node expr = new ExprParser().parse(session, ctx, complete);
096
097        // Finish parse
098        ctx.skipWhitespace();
099        if (!ctx.tryLiteral(")"))
100            throw new ParseException(ctx, "expected `)'").addCompletion(") ");
101
102        // Done
103        return new InvertInfo(jclass, fieldPath, expr);
104    }
105
106    @Override
107    public Value apply(ParseSession session, Object params) {
108        final InvertInfo info = (InvertInfo)params;
109        return new AbstractValue() {
110            @Override
111            public Object get(ParseSession session) {
112
113                // Evaluate expression
114                final Iterable<?> items = info.getExpr().evaluate(session).checkType(session, "invert()", Iterable.class);
115                final Iterable<JObject> jobjs = Iterables.transform(items, new ParseCastFunction<JObject>(JObject.class));
116
117                // Invert references to iterated objects
118                try {
119                    return JTransaction.getCurrent().invertReferencePath(info.getJClass().getType(), info.getFieldPath(), jobjs);
120                } catch (Exception e) {
121                    throw new EvalException("invert() failed: " + e, e);
122                }
123            }
124        };
125    }
126
127// InvertInfo
128
129    private static class InvertInfo {
130
131        private final JClass<?> jclass;
132        private final String fieldPath;
133        private final Node expr;
134
135        InvertInfo(JClass<?> jclass, String fieldPath, Node expr) {
136            this.jclass = jclass;
137            this.fieldPath = fieldPath;
138            this.expr = expr;
139        }
140
141        public JClass<?> getJClass() {
142            return this.jclass;
143        }
144
145        public String getFieldPath() {
146            return this.fieldPath;
147        }
148
149        public Node getExpr() {
150            return this.expr;
151        }
152    }
153}
