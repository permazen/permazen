006package org.jsimpledb.parse.func;
007
008import java.util.ArrayList;
009
010import org.jsimpledb.parse.ParseContext;
011import org.jsimpledb.parse.ParseException;
012import org.jsimpledb.parse.ParseSession;
013import org.jsimpledb.parse.SpaceParser;
014import org.jsimpledb.parse.expr.ExprParser;
015import org.jsimpledb.parse.expr.Node;
016import org.jsimpledb.parse.expr.Value;
017import org.slf4j.Logger;
018import org.slf4j.LoggerFactory;
019
020/**
021 * Superclass of all {@link ParseSession} functions.
022 *
023 * @see Function
024 */
025public abstract class AbstractFunction {
026
027    protected final Logger log = LoggerFactory.getLogger(this.getClass());
028    protected final SpaceParser spaceParser = new SpaceParser();
029    protected final String name;
030
031// Constructors
032
033    /**
034     * Constructor.
035     *
036     * @param name function name
037     */
038    protected AbstractFunction(String name) {
039        if (name == null)
040            throw new IllegalArgumentException("null name");
041        this.name = name;
042    }
043
044// Accessors
045
046    /**
047     * Get the name of this function.
048     *
049     * @return function name
050     */
051    public String getName() {
052        return this.name;
053    }
054
055    /**
056     * Get function usage string. For example: {@code pow(base, exponent)}.
057     *
058     * @return function usage string
059     */
060    public abstract String getUsage();
061
062    /**
063     * Get summarized help (typically a single line).
064     *
065     * @return one line summarized description of this function
066     */
067    public abstract String getHelpSummary();
068
069    /**
070     * Get expanded help (typically multiple lines). May refer to the {@linkplain #getUsage usage string}.
071     *
072     * <p>
073     * The implementation in {@link AbstractFunction} delegates to {@link #getHelpSummary getHelpSummary()}.
074     * </p>
075     *
076     * @return detailed description of this function
077     */
078    public String getHelpDetail() {
079        return this.getHelpSummary();
080    }
081
082// Parsing
083
084    /**
085     * Parse function parameters.
086     *
087     * <p>
088     * The {@code ctx} will be pointing at the first parameter (if any) or closing parenthesis. This method should parse
089     * (but not evaluate) function parameters up through the closing parenthesis.
090     * </p>
091     *
092     * @param session parse session
093     * @param ctx parse context
094     * @param complete false if parse is "for real", true if only for tab completion calculation
095     * @return parsed parameters object to be passed to {@link #apply apply()}
096     * @throws ParseException if parse fails, or if {@code complete} is true and there are valid completions
097     */
098    public abstract Object parseParams(ParseSession session, ParseContext ctx, boolean complete);
099
100    /**
101     * Evaluate this function. There will be a transaction open.
102     *
103     * @param session parse session
104     * @param params parsed parameters returned by {@link #parseParams parseParams()}
105     * @return value returned by this function
106     * @throws RuntimeException if there is an error
107     */
108    public abstract Value apply(ParseSession session, Object params);
109
110    /**
111     * Parse some number of Java expression function arguments. We assume we have parsed the opening parenthesis,
112     * zero or more previous arguments followed by commas, and optional whitespace. This will parse through
113     * the closing parenthesis.
114     *
115     * @param session parse session
116     * @param ctx parse context
117     * @param complete false if parse is "for real", true if only for tab completion calculation
118     * @param skippedArgs the number of arguments already parsed
119     * @param minArgs minimum number of arguments
120     * @param maxArgs maximum number of arguments
121     * @return parsed expressions
122     * @throws ParseException if parse fails, or if {@code complete} is true and there are valid completions
123     */
124    public Node[] parseExpressionParams(ParseSession session, ParseContext ctx, boolean complete,
125      int skippedArgs, int minArgs, int maxArgs) {
126
127        // Parse parameters
128        final ArrayList<Node> params = new ArrayList<Node>(Math.min(maxArgs, minArgs * 2));
129        while (true) {
130            if (ctx.isEOF()) {
131                final ParseException e = new ParseException(ctx, "truncated input");
132                if (!params.isEmpty() && params.size() < minArgs)
133                    e.addCompletion(", ");
134                else if (params.size() >= minArgs)
135                    e.addCompletion(")");
136                throw e;
137            }
138            if (ctx.tryLiteral(")")) {
139                if (params.size() < minArgs) {
140                    throw new ParseException(ctx, "at least " + (skippedArgs + minArgs) + " argument(s) are required for function "
141                      + this.getName() + "()");
142                }
143                break;
144            }
145            if (params.size() >= maxArgs) {
146                throw new ParseException(ctx, "at most " + (skippedArgs + maxArgs) + " argument(s) are allowed for function "
147                  + this.getName() + "()");
148            }
149            if (!params.isEmpty()) {
150                if (!ctx.tryLiteral(",")) {
151                    throw new ParseException(ctx, "expected `,' between " + this.getName() + "() function parameters")
152                      .addCompletion(", ");
153                }
154                this.spaceParser.parse(ctx, complete);
155            }
156            params.add(ExprParser.INSTANCE.parse(session, ctx, complete));
157            ctx.skipWhitespace();
158        }
159
160        // Done
161        return params.toArray(new Node[params.size()]);
162    }
163}
