008package org.jsimpledb.parse.func;
009
010import java.util.ArrayList;
011
012import org.jsimpledb.parse.ParseContext;
013import org.jsimpledb.parse.ParseException;
014import org.jsimpledb.parse.ParseSession;
015import org.jsimpledb.parse.SpaceParser;
016import org.jsimpledb.parse.expr.ExprParser;
017import org.jsimpledb.parse.expr.Node;
018import org.jsimpledb.parse.expr.Value;
019
020/**
021 * Simplified {@link Function} implementation for when the parameters are all normal expressions.
022 */
023public abstract class SimpleFunction extends AbstractFunction {
024
025    protected final int minArgs;
026    protected final int maxArgs;
027
028    private final SpaceParser spaceParser = new SpaceParser();
029
030// Constructors
031
032    /**
033     * Constructor.
034     *
035     * @param name function name
036     * @param minArgs minimum number of arguments (inclusive)
037     * @param maxArgs maximum number of arguments (inclusive)
038     */
039    protected SimpleFunction(String name, int minArgs, int maxArgs) {
040        super(name);
041        if (minArgs < 0 || minArgs > maxArgs)
042            throw new IllegalArgumentException("invalid minArgs = " + minArgs + ", maxArgs = " + maxArgs);
043        this.minArgs = minArgs;
044        this.maxArgs = maxArgs;
045    }
046
047// Accessors
048
049    /**
050     * Get the minimum number of arguments allowed (inclusive).
051     */
052    public int getMinArgs() {
053        return this.minArgs;
054    }
055
056    /**
057     * Get the maximum number of arguments allowed (inclusive).
058     */
059    public int getMaxArgs() {
060        return this.maxArgs;
061    }
062
063// Parsing
064
065    @Override
066    public final Node[] parseParams(ParseSession session, ParseContext ctx, boolean complete) {
067
068        // Parse parameters
069        final ArrayList<Node> params = new ArrayList<Node>(Math.min(this.maxArgs, this.minArgs * 2));
070        while (true) {
071            if (ctx.isEOF()) {
072                final ParseException e = new ParseException(ctx, "truncated input");
073                if (!params.isEmpty() && params.size() < this.minArgs)
074                    e.addCompletion(", ");
075                else if (params.size() >= this.minArgs)
076                    e.addCompletion(") ");
077                throw e;
078            }
079            if (ctx.tryLiteral(")"))
080                break;
081            if (!params.isEmpty()) {
082                if (!ctx.tryLiteral(","))
083                    throw new ParseException(ctx, "expected `,' between " + this.name + "() function parameters")
084                      .addCompletion(", ");
085                this.spaceParser.parse(ctx, complete);
086            }
087            params.add(ExprParser.INSTANCE.parse(session, ctx, complete));
088            ctx.skipWhitespace();
089        }
090
091        // Check number
092        if (params.size() < this.minArgs) {
093            throw new ParseException(ctx, "at least " + this.minArgs + " argument(s) are required for function "
094              + this.getName() + "()");
095        } else if (params.size() > this.maxArgs) {
096            throw new ParseException(ctx, "at most " + this.maxArgs + " argument(s) are allowed for function "
097              + this.getName() + "()");
098        }
099
100        // Done
101        return params.toArray(new Node[params.size()]);
102    }
103
104    @Override
105    public final Value apply(ParseSession session, Object info) {
106        final Node[] params = (Node[])info;
107        final Value[] values = new Value[params.length];
108        for (int i = 0; i < params.length; i++)
109            values[i] = params[i].evaluate(session);
110        return this.apply(session, values);
111    }
112
113    /**
114     * Apply this function to the given values.
115     *
116     * @param session parse session
117     * @param params parsed parameters; will already be checked between {@link #getMinArgs} and {@link #getMaxArgs}
118     * @throws RuntimeException if there is an error
119     */
120    protected abstract Value apply(ParseSession session, Value[] params);
121}
