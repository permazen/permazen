006package org.jsimpledb.parse.func;
007
008import org.jsimpledb.JTransaction;
009import org.jsimpledb.parse.IndexedFieldParser;
010import org.jsimpledb.parse.ParseContext;
011import org.jsimpledb.parse.ParseSession;
012import org.jsimpledb.parse.expr.AbstractValue;
013import org.jsimpledb.parse.expr.Value;
014
015@Function
016public class QueryIndexFunction extends AbstractQueryFunction {
017
018    public QueryIndexFunction() {
019        super("queryIndex", 1, 1);
020    }
021
022    @Override
023    public String getHelpSummary() {
024        return "queries the index associated with an indexed field";
025    }
026
027    @Override
028    public String getUsage() {
029        return this.getName() + "(object-type, field-name, value-type) (JSimpleDB mode only)\n"
030          + "       " + this.getName() + "(type-name.field-name)"
031          + "       " + this.getName() + "(storage-id)";
032    }
033
034    @Override
035    public String getHelpDetail() {
036        return "Queries the simple index associated with an indexed field. The object-type is the type of object to be"
037          + " queried, i.e., the object type that contains the indexed field, or any super-type or sub-type; a strict"
038          + " sub-type will cause the returned index to be restricted to that sub-type. The field-name"
039          + " is the name of the field to query; for collection fields, this must include the sub-field name, e.g.,"
040          + " `mylist.element' or `mymap.value'. The value-type is the field's value type; in the case of reference fields,"
041          + " a super-type or more restrictive sub-type may also be specified, otherwise the field type must exactly"
042          + " match the field."
043          + "\n\nThe first form is only valid in JSimpleDB mode; the second and third forms may be used in either JSimpleDB"
044          + " mode or Core API mode.";
045    }
046
047    @Override
048    protected int parseName(ParseSession session, ParseContext ctx, boolean complete) {
049        return new IndexedFieldParser().parse(session, ctx, complete).getField().getStorageId();
050    }
051
052    @Override
053    protected Value apply(ParseSession session, final Class<?> objectType, final String fieldName, final Class<?>[] valueTypes) {
054        return new AbstractValue() {
055            @Override
056            public Object get(ParseSession session) {
057                return JTransaction.getCurrent().queryIndex(objectType, fieldName, valueTypes[0]);
058            }
059        };
060    }
061}
