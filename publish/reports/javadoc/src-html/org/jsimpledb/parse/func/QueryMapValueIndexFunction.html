006package org.jsimpledb.parse.func;
007
008import com.google.common.collect.Iterables;
009
010import org.jsimpledb.JTransaction;
011import org.jsimpledb.core.Field;
012import org.jsimpledb.core.MapField;
013import org.jsimpledb.core.SimpleField;
014import org.jsimpledb.parse.IndexedFieldParser;
015import org.jsimpledb.parse.ParseContext;
016import org.jsimpledb.parse.ParseSession;
017import org.jsimpledb.parse.ParseUtil;
018import org.jsimpledb.parse.expr.AbstractValue;
019import org.jsimpledb.parse.expr.Value;
020import org.jsimpledb.parse.util.InstancePredicate;
021
022@Function
023public class QueryMapValueIndexFunction extends AbstractQueryFunction {
024
025    public QueryMapValueIndexFunction() {
026        super("queryMapValueIndex", 2, 2);
027    }
028
029    @Override
030    public String getHelpSummary() {
031        return "queries the composite index on a map value field that includes the map keys";
032    }
033
034    @Override
035    public String getUsage() {
036        return this.getName() + "(object-type, field-name, value-type, key-type) (JSimpleDB mode only)\n"
037          + "       " + this.getName() + "(type-name.field-name)"
038          + "       " + this.getName() + "(storage-id)";
039    }
040
041    @Override
042    public String getHelpDetail() {
043        return "Queries the composite index associated with an indexed map value field. The object-type is the type of object"
044          + " to be queried, i.e., the object type that contains the map field, or any super-type or sub-type; a strict"
045          + " sub-type will cause the returned index to be restricted to that sub-type. The field-name"
046          + " is the name of the map value field; it must include the `value' sub-field name, e.g., `mymap.value'."
047          + " The key-type and value-type are the map key and value types; in the case of reference fields,"
048          + " a super-type or more restrictive sub-type may also be specified, otherwise the type must exactly match."
049          + "\n\nThe first form is only valid in JSimpleDB mode; the second and third forms may be used in either JSimpleDB"
050          + " mode or Core API mode.";
051    }
052
053    @Override
054    protected int parseName(ParseSession session, ParseContext ctx, boolean complete) {
055        return new IndexedFieldParser() {
056            @Override
057            protected Iterable<? extends Field<?>> filterFields(Iterable<? extends Field<?>> fields) {
058                return Iterables.filter(fields, new InstancePredicate(MapField.class));
059            }
060            @Override
061            protected Iterable<? extends SimpleField<?>> filterSubFields(Iterable<? extends SimpleField<?>> subFields) {
062                return Iterables.filter(subFields, new ParseUtil.HasNamePredicate(MapField.VALUE_FIELD_NAME));
063            }
064        }.parse(session, ctx, complete).getField().getStorageId();
065    }
066
067    @Override
068    protected Value apply(ParseSession session, final Class<?> objectType, final String fieldName, final Class<?>[] valueTypes) {
069        return new AbstractValue() {
070            @Override
071            public Object get(ParseSession session) {
072                return JTransaction.getCurrent().queryMapValueIndex(objectType, fieldName, valueTypes[0], valueTypes[1]);
073            }
074        };
075    }
076}
