006package org.jsimpledb.parse.func;
007
008import com.google.common.base.Predicate;
009import com.google.common.collect.Iterables;
010import com.google.common.collect.Sets;
011
012import java.util.Map;
013import java.util.NavigableSet;
014import java.util.Set;
015
016import org.jsimpledb.parse.ParseSession;
017import org.jsimpledb.parse.expr.AbstractValue;
018import org.jsimpledb.parse.expr.ConstValue;
019import org.jsimpledb.parse.expr.EvalException;
020import org.jsimpledb.parse.expr.Value;
021
022@Function
023public class FilterFunction extends ApplyExprFunction {
024
025    public FilterFunction() {
026        super("filter");
027    }
028
029    @Override
030    public String getHelpSummary() {
031        return "filters a collection";
032    }
033
034    @Override
035    public String getUsage() {
036        return "filters(items, variable, expression)";
037    }
038
039    @Override
040    public String getHelpDetail() {
041        return "Creates a filtered view of an Iterable, where items are included only if when the item is assigned to the"
042          + " specified variable the specified expression evaluates to true. Maps are also supported, in which case the map's"
043          + " entrySet() is filtered.";
044    }
045
046    @Override
047    protected Value apply(ParseSession session, final ParamInfo params) {
048        return new AbstractValue() {
049            @Override
050            public Object get(final ParseSession session) {
051
052                // Build predicate
053                final Predicate<Object> predicate = new Predicate<Object>() {
054                    @Override
055                    public boolean apply(Object item) {
056                        return FilterFunction.this.evaluate(session, params.getVariable(),
057                          new ConstValue(item), params.getExpr()).checkBoolean(session, "filter()");
058                    }
059                };
060
061                // Evaluate items
062                Object items = params.getItems().evaluate(session).checkNotNull(session, "filter()");
063                if (items instanceof Map)
064                    items = ((Map<?, ?>)items).entrySet();
065
066                // Preserve NavigableSet if possible
067                if (items instanceof NavigableSet)
068                    return Sets.filter((NavigableSet<?>)items, predicate);
069
070                // Preserve Set if possible
071                if (items instanceof Set)
072                    return Sets.filter((Set<?>)items, predicate);
073
074                // Return Iterable view
075                if (items instanceof Iterable)
076                    return Iterables.filter((Iterable<?>)items, predicate);
077
078                // Wrong type
079                throw new EvalException("invalid filter() operation on non-Iterable object of type " + items.getClass().getName());
080            }
081        };
082    }
083}
