006package org.jsimpledb;
007
008import com.google.common.base.Converter;
009import com.google.common.base.Function;
010import com.google.common.collect.EnumHashBiMap;
011import com.google.common.collect.Maps;
012
013import java.util.EnumSet;
014
015import org.jsimpledb.core.EnumValue;
016
017/**
018 * Converts between core database {@link EnumValue} objects and the corresponding Java {@link Enum} model values.
019 */
020public class EnumConverter<T extends Enum<T>> extends Converter<T, EnumValue> {
021
022    private final Class<T> enumType;
023    private final EnumHashBiMap<T, EnumValue> valueMap;
024
025    /**
026     * Constructor.
027     *
028     * @param enumType {@link Enum} type
029     * @throws IllegalArgumentException if {@code enumType} is null
030     */
031    public EnumConverter(Class<T> enumType) {
032        if (enumType == null)
033            throw new IllegalArgumentException("null enumType");
034        enumType.asSubclass(Enum.class);                            // verify it's really an Enum
035        this.enumType = enumType;
036        this.valueMap = EnumHashBiMap.<T, EnumValue>create(Maps.asMap(EnumSet.allOf(this.enumType), new Function<T, EnumValue>() {
037            @Override
038            public EnumValue apply(T value) {
039                return new EnumValue(value);
040            }
041        }));
042    }
043
044    @Override
045    protected EnumValue doForward(T value) {
046        if (value == null)
047            return null;
048        final EnumValue enumValue = this.valueMap.get(value);
049        if (enumValue == null)
050            throw new IllegalArgumentException("invalid enum value " + value + " not an instance of " + this.enumType);
051        return enumValue;
052    }
053
054    @Override
055    protected T doBackward(EnumValue enumValue) {
056        if (enumValue == null)
057            return null;
058        final T value = this.valueMap.inverse().get(enumValue);
059        if (value == null)
060            throw new IllegalArgumentException("invalid value " + enumValue + " not found in " + this.enumType);
061        return value;
062    }
063
064    /**
065     * Get the {@link Enum} type associated with this instance.
066     *
067     * @return associated {@link Enum} type
068     */
069    public Class<T> getEnumType() {
070        return this.enumType;
071    }
072
073    /**
074     * Convenience "constructor" allowing wildcard caller {@link Enum} types.
075     *
076     * @param enumType type for the created converter
077     * @return new converter
078     */
079    @SuppressWarnings({ "unchecked", "rawtypes" })
080    public static EnumConverter<?> createEnumConverter(Class<? extends Enum<?>> enumType) {
081        return new EnumConverter(enumType);
082    }
083
084// Object
085
086    @Override
087    public boolean equals(Object obj) {
088        if (obj == this)
089            return true;
090        if (obj == null || obj.getClass() != this.getClass())
091            return false;
092        final EnumConverter<?> that = (EnumConverter<?>)obj;
093        return this.enumType == that.enumType;
094    }
095
096    @Override
097    public int hashCode() {
098        return this.enumType.hashCode();
099    }
100
101    @Override
102    public String toString() {
103        return this.getClass().getSimpleName() + "[type=" + this.enumType.getName() + "]";
104    }
105}
