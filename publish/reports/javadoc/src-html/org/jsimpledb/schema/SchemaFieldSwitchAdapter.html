006package org.jsimpledb.schema;
007
008/**
009 * Adapter class for {@link SchemaFieldSwitch}.
010 *
011 * @param <R> switch method return type
012 */
013public class SchemaFieldSwitchAdapter<R> implements SchemaFieldSwitch<R> {
014
015    /**
016     * Handle a {@link SetSchemaField}.
017     *
018     * <p>
019     * The implementation in {@link SchemaFieldSwitchAdapter} delegates to
020     * {@link #caseCollectionSchemaField caseCollectionSchemaField()}.
021     * </p>
022     */
023    @Override
024    public R caseSetSchemaField(SetSchemaField field) {
025        return this.caseCollectionSchemaField(field);
026    }
027
028    /**
029     * Handle a {@link ListSchemaField}.
030     *
031     * <p>
032     * The implementation in {@link SchemaFieldSwitchAdapter} delegates to
033     * {@link #caseCollectionSchemaField caseCollectionSchemaField()}.
034     * </p>
035     */
036    @Override
037    public R caseListSchemaField(ListSchemaField field) {
038        return this.caseCollectionSchemaField(field);
039    }
040
041    /**
042     * Handle a {@link MapSchemaField}.
043     *
044     * <p>
045     * The implementation in {@link SchemaFieldSwitchAdapter} delegates to
046     * {@link #caseComplexSchemaField caseComplexSchemaField()}.
047     * </p>
048     */
049    @Override
050    public R caseMapSchemaField(MapSchemaField field) {
051        return this.caseComplexSchemaField(field);
052    }
053
054    /**
055     * Handle a {@link SimpleSchemaField}.
056     *
057     * <p>
058     * The implementation in {@link SchemaFieldSwitchAdapter} delegates to
059     * {@link #caseDefault caseDefault()}.
060     * </p>
061     */
062    @Override
063    public R caseSimpleSchemaField(SimpleSchemaField field) {
064        return this.caseDefault(field);
065    }
066
067    /**
068     * Handle a {@link ReferenceSchemaField}.
069     *
070     * <p>
071     * The implementation in {@link SchemaFieldSwitchAdapter} delegates to
072     * {@link #caseSimpleSchemaField caseSimpleSchemaField()}.
073     * </p>
074     */
075    @Override
076    public R caseReferenceSchemaField(ReferenceSchemaField field) {
077        return this.caseSimpleSchemaField(field);
078    }
079
080    /**
081     * Handle a {@link EnumSchemaField}.
082     *
083     * <p>
084     * The implementation in {@link SchemaFieldSwitchAdapter} delegates to
085     * {@link #caseSimpleSchemaField caseSimpleSchemaField()}.
086     * </p>
087     */
088    @Override
089    public R caseEnumSchemaField(EnumSchemaField field) {
090        return this.caseSimpleSchemaField(field);
091    }
092
093    /**
094     * Handle a {@link CounterSchemaField}.
095     *
096     * <p>
097     * The implementation in {@link SchemaFieldSwitchAdapter} delegates to {@link #caseDefault caseDefault()}.
098     * </p>
099     */
100    @Override
101    public R caseCounterSchemaField(CounterSchemaField field) {
102        return this.caseDefault(field);
103    }
104
105    /**
106     * Adapter class roll-up method.
107     *
108     * <p>
109     * The implementation in {@link SchemaFieldSwitchAdapter} delegates to {@link #caseComplexSchemaField caseComplexSchemaField()}.
110     * </p>
111     *
112     * @param field visiting field
113     * @return visitor return value
114     */
115    protected R caseCollectionSchemaField(CollectionSchemaField field) {
116        return this.caseComplexSchemaField(field);
117    }
118
119    /**
120     * Adapter class roll-up method.
121     *
122     * <p>
123     * The implementation in {@link SchemaFieldSwitchAdapter} delegates to {@link #caseDefault caseDefault()}.
124     * </p>
125     *
126     * @param field visiting field
127     * @return visitor return value
128     */
129    protected R caseComplexSchemaField(ComplexSchemaField field) {
130        return this.caseDefault(field);
131    }
132
133    /**
134     * Adapter class roll-up method.
135     *
136     * <p>
137     * The implementation in {@link SchemaFieldSwitchAdapter} always throws {@link UnsupportedOperationException}.
138     * </p>
139     *
140     * @param field visiting field
141     * @return visitor return value
142     */
143    protected R caseDefault(SchemaField field) {
144        throw new UnsupportedOperationException("field type not handled: " + field);
145    }
146}
