006package org.jsimpledb.schema;
007
008import java.util.ArrayList;
009import java.util.HashSet;
010import java.util.List;
011
012import javax.xml.stream.XMLStreamException;
013import javax.xml.stream.XMLStreamReader;
014import javax.xml.stream.XMLStreamWriter;
015
016import org.jsimpledb.core.Database;
017import org.jsimpledb.core.InvalidSchemaException;
018import org.jsimpledb.util.DiffGenerating;
019import org.jsimpledb.util.Diffs;
020
021/**
022 * A composite index associated with a {@link SchemaObjectType}.
023 */
024public class SchemaCompositeIndex extends AbstractSchemaItem implements DiffGenerating<SchemaCompositeIndex> {
025
026    private /*final*/ ArrayList<Integer> indexedFields = new ArrayList<>();
027
028    /**
029     * Get the fields that comprise this index.
030     *
031     * @return storage IDs of indexed fields
032     */
033    public List<Integer> getIndexedFields() {
034        return this.indexedFields;
035    }
036
037    @Override
038    void validate() {
039        super.validate();
040        if (this.indexedFields.size() < 2 || this.indexedFields.size() > Database.MAX_INDEXED_FIELDS) {
041            throw new InvalidSchemaException("invalid " + this + ": between 2 and "
042              + Database.MAX_INDEXED_FIELDS + " fields must be specified");
043        }
044        final HashSet<Integer> idsSeen = new HashSet<Integer>();
045        for (int i = 0; i < this.indexedFields.size(); i++) {
046            final int storageId = this.indexedFields.get(i);
047            if (!idsSeen.add(storageId))
048                throw new InvalidSchemaException("invalid " + this + ": duplicate field in composite index: " + storageId);
049        }
050    }
051
052    @Override
053    boolean isCompatibleWithInternal(AbstractSchemaItem that0) {
054        final SchemaCompositeIndex that = (SchemaCompositeIndex)that0;
055        if (!this.indexedFields.equals(that.indexedFields))
056            return false;
057        return true;
058    }
059
060// XML Reading
061
062    @Override
063    void readSubElements(XMLStreamReader reader, int formatVersion) throws XMLStreamException {
064        this.indexedFields.clear();
065        while (this.expect(reader, true, INDEXED_FIELD_TAG)) {
066            this.indexedFields.add(this.getIntAttr(reader, STORAGE_ID_ATTRIBUTE));
067            this.expectClose(reader);   // </IndexedField>
068        }
069        this.indexedFields.trimToSize();
070    }
071
072// XML Writing
073
074    @Override
075    void writeXML(XMLStreamWriter writer) throws XMLStreamException {
076        writer.writeStartElement(COMPOSITE_INDEX_TAG.getNamespaceURI(), COMPOSITE_INDEX_TAG.getLocalPart());
077        this.writeAttributes(writer, true);
078        for (int storageId : this.indexedFields) {
079            writer.writeEmptyElement(INDEXED_FIELD_TAG.getNamespaceURI(), INDEXED_FIELD_TAG.getLocalPart());
080            writer.writeAttribute(STORAGE_ID_ATTRIBUTE.getNamespaceURI(), STORAGE_ID_ATTRIBUTE.getLocalPart(), "" + storageId);
081        }
082        writer.writeEndElement();           // </CompositeIndex>
083    }
084
085// DiffGenerating
086
087    @Override
088    public Diffs differencesFrom(SchemaCompositeIndex that) {
089        final Diffs diffs = new Diffs(super.differencesFrom(that));
090        if (!this.indexedFields.equals(that.indexedFields))
091            diffs.add("changed indexed field storage IDs from " + that.indexedFields + " to " + this.indexedFields);
092        return diffs;
093    }
094
095// Object
096
097    @Override
098    public String toString() {
099        return "composite index " + super.toString();
100    }
101
102    @Override
103    public boolean equals(Object obj) {
104        if (obj == this)
105            return true;
106        if (!super.equals(obj))
107            return false;
108        final SchemaCompositeIndex that = (SchemaCompositeIndex)obj;
109        return this.indexedFields.equals(that.indexedFields);
110    }
111
112    @Override
113    public int hashCode() {
114        return super.hashCode() ^ this.indexedFields.hashCode();
115    }
116
117// Cloneable
118
119    @Override
120    @SuppressWarnings("unchecked")
121    public SchemaCompositeIndex clone() {
122        final SchemaCompositeIndex clone = (SchemaCompositeIndex)super.clone();
123        clone.indexedFields = (ArrayList<Integer>)clone.indexedFields.clone();
124        return clone;
125    }
126}
