006package org.jsimpledb.schema;
007
008import java.util.Collections;
009import java.util.SortedSet;
010import java.util.TreeMap;
011
012/**
013 * An index of object, field, and index names in a {@link SchemaModel}.
014 */
015public class NameIndex {
016
017    private final SchemaModel schemaModel;
018    private final TreeMap<String, SchemaObjectType> typeMap = new TreeMap<>();
019    private final TreeMap<Integer, TreeMap<String, SchemaField>> typeFieldMap = new TreeMap<>();
020    private final TreeMap<Integer, TreeMap<String, SchemaCompositeIndex>> typeCompositeIndexMap = new TreeMap<>();
021
022    /**
023     * Constructor.
024     *
025     * @param schemaModel schema model to index
026     * @throws IllegalArgumentException if a name conflict is detected (implies an invalid model)
027     */
028    public NameIndex(SchemaModel schemaModel) {
029
030        // Initialize
031        if (schemaModel == null)
032            throw new IllegalArgumentException("null schemaModel");
033        this.schemaModel = schemaModel;
034
035        // Index type names
036        for (SchemaObjectType type : schemaModel.getSchemaObjectTypes().values()) {
037
038            // Index type name
039            if (this.typeMap.put(type.getName(), type) != null)
040                throw new IllegalArgumentException("schema model is invalid");
041
042            // Index field names
043            final TreeMap<String, SchemaField> fieldMap = new TreeMap<>();
044            this.typeFieldMap.put(type.getStorageId(), fieldMap);
045            for (SchemaField field : type.getSchemaFields().values()) {
046                if (fieldMap.put(field.getName(), field) != null)
047                    throw new IllegalArgumentException("schema model is invalid");
048            }
049
050            // Index composite index names
051            final TreeMap<String, SchemaCompositeIndex> compositeIndexMap = new TreeMap<>();
052            this.typeCompositeIndexMap.put(type.getStorageId(), compositeIndexMap);
053            for (SchemaCompositeIndex compositeIndex : type.getSchemaCompositeIndexes().values()) {
054                if (compositeIndexMap.put(compositeIndex.getName(), compositeIndex) != null)
055                    throw new IllegalArgumentException("schema model is invalid");
056            }
057        }
058    }
059
060    /**
061     * Get the {@link SchemaModel} on which this index is based.
062     *
063     * @return associated schema
064     */
065    public SchemaModel getSchemaModel() {
066        return this.schemaModel;
067    }
068
069    /**
070     * Get the {@link SchemaObjectType} with the given name.
071     *
072     * @param name type name
073     * @throws IllegalArgumentException if {@code name} is null
074     * @return the unique {@link SchemaObjectType} with name {@code name}, or null if not found
075     */
076    public SchemaObjectType getSchemaObjectType(String name) {
077        if (name == null)
078            throw new IllegalArgumentException("null name");
079        return this.typeMap.get(name);
080    }
081
082    /**
083     * Get the names of all {@link SchemaObjectType}s.
084     *
085     * @return unmodifiable set of {@link SchemaObjectType} names
086     */
087    public SortedSet<String> getSchemaObjectTypeNames() {
088        return Collections.unmodifiableSortedSet(this.typeMap.navigableKeySet());
089    }
090
091    /**
092     * Get the {@link SchemaField} with the given name in the given {@link SchemaObjectType}.
093     *
094     * @param type object type
095     * @param name field name
096     * @throws IllegalArgumentException if either paramter is null
097     * @return the unique {@link SchemaField} with name {@code name} in {@code type}, or null if not found
098     * @throws IllegalArgumentException if {@code type} is not indexed by this instance
099     */
100    public SchemaField getSchemaField(SchemaObjectType type, String name) {
101        if (type == null)
102            throw new IllegalArgumentException("null type");
103        if (name == null)
104            throw new IllegalArgumentException("null name");
105        final TreeMap<String, SchemaField> fieldMap = this.typeFieldMap.get(type.getStorageId());
106        if (fieldMap == null)
107            throw new IllegalArgumentException("unknown type `" + type.getName() + "' with storage ID " + type.getStorageId());
108        return fieldMap.get(name);
109    }
110
111    /**
112     * Get all of the names of {@link SchemaField}s in the given {@link SchemaObjectType}.
113     *
114     * @param type schema object
115     * @return unmodifiable set of {@link SchemaField} names in {@code type}
116     * @throws IllegalArgumentException if {@code type} is not indexed by this instance
117     */
118    public SortedSet<String> getSchemaFieldNames(SchemaObjectType type) {
119        if (type == null)
120            throw new IllegalArgumentException("null type");
121        final TreeMap<String, SchemaField> fieldMap = this.typeFieldMap.get(type.getStorageId());
122        if (fieldMap == null)
123            throw new IllegalArgumentException("unknown type `" + type.getName() + "' with storage ID " + type.getStorageId());
124        return Collections.unmodifiableSortedSet(fieldMap.navigableKeySet());
125    }
126
127    /**
128     * Get the {@link SchemaCompositeIndex} with the given name in the given {@link SchemaObjectType}.
129     *
130     * @param type object type
131     * @param name index name
132     * @throws IllegalArgumentException if either paramter is null
133     * @return the unique {@link SchemaCompositeIndex} with name {@code name} in {@code type}, or null if not found
134     * @throws IllegalArgumentException if {@code type} is not indexed by this instance
135     */
136    public SchemaCompositeIndex getSchemaCompositeIndex(SchemaObjectType type, String name) {
137        if (type == null)
138            throw new IllegalArgumentException("null type");
139        if (name == null)
140            throw new IllegalArgumentException("null name");
141        final TreeMap<String, SchemaCompositeIndex> indexMap = this.typeCompositeIndexMap.get(type.getStorageId());
142        if (indexMap == null)
143            throw new IllegalArgumentException("unknown type `" + type.getName() + "' with storage ID " + type.getStorageId());
144        return indexMap.get(name);
145    }
146
147    /**
148     * Get all of the names of {@link SchemaCompositeIndex}s in the given {@link SchemaObjectType}.
149     *
150     * @param type schema object
151     * @return unmodifiable set of {@link SchemaCompositeIndex} names in {@code type}
152     * @throws IllegalArgumentException if {@code type} is not indexed by this instance
153     */
154    public SortedSet<String> getSchemaCompositeIndexNames(SchemaObjectType type) {
155        if (type == null)
156            throw new IllegalArgumentException("null type");
157        final TreeMap<String, SchemaCompositeIndex> indexMap = this.typeCompositeIndexMap.get(type.getStorageId());
158        if (indexMap == null)
159            throw new IllegalArgumentException("unknown type `" + type.getName() + "' with storage ID " + type.getStorageId());
160        return Collections.unmodifiableSortedSet(indexMap.navigableKeySet());
161    }
162}
