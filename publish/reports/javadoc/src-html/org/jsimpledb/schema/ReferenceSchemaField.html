006package org.jsimpledb.schema;
007
008import java.util.SortedSet;
009import java.util.TreeSet;
010
011import javax.xml.stream.XMLStreamException;
012import javax.xml.stream.XMLStreamReader;
013import javax.xml.stream.XMLStreamWriter;
014
015import org.jsimpledb.core.DeleteAction;
016import org.jsimpledb.core.FieldType;
017import org.jsimpledb.core.InvalidSchemaException;
018import org.jsimpledb.util.Diffs;
019
020/**
021 * A reference field in a {@link SchemaObjectType}.
022 */
023public class ReferenceSchemaField extends SimpleSchemaField {
024
025    private DeleteAction onDelete;
026    private boolean cascadeDelete;
027    private SortedSet<Integer> objectTypes;
028
029    public ReferenceSchemaField() {
030        this.setType(FieldType.REFERENCE_TYPE_NAME);
031        this.setIndexed(true);
032    }
033
034    /**
035     * Get the desired behavior when an object referred to by this field is deleted.
036     *
037     * @return desired behavior when a referenced object is deleted
038     */
039    public DeleteAction getOnDelete() {
040        return this.onDelete;
041    }
042    public void setOnDelete(DeleteAction onDelete) {
043        this.onDelete = onDelete;
044    }
045
046    /**
047     * Determine whether the referred-to object should be deleted when an object containing this field is deleted.
048     *
049     * @return whether deletion should cascade to the referred-to object
050     */
051    public boolean isCascadeDelete() {
052        return this.cascadeDelete;
053    }
054    public void setCascadeDelete(boolean cascadeDelete) {
055        this.cascadeDelete = cascadeDelete;
056    }
057
058    /**
059     * Get the object types this field is allowed to reference, if so restricted.
060     *
061     * @return storage IDs of allowed object types, or null if there is no restriction
062     */
063    public SortedSet<Integer> getObjectTypes() {
064        return this.objectTypes;
065    }
066    public void setObjectTypes(SortedSet<Integer> objectTypes) {
067        this.objectTypes = objectTypes;
068    }
069
070    @Override
071    void validate() {
072        super.validate();
073        if (!FieldType.REFERENCE_TYPE_NAME.equals(this.getType())) {
074            throw new InvalidSchemaException("invalid " + this + ": reference fields must have type `"
075              + FieldType.REFERENCE_TYPE_NAME + "'");
076        }
077        if (!this.isIndexed())
078            throw new IllegalArgumentException("invalid " + this + ": reference fields must always be indexed");
079        if (this.onDelete == null)
080            throw new InvalidSchemaException("invalid " + this + ": no delete action specified");
081    }
082
083    @Override
084    public <R> R visit(SchemaFieldSwitch<R> target) {
085        return target.caseReferenceSchemaField(this);
086    }
087
088    @Override
089    boolean isCompatibleWithInternal(AbstractSchemaItem that0) {
090        final ReferenceSchemaField that = (ReferenceSchemaField)that0;
091        if (!super.isCompatibleWithInternal(that))
092            return false;
093        if (!this.onDelete.equals(that.onDelete))
094            return false;
095        if (this.cascadeDelete != that.cascadeDelete)
096            return false;
097        if (!(this.objectTypes != null ? this.objectTypes.equals(that.objectTypes) : that.objectTypes == null))
098            return false;
099        return true;
100    }
101
102// DiffGenerating
103
104    @Override
105    public Diffs differencesFrom(SimpleSchemaField other) {
106        final Diffs diffs = new Diffs(super.differencesFrom(other));
107        if (!(other instanceof ReferenceSchemaField)) {
108            diffs.add("change type from " + other.getClass().getSimpleName() + " to " + this.getClass().getSimpleName());
109            return diffs;
110        }
111        final ReferenceSchemaField that = (ReferenceSchemaField)other;
112        if (!(this.onDelete != null ? this.onDelete.equals(that.onDelete) : that.onDelete == null))
113            diffs.add("changed on-delete action from " + that.onDelete + " to " + this.onDelete);
114        if (this.cascadeDelete != that.cascadeDelete)
115            diffs.add("changed cascade delete from " + that.cascadeDelete + " to " + this.cascadeDelete);
116        if (!(this.objectTypes != null ? this.objectTypes.equals(that.objectTypes) : that.objectTypes == null))
117            diffs.add("changed allowed object type storage IDs from " + that.objectTypes + " to " + this.objectTypes);
118        return diffs;
119    }
120
121// XML Reading
122
123    @Override
124    void readAttributes(XMLStreamReader reader, int formatVersion) throws XMLStreamException {
125        super.readAttributes(reader, formatVersion);
126        final String text = this.getAttr(reader, ON_DELETE_ATTRIBUTE, false);
127        final DeleteAction action;
128        if (text != null) {
129            if ((action = Enum.valueOf(DeleteAction.class, text)) == null) {
130                throw new XMLStreamException("invalid value `" + text
131                  + " for \"" + ON_DELETE_ATTRIBUTE.getLocalPart() + "\" attribute in " + this, reader.getLocation());
132            }
133        } else
134            action = DeleteAction.EXCEPTION;
135        this.setOnDelete(action);
136        final Boolean cascadeDeleteAttr = this.getBooleanAttr(reader, CASCADE_DELETE_ATTRIBUTE, false);
137        if (cascadeDeleteAttr != null)
138            this.setCascadeDelete(cascadeDeleteAttr);
139    }
140
141    @Override
142    void readSubElements(XMLStreamReader reader, int formatVersion) throws XMLStreamException {
143
144        // Any restrictions?
145        if (!this.expect(reader, true, OBJECT_TYPES_TAG)) {
146            this.objectTypes = null;
147            return;
148        }
149
150        // Read list of zero or more permitted storage ID
151        this.objectTypes = new TreeSet<>();
152        while (this.expect(reader, true, OBJECT_TYPE_TAG)) {
153            this.objectTypes.add(this.getIntAttr(reader, STORAGE_ID_ATTRIBUTE));
154            this.expectClose(reader);           // </ObjectType>
155        }
156
157        // Read closing </ReferenceField>
158        this.expectClose(reader);
159    }
160
161// XML Writing
162
163    @Override
164    void writeXML(XMLStreamWriter writer, boolean includeName) throws XMLStreamException {
165        if (this.objectTypes != null)
166            writer.writeStartElement(REFERENCE_FIELD_TAG.getNamespaceURI(), REFERENCE_FIELD_TAG.getLocalPart());
167        else
168            writer.writeEmptyElement(REFERENCE_FIELD_TAG.getNamespaceURI(), REFERENCE_FIELD_TAG.getLocalPart());
169        this.writeAttributes(writer, includeName);
170        if (this.onDelete != null)
171            writer.writeAttribute(ON_DELETE_ATTRIBUTE.getNamespaceURI(), ON_DELETE_ATTRIBUTE.getLocalPart(), this.onDelete.name());
172        if (this.objectTypes != null) {
173            writer.writeStartElement(OBJECT_TYPES_TAG.getNamespaceURI(), OBJECT_TYPES_TAG.getLocalPart());
174            for (int storageId : this.objectTypes) {
175                writer.writeEmptyElement(OBJECT_TYPE_TAG.getNamespaceURI(), OBJECT_TYPE_TAG.getLocalPart());
176                writer.writeAttribute(STORAGE_ID_ATTRIBUTE.getNamespaceURI(), STORAGE_ID_ATTRIBUTE.getLocalPart(), "" + storageId);
177            }
178            writer.writeEndElement();           // </ObjectTypes>
179            writer.writeEndElement();           // </ReferenceField>
180        }
181    }
182
183    @Override
184    void writeSimpleAttributes(XMLStreamWriter writer) throws XMLStreamException {
185        // don't need to write type or indexed
186        if (this.cascadeDelete) {
187            writer.writeAttribute(CASCADE_DELETE_ATTRIBUTE.getNamespaceURI(), CASCADE_DELETE_ATTRIBUTE.getLocalPart(),
188              "" + this.cascadeDelete);
189        }
190    }
191
192// Object
193
194    @Override
195    public boolean equals(Object obj) {
196        if (obj == this)
197            return true;
198        if (!super.equals(obj))
199            return false;
200        final ReferenceSchemaField that = (ReferenceSchemaField)obj;
201        return this.onDelete == that.onDelete
202          && this.cascadeDelete == that.cascadeDelete
203          && (this.objectTypes != null ? this.objectTypes.equals(that.objectTypes) : that.objectTypes == null);
204    }
205
206    @Override
207    public int hashCode() {
208        return super.hashCode()
209          ^ (this.cascadeDelete ? 1 : 0)
210          ^ (this.onDelete != null ? this.onDelete.hashCode() : 0)
211          ^ (this.objectTypes != null ? this.objectTypes.hashCode() : 0);
212    }
213
214// Cloneable
215
216    @Override
217    public ReferenceSchemaField clone() {
218        final ReferenceSchemaField clone = (ReferenceSchemaField)super.clone();
219        if (clone.objectTypes != null)
220            clone.objectTypes = new TreeSet<Integer>(clone.objectTypes);
221        return clone;
222    }
223}
