008package org.jsimpledb.schema;
009
010import java.util.SortedMap;
011import java.util.TreeMap;
012
013import javax.xml.stream.XMLStreamException;
014import javax.xml.stream.XMLStreamReader;
015import javax.xml.stream.XMLStreamWriter;
016
017import org.jsimpledb.core.InvalidSchemaException;
018
019/**
020 * One object type in a {@link SchemaModel}.
021 */
022public class SchemaObject extends AbstractSchemaItem {
023
024    private SortedMap<Integer, SchemaField> schemaFields = new TreeMap<>();
025
026    public SortedMap<Integer, SchemaField> getSchemaFields() {
027        return this.schemaFields;
028    }
029    public void setSchemaFields(SortedMap<Integer, SchemaField> schemaFields) {
030        this.schemaFields = schemaFields;
031    }
032
033    @Override
034    public void validate() {
035        super.validate();
036        if (this.getName() == null || this.getName().length() == 0)
037            throw new InvalidSchemaException(this + " must have a name");
038        for (SchemaField field : this.schemaFields.values()) {
039            if (field.getName() == null || field.getName().length() == 0)
040                throw new InvalidSchemaException(field + " of " + this + " must have a name");
041            field.validate();
042        }
043    }
044
045    @Override
046    void readSubElements(XMLStreamReader reader) throws XMLStreamException {
047        while (this.expect(reader, true, COUNTER_FIELD_TAG, LIST_FIELD_TAG, MAP_FIELD_TAG,
048          REFERENCE_FIELD_TAG, SET_FIELD_TAG, SIMPLE_FIELD_TAG)) {
049            SchemaField field;
050            if (reader.getName().equals(COUNTER_FIELD_TAG))
051                field = new CounterSchemaField();
052            else if (reader.getName().equals(LIST_FIELD_TAG))
053                field = new ListSchemaField();
054            else if (reader.getName().equals(MAP_FIELD_TAG))
055                field = new MapSchemaField();
056            else if (reader.getName().equals(REFERENCE_FIELD_TAG))
057                field = new ReferenceSchemaField();
058            else if (reader.getName().equals(SET_FIELD_TAG))
059                field = new SetSchemaField();
060            else if (reader.getName().equals(SIMPLE_FIELD_TAG))
061                field = new SimpleSchemaField();
062            else
063                throw new RuntimeException("internal error");
064            field.readXML(reader);
065            final int storageId = field.getStorageId();
066            final SchemaField previous = this.schemaFields.put(storageId, field);
067            if (previous != null) {
068                throw new InvalidSchemaException("duplicate use of storage ID " + storageId
069                  + " for both " + previous + " and " + field + " in " + this);
070            }
071        }
072    }
073
074    @Override
075    void writeXML(XMLStreamWriter writer) throws XMLStreamException {
076        if (this.schemaFields.isEmpty())
077            writer.writeEmptyElement(OBJECT_TAG.getNamespaceURI(), OBJECT_TAG.getLocalPart());
078        else
079            writer.writeStartElement(OBJECT_TAG.getNamespaceURI(), OBJECT_TAG.getLocalPart());
080        this.writeAttributes(writer);
081        for (SchemaField schemaField : this.schemaFields.values())
082            schemaField.writeXML(writer);
083        if (!this.schemaFields.isEmpty())
084            writer.writeEndElement();
085    }
086
087// Object
088
089    @Override
090    public String toString() {
091        return "object " + super.toString();
092    }
093
094    @Override
095    public boolean equals(Object obj) {
096        if (obj == this)
097            return true;
098        if (!super.equals(obj))
099            return false;
100        final SchemaObject that = (SchemaObject)obj;
101        return this.schemaFields.equals(that.schemaFields);
102    }
103
104    @Override
105    public int hashCode() {
106        return super.hashCode() ^ this.schemaFields.hashCode();
107    }
108
109// Cloneable
110
111    @Override
112    public SchemaObject clone() {
113        final SchemaObject clone = (SchemaObject)super.clone();
114        clone.schemaFields = new TreeMap<>();
115        for (SchemaField schemaField : this.schemaFields.values())
116            clone.getSchemaFields().put(schemaField.getStorageId(), schemaField.clone());
117        return clone;
118    }
119}
