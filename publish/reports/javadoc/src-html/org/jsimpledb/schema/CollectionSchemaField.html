006package org.jsimpledb.schema;
007
008import java.util.Collections;
009import java.util.Map;
010
011import javax.xml.stream.XMLStreamException;
012import javax.xml.stream.XMLStreamReader;
013
014import org.jsimpledb.core.CollectionField;
015import org.jsimpledb.util.Diffs;
016
017/**
018 * A collection field in one version of a {@link SchemaObjectType}.
019 */
020public abstract class CollectionSchemaField extends ComplexSchemaField {
021
022    private SimpleSchemaField elementField;
023
024    public SimpleSchemaField getElementField() {
025        return this.elementField;
026    }
027    public void setElementField(SimpleSchemaField elementField) {
028        this.elementField = elementField;
029    }
030
031    @Override
032    public Map<String, SimpleSchemaField> getSubFields() {
033        return Collections.<String, SimpleSchemaField>singletonMap(CollectionField.ELEMENT_FIELD_NAME, this.elementField);
034    }
035
036    @Override
037    void readSubElements(XMLStreamReader reader, int formatVersion) throws XMLStreamException {
038        this.elementField = this.readSubField(reader, formatVersion, CollectionField.ELEMENT_FIELD_NAME);
039        this.expectClose(reader);
040    }
041
042// DiffGenerating
043
044    protected Diffs differencesFrom(CollectionSchemaField that) {
045        final Diffs diffs = new Diffs(super.differencesFrom(that));
046        final Diffs elementDiffs = this.elementField.differencesFrom(that.elementField);
047        if (!elementDiffs.isEmpty())
048            diffs.add("changed element field", elementDiffs);
049        return diffs;
050    }
051
052// Object
053
054    @Override
055    public String toString() {
056        return super.toString() + (this.elementField != null ? " with element " + this.elementField : "");
057    }
058
059    @Override
060    public boolean equals(Object obj) {
061        if (obj == this)
062            return true;
063        if (!super.equals(obj))
064            return false;
065        final CollectionSchemaField that = (CollectionSchemaField)obj;
066        return this.elementField != null ? this.elementField.equals(that.elementField) : that.elementField == null;
067    }
068
069    @Override
070    public int hashCode() {
071        return super.hashCode() ^ (this.elementField != null ? this.elementField.hashCode() : 0);
072    }
073
074// Cloneable
075
076    @Override
077    public CollectionSchemaField clone() {
078        final CollectionSchemaField clone = (CollectionSchemaField)super.clone();
079        clone.elementField = this.elementField.clone();
080        return clone;
081    }
082}
