008package org.jsimpledb.cli;
009
010import com.google.common.base.Function;
011
012import org.dellroad.stuff.string.ParseContext;
013
014/**
015 * CLI command.
016 */
017public abstract class Command {
018
019    public static final Function<Command, String> NAME_FUNCTION = new Function<Command, String>() {
020        @Override
021        public String apply(Command command) {
022            return command.getName();
023        }
024    };
025
026    private final String prefix;
027    private final String name;
028
029    protected Command(String name) {
030        this((String)null, name);
031    }
032
033    protected Command(AggregateCommand parent, String name) {
034        this(parent instanceof RootCommand ? null : parent.getFullName(), name);
035    }
036
037    protected Command(String prefix, String name) {
038        this.prefix = prefix;
039        if (name == null)
040            throw new IllegalArgumentException("null name");
041        this.name = name;
042    }
043
044    public String getPrefix() {
045        return this.prefix;
046    }
047
048    public String getName() {
049        return this.name;
050    }
051
052    public String getFullName() {
053        return this.getFullName(this.name);
054    }
055
056    public String getFullName(String commandName) {
057        return (this.prefix != null ? this.prefix + " " : "") + commandName;
058    }
059
060    /**
061     * Parse the command.
062     *
063     * @throws ParseException if parse fails, preferably with completions
064     */
065    public abstract Action parse(Session session, ParseContext ctx) throws ParseException;
066
067    /**
068     * Get a one-line help summary.
069     */
070    public abstract String getHelpSummary();
071
072    /**
073     * Get help information.
074     *
075     * <p>
076     * The implementation in {@link Command} just delegates to {@link #getHelpSummary}.
077     * </p>
078     */
079    public String getHelpDetail() {
080        return this.getHelpSummary();
081    }
082}
