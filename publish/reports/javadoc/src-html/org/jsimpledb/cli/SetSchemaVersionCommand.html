008package org.jsimpledb.cli;
009
010import org.jsimpledb.util.ParseContext;
011
012public class SetSchemaVersionCommand extends AbstractCommand {
013
014    public SetSchemaVersionCommand() {
015        super("set-schema-version");
016    }
017
018    @Override
019    public String getUsage() {
020        return this.name + " version";
021    }
022
023    @Override
024    public String getHelpSummary() {
025        return "Sets the expected schema version";
026    }
027
028    @Override
029    public String getHelpDetail() {
030        return "Sets the expected schema version version number. If no such schema version is recorded in the database,"
031          + " and `set-allow-new-schema true' has been invoked, then the current schema will be recorded anew under the"
032          + " specified version number.";
033    }
034
035    @Override
036    public Action parseParameters(Session session, ParseContext ctx) {
037        final String value = new ParamParser(1, 1, this.getUsage()).parse(ctx).getParam(0);
038        final int version;
039        try {
040            version = Integer.parseInt(value);
041            if (version < 0)
042                throw new IllegalArgumentException("schema version is negative");
043        } catch (IllegalArgumentException e) {
044            throw new ParseException(ctx, "invalid schema version `" + value + "'");
045        }
046        return new Action() {
047            @Override
048            public void run(Session session) throws Exception {
049                session.setSchemaVersion(version);
050                session.getWriter().println("Set schema version to " + version);
051            }
052        };
053    }
054}
