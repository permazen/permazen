039/**
040 * Represents one CLI console session.
041 */
042public class CliSession extends ParseSession {
043
044    private final PrintWriter writer;
045    private final TreeMap<String, AbstractCommand> commands = new TreeMap<>();
046
047    private boolean done;
048    private boolean verbose;
049    private int lineLimit = 16;
050
051// Constructors
052
053    /**
054     * Constructor for core level access only.
055     *
056     * @param db core database
057     * @param writer console output
058     * @throws IllegalArgumentException if either parameter is null
059     */
060    public CliSession(Database db, PrintWriter writer) {
061        super(db);
062        if (writer == null)
063            throw new IllegalArgumentException("null writer");
064        this.writer = writer;
065    }
066
067    /**
068     * Constructor for {@link JSimpleDB} level access.
069     *
070     * @param jdb database
071     * @param writer console output
072     * @throws IllegalArgumentException if either parameter is null
073     */
074    public CliSession(JSimpleDB jdb, PrintWriter writer) {
075        super(jdb);
076        if (writer == null)
077            throw new IllegalArgumentException("null writer");
078        this.writer = writer;
079    }
080
081// Accessors
082
083    /**
084     * Get the output {@link PrintWriter} for this CLI session.
085     *
086     * @return output writer
087     */
088    public PrintWriter getWriter() {
089        return this.writer;
090    }
091
092    /**
093     * Get the {@link AbstractCommand}s registered with this instance.
094     *
095     * @return registered commands indexed by name

099    }

101    public int getLineLimit() {
102        return this.lineLimit;
103    }

105        this.lineLimit = lineLimit;
106    }
107
108    public boolean isVerbose() {
109        return this.verbose;
110    }
111    public void setVerbose(boolean verbose) {
112        this.verbose = verbose;
113    }
114
115    public boolean isDone() {
116        return this.done;
117    }
118    public void setDone(boolean done) {
119        this.done = done;
120    }

122// Command registration
123
124    /**
125     * Register the standard CLI built-in functions such as {@code print()}, {@code dump()}, etc.
126     */
127    @Override
128    public void registerStandardFunctions() {
129        super.registerStandardFunctions();
130
131        // We don't use AnnotatedClassScanner here to avoid having a dependency on the spring classes
132        this.registerFunction(DumpFunction.class);
133        this.registerFunction(PrintFunction.class);
134    }
135
136    /**
137     * Register the standard CLI built-in commands.
138     */
139    public void registerStandardCommands() {
140
141        // We don't use AnnotatedClassScanner here to avoid having a dependency on the spring classes
142        this.registerCommand(CompareSchemasCommand.class);
143        this.registerCommand(DeleteSchemaVersionCommand.class);
144        this.registerCommand(EvalCommand.class);
145        this.registerCommand(HelpCommand.class);
146        this.registerCommand(ImportCommand.class);
147        this.registerCommand(InfoCommand.class);
148        this.registerCommand(LoadCommand.class);
149        this.registerCommand(QuitCommand.class);
150        this.registerCommand(RaftAddCommand.class);
151        this.registerCommand(RaftRemoveCommand.class);
152        this.registerCommand(SaveCommand.class);
153        this.registerCommand(SetAllowNewSchemaCommand.class);
154        this.registerCommand(SetSchemaVersionCommand.class);
155        this.registerCommand(SetValidationModeCommand.class);
156        this.registerCommand(ShowAllSchemasCommand.class);
157        this.registerCommand(ShowSchemaCommand.class);
158    }
159
160    /**
161     * Create an instance of the specified class and register it as a {@link AbstractCommand}.
162     * The class must have a public constructor taking either a single {@link CliSession} parameter
163     * or no parameters; they will be tried in that order.
164     *
165     * @param cl command class
166     * @throws IllegalArgumentException if {@code cl} has no suitable constructor
167     * @throws IllegalArgumentException if {@code cl} instantiation fails
168     * @throws IllegalArgumentException if {@code cl} does not subclass {@link AbstractCommand}
169     */
170    public void registerCommand(Class<?> cl) {
171        if (!AbstractCommand.class.isAssignableFrom(cl))
172            throw new IllegalArgumentException(cl + " does not subclass " + AbstractCommand.class.getName());
173        final AbstractCommand command = this.instantiate(cl.asSubclass(AbstractCommand.class));
174        this.commands.put(command.getName(), command);
175    }
176
177    private <T> T instantiate(Class<T> cl) {
178        Throwable failure;
179        try {
180            return cl.getConstructor(CliSession.class).newInstance(this);
181        } catch (NoSuchMethodException e) {
182            try {
183                return cl.getConstructor().newInstance();
184            } catch (NoSuchMethodException e2) {
185                throw new IllegalArgumentException("no suitable constructor found in class " + cl.getName());
186            } catch (Exception e2) {
187                failure = e2;
188            }
189        } catch (Exception e) {
190            failure = e;
191        }
192        if (failure instanceof InvocationTargetException)
193            failure = failure.getCause();
194        throw new IllegalArgumentException("unable to instantiate class " + cl.getName() + ": " + failure, failure);
195    }
196
197// Errors
198
199    @Override
200    protected void reportException(Exception e) {
201        final String message = e.getLocalizedMessage();
202        if (e instanceof ParseException && message != null)
203            this.writer.println("Error: " + message);
204        else
205            this.writer.println("Error: " + e.getClass().getSimpleName() + (message != null ? ": " + message : ""));
206        if (this.verbose || this.showStackTrace(e))
207            e.printStackTrace(this.writer);
208    }
209
210    protected boolean showStackTrace(Exception e) {
211        return e instanceof NullPointerException || (e instanceof ParseException && e.getLocalizedMessage() == null);
212    }
213
214// Action
215
216    /**
217     * Perform the given action. This is a convenience method, equivalent to: {@code perform(null, action)}
218     *
219     * @param action action to perform
220     * @return true if {@code action} completed successfully, false if the transaction could not be created
221     *  or {@code action} threw an exception
222     * @throws IllegalArgumentException if {@code action} is null
223     */
224    public boolean perform(Action action) {
225        return this.perform(null, action);
226    }
227
228    /**
229     * Perform the given action within the given existing transaction, if any, otherwise within a new transaction.
230     * If {@code tx} is not null, it will used and left open when this method returns. Otherwise,
231     * if there is already an open transaction associated with this instance, it will be used;
232     * otherwise, a new transaction is created for the duration of {@code action} and then committed.
233     *
234     * <p>
235     * If {@code tx} is not null and there is already an open transaction associated with this instance and they
236     * are not the same transaction, an {@link IllegalStateException} is thrown.
237     * </p>
238     *
239     * @param tx transaction in which to perform the action, or null to create a new one (if necessary)
240     * @param action action to perform
241     * @return true if {@code action} completed successfully, false if the transaction could not be created
242     *  or {@code action} threw an exception
243     * @throws IllegalStateException if {@code tx} conflict with the already an open transaction associated with this instance
244     * @throws IllegalArgumentException if {@code action} is null
245     */
246    public boolean perform(Transaction tx, final Action action) {
247        return this.perform(tx, new Session.Action() {
248            @Override
249            public void run(Session session) throws Exception {
250                action.run((CliSession)session);
251            }
252        });
253    }
254
255    /**
256     * Callback interface used by {@link CliSession#perform CliSession.perform()}.
257     */
258    public interface Action {
259
260        /**
261         * Perform some action using the given {@link CliSession} while a transaction is open.
262         *
263         * @param session session with open transaction
264         * @throws Exception if an error occurs
265         */
266        void run(CliSession session) throws Exception;
267    }
268}
