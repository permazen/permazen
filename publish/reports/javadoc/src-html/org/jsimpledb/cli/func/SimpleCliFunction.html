006package org.jsimpledb.cli.func;
007
008import org.jsimpledb.cli.CliSession;
009import org.jsimpledb.parse.ParseSession;
010import org.jsimpledb.parse.expr.Value;
011import org.jsimpledb.parse.func.SimpleFunction;
012
013/**
014 * CLI extension of {@link SimpleFunction}.
015 */
016public abstract class SimpleCliFunction extends SimpleFunction {
017
018// Constructors
019
020    /**
021     * Constructor.
022     *
023     * @param name function name
024     * @param minArgs minimum number of arguments (inclusive)
025     * @param maxArgs maximum number of arguments (inclusive)
026     */
027    protected SimpleCliFunction(String name, int minArgs, int maxArgs) {
028        super(name, minArgs, maxArgs);
029    }
030
031    @Override
032    protected final Value apply(ParseSession session, Value[] params) {
033        return this.apply((CliSession)session, params);
034    }
035
036    /**
037     * Apply this function to the given values.
038     *
039     * @param session CLI session
040     * @param params parsed parameters; will already be checked between {@link #getMinArgs} and {@link #getMaxArgs}
041     * @return value returned by this function
042     * @throws RuntimeException if there is an error
043     */
044    protected abstract Value apply(CliSession session, Value[] params);
045}
