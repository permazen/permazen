006package org.jsimpledb.cli;
007
008import java.io.BufferedInputStream;
009import java.io.File;
010import java.io.FileDescriptor;
011import java.io.FileInputStream;
012import java.io.InputStream;
013import java.util.ArrayDeque;
014import java.util.ArrayList;
015import java.util.Arrays;
016import java.util.LinkedHashSet;
017
018import org.jsimpledb.JSimpleDB;
019import org.jsimpledb.cli.cmd.Command;
020import org.jsimpledb.core.Database;
021import org.jsimpledb.parse.func.Function;
022import org.jsimpledb.schema.SchemaModel;
023import org.jsimpledb.spring.AnnotatedClassScanner;
024import org.jsimpledb.util.AbstractMain;
025
026/**
027 * CLI main entry point.
028 */
029public class Main extends AbstractMain {
030
031    public static final String HISTORY_FILE = ".jsimpledb_history";
032
033    private File schemaFile;
034    private boolean coreMode;
035    private final LinkedHashSet<Class<?>> commandClasses = new LinkedHashSet<>();
036    private final LinkedHashSet<Class<?>> functionClasses = new LinkedHashSet<>();
037    private final ArrayList<String> oneShotCommands = new ArrayList<>();
038
039    @Override
040    protected boolean parseOption(String option, ArrayDeque<String> params) {
041        if (option.equals("--schema-file")) {
042            if (params.isEmpty())
043                this.usageError();
044            this.schemaFile = new File(params.removeFirst());
045            this.allowAutoDemo = false;
046        } else if (option.equals("--command") || option.equals("-c")) {
047            if (params.isEmpty())
048                this.usageError();
049            this.oneShotCommands.add(params.removeFirst());
050        } else if (option.equals("--core-mode"))
051            this.coreMode = true;
052        else if (option.equals("--cmd-pkg")) {
053            if (params.isEmpty())
054                this.usageError();
055            this.scanCommandClasses(params.removeFirst());
056        } else if (option.equals("--func-pkg")) {
057            if (params.isEmpty())
058                this.usageError();
059            this.scanFunctionClasses(params.removeFirst());
060        } else
061            return false;
062        return true;
063    }
064
065    private void scanCommandClasses(String pkgname) {
066        for (String className : new AnnotatedClassScanner(Command.class).scanForClasses(pkgname.split("[\\s,]")))
067            this.commandClasses.add(this.loadClass(className));
068    }
069
070    private void scanFunctionClasses(String pkgname) {
071        for (String className : new AnnotatedClassScanner(Function.class).scanForClasses(pkgname.split("[\\s,]")))
072            this.functionClasses.add(this.loadClass(className));
073    }
074
075    @Override
076    public int run(String[] args) throws Exception {
077
078        // Parse command line
079        final ArrayDeque<String> params = new ArrayDeque<String>(Arrays.asList(args));
080        final int result = this.parseOptions(params);
081        if (result != -1)
082            return result;
083        switch (params.size()) {
084        case 0:
085            break;
086        default:
087            this.usageError();
088            return 1;
089        }
090
091        // Read schema file from `--schema-file' (if any)
092        SchemaModel schemaModel = null;
093        if (this.schemaFile != null) {
094            try {
095                final InputStream input = new BufferedInputStream(new FileInputStream(this.schemaFile));
096                try {
097                    schemaModel = SchemaModel.fromXML(input);
098                } finally {
099                    input.close();
100                }
101            } catch (Exception e) {
102                System.err.println(this.getName() + ": can't load schema from `" + this.schemaFile + "': " + e.getMessage());
103                if (this.verbose)
104                    e.printStackTrace(System.err);
105                return 1;
106            }
107        }
108
109        // Set up Database
110        final Database db = this.startupKVDatabase();
111
112        // Load JSimpleDB layer, if specified
113        final JSimpleDB jdb = this.schemaClasses != null ? this.getJSimpleDBFactory(db).newJSimpleDB() : null;
114
115        // Sanity check consistent schema model if both --schema-file and --model-pkg were specified
116        if (jdb != null) {
117            if (schemaModel != null) {
118                if (!schemaModel.equals(jdb.getSchemaModel())) {
119                    System.err.println(this.getName() + ": schema from `" + this.schemaFile + "' conflicts with schema generated"
120                      + " from scanned classes");
121                    System.err.println(schemaModel.differencesFrom(jdb.getSchemaModel()));
122                    return 1;
123                }
124            } else
125                schemaModel = jdb.getSchemaModel();
126        }
127
128        // Core API mode or JSimpleDB mode?
129        this.coreMode |= jdb == null;
130
131        // Perform test transaction
132        try {
133            if (this.coreMode)
134                this.performTestTransaction(db, schemaModel);
135            else
136                this.performTestTransaction(jdb);
137        } catch (Exception e) {
138            System.err.println(this.getName() + ": warning: test transaction failed: " + e.getMessage());
139            if (this.verbose)
140                e.printStackTrace(System.err);
141        }
142
143        // Set up console
144        final Console console = this.coreMode ?
145          new Console(db, new FileInputStream(FileDescriptor.in), System.out) :
146          new Console(jdb, new FileInputStream(FileDescriptor.in), System.out);
147        console.setHistoryFile(new File(new File(System.getProperty("user.home")), HISTORY_FILE));
148
149        // Set up CLI session
150        final CliSession session = console.getSession();
151        session.setDatabaseDescription(this.getDatabaseDescription());
152        session.setReadOnly(this.readOnly);
153        session.setVerbose(this.verbose);
154        session.setSchemaModel(schemaModel);
155        session.setSchemaVersion(this.schemaVersion);
156        session.setAllowNewSchema(this.allowNewSchema);
157        session.registerStandardFunctions();
158        session.registerStandardCommands();
159        try {
160            for (Class<?> cl : this.commandClasses) {
161                final Command annotation = cl.getAnnotation(Command.class);
162                if (!this.coreMode || annotation.worksInCoreAPIMode())
163                    session.registerCommand(cl);
164            }
165            for (Class<?> cl : this.functionClasses) {
166                final Function annotation = cl.getAnnotation(Function.class);
167                if (!this.coreMode || annotation.worksInCoreAPIMode())
168                    session.registerFunction(cl);
169            }
170        } catch (IllegalArgumentException e) {
171            System.err.println(this.getName() + ": " + e.getMessage());
172            if (this.verbose)
173                e.printStackTrace(System.err);
174            return 1;
175        }
176
177        // Handle one-shot command mode
178        if (!this.oneShotCommands.isEmpty()) {
179            for (String text : this.oneShotCommands) {
180                for (CliSession.Action action : console.parseCommand(text)) {
181                    if (!session.perform(action))
182                        return 1;
183                }
184            }
185            return 0;
186        }
187
188        // Run console
189        console.run();
190
191        // Shut down KV database
192        this.shutdownKVDatabase();
193
194        // Done
195        return 0;
196    }
197
198    @Override
199    protected String getName() {
200        return "jsimpledb";
201    }
202
203    @Override
204    protected void usageMessage() {
205        System.err.println("Usage:");
206        System.err.println("  " + this.getName() + " [options]");
207        System.err.println("Options:");
208        this.outputFlags(new String[][] {
209          { "--schema-file file",       "Load core database schema from XML file" },
210          { "--cmd-pkg package",        "Register @Command-annotated classes found under the specified Java package" },
211          { "--func-pkg package",       "Register @Function-annotated classes found under the specified Java package" },
212          { "--core-mode",              "Force core API mode even though Java model classes are provided" },
213          { "--command, -c command",    "Execute the given command and then exit (may be repeated)" },
214        });
215    }
216
217    public static void main(String[] args) throws Exception {
218        new Main().doMain(args);
219    }
220}
