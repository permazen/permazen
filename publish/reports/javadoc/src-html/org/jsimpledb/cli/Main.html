008package org.jsimpledb.cli;
009
010import java.io.BufferedInputStream;
011import java.io.File;
012import java.io.FileDescriptor;
013import java.io.FileInputStream;
014import java.io.InputStream;
015import java.util.ArrayDeque;
016import java.util.Arrays;
017import java.util.LinkedHashSet;
018
019import org.jsimpledb.JSimpleDB;
020import org.jsimpledb.cli.cmd.Command;
021import org.jsimpledb.cli.func.SimpleCliFunction;
022import org.jsimpledb.core.Database;
023import org.jsimpledb.parse.func.Function;
024import org.jsimpledb.schema.SchemaModel;
025import org.jsimpledb.spring.AnnotatedClassScanner;
026import org.jsimpledb.util.AbstractMain;
027
028/**
029 * CLI main entry point.
030 */
031public class Main extends AbstractMain {
032
033    private File schemaFile;
034    private final LinkedHashSet<Class<?>> commandClasses = new LinkedHashSet<>();
035    private final LinkedHashSet<Class<?>> functionClasses = new LinkedHashSet<>();
036
037    @Override
038    protected boolean parseOption(String option, ArrayDeque<String> params) {
039        if (option.equals("--schema-file")) {
040            if (params.isEmpty())
041                this.usageError();
042            this.schemaFile = new File(params.removeFirst());
043        } else if (option.equals("--cmdpkg")) {
044            if (params.isEmpty())
045                this.usageError();
046            this.scanCommandClasses(params.removeFirst());
047        } else if (option.equals("--funcpkg")) {
048            if (params.isEmpty())
049                this.usageError();
050            this.scanFunctionClasses(params.removeFirst());
051        } else
052            return false;
053        return true;
054    }
055
056    private void scanCommandClasses(String pkgname) {
057        for (String className : new AnnotatedClassScanner(Command.class).scanForClasses(pkgname.split("[\\s,]")))
058            this.commandClasses.add(this.loadClass(className));
059    }
060
061    private void scanFunctionClasses(String pkgname) {
062        for (String className : new AnnotatedClassScanner(Function.class).scanForClasses(pkgname.split("[\\s,]")))
063            this.functionClasses.add(this.loadClass(className));
064    }
065
066    @Override
067    public int run(String[] args) throws Exception {
068
069        // Register built-in commands and functions
070        this.scanCommandClasses(Command.class.getPackage().getName());
071        this.scanFunctionClasses(Function.class.getPackage().getName());
072        this.scanFunctionClasses(SimpleCliFunction.class.getPackage().getName());
073
074        // Parse command line
075        final ArrayDeque<String> params = new ArrayDeque<String>(Arrays.asList(args));
076        final int result = this.parseOptions(params);
077        if (result != -1)
078            return result;
079        switch (params.size()) {
080        case 0:
081            break;
082        default:
083            this.usageError();
084            return 1;
085        }
086
087        // Read schema file from `--schema-file' (if any)
088        SchemaModel schemaModel = null;
089        if (this.schemaFile != null) {
090            try {
091                final InputStream input = new BufferedInputStream(new FileInputStream(this.schemaFile));
092                try {
093                    schemaModel = SchemaModel.fromXML(input);
094                } finally {
095                    input.close();
096                }
097            } catch (Exception e) {
098                System.err.println(this.getName() + ": can't load schema from `" + this.schemaFile + "': " + e.getMessage());
099                if (this.verbose)
100                    e.printStackTrace(System.err);
101                return 1;
102            }
103        }
104
105        // Start up KV database
106        this.startupKVDatabase();
107        final Database db = new Database(this.kvdb);
108
109        // Load JSimpleDB layer, if specified
110        final JSimpleDB jdb = this.schemaClasses != null ? new JSimpleDB(db, this.schemaVersion, this.schemaClasses) : null;
111
112        // Sanity check consistent schema model if both --schema-file and --schema-pkg were specified
113        if (jdb != null && schemaModel != null) {
114            if (!schemaModel.equals(jdb.getSchemaModel())) {
115                System.err.println(this.getName() + ": schema from `" + this.schemaFile + "' conflicts with schema generated"
116                  + " from scanned classes " + this.schemaClasses);
117                return 1;
118            }
119        }
120
121        // Set up console
122        final Console console = jdb != null ?
123          new Console(jdb, new FileInputStream(FileDescriptor.in), System.out) :
124          new Console(db, new FileInputStream(FileDescriptor.in), System.out);
125        final CliSession session = console.getSession();
126        console.setHistoryFile(new File(new File(System.getProperty("user.home")), ".jsimpledb_history"));
127        session.setReadOnly(this.readOnly);
128        session.setVerbose(this.verbose);
129        session.setSchemaModel(schemaModel);
130        session.setSchemaVersion(this.schemaVersion);
131        session.setAllowNewSchema(this.allowNewSchema);
132        try {
133            for (Class<?> cl : this.commandClasses) {
134                final Command annotation = cl.getAnnotation(Command.class);
135                if (jdb != null || annotation.worksInCoreAPIMode())
136                    session.registerCommand(cl);
137            }
138            for (Class<?> cl : this.functionClasses) {
139                final Function annotation = cl.getAnnotation(Function.class);
140                if (jdb != null || annotation.worksInCoreAPIMode())
141                    session.registerFunction(cl);
142            }
143        } catch (IllegalArgumentException e) {
144            System.err.println(this.getName() + ": " + e.getMessage());
145            if (this.verbose)
146                e.printStackTrace(System.err);
147            return 1;
148        }
149
150        // Run console
151        console.run();
152
153        // Shut down KV database
154        this.shutdownKVDatabase();
155
156        // Done
157        return 0;
158    }
159
160    @Override
161    protected String getName() {
162        return "jsimpledb";
163    }
164
165    @Override
166    protected void usageMessage() {
167        System.err.println("Usage:");
168        System.err.println("  " + this.getName() + " [options]");
169        System.err.println("Options:");
170        this.outputFlags(new String[][] {
171          { "--schema-file file",   "Load core database schema from XML file" },
172          { "--cmdpkg package",     "Register @Command-annotated classes found under the specified Java package" },
173          { "--funcpkg package",    "Register @Function-annotated classes found under the specified Java package" },
174        });
175    }
176
177    public static void main(String[] args) throws Exception {
178        new Main().doMain(args);
179    }
180}
