008package org.jsimpledb.cli;
009
010import com.google.common.collect.Iterables;
011
012import org.jsimpledb.util.ParseContext;
013
014public class LimitCommand extends AbstractCommand {
015
016    public LimitCommand() {
017        super("limit");
018    }
019
020    @Override
021    public String getUsage() {
022        return this.name + " [offset] count";
023    }
024
025    @Override
026    public String getHelpSummary() {
027        return "discards items beyond a maximum count, and optionally prior to a minimum offset";
028    }
029
030    @Override
031    public String getHelpDetail() {
032        return "Truncates data items in the top channel after the specified number have been copied. If an offset is specified,"
033          + " that many data items are skipped over first.";
034    }
035
036    @Override
037    public Action parseParameters(Session session, ParseContext ctx) {
038
039        // Parse parameters
040        final ParamParser parser = new ParamParser(1, 2, this.getUsage()).parse(ctx);
041        final String offsetParam = parser.getParam(0);
042        final int offset;
043        if (parser.getParams().size() > 1) {
044            try {
045                offset = Integer.parseInt(offsetParam);
046                if (offset < 0)
047                    throw new IllegalArgumentException("offset is negative");
048            } catch (IllegalArgumentException e) {
049                throw new ParseException(ctx, "invalid offset `" + offsetParam + "'");
050            }
051        } else
052            offset = 0;
053        final String limitParam = parser.getParam(parser.getParams().size() == 1 ? 0 : 1);
054        final int limit;
055        try {
056            limit = Integer.parseInt(limitParam);
057            if (limit < 0)
058                throw new IllegalArgumentException("limit is negative");
059        } catch (IllegalArgumentException e) {
060            throw new ParseException(ctx, "invalid limit `" + limitParam + "'");
061        }
062
063        // Return action
064        return new Action() {
065            @Override
066            public void run(Session session) throws Exception {
067                LimitCommand.this.limit(session, LimitCommand.this.pop(session), offset, limit);
068            }
069        };
070    }
071
072    // This method exists solely to bind the generic type parameters
073    private <T> void limit(Session session, final Channel<T> channel, final int offset, final int limit) throws Exception {
074        LimitCommand.this.push(session, new AbstractChannel<T>(channel.getItemType()) {
075            @Override
076            public Iterable<T> getItems(Session session) {
077                return Iterables.limit(Iterables.skip(channel.getItems(session), offset), limit);
078            }
079        });
080    }
081}
