036import jline.console.completer.FileNameCompleter;
037
038@Command
039public class SaveCommand extends AbstractCommand {
040
041    public SaveCommand() {
042        super("save --storage-id-format:storageIdFormat file.xml:file expr:expr");
043    }
044
045    @Override
046    public String getHelpSummary() {
047        return "exports objects to an XML file";
048    }
049
050    @Override
051    public String getHelpDetail() {
052        return "Evaluates the expression, which must evaluate to an Iterator (or Iterable) of database objects,"
053          + " and writes the objects to the specified XML file. Objects can be read back in later via `load'.";
054    }
055
056    @Override
057    protected Parser<?> getParser(String typeName) {
058        return "file".equals(typeName) ? new FileParser() : super.getParser(typeName);
059    }
060
061    @Override
062    public CliSession.Action getAction(CliSession session, ParseContext ctx, boolean complete, Map<String, Object> params) {
063
064        // Parse parameters
065        final boolean nameFormat = !params.containsKey("storageIdFormat");
066        final File file = (File)params.get("file.xml");
067        final Node expr = (Node)params.get("expr");
068
069        // Return action
070        return new CliSession.Action() {
071            @Override
072            public void run(CliSession session) throws Exception {
073                final Value value = expr.evaluate(session);
074                final Iterable<?> i = value.checkType(session, "save", Iterable.class);
075                final AtomicUpdateFileOutputStream updateOutput = new AtomicUpdateFileOutputStream(file);
076                final BufferedOutputStream output = new BufferedOutputStream(updateOutput);
077                boolean success = false;
078                final int count;
079                try {
080                    final XMLStreamWriter writer = new IndentXMLStreamWriter(
081                      XMLOutputFactory.newInstance().createXMLStreamWriter(output, "UTF-8"));
082                    writer.writeStartDocument("UTF-8", "1.0");
083                    count = new XMLObjectSerializer(session.getTransaction()).write(writer, nameFormat,
084                      Iterables.transform(i, new ParseCastFunction<ObjId>(ObjId.class) {
085                        @Override
086                        public ObjId apply(Object obj) {
087                            return obj instanceof JObject ? ((JObject)obj).getObjId() : super.apply(obj);
088                        }
089                      }));
090                    success = true;
091                } finally {
092                    if (success) {
093                        try {
094                            output.close();
095                        } catch (IOException e) {
096                            // ignore
097                        }
098                    } else
099                        updateOutput.cancel();
100                }
101                session.getWriter().println("Wrote " + count + " objects to `" + file + "'");
102            }
103        };
104    }
105
106// FileParser
107
108    private class FileParser implements Parser<File> {
109
110        @Override
111        public File parse(ParseSession session, ParseContext ctx, boolean complete) {
112
113            // Get filename
114            final Matcher matcher = ctx.tryPattern("[^\\s;]*");
115            if (matcher == null)
116                throw new ParseException(ctx);
117            final String path = matcher.group();
118
119            // Check file
120            final File file = new File(path);
121            if (file.isDirectory() || (!file.exists() && complete)) {
122                final ArrayList<CharSequence> list = new ArrayList<>();
123                final int index = new FileNameCompleter().complete(path, path.length(), list);
124                throw new ParseException(ctx, "can't write to file `" + path + "'").addCompletions(
125                  Lists.transform(Lists.transform(list, new ParseCastFunction<String>(String.class)),
126                    new StripPrefixFunction(path.substring(index))));
127            }
128
129            // Done
130            return file;
131        }
132    }
133}
