006package org.jsimpledb.cli.cmd;
007
008import java.util.Map;
009import java.util.regex.Matcher;
010import java.util.regex.Pattern;
011
012import org.jsimpledb.cli.CliSession;
013import org.jsimpledb.cli.ParamParser;
014import org.jsimpledb.parse.FieldTypeParser;
015import org.jsimpledb.parse.ObjIdParser;
016import org.jsimpledb.parse.ObjTypeParser;
017import org.jsimpledb.parse.ParseContext;
018import org.jsimpledb.parse.ParseSession;
019import org.jsimpledb.parse.Parser;
020import org.jsimpledb.parse.expr.ExprParser;
021import org.slf4j.Logger;
022import org.slf4j.LoggerFactory;
023
024/**
025 * Superclass of all CLI commands.
026 *
027 * @see Command
028 */
029public abstract class AbstractCommand implements Parser<CliSession.Action> {
030
031    protected final Logger log = LoggerFactory.getLogger(this.getClass());
032    protected final String name;
033    protected final ParamParser paramParser;
034
035// Constructors
036
037    /**
038     * Constructor.
039     *
040     * @param spec {@link ParamParser} spec string, possibly containing custom type names (see {@link #getParser getParser()})
041     */
042    protected AbstractCommand(String spec) {
043        if (spec == null)
044            throw new IllegalArgumentException("null spec");
045        final Matcher matcher = Pattern.compile("([^\\s]+)(\\s+(.*))?").matcher(spec);
046        if (!matcher.matches())
047            throw new IllegalArgumentException("invalid command specification `" + spec + "'");
048        this.name = matcher.group(1);
049        final String paramSpec = matcher.group(3);
050        this.paramParser = new ParamParser(paramSpec != null ? paramSpec : "") {
051            @Override
052            protected Parser<?> getParser(String typeName) {
053                try {
054                    return super.getParser(typeName);
055                } catch (IllegalArgumentException e) {
056                    return AbstractCommand.this.getParser(typeName);
057                }
058            }
059        };
060    }
061
062// Command stuff
063
064    /**
065     * Get the name of this command.
066     *
067     * @return command name
068     */
069    public String getName() {
070        return this.name;
071    }
072
073    /**
074     * Get command usage string.
075     *
076     * <p>
077     * The implementation in {@link Command} delegates to {@link ParamParser#getUsage ParamParser.getUsage()}.
078     * </p>
079     *
080     * @return command usage string
081     */
082    public String getUsage() {
083        return this.paramParser.getUsage(this.name);
084    }
085
086    /**
087     * Get summarized help (typically a single line).
088     *
089     * @return one line command summary
090     */
091    public abstract String getHelpSummary();
092
093    /**
094     * Get expanded help (typically multiple lines).
095     *
096     * <p>
097     * The implementation in {@link Command} delegates to {@link #getHelpSummary getHelpSummary()}.
098     * </p>
099     *
100     * @return detailed command description
101     */
102    public String getHelpDetail() {
103        return this.getHelpSummary();
104    }
105
106    /**
107     * Parse command line and return command action.
108     *
109     * <p>
110     * The implementation in {@link ParamParser} parses the parameters and delegates to {@link #getAction} with the result.
111     * </p>
112     *
113     * @param session CLI session
114     * @param ctx input to parse
115     * @param complete false if parse is "for real", true if only for tab completion calculation
116     * @return action to perform for the parsed command
117     * @throws org.jsimpledb.parse.ParseException if parse fails, or if {@code complete} is true and there are valid completions
118     * @throws org.jsimpledb.parse.ParseException if parameters are invalid
119     */
120    @Override
121    public CliSession.Action parse(ParseSession session, ParseContext ctx, boolean complete) {
122        return this.getAction((CliSession)session, ctx, complete, this.paramParser.parse(session, ctx, complete));
123    }
124
125    /**
126     * Process command line parameters and return action.
127     *
128     * @param session CLI session
129     * @param ctx input to parse
130     * @param complete false if parse is "for real", true if only for tab completion calculation
131     * @param params parsed parameters indexed by name
132     * @return action to perform for the parsed command
133     * @throws org.jsimpledb.parse.ParseException if parse fails, or if {@code complete} is true and there are valid completions
134     * @throws org.jsimpledb.parse.ParseException if parameters are invalid
135     */
136    public abstract CliSession.Action getAction(CliSession session, ParseContext ctx, boolean complete, Map<String, Object> params);
137
138    /**
139     * Convert parameter spec type name into a {@link Parser}. Used for custom type names not supported by {@link ParamParser}.
140     *
141     * <p>
142     * The implementation in {@link ParamParser} supports all {@link org.jsimpledb.core.FieldType}s registered with the database,
143     * {@code type} for an object type name (returns {@link Integer}), and {@code objid} for an object ID
144     * (returns {@link org.jsimpledb.core.ObjId}).
145     * </p>
146     *
147     * @param typeName parameter type name
148     * @return parser for parameters of the specified type
149     */
150    protected Parser<?> getParser(String typeName) {
151        if (typeName == null)
152            throw new IllegalArgumentException("null typeName");
153        if (typeName.equals("type"))
154            return new ObjTypeParser();
155        if (typeName.equals("objid"))
156            return new ObjIdParser();
157        if (typeName.equals("expr"))
158            return new ExprParser();
159        return FieldTypeParser.getFieldTypeParser(typeName);
160    }
161}
