006package org.jsimpledb.cli.cmd;
007
008import java.util.Map;
009
010import org.jsimpledb.cli.CliSession;
011import org.jsimpledb.kv.KVTransaction;
012import org.jsimpledb.kv.raft.RaftKVTransaction;
013import org.jsimpledb.parse.ParseContext;
014
015@Command
016public class RaftAddCommand extends AbstractCommand {
017
018    public RaftAddCommand() {
019        super("raft-add node address");
020    }
021
022    @Override
023    public String getHelpSummary() {
024        return "adds a node to a Raft clustered database";
025    }
026
027    @Override
028    public CliSession.Action getAction(CliSession session, ParseContext ctx, boolean complete, Map<String, Object> params) {
029        final String node = (String)params.get("node");
030        final String address = (String)params.get("address");
031        return new CliSession.Action() {
032            @Override
033            public void run(CliSession session) throws Exception {
034                final KVTransaction kvtx = session.getTransaction().getKVTransaction();
035                if (!(kvtx instanceof RaftKVTransaction))
036                    throw new Exception("key/value store is not Raft");
037                final RaftKVTransaction raftx = (RaftKVTransaction)kvtx;
038                raftx.configChange(node, address);
039            }
040        };
041    }
042}
