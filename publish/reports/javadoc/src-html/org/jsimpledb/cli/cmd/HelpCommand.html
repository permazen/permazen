006package org.jsimpledb.cli.cmd;
007
008import java.io.PrintWriter;
009import java.util.HashSet;
010import java.util.Map;
011
012import org.jsimpledb.cli.CliSession;
013import org.jsimpledb.parse.ParseContext;
014import org.jsimpledb.parse.Parser;
015import org.jsimpledb.parse.WordParser;
016import org.jsimpledb.parse.func.AbstractFunction;
017
018@Command
019public class HelpCommand extends AbstractCommand {
020
021    private final CliSession session;
022
023    public HelpCommand(CliSession session) {
024        super("help command-or-function:cmdfunc?");
025        this.session = session;
026    }
027
028    @Override
029    public String getHelpSummary() {
030        return "display help information";
031    }
032
033    @Override
034    public String getHelpDetail() {
035        return "Displays the list of known commands and functions, or help information about a specific command or function.";
036    }
037
038    @Override
039    protected Parser<?> getParser(String typeName) {
040        if ("cmdfunc".equals(typeName)) {
041            return new WordParser("command/function") {
042                @Override
043                protected HashSet<String> getWords() {
044                    final HashSet<String> names = new HashSet<>();
045                    names.addAll(HelpCommand.this.session.getCommands().keySet());
046                    names.addAll(HelpCommand.this.session.getFunctions().keySet());
047                    return names;
048                }
049            };
050        }
051        return super.getParser(typeName);
052    }
053
054    @Override
055    public CliSession.Action getAction(CliSession session, ParseContext ctx, boolean complete, Map<String, Object> params) {
056        final String name = (String)params.get("command-or-function");
057        return new CliSession.Action() {
058            @Override
059            public void run(CliSession session) throws Exception {
060                final PrintWriter writer = session.getWriter();
061                if (name == null) {
062                    writer.println("Available commands:");
063                    for (AbstractCommand availableCommand : session.getCommands().values())
064                        writer.println(String.format("%24s - %s", availableCommand.getName(), availableCommand.getHelpSummary()));
065                    writer.println("Available functions:");
066                    for (AbstractFunction availableFunction : session.getFunctions().values())
067                        writer.println(String.format("%24s - %s", availableFunction.getName(), availableFunction.getHelpSummary()));
068                } else {
069                    final AbstractCommand command = session.getCommands().get(name);
070                    if (command != null) {
071                        writer.println("Usage: " + command.getUsage());
072                        writer.println(command.getHelpDetail());
073                    }
074                    final AbstractFunction function = session.getFunctions().get(name);
075                    if (function != null) {
076                        writer.println("Usage: " + function.getUsage());
077                        writer.println(function.getHelpDetail());
078                    }
079                    if (command == null && function == null)
080                        writer.println("No command or function named `" + name + "' exists.");
081                }
082            }
083        };
084    }
085}
