006package org.jsimpledb.cli.cmd;
007
008import java.util.Map;
009
010import org.jsimpledb.cli.CliSession;
011import org.jsimpledb.parse.ParseContext;
012
013@Command
014public class SetVerboseCommand extends AbstractCommand {
015
016    public SetVerboseCommand() {
017        super("set-verbose verbose:boolean");
018    }
019
020    @Override
021    public String getHelpSummary() {
022        return "Enables or disables verbose mode, which displays the complete stack trace when an exceptions occurs.";
023    }
024
025    @Override
026    public CliSession.Action getAction(CliSession session, ParseContext ctx, boolean complete, Map<String, Object> params) {
027        final boolean verbose = (Boolean)params.get("verbose");
028        return new CliSession.Action() {
029            @Override
030            public void run(CliSession session) throws Exception {
031                session.setVerbose(verbose);
032                session.getWriter().println((verbose ? "En" : "Dis") + "abled verbose mode.");
033            }
034        };
035    }
036}
