006package org.jsimpledb.cli.cmd;
007
008import java.util.Map;
009
010import org.jsimpledb.cli.CliSession;
011import org.jsimpledb.parse.ParseContext;
012
013@Command
014public class DeleteSchemaVersionCommand extends AbstractCommand {
015
016    public DeleteSchemaVersionCommand() {
017        super("delete-schema-version version:int");
018    }
019
020    @Override
021    public String getHelpSummary() {
022        return "Deletes the specified schema version from the database";
023    }
024
025    @Override
026    public String getHelpDetail() {
027        return "This command deletes a schema version recorded in the database. The schema version must not be the"
028          + " currently configured schema version and there must not be any objects having that version in the database";
029    }
030
031    @Override
032    public CliSession.Action getAction(CliSession session, ParseContext ctx, boolean complete, Map<String, Object> params) {
033        final int version = (Integer)params.get("version");
034        return new CliSession.Action() {
035            @Override
036            public void run(CliSession session) throws Exception {
037                final boolean deleted = session.getTransaction().deleteSchemaVersion(version);
038                session.getWriter().println("Schema version " + version + " " + (deleted ? "deleted" : "not found"));
039            }
040        };
041    }
042}
