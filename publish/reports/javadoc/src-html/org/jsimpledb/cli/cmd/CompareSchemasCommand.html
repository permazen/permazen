006package org.jsimpledb.cli.cmd;
007
008import java.util.Map;
009
010import org.jsimpledb.cli.CliSession;
011import org.jsimpledb.core.Schema;
012import org.jsimpledb.parse.ParseContext;
013import org.jsimpledb.util.Diffs;
014
015@Command
016public class CompareSchemasCommand extends AbstractCommand {
017
018    public CompareSchemasCommand() {
019        super("compare-schemas version1:int version2:int");
020    }
021
022    @Override
023    public String getHelpSummary() {
024        return "Shows the differences between two schema versions recorded in the database";
025    }
026
027    @Override
028    public CliSession.Action getAction(CliSession session, ParseContext ctx, boolean complete, Map<String, Object> params) {
029        final int version1 = (Integer)params.get("version1");
030        final int version2 = (Integer)params.get("version2");
031        return new CliSession.Action() {
032            @Override
033            public void run(CliSession session) throws Exception {
034                final Schema schema1 = session.getTransaction().getSchemas().getVersions().get(version1);
035                if (schema1 == null)
036                    session.getWriter().println("Schema version " + version1 + " " + "not found");
037                final Schema schema2 = session.getTransaction().getSchemas().getVersions().get(version2);
038                if (schema2 == null)
039                    session.getWriter().println("Schema version " + version2 + " " + "not found");
040                final Diffs diffs = schema2.getSchemaModel().differencesFrom(schema1.getSchemaModel());
041                if (diffs.isEmpty())
042                    session.getWriter().println("No differences found between schema versions " + version1 + " and " + version2);
043                else {
044                    session.getWriter().println("Found differences found between schema versions " + version1 + " and "
045                      + version2 + "\n" + diffs);
046                }
047            }
048        };
049    }
050}
