006package org.jsimpledb.cli.cmd;
007
008import com.google.common.collect.Lists;
009
010import java.io.BufferedInputStream;
011import java.io.File;
012import java.io.FileInputStream;
013import java.io.IOException;
014import java.util.ArrayList;
015import java.util.Map;
016import java.util.regex.Matcher;
017
018import org.jsimpledb.cli.CliSession;
019import org.jsimpledb.parse.ParseContext;
020import org.jsimpledb.parse.ParseException;
021import org.jsimpledb.parse.ParseSession;
022import org.jsimpledb.parse.Parser;
023import org.jsimpledb.parse.util.ParseCastFunction;
024import org.jsimpledb.parse.util.StripPrefixFunction;
025import org.jsimpledb.util.XMLObjectSerializer;
026
027import jline.console.completer.FileNameCompleter;
028
029@Command
030public class LoadCommand extends AbstractCommand {
031
032    public LoadCommand() {
033        super("load file.xml:file");
034    }
035
036    @Override
037    public String getHelpSummary() {
038        return "Imports objects from an XML file created previously via `save'";
039    }
040
041    @Override
042    protected Parser<?> getParser(String typeName) {
043        return "file".equals(typeName) ? new FileParser() : super.getParser(typeName);
044    }
045
046    @Override
047    public CliSession.Action getAction(CliSession session, ParseContext ctx, boolean complete, Map<String, Object> params) {
048        final File file = (File)params.get("file.xml");
049
050        // Return import action
051        return new CliSession.Action() {
052            @Override
053            public void run(CliSession session) throws Exception {
054                final BufferedInputStream input = new BufferedInputStream(new FileInputStream(file));
055                final int count;
056                try {
057                    count = new XMLObjectSerializer(session.getTransaction()).read(input);
058                } finally {
059                    try {
060                        input.close();
061                    } catch (IOException e) {
062                        // ignore
063                    }
064                }
065                session.getWriter().println("Read " + count + " objects from `" + file + "'");
066            }
067        };
068    }
069
070// FileParser
071
072    private class FileParser implements Parser<File> {
073
074        @Override
075        public File parse(ParseSession session, ParseContext ctx, boolean complete) {
076
077            // Get filename
078            final Matcher matcher = ctx.tryPattern("[^\\s;]*");
079            if (matcher == null)
080                throw new ParseException(ctx);
081            final String path = matcher.group();
082
083            // Check file
084            final File file = new File(path);
085            if (!file.exists() || file.isDirectory() || !file.canRead()) {
086                final ArrayList<CharSequence> list = new ArrayList<>();
087                final int index = new FileNameCompleter().complete(path, path.length(), list);
088                throw new ParseException(ctx, "can't read file `" + file + "'").addCompletions(
089                  Lists.transform(Lists.transform(list, new ParseCastFunction<String>(String.class)),
090                    new StripPrefixFunction(path.substring(index))));
091            }
092
093            // Done
094            return file;
095        }
096    }
097}
