006package org.jsimpledb.cli;
007
008import org.jsimpledb.parse.ParseContext;
009import org.jsimpledb.parse.ParseSession;
010import org.jsimpledb.parse.Parser;
011import org.jsimpledb.parse.WordParser;
012
013public class CommandParser implements Parser<CliSession.Action> {
014
015    /**
016     * Parse a command and return the corresponding {@link CliSession.Action}.
017     */
018    @Override
019    public CliSession.Action parse(ParseSession parseSession, ParseContext ctx, boolean complete) {
020        final CliSession session = (CliSession)parseSession;
021        final String commandName = new WordParser(session.getCommands().keySet(), "command").parse(session, ctx, complete);
022        return session.getCommands().get(commandName).parse(session, ctx, complete);
023    }
024}
