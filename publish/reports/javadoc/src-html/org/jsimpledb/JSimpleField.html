006package org.jsimpledb;
007
008import com.google.common.base.Converter;
009import com.google.common.reflect.TypeToken;
010
011import java.lang.reflect.Method;
012import java.util.ArrayList;
013import java.util.Collections;
014import java.util.Comparator;
015
016import org.dellroad.stuff.java.Primitive;
017import org.jsimpledb.core.FieldType;
018import org.jsimpledb.schema.SimpleSchemaField;
019import org.objectweb.asm.ClassWriter;
020import org.objectweb.asm.MethodVisitor;
021import org.objectweb.asm.Opcodes;
022import org.objectweb.asm.Type;
023
024/**
025 * Represents a simple field in a {@link JClass} or a simple sub-field of a complex field in a {@link JClass}.
026 */
027public class JSimpleField extends JField {
028
029    final TypeToken<?> typeToken;
030    final FieldType<?> fieldType;
031    final boolean indexed;
032    final boolean unique;
033    final ArrayList<Object> uniqueExcludes;         // note: these are core API values, sorted by this.fieldType
034    final Method setter;
035
036    JSimpleField(JSimpleDB jdb, String name, int storageId, TypeToken<?> typeToken, String typeName, boolean indexed,
037      org.jsimpledb.annotation.JField annotation, String description, Method getter, Method setter) {
038        this(jdb, name, storageId, typeToken,
039          jdb.db.getFieldTypeRegistry().getFieldType(typeName), indexed, annotation, description, getter, setter);
040    }
041
042    @SuppressWarnings("unchecked")
043    JSimpleField(JSimpleDB jdb, String name, int storageId, TypeToken<?> typeToken, FieldType<?> fieldType, boolean indexed,
044      org.jsimpledb.annotation.JField annotation, String description, Method getter, Method setter) {
045        super(jdb, name, storageId, description, getter);
046        if (typeToken == null)
047            throw new IllegalArgumentException("null typeToken");
048        if (fieldType == null)
049            throw new IllegalArgumentException("null fieldType");
050        if (annotation == null)
051            throw new IllegalArgumentException("null annotation");
052        this.typeToken = typeToken;
053        this.fieldType = fieldType;
054        this.indexed = indexed;
055        this.unique = annotation.unique();
056        this.setter = setter;
057
058        // Parse uniqueExcludes
059        final int numExcludes = annotation.uniqueExclude().length + (annotation.uniqueExcludeNull() ? 1 : 0);
060        if (numExcludes > 0) {
061            this.uniqueExcludes = new ArrayList<>(numExcludes);
062            if (annotation.uniqueExcludeNull())
063                this.uniqueExcludes.add(null);
064            for (String string : annotation.uniqueExclude()) {
065                try {
066                    this.uniqueExcludes.add(this.fieldType.fromString(string));
067                } catch (IllegalArgumentException e) {
068                    throw new IllegalArgumentException("invalid uniqueExclude() value `" + string + "': " + e.getMessage(), e);
069                }
070            }
071            Collections.sort(this.uniqueExcludes, (Comparator<Object>)this.fieldType);
072        } else
073            this.uniqueExcludes = null;
074    }
075
076    /**
077     * Get the {@link JComplexField} of which this instance is a sub-field, if any.
078     *
079     * @return parent {@link JComplexField}, or null if this instance is not a sub-field
080     */
081    public JComplexField getParentField() {
082        return this.parent instanceof JComplexField ? (JComplexField)this.parent : null;
083    }
084
085    /**
086     * Get the type of this field.
087     *
088     * @return this field's Java type
089     */
090    public TypeToken<?> getTypeToken() {
091        return this.typeToken;
092    }
093
094    /**
095     * Get the {@link org.jsimpledb.core.FieldType} used by the core API to encode this field's values.
096     *
097     * <p>
098     * Note that for {@link Enum} and reference fields, the core API uses a different type than the Java model
099     * classes ({@link org.jsimpledb.core.EnumValue} and {@link org.jsimpledb.core.ObjId}, respectively).
100     * Values can always be properly converted using the {@link Converter} returned by {@link #getConverter getConverter()}.
101     * </p>
102     *
103     * @return this field's core-layer type definition
104     */
105    public FieldType<?> getFieldType() {
106        return this.fieldType;
107    }
108
109    /**
110     * Get whether this field is indexed.
111     *
112     * @return whether this field is indexed
113     */
114    public boolean isIndexed() {
115        return this.indexed;
116    }
117
118    /**
119     * Get the setter method associated with this field.
120     *
121     * @return field property setter method, or null if this field is a sub-field of a complex field
122     */
123    public Method getSetter() {
124        return this.setter;
125    }
126
127    @Override
128    public Object getValue(JObject jobj) {
129        if (jobj == null)
130            throw new IllegalArgumentException("null jobj");
131        return jobj.getTransaction().readSimpleField(jobj, this.storageId, false);
132    }
133
134    @Override
135    public <R> R visit(JFieldSwitch<R> target) {
136        return target.caseJSimpleField(this);
137    }
138
139    /**
140     * Get a {@link Converter} that converts this field values between core API type and Java model type.
141     * Only {@link Enum} and reference types require conversion; for all other types, this returns an identity converter.
142     *
143     * @param jtx transaction
144     * @return {@link Converter} from core API field type to Java model field type
145     */
146    public Converter<?, ?> getConverter(JTransaction jtx) {
147        return Converter.<Object>identity();
148    }
149
150    /**
151     * Set the Java value of this field in the given object.
152     * Does not alter the schema version of the object.
153     *
154     * @param jobj object containing this field
155     * @param value new value
156     * @throws org.jsimpledb.core.DeletedObjectException if {@code jobj} does not exist in its associated {@link JTransaction}
157     * @throws org.jsimpledb.core.StaleTransactionException if the {@link JTransaction} associated with {@code jobj}
158     *  is no longer usable
159     * @throws IllegalArgumentException if {@code value} is not an appropriate value for this field
160     * @throws IllegalArgumentException if {@code jobj} is null
161     */
162    public void setValue(JObject jobj, Object value) {
163        if (jobj == null)
164            throw new IllegalArgumentException("null jobj");
165        jobj.getTransaction().writeSimpleField(jobj, this.storageId, value, false);
166    }
167
168    @Override
169    SimpleSchemaField toSchemaItem(JSimpleDB jdb) {
170        final SimpleSchemaField schemaField = new SimpleSchemaField();
171        this.initialize(jdb, schemaField);
172        return schemaField;
173    }
174
175    @Override
176    void calculateRequiresValidation() {
177        super.calculateRequiresValidation();
178        this.requiresValidation |= this.unique;
179    }
180
181    void initialize(JSimpleDB jdb, SimpleSchemaField schemaField) {
182        super.initialize(jdb, schemaField);
183        schemaField.setType(this.fieldType.getName());
184        schemaField.setIndexed(this.indexed);
185    }
186
187    @Override
188    void outputMethods(final ClassGenerator<?> generator, ClassWriter cw) {
189
190        // Getter
191        this.outputReadMethod(generator, cw, ClassGenerator.READ_SIMPLE_FIELD_METHOD);
192
193        // Setter
194        final Method writeMethod = ClassGenerator.WRITE_SIMPLE_FIELD_METHOD;
195        generator.overrideBeanMethod(cw, this.setter, this.storageId, new ClassGenerator.CodeEmitter() {
196            @Override
197            public void emit(MethodVisitor mv) {
198
199                // Push field value
200                mv.visitVarInsn(Type.getType(JSimpleField.this.typeToken.getRawType()).getOpcode(Opcodes.ILOAD), 1);
201
202                // Wrap result if needed
203                if (JSimpleField.this.typeToken.isPrimitive())
204                    generator.wrap(mv, Primitive.get(JSimpleField.this.typeToken.getRawType()));
205
206                // Push "true"
207                mv.visitInsn(Opcodes.ICONST_1);
208
209                // Invoke Transaction.writeSimpleField()
210                mv.visitMethodInsn(Opcodes.INVOKEVIRTUAL, Type.getInternalName(JTransaction.class),
211                  writeMethod.getName(), Type.getMethodDescriptor(writeMethod), false);
212            }
213        });
214    }
215
216    @Override
217    final JSimpleFieldInfo toJFieldInfo() {
218        return this.toJFieldInfo(0);
219    }
220
221    JSimpleFieldInfo toJFieldInfo(int parentStorageId) {
222        return new JSimpleFieldInfo(this, parentStorageId);
223    }
224}
