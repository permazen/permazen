006package org.jsimpledb;
007
008import java.lang.reflect.Method;
009import java.util.Arrays;
010import java.util.List;
011import java.util.NavigableMap;
012
013import org.jsimpledb.core.MapField;
014import org.jsimpledb.schema.MapSchemaField;
015import org.objectweb.asm.ClassWriter;
016
017/**
018 * Represents a map field in a {@link JClass}.
019 */
020public class JMapField extends JComplexField {
021
022    final JSimpleField keyField;
023    final JSimpleField valueField;
024
025    JMapField(JSimpleDB jdb, String name, int storageId,
026      JSimpleField keyField, JSimpleField valueField, String description, Method getter) {
027        super(jdb, name, storageId, description, getter);
028        if (keyField == null)
029            throw new IllegalArgumentException("null keyField");
030        if (valueField == null)
031            throw new IllegalArgumentException("null valueField");
032        this.keyField = keyField;
033        this.valueField = valueField;
034    }
035
036    /**
037     * Get the key sub-field.
038     *
039     * @return this field's key sub-field
040     */
041    public JSimpleField getKeyField() {
042        return this.keyField;
043    }
044
045    /**
046     * Get the value sub-field.
047     *
048     * @return this field's value sub-field
049     */
050    public JSimpleField getValueField() {
051        return this.valueField;
052    }
053
054    @Override
055    public NavigableMap<?, ?> getValue(JObject jobj) {
056        if (jobj == null)
057            throw new IllegalArgumentException("null jobj");
058        return jobj.getTransaction().readMapField(jobj, this.storageId, false);
059    }
060
061    @Override
062    public <R> R visit(JFieldSwitch<R> target) {
063        return target.caseJMapField(this);
064    }
065
066    @Override
067    public List<JSimpleField> getSubFields() {
068        return Arrays.asList(this.keyField, this.valueField);
069    }
070
071    @Override
072    public JSimpleField getSubField(String name) {
073        if (MapField.KEY_FIELD_NAME.equals(name))
074            return this.keyField;
075        if (MapField.VALUE_FIELD_NAME.equals(name))
076            return this.valueField;
077        throw new IllegalArgumentException("unknown sub-field `" + name
078          + "' (did you mean `" + MapField.KEY_FIELD_NAME + "' or `" + MapField.VALUE_FIELD_NAME + "' instead?)");
079    }
080
081    @Override
082    String getSubFieldName(JSimpleField subField) {
083        if (subField == this.keyField)
084            return MapField.KEY_FIELD_NAME;
085        if (subField == this.valueField)
086            return MapField.VALUE_FIELD_NAME;
087        throw new IllegalArgumentException("unknown sub-field");
088    }
089
090    @Override
091    MapSchemaField toSchemaItem(JSimpleDB jdb) {
092        final MapSchemaField schemaField = new MapSchemaField();
093        super.initialize(jdb, schemaField);
094        schemaField.setKeyField(this.keyField.toSchemaItem(jdb));
095        schemaField.setValueField(this.valueField.toSchemaItem(jdb));
096        return schemaField;
097    }
098
099    @Override
100    void outputMethods(ClassGenerator<?> generator, ClassWriter cw) {
101        this.outputReadMethod(generator, cw, ClassGenerator.READ_MAP_FIELD_METHOD);
102    }
103
104    @Override
105    JMapFieldInfo toJFieldInfo() {
106        return new JMapFieldInfo(this);
107    }
108}
