006package org.jsimpledb.tuple;
007
008import com.google.common.collect.Lists;
009
010import java.util.Arrays;
011
012/**
013 * Tuple utility methods.
014 */
015public final class Tuples {
016
017    private Tuples() {
018    }
019
020    /**
021     * Create a {@link Tuple} of the appropriate cardinality for the given number of objects.
022     *
023     * @param values tuple values
024     * @return newly created tuple
025     * @throws IllegalArgumentException if {@code values} is null or empty
026     * @throws UnsupportedOperationException if {@code values} has more elements than exist concrete {@link Tuple} classes
027     */
028    public static Tuple createTuple(Iterable<?> values) {
029        if (values == null)
030            throw new IllegalArgumentException("null values");
031        return Tuples.createTuple(Lists.newArrayList(values).toArray());
032    }
033
034    /**
035     * Create a {@link Tuple} of the appropriate cardinality for the given number of objects.
036     *
037     * @param values tuple values
038     * @return newly created tuple
039     * @throws IllegalArgumentException if {@code values} is null or empty
040     * @throws UnsupportedOperationException if {@code values} has more elements than exist concrete {@link Tuple} classes
041     */
042    public static Tuple createTuple(Object... values) {
043        if (values == null)
044            throw new IllegalArgumentException("null values");
045        final Class<? extends Tuple> tupleClass = Tuples.tupleClassForSize(values.length);
046        final Class<?>[] parameterTypes = new Class<?>[values.length];
047        Arrays.fill(parameterTypes, Object.class);
048        try {
049            return tupleClass.getConstructor(parameterTypes).newInstance(values);
050        } catch (Exception e) {
051            throw new UnsupportedOperationException("can't instantiate " + tupleClass, e);
052        }
053    }
054
055    /**
056     * Get the {@link Tuple} subclass having the specified cardinality.
057     *
058     * @param size number of objects in the tuple
059     * @return {@link Tuple} subclass with cardinality {@code size}
060     * @throws IllegalArgumentException if {@code size} is less than one
061     * @throws UnsupportedOperationException if {@code size} is more than exist concrete {@link Tuple} classes
062     */
063    public static Class<? extends Tuple> tupleClassForSize(int size) {
064        if (size < 1)
065            throw new IllegalArgumentException("invalid size " + size);
066        final String name = Tuple.class.getName() + size;
067        try {
068            return Class.forName(name, false, Tuple.class.getClassLoader()).asSubclass(Tuple.class);
069        } catch (Exception e) {
070            throw new UnsupportedOperationException("can't find class `" + name + "' of size " + size, e);
071        }
072    }
073
074    /**
075     * Get the cardinality of the given {@link Tuple} subclass.
076     *
077     * @param tupleClass tuple class
078     * @return cardinality of {@code tupleClass}
079     * @throws IllegalArgumentException if {@code tupleClass} is not a recognized tuple class
080     */
081    public static int tupleSizeOf(Class<? extends Tuple> tupleClass) {
082        for (int size = 1; true; size++) {
083            try {
084                if (Tuples.tupleClassForSize(size).isAssignableFrom(tupleClass))
085                    return size;
086            } catch (UnsupportedOperationException e) {
087                throw new IllegalArgumentException("not a tuple (sub)class: " + tupleClass);
088            }
089        }
090    }
091}
