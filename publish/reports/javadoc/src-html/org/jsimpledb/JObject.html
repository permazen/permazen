006package org.jsimpledb;
007
008import org.jsimpledb.core.ObjId;
009
010/**
011 * Interface implemented by {@link JSimpleDB} Java model objects.
012 *
013 * <p>
014 * All {@link JSimpleDB} database objects are instances of runtime-generated sub-classes of user-provided Java model classes.
015 * These generated subclasses will always implement this interface, providing convenient access to database operations.
016 * Therefore, it is conveninent to declare Java model classes {@code abstract} and {@code implements JObject}.
017 * However, this is not strictly necessary; all of the methods declared here ultimately delegate to one of the
018 * {@link JTransaction} support methods.
019 * </p>
020 *
021 * <p>
022 * There are two types of {@link JObject}s: normal instances, which always reflect the state of the {@link JTransaction}
023 * {@linkplain JTransaction#getCurrent associated} with the current thread, and "snapshot" instances that reflect
024 * the state of their associated {@link SnapshotJTransaction}. Use {@link #isSnapshot} to distinguish if necessary
025 * Use {@link #copyIn copyIn()} and {@link #copyOut copyOut()} to copy data between normal and snapshot transactions.
026 * </p>
027 *
028 * <p>
029 * The {@link #getTransaction} method returns the transaction associated with an instance. For a normal instance,
030 * this is just the {@link JTransaction} {@linkplain JTransaction#getCurrent associated} with the current thread;
031 * if there is no such transaction, an {@link IllegalStateException} is thrown. For a "snapshot" instance,
032 * {@link #getTransaction} always returns the corresponding {@link SnapshotJTransaction}.
033 * </p>
034 */
035public interface JObject {
036
037    /**
038     * Get this instance's unique JSimpleDB object identifier.
039     *
040     * <p>
041     * This method always succeeds.
042     * </p>
043     *
044     * @return unique database identifier for this instance
045     */
046    ObjId getObjId();
047
048    /**
049     * Get this instance's current schema version. Does not change this instance's schema version.
050     *
051     * @return the schema version of this instance
052     * @throws org.jsimpledb.core.DeletedObjectException
053     *  if this object does not exist in the {@link JTransaction} associated with this instance
054     * @throws IllegalStateException if this is not a snapshot instance and there is no {@link JTransaction}
055     *  associated with the current thread
056     * @throws org.jsimpledb.core.StaleTransactionException
057     *  if this is not a snapshot instance and the transaction associated with the current thread is no longer usable
058     */
059    int getSchemaVersion();
060
061    /**
062     * Get this instance's associated {@link JTransaction}.
063     *
064     * <p>
065     * If this is a regular database instance, this returns the {@link JTransaction}
066     * {@linkplain JTransaction#getCurrent associated} with the current thread. Otherwise, this instance
067     * is a snapshot instance and this method returns the associated {@link SnapshotJTransaction}.
068     * </p>
069     *
070     * @return the {@link JTransaction} that contains this instance's field state
071     * @throws IllegalStateException if this is not a snapshot instance and there is no {@link JTransaction}
072     *  associated with the current thread
073     */
074    JTransaction getTransaction();
075
076    /**
077     * Delete this instance, if it exists, in this instance's associated transaction.
078     *
079     * <p>
080     * See {@link org.jsimpledb.core.Transaction#delete Transaction.delete()} for details on secondary deletions from
081     * {@link org.jsimpledb.core.DeleteAction#DELETE} and {@link org.jsimpledb.annotation.JField#cascadeDelete}.
082     * </p>
083     *
084     * @return true if instance was deleted, false if it did not exist
085     * @throws IllegalStateException if this is not a snapshot instance and there is no {@link JTransaction}
086     *  associated with the current thread
087     * @throws org.jsimpledb.core.StaleTransactionException
088     *  if the transaction associated with the current thread is no longer usable
089     * @throws org.jsimpledb.core.ReferencedObjectException if the object is referenced by some other object
090     *  through a reference field configured for {@link org.jsimpledb.core.DeleteAction#EXCEPTION}
091     */
092    boolean delete();
093
094    /**
095     * Determine whether this instance still exists in its associated transaction.
096     *
097     * @return true if instance exists, otherwise false
098     * @throws IllegalStateException if this is not a snapshot instance and there is no {@link JTransaction}
099     *  associated with the current thread
100     * @throws org.jsimpledb.core.StaleTransactionException
101     *  if this is not a snapshot instance and the transaction associated with the current thread is no longer usable
102     */
103    boolean exists();
104
105    /**
106     * Determine whether this instance is a normal instance or is a "snapshot" instance associated
107     * with a {@link SnapshotJTransaction}.
108     *
109     * @return true if instance is a snapshot instance
110     */
111    boolean isSnapshot();
112
113    /**
114     * Recreate a deleted instance, if it does not exist, in its associated transaction.
115     * The fields of a recreated object are set to their initial values. If the object already exists, nothing changes.
116     *
117     * @return true if instance was recreated, false if it already existed
118     * @throws IllegalStateException if this is not a snapshot instance and there is no {@link JTransaction}
119     *  associated with the current thread
120     * @throws org.jsimpledb.core.StaleTransactionException
121     *  if this is not a snapshot instance and the transaction associated with the current thread is no longer usable
122     */
123    boolean recreate();
124
125    /**
126     * Add this instance to the validation queue for validation in its associated transaction.
127     *
128     * <p>
129     * The actual validation will occur either during {@link JTransaction#commit}
130     * or at the next invocation of {@link JTransaction#validate}, whichever occurs first.
131     * </p>
132     *
133     * @throws org.jsimpledb.core.DeletedObjectException
134     *  if this object does not exist in the {@link JTransaction} associated with this instance
135     * @throws IllegalStateException if transaction commit is already in progress
136     * @throws IllegalStateException if this is not a snapshot instance and there is no {@link JTransaction}
137     *  associated with the current thread
138     * @throws org.jsimpledb.core.StaleTransactionException
139     *  if this is not a snapshot instance and the transaction associated with the current thread is no longer usable
140     */
141    void revalidate();
142
143    /**
144     * Update the schema version of this instance, if necessary, so that it matches the schema version
145     * of its associated transaction.
146     *
147     * <p>
148     * If a version change occurs, matching {@link org.jsimpledb.annotation.OnVersionChange &#64;OnVersionChange}
149     * methods will be invoked prior to this method returning.
150     * </p>
151     *
152     * @return true if the object's schema version was changed, false if it was already updated
153     * @throws org.jsimpledb.core.DeletedObjectException
154     *  if this object does not exist in the {@link JTransaction} associated with this instance
155     * @throws IllegalStateException if this is not a snapshot instance and there is no {@link JTransaction}
156     *  associated with the current thread
157     * @throws org.jsimpledb.core.StaleTransactionException
158     *  if this is not a snapshot instance and the transaction associated with the current thread is no longer usable
159     */
160    boolean upgrade();
161
162    /**
163     * Copy this instance, and other instances it references, into a (possibly) different {@link JTransaction}.
164     * This is a more general method; see {@link #copyIn copyIn()} and {@link #copyOut copyOut()} for more common
165     * and specific use cases.
166     *
167     * <p>
168     * This method will copy this object's fields into the object with ID {@code target} (or this instance's object ID if
169     * {@code target} is null) in the {@code dest} transaction, overwriting any previous values there, along with all other
170     * objects reachable from this instance through any of the specified {@linkplain ReferencePath reference paths} (including
171     * intermediate objects visited). This instance will first be {@link #upgrade}ed if necessary.
172     * If {@code target} (or any other referenced object) already exists in {@code dest}, it will have its schema version
173     * updated first, if necessary, otherwise it will be created.
174     * Any {@link org.jsimpledb.annotation.OnCreate &#64;OnVersionChange}, {@link org.jsimpledb.annotation.OnCreate &#64;OnCreate},
175     * and {@link org.jsimpledb.annotation.OnCreate &#64;OnChange} methods will be notified accordingly as usual (in {@code dest});
176     * however, for {@link org.jsimpledb.annotation.OnCreate &#64;OnCreate} and
177     * {@link org.jsimpledb.annotation.OnCreate &#64;OnChange}, the annotation must have {@code snapshotTransactions = true}
178     * if {@code dest} is a {@link SnapshotJTransaction}.
179     * </p>
180     *
181     * <p>
182     * The two transactions must be compatible in that for any schema versions encountered, those schema versions
183     * must be identical in both transactions.
184     * </p>
185     *
186     * <p>
187     * Circular references are handled properly: if an object is encountered more than once, it is not copied again.
188     * The {@code copyState} parameter can be used to keep track of objects that have already been copied and/or traversed
189     * along some reference path (however, if an object is marked as copied in {@code copyState} and is traversed, but does not
190     * actually already exist in {@code dest}, an exception is thrown).
191     * For a "fresh" copy operation, pass a newly created {@code CopyState}; for a copy operation that is a continuation
192     * of a previous copy, {@code copyState} may be reused.
193     * </p>
194     *
195     * <p>
196     * Note: if {@code target} is not equal to this instance's object ID, and through one of the {@code refPaths} there
197     * is a circular reference back to this instance, then that reference is copied as-is (i.e., it is not copied
198     * onto {@code target}).
199     * </p>
200     *
201     * <p>
202     * Warning: if two threads attempt to copy objects between the same two transactions at the same time
203     * but in opposite directions, deadlock could result.
204     * </p>
205     *
206     * @param dest destination transaction for copies
207     * @param target target object ID in {@code dest} onto which to copy this instance's fields, or null for this instance
208     * @param copyState tracks which indirectly referenced objects have already been copied
209     * @param refPaths zero or more reference paths that refer to additional objects to be copied
210     * @return the copied version of this instance in {@code dest}
211     * @throws org.jsimpledb.core.DeletedObjectException
212     *  if this object does not exist in the {@link JTransaction} associated with this instance
213     *  (no exception is thrown however if an indirectly referenced object does not exist unless it is traversed)
214     * @throws org.jsimpledb.core.DeletedObjectException if an object in {@code copied} is traversed but does not actually exist
215     * @throws IllegalStateException if this is not a snapshot instance and there is no {@link JTransaction}
216     *  associated with the current thread
217     * @throws org.jsimpledb.core.SchemaMismatchException
218     *  if the schema corresponding to this object's version is not identical in both the {@link JTransaction}
219     *  associated with this instance and {@code dest} (as well for any referenced objects)
220     * @throws IllegalArgumentException if {@code dest}, {@code copied}, or {@code refPaths} is null
221     * @throws IllegalArgumentException if any path in {@code refPaths} is invalid
222     * @see #copyIn copyIn()
223     * @see #copyOut copyOut()
224     * @see JTransaction#copyTo(JTransaction, JObject, ObjId, CopyState, String[]) JTransaction.copyTo()
225     */
226    JObject copyTo(JTransaction dest, ObjId target, CopyState copyState, String... refPaths);
227
228    /**
229     * Snapshot this instance and other instances it references.
230     *
231     * <p>
232     * This method will copy this object and all of its fields, along with all other objects reachable through
233     * any of the specified {@linkplain ReferencePath reference paths} into the {@link SnapshotJTransaction}
234     * {@linkplain JTransaction#getSnapshotTransaction corresponding} to this instance's associated transaction
235     * (including intermediate objects visited). This instance will first be {@link #upgrade}ed if necessary.
236     * If any object already exists there, it will be overwritten, otherwise it will be created.
237     * {@link org.jsimpledb.annotation.OnCreate &#64;OnCreate} and {@link org.jsimpledb.annotation.OnCreate &#64;OnChange}
238     * notifications will be delivered accordingly; however, the annotation must have {@code snapshotTransactions = true}.
239     * </p>
240     *
241     * <p>
242     * Normally this method would only be invoked on a regular database {@link JObject}.
243     * The returned object will always be a snapshot {@link JObject}.
244     * </p>
245     *
246     * <p>
247     * This is a convenience method, and is equivalent to invoking:
248     * <blockquote><code>
249     * this.copyTo(this.getTransaction().getSnapshotTransaction(), null, new CopyState(), refPaths);
250     * </code></blockquote>
251     *
252     * @param refPaths zero or more reference paths that refer to additional objects to be copied
253     * @return the snapshot {@link JObject} copy of this instance
254     * @throws org.jsimpledb.core.DeletedObjectException
255     *  if this object does not exist in the {@link JTransaction} associated with this instance
256     *  (no exception is thrown however if an indirectly referenced object does not exist)
257     * @throws IllegalStateException if this is not a snapshot instance and there is no {@link JTransaction}
258     *  associated with the current thread
259     * @throws IllegalArgumentException if any path in {@code refPaths} is invalid
260     * @see #copyIn copyIn()
261     */
262    JObject copyOut(String... refPaths);
263
264    /**
265     * Copy this instance and other instances it references into the transaction associated with the current thread.
266     *
267     * <p>
268     * This method will copy this object and all of its fields, along with all other objects reachable through any of the
269     * specified {@linkplain ReferencePath reference paths} into the {@link JTransaction}
270     * {@linkplain JTransaction#getCurrent associated} with the current thread (including intermediate objects visited).
271     * If any object already exists in the current thread's transaction, it will be overwritten, otherwise it will be created.
272     * {@link org.jsimpledb.annotation.OnCreate &#64;OnCreate} and {@link org.jsimpledb.annotation.OnCreate &#64;OnChange}
273     * notifications will be delivered accordingly.
274     * </p>
275     *
276     * <p>
277     * Normally this method would only be invoked on a snapshot {@link JObject}.
278     * The returned object will always be a regular database {@link JObject}.
279     * </p>
280     *
281     * <p>
282     * This is a convenience method, and is equivalent to invoking:
283     * <blockquote><code>
284     * this.copyTo(JTransaction.getCurrent(), null, new CopyState(), refPaths)
285     * </code></blockquote>
286     *
287     * @param refPaths zero or more reference paths that refer to additional objects to be copied
288     * @return the regular database copy of this instance
289     * @throws org.jsimpledb.core.DeletedObjectException
290     *  if this object does not exist in the {@link JTransaction} associated with this instance
291     *  (no exception is thrown however if an indirectly referenced object does not exist)
292     * @throws IllegalStateException if this is not a snapshot instance and there is no {@link JTransaction}
293     *  associated with the current thread
294     * @throws org.jsimpledb.core.SchemaMismatchException
295     *  if the schema corresponding to this object's version is not identical in both transactions
296     * @throws IllegalArgumentException if any path in {@code refPaths} is invalid
297     * @see #copyOut copyOut()
298     */
299    JObject copyIn(String... refPaths);
300}
