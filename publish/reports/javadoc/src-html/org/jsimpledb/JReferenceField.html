006package org.jsimpledb;
007
008import com.google.common.base.Converter;
009import com.google.common.reflect.TypeToken;
010
011import java.lang.reflect.Method;
012import java.util.TreeSet;
013
014import org.jsimpledb.core.DeleteAction;
015import org.jsimpledb.core.ObjId;
016import org.jsimpledb.core.ReferenceFieldType;
017import org.jsimpledb.schema.ReferenceSchemaField;
018
019/**
020 * Represents a reference field in a {@link JClass} or a reference sub-field of a complex field in a {@link JClass}.
021 */
022public class JReferenceField extends JSimpleField {
023
024    final DeleteAction onDelete;
025    final boolean cascadeDelete;
026
027    JReferenceField(JSimpleDB jdb, String name, int storageId, String description, TypeToken<?> typeToken,
028      org.jsimpledb.annotation.JField annotation, Method getter, Method setter) {
029        super(jdb, name, storageId, typeToken, new ReferenceFieldType(), true, annotation, description, getter, setter);
030        this.onDelete = annotation.onDelete();
031        this.cascadeDelete = annotation.cascadeDelete();
032    }
033
034    @Override
035    public JObject getValue(JObject jobj) {
036        return (JObject)super.getValue(jobj);
037    }
038
039    @Override
040    public <R> R visit(JFieldSwitch<R> target) {
041        return target.caseJReferenceField(this);
042    }
043
044    @Override
045    public Converter<ObjId, JObject> getConverter(JTransaction jtx) {
046        return new ReferenceConverter<JObject>(jtx, JObject.class).reverse();
047    }
048
049    /**
050     * Get the {@link DeleteAction} configured for this field.
051     *
052     * @return this field's {@link DeleteAction}
053     */
054    public DeleteAction getOnDelete() {
055        return this.onDelete;
056    }
057
058    /**
059     * Determine whether the referred-to object should be deleted when an object containing this field is deleted.
060     *
061     * @return this field's delete cascade setting
062     */
063    public boolean isCascadeDelete() {
064        return this.cascadeDelete;
065    }
066
067    @Override
068    ReferenceSchemaField toSchemaItem(JSimpleDB jdb) {
069        final ReferenceSchemaField schemaField = new ReferenceSchemaField();
070        super.initialize(jdb, schemaField);
071        schemaField.setOnDelete(this.onDelete);
072        schemaField.setCascadeDelete(this.cascadeDelete);
073        final TreeSet<Integer> objectTypes = new TreeSet<>();
074        for (JClass<?> jclass : jdb.getJClasses(this.typeToken.getRawType()))
075            objectTypes.add(jclass.storageId);
076        schemaField.setObjectTypes(objectTypes);
077        return schemaField;
078    }
079
080    @Override
081    JReferenceFieldInfo toJFieldInfo(int parentStorageId) {
082        return new JReferenceFieldInfo(this, parentStorageId);
083    }
084}
