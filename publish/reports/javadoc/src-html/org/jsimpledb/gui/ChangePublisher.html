006package org.jsimpledb.gui;
007
008import org.dellroad.stuff.vaadin7.VaadinConfigurable;
009import org.jsimpledb.JObject;
010import org.jsimpledb.JTransaction;
011import org.jsimpledb.change.Change;
012import org.jsimpledb.change.SimpleFieldChange;
013import org.jsimpledb.core.ObjId;
014import org.jsimpledb.core.Transaction;
015import org.springframework.beans.factory.annotation.Autowired;
016import org.springframework.beans.factory.annotation.Qualifier;
017import org.springframework.context.event.ApplicationEventMulticaster;
018import org.springframework.stereotype.Component;
019
020/**
021 * Publishes change notifications.
022 */
023@Component
024@VaadinConfigurable
025public class ChangePublisher {
026
027    @Autowired
028    @Qualifier("jsimpledbGuiEventMulticaster")
029    private ApplicationEventMulticaster eventMulticaster;
030
031    /**
032     * Publish the given {@link Change} if the current transaction is successful.
033     *
034     * @param change change during the current transaction
035     */
036    public void publishChangeOnCommit(Change<?> change) {
037        JTransaction.getCurrent().getTransaction().addCallback(
038          new PublishChangeCallback(this.eventMulticaster, new DataChangeEvent(this, change)));
039    }
040
041    /**
042     * Publish that the specified object has changed if the current transaction is successful.
043     *
044     * @param jobj object changed during the current transaction
045     */
046    public void publishChangeOnCommit(JObject jobj) {
047
048        // We rely here on the fact that publishing a change in any property reloads all properties for that object
049        this.publishChangeOnCommit(new SimpleFieldChange<Object, ObjId>(jobj,
050          Integer.MAX_VALUE, JObjectContainer.OBJECT_ID_PROPERTY, jobj.getObjId(), jobj.getObjId()));
051    }
052
053// PublishChangeCallback - notifies the rest of the application when a data instance has been added/removed/changed
054
055    private static class PublishChangeCallback extends Transaction.CallbackAdapter {
056
057        private final ApplicationEventMulticaster eventMulticaster;
058        private final DataChangeEvent event;
059
060        PublishChangeCallback(ApplicationEventMulticaster eventMulticaster, DataChangeEvent event) {
061            if (eventMulticaster == null)
062                throw new IllegalArgumentException("null eventMulticaster");
063            if (event == null)
064                throw new IllegalArgumentException("null event");
065            this.eventMulticaster = eventMulticaster;
066            this.event = event;
067        }
068
069        @Override
070        public void afterCommit() {
071            this.eventMulticaster.multicastEvent(this.event);
072        }
073
074        @Override
075        public boolean equals(Object obj) {
076            if (obj == null || obj.getClass() != this.getClass())
077                return false;
078            final PublishChangeCallback that = (PublishChangeCallback)obj;
079            return this.event.equals(that.event);
080        }
081
082        @Override
083        public int hashCode() {
084            return this.event.hashCode();
085        }
086    }
087}
