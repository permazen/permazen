006package org.jsimpledb.gui;
007
008import com.vaadin.server.UIClassSelectionEvent;
009import com.vaadin.server.UIProvider;
010import com.vaadin.ui.UI;
011
012/**
013 * {@link UIProvider} for this GUI.
014 */
015@SuppressWarnings("serial")
016public class GUIUIProvider extends UIProvider {
017
018    @Override
019    public Class<? extends UI> getUIClass(UIClassSelectionEvent event) {
020
021        // Get requested URI path
022        String pathInfo = event.getRequest().getPathInfo();
023        if (pathInfo != null)
024            pathInfo = pathInfo.substring(1);
025
026        // Look for well-known patterns... TODO
027
028        // Default to main root
029        return MainUI.class;
030    }
031}
