006package org.jsimpledb.gui;
007
008import org.jsimpledb.JSimpleDB;
009
010/**
011 * GUI configuration object.
012 */
013public interface GUIConfig {
014
015    /**
016     * Get the {@link JSimpleDB}.
017     *
018     * @return configured database
019     */
020    JSimpleDB getJSimpleDB();
021
022    /**
023     * Get a short description of the database.
024     *
025     * @return configured database description
026     */
027    String getDatabaseDescription();
028
029    /**
030     * Determine the schema version associated with the {@link JSimpleDB}.
031     *
032     * @return configured schema version
033     */
034    int getSchemaVersion();
035
036    /**
037     * Determine whether we are allowed to create a new schema version.
038     *
039     * @return configured new schema setting
040     */
041    boolean isAllowNewSchema();
042
043    /**
044     * Determine whether the underlying database is read-only.
045     *
046     * @return configured read-only setting
047     */
048    boolean isReadOnly();
049
050    /**
051     * Determine whether verbose mode is enabled.
052     *
053     * @return configured verbose setting
054     */
055    boolean isVerbose();
056
057    /**
058     * Get any custom {@link org.jsimpledb.parse.func.AbstractFunction} classes.
059     *
060     * @return custom function classes, or null for none
061     */
062    Iterable<? extends Class<?>> getFunctionClasses();
063}
