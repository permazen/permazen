006package org.jsimpledb.gui;
007
008import com.vaadin.ui.Component;
009
010import org.dellroad.stuff.vaadin7.VaadinConfigurable;
011import org.springframework.beans.factory.annotation.Autowired;
012
013/**
014 * Main GUI {@link com.vaadin.ui.UI} component.
015 */
016@SuppressWarnings("serial")
017@VaadinConfigurable
018public class MainUI extends AbstractUI {
019
020    public static final String URI_PATH = "main";
021
022    @Autowired
023    private GUIConfig guiConfig;
024
025    @Override
026    protected String getTitle() {
027        return "JSimpleDB Viewer";
028    }
029
030    @Override
031    protected Component getTopRightLabel() {
032        return new SizedLabel(this.guiConfig.getDatabaseDescription());
033    }
034
035    @Override
036    protected Component buildMainPanel() {
037        return new MainPanel(this.guiConfig);
038    }
039}
