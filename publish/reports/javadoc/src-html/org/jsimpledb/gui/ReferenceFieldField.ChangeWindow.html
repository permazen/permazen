028/**
029 * A Vaadin {@link com.vaadin.ui.Field} that edits a database object reference. Supports displaying,
030 * but not selecting, a null value; wrap in a {@link NullableField} to allow for that.
031 */
032@SuppressWarnings("serial")
033public class ReferenceFieldField extends CustomField<JObject> {
034
035    private final JTransaction dest;
036    private final HorizontalLayout layout = new HorizontalLayout();
037    private final ParseSession session;
038    private final String name;
039    private final Class<?> type;
040    private final JObjectContainer.RefLabelPropertyDef refLabelPropertyDef = new JObjectContainer.RefLabelPropertyDef();
041    private final SmallButton changeButton = new SmallButton("Change...", new Button.ClickListener() {
042        @Override
043        public void buttonClick(Button.ClickEvent event) {
044            ReferenceFieldField.this.change();
045        }
046    });
047
048    private Component refLabel = new Label();
049
050    /**
051     * Constructor.
052     *
053     * @param dest target transaction for the chosen {@link JObject}
054     * @param session session for expression parsing
055     * @param name name of the property
056     * @param type type restriction, or null for none
057     */
058    public ReferenceFieldField(JTransaction dest, ParseSession session, String name, Class<?> type) {
059
060        // Initialize
061        if (dest == null)
062            throw new IllegalArgumentException("null dest");
063        if (session == null)
064            throw new IllegalArgumentException("null session");
065        if (name == null)
066            throw new IllegalArgumentException("null name");
067        this.dest = dest;
068        this.session = session;
069        this.name = name;
070        this.type = type;
071
072        // Listen for property value and read-only status changes
073        this.addValueChangeListener(new Property.ValueChangeListener() {
074            @Override
075            public void valueChange(Property.ValueChangeEvent event) {
076                ReferenceFieldField.this.updateDisplay();
077            }
078        });
079        this.addReadOnlyStatusChangeListener(new Property.ReadOnlyStatusChangeListener() {
080            @Override
081            public void readOnlyStatusChange(Property.ReadOnlyStatusChangeEvent event) {
082                ReferenceFieldField.this.updateDisplay();
083            }
084        });
085    }
086
087// CustomField
088
089    @Override
090    public Class<JObject> getType() {
091        return JObject.class;
092    }
093
094    @Override
095    protected HorizontalLayout initContent() {
096        this.layout.setMargin(false);
097        this.layout.setSpacing(true);
098        this.layout.addComponent(this.refLabel);
099        this.layout.addComponent(new Label("\u00a0\u00a0"));
100        this.layout.addComponent(this.changeButton);
101        this.layout.setComponentAlignment(this.changeButton, Alignment.MIDDLE_LEFT);
102        this.updateDisplay();
103        return this.layout;
104    }
105
106    @Override
107    protected void setInternalValue(JObject jobj) {
108        super.setInternalValue(jobj);
109        this.updateDisplay();
110    }
111
112// Internal methods
113
114    // Update components to reflect new value
115    private void updateDisplay() {
116        final JObject jobj = this.getValue();
117        final boolean readOnly = this.isReadOnly();
118        final Component newLabel = jobj != null ?
119          this.refLabelPropertyDef.extract(jobj) : new SizedLabel("<i>Null</i>&#160;", ContentMode.HTML);
120        this.layout.replaceComponent(this.refLabel, newLabel);
121        this.refLabel = newLabel;
122        this.changeButton.setEnabled(!readOnly);
123    }
124
125    // Handle change button click
126    private void change() {
127        this.new ChangeWindow().show();
128    }
129
130// ChangeWindow
131
132    public class ChangeWindow extends ConfirmWindow {
133
134        private final JObjectChooser objectChooser;
135
136        ChangeWindow() {
137            super(ReferenceFieldField.this.getUI(), "Select " + ReferenceFieldField.this.name);
138            this.setWidth(800, Sizeable.Unit.PIXELS);
139            this.setHeight(500, Sizeable.Unit.PIXELS);
140            this.objectChooser = new JObjectChooser(ReferenceFieldField.this.dest.getJSimpleDB(),
141              ReferenceFieldField.this.session, ReferenceFieldField.this.type, false);
142        }
143
144        @Override
145        protected void addContent(VerticalLayout layout) {
146            final HorizontalSplitPanel objectPanel = this.objectChooser.getObjectPanel();
147            objectPanel.setHeight(200, Sizeable.Unit.PIXELS);
148            objectPanel.setSplitPosition(40);
149            layout.addComponent(objectPanel);
150            layout.addComponent(this.objectChooser.getShowForm());
151        }
152
153        @Override
154        @RetryTransaction
155        @org.springframework.transaction.annotation.Transactional("jsimpledbGuiTransactionManager")
156        protected boolean execute() {
157            final ObjId id = this.objectChooser.getObjId();
158            if (id == null)
159                return true;
160            final JObject jobj = JTransaction.getCurrent().getJObject(id);
161            if (!jobj.exists()) {
162                Notification.show("Object " + id + " no longer exists", null, Notification.Type.WARNING_MESSAGE);
163                return false;
164            }
165            try {
166                ReferenceFieldField.this.setValue(jobj.copyTo(ReferenceFieldField.this.dest, id, new CopyState()));
167            } catch (Exception e) {
168                Notification.show("Error: " + e, null, Notification.Type.ERROR_MESSAGE);
169            }
170            return true;
171        }
172    }
173}
