011import com.google.common.collect.Lists;
012import com.google.common.reflect.TypeToken;
013import com.vaadin.data.Container;
014import com.vaadin.shared.ui.label.ContentMode;
015
016import java.util.ArrayList;
017import java.util.Collection;
018import java.util.Collections;
019import java.util.List;
020
021import org.dellroad.stuff.vaadin7.ProvidesProperty;
022import org.dellroad.stuff.vaadin7.SimpleKeyedContainer;
023import org.jsimpledb.JClass;
024import org.jsimpledb.JSimpleDB;
025import org.slf4j.Logger;
026import org.slf4j.LoggerFactory;

028/**
029 * Hierarical container that contains the tree of all Java model classes in the database schema.
030 */
031@SuppressWarnings("serial")
032public class TypeContainer extends SimpleKeyedContainer<TypeToken<?>, TypeContainer.Node> implements Container.Hierarchical {
033
034    public static final String NAME_PROPERTY = "name";
035    public static final String STORAGE_ID_PROPERTY = "storageId";
036    public static final String TYPE_PROPERTY = "type";
037
038    protected final Logger log = LoggerFactory.getLogger(this.getClass());
039
040    private final JSimpleDB jdb;
041    private final Class<?> type;
042    private final ArrayList<Node> rootList = new ArrayList<>();
043
044    /**
045     * Constructor.
046     */
047    public TypeContainer(JSimpleDB jdb) {
048        this(jdb, null);
049    }
050
051    /**
052     * Constructor.
053     */
054    public TypeContainer(JSimpleDB jdb, Class<?> type) {
055        super(Node.class);
056        this.jdb = jdb;
057        this.type = type;
058    }
059
060    @Override
061    public void connect() {
062        super.connect();
063        this.reload();
064    }
065
066    @Override
067    public TypeToken<?> getKeyFor(Node node) {
068        return node.getTypeToken();
069    }
070
071    public void reload() {
072
073        // Node set
074        final ArrayList<Node> nodes = new ArrayList<>();
075
076        // Create root node for type if it doesn't correspond to a JClass
077        final TypeToken<?> topType = this.type != null ? TypeToken.of(this.type) : TypeToken.of(Object.class);
078        boolean needsTop = this.type == null;
079        if (!needsTop) {
080            try {
081                this.jdb.getJClass(type);
082            } catch (IllegalArgumentException e) {
083                needsTop = true;
084            }
085        }
086        if (needsTop)
087            nodes.add(new Node(topType));
088
089        // Create Node's for each JClass
090        for (JClass<?> jclass : this.jdb.getJClassesByStorageId().values()) {
091            if (topType.isAssignableFrom(jclass.getTypeToken()))
092                nodes.add(new Node(jclass));
093        }
094
095        // Determine parents by inspecting Java types
096        for (Node node1 : nodes) {
097            final TypeToken<?> type1 = node1.getTypeToken();
098            for (Node node2 : nodes) {
099                if (node2 == node1)
100                    continue;
101                final TypeToken<?> type2 = node2.getTypeToken();
102                if (type2.isAssignableFrom(type1)) {
103                    final Node previous = node1.getParent();
104                    if (previous == null || previous.getTypeToken().isAssignableFrom(type2))
105                        node1.setParent(node2);
106                }
107            }
108        }
109
110        // Derive children
111        for (Node node1 : nodes) {
112            for (Node node2 : nodes) {
113                if (node2 == node1.getParent())
114                    node2.getChilds().add(node1);
115            }
116        }
117
118        // Derive roots
119        this.rootList.clear();
120        for (Node node : nodes) {
121            if (node.getParent() == null)
122                this.rootList.add(node);
123        }
124
125        // Load container
126        this.load(nodes);
127    }
128
129// Node
130
131    public static class Node {
132
133        public static final Function<Node, TypeToken<?>> TYPE_TOKEN_FUNCTION = new Function<Node, TypeToken<?>>() {
134            @Override
135            public TypeToken<?> apply(Node node) {
136                return node.getTypeToken();
137            }
138        };
139
140        private final TypeToken<?> typeToken;
141        private final JClass<?> jclass;
142        private final ArrayList<Node> childs = new ArrayList<>();
143
144        private Node parent;
145
146        public Node(JClass<?> jclass) {
147            this(jclass.getTypeToken(), jclass);
148        }
149
150        public Node(TypeToken<?> typeToken) {
151            this(typeToken, null);
152        }
153
154        private Node(TypeToken<?> typeToken, JClass<?> jclass) {
155            if (typeToken == null)
156                throw new IllegalArgumentException("null typeToken");
157            this.typeToken = typeToken;
158            this.jclass = jclass;
159        }
160
161        public TypeToken<?> getTypeToken() {
162            return this.typeToken;
163        }
164
165        public Node getParent() {
166            return this.parent;
167        }
168        public void setParent(Node parent) {
169            this.parent = parent;
170        }
171
172        public List<Node> getChilds() {
173            return this.childs;
174        }
175
176        @ProvidesProperty(TypeContainer.NAME_PROPERTY)
177        public String getName() {
178            return this.jclass != null ? this.jclass.getName() : this.typeToken.getRawType().getSimpleName();
179        }
180
181        @ProvidesProperty(TypeContainer.STORAGE_ID_PROPERTY)
182        public Integer getStorageId() {
183            return this.jclass != null ? this.jclass.getStorageId() : null;
184        }
185
186        @ProvidesProperty(TypeContainer.TYPE_PROPERTY)
187        public SizedLabel getType() {
188            return new SizedLabel("<code>" + this.getTypeToken().toString() + "</code>", ContentMode.HTML);
189        }
190    }
191
192// Container.Hierarchical methods
193
194    @Override
195    public Collection<TypeToken<?>> getChildren(Object itemId) {
196        final Node node = this.getJavaObject(itemId);
197        return node != null ? Lists.transform(node.getChilds(), Node.TYPE_TOKEN_FUNCTION) : Collections.<TypeToken<?>>emptySet();
198    }
199
200    @Override
201    public TypeToken<?> getParent(Object itemId) {
202        final Node child = this.getJavaObject(itemId);
203        if (child == null)
204            return null;
205        final Node parent = child.getParent();
206        if (parent == null)
207            return null;
208        return parent.getTypeToken();
209    }
210
211    @Override
212    public Collection<TypeToken<?>> rootItemIds() {
213        return Lists.transform(this.rootList, Node.TYPE_TOKEN_FUNCTION);
214    }
215
216    @Override
217    public boolean areChildrenAllowed(Object itemId) {
218        final Node node = this.getJavaObject(itemId);
219        return node != null && !node.getChilds().isEmpty();
220    }
221
222    @Override
223    public boolean isRoot(Object itemId) {
224        final Node node = this.getJavaObject(itemId);
225        return node != null && node.getParent() == null;
226    }
227
228    @Override
229    public boolean hasChildren(Object itemId) {
230        final Node node = this.getJavaObject(itemId);
231        return node != null && !node.getChilds().isEmpty();
232    }
233
234// Mutators are not supported
235
236    @Override
237    public boolean setParent(Object itemId, Object newParentId) {
238        throw new UnsupportedOperationException();
239    }
240
241    @Override
242    public boolean setChildrenAllowed(Object itemId, boolean areChildrenAllowed) {
243        throw new UnsupportedOperationException();
244    }
245
246    @Override
247    public boolean removeItem(Object itemId) {
248        throw new UnsupportedOperationException();
249    }
250}
