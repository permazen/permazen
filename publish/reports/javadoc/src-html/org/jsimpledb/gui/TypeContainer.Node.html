028/**
029 * Hierarical container that contains the tree of all Java model classes in the database schema.
030 */
031@SuppressWarnings("serial")
032public class TypeContainer extends SimpleKeyedContainer<Class<?>, TypeContainer.Node> implements Container.Hierarchical {
033
034    public static final String NAME_PROPERTY = "name";
035    public static final String STORAGE_ID_PROPERTY = "storageId";
036    public static final String TYPE_PROPERTY = "type";
037
038    protected final Logger log = LoggerFactory.getLogger(this.getClass());
039
040    private final JSimpleDB jdb;
041    private final Class<?> rootType;
042    private final ArrayList<Node> rootList = new ArrayList<>();
043
044    /**
045     * Constructor.
046     *
047     * @param jdb underlying database
048     */
049    public TypeContainer(JSimpleDB jdb) {
050        this(jdb, null);
051    }
052
053    /**
054     * Constructor.
055     *
056     * @param jdb underlying database
057     * @param type type restrictions, or null for none
058     */
059    public TypeContainer(JSimpleDB jdb, Class<?> type) {
060        super(Node.class);
061        if (jdb == null)
062            throw new IllegalArgumentException("null jdb");
063        this.jdb = jdb;
064
065        // Get the types of all JClasses assignable to the given type, and use lowest common ancestor as the "top type"
066        final HashSet<Class<?>> types = new HashSet<>();
067        for (JClass<?> jclass : this.jdb.getJClasses().values()) {
068            if (type == null || type.isAssignableFrom(jclass.getType()))
069                types.add(jclass.getType());
070        }
071        this.rootType = !types.isEmpty() ? Util.findLowestCommonAncestorOfClasses(types).getRawType() : Object.class;
072    }
073
074    @Override
075    public void connect() {
076        super.connect();
077        this.reload();
078    }
079
080    @Override
081    public Class<?> getKeyFor(Node node) {
082        return node.getType();
083    }
084
085    public Class<?> getRootType() {
086        return this.rootType;
087    }
088
089    public void reload() {
090
091        // Node set
092        final ArrayList<Node> nodes = new ArrayList<>();
093
094        // Create Node's for each JClass
095        boolean addedRoot = false;
096        for (JClass<?> jclass : this.jdb.getJClasses().values()) {
097            if (this.rootType.isAssignableFrom(jclass.getType())) {
098                nodes.add(new Node(jclass));
099                if (jclass.getType() == this.rootType)
100                    addedRoot = true;
101            }
102        }
103
104        // Ensure there is a root node
105        if (!addedRoot)
106            nodes.add(new Node(this.rootType));
107
108        // Sort by name
109        Collections.sort(nodes, new Comparator<Node>() {
110            @Override
111            public int compare(Node node1, Node node2) {
112                return node1.propertyName().compareTo(node2.propertyName());
113            }
114        });
115
116        // Determine parents by inspecting Java types
117        for (Node node1 : nodes) {
118            final Class<?> type1 = node1.getType();
119            for (Node node2 : nodes) {
120                if (node2 == node1)
121                    continue;
122                final Class<?> type2 = node2.getType();
123                if (type2.isAssignableFrom(type1)) {
124                    final Node previous = node1.getParent();
125                    if (previous == null || previous.getType().isAssignableFrom(type2))
126                        node1.setParent(node2);
127                }
128            }
129        }
130
131        // Derive children
132        for (Node node1 : nodes) {
133            for (Node node2 : nodes) {
134                if (node2 == node1.getParent())
135                    node2.getChilds().add(node1);
136            }
137        }
138
139        // Derive roots
140        this.rootList.clear();
141        for (Node node : nodes) {
142            if (node.getParent() == null)
143                this.rootList.add(node);
144        }
145
146        // Load container
147        this.load(nodes);
148    }
149
150// Node
151
152    public static class Node {
153
154        public static final Function<Node, Class<?>> TYPE_FUNCTION = new Function<Node, Class<?>>() {
155            @Override
156            public Class<?> apply(Node node) {
157                return node.getType();
158            }
159        };
160
161        private final Class<?> type;
162        private final JClass<?> jclass;
163        private final ArrayList<Node> childs = new ArrayList<>();
164
165        private Node parent;
166
167        public Node(JClass<?> jclass) {
168            this(jclass.getType(), jclass);
169        }
170
171        public Node(Class<?> type) {
172            this(type, null);
173        }
174
175        private Node(Class<?> type, JClass<?> jclass) {
176            if (type == null)
177                throw new IllegalArgumentException("null type");
178            this.type = type;
179            this.jclass = jclass;
180        }
181
182        public Class<?> getType() {
183            return this.type;
184        }
185
186        public Node getParent() {
187            return this.parent;
188        }
189        public void setParent(Node parent) {
190            this.parent = parent;
191        }
192
193        public List<Node> getChilds() {
194            return this.childs;
195        }
196
197        @ProvidesProperty(TypeContainer.NAME_PROPERTY)
198        public String propertyName() {
199            return this.jclass != null ? this.jclass.getName() : this.type.getSimpleName();
200        }
201
202        @ProvidesProperty(TypeContainer.STORAGE_ID_PROPERTY)
203        public Integer propertyStorageId() {
204            return this.jclass != null ? this.jclass.getStorageId() : null;
205        }
206
207        @ProvidesProperty(TypeContainer.TYPE_PROPERTY)
208        public SizedLabel proeprtyType() {
209            return new SizedLabel("<code>" + this.getType().toString() + "</code>", ContentMode.HTML);
210        }
211    }
212
213// Container.Hierarchical methods
214
215    @Override
216    public Collection<Class<?>> getChildren(Object itemId) {
217        final Node node = this.getJavaObject(itemId);
218        return node != null ? Lists.transform(node.getChilds(), Node.TYPE_FUNCTION) : Collections.<Class<?>>emptySet();
219    }
220
221    @Override
222    public Class<?> getParent(Object itemId) {
223        final Node child = this.getJavaObject(itemId);
224        if (child == null)
225            return null;
226        final Node parent = child.getParent();
227        if (parent == null)
228            return null;
229        return parent.getType();
230    }
231
232    @Override
233    public Collection<Class<?>> rootItemIds() {
234        return Lists.transform(this.rootList, Node.TYPE_FUNCTION);
235    }
236
237    @Override
238    public boolean areChildrenAllowed(Object itemId) {
239        final Node node = this.getJavaObject(itemId);
240        return node != null && !node.getChilds().isEmpty();
241    }
242
243    @Override
244    public boolean isRoot(Object itemId) {
245        final Node node = this.getJavaObject(itemId);
246        return node != null && node.getParent() == null;
247    }
248
249    @Override
250    public boolean hasChildren(Object itemId) {
251        final Node node = this.getJavaObject(itemId);
252        return node != null && !node.getChilds().isEmpty();
253    }
254
255// Mutators are not supported
256
257    @Override
258    public boolean setParent(Object itemId, Object newParentId) {
259        throw new UnsupportedOperationException();
260    }
261
262    @Override
263    public boolean setChildrenAllowed(Object itemId, boolean areChildrenAllowed) {
264        throw new UnsupportedOperationException();
265    }
266
267    @Override
268    public boolean removeItem(Object itemId) {
269        throw new UnsupportedOperationException();
270    }
271}
