006package org.jsimpledb.gui;
007
008import com.google.common.base.Function;
009import com.google.common.collect.Iterators;
010
011import java.util.Iterator;
012
013import org.jsimpledb.CopyState;
014import org.jsimpledb.JObject;
015import org.jsimpledb.JSimpleDB;
016import org.jsimpledb.JTransaction;
017import org.jsimpledb.SnapshotJTransaction;
018import org.jsimpledb.ValidationMode;
019
020/**
021 * Vaadin {@link com.vaadin.data.Container} backed by {@link JSimpleDB} Java model objects acquired by performing
022 * a transactional query and copying out the results into a {@link SnapshotJTransaction}.
023 *
024 * <p>
025 * The subclass method {@linkplain #queryForObjects queryForObjects()} performs the query returning the backing objects
026 * within an already-opened transaction.  The {@link com.vaadin.data.Item}s in the container are backed by in-memory copies
027 * of the returned {@link JObject}s; these live inside a {@link SnapshotJTransaction} and therefore may persist indefinitely
028 * after the query transaction closes.
029 *
030 * <p>
031 * Instances may be (re)loaded at any time by invoking {@link #reload}. During reload, the container opens a new
032 * {@link JTransaction}, queries for objects using {@link #queryForObjects queryForObjects()}, and copies each returned
033 * object into the associated {@link SnapshotJTransaction} via {@link #copyWithRelated copyWithRelated()},
034 * which copies the object and any related objects necessary for resolving container properties. The set of related
035 * objects associated with an object is determined by {@link #getRelatedObjects getRelatedObjects()}; by default,
036 * this is all directly referenced objects. Subclasses may override to customize.
037 *
038 * <p>
039 * It is up to {@link #queryForObjects queryForObjects()} to determine what and how many objects are returned, their
040 * sort order, etc. Any objects returned by {@link #queryForObjects queryForObjects()} that are not instances of the
041 * container's {@linkplain #setType configured type} are ignored.
042 *
043 * <p>
044 * <b>{@link org.dellroad.stuff.vaadin7.ProvidesProperty &#64;ProvidesProperty} Limitations</b>
045 *
046 * <p>
047 * The use of {@link org.dellroad.stuff.vaadin7.ProvidesProperty &#64;ProvidesProperty} methods has certain implications.
048 * First, if the method reads any of the object's field(s) via their Java getter methods (as would normally be expected),
049 * this will trigger a schema upgrade of the object if needed; however, this schema upgrade will occur in the
050 * container's in-memory {@link SnapshotJTransaction} rather than in a real database transaction, so the
051 * {@link #VERSION_PROPERTY} will return a different schema version from what's in the database. The automatic schema
052 * upgrade can be avoided if desired by reading the field using the appropriate {@link JTransaction} field access method
053 * (e.g., {@link JTransaction#readSimpleField JTransaction.readSimpleField()}) and being prepared to handle a
054 * {@link org.jsimpledb.core.UnknownFieldException} if/when the object has an older schema version that does not contain
055 * the requested field.
056 *
057 * <p>
058 * Secondly, because the values of reference fields (including complex sub-fields) are displayed using reference labels,
059 * and these are typically derived from the referenced object's fields, those indirectly referenced objects need to be
060 * copied into the container's {@link SnapshotJTransaction} as well. The easiest way to ensure these indirectly
061 * referenced objects are copied is by overriding {@link #getRelatedObjects getRelatedObjects()} as described above.
062 */
063@SuppressWarnings("serial")
064public abstract class QueryJObjectContainer extends ReloadableJObjectContainer {
065
066    /**
067     * Constructor.
068     *
069     * @param jdb {@link JSimpleDB} database
070     * @param type type restriction, or null for no restriction
071     * @throws IllegalArgumentException if {@code jdb} is null
072     */
073    protected QueryJObjectContainer(JSimpleDB jdb, Class<?> type) {
074        super(jdb, type);
075    }
076
077    /**
078     * (Re)load this container.
079     *
080     * <p>
081     * This creates a new {@link JTransaction}, invokes {@link #queryForObjects} to query for backing objects,
082     * copies them into an in-memory {@link SnapshotJTransaction} via {@link #copyWithRelated copyWithRelated()},
083     * and builds the container from the result.
084     */
085    @Override
086    public void reload() {
087        this.doInTransaction(new Runnable() {
088            @Override
089            public void run() {
090                QueryJObjectContainer.this.reloadInTransaction();
091            }
092        });
093    }
094
095    // This method runs within a transaction
096    private void reloadInTransaction() {
097
098        // Get objects from subclass
099        Iterator<? extends JObject> jobjs = this.queryForObjects();
100
101        // Copy objects (and their related object friends) into a snapshot transaction
102        final SnapshotJTransaction snapshotTx = JTransaction.getCurrent().createSnapshotTransaction(ValidationMode.DISABLED);
103        final CopyState copyState = new CopyState();
104        jobjs = Iterators.transform(jobjs, new Function<JObject, JObject>() {
105            @Override
106            public JObject apply(JObject jobj) {
107                return QueryJObjectContainer.this.copyWithRelated(jobj, snapshotTx, copyState);
108            }
109        });
110
111        // Now actually load the objects
112        this.load(jobjs);
113    }
114
115    /**
116     * Copy the given database object, and any related objects needed by any
117     * {@link org.dellroad.stuff.vaadin7.ProvidesProperty &#64;ProvidesProperty}-annotated methods,
118     * into the specified transaction.
119     *
120     * <p>
121     * The implementation in {@link JObjectContainer} copies {@code jobj}, and all of {@code jobj}'s related objects returned
122     * by {@link #getRelatedObjects getRelatedObjects()}, via {@link JTransaction#copyTo(JTransaction, CopyState, Iterable)}.
123     *
124     * @param target the object to copy, or null (ignored)
125     * @param dest destination transaction
126     * @param copyState tracks what's already been copied
127     * @return the copy of {@code target} in {@code dest}, or null if {@code target} is null
128     * @see #getRelatedObjects getRelatedObjects()
129     */
130    public JObject copyWithRelated(JObject target, JTransaction dest, CopyState copyState) {
131
132        // Ignore null
133        if (target == null)
134            return null;
135
136        // Copy out target object
137        final JTransaction jtx = target.getTransaction();
138        final JObject copy = target.copyTo(dest, null, copyState);
139
140        // Copy out target's related objects
141        final Iterable<? extends JObject> relatedObjects = this.getRelatedObjects(target);
142        if (relatedObjects != null)
143            jtx.copyTo(dest, copyState, relatedObjects);
144
145        // Done
146        return copy;
147    }
148
149    /**
150     * Find objects related to the specified object that are needed by any
151     * {@link org.dellroad.stuff.vaadin7.ProvidesProperty &#64;ProvidesProperty}-annotated
152     * methods.
153     *
154     * <p>
155     * This defines all of the other objects on which any container property of {@code jobj} may depend.
156     * These related objects will copied along with {@code obj} into the container's
157     * {@link SnapshotJTransaction} when the container is (re)loaded.
158     *
159     * <p>
160     * The implementation in {@link JObjectContainer} returns all objects that are directly referenced by {@code jobj},
161     * delegating to {@link JSimpleDB#getReferencedObjects JSimpleDB.getReferencedObjects()}.
162     * Subclasses may override this method to refine the selection.
163     *
164     * @param jobj the object being copied
165     * @return {@link Iterable} of additional objects to be copied, or null for none; any null values are ignored
166     * @throws IllegalArgumentException if {@code jobj} is null
167     */
168    protected Iterable<? extends JObject> getRelatedObjects(JObject jobj) {
169        return this.jdb.getReferencedObjects(jobj);
170    }
171
172    /**
173     * Query for the database objects that will be used to fill this container. Objects should be returned in the
174     * desired order; duplicates and null values will be ignored.
175     *
176     * <p>
177     * A {@link JTransaction} will be open in the current thread when this method is invoked.
178     *
179     * @return database objects
180     */
181    protected abstract Iterator<? extends JObject> queryForObjects();
182}
