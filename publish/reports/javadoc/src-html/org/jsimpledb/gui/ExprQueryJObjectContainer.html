006package org.jsimpledb.gui;
007
008import com.google.common.collect.Iterables;
009
010import java.util.Collections;
011
012import org.dellroad.stuff.vaadin7.VaadinApplicationListener;
013import org.dellroad.stuff.vaadin7.VaadinConfigurable;
014import org.jsimpledb.JObject;
015import org.jsimpledb.JSimpleDB;
016import org.jsimpledb.parse.ParseContext;
017import org.jsimpledb.parse.ParseException;
018import org.jsimpledb.parse.ParseSession;
019import org.jsimpledb.parse.expr.EvalException;
020import org.jsimpledb.parse.expr.ExprParser;
021import org.jsimpledb.parse.expr.Node;
022import org.jsimpledb.util.CastFunction;
023import org.springframework.beans.factory.annotation.Autowired;
024import org.springframework.beans.factory.annotation.Qualifier;
025import org.springframework.context.event.ApplicationEventMulticaster;
026
027/**
028 * {@link JObjectContainer} whose contents are determined by a Java expression.
029 * Listens for {@link DataChangeEvent}s broadcast by an autowired {@link ApplicationEventMulticaster}.
030 */
031@SuppressWarnings("serial")
032@VaadinConfigurable(preConstruction = true)
033public class ExprQueryJObjectContainer extends JObjectContainer {
034
035    private final ParseSession session;
036
037    private DataChangeListener dataChangeListener;
038    private String contentExpression;
039
040    @Autowired(required = false)
041    @Qualifier("jsimpledbGuiEventMulticaster")
042    private ApplicationEventMulticaster eventMulticaster;
043
044    /**
045     * Constructor.
046     *
047     * @param jdb underlying database
048     * @param session session for parsing expressions
049     */
050    public ExprQueryJObjectContainer(JSimpleDB jdb, ParseSession session) {
051        this(jdb, null, session);
052    }
053
054    /**
055     * Constructor.
056     *
057     * @param jdb underlying database
058     * @param type type restriction, or null for no restriction
059     * @param session session for parsing expressions
060     */
061    public ExprQueryJObjectContainer(JSimpleDB jdb, Class<?> type, ParseSession session) {
062        super(jdb, type);
063        if (session == null)
064            throw new IllegalArgumentException("null session");
065        this.session = session;
066    }
067
068    /**
069     * Configure the expression that returns this container's content. Container will automatically reload.
070     *
071     * @param contentExpression Java expression for container content
072     */
073    public void setContentExpression(String contentExpression) {
074        this.contentExpression = contentExpression;
075        this.reload();
076    }
077
078    @Override
079    protected void doInTransaction(final Runnable action) {
080        this.session.perform(new ParseSession.Action() {
081            @Override
082            public void run(ParseSession session) {
083                action.run();
084            }
085        });
086    }
087
088    @Override
089    protected Iterable<? extends JObject> queryForObjects() {
090
091        // Any content?
092        if (this.contentExpression == null)
093            return Collections.<JObject>emptySet();
094
095        // Parse expression
096        final ParseContext ctx = new ParseContext(this.contentExpression);
097        final Node node = new ExprParser().parse(this.session, ctx, false);
098        ctx.skipWhitespace();
099        if (!ctx.isEOF())
100            throw new ParseException(ctx, "syntax error");
101
102        // Evaluate parsed expression
103        final Object content = node.evaluate(this.session).get(this.session);
104        if (!(content instanceof Iterable)) {
105            throw new EvalException("expression must evaluate to an Iterable; found "
106              + (content != null ? content : "null") + " instead");
107        }
108
109        // Reload container with results of expression
110        return Iterables.transform((Iterable<?>)content, new CastFunction<JObject>(JObject.class));
111    }
112
113// Connectable
114
115    @Override
116    public void connect() {
117        super.connect();
118        if (this.eventMulticaster != null) {
119            this.dataChangeListener = new DataChangeListener();
120            this.dataChangeListener.register();
121        }
122    }
123
124    @Override
125    public void disconnect() {
126        if (this.dataChangeListener != null) {
127            this.dataChangeListener.unregister();
128            this.dataChangeListener = null;
129        }
130        super.disconnect();
131    }
132
133// DataChangeListener
134
135    private class DataChangeListener extends VaadinApplicationListener<DataChangeEvent> {
136
137        DataChangeListener() {
138            super(ExprQueryJObjectContainer.this.eventMulticaster);
139            this.setAsynchronous(true);
140        }
141
142        @Override
143        protected void onApplicationEventInternal(DataChangeEvent event) {
144            ExprQueryJObjectContainer.this.handleChange(event.getChange());
145        }
146    }
147}
