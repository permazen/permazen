006package org.jsimpledb.gui;
007
008import com.vaadin.ui.DefaultFieldFactory;
009import com.vaadin.ui.Table;
010
011import org.jsimpledb.JSimpleDB;
012import org.jsimpledb.Session;
013
014/**
015 * Table showing all objects of a certain type, backed by an {@link JObjectContainer}.
016 */
017@SuppressWarnings("serial")
018public class JObjectTable extends AbstractTable<JObjectContainer> {
019
020    private final JSimpleDB jdb;
021    private final JObjectContainer container;
022    private final Session session;
023    private final boolean showFields;
024
025    public JObjectTable(JSimpleDB jdb, JObjectContainer container, Session session) {
026        this(jdb, container, session, true);
027    }
028
029    public JObjectTable(JSimpleDB jdb, JObjectContainer container, Session session, boolean showFields) {
030        if (jdb == null)
031            throw new IllegalArgumentException("null jdb");
032        if (container == null)
033            throw new IllegalArgumentException("null container");
034        if (session == null)
035            throw new IllegalArgumentException("null session");
036        this.jdb = jdb;
037        this.container = container;
038        this.session = session;
039        this.showFields = showFields;
040        this.setSelectable(true);
041        this.setImmediate(true);
042        this.setSizeFull();
043    }
044
045    @Override
046    protected JObjectContainer buildContainer() {
047        return this.container;
048    }
049
050    @Override
051    protected void configureColumns() {
052        this.setColumnCollapsingAllowed(true);
053        for (String propertyName : this.getContainer().getOrderedPropertyNames()) {
054            String title = DefaultFieldFactory.createCaptionByPropertyId(propertyName);
055            Table.Align align = Table.Align.CENTER;
056            int width = 120;
057            boolean showField = this.showFields;
058            switch (propertyName) {
059            case JObjectContainer.REFERENCE_LABEL_PROPERTY:
060                title = "Label";
061                width = 120;
062                showField = true;
063                break;
064            case JObjectContainer.OBJECT_ID_PROPERTY:
065                title = "ID";
066                width = 120;
067                showField = false;
068                break;
069            case JObjectContainer.TYPE_PROPERTY:
070                title = "Type";
071                width = 80;
072                break;
073            case JObjectContainer.VERSION_PROPERTY:
074                title = "Schema Version";
075                width = 40;
076                showField = false;
077                break;
078            default:
079                break;
080            }
081            this.addColumn(propertyName, title, width, align);
082            this.setColumnExpandRatio(propertyName, width / 120.0f);
083            this.setColumnCollapsed(propertyName, !showField);
084        }
085    }
086}
