006package org.jsimpledb.gui;
007
008import com.vaadin.data.Container;
009import com.vaadin.ui.Table;
010
011import java.util.ArrayList;
012
013import org.dellroad.stuff.vaadin7.Connectable;
014import org.slf4j.Logger;
015import org.slf4j.LoggerFactory;
016
017/**
018 * Support superclass for {@link Table}s that are based on {@link Connectable} {@link Container}s.
019 *
020 * @param <C> container type
021 */
022@SuppressWarnings("serial")
023public abstract class AbstractTable<C extends Container & Connectable> extends Table {
024
025    protected final Logger log = LoggerFactory.getLogger(this.getClass());
026
027    private C container;
028
029    private ArrayList<String> columnIds;
030
031    protected AbstractTable() {
032    }
033
034    protected AbstractTable(String caption) {
035        super(caption);
036    }
037
038    protected void addColumn(String property, String name, int width, Table.Align alignment) {
039        if (property == null)
040            throw new IllegalArgumentException("null property");
041        if (name == null)
042            throw new IllegalArgumentException("null name");
043        this.setColumnHeader(property, name);
044        this.setColumnWidth(property, width);
045        if (alignment != null)
046            this.setColumnAlignment(property, alignment);
047        this.columnIds.add(property);
048    }
049
050    public C getContainer() {
051        return this.container;
052    }
053
054    protected abstract C buildContainer();
055
056    protected abstract void configureColumns();
057
058// Vaadin lifecycle
059
060    @Override
061    public void attach() {
062        super.attach();
063
064        // Connect container
065        this.container = this.buildContainer();
066        this.container.connect();
067        this.setContainerDataSource(this.container);
068
069        // Setup columns
070        this.columnIds = new ArrayList<String>();
071        this.configureColumns();
072        this.setVisibleColumns(this.columnIds.toArray());
073    }
074
075    @Override
076    public void detach() {
077        if (this.container != null) {
078            this.container.disconnect();
079            this.container = null;
080        }
081        super.detach();
082    }
083}
