031/**
032 * Contains widgets that allow choosing an object reference. Supports searching via Java expression and filtering.
033 */
034@SuppressWarnings({ "serial", "deprecation" })
035public class JObjectChooser implements Property.ValueChangeNotifier {
036
037    protected final Logger log = LoggerFactory.getLogger(this.getClass());
038
039    private final Button showButton = new Button("Show", new Button.ClickListener() {
040        @Override
041        public void buttonClick(Button.ClickEvent event) {
042            JObjectChooser.this.showButtonClicked();
043        }
044    });
045    private final CheckBox reverseCheckBox = new CheckBox("Reverse sort");
046
047    private final JSimpleDB jdb;
048    private final ParseSession session;
049    private final boolean showFields;
050    private final TypeContainer typeContainer;
051    private final ExprQueryJObjectContainer objectContainer;
052    private final HashSet<Property.ValueChangeListener> listeners = new HashSet<>();
053
054    private final HorizontalSplitPanel splitPanel = new HorizontalSplitPanel();
055    private final FormLayout showForm = new FormLayout();
056    private final TypeTable typeTable;
057    private final TextArea exprField = new TextArea();
058    private final ComboBox sortComboBox = new ComboBox();
059
060    private JObjectTable objectTable;
061    private SortKeyContainer sortKeyContainer;
062
063    /**
064     * Constructor.
065     *
066     * @param jdb database
067     * @param session session for evaluating expressions
068     * @param type type restriction, or null for none
069     * @param showFields true to show all object fields, false for just reference label
070     */
071    public JObjectChooser(JSimpleDB jdb, ParseSession session, Class<?> type, boolean showFields) {
072        if (jdb == null)
073            throw new IllegalArgumentException("null jdb");
074        if (session == null)
075            throw new IllegalArgumentException("null session");
076        this.jdb = jdb;
077        this.session = session;
078        this.showFields = showFields;
079        this.typeContainer = new TypeContainer(this.jdb, type);
080        this.typeTable = new TypeTable(this.typeContainer);
081        this.objectContainer = new ExprQueryJObjectContainer(this.jdb, this.typeContainer.getRootType(), this.session);
082
083        // Build object panel
084        this.splitPanel.setWidth("100%");
085        this.splitPanel.setHeight(300, Sizeable.Unit.PIXELS);
086        this.splitPanel.setFirstComponent(this.typeTable);
087        this.splitPanel.setSecondComponent(new SizedLabel(" "));
088        this.splitPanel.setSplitPosition(20);
089
090        // Build show form
091        this.showForm.setMargin(false);
092        this.showForm.setWidth("100%");
093
094        // Add "Sort by" combo box and "Reverse sort" check box
095        final HorizontalLayout sortLayout = new HorizontalLayout();
096        sortLayout.setCaption("Sort by:");
097        sortLayout.setSpacing(true);
098        sortLayout.setMargin(false);
099        this.sortComboBox.setNullSelectionAllowed(false);
100        this.sortComboBox.setNewItemsAllowed(false);
101        this.sortComboBox.setTextInputAllowed(false);
102        this.sortComboBox.setFilteringMode(FilteringMode.OFF);
103        this.sortComboBox.setItemCaptionMode(AbstractSelect.ItemCaptionMode.PROPERTY);
104        this.sortComboBox.setItemCaptionPropertyId(SortKeyContainer.DESCRIPTION_PROPERTY);
105        this.sortComboBox.setImmediate(true);
106        this.reverseCheckBox.setImmediate(true);
107        sortLayout.addComponent(this.sortComboBox);
108        sortLayout.addComponent(this.reverseCheckBox);
109        this.showForm.addComponent(sortLayout);
110
111        // Add show/expression field
112        this.exprField.setCaption("Expression:");
113        this.exprField.setRows(6);
114        this.exprField.setWidth("100%");
115        this.exprField.addStyleName("jsdb-fixed-width");
116        this.showForm.addComponent(this.exprField);
117        this.showForm.addComponent(this.showButton);
118
119        // Listen to type selections
120        this.typeTable.addValueChangeListener(new Property.ValueChangeListener() {
121            @Override
122            public void valueChange(Property.ValueChangeEvent event) {
123                JObjectChooser.this.selectType((Class<?>)event.getProperty().getValue(), false);
124            }
125        });
126
127        // Listen to "Sort by" and "Reverse sort" selections
128        final Property.ValueChangeListener sortListener = new Property.ValueChangeListener() {
129            @Override
130            public void valueChange(Property.ValueChangeEvent event) {
131                JObjectChooser.this.selectSort();
132            }
133        };
134        this.sortComboBox.addValueChangeListener(sortListener);
135        this.reverseCheckBox.addValueChangeListener(sortListener);
136
137        // Populate table
138        this.selectType(type != null ? type : this.typeContainer.getRootType(), true);
139    }

195    /**
196     * Get the {@link JClass} corresponding to the currently selected type, if any.
197     *
198     * @return selected type object type
199     */
200    public JClass<?> getJClass() {
201        final Class<?> type = this.objectContainer.getType();
202        if (type == null)
203            return null;
204        try {
205            return this.jdb.getJClass(type);
206        } catch (IllegalArgumentException e) {
207            return null;
208        }
209    }

211// Property.ValueChangeNotifier
212
213    @Override
214    public void addValueChangeListener(Property.ValueChangeListener listener) {
215        if (listener == null)
216            throw new IllegalArgumentException("null listener");
217        this.listeners.add(listener);
218        if (this.objectTable != null)
219            this.objectTable.addValueChangeListener(listener);
220    }
221
222    @Override
223    public void removeValueChangeListener(Property.ValueChangeListener listener) {
224        if (listener == null)
225            return;
226        this.listeners.remove(listener);
227        if (this.objectTable != null)
228            this.objectTable.removeValueChangeListener(listener);
229    }
230
231    @Override
232    public void addListener(Property.ValueChangeListener listener) {
233        this.addValueChangeListener(listener);
234    }
235
236    @Override
237    public void removeListener(Property.ValueChangeListener listener) {
238        this.removeValueChangeListener(listener);
239    }

241// GUI Updates
242
243    // Invoked when a type is clicked on
244    private void selectType(Class<?> type, boolean force) {
245
246        // Anything to do?
247        if (type == null)
248            return;
249        if (!this.setNewType(!type.equals(Object.class) ? type : null, force))
250            return;
251
252        // Get previous sort key
253        final SortKeyContainer.SortKey previousSort = (SortKeyContainer.SortKey)this.sortComboBox.getValue();
254
255        // Rebuild the sort combobox
256        final JClass<?> jclass = this.getJClass();
257        this.sortKeyContainer = jclass != null ?
258          new SortKeyContainer(this.jdb, jclass) : new SortKeyContainer(this.jdb, this.objectContainer.getType());
259        this.sortComboBox.setContainerDataSource(this.sortKeyContainer);
260
261        // Try to restore previous sort, otherwise default to sorting by object ID
262        SortKeyContainer.SortKey newSort = this.sortKeyContainer.getJavaObject(previousSort);
263        if (newSort == null)
264            newSort = this.sortKeyContainer.new ObjectIdSortKey();
265        this.sortComboBox.setValue(newSort);
266        this.selectSort();
267        this.showObjects();
268    }
269
270    // Update sort based on current selection
271    private void selectSort() {
272        this.selectSort((SortKeyContainer.SortKey)this.sortComboBox.getValue(), this.reverseCheckBox.getValue());
273    }
274
275    // Invoked when a sort choice is made
276    private void selectSort(SortKeyContainer.SortKey sortKey, boolean reverse) {
277        if (sortKey == null)
278            return;
279        this.exprField.setValue(sortKey.getExpression(this.session, null, reverse));          // TODO: starting point
280        this.showObjects();
281    }
282
283    // Invoked when "Show" button is clicked
284    private void showButtonClicked() {
285        this.setNewType(null, false);
286        this.typeTable.setValue(null);
287        this.showObjects();
288    }
289
290    private void showObjects() {
291        this.objectContainer.setContentExpression(this.exprField.getValue());
292    }
293
294    private boolean setNewType(Class<?> type, boolean force) {
295        final Class<?> currentType = this.objectContainer.getType();
296        if (!force && (currentType != null ? currentType.equals(type) : type == null))
297            return false;
298        if (this.objectTable != null) {
299            for (Property.ValueChangeListener listener : this.listeners)
300                this.objectTable.removeValueChangeListener(listener);
301            this.splitPanel.removeComponent(this.objectTable);
302        }
303        this.objectContainer.setType(type);
304        this.objectContainer.load(Collections.<JObject>emptySet());
305        this.objectTable = new JObjectTable(this.jdb, this.objectContainer, this.session, this.showFields);
306        for (Property.ValueChangeListener listener : this.listeners)
307            this.objectTable.addValueChangeListener(listener);
308        this.splitPanel.setSecondComponent(this.objectTable);
309        this.objectTable.setValue(null);
310        final Property.ValueChangeEvent event = new Property.ValueChangeEvent() {
311            @Override
312            public Property<?> getProperty() {
313                return JObjectChooser.this.objectTable;
314            }
315        };
316        for (Property.ValueChangeListener listener : new HashSet<Property.ValueChangeListener>(this.listeners)) {
317            try {
318                listener.valueChange(event);
319            } catch (Exception e) {
320                this.log.error("exception thrown by value change listener", e);
321            }
322        }
323        return true;
324    }
