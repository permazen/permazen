076    private final GUIConfig guiConfig;

081    @Autowired(required = false)

084    public MainPanel(GUIConfig guiConfig) {
085        this.guiConfig = guiConfig;
086        this.jdb = this.guiConfig.getJSimpleDB();
087
088        // Setup parse session
089        this.session = new ParseSession(this.jdb) {
090            @Override
091            protected void reportException(Exception e) {
092                Notification.show("Error: " + e.getMessage(), null, Notification.Type.ERROR_MESSAGE);
093                if (MainPanel.this.guiConfig.isVerbose())
094                    MainPanel.this.log.info("exception in parse session", e);
095            }
096        };
097        this.session.setReadOnly(this.guiConfig.isReadOnly());
098        this.session.setSchemaModel(this.jdb.getSchemaModel());
099        this.session.setSchemaVersion(this.guiConfig.getSchemaVersion());
100        this.session.setAllowNewSchema(this.guiConfig.isAllowNewSchema());
101
102        // Register built-in functions
103        this.session.registerStandardFunctions();
104
105        // Register custom functions
106        final Iterable<? extends Class<?>> customFunctionClasses = this.guiConfig.getFunctionClasses();
107        if (customFunctionClasses != null) {
108            for (Class<?> cl : customFunctionClasses) {
109                try {
110                    this.session.registerFunction(cl);
111                } catch (IllegalArgumentException e) {
112                    this.log.warn("failed to register function " + cl + ": " + e.getMessage());
113                }
114            }
115        }
116
117        // Setup object chooser
118        this.objectChooser = new JObjectChooser(this.jdb, this.session, null, true);
119
120        // Listen to object selections
121        this.objectChooser.addValueChangeListener(new Property.ValueChangeListener() {
122            @Override
123            public void valueChange(Property.ValueChangeEvent event) {
124                MainPanel.this.selectObject(MainPanel.this.objectChooser.getObjId());
125            }
126        });
127    }

168// GUI Updates

170    // Invoked when an object is clicked on
171    protected void selectObject(ObjId id) {
172
173        // New button
174        this.newButton.setEnabled(this.objectChooser.getJClass() != null);
175
176        // Update buttons
177        if (id == null) {
178            this.editButton.setEnabled(false);
179            this.deleteButton.setEnabled(false);
180            this.upgradeButton.setEnabled(false);
181        } else {
182            this.editButton.setEnabled(true);
183            this.deleteButton.setEnabled(true);
184            this.upgradeButton.setEnabled(this.canUpgrade(id));
185        }
186    }

188// Edit
189
190    private void editButtonClicked(ObjId id) {
191        this.log.info("editing object " + id);
192
193        // Copy object
194        final JObject jobj = this.doCopyForEdit(id);
195        if (jobj == null) {
196            Notification.show("Object " + id + " no longer exists", null, Notification.Type.WARNING_MESSAGE);
197            return;
198        }
199
200        // Ensure object type is known
201        if (jobj instanceof UntypedJObject) {
202            Notification.show("Can't edit object " + id + " having unknown object type",
203              "Storage ID " + id.getStorageId() + " not defined in the current schema version",
204              Notification.Type.WARNING_MESSAGE);
205            return;
206        }
207
208        // Build title component from reference label
209        Object refLabel = MainPanel.this.objectChooser.getJObjectContainer().getContainerProperty(
210          id, JObjectContainer.REFERENCE_LABEL_PROPERTY).getValue();
211        final Component titleComponent = refLabel instanceof Component ? (Component)refLabel : new Label("" + refLabel);
212
213        // Open window
214        new JObjectEditorWindow(this.getUI(), this.session, this.jdb.getJClass(id), jobj, titleComponent).show();
215    }
216
217    @RetryTransaction
218    @Transactional("jsimpledbGuiTransactionManager")
219    private JObject doCopyForEdit(ObjId id) {
220
221        // Get object
222        final JObject jobj = JTransaction.getCurrent().getJObject(id);
223        if (!jobj.exists())
224            return null;
225
226        // Copy object and dependencies
227        return this.objectChooser.getJObjectContainer().copyOut(jobj, new CopyState());
228    }

232    private void newButtonClicked() {
233        final JClass<?> jclass = this.objectChooser.getJClass();
234        if (jclass == null) {
235            Notification.show("Can't create object having unknown type",
236              "Please select an object type first", Notification.Type.WARNING_MESSAGE);
237            return;
238        }
239        this.log.info("creating new object of type " + jclass.getType().getName());
240        new JObjectEditorWindow(this.getUI(), this.session, jclass).show();
241    }

243// Delete
244
245    private void deleteButtonClicked(ObjId id) {
246        this.log.info("deleting object " + id);
247        final boolean deleted;
248        try {
249            deleted = this.doDelete(id);
250        } catch (DeletedObjectException e) {
251            Notification.show("Object " + e.getId() + " no longer exists", null, Notification.Type.WARNING_MESSAGE);
252            return;
253        } catch (ReferencedObjectException e) {
254            Notification.show("Object " + id + " is referenced by " + e.getReferrer(),
255              e.getMessage(), Notification.Type.ERROR_MESSAGE);
256            return;
257        }
258        if (deleted)
259            Notification.show("Removed object " + id);
260        else
261            Notification.show("Could not delete object " + id, null, Notification.Type.WARNING_MESSAGE);
262        this.selectObject(null);
263    }
264
265    @RetryTransaction
266    @Transactional("jsimpledbGuiTransactionManager")
267    private boolean doDelete(ObjId id) {
268        final JObject jobj = JTransaction.getCurrent().getJObject(id);
269        final boolean deleted = jobj.delete();
270        if (deleted && this.changePublisher != null)
271            this.changePublisher.publishChangeOnCommit(new ObjectDelete<Object>(jobj));
272        return deleted;
273    }
274
275// Upgrade
276
277    private void upgradeButtonClicked(ObjId id) {
278        final int newVersion = this.jdb.getActualVersion();
279        this.log.info("upgrading object " + id + " to schema version " + newVersion);
280        final int oldVersion = this.doUpgrade(id);
281        switch (oldVersion) {
282        case -1:
283            Notification.show("Object " + id + " no longer exists", null, Notification.Type.WARNING_MESSAGE);
284            break;
285        case 0:
286            Notification.show("Object " + id + " was already upgraded", null, Notification.Type.WARNING_MESSAGE);
287            break;
288        default:
289            Notification.show("Upgraded object " + id + " version from " + oldVersion + " to " + newVersion);
290            break;
291        }
292    }
293
294    @RetryTransaction
295    @Transactional("jsimpledbGuiTransactionManager")
296    private int doUpgrade(ObjId id) {
297        final JObject jobj = JTransaction.getCurrent().getJObject(id);
298        final int oldVersion;
299        try {
300            oldVersion = jobj.getSchemaVersion();
301        } catch (DeletedObjectException e) {
302            return -1;
303        }
304        final boolean upgraded = jobj.upgrade();
305        if (upgraded && this.changePublisher != null)
306            this.changePublisher.publishChangeOnCommit(jobj);
307        return upgraded ? oldVersion : 0;
308    }
309
310    @RetryTransaction
311    @Transactional("jsimpledbGuiTransactionManager")
312    private boolean canUpgrade(ObjId id) {
313        final JObject jobj = JTransaction.getCurrent().getJObject(id);
314        return jobj.exists() && jobj.getSchemaVersion() != this.jdb.getActualVersion();
315    }
