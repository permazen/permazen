006package org.jsimpledb.gui;
007
008import org.jsimpledb.change.Change;
009import org.jsimpledb.change.ChangeCopier;
010import org.springframework.context.ApplicationEvent;
011
012/**
013 * Spring application event that notifies about a {@link Change} to the database.
014 */
015@SuppressWarnings("serial")
016public class DataChangeEvent extends ApplicationEvent {
017
018    private final Change<?> change;
019
020    /**
021     * Constructor. Assumes there is a current transaction open.
022     *
023     * @param source event source
024     * @param change change information; will be copied out of the current transaction with a {@link ChangeCopier}
025     */
026    public DataChangeEvent(Object source, Change<?> change) {
027        super(source);
028        if (change == null)
029            throw new IllegalArgumentException("null change");
030        this.change = change.visit(new ChangeCopier());
031    }
032
033    /**
034     * Get the change that occurred, already copied out of the transaction with a {@link ChangeCopier}.
035     * Related objects are also copied.
036     *
037     * @return the associated change
038     */
039    public Change<?> getChange() {
040        return this.change;
041    }
042}
