006package org.jsimpledb.gui;
007
008import com.vaadin.shared.ui.datefield.Resolution;
009import com.vaadin.ui.CheckBox;
010import com.vaadin.ui.Field;
011import com.vaadin.ui.PopupDateField;
012import com.vaadin.ui.TextField;
013
014import java.util.Date;
015
016import org.dellroad.stuff.vaadin7.EnumComboBox;
017import org.jsimpledb.JReferenceField;
018import org.jsimpledb.JSimpleField;
019import org.jsimpledb.JTransaction;
020import org.jsimpledb.core.FieldType;
021import org.jsimpledb.parse.ParseSession;
022
023/**
024 * Builds {@link Field}s for editing {@link JSimpleField} values.
025 */
026@SuppressWarnings("serial")
027public class SimpleFieldFieldBuilder {
028
029    private final JTransaction jtx;
030    private final JSimpleField jfield;
031    private final ParseSession session;
032    private final boolean allowNull;
033
034    /**
035     * Constructor.
036     *
037     * @param jtx target transaction used by {@link JObjectChooser}
038     * @param jfield the database {@link JSimpleField} for which to build a Vaadin {@link Field}
039     * @param session session used by {@link JObjectChooser}
040     * @param allowNull whether null values are allowed
041     */
042    public SimpleFieldFieldBuilder(JTransaction jtx, JSimpleField jfield, ParseSession session, boolean allowNull) {
043        if (jtx == null)
044            throw new IllegalArgumentException("null jtx");
045        if (jfield == null)
046            throw new IllegalArgumentException("null jfield");
047        if (session == null)
048            throw new IllegalArgumentException("null session");
049        this.jtx = jtx;
050        this.jfield = jfield;
051        this.session = session;
052        this.allowNull = allowNull && !jfield.getTypeToken().isPrimitive();
053    }
054
055    /**
056     * Build a {@link Field} appropriate for the configured {@link JSimpleField}.
057     *
058     * @return Vaadin {@link Field} for editing the field value
059     */
060    public Field<?> buildField() {
061
062        // Get property type
063        final Class<?> propertyType = jfield.getTypeToken().getRawType();
064
065        // Build an appropriate field
066        Field<?> field;
067        if (jfield instanceof JReferenceField)                                          // use object choosers for references
068            field = new ReferenceFieldField(this.jtx, this.session, jfield.getName(), jfield.getTypeToken().getRawType());
069        else if (Enum.class.isAssignableFrom(propertyType)) {                           // use ComboBox for Enum's
070            final EnumComboBox comboBox = this.createEnumComboBox(propertyType.asSubclass(Enum.class));
071            comboBox.setInputPrompt("Null");
072            field = comboBox;
073        } else if (propertyType.isAssignableFrom(Date.class)) {                         // use DatePicker for dates
074            final PopupDateField dateField = new PopupDateField();
075            dateField.setResolution(Resolution.SECOND);
076            field = dateField;
077        } else if (propertyType == boolean.class || propertyType == Boolean.class)      // use CheckBox for booleans
078            field = new CheckBox();
079        else {                                                                          // use text field for all others
080            final TextField textField = new TextField();
081            textField.setWidth("100%");
082            textField.setNullRepresentation("");
083            textField.setNullSettingAllowed(false);
084            final FieldType<?> fieldType = jfield.getFieldType();
085            textField.setConverter(this.buildSimpleFieldConverter(fieldType));
086            field = textField;
087        }
088
089        // Add a "Null" button if field can be null
090        if (this.allowNull)
091            field = this.addNullButton(field);
092
093        // Done
094        return field;
095    }
096
097    @SuppressWarnings({ "unchecked", "rawtypes" })
098    private <T extends Enum> EnumComboBox createEnumComboBox(Class<T> enumType) {
099        return new EnumComboBox(enumType, this.allowNull);
100    }
101
102    // This method exists solely to bind the generic type parameters
103    private <T> SimpleFieldConverter<T> buildSimpleFieldConverter(FieldType<T> fieldType) {
104        return new SimpleFieldConverter<T>(fieldType);
105    }
106
107    // This method exists solely to bind the generic type parameters
108    private <T> NullableField<T> addNullButton(Field<T> field) {
109        return new NullableField<T>(field);
110    }
111}
