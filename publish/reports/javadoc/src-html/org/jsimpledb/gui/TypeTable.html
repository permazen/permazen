006package org.jsimpledb.gui;
007
008import com.vaadin.ui.Table;
009import com.vaadin.ui.TreeTable;
010
011/**
012 * Table showing object types.
013 */
014@SuppressWarnings("serial")
015public class TypeTable extends TreeTable {
016
017    public TypeTable(TypeContainer container) {
018        super(null, container);
019
020        this.setSelectable(true);
021        this.setImmediate(true);
022        this.setSizeFull();
023        this.setHierarchyColumn(TypeContainer.NAME_PROPERTY);
024        this.setAnimationsEnabled(true);
025
026        this.addColumn(TypeContainer.NAME_PROPERTY, "Type", 140, Table.Align.LEFT);
027        this.addColumn(TypeContainer.STORAGE_ID_PROPERTY, "SID", 40, Table.Align.CENTER);
028        this.addColumn(TypeContainer.TYPE_PROPERTY, "Java Type", 250, Table.Align.CENTER);
029
030        this.setColumnExpandRatio(TypeContainer.NAME_PROPERTY, 1.0f);
031
032        this.setVisibleColumns(TypeContainer.NAME_PROPERTY, TypeContainer.STORAGE_ID_PROPERTY, TypeContainer.TYPE_PROPERTY);
033        this.setColumnCollapsingAllowed(true);
034        this.setColumnCollapsed(TypeContainer.STORAGE_ID_PROPERTY, true);
035        this.setColumnCollapsed(TypeContainer.TYPE_PROPERTY, true);
036    }
037
038    protected void addColumn(String property, String name, int width, Table.Align alignment) {
039        this.setColumnHeader(property, name);
040        this.setColumnWidth(property, width);
041        if (alignment != null)
042            this.setColumnAlignment(property, alignment);
043    }
044
045    public TypeContainer getContainer() {
046        return (TypeContainer)this.getContainerDataSource();
047    }
048
049// Vaadin lifecycle
050
051    @Override
052    public void attach() {
053        super.attach();
054        this.getContainer().connect();
055
056        // Expand all root nodes
057        for (Class<?> type : this.getContainer().rootItemIds())
058            this.setCollapsed(type, false);
059    }
060
061    @Override
062    public void detach() {
063        this.getContainer().disconnect();
064        super.detach();
065    }
066}
