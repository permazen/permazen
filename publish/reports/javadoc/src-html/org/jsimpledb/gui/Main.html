006package org.jsimpledb.gui;
007
008import java.io.File;
009import java.net.URI;
010import java.util.ArrayDeque;
011import java.util.Arrays;
012import java.util.LinkedHashSet;
013import java.util.Set;
014
015import org.eclipse.jetty.annotations.AnnotationConfiguration;
016import org.eclipse.jetty.plus.webapp.EnvConfiguration;
017import org.eclipse.jetty.plus.webapp.PlusConfiguration;
018import org.eclipse.jetty.server.Server;
019import org.eclipse.jetty.util.resource.Resource;
020import org.eclipse.jetty.webapp.Configuration;
021import org.eclipse.jetty.webapp.FragmentConfiguration;
022import org.eclipse.jetty.webapp.MetaInfConfiguration;
023import org.eclipse.jetty.webapp.WebAppContext;
024import org.eclipse.jetty.webapp.WebInfConfiguration;
025import org.eclipse.jetty.webapp.WebXmlConfiguration;
026import org.jsimpledb.JSimpleDB;
027import org.jsimpledb.core.Database;
028import org.jsimpledb.parse.func.Function;
029import org.jsimpledb.spring.AnnotatedClassScanner;
030import org.jsimpledb.util.AbstractMain;
031
032/**
033 * GUI main entry point.
034 */
035public class Main extends AbstractMain implements GUIConfig {
036
037    private static final int DEFAULT_HTTP_PORT = 8080;
038
039    private static Main instance;
040
041    private JSimpleDB jdb;
042    private Server server;
043    private int port = DEFAULT_HTTP_PORT;
044    private URI root;
045    private final LinkedHashSet<Class<?>> functionClasses = new LinkedHashSet<>();
046
047    @Override
048    protected boolean parseOption(String option, ArrayDeque<String> params) {
049        if (option.equals("--root")) {
050            if (params.isEmpty())
051                this.usageError();
052            this.root = new File(params.removeFirst()).toURI();
053        } else if (option.equals("--port")) {
054            if (params.isEmpty())
055                this.usageError();
056            this.port = Integer.parseInt(params.removeFirst());
057        } else if (option.equals("--func-pkg")) {
058            if (params.isEmpty())
059                this.usageError();
060            this.scanFunctionClasses(params.removeFirst());
061        } else
062            return false;
063        return true;
064    }
065
066    private void scanFunctionClasses(String pkgname) {
067        for (String className : new AnnotatedClassScanner(Function.class).scanForClasses(pkgname.split("[\\s,]")))
068            this.functionClasses.add(this.loadClass(className));
069    }
070
071    @Override
072    public int run(String[] args) throws Exception {
073
074        // Set singleton
075        Main.instance = this;
076
077        // Parse command line
078        final ArrayDeque<String> params = new ArrayDeque<String>(Arrays.asList(args));
079        final int result = this.parseOptions(params);
080        if (result != -1)
081            return result;
082        switch (params.size()) {
083        case 0:
084            break;
085        default:
086            this.usageError();
087            return 1;
088        }
089        if (this.schemaClasses == null) {
090            System.err.println(this.getName() + ": no schema classes defined; please specify using `--pkg'");
091            return 1;
092        }
093
094        // Try to infer root directory from classpath
095        if (this.root == null) {
096            try {
097                URI uri = this.getClass().getResource("/WEB-INF/web.xml").toURI();
098                if (uri.toString().startsWith("jar:"))
099                    this.root = new URI(uri.toString().substring(0, uri.toString().lastIndexOf('!') + 2));
100                else
101                    this.root = this.getClass().getResource("/WEB-INF/web.xml").toURI().resolve("..");
102            } catch (Exception e) {
103                // ignore
104            }
105            if (this.root == null) {
106                System.err.println(this.getName() + ": can't determine install directory; please specify `--root dir'");
107                return 1;
108            }
109        }
110        this.log.debug("using root directory " + this.root);
111
112        // Set up database
113        final Database db = this.startupKVDatabase();
114        try {
115
116            // Create JSimpleDB instance
117            this.jdb = this.getJSimpleDBFactory(db).newJSimpleDB();
118
119            // Perform test transaction
120            this.performTestTransaction(jdb);
121
122            // Create web server with Spring application context
123            this.server = new Server(this.port);
124            final WebAppContext context = new WebAppContext();
125            context.setBaseResource(Resource.newResource(this.root));
126            context.setConfigurations(new Configuration[] {
127                                new AnnotationConfiguration(),
128                                new WebXmlConfiguration(),
129                                new WebInfConfiguration(),
130                                new PlusConfiguration(),
131                                new MetaInfConfiguration(),
132                                new FragmentConfiguration(),
133                                new EnvConfiguration() });
134            context.setContextPath("/");
135            context.setParentLoaderPriority(true);
136            this.server.setHandler(context);
137
138            // Start server
139            this.server.start();
140
141            // Wait for server to stop
142            this.server.join();
143
144            // Done
145            return 0;
146        } finally {
147
148            // Shut down KV database
149            this.shutdownKVDatabase();
150        }
151    }
152
153    @Override
154    public JSimpleDB getJSimpleDB() {
155        return this.jdb;
156    }
157
158    @Override
159    public boolean isVerbose() {
160        return this.verbose;
161    }
162
163    @Override
164    public Set<Class<?>> getFunctionClasses() {
165        return this.functionClasses;
166    }
167
168    @Override
169    protected String getName() {
170        return "jsimpledb-gui";
171    }
172
173    @Override
174    protected void usageMessage() {
175        System.err.println("Usage:");
176        System.err.println("  " + this.getName() + " --pkg package [options]");
177        System.err.println("Options:");
178        this.outputFlags(new String[][] {
179          { "--func-pkg package",   "Register @Function-annotated classes found under the specified Java package" },
180          { "--port port",          "Specify HTTP port (default " + DEFAULT_HTTP_PORT + ")" },
181          { "--root directory",     "Specify GUI install directory" },
182        });
183    }
184
185    public static void main(String[] args) throws Exception {
186        new Main().doMain(args);
187    }
188
189    public static Main getInstance() {
190        return Main.instance;
191    }
192}
