097    /**
098     * Constructor for editing an existing object.
099     *
100     * @param ui associated {@link UI}
101     * @param session parse session for {@link JObjectChooser}
102     * @param jclass object type
103     * @param jobj object to edit; should be contained in a snapshot transaction
104     * @param titleComponent title for edit panel
105     */
106    @SuppressWarnings("unchecked")
107    public JObjectEditorWindow(UI ui, ParseSession session, JClass<?> jclass, JObject jobj, Component titleComponent) {
108        super(ui, (jobj != null ? "Edit " : "New ") + jclass.getName());
109        this.setWidth(600, Sizeable.Unit.PIXELS);
110        this.setHeight(450, Sizeable.Unit.PIXELS);
111        this.jclass = jclass;
112        this.jobj = jobj != null ? jobj : this.doCreateForEdit();
113        this.create = jobj == null;
114        this.session = session;
115        this.titleComponent = titleComponent;
116
117        // Create PropertysetItem to hold our editable properties
118        final PropertysetItem item = new PropertysetItem();
119
120        // Introspect for any @FieldBuilder.* annotations
121        final BeanFieldGroup<?> beanFieldGroup = FieldBuilder.buildFieldGroup(this.jobj);
122        for (Object id : beanFieldGroup.getBoundPropertyIds()) {
123            final String fieldName = (String)id;
124            final Field<?> field = beanFieldGroup.getField(id);
125            final Property<?> property = beanFieldGroup.getItemDataSource().getItemProperty(id);
126            this.fieldMap.put(fieldName, field);
127            item.addItemProperty(fieldName, property);
128        }
129
130        // Build fields and components for all remaining database properties
131        final SortedMap<String, JField> jfieldMap = jclass.getJFieldsByName();
132        for (Map.Entry<String, JField> entry : jfieldMap.entrySet()) {
133            final String fieldName = entry.getKey();
134            final JField jfield = entry.getValue();
135
136            // If a Field already exists for this database field, just use it, otherwise build one
137            Field<?> field = this.fieldMap.get(fieldName);
138            if (field == null) {
139                field = this.buildFieldField(fieldName, jfield);
140                this.fieldMap.put(fieldName, field);
141            }
142
143            // Build an appropriate Vaadin Property for the field
144            item.addItemProperty(fieldName, this.buildFieldProperty(this.jobj, jfield));
145
146            // Set the field's caption
147            field.setCaption(this.buildCaption(jfield.getName(), !(field instanceof CheckBox)));
148        }
149
150        // Connect fields to properties via FieldGroup
151        this.fieldGroup.setItemDataSource(item);
152
153        // Bind fields into FieldGroup
154        for (Map.Entry<String, Field<?>> entry : this.fieldMap.entrySet())
155            this.fieldGroup.bind(entry.getValue(), entry.getKey());
156    }
157
158    @Override
159    protected void addContent(VerticalLayout layout) {
160        if (this.titleComponent != null)
161            layout.addComponent(this.titleComponent);
162        final FormLayout formLayout = new FormLayout();
163        for (Field<?> field : this.fieldMap.values())
164            formLayout.addComponent(field);
165        layout.addComponent(formLayout);
166    }
167
168    @Override
169    protected boolean execute() {
170
171        // Commit fields in field group
172        try {
173            this.fieldGroup.commit();
174        } catch (FieldGroup.CommitException e) {
175            Notification.show("Invalid value(s)", null, Notification.Type.WARNING_MESSAGE);
176            throw new RuntimeException(e);
177        }
178
179        // Commit transaction, if possible
180        try {
181            return this.writeBack();
182        } catch (UnexpectedRollbackException e) {
183            this.log.debug("ignoring UnexpectedRollbackException presumably caused by validation failure");
184            return false;
185        }
186    }
187
188    @RetryTransaction
189    @Transactional("jsimpledbGuiTransactionManager")
190    private JObject doCreateForEdit() {
191        return (JObject)JTransaction.getCurrent().getSnapshotTransaction().create(this.jclass);
192    }
193
194    @RetryTransaction
195    @Transactional("jsimpledbGuiTransactionManager")
196    protected boolean writeBack() {
197
198        // Find/create target object in current transaction
199        final JTransaction jtx = JTransaction.getCurrent();
200        final ObjId id = this.jobj.getObjId();
201        final JObject target = this.create ? (JObject)jtx.create(this.jclass) : jtx.getJObject(id);
202
203        // Verify object still exists when editing
204        if (!this.create && !target.exists()) {
205            Notification.show("Object " + id + " no longer exists", null, Notification.Type.WARNING_MESSAGE);
206            return true;
207        }
208
209        // Copy fields
210        this.jobj.copyTo(jtx, target.getObjId(), new CopyState());
211
212        // Run validation queue
213        try {
214            jtx.validate();
215        } catch (ValidationException e) {
216            Notification.show("Validation failed", e.getMessage(), Notification.Type.ERROR_MESSAGE);
217            jtx.getTransaction().setRollbackOnly();
218            return false;
219        }
220
221        // Broadcast update event after successful commit
222        if (this.changePublisher != null) {
223            if (create)
224                this.changePublisher.publishChangeOnCommit(new ObjectCreate<Object>(target));
225            else
226                this.changePublisher.publishChangeOnCommit(target);
227        }
228
229        // Show notification after successful commit
230        final VaadinSession vaadinSession = VaadinUtil.getCurrentSession();
231        jtx.getTransaction().addCallback(new Transaction.CallbackAdapter() {
232            @Override
233            public void afterCommit() {
234                VaadinUtil.invoke(vaadinSession, new Runnable() {
235                    @Override
236                    public void run() {
237                        Notification.show((JObjectEditorWindow.this.create ? "Created" : "Updated") + " object " + id);
238                    }
239                });
240            }
241        });
242        return true;
243    }

247    private Field<?> buildFieldField(String fieldName, JField jfield) {
248        return jfield.visit(new JFieldSwitchAdapter<Field<?>>() {
249            @Override
250            public Field<?> caseJSimpleField(JSimpleField jfield) {
251                final boolean allowNull = jfield.getGetter().getAnnotation(NotNull.class) == null
252                  && !jfield.getTypeToken().isPrimitive();
253                return new SimpleFieldFieldBuilder(JObjectEditorWindow.this.jobj.getTransaction(),
254                  jfield, JObjectEditorWindow.this.session, allowNull).buildField();
255            }
256            @Override
257            public Field<?> caseJCounterField(JCounterField jfield) {
258                final TextField field = new TextField();
259                field.setWidth("100%");
260                field.setNullSettingAllowed(false);
261                field.setConverter(new SimpleFieldConverter<Long>(JObjectEditorWindow.this.jclass.getJSimpleDB()
262                  .getDatabase().getFieldTypeRegistry().getFieldType(TypeToken.of(long.class))));
263                return field;
264            }
265            @Override
266            public Field<?> caseJSetField(JSetField jfield) {
267                //return new SetFieldFieldBuilder(jfield, JObjectEditorWindow.this.session).buildField();
268                return new PlaceHolderField(jfield);                                       // TODO
269            }
270            @Override
271            public Field<?> caseJListField(JListField jfield) {
272                //return new ListFieldFieldBuilder(jfield, JObjectEditorWindow.this.session).buildField();
273                return new PlaceHolderField(jfield);                                       // TODO
274            }
275            @Override
276            public Field<?> caseJMapField(JMapField jfield) {
277                //return new MapFieldFieldBuilder(jfield, JObjectEditorWindow.this.session).buildField();
278                return new PlaceHolderField(jfield);                                       // TODO
279            }
280        });
281    }

283    private Property<?> buildFieldProperty(final JObject jobj, JField jfield) {
284        return jfield.visit(new JFieldSwitchAdapter<Property<?>>() {
285            @Override
286            @SuppressWarnings({ "unchecked", "rawtypes" })
287            public Property<?> caseJSimpleField(JSimpleField jfield) {
288                return new MethodProperty(jfield.getTypeToken().getRawType(), jobj, jfield.getGetter(), jfield.getSetter());
289            }
290            @Override
291            public Property<?> caseJCounterField(JCounterField jfield) {
292                return new CounterProperty(jfield.getValue(jobj));
293            }
294            @Override
295            @SuppressWarnings("rawtypes")
296            public Property<?> caseJSetField(JSetField jfield) {
297                return new CollectionProperty(jfield.getValue(jobj));
298            }
299            @Override
300            @SuppressWarnings("rawtypes")
301            public Property<?> caseJListField(JListField jfield) {
302                return new CollectionProperty(jfield.getValue(jobj));
303            }
304            @Override
305            public Property<?> caseJMapField(JMapField jfield) {
306                return new ObjectProperty<Void>(null, Void.class);                          // TODO
307            }
308        });
309    }
