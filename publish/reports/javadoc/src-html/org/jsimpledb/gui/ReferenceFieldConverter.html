006package org.jsimpledb.gui;
007
008import com.vaadin.data.util.converter.Converter;
009
010import java.util.Locale;
011
012import org.jsimpledb.JObject;
013import org.jsimpledb.JSimpleDB;
014import org.jsimpledb.core.ObjId;
015
016/**
017 * Vaadin {@link Converter} for that converts between {@link ObjId}'s and {@link JObject}s.
018 */
019@SuppressWarnings("serial")
020public class ReferenceFieldConverter implements Converter<ObjId, JObject> {
021
022    private final JSimpleDB jdb;
023
024    public ReferenceFieldConverter(JSimpleDB jdb) {
025        if (jdb == null)
026            throw new IllegalArgumentException("null jdb");
027        this.jdb = jdb;
028    }
029
030    @Override
031    public Class<ObjId> getPresentationType() {
032        return ObjId.class;
033    }
034
035    @Override
036    public Class<JObject> getModelType() {
037        return JObject.class;
038    }
039
040    @Override
041    public ObjId convertToPresentation(JObject jobj, Class<? extends ObjId> targetType, Locale locale) {
042        if (jobj == null)
043            return null;
044        return jobj.getObjId();
045    }
046
047    @Override
048    public JObject convertToModel(ObjId id, Class<? extends JObject> targetType, Locale locale) {
049        if (id == null)
050            return null;
051        return this.jdb.getJObject(id);
052    }
053}
