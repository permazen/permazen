006package org.jsimpledb;
007
008import java.lang.reflect.Method;
009
010import org.jsimpledb.schema.CounterSchemaField;
011import org.objectweb.asm.ClassWriter;
012
013/**
014 * Represents a counter field in a {@link JClass}.
015 *
016 * @see org.jsimpledb.Counter
017 */
018public class JCounterField extends JField {
019
020    JCounterField(JSimpleDB jdb, String name, int storageId, String description, Method getter) {
021        super(jdb, name, storageId, description, getter);
022    }
023
024    @Override
025    public Counter getValue(JObject jobj) {
026        if (jobj == null)
027            throw new IllegalArgumentException("null jobj");
028        return jobj.getTransaction().readCounterField(jobj, this.storageId, false);
029    }
030
031    @Override
032    public <R> R visit(JFieldSwitch<R> target) {
033        return target.caseJCounterField(this);
034    }
035
036    @Override
037    CounterSchemaField toSchemaItem(JSimpleDB jdb) {
038        final CounterSchemaField schemaField = new CounterSchemaField();
039        this.initialize(jdb, schemaField);
040        return schemaField;
041    }
042
043    @Override
044    void outputMethods(final ClassGenerator<?> generator, ClassWriter cw) {
045        this.outputReadMethod(generator, cw, ClassGenerator.READ_COUNTER_FIELD_METHOD);
046    }
047
048    @Override
049    JCounterFieldInfo toJFieldInfo() {
050        return new JCounterFieldInfo(this);
051    }
052}
