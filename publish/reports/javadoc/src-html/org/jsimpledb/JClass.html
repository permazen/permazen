112    /**
113     * Get the {@link JField} in this instance associated with the specified storage ID, cast to the given type.
114     *
115     * @param storageId field storage ID
116     * @param type required type
117     * @param <T> expected field type
118     * @return {@link JField} in this instance corresponding to {@code storageId}
119     * @throws UnknownFieldException if {@code storageId} does not correspond to any field in this instance
120     * @throws UnknownFieldException if the field is not an instance of of {@code type}
121     */
122    public <T extends JField> T getJField(int storageId, Class<T> type) {
123        if (type == null)
124            throw new IllegalArgumentException("null type");
125        final JField jfield = this.jfields.get(storageId);
126        if (jfield == null)
127            throw new UnknownFieldException(storageId, "object type `" + this.name + "' has no field with storage ID " + storageId);
128        try {
129            return type.cast(jfield);
130        } catch (ClassCastException e) {
131            throw new UnknownFieldException(storageId, "object type `" + this.name + "' has no field with storage ID "
132              + storageId + " of type " + type.getName() + " (found " + jfield + " instead)");
133        }
134    }

138    void createFields(JSimpleDB jdb) {
139
140        // Auto-generate properties?
141        final boolean autogenFields = this.type.getAnnotation(JSimpleClass.class).autogenFields();
142
143        // Scan for Simple fields
144        final JFieldScanner<T> simpleFieldScanner = new JFieldScanner<T>(this, autogenFields);
145        for (JFieldScanner<T>.MethodInfo info : simpleFieldScanner.findAnnotatedMethods()) {
146
147            // Get info
148            final org.jsimpledb.annotation.JField annotation = info.getAnnotation();
149            final Method getter = info.getMethod();
150            final String description = simpleFieldScanner.getAnnotationDescription() + " annotation on method " + getter;
151            final String fieldName = this.getFieldName(annotation.name(), info, description);
152            final TypeToken<?> fieldTypeToken = TypeToken.of(this.type).resolveType(getter.getGenericReturnType());
153            if (this.log.isTraceEnabled())
154                this.log.trace("found " + description);
155
156            // Get storage ID
157            int storageId = annotation.storageId();
158            if (storageId == 0)
159                storageId = jdb.getStorageIdGenerator(annotation, getter).generateFieldStorageId(getter, fieldName);
160
161            // Handle Counter fields
162            if (fieldTypeToken.equals(TypeToken.of(Counter.class))) {
163
164                // Sanity check annotation
165                if (annotation.type().length() != 0)
166                    throw new IllegalArgumentException("invalid " + description + ": counter fields must not specify a type");
167                if (annotation.indexed())
168                    throw new IllegalArgumentException("invalid " + description + ": counter fields cannot be indexed");
169
170                // Create counter field
171                final JCounterField jfield = new JCounterField(this.jdb, fieldName, storageId,
172                  "counter field `" + fieldName + "' of object type `" + this.name + "'", getter);
173                jfield.parent = this;
174
175                // Add field
176                this.addField(jfield);
177                continue;
178            }
179
180            // Find corresponding setter method
181            final Method setter;
182            try {
183                setter = Util.findSetterMethod(this.type, getter);
184            } catch (IllegalArgumentException e) {
185                throw new IllegalArgumentException("invalid " + description + ": " + e.getMessage());
186            }
187
188            // Create simple field
189            final JSimpleField jfield = this.createSimpleField(description, fieldTypeToken,
190              fieldName, storageId, annotation, getter, setter, "field `" + fieldName + "' of object type `" + this.name + "'");
191            jfield.parent = this;
192
193            // Add field
194            this.addField(jfield);
195        }
196
197        // Scan for Set fields
198        final JSetFieldScanner<T> setFieldScanner = new JSetFieldScanner<T>(this, autogenFields);
199        for (JSetFieldScanner<T>.MethodInfo info : setFieldScanner.findAnnotatedMethods()) {
200
201            // Get info
202            final org.jsimpledb.annotation.JSetField annotation = info.getAnnotation();
203            final org.jsimpledb.annotation.JField elementAnnotation = annotation.element();
204            final Method getter = info.getMethod();
205            final String description = setFieldScanner.getAnnotationDescription() + " annotation on method " + getter;
206            final String fieldName = this.getFieldName(annotation.name(), info, description);
207            if (this.log.isTraceEnabled())
208                this.log.trace("found " + description);
209
210            // Get storage ID's
211            int storageId = annotation.storageId();
212            if (storageId == 0)
213                storageId = jdb.getStorageIdGenerator(annotation, getter).generateFieldStorageId(getter, fieldName);
214            int elementStorageId = elementAnnotation.storageId();
215            if (elementStorageId == 0) {
216                elementStorageId = jdb.getStorageIdGenerator(elementAnnotation, getter)
217                  .generateSetElementStorageId(getter, fieldName);
218            }
219
220            // Get element type (the raw return type has already been validated by the annotation scanner)
221            final TypeToken<?> elementType = TypeToken.of(this.type).resolveType(this.getParameterType(description, getter, 0));
222
223            // Create element sub-field
224            final JSimpleField elementField = this.createSimpleField("element() property of " + description, elementType,
225              SetField.ELEMENT_FIELD_NAME, elementStorageId, elementAnnotation, null, null,
226              "element field of set field `" + fieldName + "' in object type `" + this.name + "'");
227
228            // Create set field
229            final JSetField jfield = new JSetField(this.jdb, fieldName, storageId, elementField,
230              "set field `" + fieldName + "' in object type `" + this.name + "'", getter);
231            elementField.parent = jfield;
232
233            // Add field
234            this.addField(jfield);
235        }
236
237        // Scan for List fields
238        final JListFieldScanner<T> listFieldScanner = new JListFieldScanner<T>(this, autogenFields);
239        for (JListFieldScanner<T>.MethodInfo info : listFieldScanner.findAnnotatedMethods()) {
240
241            // Get info
242            final org.jsimpledb.annotation.JListField annotation = info.getAnnotation();
243            final org.jsimpledb.annotation.JField elementAnnotation = annotation.element();
244            final Method getter = info.getMethod();
245            final String description = listFieldScanner.getAnnotationDescription() + " annotation on method " + getter;
246            final String fieldName = this.getFieldName(annotation.name(), info, description);
247            if (this.log.isTraceEnabled())
248                this.log.trace("found " + description);
249
250            // Get storage ID's
251            int storageId = annotation.storageId();
252            if (storageId == 0)
253                storageId = jdb.getStorageIdGenerator(annotation, getter).generateFieldStorageId(getter, fieldName);
254            int elementStorageId = elementAnnotation.storageId();
255            if (elementStorageId == 0) {
256                elementStorageId = jdb.getStorageIdGenerator(elementAnnotation, getter)
257                  .generateListElementStorageId(getter, fieldName);
258            }
259
260            // Get element type (the raw return type has already been validated by the annotation scanner)
261            final TypeToken<?> elementType = TypeToken.of(this.type).resolveType(this.getParameterType(description, getter, 0));
262
263            // Create element sub-field
264            final JSimpleField elementField = this.createSimpleField("element() property of " + description, elementType,
265              ListField.ELEMENT_FIELD_NAME, elementStorageId, elementAnnotation, null, null,
266              "element field of list field `" + fieldName + "' in object type `" + this.name + "'");
267
268            // Create list field
269            final JListField jfield = new JListField(this.jdb, fieldName, storageId, elementField,
270              "list field `" + fieldName + "' in object type `" + this.name + "'", getter);
271            elementField.parent = jfield;
272
273            // Add field
274            this.addField(jfield);
275        }
276
277        // Scan for Map fields
278        final JMapFieldScanner<T> mapFieldScanner = new JMapFieldScanner<T>(this, autogenFields);
279        for (JMapFieldScanner<T>.MethodInfo info : mapFieldScanner.findAnnotatedMethods()) {
280
281            // Get info
282            final org.jsimpledb.annotation.JMapField annotation = info.getAnnotation();
283            final org.jsimpledb.annotation.JField keyAnnotation = annotation.key();
284            final org.jsimpledb.annotation.JField valueAnnotation = annotation.value();
285            final Method getter = info.getMethod();
286            final String description = mapFieldScanner.getAnnotationDescription() + " annotation on method " + getter;
287            final String fieldName = this.getFieldName(annotation.name(), info, description);
288            if (this.log.isTraceEnabled())
289                this.log.trace("found " + description);
290
291            // Get storage ID's
292            int storageId = annotation.storageId();
293            if (storageId == 0)
294                storageId = jdb.getStorageIdGenerator(annotation, getter).generateFieldStorageId(getter, fieldName);
295            int keyStorageId = keyAnnotation.storageId();
296            if (keyStorageId == 0)
297                keyStorageId = jdb.getStorageIdGenerator(keyAnnotation, getter).generateMapKeyStorageId(getter, fieldName);
298            int valueStorageId = valueAnnotation.storageId();
299            if (valueStorageId == 0)
300                valueStorageId = jdb.getStorageIdGenerator(valueAnnotation, getter).generateMapValueStorageId(getter, fieldName);
301
302            // Get key and value types (the raw return type has already been validated by the annotation scanner)
303            final TypeToken<?> keyType = TypeToken.of(this.type).resolveType(this.getParameterType(description, getter, 0));
304            final TypeToken<?> valueType = TypeToken.of(this.type).resolveType(this.getParameterType(description, getter, 1));
305
306            // Create key and value sub-fields
307            final JSimpleField keyField = this.createSimpleField("key() property of " + description, keyType,
308              MapField.KEY_FIELD_NAME, keyStorageId, keyAnnotation, null, null,
309              "key field of map field `" + fieldName + "' in object type `" + this.name + "'");
310            final JSimpleField valueField = this.createSimpleField("value() property of " + description, valueType,
311              MapField.VALUE_FIELD_NAME, valueStorageId, valueAnnotation, null, null,
312              "value field of map field `" + fieldName + "' in object type `" + this.name + "'");
313
314            // Create map field
315            final JMapField jfield = new JMapField(this.jdb, fieldName, storageId, keyField, valueField,
316              "map field `" + fieldName + "' in object type `" + this.name + "'", getter);
317            keyField.parent = jfield;
318            valueField.parent = jfield;
319
320            // Add field
321            this.addField(jfield);
322        }
323
324        // Calculate which fields require validation
325        for (JField jfield : this.jfields.values())
326            jfield.calculateRequiresValidation();
327    }

329    void addCompositeIndex(JSimpleDB jdb, org.jsimpledb.annotation.JCompositeIndex annotation) {
330
331        // Get info
332        final String indexName = annotation.name();
333
334        // Resolve field names
335        final String[] fieldNames = annotation.fields();
336        final JSimpleField[] indexFields = new JSimpleField[fieldNames.length];
337        final int[] indexFieldStorageIds = new int[fieldNames.length];
338        final HashSet<String> seenFieldNames = new HashSet<String>();
339        for (int i = 0; i < fieldNames.length; i++) {
340            final String fieldName = fieldNames[i];
341            if (!seenFieldNames.add(fieldName))
342                throw this.invalidIndex(annotation, "field `" + fieldName + "' appears more than once");
343            final JField jfield = this.jfieldsByName.get(fieldName);
344            if (!(jfield instanceof JSimpleField)) {
345                throw this.invalidIndex(annotation, "field `" + fieldName + "' "
346                  + (jfield != null ? "is not a simple field" : "not found"));
347            }
348            indexFields[i] = (JSimpleField)jfield;
349            indexFieldStorageIds[i] = jfield.storageId;
350        }
351
352        // Get storage ID
353        int storageId = annotation.storageId();
354        if (storageId == 0) {
355            storageId = jdb.getStorageIdGenerator(annotation, type)
356              .generateCompositeIndexStorageId(this.type, indexName, indexFieldStorageIds);
357        }
358
359        // Create and add index
360        final JCompositeIndex index = new JCompositeIndex(this.jdb, indexName, storageId, indexFields);
361        if (this.jcompositeIndexes.put(index.storageId, index) != null)
362            throw this.invalidIndex(annotation, "duplicate use of storage ID " + index.storageId);
363        if (this.jcompositeIndexesByName.put(index.name, index) != null)
364            throw this.invalidIndex(annotation, "duplicate use of composite index name `" + index.name + "'");
365    }

367    void scanAnnotations() {
368        this.onCreateMethods = new OnCreateScanner<T>(this).findAnnotatedMethods();
369        this.onDeleteMethods = new OnDeleteScanner<T>(this).findAnnotatedMethods();
370        this.onChangeMethods = new OnChangeScanner<T>(this).findAnnotatedMethods();
371        this.validateMethods = new ValidateScanner<T>(this).findAnnotatedMethods();
372        final OnVersionChangeScanner<T> onVersionChangeScanner = new OnVersionChangeScanner<T>(this);
373        this.onVersionChangeMethods = new ArrayList<>(onVersionChangeScanner.findAnnotatedMethods());
374        Collections.sort(this.onVersionChangeMethods, onVersionChangeScanner);
375    }

377    void calculateValidationRequirement() {
378
379        // Check for JSR 303 or @Validate annotations
380        if ((this.requiresValidation = Util.requiresValidation(this.type)))
381            return;
382
383        // Check for any uniqueness constraints
384        for (JSimpleField jfield : Iterables.filter(this.jfields.values(), JSimpleField.class)) {
385            if (jfield.unique) {
386                this.requiresValidation = true;
387                return;
388            }
389        }
390    }

392    @Override
393    SchemaObjectType toSchemaItem(JSimpleDB jdb) {
394        final SchemaObjectType schemaObjectType = new SchemaObjectType();
395        this.initialize(jdb, schemaObjectType);
396        for (JField field : this.jfields.values()) {
397            final SchemaField schemaField = field.toSchemaItem(jdb);
398            schemaObjectType.getSchemaFields().put(schemaField.getStorageId(), schemaField);
399        }
400        for (JCompositeIndex index : this.jcompositeIndexes.values()) {
401            final SchemaCompositeIndex schemaIndex = index.toSchemaItem(jdb);
402            schemaObjectType.getSchemaCompositeIndexes().put(index.getStorageId(), schemaIndex);
403        }
404        return schemaObjectType;
405    }

407    private IllegalArgumentException invalidIndex(org.jsimpledb.annotation.JCompositeIndex annotation, String message) {
408        return new IllegalArgumentException("invalid @JCompositeIndex annotation for index `"
409          + annotation.name() + "' on " + this.type + ": " + message);

412    // Add new JField (and sub-fields, if any), checking for name and storage ID conflicts
413    private void addField(JField jfield) {
414
415        // Check for storage ID conflict
416        JField other = this.jfields.get(jfield.storageId);
417        if (other != null) {
418            throw new IllegalArgumentException("illegal duplicate use of storage ID "
419              + jfield.storageId + " for both " + other + " and " + jfield);
420        }
421        this.jfields.put(jfield.storageId, jfield);
422
423        // Check for name conflict
424        if ((other = this.jfieldsByName.get(jfield.name)) != null)
425            throw new IllegalArgumentException("illegal duplicate use of field name `" + jfield.name + "' in " + this);
426        this.jfieldsByName.put(jfield.name, jfield);
427
428        // Logging
429        if (this.log.isTraceEnabled())
430            this.log.trace("added " + jfield + " to object type `" + this.name + "'");
431    }

433    // Get field name, deriving it from the getter property name if necessary
434    private String getFieldName(String fieldName, AnnotationScanner<T, ?>.MethodInfo info, String description) {
435        if (fieldName.length() > 0)
436            return fieldName;
437        try {
438            return info.getMethodPropertyName();
439        } catch (IllegalArgumentException e) {
440            throw new IllegalArgumentException("invalid " + description + ": can't infer field name: " + e, e);
441        }
442    }

453    // Create a simple field, either regular object field or sub-field of complex field
454    @SuppressWarnings("unchecked")
455    private JSimpleField createSimpleField(String description, TypeToken<?> fieldTypeToken, String fieldName,
456      int storageId, org.jsimpledb.annotation.JField annotation, Method getter, Method setter, String fieldDescription) {
457
458        // Get explicit type name, if any
459        final String typeName = annotation.type().length() > 0 ? annotation.type() : null;
460
461        // Include containing type for annotation description; with autogenProperties it can be more than one
462        description += " in " + this.type;
463
464        // Complex sub-field?
465        final boolean isSubField = getter == null;
466
467        // Sanity check annotation
468        if (isSubField && annotation.unique())
469            throw new IllegalArgumentException("invalid " + description + ": unique() constraint not allowed on complex sub-field");
470        if (annotation.uniqueExclude().length > 0 && !annotation.unique())
471            throw new IllegalArgumentException("invalid " + description + ": use of uniqueExclude() requires unique = true");
472        if (annotation.uniqueExcludeNull() && !annotation.unique())
473            throw new IllegalArgumentException("invalid " + description + ": use of uniqueExcludeNull() requires unique = true");
474
475        // See if field type encompasses one or more JClass types and is therefore a reference type
476        boolean isReferenceType = false;
477        for (JClass<?> jclass : this.jdb.jclasses.values()) {
478            if (fieldTypeToken.getRawType().isAssignableFrom(jclass.type)) {
479                isReferenceType = true;
480                break;
481            }
482        }
483
484        // See if field type is a simple type, known either by explicitly-given name or type
485        FieldType<?> nonReferenceType = null;
486        if (typeName != null) {
487
488            // Field type is explicitly specified by name
489            if ((nonReferenceType = this.jdb.db.getFieldTypeRegistry().getFieldType(typeName)) == null)
490                throw new IllegalArgumentException("invalid " + description + ": unknown simple field type `" + typeName + "'");
491
492            // Verify field type matches what we expect
493            final TypeToken<?> expectedType = isSubField ? nonReferenceType.getTypeToken().wrap() : nonReferenceType.getTypeToken();
494            if (!expectedType.equals(fieldTypeToken)) {
495                throw new IllegalArgumentException("invalid " + description + ": field type `" + typeName
496                  + "' supports values of type " + nonReferenceType.getTypeToken() + " but " + fieldTypeToken
497                  + " is required (according to the getter method's return type)");
498            }
499        } else {
500
501            // Try to find a field type supporting getter method return type
502            final List<? extends FieldType<?>> fieldTypes = this.jdb.db.getFieldTypeRegistry().getFieldTypes(fieldTypeToken);
503            switch (fieldTypes.size()) {
504            case 0:
505                nonReferenceType = null;
506                break;
507            case 1:
508                nonReferenceType = fieldTypes.get(0);
509                break;
510            default:
511                if (!isReferenceType) {
512                    throw new IllegalArgumentException("invalid " + description + ": an explicit type() must be specified"
513                      + " because type " + fieldTypeToken + " is supported by multiple registered simple field types: "
514                      + fieldTypes);
515                }
516                break;
517            }
518        }
519
520        // Detect enum types
521        final Class<? extends Enum<?>> enumType = Enum.class.isAssignableFrom(fieldTypeToken.getRawType()) ?
522          (Class<? extends Enum<?>>)fieldTypeToken.getRawType().asSubclass(Enum.class) : null;
523
524        // If field type neither refers to a JClass type, nor is a registered field type, nor is an enum type, fail
525        if (!isReferenceType && nonReferenceType == null && enumType == null) {
526            throw new IllegalArgumentException("invalid " + description + ": an explicit type() must be specified"
527              + " because no known type supports values of type " + fieldTypeToken);
528        }
529
530        // Handle ambiguity between reference vs. non-reference
531        if (isReferenceType && nonReferenceType != null) {
532
533            // If an explicit type name was provided, assume they want the specified non-reference type
534            if (typeName != null)
535                isReferenceType = false;
536            else {
537                throw new IllegalArgumentException("invalid " + description + ": an explicit type() must be specified"
538                  + " because type " + fieldTypeToken + " is ambiguous, being both a @" + JSimpleClass.class.getSimpleName()
539                  + " reference type and a simple Java type supported by type `" + nonReferenceType + "'");
540            }
541        }
542
543        // Sanity check annotation some more
544        if (!isReferenceType && annotation.onDelete() != DeleteAction.EXCEPTION)
545            throw new IllegalArgumentException("invalid " + description + ": onDelete() only allowed on reference fields");
546        if (!isReferenceType && annotation.cascadeDelete())
547            throw new IllegalArgumentException("invalid " + description + ": cascadeDelete() only allowed on reference fields");
548        if (!isReferenceType && annotation.unique() && !annotation.indexed())
549            throw new IllegalArgumentException("invalid " + description + ": unique() constraint requires field to be indexed");
550        if (nonReferenceType != null && nonReferenceType.getTypeToken().isPrimitive() && annotation.uniqueExcludeNull()) {
551            throw new IllegalArgumentException("invalid " + description + ": uniqueExcludeNull() is incompatible with fields"
552              + " having primitive type");
553        }
554
555        // Create simple, enum, or reference field
556        try {
557            return
558              isReferenceType ?
559                new JReferenceField(this.jdb, fieldName, storageId, fieldDescription, fieldTypeToken, annotation, getter, setter) :
560              enumType != null ?
561                new JEnumField(this.jdb, fieldName, storageId, enumType, annotation, fieldDescription, getter, setter) :
562                new JSimpleField(this.jdb, fieldName, storageId, fieldTypeToken,
563                  nonReferenceType, annotation.indexed(), annotation, fieldDescription, getter, setter);
564        } catch (IllegalArgumentException e) {
565            throw new IllegalArgumentException("invalid " + description + ": " + e.getMessage(), e);
566        }
567    }
