111    /**
112     * Get the {@link JField} in this instance associated with the specified storage ID, cast to the given type.
113     *
114     * @param storageId field storage ID
115     * @param type required type
116     * @return {@link JField} in this instance corresponding to {@code storageId}
117     * @throws UnknownFieldException if {@code storageId} does not correspond to any field in this instance
118     * @throws UnknownFieldException if the field is not an instance of of {@code type}
119     */
120    public <T extends JField> T getJField(int storageId, Class<T> type) {
121        if (type == null)
122            throw new IllegalArgumentException("null type");
123        final JField jfield = this.jfields.get(storageId);
124        if (jfield == null)
125            throw new UnknownFieldException(storageId, "object type `" + this.name + "' has no field with storage ID " + storageId);
126        try {
127            return type.cast(jfield);
128        } catch (ClassCastException e) {
129            throw new UnknownFieldException(storageId, "object type `" + this.name + "' has no field with storage ID "
130              + storageId + " of type " + type.getName() + " (found " + jfield + " instead)");
131        }
132    }

136    void createFields(JSimpleDB jdb) {
137
138        // Auto-generate properties?
139        final boolean autogenFields = this.type.getAnnotation(JSimpleClass.class).autogenFields();
140
141        // Scan for Simple fields
142        final JFieldScanner<T> simpleFieldScanner = new JFieldScanner<T>(this, autogenFields);
143        for (JFieldScanner<T>.MethodInfo info : simpleFieldScanner.findAnnotatedMethods()) {
144
145            // Get info
146            final org.jsimpledb.annotation.JField annotation = info.getAnnotation();
147            final Method getter = info.getMethod();
148            final String description = simpleFieldScanner.getAnnotationDescription() + " annotation on method " + getter;
149            final String fieldName = this.getFieldName(annotation.name(), info, description);
150            final TypeToken<?> fieldTypeToken = TypeToken.of(this.type).resolveType(getter.getGenericReturnType());
151            if (this.log.isTraceEnabled())
152                this.log.trace("found " + description);
153
154            // Get storage ID
155            int storageId = annotation.storageId();
156            if (storageId == 0)
157                storageId = jdb.getStorageIdGenerator(annotation, getter).generateFieldStorageId(getter, fieldName);
158
159            // Handle Counter fields
160            if (fieldTypeToken.equals(TypeToken.of(Counter.class))) {
161
162                // Sanity check annotation
163                if (annotation.type().length() != 0)
164                    throw new IllegalArgumentException("invalid " + description + ": counter fields must not specify a type");
165                if (annotation.indexed())
166                    throw new IllegalArgumentException("invalid " + description + ": counter fields cannot be indexed");
167
168                // Create counter field
169                final JCounterField jfield = new JCounterField(this.jdb, fieldName, storageId,
170                  "counter field `" + fieldName + "' of object type `" + this.name + "'", getter);
171                jfield.parent = this;
172
173                // Add field
174                this.addField(jfield);
175                continue;
176            }
177
178            // Find corresponding setter method
179            final Method setter;
180            try {
181                setter = Util.findSetterMethod(this.type, getter);
182            } catch (IllegalArgumentException e) {
183                throw new IllegalArgumentException("invalid " + description + ": " + e.getMessage());
184            }
185
186            // Create simple field
187            final JSimpleField jfield = this.createSimpleField(description, fieldTypeToken,
188              fieldName, annotation.type(), storageId, annotation.indexed(), annotation.onDelete(), annotation.cascadeDelete(),
189              getter, setter, "field `" + fieldName + "' of object type `" + this.name + "'");
190            jfield.parent = this;
191
192            // Add field
193            this.addField(jfield);
194        }
195
196        // Scan for Set fields
197        final JSetFieldScanner<T> setFieldScanner = new JSetFieldScanner<T>(this, autogenFields);
198        for (JSetFieldScanner<T>.MethodInfo info : setFieldScanner.findAnnotatedMethods()) {
199
200            // Get info
201            final org.jsimpledb.annotation.JSetField annotation = info.getAnnotation();
202            final org.jsimpledb.annotation.JField elementAnnotation = annotation.element();
203            final Method getter = info.getMethod();
204            final String description = setFieldScanner.getAnnotationDescription() + " annotation on method " + getter;
205            final String fieldName = this.getFieldName(annotation.name(), info, description);
206            if (this.log.isTraceEnabled())
207                this.log.trace("found " + description);
208
209            // Get storage ID's
210            int storageId = annotation.storageId();
211            if (storageId == 0)
212                storageId = jdb.getStorageIdGenerator(annotation, getter).generateFieldStorageId(getter, fieldName);
213            int elementStorageId = elementAnnotation.storageId();
214            if (elementStorageId == 0) {
215                elementStorageId = jdb.getStorageIdGenerator(elementAnnotation, getter)
216                  .generateSetElementStorageId(getter, fieldName);
217            }
218
219            // Get element type (the raw return type has already been validated by the annotation scanner)
220            final TypeToken<?> elementType = TypeToken.of(this.type).resolveType(this.getParameterType(description, getter, 0));
221
222            // Create element sub-field
223            final JSimpleField elementField = this.createSimpleField("element() property of " + description,
224              elementType, SetField.ELEMENT_FIELD_NAME, elementAnnotation.type(), elementStorageId,
225              elementAnnotation.indexed(), elementAnnotation.onDelete(), elementAnnotation.cascadeDelete(), null, null,
226              "element field of set field `" + fieldName + "' in object type `" + this.name + "'");
227
228            // Create set field
229            final JSetField jfield = new JSetField(this.jdb, fieldName, storageId, elementField,
230              "set field `" + fieldName + "' in object type `" + this.name + "'", getter);
231            elementField.parent = jfield;
232
233            // Add field
234            this.addField(jfield);
235        }
236
237        // Scan for List fields
238        final JListFieldScanner<T> listFieldScanner = new JListFieldScanner<T>(this, autogenFields);
239        for (JListFieldScanner<T>.MethodInfo info : listFieldScanner.findAnnotatedMethods()) {
240
241            // Get info
242            final org.jsimpledb.annotation.JListField annotation = info.getAnnotation();
243            final org.jsimpledb.annotation.JField elementAnnotation = annotation.element();
244            final Method getter = info.getMethod();
245            final String description = listFieldScanner.getAnnotationDescription() + " annotation on method " + getter;
246            final String fieldName = this.getFieldName(annotation.name(), info, description);
247            if (this.log.isTraceEnabled())
248                this.log.trace("found " + description);
249
250            // Get storage ID's
251            int storageId = annotation.storageId();
252            if (storageId == 0)
253                storageId = jdb.getStorageIdGenerator(annotation, getter).generateFieldStorageId(getter, fieldName);
254            int elementStorageId = elementAnnotation.storageId();
255            if (elementStorageId == 0) {
256                elementStorageId = jdb.getStorageIdGenerator(elementAnnotation, getter)
257                  .generateListElementStorageId(getter, fieldName);
258            }
259
260            // Get element type (the raw return type has already been validated by the annotation scanner)
261            final TypeToken<?> elementType = TypeToken.of(this.type).resolveType(this.getParameterType(description, getter, 0));
262
263            // Create element sub-field
264            final JSimpleField elementField = this.createSimpleField("element() property of " + description,
265              elementType, ListField.ELEMENT_FIELD_NAME, elementAnnotation.type(), elementStorageId,
266              elementAnnotation.indexed(), elementAnnotation.onDelete(), elementAnnotation.cascadeDelete(), null, null,
267              "element field of list field `" + fieldName + "' in object type `" + this.name + "'");
268
269            // Create list field
270            final JListField jfield = new JListField(this.jdb, fieldName, storageId, elementField,
271              "list field `" + fieldName + "' in object type `" + this.name + "'", getter);
272            elementField.parent = jfield;
273
274            // Add field
275            this.addField(jfield);
276        }
277
278        // Scan for Map fields
279        final JMapFieldScanner<T> mapFieldScanner = new JMapFieldScanner<T>(this, autogenFields);
280        for (JMapFieldScanner<T>.MethodInfo info : mapFieldScanner.findAnnotatedMethods()) {
281
282            // Get info
283            final org.jsimpledb.annotation.JMapField annotation = info.getAnnotation();
284            final org.jsimpledb.annotation.JField keyAnnotation = annotation.key();
285            final org.jsimpledb.annotation.JField valueAnnotation = annotation.value();
286            final Method getter = info.getMethod();
287            final String description = mapFieldScanner.getAnnotationDescription() + " annotation on method " + getter;
288            final String fieldName = this.getFieldName(annotation.name(), info, description);
289            if (this.log.isTraceEnabled())
290                this.log.trace("found " + description);
291
292            // Get storage ID's
293            int storageId = annotation.storageId();
294            if (storageId == 0)
295                storageId = jdb.getStorageIdGenerator(annotation, getter).generateFieldStorageId(getter, fieldName);
296            int keyStorageId = keyAnnotation.storageId();
297            if (keyStorageId == 0)
298                keyStorageId = jdb.getStorageIdGenerator(keyAnnotation, getter).generateMapKeyStorageId(getter, fieldName);
299            int valueStorageId = valueAnnotation.storageId();
300            if (valueStorageId == 0)
301                valueStorageId = jdb.getStorageIdGenerator(valueAnnotation, getter).generateMapValueStorageId(getter, fieldName);
302
303            // Get key and value types (the raw return type has already been validated by the annotation scanner)
304            final TypeToken<?> keyType = TypeToken.of(this.type).resolveType(this.getParameterType(description, getter, 0));
305            final TypeToken<?> valueType = TypeToken.of(this.type).resolveType(this.getParameterType(description, getter, 1));
306
307            // Create key and value sub-fields
308            final JSimpleField keyField = this.createSimpleField("key() property of " + description,
309              keyType, MapField.KEY_FIELD_NAME, keyAnnotation.type(), keyStorageId, keyAnnotation.indexed(),
310              keyAnnotation.onDelete(), keyAnnotation.cascadeDelete(), null, null,
311              "key field of map field `" + fieldName + "' in object type `" + this.name + "'");
312            final JSimpleField valueField = this.createSimpleField("value() property of " + description,
313              valueType, MapField.VALUE_FIELD_NAME, valueAnnotation.type(), valueStorageId, valueAnnotation.indexed(),
314              valueAnnotation.onDelete(), valueAnnotation.cascadeDelete(), null, null,
315              "value field of map field `" + fieldName + "' in object type `" + this.name + "'");
316
317            // Create map field
318            final JMapField jfield = new JMapField(this.jdb, fieldName, storageId, keyField, valueField,
319              "map field `" + fieldName + "' in object type `" + this.name + "'", getter);
320            keyField.parent = jfield;
321            valueField.parent = jfield;
322
323            // Add field
324            this.addField(jfield);
325        }
326    }

328    void addCompositeIndex(JSimpleDB jdb, org.jsimpledb.annotation.JCompositeIndex annotation) {
329
330        // Get info
331        final String indexName = annotation.name();
332
333        // Resolve field names
334        final String[] fieldNames = annotation.fields();
335        final JSimpleField[] indexFields = new JSimpleField[fieldNames.length];
336        final int[] indexFieldStorageIds = new int[fieldNames.length];
337        final HashSet<String> seenFieldNames = new HashSet<String>();
338        for (int i = 0; i < fieldNames.length; i++) {
339            final String fieldName = fieldNames[i];
340            if (!seenFieldNames.add(fieldName))
341                throw this.invalidIndex(annotation, "field `" + fieldName + "' appears more than once");
342            final JField jfield = this.jfieldsByName.get(fieldName);
343            if (!(jfield instanceof JSimpleField)) {
344                throw this.invalidIndex(annotation, "field `" + fieldName + "' "
345                  + (jfield != null ? "is not a simple field" : "not found"));
346            }
347            indexFields[i] = (JSimpleField)jfield;
348            indexFieldStorageIds[i] = jfield.storageId;
349        }
350
351        // Get storage ID
352        int storageId = annotation.storageId();
353        if (storageId == 0) {
354            storageId = jdb.getStorageIdGenerator(annotation, type)
355              .generateCompositeIndexStorageId(this.type, indexName, indexFieldStorageIds);
356        }
357
358        // Create and add index
359        final JCompositeIndex index = new JCompositeIndex(this.jdb, indexName, storageId, indexFields);
360        if (this.jcompositeIndexes.put(index.storageId, index) != null)
361            throw this.invalidIndex(annotation, "duplicate use of storage ID " + index.storageId);
362        if (this.jcompositeIndexesByName.put(index.name, index) != null)
363            throw this.invalidIndex(annotation, "duplicate use of composite index name `" + index.name + "'");
364    }

366    void scanAnnotations() {
367        this.onCreateMethods = new OnCreateScanner<T>(this).findAnnotatedMethods();
368        this.onDeleteMethods = new OnDeleteScanner<T>(this).findAnnotatedMethods();
369        this.onChangeMethods = new OnChangeScanner<T>(this).findAnnotatedMethods();
370        this.validateMethods = new ValidateScanner<T>(this).findAnnotatedMethods();
371        final OnVersionChangeScanner<T> onVersionChangeScanner = new OnVersionChangeScanner<T>(this);
372        this.onVersionChangeMethods = new ArrayList<>(onVersionChangeScanner.findAnnotatedMethods());
373        Collections.sort(this.onVersionChangeMethods, onVersionChangeScanner);
374    }

380    @Override
381    SchemaObjectType toSchemaItem(JSimpleDB jdb) {
382        final SchemaObjectType schemaObjectType = new SchemaObjectType();
383        this.initialize(jdb, schemaObjectType);
384        for (JField field : this.jfields.values()) {
385            final SchemaField schemaField = field.toSchemaItem(jdb);
386            schemaObjectType.getSchemaFields().put(schemaField.getStorageId(), schemaField);
387        }
388        for (JCompositeIndex index : this.jcompositeIndexes.values()) {
389            final SchemaCompositeIndex schemaIndex = index.toSchemaItem(jdb);
390            schemaObjectType.getSchemaCompositeIndexes().put(index.getStorageId(), schemaIndex);
391        }
392        return schemaObjectType;
393    }

397          + annotation.name() + "' on " + this.type + ": " + message);

400    // Add new JField (and sub-fields, if any), checking for name and storage ID conflicts
401    private void addField(JField jfield) {
402
403        // Check for storage ID conflict
404        JField other = this.jfields.get(jfield.storageId);
405        if (other != null) {
406            throw new IllegalArgumentException("illegal duplicate use of storage ID "
407              + jfield.storageId + " for both " + other + " and " + jfield);
408        }
409        this.jfields.put(jfield.storageId, jfield);
410
411        // Check for name conflict
412        if ((other = this.jfieldsByName.get(jfield.name)) != null)
413            throw new IllegalArgumentException("illegal duplicate use of field name `" + jfield.name + "' in " + this);
414        this.jfieldsByName.put(jfield.name, jfield);
415
416        // Logging
417        if (this.log.isTraceEnabled())
418            this.log.trace("added " + jfield + " to object type `" + this.name + "'");
419    }

421    // Get field name, deriving it from the getter property name if necessary
422    private String getFieldName(String fieldName, AnnotationScanner<T, ?>.MethodInfo info, String description) {
423        if (fieldName.length() > 0)
424            return fieldName;
425        try {
426            return info.getMethodPropertyName();
427        } catch (IllegalArgumentException e) {
428            throw new IllegalArgumentException("invalid " + description + ": can't infer field name: " + e, e);
429        }
430    }

441    // Create a simple field, either regular object field or sub-field of complex field
442    @SuppressWarnings("unchecked")
443    private JSimpleField createSimpleField(String description, TypeToken<?> fieldTypeToken, String fieldName,
444      String typeName, int storageId, boolean indexed, DeleteAction onDelete, boolean cascadeDelete, Method getter, Method setter,
445      String fieldDescription) {
446
447        // Include containing type for annotation description; with autogenProperties it can be more than one
448        description += " in " + this.type;
449
450        // Complex sub-field?
451        final boolean isSubField = getter == null;
452
453        // Empty type name same as null
454        if (typeName.length() == 0)
455            typeName = null;
456
457        // See if field type encompasses one or more JClass types and is therefore a reference type
458        boolean isReferenceType = false;
459        for (JClass<?> jclass : this.jdb.jclasses.values()) {
460            if (fieldTypeToken.getRawType().isAssignableFrom(jclass.type)) {
461                isReferenceType = true;
462                break;
463            }
464        }
465
466        // See if field type is a simple type, known either by explicitly-given name or type
467        FieldType<?> nonReferenceType = null;
468        if (typeName != null) {
469
470            // Field type is explicitly specified by name
471            if ((nonReferenceType = this.jdb.db.getFieldTypeRegistry().getFieldType(typeName)) == null)
472                throw new IllegalArgumentException("invalid " + description + ": unknown simple field type `" + typeName + "'");
473
474            // Verify field type matches what we expect
475            final TypeToken<?> expectedType = isSubField ? nonReferenceType.getTypeToken().wrap() : nonReferenceType.getTypeToken();
476            if (!expectedType.equals(fieldTypeToken)) {
477                throw new IllegalArgumentException("invalid " + description + ": field type `" + typeName
478                  + "' supports values of type " + nonReferenceType.getTypeToken() + " but " + fieldTypeToken
479                  + " is required (according to the getter method's return type)");
480            }
481        } else {
482
483            // Try to find a field type supporting getter method return type
484            final List<? extends FieldType<?>> fieldTypes = this.jdb.db.getFieldTypeRegistry().getFieldTypes(fieldTypeToken);
485            switch (fieldTypes.size()) {
486            case 0:
487                nonReferenceType = null;
488                break;
489            case 1:
490                nonReferenceType = fieldTypes.get(0);
491                break;
492            default:
493                if (!isReferenceType) {
494                    throw new IllegalArgumentException("invalid " + description + ": an explicit type() must be specified"
495                      + " because type " + fieldTypeToken + " is supported by multiple registered simple field types: "
496                      + fieldTypes);
497                }
498                break;
499            }
500        }
501
502        // Detect enum types
503        final Class<? extends Enum<?>> enumType = Enum.class.isAssignableFrom(fieldTypeToken.getRawType()) ?
504          (Class<? extends Enum<?>>)fieldTypeToken.getRawType().asSubclass(Enum.class) : null;
505
506        // If field type neither refers to a JClass type, nor is a registered field type, nor is an enum type, fail
507        if (!isReferenceType && nonReferenceType == null && enumType == null) {
508            throw new IllegalArgumentException("invalid " + description + ": an explicit type() must be specified"
509              + " because no known type supports values of type " + fieldTypeToken);
510        }
511
512        // Handle ambiguity between reference vs. non-reference
513        if (isReferenceType && nonReferenceType != null) {
514
515            // If an explicit type name was provided, assume they want the specified non-reference type
516            if (typeName != null)
517                isReferenceType = false;
518            else {
519                throw new IllegalArgumentException("invalid " + description + ": an explicit type() must be specified"
520                  + " because type " + fieldTypeToken + " is ambiguous, being both a @" + JSimpleClass.class.getSimpleName()
521                  + " reference type and a simple Java type supported by type `" + nonReferenceType + "'");
522            }
523        }
524
525        // Create simple, enum, or reference field
526        try {
527            return
528              isReferenceType ?
529                new JReferenceField(this.jdb, fieldName, storageId, fieldDescription,
530                  fieldTypeToken, onDelete, cascadeDelete, getter, setter) :
531              enumType != null ?
532                new JEnumField(this.jdb, fieldName, storageId, enumType, indexed, fieldDescription, getter, setter) :
533                new JSimpleField(this.jdb, fieldName, storageId, fieldTypeToken,
534                 nonReferenceType.getName(), indexed, fieldDescription, getter, setter);
535        } catch (IllegalArgumentException e) {
536            throw new IllegalArgumentException("invalid " + description + ": " + e, e);
537        }
538    }
