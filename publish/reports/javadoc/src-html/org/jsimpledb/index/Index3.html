006package org.jsimpledb.index;
007
008import java.util.NavigableMap;
009import java.util.NavigableSet;
010
011import org.jsimpledb.tuple.Tuple2;
012import org.jsimpledb.tuple.Tuple3;
013import org.jsimpledb.tuple.Tuple4;
014
015/**
016 * An index on three fields.
017 *
018 * <p>
019 * Indexes are read-only and "live", always reflecting the current transaction state.
020 * </p>
021 *
022 * @param <V1> first indexed value type
023 * @param <V2> second indexed value type
024 * @param <V3> third indexed value type
025 * @param <T> index target type
026 * @see org.jsimpledb.index
027 */
028public interface Index3<V1, V2, V3, T> {
029
030    /**
031     * View this index as a {@link NavigableSet} of tuples.
032     *
033     * @return {@link NavigableSet} of tuples containing indexed values and target value
034     */
035    NavigableSet<Tuple4<V1, V2, V3, T>> asSet();
036
037    /**
038     * View this index as a {@link NavigableMap} of target values keyed by indexed value tuples.
039     *
040     * @return {@link NavigableMap} from indexed value tuple to the corresponding set of target objects
041     */
042    NavigableMap<Tuple3<V1, V2, V3>, NavigableSet<T>> asMap();
043
044    /**
045     * View this index as a {@link NavigableMap} of {@link Index}s keyed by the first two values.
046     *
047     * @return {@link NavigableMap} from first two values to {@link Index}
048     */
049    NavigableMap<Tuple2<V1, V2>, Index<V3, T>> asMapOfIndex();
050
051    /**
052     * View this index as a {@link NavigableMap} of {@link Index2}s keyed by the first value.
053     *
054     * @return {@link NavigableMap} from first value to {@link Index2}
055     */
056    NavigableMap<V1, Index2<V2, V3, T>> asMapOfIndex2();
057
058    /**
059     * Get the prefix of this instance that only includes the first three indexed fields.
060     *
061     * @return prefix of this index
062     */
063    Index2<V1, V2, V3> asIndex2();
064
065    /**
066     * Get the prefix of this instance that only includes the first two indexed fields.
067     *
068     * @return prefix of this index
069     */
070    Index<V1, V2> asIndex();
071}
