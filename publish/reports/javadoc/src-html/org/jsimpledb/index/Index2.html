006package org.jsimpledb.index;
007
008import java.util.NavigableMap;
009import java.util.NavigableSet;
010
011import org.jsimpledb.tuple.Tuple2;
012import org.jsimpledb.tuple.Tuple3;
013
014/**
015 * An index on two fields.
016 *
017 * <p>
018 * Indexes are read-only and "live", always reflecting the current transaction state.
019 * </p>
020 *
021 * @param <V1> first indexed value type
022 * @param <V2> second indexed value type
023 * @param <T> index target type
024 * @see org.jsimpledb.index
025 */
026public interface Index2<V1, V2, T> {
027
028    /**
029     * View this index as a {@link NavigableSet} of tuples.
030     *
031     * @return {@link NavigableSet} of tuples containing indexed values and target value
032     */
033    NavigableSet<Tuple3<V1, V2, T>> asSet();
034
035    /**
036     * View this index as a {@link NavigableMap} of target values keyed by indexed value tuples.
037     *
038     * @return {@link NavigableMap} from indexed value tuple to the corresponding set of target objects
039     */
040    NavigableMap<Tuple2<V1, V2>, NavigableSet<T>> asMap();
041
042    /**
043     * View this index as a {@link NavigableMap} of {@link Index}s keyed by the first value.
044     *
045     * @return {@link NavigableMap} from first value to {@link Index}
046     */
047    NavigableMap<V1, Index<V2, T>> asMapOfIndex();
048
049    /**
050     * Get the prefix of this instance that only includes the first two indexed fields.
051     *
052     * @return prefix of this index
053     */
054    Index<V1, V2> asIndex();
055}
