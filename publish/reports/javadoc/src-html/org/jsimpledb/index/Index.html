006package org.jsimpledb.index;
007
008import java.util.NavigableMap;
009import java.util.NavigableSet;
010
011import org.jsimpledb.tuple.Tuple2;
012
013/**
014 * An index on a single field.
015 *
016 * <p>
017 * Indexes are read-only and "live", always reflecting the current transaction state.
018 * </p>
019 *
020 * @param <V> indexed value type
021 * @param <T> index target type
022 * @see org.jsimpledb.index
023 */
024public interface Index<V, T> {
025
026    /**
027     * View this index as a {@link NavigableSet} of tuples.
028     *
029     * @return {@link NavigableSet} of tuples containing indexed value and target value
030     */
031    NavigableSet<Tuple2<V, T>> asSet();
032
033    /**
034     * View this index as a {@link NavigableMap} of target values keyed by indexed value.
035     *
036     * @return {@link NavigableMap} from indexed value to the corresponding set of target objects
037     */
038    NavigableMap<V, NavigableSet<T>> asMap();
039}
