006package org.jsimpledb.demo;
007
008import java.util.NavigableSet;
009
010import org.jsimpledb.JObject;
011import org.jsimpledb.util.NavigableSets;
012
013/**
014 * Support superclass for {@link HasSatellites} implementations.
015 */
016public abstract class AbstractHasSatellites<S extends Satellite<?>> extends AbstractBody implements HasSatellites<S> {
017
018    private final Class<S> satelliteType;
019
020    protected AbstractHasSatellites(Class<S> satelliteType) {
021        this.satelliteType = satelliteType;
022    }
023
024    @Override
025    public NavigableSet<S> getSatellites() {
026        final NavigableSet<S> satellites = this.getTransaction().queryIndex(
027          this.satelliteType, "parent", JObject.class).asMap().get(this);
028        return satellites != null ? satellites : NavigableSets.<S>empty();
029    }
030}
