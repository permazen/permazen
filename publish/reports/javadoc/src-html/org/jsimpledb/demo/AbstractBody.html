006package org.jsimpledb.demo;
007
008import com.vaadin.server.StreamResource;
009import com.vaadin.shared.ui.label.ContentMode;
010import com.vaadin.ui.Component;
011import com.vaadin.ui.Field;
012import com.vaadin.ui.Image;
013
014import java.io.ByteArrayInputStream;
015
016import org.dellroad.stuff.vaadin7.BlobField;
017import org.dellroad.stuff.vaadin7.FieldBuilder;
018import org.dellroad.stuff.vaadin7.ProvidesProperty;
019import org.jsimpledb.gui.NullableField;
020import org.jsimpledb.gui.SizedLabel;
021
022/**
023 * Support superclass for {@code Body} implementations.
024 */
025public abstract class AbstractBody implements Body {
026
027// Object
028
029    // Show the database object ID instead of system hash code
030    @Override
031    public String toString() {
032        return this.getClass().getName() + "@" + this.getObjId();
033    }
034
035// Vaadin GUI customizations
036
037    // The @ProvidesProperty annotation here indicates that this method should be invoked
038    // to build the Vaadin property named "image" instead of doing the default thing (using toString()).
039    // We are converting the byte[] value into a Vaadin PNG Image object.
040
041    @ProvidesProperty("image")
042    @SuppressWarnings("serial")
043    private Component propertyImage() {
044        final byte[] imageData = this.getImage();
045        if (imageData == null)
046            return new SizedLabel("<i>Null</i>", ContentMode.HTML);
047        return new Image(null, new StreamResource(new StreamResource.StreamSource() {
048            @Override
049            public ByteArrayInputStream getStream() {
050                return new ByteArrayInputStream(imageData);
051            }
052        }, "image.png"));
053    }
054
055    // The @FieldBuilder.ProvidesField annotation indicates that if anyone wants to edit
056    // the "image" field, they should use this method to build an appropriate Vaaddin Field instead
057    // of doing the default thing (i.e., using a TextField).
058
059    @FieldBuilder.ProvidesField("image")
060    private Field<byte[]> buildImagePropertyField() {
061        return new NullableField<byte[]>(new BlobField());
062    }
063}
