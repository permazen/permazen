006package org.jsimpledb;
007
008import org.jsimpledb.core.ObjId;
009import org.jsimpledb.core.Transaction;
010
011/**
012 * Represents a 64-bit counter value that can be adjusted concurrently by multiple transactions,
013 * typically without locking (depending on the underlying key/value store).
014 *
015 * <p>
016 * To define a field of type {@link Counter}, annotate the field's getter method as a normal field using
017 * {@link org.jsimpledb.annotation.JField &#64;JField}. No setter method should be defined.
018 * Counter fields do not support indexing or change listeners.
019 * </p>
020 *
021 * <p>
022 * Note: during schema version change notification, counter field values appear as plain {@code long} values.
023 * </p>
024 */
025public class Counter {
026
027    private final Transaction tx;
028    private final ObjId id;
029    private final int storageId;
030    private final boolean updateVersion;
031
032    Counter(Transaction tx, ObjId id, int storageId, boolean updateVersion) {
033        if (tx == null)
034            throw new IllegalArgumentException("null tx");
035        if (id == null)
036            throw new IllegalArgumentException("null id");
037        this.tx = tx;
038        this.id = id;
039        this.storageId = storageId;
040        this.updateVersion = updateVersion;
041    }
042
043    /**
044     * Read the counter's current value. Invoking this method will typically disable the lock-free
045     * behavior of {@link #adjust adjust()} in the current transaction.
046     *
047     * @return current value of the counter
048     * @throws org.jsimpledb.kv.StaleTransactionException if the transaction from which this instance
049     *  was read is no longer usable
050     * @throws org.jsimpledb.core.DeletedObjectException if the object from which this instance was read no longer exists
051     */
052    public long get() {
053        return this.tx.readCounterField(this.id, this.storageId, this.updateVersion);
054    }
055
056    /**
057     * Set the counter's value. Invoking this method will typically disable the lock-free
058     * behavior of {@link #adjust adjust()} in the current transaction.
059     *
060     * @param value new value for the counter
061     * @throws org.jsimpledb.kv.StaleTransactionException if the transaction from which this instance
062     *  was read is no longer usable
063     * @throws org.jsimpledb.core.DeletedObjectException if the object from which this instance was read no longer exists
064     */
065    public void set(long value) {
066        this.tx.writeCounterField(this.id, this.storageId, value, this.updateVersion);
067    }
068
069    /**
070     * Adjust the counter's value by the specified amount.
071     *
072     * @param offset amount to add to counter
073     * @throws org.jsimpledb.kv.StaleTransactionException if the transaction from which this instance
074     *  was read is no longer usable
075     * @throws org.jsimpledb.core.DeletedObjectException if the object from which this instance was read no longer exists
076     */
077    public void adjust(long offset) {
078        this.tx.adjustCounterField(this.id, this.storageId, offset, this.updateVersion);
079    }
080}
