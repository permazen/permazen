060    /**
061     * Determine if instances of the given type require any validation.
062     *
063     * <p>
064     * This will be true if {@code type} or any of its declared methods has a JSR 303 (<i>public</i> methods only)
065     * or {@link Validate &#64;Validate} annotation, or if any of its super-types requires validation.
066     * </p>
067     *
068     * @param type object type
069     * @return true if {@code type} has any validation requirements
070     * @throws IllegalArgumentException if {@code type} is null
071     * @see ValidationMode
072     */
073    public static boolean requiresValidation(Class<?> type) {
074        if (type == null)
075            throw new IllegalArgumentException("null type");
076        if (Util.hasValidationAnnotation(type))
077            return true;
078        for (Method method : type.getDeclaredMethods()) {
079            if (method.isAnnotationPresent(Validate.class)
080              || ((method.getModifiers() & Modifier.PUBLIC) != 0 && Util.hasValidationAnnotation(method)))
081                return true;
082        }
083        for (TypeToken<?> typeToken : TypeToken.of(type).getTypes()) {
084            final Class<?> superType = typeToken.getRawType();
085            if (superType != type && Util.requiresValidation(superType))
086                return true;
087        }
088        return false;
089    }

091    /**
092     * Determine if the given getter method, or any method it overrides, has a JSR 303 validation constraint.
093     *
094     * @param method annotated method
095     * @return true if {@code obj} has one or more JSR 303 annotations
096     * @throws IllegalArgumentException if {@code method} is null
097     */
098    public static boolean requiresValidation(Method method) {
099        if (method == null)
100            throw new IllegalArgumentException("null method");
101        for (TypeToken<?> typeToken : TypeToken.of(method.getDeclaringClass()).getTypes()) {
102            final Class<?> superType = typeToken.getRawType();
103            try {
104                method = superType.getMethod(method.getName(), method.getParameterTypes());
105            } catch (NoSuchMethodException e) {
106                continue;
107            }
108            if (Util.hasValidationAnnotation(method))
109                return true;
110        }
111        return false;
112    }
113
114    /**
115     * Determine whether the given object has any JSR 303 annotation(s).
116     *
117     * @param obj annotated element
118     * @return true if {@code obj} has one or more JSR 303 annotations
119     * @throws IllegalArgumentException if {@code obj} is null
120     */
121    public static boolean hasValidationAnnotation(AnnotatedElement obj) {
122        if (obj == null)
123            throw new IllegalArgumentException("null obj");
124        for (Annotation annotation : obj.getAnnotations()) {
125            final Class<?> annotationType = annotation.annotationType();
126            if (annotationType.isAnnotationPresent(Constraint.class))
127                return true;
128        }
129        return false;
130    }

132    /**
133     * Find the setter method corresponding to a getter method.
134     *
135     * @param type Java type (possibly a sub-type of the type in which {@code getter} is declared)
136     * @param getter Java bean property getter method
137     * @return Java bean property setter method
138     * @throws IllegalArgumentException if no corresponding setter method exists
139     */
140    public static Method findSetterMethod(Class<?> type, Method getter) {
141        final Matcher matcher = Pattern.compile("(is|get)(.+)").matcher(getter.getName());
142        if (!matcher.matches()) {
143            throw new IllegalArgumentException("can't infer setter method name from getter method "
144              + getter.getName() + "() because name does not follow Java bean naming conventions");
145        }
146        final String setterName = "set" + matcher.group(2);
147        for (TypeToken<?> superType : TypeToken.of(type).getTypes()) {
148            try {
149                final Method setter = superType.getRawType().getMethod(setterName, getter.getReturnType());
150                if (setter.getReturnType() != Void.TYPE)
151                    continue;
152                return setter;
153            } catch (NoSuchMethodException e) {
154                continue;
155            }
156        }
157        throw new IllegalArgumentException("can't find any setter method " + setterName
158          + "() corresponding to getter method " + getter.getName() + "() taking " + getter.getReturnType()
159          + " and returning void");
160    }

162    /**
163     * Find the narrowest type that is a supertype of all of the given types.
164     *
165     * <p>
166     * This method delegates to {@link #findLowestCommonAncestor findLowestCommonAncestor()}
167     * after converting the {@link Class} instances to {@link TypeToken}s.
168     * </p>
169     *
170     * @param types sub-types
171     * @return narrowest common super-type
172     * @see #findLowestCommonAncestor findLowestCommonAncestor()
173     */
174    public static TypeToken<?> findLowestCommonAncestorOfClasses(Iterable<Class<?>> types) {
175        return Util.findLowestCommonAncestor(Iterables.transform(types, new Function<Class<?>, TypeToken<?>>() {
176            @Override
177            public TypeToken<?> apply(Class<?> type) {
178                return TypeToken.of(type);
179            }
180        }));
181    }
182
183    /**
184     * Find the narrowest type that is a supertype of all of the given types.
185     *
186     * <p>
187     * When there is more than one choice, heuristics are used. For example, we prefer
188     * non-interface types, and {@link JObject} over other interface types.
189     * </p>
190     *
191     * @param types sub-types
192     * @return narrowest common super-type
193     */
194    public static TypeToken<?> findLowestCommonAncestor(Iterable<TypeToken<?>> types) {
195
196        // Gather all supertypes of types recursively
197        final HashSet<TypeToken<?>> supertypes = new HashSet<>();
198        for (TypeToken<?> type : types)
199            Util.addSupertypes(supertypes, type);
200
201        // Throw out all supertypes that are not supertypes of every type
202        for (Iterator<TypeToken<?>> i = supertypes.iterator(); i.hasNext(); ) {
203            final TypeToken<?> supertype = i.next();
204            for (TypeToken<?> type : types) {
205                if (!supertype.isAssignableFrom(type)) {
206                    i.remove();
207                    break;
208                }
209            }
210        }
211
212        // Throw out all supertypes that are supertypes of some other supertype
213        for (Iterator<TypeToken<?>> i = supertypes.iterator(); i.hasNext(); ) {
214            final TypeToken<?> supertype = i.next();
215            for (TypeToken<?> supertype2 : supertypes) {
216                if (supertype2 != supertype && supertype.isAssignableFrom(supertype2)) {
217                    i.remove();
218                    break;
219                }
220            }
221        }
222
223        // Pick the best candidate that's not Object, if possible
224        final TypeToken<Object> objectType = TypeToken.of(Object.class);
225        supertypes.remove(objectType);
226        switch (supertypes.size()) {
227        case 0:
228            return objectType;
229        case 1:
230            return supertypes.iterator().next();
231        default:
232            break;
233        }
234
235        // Pick the one that's not an interface, if any (it will be the the only non-interface type)
236        for (TypeToken<?> supertype : supertypes) {
237            if (!supertype.getRawType().isInterface())
238                return supertype;
239        }
240
241        // There are now only interfaces to choose from (or Object)... try JObject
242        final TypeToken<JObject> jobjectType = TypeToken.of(JObject.class);
243        if (supertypes.contains(jobjectType))
244            return jobjectType;
245
246        // Last resort is Object
247        return objectType;
248    }

250    @SuppressWarnings("unchecked")
251    private static <T> void addSupertypes(Set<TypeToken<?>> types, TypeToken<T> type) {
252        if (type == null || !types.add(type))
253            return;
254        final Class<? super T> rawType = type.getRawType();
255        types.add(TypeToken.of(rawType));
256        types.add(Util.getWildcardedType(rawType));
257        final Class<? super T> superclass = rawType.getSuperclass();
258        if (superclass != null) {
259            Util.addSupertypes(types, TypeToken.of(superclass));
260            Util.addSupertypes(types, type.getSupertype(superclass));
261        }
262        for (Class<?> iface : rawType.getInterfaces())
263            Util.addSupertypes(types, type.getSupertype((Class<? super T>)iface));
264    }

266    /**
267     * Parameterize the raw type with wildcards.
268     *
269     * @param type raw type
270     * @param <T> raw type
271     * @return {@code type} genericized with wildcards
272     */
273    public static <T> TypeToken<? extends T> getWildcardedType(Class<T> type) {
274        if (type == null)
275            throw new IllegalArgumentException("null type");
276        final TypeVariable<Class<T>>[] typeVariables = type.getTypeParameters();
277        if (typeVariables.length == 0)
278            return TypeToken.of(type);
279        final WildcardType[] questionMarks = new WildcardType[typeVariables.length];
280        Arrays.fill(questionMarks, QUESTION_MARK);
281        return Util.newParameterizedType(type, questionMarks);
282    }

284    /**
285     * Get the n'th generic type parameter.
286     *
287     * @param type parameterized generic type
288     * @param index type parameter index (zero based)
289     * @return type parameter at {@code index}
290     * @throws IllegalArgumentException if {@code type} is not a parameterized type with more than {@code index} type variables
291     */
292    public static Type getTypeParameter(Type type, int index) {
293        if (!(type instanceof ParameterizedType))
294            throw new IllegalArgumentException("type is missing generic type parameter(s)");
295        final ParameterizedType parameterizedType = (ParameterizedType)type;
296        final Type[] parameters = parameterizedType.getActualTypeArguments();
297        if (index >= parameters.length)
298            throw new IllegalArgumentException("type is missing generic type parameter(s)");
299        return parameters[index];
300    }

302    /**
303     * Invoke method via reflection and re-throw any checked exception wrapped in an {@link JSimpleDBException}.
304     *
305     * @param method method to invoke
306     * @param target instance, or null if method is static
307     * @param params method parameters
308     * @return method return value
309     */
310    public static Object invoke(Method method, Object target, Object... params) {
311        try {
312            return method.invoke(target, params);
313        } catch (InvocationTargetException e) {
314            if (e.getCause() instanceof Error)
315                throw (Error)e.getCause();
316            if (e.getCause() instanceof RuntimeException)
317                throw (RuntimeException)e.getCause();
318            throw new JSimpleDBException("unexpected error invoking method " + method + " on " + target, e);
319        } catch (Error e) {
320            throw e;
321        } catch (RuntimeException e) {
322            throw e;
323        } catch (Exception e) {
324            throw new JSimpleDBException("unexpected error invoking method " + method + " on " + target, e);
325        }
326    }

328    /**
329     * Convert a raw class back into its generic type using caller-supplied type parameters.
330     *
331     * @param target raw class
332     * @param params type parameters
333     * @param <T> raw class type
334     * @return generic {@link TypeToken} for {@code target}
335     * @see <a href="https://github.com/google/guava/issues/1645">Guava Issue #1645</a>
336     */
337    @SuppressWarnings("unchecked")
338    public static <T> TypeToken<? extends T> newParameterizedType(Class<T> target, Type[] params) {
339        Type type;
340        try {
341            if (Util.newParameterizedTypeMethod == null) {
342                Util.newParameterizedTypeMethod = Class.forName("com.google.common.reflect.Types",
343                  false, Thread.currentThread().getContextClassLoader()).getDeclaredMethod(
344                  "newParameterizedType", Class.class, Type[].class);
345                Util.newParameterizedTypeMethod.setAccessible(true);
346            }
347            type = (Type)Util.newParameterizedTypeMethod.invoke(null, target, params);
348        } catch (RuntimeException e) {
349            throw e;
350        } catch (Exception e) {
351            throw new RuntimeException("unexpected exception", e);
352        }
353        return (TypeToken<T>)TypeToken.of(type);
354    }
