085
086    private Session(JSimpleDB jdb, Database db, KVDatabase kvdb) {
087        Preconditions.checkArgument(kvdb != null, "null kvdb");
088        if (jdb != null) {
089            Preconditions.checkArgument(db != null, "null db");
090            this.mode = SessionMode.JSIMPLEDB;
091        } else if (db != null)
092            this.mode = SessionMode.CORE_API;
093        else
094            this.mode = SessionMode.KEY_VALUE;
095        this.jdb = jdb;
096        this.db = db;
097        this.kvdb = kvdb;
098    }

112     * Change this instance's {@link SessionMode}.
113     *
114     * @param mode new {@link SessionMode}
115     * @throws IllegalArgumentException if {@code mode} is null
116     * @throws IllegalArgumentException if {@code mode} requires a {@link JSimpleDB} (i.e., {@link SessionMode#JSIMPLEDB}),
117     *  or {@link Database} (i.e., {@link SessionMode#CORE_API}) instance, but none was provided at construction
118     */
119    public void setMode(SessionMode mode) {
120        Preconditions.checkNotNull(mode, "null mode");
121        switch (mode) {
122        case KEY_VALUE:
123            break;
124        case CORE_API:
125            Preconditions.checkNotNull(this.db, "session is not configured with a Core API Database instance");
126            break;
127        case JSIMPLEDB:
128            Preconditions.checkNotNull(this.jdb, "session is not configured with a JSimpleDB instance");
129            break;
130        default:
131            throw new IllegalArgumentException("internal error");
132        }
133        this.mode = mode;
134    }

321    /**
322     * Associate the current {@link JTransaction} with this instance while performing the given action.
323     *
324     * <p>
325     * If {@code action} throws an {@link Exception}, it will be caught and handled by {@link #reportException reportException()}
326     * and then false returned.
327     *
328     * <p>
329     * There must be a {@link JTransaction} open and {@linkplain JTransaction#getCurrent associated with the current thread}.
330     * It will be left open when this method returns.
331     *
332     * <p>
333     * This method safely handles re-entrant invocation.
334     *
335     * @param action action to perform
336     * @return true if {@code action} completed successfully, false if {@code action} threw an exception
337     * @throws IllegalArgumentException if {@code action} is null
338     * @throws IllegalStateException if there is already an open transaction associated with this instance
339     * @throws IllegalStateException if this instance is not in mode {@link SessionMode#JSIMPLEDB}
340     */
341    public boolean performSessionActionWithCurrentTransaction(Action action) {
342
343        // Sanity check
344        Preconditions.checkArgument(action != null, "null action");
345        Preconditions.checkArgument(SessionMode.JSIMPLEDB.equals(this.mode), "session is not in JSimpleDB mode");
346
347        // Check for re-entrant invocation, otherwise verify no other transaction is associated
348        final Transaction currentTx = JTransaction.getCurrent().getTransaction();
349        final KVTransaction currentKVT = currentTx.getKVTransaction();
350        final boolean associate;
351        if (this.tx == null && this.kvt == null)
352            associate = true;
353        else if (this.tx == currentTx && this.kvt == currentKVT)
354            associate = false;
355        else
356            throw new IllegalStateException("a different transaction is already open in this session");
357
358        // Perform action
359        if (associate) {
360            this.tx = currentTx;
361            this.kvt = currentKVT;
362        }
363        try {
364            action.run(this);
365            return true;
366        } catch (Exception e) {
367            this.reportException(e);
368            return false;
369        } finally {
370            if (associate) {
371                this.tx = null;
372                this.kvt = null;
373            }
374        }
375    }

377    /**
378     * Perform the given action within a new transaction associated with this instance.
379     *
380     * <p>
381     * If {@code action} throws an {@link Exception}, it will be caught and handled by {@link #reportException reportException()}
382     * and then false returned.
383     *
384     * @param action action to perform
385     * @return true if {@code action} completed successfully, false if the transaction could not be created
386     *  or {@code action} threw an exception
387     * @throws IllegalArgumentException if {@code action} is null
388     * @throws IllegalStateException if there is already an open transaction associated with this instance
389     */
390    public boolean performSessionAction(Action action) {
391
392        // Sanity check
393        Preconditions.checkArgument(action != null, "null action");
394
395        // Perform action within new transaction
396        try {
397            if (!this.openTransaction())
398                return false;
399            boolean success = false;
400            this.rollbackOnly = false;
401            try {
402                action.run(this);
403                success = true;
404            } finally {
405                success &= this.closeTransaction(success && !this.rollbackOnly);
406            }
407            return success;
408        } catch (Exception e) {
409            this.reportException(e);
410            return false;
411        } finally {
412            this.tx = null;
413        }
414    }

416    /**

418     *
419     * @throws IllegalStateException if {@link #performSessionAction performSessionAction()} is not currently being invoked

421    public void setRollbackOnly() {
422        Preconditions.checkState(this.kvt != null, "no transaction is open in this session");
423        this.rollbackOnly = true;
424    }

426    private boolean openTransaction() {
427        try {
428
429            // Sanity check
430            Preconditions.checkState(this.tx == null && this.kvt == null, "a transaction is already open in this session");
431
432            // Open transaction at the appropriate level
433            switch (this.mode) {
434            case KEY_VALUE:
435                this.kvt = this.kvdb.createTransaction();
436                break;
437            case CORE_API:
438                this.tx = this.db.createTransaction(this.schemaModel, this.schemaVersion, this.allowNewSchema);
439                this.kvt = this.tx.getKVTransaction();
440                break;
441            case JSIMPLEDB:
442                Preconditions.checkState(!Session.isCurrentJTransaction(),
443                  "a JSimpleDB transaction is already open in the current thread");
444                final JTransaction jtx = this.jdb.createTransaction(this.allowNewSchema,
445                  validationMode != null ? validationMode : ValidationMode.AUTOMATIC);
446                JTransaction.setCurrent(jtx);
447                this.tx = jtx.getTransaction();
448                this.kvt = this.tx.getKVTransaction();
449                break;
450            default:
451                assert false;
452                break;
453            }
454
455            // Update schema model and version (if appropriate)
456            if (this.tx != null) {
457                final Schema schema = this.tx.getSchema();
458                this.setSchemaModel(schema.getSchemaModel());
459                this.setSchemaVersion(schema.getVersionNumber());
460                this.tx.setReadOnly(this.readOnly);
461            }
462
463            // Done
464            return true;
465        } catch (Exception e) {
466            this.tx = null;
467            this.kvt = null;
468            this.reportException(e);
469            return false;
470        }
471    }
472
473    @SuppressWarnings("fallthrough")
474    private boolean closeTransaction(boolean commit) {
475        final SessionMode currentMode = this.mode;
476        try {
477            Preconditions.checkState(this.tx != null || this.kvt != null, "no transaction");
478            switch (currentMode) {
479            case JSIMPLEDB:
480                if (commit && !this.tx.isRollbackOnly())
481                    JTransaction.getCurrent().commit();
482                else
483                    JTransaction.getCurrent().rollback();
484                break;
485            case CORE_API:
486                if (commit && !this.tx.isRollbackOnly())
487                    this.tx.commit();
488                else
489                    this.tx.rollback();
490                break;
491            case KEY_VALUE:
492                if (commit)
493                    this.kvt.commit();
494                else
495                    this.kvt.rollback();
496                break;
497            default:
498                assert false;
499                break;
500            }
501            return true;
502        } catch (Exception e) {
503            this.reportException(e);
504            return false;
505        } finally {
506            switch (currentMode) {
507            case JSIMPLEDB:
508                JTransaction.setCurrent(null);
509                // FALLTHROUGH
510            case CORE_API:
511                this.tx = null;
512                // FALLTHROUGH
513            case KEY_VALUE:
514                this.kvt = null;
515                // FALLTHROUGH
516            default:
517                break;
518            }
519        }
520    }
521
522    private static boolean isCurrentJTransaction() {
523        try {
524            JTransaction.getCurrent();
525            return true;
526        } catch (IllegalStateException e) {
527            return false;
528        }
529    }

533    /**
534     * Callback interface used by {@link Session#performSessionAction Session.performSessionAction()}
535     * and {@link Session#performSessionActionWithCurrentTransaction Session.performSessionActionWithCurrentTransaction()}.

537    public interface Action {
538

540         * Perform some action using the given {@link Session} while a transaction is open.

542         * @param session session with open transaction

544         */
