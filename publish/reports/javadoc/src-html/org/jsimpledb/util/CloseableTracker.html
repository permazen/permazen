006package org.jsimpledb.util;
007
008import java.io.Closeable;
009import java.lang.ref.ReferenceQueue;
010import java.lang.ref.WeakReference;
011import java.util.ArrayList;
012import java.util.HashSet;
013
014import org.slf4j.Logger;
015import org.slf4j.LoggerFactory;
016
017/**
018 * Keeps track of {@link Closeable} objects (termed <i>items</i>) that must be reliably {@linkplain Closeable#close closed}
019 * prior to shutting down some associated context, but that also must not be closed prior to becoming unreachable, when
020 * the straightforward strategy of just storing them until context shutdown would require too much memory. In a
021 * typical scenario, these items are returned (indirectly by reference from some associated <i>holder</i> object) from some
022 * method to callers who cannot be expected to reliably close them. Following the simple strategy of just storing all unclosed
023 * items until context shutdown means unbounded memory use can occur; this class solves that problem.
024 *
025 * <p>
026 * For each such context, an instance of this class may be used to register and track the associated items,
027 * guaranteeing that they will always eventually be closed, but doing so in a way that avoids memory leaks:
028 * For each item, there must be a corresponding <i>holder</i> object containing a reference to it. The holder objects are
029 * then tracked by this class using weak references. When a holder object is no longer strongly referenced, the corresponding
030 * item is closed. With this scheme, no memory leak occurs due to this tracking, even when arbitrarily many items are created.
031 * This of course assumes that when a holder object is no longer referenced, the associated item may be safely closed.
032 * </p>
033 *
034 * <p>
035 * A registered {@link Closeable} item is closed at the first occurrence of any of the following:
036 * <ul>
037 *  <li>
038 *  <li>The application itself invokes {@link Closeable#close Closeable.close()} on the item</li>
039 *  <li>The associated holder object is no longer strongly referenced, and then {@link #poll} is invoked</li>
040 *  <li>{@link CloseableTracker#close} is invoked</li>
041 *  <li>{@link CloseableTracker#finalize} is invoked (i.e., this instance is garbage collected)</li>
042 * </ul>
043 *
044 * <p>
045 * Note that {@link Closeable#close Closeable.close()} is required to be idempotent, so application usage
046 * that results in eager closing of items is appropriate and encouraged. Use of this class may occasionally
047 * result in {@link Closeable#close Closeable.close()} being invoked more than once for registered items.
048 * </p>
049 *
050 * <p>
051 * Instances of this class are thread safe.
052 * </p>
053 */
054public class CloseableTracker implements Closeable {
055
056    protected final Logger log = LoggerFactory.getLogger(this.getClass());
057
058    private HashSet<HolderRef> unclosedItems;
059    private ReferenceQueue<Object> queue;
060
061// Constructors
062
063    /**
064     * Constructor.
065     */
066    public CloseableTracker() {
067        this.reset();
068    }
069
070// Public API
071
072    /**
073     * Add an item to the set of items being tracked by this instance.
074     *
075     * @param holder item's holder
076     * @param item item to track
077     * @throws IllegalArgumentException if either parameter is null
078     */
079    public synchronized void add(Object holder, Closeable item) {
080        this.unclosedItems.add(new HolderRef(holder, item, this.queue));
081    }
082
083    /**
084     * Poll the internal weak reference queue and close any unclosed items whose holders are no longer reachable.
085     *
086     * <p>
087     * Applications should invoke this method periodically to avoid memory leaks.
088     * </p>
089     */
090    public void poll() {
091
092        // Poll for refs and close the associated items
093        ArrayList<HolderRef> closedItems = null;
094        for (HolderRef ref; (ref = (HolderRef)this.queue.poll()) != null; ) {
095            if (closedItems == null)
096                closedItems = new ArrayList<>();
097            final Closeable item = ref.getItem();
098            try {
099                item.close();
100            } catch (Throwable e) {
101                this.exceptionDuringClose(item, e);
102            }
103            closedItems.add(ref);
104        }
105
106        // Remove closed items from the set of unclosed items
107        if (closedItems != null) {
108            synchronized (this) {
109                this.unclosedItems.removeAll(closedItems);
110            }
111        }
112    }
113
114    /**
115     * Close all unclosed items associated with this instance.
116     *
117     * <p>
118     * The implementation in {@link CloseableTracker} just invokes {@link #reset}.
119     * </p>
120     */
121    @Override
122    public void close() {
123        this.reset();
124    }
125
126    /**
127     * Reset this instance.
128     *
129     * <p>
130     * Results in the forced closing of all unclosed items associated with this instance.
131     * </p>
132     *
133     * <p>
134     * This instance may be reused after invoking this method.
135     * </p>
136     */
137    public void reset() {
138
139        // Snapshot remaining items and reset state
140        final HashSet<HolderRef> itemsToClose;
141        synchronized (this) {
142            itemsToClose = this.unclosedItems;
143            this.unclosedItems = new HashSet<>();
144            this.queue = new ReferenceQueue<>();            // discard old queue, everything on it is about to be closed
145        }
146
147        // This case happens during construction
148        if (itemsToClose == null)
149            return;
150
151        // Close unclosed items
152        for (HolderRef ref : itemsToClose) {
153            final Closeable item = ref.getItem();
154            try {
155                item.close();
156            } catch (Throwable e) {
157                this.exceptionDuringClose(item, e);
158            }
159        }
160    }
161
162// Subclass methods
163
164    /**
165     * Handle an exception thrown while attempting to {@link Closeable#close close()} and item.
166     *
167     * <p>
168     * The implementation in {@link CloseableTracker} just logs the exception at debug level and returns.
169     * </p>
170     *
171     * @param item item that was closed
172     * @param e exeption thrown
173     */
174    protected void exceptionDuringClose(Closeable item, Throwable e) {
175        if (this.log.isDebugEnabled())
176            this.log.debug("caught exception attempting to close " + item + " (ignoring)", e);
177    }
178
179// Object
180
181    @Override
182    protected void finalize() throws Throwable {
183        try {
184            this.close();
185        } finally {
186            super.finalize();
187        }
188    }
189
190// HolderRef
191
192    private static class HolderRef extends WeakReference<Object> {
193
194        private final Closeable item;
195
196        HolderRef(Object holder, Closeable item, ReferenceQueue<Object> queue) {
197            super(holder, queue);
198            if (holder == null)
199                throw new IllegalArgumentException("null holder");
200            if (item == null)
201                throw new IllegalArgumentException("null item");
202            this.item = item;
203        }
204
205        public Closeable getItem() {
206            return this.item;
207        }
208    }
209}
