018/**
019 * A map with {@link ObjId} keys.
020 *
021 * <p>
022 * This implementation is space optimized for the 64-bits of information contained in an {@link ObjId}.
023 * Instances do not accept null keys and are not thread safe.
024 * </p>
025 */
026public class ObjIdMap<V> extends AbstractMap<ObjId, V> implements Cloneable {
027
028    // Algorithm described here: http://en.wikipedia.org/wiki/Open_addressing
029
030    private static final float EXPAND_THRESHOLD = 0.70f;    // expand array when > 70% full
031    private static final float SHRINK_THRESHOLD = 0.25f;    // shrink array when < 25% full
032
033    private static final int MIN_LOG2_LENGTH = 4;           // minimum array length = 16 slots
034    private static final int MAX_LOG2_LENGTH = 30;          // maximum array length = 1 billion slots
035
036    private long[] keys;                                    // has length always a power of 2
037    private V[] values;                                     // will be null if we are being used to implement ObjIdSet
038    private int size;                                       // the number of entries in the map
039    private int log2len;                                    // log2 of keys.length and values.length (if not null)
040    private int upperSizeLimit;                             // size threshold when to grow array
041    private int lowerSizeLimit;                             // size threshold when to shrink array
042    private int numHashShifts;                              // used by hash() function
043    private volatile int modcount;
044
045// Constructors
046
047    /**
048     * Constructs an empty instance.
049     */
050    public ObjIdMap() {
051        this(0, true);
052    }
053
054    /**
055     * Constructs an instance with the given initial capacity.
056     *
057     * @param capacity initial capacity
058     * @throws IllegalArgumentException if {@code capacity} is negative
059     */
060    public ObjIdMap(int capacity) {
061        this(capacity, true);
062    }
063
064    /**
065     * Constructs an instance initialized from the given map.
066     *
067     * @param map initial contents for this instance
068     * @throws NullPointerException if {@code map} is null
069     * @throws IllegalArgumentException if {@code map} contains a null key
070     */
071    public ObjIdMap(Map<ObjId, ? extends V> map) {
072        this(map.size(), true);
073        for (Map.Entry<ObjId, ? extends V> entry : map.entrySet())
074            this.put(entry.getKey(), entry.getValue());
075    }
076
077    // Internal constructor
078    ObjIdMap(int capacity, boolean withValues) {
079        if (capacity < 0)
080            throw new IllegalArgumentException("capacity < 0");
081        capacity &= 0x3fffffff;                                                 // avoid integer overflow from large values
082        capacity = (int)(capacity / EXPAND_THRESHOLD);                          // increase to account for overhead
083        capacity = Math.max(1, capacity);                                       // avoid zero, on which the next line fails
084        this.log2len = 32 - Integer.numberOfLeadingZeros(capacity - 1);         // round up to next power of 2
085        this.log2len = Math.max(MIN_LOG2_LENGTH, this.log2len);                 // clip to bounds
086        this.log2len = Math.min(MAX_LOG2_LENGTH, this.log2len);
087        this.createArrays(withValues);
088    }

099        return this.size == 0;

102    @Override
103    public boolean containsKey(Object obj) {
104
105        // Check type
106        if (!(obj instanceof ObjId))
107            return false;
108        final long value = ((ObjId)obj).asLong();
109        assert value != 0;
110
111        // Check slot for value
112        final int slot = this.findSlot(value);
113        if (this.keys[slot] == value)
114            return true;
115        assert this.keys[slot] == 0;
116        return false;
117    }
118
119    @Override
120    public V get(Object obj) {
121
122        // Check type
123        if (!(obj instanceof ObjId))
124            return null;
125        final long value = ((ObjId)obj).asLong();
126        assert value != 0;
127
128        // Check slot for value
129        final int slot = this.findSlot(value);
130        if (this.keys[slot] == value)
131            return this.values != null ? this.values[slot] : null;
132        assert this.keys[slot] == 0;
133        return null;
134    }
135
136    @Override
137    public V put(ObjId id, V value) {
138        if (id == null)
139            throw new IllegalArgumentException("illegal null key");
140        final long key = id.asLong();
141        assert key != 0;
142        return this.insert(key, value);
143    }
144
145    @Override
146    public V remove(Object obj) {
147        if (!(obj instanceof ObjId))
148            return null;
149        final long key = ((ObjId)obj).asLong();
150        assert key != 0;
151        return this.exsert(key);
152    }
153
154    @Override
155    public void clear() {
156        this.log2len = MIN_LOG2_LENGTH;
157        this.createArrays(this.values != null);
158        this.size = 0;
159        this.modcount++;
160    }

194    public ObjIdMap<V> clone() {
195        final ObjIdMap<V> clone;
196        try {
197            clone = (ObjIdMap<V>)super.clone();
198        } catch (CloneNotSupportedException e) {
199            throw new RuntimeException(e);
200        }
201        clone.keys = clone.keys.clone();
202        if (clone.values != null)
203            clone.values = clone.values.clone();
204        return clone;
205    }

223    private V insert(long key, V value) {
224
225        // Find slot for key
226        assert key != 0;
227        final int slot = this.findSlot(key);
228
229        // Key already exists? Just replace value
230        if (this.keys[slot] == key) {
231            if (this.values == null)
232                return null;
233            final V prev = this.values[slot];
234            this.values[slot] = value;
235            return prev;
236        }
237
238        // Insert new key/value pair
239        assert this.keys[slot] == 0;
240        this.keys[slot] = key;
241        if (this.values != null) {
242            assert this.values[slot] == null;
243            this.values[slot] = value;
244        }
245
246        // Expand if necessary
247        if (++this.size > this.upperSizeLimit && this.log2len < MAX_LOG2_LENGTH) {
248            this.log2len++;
249            this.resize();
250        }
251        this.modcount++;
252        return null;
253    }
254
255    private V exsert(long key) {
256
257        // Find slot for key
258        final int slot = this.findSlot(key);
259        if (this.keys[slot] == 0) {
260            assert this.values == null || this.values[slot] == null;
261            return null;
262        }
263        assert this.keys[slot] == key;
264
265        // Remove key
266        return this.exsert(slot);
267    }
268
269    private V exsert(final int slot) {
270
271        // Sanity check
272        assert this.keys[slot] != 0;
273        final V ovalue = this.values != null ? this.values[slot] : null;
274
275        // Remove key/value pair and fixup subsequent entries
276        int i = slot;                                                   // i points to the new empty slot
277        int j = slot;                                                   // j points to the next slot to fixup
278loop:   while (true) {
279            this.keys[i] = 0;
280            if (this.values != null)
281                this.values[i] = null;
282            long jkey;
283            V jvalue;
284            while (true) {
285                j = (j + 1) & (this.keys.length - 1);
286                jkey = this.keys[j];
287                if (jkey == 0)                                          // end of hash chain, no more fixups required
288                    break loop;
289                jvalue = this.values != null ? this.values[j] : null;   // get corresponding value
290                final int k = this.hash(jkey);                          // find where jkey's hash chain started
291                if (i <= j ? (i < k && k <= j) : (i < k || k <= j))     // jkey is between i and j, so it's not cut off
292                    continue;
293                break;                                                  // jkey is cut off from its hash chain, need to fix
294            }
295            this.keys[i] = jkey;                                        // move jkey back into its hash chain
296            if (this.values != null)
297                this.values[i] = jvalue;
298            i = j;                                                      // restart fixups at jkey's old location
299        }
300
301        // Shrink if necessary
302        if (--this.size < this.lowerSizeLimit && this.log2len > MIN_LOG2_LENGTH) {
303            this.log2len--;
304            this.resize();
305        }
306        this.modcount++;
307        return ovalue;
308    }
309
310    private int findSlot(long value) {
311        assert value != 0;
312        int slot = this.hash(value);
313        while (true) {
314            final long existing = this.keys[slot];
315            if (existing == 0 || existing == value)
316                return slot;
317            slot = (slot + 1) & (this.keys.length - 1);
318        }
319    }
320
321    private int hash(long value) {
322        final int shift = this.log2len;
323        int hash = (int)value;
324        for (int i = 0; i < this.numHashShifts; i++) {
325            value >>>= shift;
326            hash ^= (int)value;
327        }
328        return hash & (this.keys.length - 1);
329    }
330
331    private void resize() {
332
333        // Grab a copy of old arrays and create new ones
334        final long[] oldKeys = this.keys;
335        final V[] oldValues = this.values;
336        assert oldValues == null || oldValues.length == oldKeys.length;
337        this.createArrays(oldValues != null);
338
339        // Rehash key/value pairs from old array into new array
340        for (int oldSlot = 0; oldSlot < oldKeys.length; oldSlot++) {
341            final long key = oldKeys[oldSlot];
342            if (key == 0) {
343                assert oldValues == null || oldValues[oldSlot] == null;
344                continue;
345            }
346            final int newSlot = this.findSlot(key);
347            assert this.keys[newSlot] == 0;
348            this.keys[newSlot] = key;
349            if (this.values != null) {
350                assert this.values[newSlot] == null;
351                this.values[newSlot] = oldValues[oldSlot];
352            }
353        }
354    }
355
356    @SuppressWarnings("unchecked")
357    private void createArrays(boolean withValues) {
358        assert this.log2len >= MIN_LOG2_LENGTH;
359        assert this.log2len <= MAX_LOG2_LENGTH;
360        final int arrayLength = 1 << this.log2len;
361        this.lowerSizeLimit = this.log2len > MIN_LOG2_LENGTH ? (int)(SHRINK_THRESHOLD * arrayLength) : 0;
362        this.upperSizeLimit = this.log2len < MAX_LOG2_LENGTH ? (int)(EXPAND_THRESHOLD * arrayLength) : arrayLength;
363        this.numHashShifts = (64 + (this.log2len - 1)) / this.log2len;
364        this.numHashShifts = Math.min(12, this.numHashShifts);
365        this.keys = new long[arrayLength];
366        if (withValues)
367            this.values = (V[])new Object[arrayLength];
368    }

372    class EntrySet extends AbstractSet<Map.Entry<ObjId, V>> {
373
374        @Override
375        public Iterator<Map.Entry<ObjId, V>> iterator() {
376            return new EntrySetIterator();
377        }
378
379        @Override
380        public int size() {
381            return ObjIdMap.this.size;
382        }
383
384        @Override
385        public boolean contains(Object obj) {
386            if (!(obj instanceof Map.Entry))
387                return false;
388            final Map.Entry<?, ?> entry = (Map.Entry<?, ?>)obj;
389            final Object key = entry.getKey();
390            final V actualValue = ObjIdMap.this.get(key);
391            if (actualValue == null && !ObjIdMap.this.containsKey(key))
392                return false;
393            return entry.equals(new AbstractMap.SimpleEntry<ObjId, V>((ObjId)key, actualValue));
394        }
395
396        @Override
397        public boolean remove(Object obj) {
398            if (!(obj instanceof Map.Entry))
399                return false;
400            final Map.Entry<?, ?> entry = (Map.Entry<?, ?>)obj;
401            final Object key = entry.getKey();
402            final V actualValue = ObjIdMap.this.get(key);
403            if (actualValue == null && !ObjIdMap.this.containsKey(key))
404                return false;
405            if (actualValue != null ? actualValue.equals(entry.getValue()) : entry.getValue() == null) {
406                ObjIdMap.this.remove(key);
407                return true;
408            }
409            return false;
410        }
411
412        @Override
413        public void clear() {
414            ObjIdMap.this.clear();
415        }
416
417        // This works because ObjId.hashCode() == ObjId.asLong().hashCode()
418        @Override
419        public int hashCode() {
420            final long[] keyArray = ObjIdMap.this.keys;
421            final V[] valueArray = ObjIdMap.this.values;
422            int hash = 0;
423            for (int i = 0; i < keyArray.length; i++) {
424                final long key = keyArray[i];
425                if (key != 0) {
426                    int entryHash = (int)(key >>> 32) ^ (int)key;
427                    if (valueArray != null) {
428                        final V value = valueArray[i];
429                        if (value != null)
430                            entryHash ^= value.hashCode();
431                    }
432                    hash += entryHash;
433                }
434            }
435            return hash;
436        }
437    }

441    class EntrySetIterator implements Iterator<Map.Entry<ObjId, V>> {
442
443        private int modcount = ObjIdMap.this.modcount;
444        private int removeSlot = -1;
445        private int nextSlot;
446
447        @Override
448        public boolean hasNext() {
449            return this.findNext(false) != -1;
450        }
451
452        @Override
453        public Entry next() {
454            final int slot = this.findNext(true);
455            if (slot == -1)
456                throw new NoSuchElementException();
457            final long key = ObjIdMap.this.keys[slot];
458            assert key != 0;
459            this.removeSlot = slot;
460            return new Entry(slot);
461        }
462
463        @Override
464        public void remove() {
465            if (this.removeSlot == -1)
466                throw new IllegalStateException();
467            if (this.modcount != ObjIdMap.this.modcount)
468                throw new ConcurrentModificationException();
469            ObjIdMap.this.exsert(this.removeSlot);
470            this.removeSlot = -1;
471            this.modcount++;                            // keep synchronized with ObjIdMap.this.modcount
472        }
473
474        private int findNext(boolean advance) {
475            if (this.modcount != ObjIdMap.this.modcount)
476                throw new ConcurrentModificationException();
477            for (int slot = this.nextSlot; slot < ObjIdMap.this.keys.length; slot++) {
478                if (ObjIdMap.this.keys[slot] == 0)
479                    continue;
480                this.nextSlot = advance ? slot + 1 : slot;
481                return slot;
482            }
483            return -1;
484        }
485    }

489    @SuppressWarnings("serial")
490    class Entry extends AbstractMap.SimpleEntry<ObjId, V> {
491
492        private final int modcount = ObjIdMap.this.modcount;
493        private final int slot;
494
495        public Entry(int slot) {
496            super(new ObjId(ObjIdMap.this.keys[slot]), ObjIdMap.this.values != null ? ObjIdMap.this.values[slot] : null);
497            this.slot = slot;
498        }
499
500        @Override
501        public V setValue(V value) {
502            if (ObjIdMap.this.modcount != this.modcount)
503                throw new ConcurrentModificationException();
504            if (ObjIdMap.this.values != null)
505                ObjIdMap.this.values[this.slot] = value;
506            return super.setValue(value);
507        }
508    }
