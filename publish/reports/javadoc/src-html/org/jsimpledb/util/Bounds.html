006package org.jsimpledb.util;
007
008import java.util.Comparator;
009
010/**
011 * Utility class used by {@link AbstractNavigableSet} and {@link AbstractNavigableMap} to define
012 * the (optional) upper and lower bounds of a restricted range.
013 *
014 * @param <T> Java type of range bounds
015 */
016public class Bounds<T> {
017
018    private final T lowerBound;
019    private final T upperBound;
020    private final BoundType lowerBoundType;
021    private final BoundType upperBoundType;
022
023    /**
024     * Convenience constructor to create a new instance with no upper or lower bounds.
025     */
026    public Bounds() {
027        this(null, BoundType.NONE, null, BoundType.NONE);
028    }
029
030    /**
031     * Create a one-sided bound.
032     *
033     * @param bound bound restriction value; ignored if {@code boundType} is {@link BoundType#NONE}
034     * @param boundType type of bound for {@code bound}, or {@link BoundType#NONE} if there is no lower bound
035     * @param upper true to create an upper bound, false to create a lower bound
036     * @throws IllegalArgumentException if {@code boundType} is null
037     */
038    public Bounds(T bound, BoundType boundType, boolean upper) {
039        this(upper ? null : bound, upper ? BoundType.NONE : boundType, upper ? bound : null, upper ? boundType : BoundType.NONE);
040    }
041
042    /**
043     * Create two-sided bounds with inclusive lower bound and exclusive upper bound.
044     *
045     * @param lowerBound lower bound restriction value (inclusive)
046     * @param upperBound upper bound restriction value (exclusive)
047     */
048    public Bounds(T lowerBound, T upperBound) {
049        this(lowerBound, BoundType.INCLUSIVE, upperBound, BoundType.EXCLUSIVE);
050    }
051
052    /**
053     * Primary constructor.
054     *
055     * @param lowerBound lower bound restriction value; ignored if {@code lowerBoundType} is {@link BoundType#NONE}
056     * @param lowerBoundType type of bound for {@code lowerBound}, or {@link BoundType#NONE} if there is no lower bound
057     * @param upperBound upper bound restriction value; ignored if {@code upperBoundType} is {@link BoundType#NONE}
058     * @param upperBoundType type of bound for {@code upperBound}, or {@link BoundType#NONE} if there is no upper bound
059     * @throws IllegalArgumentException if {@code lowerBoundType} or {@code upperBoundType} is null
060     */
061    public Bounds(T lowerBound, BoundType lowerBoundType, T upperBound, BoundType upperBoundType) {
062        if (lowerBoundType == null)
063            throw new IllegalArgumentException("null lowerBoundType");
064        if (upperBoundType == null)
065            throw new IllegalArgumentException("null upperBoundType");
066        this.lowerBound = lowerBoundType != BoundType.NONE ? lowerBound : null;
067        this.upperBound = upperBoundType != BoundType.NONE ? upperBound : null;
068        this.lowerBoundType = lowerBoundType;
069        this.upperBoundType = upperBoundType;
070    }
071
072    /**
073     * Get the Java value corresponding to the lower bound restriction, if any.
074     * Returns null when {@link #getLowerBoundType} returns {@link BoundType#NONE}.
075     *
076     * @return lower bound Java value, or null if there is no lower bound
077     */
078    public T getLowerBound() {
079        return this.lowerBound;
080    }
081
082    /**
083     * Get the Java value corresponding to the upper bound restriction, if any.
084     * Returns null when {@link #getUpperBoundType} returns {@link BoundType#NONE}.
085     *
086     * @return upper bound Java value, or null if there is no upper bound
087     */
088    public T getUpperBound() {
089        return this.upperBound;
090    }
091
092    /**
093     * Get the type of the lower bound that corresponds to {@link #getLowerBound}.
094     *
095     * @return lower bound restriction type, never null
096     */
097    public BoundType getLowerBoundType() {
098        return this.lowerBoundType;
099    }
100
101    /**
102     * Get the type of the upper bound that corresponds to {@link #getUpperBound}.
103     *
104     * @return upper bound restriction type, never null
105     */
106    public BoundType getUpperBoundType() {
107        return this.upperBoundType;
108    }
109
110    /**
111     * Create an instance like this instance but with the upper and lower bounds reversed.
112     * Obviously, the result is only sensible when a reversed {@link Comparator} is also used.
113     *
114     * @return reversal of this instance
115     */
116    public Bounds<T> reverse() {
117        return new Bounds<T>(this.upperBound, this.upperBoundType, this.lowerBound, this.lowerBoundType);
118    }
119
120    /**
121     * Create an instance like this instance but with a different lower bound.
122     *
123     * @param newLowerBound new lower bound restriction value; ignored if {@code newLowerBoundType} is {@link BoundType#NONE}
124     * @param newLowerBoundType type of bound for {@code newLowerBound}, or {@link BoundType#NONE} if there is no lower bound
125     * @return instance with new lower bound
126     * @throws IllegalArgumentException if {@code newLowerBoundType} is null
127     */
128    public Bounds<T> withLowerBound(T newLowerBound, BoundType newLowerBoundType) {
129        return new Bounds<T>(newLowerBound, newLowerBoundType, this.upperBound, this.upperBoundType);
130    }
131
132    /**
133     * Create an instance like this instance but with a different upper bound.
134     *
135     * @param newUpperBound new upper bound restriction value; ignored if {@code newUpperBoundType} is {@link BoundType#NONE}
136     * @param newUpperBoundType type of bound for {@code newUpperBound}, or {@link BoundType#NONE} if there is no upper bound
137     * @return instance with new upper bound
138     * @throws IllegalArgumentException if {@code newUpperBoundType} is null
139     */
140    public Bounds<T> withUpperBound(T newUpperBound, BoundType newUpperBoundType) {
141        return new Bounds<T>(this.lowerBound, this.lowerBoundType, newUpperBound, newUpperBoundType);
142    }
143
144    /**
145     * Create an instance like this instance but with the lower bound removed.
146     *
147     * @return new instance
148     */
149    public Bounds<T> withoutLowerBound() {
150        return new Bounds<T>(null, BoundType.NONE, this.upperBound, this.upperBoundType);
151    }
152
153    /**
154     * Create an instance like this instance but with the upper bound removed.
155     *
156     * @return new instance
157     */
158    public Bounds<T> withoutUpperBound() {
159        return new Bounds<T>(this.lowerBound, this.lowerBoundType, null, BoundType.NONE);
160    }
161
162    /**
163     * Check whether the given value is within the lower bound of this instance.
164     *
165     * @param comparator comparator used to compare values, or null for natural ordering
166     * @param value value to check
167     * @return true if {@code value} is within bounds, false otherwise
168     */
169    public boolean isWithinLowerBound(Comparator<? super T> comparator, T value) {
170        return this.isWithinBound(comparator, value, true, false);
171    }
172
173    /**
174     * Check whether the given value is within the upper bound of this instance.
175     *
176     * @param comparator comparator used to compare values, or null for natural ordering
177     * @param value value to check
178     * @return true if {@code value} is within bounds, false otherwise
179     */
180    public boolean isWithinUpperBound(Comparator<? super T> comparator, T value) {
181        return this.isWithinBound(comparator, value, true, true);
182    }
183
184    /**
185     * Check whether the given value is within the bounds of this instance.
186     *
187     * <p>
188     * Equivalent to:
189     * <blockquote><code>
190     * isWithinLowerBound(comparator, value) &amp;&amp; isWithinUpperBound(comparator, value)
191     * </code></blockquote>
192     *
193     * @param comparator comparator used to compare values, or null for natural ordering
194     * @param value value to check
195     * @return true if {@code value} is within bounds, false otherwise
196     */
197    public boolean isWithinBounds(Comparator<? super T> comparator, T value) {
198        return this.isWithinLowerBound(comparator, value) && this.isWithinUpperBound(comparator, value);
199    }
200
201    /**
202     * Determine whether the given new {@link Bounds} is within of the bounds of this instance.
203     *
204     * @param comparator comparator used to compare values, or null for natural ordering
205     * @param newBounds new bounds
206     * @return true if {@code newBounds} is in range, false otherwise
207     * @throws IllegalArgumentException if {@code newBounds} is null
208     */
209    public boolean isWithinBounds(Comparator<? super T> comparator, Bounds<? extends T> newBounds) {
210        if (newBounds == null)
211            throw new IllegalArgumentException("null newBounds");
212        if (newBounds.lowerBoundType != BoundType.NONE
213          && !this.isWithinBound(comparator, newBounds.lowerBound, newBounds.lowerBoundType.isInclusive(), false))
214            return false;
215        if (newBounds.upperBoundType != BoundType.NONE
216          && !this.isWithinBound(comparator, newBounds.upperBound, newBounds.upperBoundType.isInclusive(), true))
217            return false;
218        return true;
219    }
220
221    /**
222     * Check whether the given value is outside one of the bounds of this instance.
223     *
224     * @param comparator comparator used to compare values, or null for natural ordering
225     * @param value value to check
226     * @param requireInclusive whether value should not be considered included when it is exactly equal to an exclusive bound
227     * @param upper true to check against upper bound, false to check against lower bound
228     * @throws IllegalArgumentException if {@link newBound} is out of range
229     * @throws IllegalArgumentException if {@link newBoundType} is null
230     */
231    @SuppressWarnings("unchecked")
232    private boolean isWithinBound(Comparator<? super T> comparator, T value, boolean requireInclusive, boolean upper) {
233
234        // Check bound type
235        final BoundType boundType = upper ? this.upperBoundType : this.lowerBoundType;
236        if (boundType == BoundType.NONE)
237            return true;
238
239        // Compare value to bound
240        final T bound = upper ? this.upperBound : this.lowerBound;
241        final int diff = comparator != null ? comparator.compare(value, bound) : ((Comparable<T>)value).compareTo(bound);
242
243        // Handle value equal to bound
244        if (diff == 0) {
245            if (boundType == BoundType.INCLUSIVE || !requireInclusive)
246                return true;
247            return false;
248        }
249
250        // Value is either inside or outside bound
251        return upper ? diff < 0 : diff > 0;
252    }
253
254// Object
255
256    @Override
257    public String toString() {
258        return "Bounds[lower(" + this.lowerBoundType + ")=" + this.lowerBound
259          + ",upper(" + this.upperBoundType + ")=" + this.upperBound + "]";
260    }
261
262    @Override
263    public boolean equals(Object obj) {
264        if (obj == this)
265            return true;
266        if (obj == null || obj.getClass() != this.getClass())
267            return false;
268        final Bounds<?> that = (Bounds<?>)obj;
269        return this.lowerBoundType == that.lowerBoundType
270          && this.upperBoundType == that.upperBoundType
271          && (this.lowerBound != null ?  this.lowerBound.equals(that.lowerBound) : that.lowerBound == null)
272          && (this.upperBound != null ?  this.upperBound.equals(that.upperBound) : that.upperBound == null);
273    }
274
275    @Override
276    public int hashCode() {
277        return this.lowerBoundType.hashCode()
278          ^ this.upperBoundType.hashCode()
279          ^ (this.lowerBound != null ?  this.lowerBound.hashCode() : 0)
280          ^ (this.upperBound != null ?  this.upperBound.hashCode() : 0);
281    }
282}
