006package org.jsimpledb.util;
007
008/**
009 * Writes bytes to a buffer.
010 */
011public class ByteWriter {
012
013    private static final int DEFAULT_CAPACITY = 20;
014
015    byte[] buf;
016    int len;
017
018    /**
019     * Default constructor.
020     */
021    public ByteWriter() {
022        this(DEFAULT_CAPACITY);
023    }
024
025    /**
026     * Constructor.
027     *
028     * @param capacity initial capacity of buffer
029     */
030    public ByteWriter(int capacity) {
031        if (capacity < 0)
032            throw new IndexOutOfBoundsException("capacity = " + capacity);
033        this.buf = new byte[capacity];
034    }
035
036    /**
037     * Retrieve all of the bytes that have been written to this instance.
038     *
039     * @return byte content written so far (not necessarily a copy; caller must not modify)
040     */
041    public byte[] getBytes() {
042        if (this.buf.length == this.len)
043            return this.buf;
044        final byte[] result = new byte[this.len];
045        System.arraycopy(this.buf, 0, result, 0, this.len);
046        return result;
047    }
048
049    /**
050     * Retrieve the bytes that have been written to this instance, starting at the given offset.
051     *
052     * @param off offset into written bytes
053     * @return byte content written so far starting at {@code off} (not necessarily a copy; caller must not modify)
054     * @throws IndexOutOfBoundsException if {@code off} or {@code len} is out of bounds
055     */
056    public byte[] getBytes(int off) {
057        return this.getBytes(off, this.len - off);
058    }
059
060    /**
061     * Retrieve a sub-range of the bytes that have been written to this instance.
062     *
063     * @param off offset into written bytes
064     * @param len desired length
065     * @return {@code len} bytes written so far starting from {@code off} (not necessarily a copy; caller must not modify)
066     * @throws IndexOutOfBoundsException if {@code off} or {@code len} is out of bounds
067     */
068    public byte[] getBytes(int off, int len) {
069        if (off == 0 && len == this.len && this.buf.length == this.len)
070            return this.buf;
071        if (off < 0 || off > this.len || len < 0 || off + len < 0 || off + len > this.len)
072            throw new IndexOutOfBoundsException("total length = " + this.len + ", off = " + off + ", len = " + len);
073        final byte[] result = new byte[len];
074        System.arraycopy(this.buf, off, result, 0, len);
075        return result;
076    }
077
078    /**
079     * Write a single byte to this instance.
080     *
081     * @param value byte to write; all but the lower 8 bits are ignored
082     */
083    public void writeByte(int value) {
084        final byte b = (byte)value;
085        this.makeRoom(1);
086        this.buf[this.len++] = b;
087    }
088
089    /**
090     * Read all remaining content from the given {@link ByteReader} and write it to this instance.
091     *
092     * @param reader source for bytes to write
093     */
094    public void write(ByteReader reader) {
095        this.write(reader.buf, reader.off, reader.max - reader.off);
096    }
097
098    /**
099     * Write an array of bytes to this instance.
100     *
101     * @param data bytes to write
102     */
103    public void write(byte[] data) {
104        this.makeRoom(data.length);
105        System.arraycopy(data, 0, this.buf, this.len, data.length);
106        this.len += data.length;
107    }
108
109    /**
110     * Write a sub-range from an array of bytes.
111     *
112     * @param data bytes to write
113     * @param off offset into {@code data}
114     * @param len the number of bytes to write
115     * @throws IndexOutOfBoundsException if {@code off} or {@code len} is out of bounds
116     */
117    public void write(byte[] data, int off, int len) {
118        if (off < 0 || off > data.length || len < 0 || off + len < 0 || off + len > data.length)
119            throw new IndexOutOfBoundsException("data.length = " + data.length + ", off = " + off + ", len = " + len);
120        this.makeRoom(len);
121        System.arraycopy(data, off, this.buf, this.len, len);
122        this.len += len;
123    }
124
125    /**
126     * Get the current buffer length. Returns the same value as {@code #mark}.
127     *
128     * @return number of bytes written so far
129     */
130    public int getLength() {
131        return this.len;
132    }
133
134    /**
135     * Mark current position. Returns the same value as {@code #getLength}.
136     *
137     * @return number of bytes written so far
138     */
139    public int mark() {
140        return this.len;
141    }
142
143    /**
144     * Reset write position to a previously marked position.
145     *
146     * @param mark value previously returned by {@link #mark}
147     * @throws IndexOutOfBoundsException if {@code mark} is out of bounds
148     */
149    public void reset(int mark) {
150        if (mark < 0 || mark > this.len)
151            throw new IndexOutOfBoundsException();
152        this.len = mark;
153    }
154
155    /**
156     * Make room for additional bytes.
157     *
158     * @param amount number of additional bytes
159     */
160    void makeRoom(int amount) {
161        if (this.len + amount > this.buf.length) {
162            final byte[] newbuf = new byte[Math.max(this.len + amount, this.buf.length * 2 + 8)];
163            System.arraycopy(this.buf, 0, newbuf, 0, this.len);
164            this.buf = newbuf;
165        }
166    }
167}
