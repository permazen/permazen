006package org.jsimpledb.util;
007
008import com.google.common.base.Preconditions;
009
010import org.jsimpledb.core.ObjId;
011
012/**
013 * A bi-directional, many-to-many mapping between {@link ObjId}s.
014 *
015 * <p>
016 * Instances can be thought of as containing a set of <i>source, target</i> ordered pairs.
017 * The many-to-many mapping can be efficiently queried and modified from either direction.
018 * </p>
019 *
020 * <p>
021 * Instances of this class are thread-safe.
022 * </p>
023 */
024public class ObjIdBiMultiMap {
025
026    private final Object lock;                      // we always synchronized on this object
027    private final ObjIdBiMultiMap inverse;          // if not null, this is my inverse
028    private /*final*/ ObjIdMap<ObjIdSet> forward;
029    private /*final*/ ObjIdMap<ObjIdSet> reverse;
030
031// Constructors
032
033    /**
034     * Default constructor.
035     */
036    public ObjIdBiMultiMap() {
037        this(0, 0);
038    }
039
040    /**
041     * Constructs an instance with the given initial capacities.
042     *
043     * @param sourceCapacity initial capacity for the number of sources
044     * @param targetCapacity initial capacity for the number of targets
045     * @throws IllegalArgumentException if either value is negative
046     */
047    public ObjIdBiMultiMap(int sourceCapacity, int targetCapacity) {
048        this(null, new ObjIdMap<ObjIdSet>(sourceCapacity), new ObjIdMap<ObjIdSet>(targetCapacity));
049    }
050
051    // Internal constructor
052    private ObjIdBiMultiMap(ObjIdBiMultiMap inverse, ObjIdMap<ObjIdSet> forward, ObjIdMap<ObjIdSet> reverse) {
053        if (inverse != null) {
054            this.lock = inverse.lock;
055            this.inverse = inverse;
056        } else {
057            this.lock = new Object();
058            this.inverse = null;
059        }
060        this.forward = forward;
061        this.reverse = reverse;
062        synchronized (this) { }
063    }
064
065// Public methods
066
067    /**
068     * Get the the number of sources that have one or more associated targets.
069     *
070     * @return the number of sources contained in this instance
071     */
072    public int getNumSources() {
073        synchronized (this.lock) {
074            return this.forward.size();
075        }
076    }

078    /**
079     * Get the the number of targets that have one or more associated sources.
080     *
081     * @return the number of targets contained in this instance
082     */
083    public int getNumTargets() {
084        synchronized (this.lock) {
085            return this.reverse.size();
086        }
087    }
088
089    /**
090     * Get all sources associated with this instance that have one or more associated targets.
091     *
092     * <p>

094     * </p>
095     *
096     * @return all source ID's associated with this instance, possibly empty
097     */
098    public ObjIdSet getSources() {
099        synchronized (this.lock) {
100            return this.forward.keySet().clone();
101        }
102    }
103
104    /**
105     * Get all targets associated with this instance that have one or more associated sources.
106     *
107     * <p>
108     * The returned {@link ObjIdSet} is mutable, but changes to it do not affect this instance.
109     * </p>
110     *
111     * @return all targets ID's associated with this instance, possibly empty
112     */
113    public ObjIdSet getTargets() {
114        synchronized (this.lock) {
115            return this.reverse.keySet().clone();
116        }
117    }

119    /**
120     * Get the sources associated with the given target, if any.
121     *
122     * <p>
123     * The returned {@link ObjIdSet} is mutable, but changes to it do not affect this instance.
124     * </p>
125     *
126     * @param target target ID
127     * @return one or more source ID's associated with {@code target}, or null if there are none
128     * @throws IllegalArgumentException if {@code target} is null
129     */
130    public ObjIdSet getSources(ObjId target) {
131        Preconditions.checkArgument(target != null, "null target");
132        synchronized (this.lock) {
133            final ObjIdSet sources = this.reverse.get(target);
134            return sources != null ? sources.clone() : null;
135        }
136    }

139     * Get the targets associated with the given source, if any.

146     * @return one or more target ID's associated with {@code source}, or null if there are none
147     * @throws IllegalArgumentException if {@code source} is null
148     */
149    public ObjIdSet getTargets(ObjId source) {
150        Preconditions.checkArgument(source != null, "null source");
151        synchronized (this.lock) {
152            final ObjIdSet targets = this.forward.get(source);
153            return targets != null ? targets.clone() : null;
154        }
155    }

157    /**
158     * Determine if this instance has any targets associated with the specified source.
159     *
160     * @param source source ID
161     * @return true if any target ID's are associated with {@code source}, otherwise false
162     * @throws IllegalArgumentException if {@code source} is null
163     */
164    public boolean containsSource(ObjId source) {
165        Preconditions.checkArgument(source != null, "null source");
166        synchronized (this.lock) {
167            return this.forward.containsKey(source);
168        }
169    }
170
171    /**
172     * Determine if this instance has any sources associated with the specified target.
173     *
174     * @param target target ID
175     * @return true if any source ID's are associated with {@code target}, otherwise false
176     * @throws IllegalArgumentException if {@code target} is null
177     */
178    public boolean containsTarget(ObjId target) {
179        Preconditions.checkArgument(target != null, "null target");
180        synchronized (this.lock) {
181            return this.reverse.containsKey(target);
182        }
183    }
184
185    /**
186     * Add an association.
187     *
188     * @param source source ID
189     * @param target target ID
190     * @return true if association was added, false if the association already existed
191     * @throws IllegalArgumentException if {@code source} or {@code target} is null
192     */
193    public boolean add(ObjId source, ObjId target) {
194        Preconditions.checkArgument(source != null, "null source");
195        Preconditions.checkArgument(target != null, "null target");
196        synchronized (this.lock) {
197            if (!ObjIdBiMultiMap.add(this.forward, source, target))
198                return false;
199            ObjIdBiMultiMap.add(this.reverse, target, source);
200            return true;
201        }
202    }
203
204    /**
205     * Add multiple associations with a given source.
206     *
207     * @param source source ID
208     * @param targets target ID's
209     * @return true if any association was added, false if all associations already existed
210     * @throws IllegalArgumentException if {@code source}, {@code targets}, or any target in the iteration is null
211     */
212    public boolean addAll(ObjId source, Iterable<? extends ObjId> targets) {
213
214        // Sanity check
215        Preconditions.checkArgument(source != null, "null source");
216        Preconditions.checkArgument(targets != null, "null targets");
217
218        // Gather targets
219        final ObjIdSet addedTargets = ObjIdBiMultiMap.gather(targets);
220        if (addedTargets.isEmpty())
221            return false;
222
223        // Update maps
224        boolean result = false;
225        synchronized (this.lock) {
226            final ObjIdSet targetSet = this.forward.get(source);
227            if (targetSet != null) {
228                assert !targetSet.isEmpty();
229                if (targetSet.size() >= addedTargets.size())
230                    result = targetSet.addAll(addedTargets);
231                else {
232                    addedTargets.addAll(targetSet);
233                    this.forward.put(source, addedTargets);
234                    result = true;
235                }
236            } else {
237                this.forward.put(source, addedTargets);
238                result = true;
239            }
240            for (ObjId target : addedTargets)
241                ObjIdBiMultiMap.add(this.reverse, target, source);
242        }
243
244        // Done
245        return result;
246    }
247
248    /**
249     * Remove an association.
250     *
251     * @param source source ID
252     * @param target target ID
253     * @return true if association was removed, false if the association did not exist
254     * @throws IllegalArgumentException if {@code source} or {@code target} is null
255     */
256    public boolean remove(ObjId source, ObjId target) {
257        Preconditions.checkArgument(source != null, "null source");
258        Preconditions.checkArgument(target != null, "null target");
259        synchronized (this.lock) {
260            if (!ObjIdBiMultiMap.remove(this.forward, source, target))
261                return false;
262            ObjIdBiMultiMap.remove(this.reverse, target, source);
263            return true;
264        }
265    }
266
267    /**
268     * Remove multiple associations with a given source.
269     *
270     * @param source source ID
271     * @param targets target ID's
272     * @return true if any association was removed, false if none of the specified associations existed
273     * @throws IllegalArgumentException if {@code source}, {@code targets}, or any target in the iteration is null
274     */
275    public boolean removeAll(ObjId source, Iterable<? extends ObjId> targets) {
276
277        // Sanity check
278        Preconditions.checkArgument(source != null, "null source");
279        Preconditions.checkArgument(targets != null, "null targets");
280
281        // Gather targets
282        final ObjIdSet removedTargets = ObjIdBiMultiMap.gather(targets);
283        if (removedTargets.isEmpty())
284            return false;
285
286        // Update maps
287        boolean result = false;
288        synchronized (this.lock) {
289            final ObjIdSet targetSet = this.forward.get(source);
290            if (targetSet == null)
291                return false;
292            assert !targetSet.isEmpty();
293            for (ObjId target : removedTargets) {
294                if (targetSet.remove(target)) {
295                    final boolean removed = ObjIdBiMultiMap.remove(this.reverse, target, source);
296                    assert removed;
297                    result = true;
298                }
299            }
300            if (targetSet.isEmpty())
301                this.forward.remove(source);
302        }
303
304        // Done
305        return result;
306    }
307
308    /**
309     * Remove all associations involving the specified source.
310     *
311     * @param source source ID
312     * @return true if any {@code source} association(s) were removed, false if {@code source} had no target associations
313     * @throws IllegalArgumentException if {@code source} is null
314     */
315    public boolean removeSource(ObjId source) {
316        Preconditions.checkArgument(source != null, "null source");
317        synchronized (this.lock) {
318            final ObjIdSet targets = this.forward.remove(source);
319            if (targets == null)
320                return false;
321            assert !targets.isEmpty();
322            for (ObjId target : targets) {
323                final ObjIdSet sources = this.reverse.get(target);
324                sources.remove(source);
325                if (sources.isEmpty())
326                    this.reverse.remove(target);
327            }
328        }
329        return true;
330    }
331
332    /**
333     * Remove all associations involving the specified target.
334     *
335     * @param target target ID
336     * @return true if any {@code target} association(s) were removed, false if {@code target} had no source associations
337     * @throws IllegalArgumentException if {@code target} is null
338     */
339    public boolean removeTarget(ObjId target) {
340        return this.inverse().removeSource(target);
341    }
342
343    /**
344     * Clear this instance.
345     */
346    public void clear() {
347        synchronized (this.lock) {
348            this.forward.clear();
349            this.reverse.clear();
350        }
351    }
352
353    /**
354     * Get an inverse view backed by this instance.
355     *
356     * <p>
357     * The returned {@link ObjIdBiMultiMap} is a <i>view</i> in which sources become targets and vice-versa;
358     * any changes are reflected back in this instance.
359     * </p>
360     *
361     * <p>
362     * This method is efficient, requiring only constant time.
363     * </p>
364     *
365     * @return inverse view of this instance
366     */
367    public ObjIdBiMultiMap inverse() {
368        return this.inverse != null ? this.inverse : new ObjIdBiMultiMap(this, this.reverse, this.forward);
369    }

383    /**
384     * Compare for equality.
385     */
386    @Override
387    public boolean equals(Object obj) {
388        if (obj == this)
389            return true;
390        if (obj == null || obj.getClass() != this.getClass())
391            return false;
392        final ObjIdBiMultiMap that = (ObjIdBiMultiMap)obj;
393        final ObjIdMap<ObjIdSet> thisForward;
394        final ObjIdMap<ObjIdSet> thatForward;
395        synchronized (this.lock) {
396            thisForward = ObjIdBiMultiMap.deepClone(this.forward);
397        }
398        synchronized (that.lock) {
399            thatForward = ObjIdBiMultiMap.deepClone(that.forward);
400        }
401        return thisForward.equals(thatForward);
402    }

416    /**
417     * Clone this instance.
418     */
419    @Override
420    @SuppressWarnings("unchecked")
421    public ObjIdBiMultiMap clone() {
422        final ObjIdBiMultiMap clone;
423        try {
424            clone = (ObjIdBiMultiMap)super.clone();
425        } catch (CloneNotSupportedException e) {
426            throw new RuntimeException(e);
427        }
428        synchronized (this.lock) {
429            clone.forward = ObjIdBiMultiMap.deepClone(clone.forward);
430            clone.reverse = ObjIdBiMultiMap.deepClone(clone.reverse);
431        }
432        return clone;
433    }

435// Internal methods
436
437    private static ObjIdSet gather(Iterable<? extends ObjId> ids) {
438        final ObjIdSet set = new ObjIdSet();
439        for (ObjId id : ids) {
440            Preconditions.checkArgument(id != null, "encountered null ObjId in iteration");
441            set.add(id);
442        }
443        return set;
444    }
445
446    private static boolean add(ObjIdMap<ObjIdSet> map, ObjId source, ObjId target) {
447        ObjIdSet set = map.get(source);
448        if (set == null) {
449            set = new ObjIdSet();
450            map.put(source, set);
451        }
452        return set.add(target);
453    }
454
455    private static boolean remove(ObjIdMap<ObjIdSet> map, ObjId source, ObjId target) {
456        final ObjIdSet set = map.get(source);
457        if (set == null || !set.remove(target))
458            return false;
459        if (set.isEmpty())
460            map.remove(source);
461        return true;
462    }
463
464    private static ObjIdMap<ObjIdSet> deepClone(ObjIdMap<ObjIdSet> map) {
465        final ObjIdMap<ObjIdSet> clone = map.clone();
466        final int numSlots = map.getKeys().length;
467        for (int i = 0; i < numSlots; i++) {
468            final ObjIdSet set = map.getValue(i);
469            if (set != null)
470                map.setValue(i, set.clone());
471        }
472        return clone;
473    }
