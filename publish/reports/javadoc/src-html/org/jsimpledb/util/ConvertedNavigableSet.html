006package org.jsimpledb.util;
007
008import com.google.common.base.Converter;
009import com.google.common.collect.Iterators;
010
011import java.util.Comparator;
012import java.util.Iterator;
013import java.util.NavigableSet;
014import java.util.NoSuchElementException;
015
016/**
017 * Provides a transformed view of a wrapped {@link NavigableSet} using a strictly invertable {@link Converter}.
018 *
019 * @param <E> element type of this set
020 * @param <W> element type of the wrapped set
021 */
022public class ConvertedNavigableSet<E, W> extends AbstractNavigableSet<E> {
023
024    private final NavigableSet<W> set;
025    private final Converter<E, W> elementConverter;
026
027    /**
028     * Constructor.
029     *
030     * @param set wrapped set
031     * @param elementConverter element converter
032     * @throws IllegalArgumentException if either parameter is null
033     */
034    public ConvertedNavigableSet(NavigableSet<W> set, Converter<E, W> elementConverter) {
035        this(set, elementConverter, new Bounds<E>());
036    }
037
038    /**
039     * Internal constructor.
040     *
041     * @param set wrapped set
042     * @param elementConverter element converter
043     * @throws IllegalArgumentException if any parameter is null
044     */
045    ConvertedNavigableSet(NavigableSet<W> set, Converter<E, W> elementConverter, Bounds<E> bounds) {
046        super(bounds);
047        if (set == null)
048            throw new IllegalArgumentException("null set");
049        if (elementConverter == null)
050            throw new IllegalArgumentException("null elementConverter");
051        this.set = set;
052        this.elementConverter = elementConverter;
053    }
054
055    public Converter<E, W> getConverter() {
056        return this.elementConverter;
057    }
058
059    @Override
060    public Comparator<? super E> comparator() {
061        return new ConvertedComparator<E, W>(this.set.comparator(), this.elementConverter);
062    }
063
064    @Override
065    @SuppressWarnings("unchecked")
066    public boolean contains(Object obj) {
067        W wobj = null;
068        if (obj != null) {
069            try {
070                wobj = this.elementConverter.convert((E)obj);
071            } catch (ClassCastException e) {
072                return false;
073            }
074        }
075        return this.set.contains(wobj);
076    }
077
078    @Override
079    public Iterator<E> iterator() {
080        return Iterators.transform(this.set.iterator(), this.elementConverter.reverse());
081    }
082
083    @Override
084    public boolean add(E obj) {
085        return this.set.add(obj != null ? this.elementConverter.convert(obj) : null);
086    }
087
088    @Override
089    @SuppressWarnings("unchecked")
090    public boolean remove(Object obj) {
091        W wobj = null;
092        if (obj != null) {
093            try {
094                wobj = this.elementConverter.convert((E)obj);
095            } catch (ClassCastException e) {
096                return false;
097            }
098        }
099        return this.set.remove(wobj);
100    }
101
102    @Override
103    public void clear() {
104        this.set.clear();
105    }
106
107    @Override
108    public boolean isEmpty() {
109        return this.set.isEmpty();
110    }
111
112    @Override
113    public int size() {
114        return this.set.size();
115    }
116
117    @Override
118    protected E searchBelow(E elem, boolean inclusive) {
119        try {
120            return super.searchBelow(elem, inclusive);
121        } catch (IllegalArgumentException e) {                      // handle case where elem is out of bounds
122            final E last;
123            try {
124                last = this.last();
125            } catch (NoSuchElementException e2) {
126                return null;
127            }
128            return this.getComparator(false).compare(elem, last) > 0 ? last : null;
129        }
130    }
131
132    @Override
133    protected E searchAbove(E elem, boolean inclusive) {
134        try {
135            return super.searchAbove(elem, inclusive);
136        } catch (IllegalArgumentException e) {                      // handle case where elem is out of bounds
137            final E first;
138            try {
139                first = this.first();
140            } catch (NoSuchElementException e2) {
141                return null;
142            }
143            return this.getComparator(false).compare(elem, first) < 0 ? first : null;
144        }
145    }
146
147    @Override
148    protected NavigableSet<E> createSubSet(boolean reverse, Bounds<E> newBounds) {
149        final E lower = newBounds.getLowerBound();
150        final E upper = newBounds.getUpperBound();
151        final W wlower = newBounds.getLowerBoundType() != BoundType.NONE && lower != null ?
152          this.elementConverter.convert(lower) : null;
153        final W wupper = newBounds.getUpperBoundType() != BoundType.NONE && upper != null ?
154          this.elementConverter.convert(upper) : null;
155        NavigableSet<W> subSet = reverse ? this.set.descendingSet() : this.set;
156        if (newBounds.getLowerBoundType() != BoundType.NONE && newBounds.getUpperBoundType() != BoundType.NONE) {
157            subSet = subSet.subSet(
158              wlower, newBounds.getLowerBoundType().isInclusive(),
159              wupper, newBounds.getUpperBoundType().isInclusive());
160        } else if (newBounds.getLowerBoundType() != BoundType.NONE)
161            subSet = subSet.tailSet(wlower, newBounds.getLowerBoundType().isInclusive());
162        else if (newBounds.getUpperBoundType() != BoundType.NONE)
163            subSet = subSet.headSet(wupper, newBounds.getUpperBoundType().isInclusive());
164        return new ConvertedNavigableSet<E, W>(subSet, this.elementConverter, newBounds);
165    }
166}
