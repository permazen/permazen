006package org.jsimpledb.util;
007
008import com.google.common.base.Converter;
009import com.google.common.collect.Iterators;
010
011import java.util.AbstractList;
012import java.util.ArrayList;
013import java.util.Collection;
014import java.util.Iterator;
015import java.util.List;
016
017/**
018 * Provides a transformed view of a wrapped {@link List} using a strictly invertable {@link Converter}.
019 *
020 * @param <E> element type of this list
021 * @param <W> element type of the wrapped list
022 */
023public class ConvertedList<E, W> extends AbstractList<E> {
024
025    private final List<W> list;
026    private final Converter<E, W> converter;
027
028    /**
029     * Constructor.
030     *
031     * @param list wrapped list
032     * @param converter element converter
033     * @throws IllegalArgumentException if either parameter is null
034     */
035    public ConvertedList(List<W> list, Converter<E, W> converter) {
036        if (list == null)
037            throw new IllegalArgumentException("null list");
038        if (converter == null)
039            throw new IllegalArgumentException("null converter");
040        this.list = list;
041        this.converter = converter;
042    }
043
044    public Converter<E, W> getConverter() {
045        return this.converter;
046    }
047
048    @Override
049    public E get(int index) {
050        final W value = this.list.get(index);
051        return value != null ? this.converter.reverse().convert(value) : null;
052    }
053
054    @Override
055    public int size() {
056        return this.list.size();
057    }
058
059    @Override
060    public boolean isEmpty() {
061        return this.list.isEmpty();
062    }
063
064    @Override
065    public E set(int index, E elem) {
066        final W welem = elem != null ? this.converter.convert(elem) : null;
067        final W prev = this.list.set(index, welem);
068        return prev != null ? this.converter.reverse().convert(prev) : null;
069    }
070
071    @Override
072    public void add(int index, E elem) {
073        final W welem = elem != null ? this.converter.convert(elem) : null;
074        this.list.add(index, welem);
075    }
076
077    @Override
078    public boolean addAll(int index, Collection<? extends E> elems) {
079        final ArrayList<W> welems = new ArrayList<>();
080        for (E elem : elems)
081            welems.add(elem != null ? this.converter.convert(elem) : null);
082        return this.list.addAll(index, welems);
083    }
084
085    @Override
086    @SuppressWarnings("unchecked")
087    public boolean contains(Object obj) {
088        W wobj = null;
089        if (obj != null) {
090            try {
091                wobj = this.converter.convert((E)obj);
092            } catch (ClassCastException e) {
093                return false;
094            }
095        }
096        return this.list.contains(wobj);
097    }
098
099    @Override
100    @SuppressWarnings("unchecked")
101    public int indexOf(Object obj) {
102        W wobj = null;
103        if (obj != null) {
104            try {
105                wobj = this.converter.convert((E)obj);
106            } catch (ClassCastException e) {
107                return -1;
108            }
109        }
110        return this.list.indexOf(wobj);
111    }
112
113    @Override
114    @SuppressWarnings("unchecked")
115    public int lastIndexOf(Object obj) {
116        W wobj = null;
117        if (obj != null) {
118            try {
119                wobj = this.converter.convert((E)obj);
120            } catch (ClassCastException e) {
121                return -1;
122            }
123        }
124        return this.list.lastIndexOf(wobj);
125    }
126
127    @Override
128    public void clear() {
129        this.list.clear();
130    }
131
132    @Override
133    public E remove(int index) {
134        final W welem = this.list.remove(index);
135        return welem != null ? this.converter.reverse().convert(welem) : null;
136    }
137
138    @Override
139    public Iterator<E> iterator() {
140        return Iterators.transform(this.list.iterator(), this.converter.reverse());
141    }
142
143    @Override
144    public List<E> subList(int min, int max) {
145        return new ConvertedList<E, W>(this.list.subList(min, max), this.converter);
146    }
147
148    @Override
149    protected void removeRange(int min, int max) {
150        this.list.subList(min, max).clear();
151    }
152}
