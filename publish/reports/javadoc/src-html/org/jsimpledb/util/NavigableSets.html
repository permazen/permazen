008package org.jsimpledb.util;
009
010import java.util.Arrays;
011import java.util.Comparator;
012import java.util.NavigableSet;
013
014/**
015 * Utility methods relating to {@link NavigableSet}.
016 */
017public final class NavigableSets {
018
019    private NavigableSets() {
020    }
021
022    /**
023     * Create a read-only view of the intersection of two or more {@link NavigableSet}s that have a consistent sort order.
024     * The {@linkplain NavigableSet#comparator Comparator}s of the sets must all be {@linkplain Comparator#equals equal}
025     * (or else all null, for natural ordering).
026     *
027     * <p>
028     * The returned intersection interates efficiently: a complete iteration takes time <i>O(N * M)</i> where
029     * <i>N</i> is the size of the smallest set, and <i>M</i> is the number of sets.
030     * </p>
031     *
032     * @param sets the sets to intersect
033     * @throws IllegalArgumentException if the {@code sets} is empty
034     * @throws IllegalArgumentException if the {@code sets} do not have equal {@link Comparator}s
035     */
036    public static <E> NavigableSet<E> intersection(Iterable<? extends NavigableSet<E>> sets) {
037        return new IntersectionNavigableSet<E>(sets);
038    }
039
040    /**
041     * Create a read-only view of the intersection of two or more {@link NavigableSet}s that have a consistent sort order.
042     * The {@linkplain NavigableSet#comparator Comparator}s of the sets must all be {@linkplain Comparator#equals equal}
043     * (or else all null, for natural ordering).
044     *
045     * <p>
046     * The returned intersection interates efficiently: a complete iteration takes time <i>O(N * M)</i> where
047     * <i>N</i> is the size of the smallest set, and <i>M</i> is the number of sets.
048     * </p>
049     *
050     * @param sets the sets to intersect
051     * @throws IllegalArgumentException if the {@code sets} is empty
052     * @throws IllegalArgumentException if the {@code sets} do not have equal {@link Comparator}s
053     */
054    @SafeVarargs
055    @SuppressWarnings("varargs")
056    public static <E> NavigableSet<E> intersection(NavigableSet<E>... sets) {
057        return new IntersectionNavigableSet<E>(Arrays.asList(sets));
058    }
059
060    /**
061     * Create a read-only view of the union of two or more {@link NavigableSet}s that have a consistent sort order.
062     * The {@linkplain NavigableSet#comparator Comparator}s of the sets must all be {@linkplain Comparator#equals equal}
063     * (or else all null, for natural ordering).
064     *
065     * @param sets the sets to union
066     * @throws IllegalArgumentException if the {@code sets} is empty
067     * @throws IllegalArgumentException if the {@code sets} do not have equal {@link Comparator}s
068     */
069    @SafeVarargs
070    @SuppressWarnings("varargs")
071    public static <E> NavigableSet<E> union(NavigableSet<E>... sets) {
072        return new UnionNavigableSet<E>(Arrays.asList(sets));
073    }
074
075    /**
076     * Create a read-only view of the union of two or more {@link NavigableSet}s that have a consistent sort order.
077     * The {@linkplain NavigableSet#comparator Comparator}s of the sets must all be {@linkplain Comparator#equals equal}
078     * (or else all null, for natural ordering).
079     *
080     * @param sets the sets to union
081     * @throws IllegalArgumentException if the {@code sets} is empty
082     * @throws IllegalArgumentException if the {@code sets} do not have equal {@link Comparator}s
083     */
084    public static <E> NavigableSet<E> union(Iterable<? extends NavigableSet<E>> sets) {
085        return new UnionNavigableSet<E>(sets);
086    }
087
088    /**
089     * Create a read-only view of the difference of two {@link NavigableSet}s that have a consistent sort order.
090     * That is, a set containing all elements contained in the first set but not the second.
091     * The {@linkplain NavigableSet#comparator Comparator}s of the sets must all be {@linkplain Comparator#equals equal}
092     * (or else all null, for natural ordering).
093     *
094     * @param set1 original set
095     * @param set2 set of elements to exclude from {@code set1}
096     * @throws IllegalArgumentException if the {@code set1} or {@code set2} is empty
097     * @throws IllegalArgumentException if the {@code set1} and {@code set2} do not have equal {@link Comparator}s
098     */
099    public static <E> NavigableSet<E> difference(NavigableSet<E> set1, NavigableSet<E> set2) {
100        return new DifferenceNavigableSet<E>(set1, set2);
101    }
102
103    /**
104     * Create a read-only view of the symmetric difference of two {@link NavigableSet}s that have a consistent sort order.
105     * That is, the set containing all elements contained in the first set or the second set, but not in both sets.
106     * The {@linkplain NavigableSet#comparator Comparator}s of the sets must all be {@linkplain Comparator#equals equal}
107     * (or else all null, for natural ordering).
108     *
109     * @param set1 first set
110     * @param set2 second set
111     * @throws IllegalArgumentException if the {@code set1} or {@code set2} is empty
112     * @throws IllegalArgumentException if the {@code set1} and {@code set2} do not have equal {@link Comparator}s
113     */
114    public static <E> NavigableSet<E> symmetricDifference(NavigableSet<E> set1, NavigableSet<E> set2) {
115        return NavigableSets.difference(NavigableSets.union(set1, set2), NavigableSets.intersection(set1, set2));
116    }
117
118    /**
119     * Create a {@link NavigableSet} containing a single element and natural ordering.
120     *
121     * @param value singleton value
122     * @throws IllegalArgumentException if {@code value} is null or
123     *  {@code value}'s class has no natural ordering (i.e., does not implement {@link Comparable})
124     */
125    public static <E> NavigableSet<E> singleton(E value) {
126        return NavigableSets.<E>singleton(null, value);
127    }
128
129    /**
130     * Create a {@link NavigableSet} containing a single element.
131     *
132     * @param comparator comparator, or null for natural ordering
133     * @param value singleton value (possibly null)
134     * @throws IllegalArgumentException if {@code comparator} is null and either {@code value} is null or
135     *  {@code value}'s class has no natural ordering (i.e., does not implement {@link Comparable})
136     */
137    public static <E> NavigableSet<E> singleton(Comparator<? super E> comparator, E value) {
138        return new SingletonNavigableSet<E>(comparator, value);
139    }
140
141    /**
142     * Create an empty {@link NavigableSet}.
143     * The returned set's {@link NavigableSet#comparator comparator()} method will return null.
144     */
145    public static <E> NavigableSet<E> empty() {
146        return NavigableSets.<E>empty(null);
147    }
148
149    /**
150     * Create an empty {@link NavigableSet} with a specified {@link Comparator}.
151     *
152     * @param comparator comparator, or null for natural ordering
153     */
154    public static <E> NavigableSet<E> empty(Comparator<? super E> comparator) {
155        return new EmptyNavigableSet<E>(comparator);
156    }
157}
