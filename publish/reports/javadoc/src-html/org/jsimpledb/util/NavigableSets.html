006package org.jsimpledb.util;
007
008import com.google.common.base.Preconditions;
009import com.google.common.base.Predicates;
010import com.google.common.collect.Iterables;
011
012import java.util.Arrays;
013import java.util.Collections;
014import java.util.Comparator;
015import java.util.NavigableSet;
016
017/**
018 * Utility methods relating to {@link NavigableSet}.
019 */
020public final class NavigableSets {
021
022    private NavigableSets() {
023    }
024
025    /**
026     * Create a read-only view of the intersection of two or more {@link NavigableSet}s that have a consistent sort order.
027     * The {@linkplain NavigableSet#comparator Comparator}s of the sets must all be {@linkplain Comparator#equals equal}
028     * (or else all null, for natural ordering).
029     *
030     * <p>
031     * The returned intersection iterates efficiently: a complete iteration requires <i>O(N * M)</i> queries, where
032     * <i>N</i> is the size of the smallest set, and <i>M</i> is the number of sets.
033     * </p>
034     *
035     * @param sets the sets to intersect
036     * @param <E> element type
037     * @return the intersection of all {@code sets}
038     * @throws IllegalArgumentException if the {@code sets} do not have equal {@link Comparator}s
039     * @throws IllegalArgumentException if {@code sets} is null
040     */
041    public static <E> NavigableSet<E> intersection(Iterable<? extends NavigableSet<E>> sets) {
042        Preconditions.checkArgument(sets != null, "null sets");
043        if (Iterables.find(sets, Predicates.instanceOf(EmptyNavigableSet.class), null) != null)
044            return new EmptyNavigableSet<E>(null);
045        return new IntersectionNavigableSet<E>(sets);
046    }
047
048    /**
049     * Create a read-only view of the intersection of two or more {@link NavigableSet}s that have a consistent sort order.
050     * The {@linkplain NavigableSet#comparator Comparator}s of the sets must all be {@linkplain Comparator#equals equal}
051     * (or else all null, for natural ordering).
052     *
053     * <p>
054     * The returned intersection iterates efficiently: a complete iteration takes time <i>O(N * M)</i> where
055     * <i>N</i> is the size of the smallest set, and <i>M</i> is the number of sets.
056     * </p>
057     *
058     * @param sets the sets to intersect
059     * @param <E> element type
060     * @return the intersection of all {@code sets}
061     * @throws IllegalArgumentException if the {@code sets} do not have equal {@link Comparator}s
062     * @throws IllegalArgumentException if {@code sets} is null
063     */
064    @SafeVarargs
065    @SuppressWarnings("varargs")
066    public static <E> NavigableSet<E> intersection(NavigableSet<E>... sets) {
067        Preconditions.checkArgument(sets != null, "null sets");
068        return new IntersectionNavigableSet<E>(Arrays.asList(sets));
069    }
070
071    /**
072     * Create a read-only view of the union of two or more {@link NavigableSet}s that have a consistent sort order.
073     * The {@linkplain NavigableSet#comparator Comparator}s of the sets must all be {@linkplain Comparator#equals equal}
074     * (or else all null, for natural ordering).
075     *
076     * @param sets the sets to union
077     * @param <E> element type
078     * @return the union of all {@code sets}
079     * @throws IllegalArgumentException if the {@code sets} do not have equal {@link Comparator}s
080     * @throws IllegalArgumentException if {@code sets} is null
081     */
082    @SafeVarargs
083    @SuppressWarnings("varargs")
084    public static <E> NavigableSet<E> union(NavigableSet<E>... sets) {
085        Preconditions.checkArgument(sets != null, "null sets");
086        return new UnionNavigableSet<E>(Arrays.asList(sets));
087    }
088
089    /**
090     * Create a read-only view of the union of two or more {@link NavigableSet}s that have a consistent sort order.
091     * The {@linkplain NavigableSet#comparator Comparator}s of the sets must all be {@linkplain Comparator#equals equal}
092     * (or else all null, for natural ordering).
093     *
094     * @param sets the sets to union
095     * @param <E> element type
096     * @return the union of all {@code sets}
097     * @throws IllegalArgumentException if the {@code sets} do not have equal {@link Comparator}s
098     * @throws IllegalArgumentException if {@code sets} is null
099     */
100    public static <E> NavigableSet<E> union(Iterable<? extends NavigableSet<E>> sets) {
101        Preconditions.checkArgument(sets != null, "null sets");
102        if (!(sets = Iterables.filter(sets, Predicates.not(Predicates.instanceOf(EmptyNavigableSet.class)))).iterator().hasNext())
103            return new EmptyNavigableSet<E>(null);
104        return new UnionNavigableSet<E>(sets);
105    }
106
107    /**
108     * Create a read-only view of the difference of two {@link NavigableSet}s that have a consistent sort order.
109     * That is, a set containing all elements contained in the first set but not the second.
110     * The {@linkplain NavigableSet#comparator Comparator}s of the sets must all be {@linkplain Comparator#equals equal}
111     * (or else all null, for natural ordering).
112     *
113     * @param set1 original set
114     * @param set2 set of elements to exclude from {@code set1}
115     * @param <E> element type
116     * @return the difference of {@code set1} and {@code set2}
117     * @throws IllegalArgumentException if the {@code set1} and {@code set2} do not have equal {@link Comparator}s
118     * @throws IllegalArgumentException if either parameter is is null
119     */
120    public static <E> NavigableSet<E> difference(NavigableSet<E> set1, NavigableSet<E> set2) {
121        Preconditions.checkArgument(set1 != null, "null set1");
122        Preconditions.checkArgument(set2 != null, "null set2");
123        if (set1 instanceof EmptyNavigableSet || set2 instanceof EmptyNavigableSet)
124            return set1;
125        return new DifferenceNavigableSet<E>(set1, set2);
126    }
127
128    /**
129     * Create a read-only view of the symmetric difference of two {@link NavigableSet}s that have a consistent sort order.
130     * That is, the set containing all elements contained in the first set or the second set, but not in both sets.
131     * The {@linkplain NavigableSet#comparator Comparator}s of the sets must all be {@linkplain Comparator#equals equal}
132     * (or else all null, for natural ordering).
133     *
134     * @param set1 first set
135     * @param set2 second set
136     * @param <E> element type
137     * @return the symmetric difference of {@code set1} and {@code set2}
138     * @throws IllegalArgumentException if the {@code set1} and {@code set2} do not have equal {@link Comparator}s
139     * @throws IllegalArgumentException if either parameter is is null
140     */
141    public static <E> NavigableSet<E> symmetricDifference(NavigableSet<E> set1, NavigableSet<E> set2) {
142        Preconditions.checkArgument(set1 != null, "null set1");
143        Preconditions.checkArgument(set2 != null, "null set2");
144        if (set1 instanceof EmptyNavigableSet)
145            return set2;
146        if (set2 instanceof EmptyNavigableSet)
147            return set1;
148        return NavigableSets.difference(NavigableSets.union(set1, set2), NavigableSets.intersection(set1, set2));
149    }
150
151    /**
152     * Create a {@link NavigableSet} containing a single element and natural ordering.
153     *
154     * @param value singleton value
155     * @param <E> element type
156     * @return singleton set
157     * @throws IllegalArgumentException if {@code value} is null or
158     *  {@code value}'s class has no natural ordering (i.e., does not implement {@link Comparable})
159     */
160    public static <E> NavigableSet<E> singleton(E value) {
161        return NavigableSets.<E>singleton(null, value);
162    }
163
164    /**
165     * Create a {@link NavigableSet} containing a single element.
166     *
167     * @param comparator comparator, or null for natural ordering
168     * @param value singleton value (possibly null)
169     * @param <E> element type
170     * @return singleton set
171     * @throws IllegalArgumentException if {@code comparator} is null and either {@code value} is null or
172     *  {@code value}'s class has no natural ordering (i.e., does not implement {@link Comparable})
173     */
174    public static <E> NavigableSet<E> singleton(Comparator<? super E> comparator, E value) {
175        return new SingletonNavigableSet<E>(comparator, value);
176    }
177
178    /**
179     * Create an empty {@link NavigableSet}.
180     * The returned set's {@link NavigableSet#comparator comparator()} method will return null.
181     *
182     * @param <E> element type
183     * @return empty set
184     */
185    public static <E> NavigableSet<E> empty() {
186        return NavigableSets.<E>empty(null);
187    }
188
189    /**
190     * Create an empty {@link NavigableSet} with a specified {@link Comparator}.
191     *
192     * @param comparator comparator, or null for natural ordering
193     * @param <E> element type
194     * @return empty set
195     */
196    public static <E> NavigableSet<E> empty(Comparator<? super E> comparator) {
197        return new EmptyNavigableSet<E>(comparator);
198    }
199
200    /**
201     * Get a non-null {@link Comparator} that sorts consistently with, and optionally reversed from, the given {@link Comparator}.
202     *
203     * @param comparator comparator, or null for natural ordering
204     * @param reversed whether to return a reversed {@link Comparator}
205     * @param <T> compared type
206     * @return a non-null {@link Comparator}
207     */
208    static <T> Comparator<T> getComparator(Comparator<T> comparator, boolean reversed) {
209        return comparator != null ?
210          (reversed ? Collections.<T>reverseOrder(comparator) : comparator) :
211          (reversed ? Collections.<T>reverseOrder() : new NaturalComparator<T>());
212    }
213
214// NaturalComparator
215
216    private static class NaturalComparator<T> implements Comparator<T> {
217
218        @Override
219        @SuppressWarnings("unchecked")
220        public int compare(T e1, T e2) {
221            return ((Comparable<T>)e1).compareTo(e2);
222        }
223    }
224}
