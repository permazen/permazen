006package org.jsimpledb.util;
007
008import java.util.AbstractSet;
009import java.util.ArrayList;
010import java.util.Iterator;
011import java.util.Set;
012
013/**
014 * Support superclass for {@link Set} implementations for which calculating {@link #size} requires
015 * an iteration through all of the set's elements to count them.
016 *
017 * <p>
018 * Superclass methods in {@link AbstractSet} that rely on {@link #size} are overridden with alternate
019 * implementations that avoid the use of {@link #size} when possible.
020 * </p>
021 *
022 * <p>
023 * For a read-only implementation, subclasses should implement {@link #contains contains()} and {@link #iterator iterator()}.
024 * For a mutable implementation, subclasses should also implement {@link #add add()}, {@link #remove remove()},
025 * and {@link #clear}, and make the {@link #iterator} mutable.
026 * </p>
027 */
028public abstract class AbstractIterationSet<E> extends AbstractSet<E> {
029
030    protected AbstractIterationSet() {
031    }
032
033    /**
034     * Calculate size.
035     *
036     * <p>
037     * The implementation in {@link AbstractIterationSet} iterates through all of the elements.
038     * </p>
039     */
040    @Override
041    public int size() {
042        int count = 0;
043        for (E elem : this)
044            count++;
045        return count;
046    }
047
048    /**
049     * Overridden in {@link AbstractIterationSet} to minimize the use of {@link #size}.
050     */
051    @Override
052    public boolean equals(Object obj) {
053        if (obj == this)
054            return true;
055        if (!(obj instanceof Set))
056            return false;
057        final Set<?> that = (Set<?>)obj;
058        final Iterator<?> i1 = this.iterator();
059        final Iterator<?> i2 = that.iterator();
060        while (true) {
061            final boolean hasNext1 = i1.hasNext();
062            final boolean hasNext2 = i2.hasNext();
063            if (!hasNext1 && !hasNext2)
064                return true;
065            if (!hasNext1 || !hasNext2)
066                return false;
067            if (!this.contains(i2.next()))
068                return false;
069            i1.next();
070        }
071    }
072
073    // This is here to silence a checkstyle warning
074    @Override
075    public int hashCode() {
076        return super.hashCode();
077    }
078
079    /**
080     * Overridden in {@link AbstractIterationSet} to minimize the use of {@link #size}.
081     */
082    @Override
083    public boolean isEmpty() {
084        return !this.iterator().hasNext();
085    }
086
087    /**
088     * Overridden in {@link AbstractIterationSet} to minimize the use of {@link #size}.
089     */
090    @Override
091    public Object[] toArray() {
092        final ArrayList<E> list = new ArrayList<>();
093        for (E elem : this)
094            list.add(elem);
095       return list.toArray();
096    }
097
098    /**
099     * Overridden in {@link AbstractIterationSet} to minimize the use of {@link #size}.
100     */
101    @Override
102    public <T> T[] toArray(T[] array) {
103        final ArrayList<E> list = new ArrayList<>();
104        for (E elem : this)
105            list.add(elem);
106       return list.toArray(array);
107    }
108}
