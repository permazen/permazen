006package org.jsimpledb.util;
007
008import com.google.common.base.Converter;
009import com.google.common.collect.Iterators;
010
011import java.util.Iterator;
012import java.util.Set;
013
014/**
015 * Provides a transformed view of a wrapped {@link Set} using a strictly invertable {@link Converter}.
016 *
017 * @param <E> element type of this set
018 * @param <W> element type of the wrapped set
019 */
020public class ConvertedSet<E, W> extends AbstractIterationSet<E> {
021
022    private final Set<W> set;
023    private final Converter<E, W> converter;
024
025    /**
026     * Constructor.
027     *
028     * @param set wrapped set
029     * @param converter element converter
030     * @throws IllegalArgumentException if any parameter is null
031     */
032    ConvertedSet(Set<W> set, Converter<E, W> converter) {
033        if (set == null)
034            throw new IllegalArgumentException("null set");
035        if (converter == null)
036            throw new IllegalArgumentException("null converter");
037        this.set = set;
038        this.converter = converter;
039    }
040
041    public Converter<E, W> getConverter() {
042        return this.converter;
043    }
044
045    @Override
046    @SuppressWarnings("unchecked")
047    public boolean contains(Object obj) {
048        W wobj = null;
049        if (obj != null) {
050            try {
051                wobj = this.converter.convert((E)obj);
052            } catch (ClassCastException e) {
053                return false;
054            }
055        }
056        return this.set.contains(wobj);
057    }
058
059    @Override
060    public Iterator<E> iterator() {
061        return Iterators.transform(this.set.iterator(), this.converter.reverse());
062    }
063
064    @Override
065    public boolean add(E obj) {
066        return this.set.add(obj != null ? this.converter.convert(obj) : null);
067    }
068
069    @Override
070    @SuppressWarnings("unchecked")
071    public boolean remove(Object obj) {
072        W wobj = null;
073        if (obj != null) {
074            try {
075                wobj = this.converter.convert((E)obj);
076            } catch (ClassCastException e) {
077                return false;
078            }
079        }
080        return this.set.remove(wobj);
081    }
082
083    @Override
084    public void clear() {
085        this.set.clear();
086    }
087
088    @Override
089    public boolean isEmpty() {
090        return this.set.isEmpty();
091    }
092
093    @Override
094    public int size() {
095        return this.set.size();
096    }
097}
