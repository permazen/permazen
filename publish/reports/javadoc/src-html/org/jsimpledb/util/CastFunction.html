006package org.jsimpledb.util;
007
008import com.google.common.base.Function;
009
010/**
011 * A {@link Function} that casts objects to some type.
012 */
013public class CastFunction<T> implements Function<Object, T> {
014
015    protected final Class<T> type;
016
017    /**
018     * Constructor.
019     *
020     * @param type desired type
021     * @throws IllegalArgumentException if {@code type} is null
022     */
023     public CastFunction(Class<T> type) {
024        if (type == null)
025            throw new IllegalArgumentException("null type");
026        this.type = type;
027    }
028
029    @Override
030    public T apply(Object obj) {
031        if (obj == null)
032            return null;
033        try {
034            return this.type.cast(obj);
035        } catch (ClassCastException e) {
036            throw this.handleFailure(obj, e);
037        }
038    }
039
040    /**
041     * Generate an exception to throw when a cast failure occurs.
042     *
043     * <p>
044     * The implementation in {@link CastFunction} just throws {@code e}.
045     * </p>
046     *
047     * @param obj object on which cast failed
048     * @param e resulting exception
049     * @return exception to throw
050     */
051    protected RuntimeException handleFailure(Object obj, ClassCastException e) {
052        return e;
053    }
054}
