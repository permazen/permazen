091    /**
092     * Parse command line options.
093     *
094     * @param params command line parameters
095     * @return -1 to proceed, otherwise process exit value
096     */
097    public int parseOptions(ArrayDeque<String> params) {
098
099        // Parse options
100        final LinkedHashSet<String> modelPackages = new LinkedHashSet<>();
101        final LinkedHashSet<String> typePackages = new LinkedHashSet<>();
102        while (!params.isEmpty() && params.peekFirst().startsWith("-")) {
103            final String option = params.removeFirst();
104            if (option.equals("-h") || option.equals("--help")) {
105                this.usageMessage();
106                return 0;
107            } else if (option.equals("-ro") || option.equals("--read-only"))
108                this.readOnly = true;
109            else if (option.equals("-cp") || option.equals("--classpath")) {
110                if (params.isEmpty())
111                    this.usageError();
112                if (!this.appendClasspath(params.removeFirst()))
113                    return 1;
114            } else if (option.equals("--verbose"))
115                this.verbose = true;
116            else if (option.equals("-v") || option.equals("--schema-version")) {
117                if (params.isEmpty())
118                    this.usageError();
119                final String vstring = params.removeFirst();
120                try {
121                    this.schemaVersion = Integer.parseInt(vstring);
122                    if (this.schemaVersion < 0)
123                        throw new IllegalArgumentException("schema version is negative");
124                } catch (Exception e) {
125                    System.err.println(this.getName() + ": invalid schema version `" + vstring + "': " + e.getMessage());
126                    return 1;
127                }
128            } else if (option.equals("--model-pkg")) {
129                if (params.isEmpty())
130                    this.usageError();
131                modelPackages.add(params.removeFirst());
132            } else if (option.equals("--type-pkg")) {
133                if (params.isEmpty())
134                    this.usageError();
135                typePackages.add(params.removeFirst());
136            } else if (option.equals("-p") || option.equals("--pkg")) {
137                if (params.isEmpty())
138                    this.usageError();
139                final String packageName = params.removeFirst();
140                modelPackages.add(packageName);
141                typePackages.add(packageName);
142            } else if (option.equals("--new-schema")) {
143                this.allowNewSchema = true;
144                this.allowAutoDemo = false;
145            } else if (option.equals("--mem"))
146                this.kvType = KV_MEM;
147            else if (option.equals("--prefix")) {
148                if (params.isEmpty())
149                    this.usageError();
150                final String value = params.removeFirst();
151                try {
152                    this.keyPrefix = ByteUtil.parse(value);
153                } catch (IllegalArgumentException e) {
154                    this.keyPrefix = value.getBytes(Charset.forName("UTF-8"));
155                }
156                if (this.keyPrefix.length > 0)
157                    this.allowAutoDemo = false;
158            } else if (option.equals("--fdb")) {
159                if (params.isEmpty())
160                    this.usageError();
161                this.kvType = KV_FDB;
162                this.fdbClusterFile = params.removeFirst();
163                if (!new File(this.fdbClusterFile).exists()) {
164                    System.err.println(this.getName() + ": file `" + this.fdbClusterFile + "' does not exist");
165                    return 1;
166                }
167            } else if (option.equals("--xml")) {
168                if (params.isEmpty())
169                    this.usageError();
170                this.kvType = KV_XML;
171                this.xmlFile = new File(params.removeFirst());
172            } else if (option.equals("--bdb")) {
173                if (params.isEmpty())
174                    this.usageError();
175                this.kvType = KV_BDB;
176                this.bdbDirectory = new File(params.removeFirst());
177                if (!this.bdbDirectory.exists()) {
178                    System.err.println(this.getName() + ": directory `" + this.bdbDirectory + "' does not exist");
179                    return 1;
180                }
181                if (!this.bdbDirectory.isDirectory()) {
182                    System.err.println(this.getName() + ": file `" + this.bdbDirectory + "' is not a directory");
183                    return 1;
184                }
185            } else if (option.equals("--bdb-database")) {
186                if (params.isEmpty())
187                    this.usageError();
188                this.bdbDatabaseName = params.removeFirst();
189            } else if (option.equals("--mysql")) {
190                if (params.isEmpty())
191                    this.usageError();
192                this.jdbcUrl = params.removeFirst();
193                this.kvType = KV_MYSQL;
194            } else if (option.equals("--leveldb")) {
195                if (params.isEmpty())
196                    this.usageError();
197                this.kvType = KV_LEVELDB;
198                this.leveldbDirectory = new File(params.removeFirst());
199                if (!this.leveldbDirectory.exists()) {
200                    System.err.println(this.getName() + ": directory `" + this.leveldbDirectory + "' does not exist");
201                    return 1;
202                }
203                if (!this.leveldbDirectory.isDirectory()) {
204                    System.err.println(this.getName() + ": file `" + this.leveldbDirectory + "' is not a directory");
205                    return 1;
206                }
207            } else if (option.equals("--raft-dir")) {
208                if (params.isEmpty())
209                    this.usageError();
210                this.kvType = KV_RAFT;
211                this.raftDirectory = new File(params.removeFirst());
212                if (!this.raftDirectory.exists() && !this.raftDirectory.mkdirs()) {
213                    System.err.println(this.getName() + ": could not create directory `" + this.raftDirectory + "'");
214                    return 1;
215                }
216                if (!this.raftDirectory.isDirectory()) {
217                    System.err.println(this.getName() + ": file `" + this.raftDirectory + "' is not a directory");
218                    return 1;
219                }
220            } else if (option.matches("--raft-((min|max)-election|heartbeat)-timeout")) {
221                if (params.isEmpty())
222                    this.usageError();
223                final String tstring = params.removeFirst();
224                final int timeout;
225                try {
226                    timeout = Integer.parseInt(tstring);
227                } catch (Exception e) {
228                    System.err.println(this.getName() + ": timeout value `" + tstring + "': " + e.getMessage());
229                    return 1;
230                }
231                if (option.equals("--raft-min-election-timeout"))
232                    this.raftMinElectionTimeout = timeout;
233                else if (option.equals("--raft-max-election-timeout"))
234                    this.raftMaxElectionTimeout = timeout;
235                else if (option.equals("--raft-heartbeat-timeout"))
236                    this.raftHeartbeatTimeout = timeout;
237                else
238                    throw new RuntimeException("internal error");
239            } else if (option.equals("--raft-identity")) {
240                if (params.isEmpty())
241                    this.usageError();
242                this.kvType = KV_RAFT;
243                this.raftIdentity = params.removeFirst();
244            } else if (option.equals("--raft-address")) {
245                if (params.isEmpty())
246                    this.usageError();
247                this.raftAddress = params.removeFirst();
248            } else if (option.equals("--raft-new-cluster"))
249                this.raftNewCluster = true;
250            else if (option.equals("--raft-port")) {
251                if (params.isEmpty())
252                    this.usageError();
253                final String pstring = params.removeFirst();
254                try {
255                    this.raftPort = Integer.parseInt(pstring);
256                    if (this.raftPort < 1 || this.raftPort > 65535)
257                        throw new IllegalArgumentException("value out of range");
258                } catch (Exception e) {
259                    System.err.println(this.getName() + ": invalid Raft TCP port `" + pstring + "': " + e.getMessage());
260                    return 1;
261                }
262            } else if (option.equals("--"))
263                break;
264            else if (!this.parseOption(option, params)) {
265                System.err.println(this.getName() + ": unknown option `" + option + "'");
266                this.usageError();
267                return 1;
268            }
269        }
270
271        // Additional logic post-processing of options
272        if (!modelPackages.isEmpty() || !typePackages.isEmpty())
273            this.allowAutoDemo = false;
274        if (this.kvType != KV_FDB && this.keyPrefix != null) {
275            System.err.println(this.getName() + ": option `--prefix' is only valid in combination with `--fdb'");
276            this.usageError();
277            return 1;
278        }
279        if (this.kvType == -1) {
280            if (this.allowAutoDemo && DEMO_XML_FILE.exists() && DEMO_SUBDIR.exists())
281                this.configureDemoMode();
282            else {
283                System.err.println(this.getName() + ": no key/value store specified; use one of `--mem', `--fdb', etc.");
284                this.usageError();
285                return 1;
286            }
287        }
288
289        // Scan for model and type classes
290        final LinkedHashSet<String> emptyPackages = new LinkedHashSet<>();
291        emptyPackages.addAll(modelPackages);
292        emptyPackages.addAll(typePackages);
293        for (String packageName : modelPackages) {
294            if (this.scanModelClasses(packageName) > 0)
295                emptyPackages.remove(packageName);
296        }
297        for (String packageName : typePackages) {
298            if (this.scanTypeClasses(packageName) > 0)
299                emptyPackages.remove(packageName);
300        }
301
302        // Warn if we didn't find anything
303        for (String packageName : emptyPackages) {
304            final boolean isModel = modelPackages.contains(packageName);
305            final boolean isType = typePackages.contains(packageName);
306            if (isModel && isType)
307                this.log.warn("no Java model or custom FieldType classes found under package `" + packageName + "'");
308            else if (isModel)
309                this.log.warn("no Java model classes found under package `" + packageName + "'");
310            else
311                this.log.warn("no custom FieldType classes found under package `" + packageName + "'");
312        }
313
314        // Done
315        return -1;
316    }

366    /**
367     * Append path(s) to the classpath.
368     *
369     * @param path classpath path component
370     * @return true if successful, false if an error occured
371     */
372    protected boolean appendClasspath(String path) {
373        this.log.trace("adding classpath `" + path + "' to system classpath");
374        try {
375
376            // Get URLClassLoader.addURL() method and make accessible
377            final Method addURLMethod = URLClassLoader.class.getDeclaredMethod("addURL", URL.class);
378            addURLMethod.setAccessible(true);
379
380            // Split path and add components
381            for (String file : path.split(System.getProperty("path.separator", ":"))) {
382                if (file.length() == 0)
383                    continue;
384                addURLMethod.invoke(ClassLoader.getSystemClassLoader(), new Object[] { new File(file).toURI().toURL() });
385                this.log.trace("added path component `" + file + "' to system classpath");
386            }
387            return true;
388        } catch (Exception e) {
389            this.log.error("can't append `" + path + " to classpath: " + e, e);
390            return false;
391        }
392    }

394    private int scanModelClasses(String pkgname) {
395        if (this.schemaClasses == null)
396            this.schemaClasses = new HashSet<>();
397        int count = 0;
398        for (String className : new JSimpleDBClassScanner().scanForClasses(pkgname.split("[\\s,]"))) {
399            this.log.debug("loading Java model class " + className);
400            this.schemaClasses.add(this.loadClass(className));
401            count++;
402        }
403        return count;
404    }
405
406    private int scanTypeClasses(String pkgname) {
407
408        // Check types of annotated classes as we scan them
409        final Function<Class<?>, Class<? extends FieldType<?>>> checkFunction
410          = new Function<Class<?>, Class<? extends FieldType<?>>>() {
411            @Override
412            @SuppressWarnings("unchecked")
413            public Class<? extends FieldType<?>> apply(Class<?> type) {
414                try {
415                    return (Class<? extends FieldType<?>>)type.asSubclass(FieldType.class);
416                } catch (ClassCastException e) {
417                    throw new IllegalArgumentException("invalid @" + JFieldType.class.getSimpleName() + " annotation on "
418                      + type + ": type is not a subclass of " + FieldType.class);
419                }
420            }
421        };
422
423        // Scan classes
424        if (this.fieldTypeClasses == null)
425            this.fieldTypeClasses = new HashSet<>();
426        int count = 0;
427        for (String className : new JSimpleDBFieldTypeScanner().scanForClasses(pkgname.split("[\\s,]"))) {
428            this.log.debug("loading custom FieldType class " + className);
429            this.fieldTypeClasses.add(checkFunction.apply(this.loadClass(className)));
430            count++;
431        }
432        return count;
433    }

435    /**
436     * Load a class.
437     *
438     * @param className class name
439     * @return class with name {@code className}
440     * @throws RuntimeException if load fails
441     */
442    protected Class<?> loadClass(String className) {
443        try {
444            return Class.forName(className, false, Thread.currentThread().getContextClassLoader());
445        } catch (ClassNotFoundException e) {
446            throw new RuntimeException("failed to load class `" + className + "'", e);
447        }
448    }

450    /**
451     * Start the {@link Database} based on the configured {@link KVDatabase} and {@link #fieldTypeClasses} and return it.
452     *
453     * @return initialized database
454     */
455    protected Database startupKVDatabase() {
456
457        // Construct KVDatabase
458        if (this.kvdb != null)
459            this.shutdownKVDatabase();
460        switch (this.kvType) {
461        case KV_MEM:
462            this.kvdb = new SimpleKVDatabase();
463            this.databaseDescription = "In-Memory Database";
464            break;
465        case KV_FDB:
466        {
467            final FoundationKVDatabase fdb = new FoundationKVDatabase();
468            fdb.setClusterFilePath(this.fdbClusterFile);
469            fdb.setKeyPrefix(this.keyPrefix);
470            fdb.start();
471            this.kvdb = fdb;
472            this.databaseDescription = "FoundationDB " + new File(this.fdbClusterFile).getName();
473            if (this.keyPrefix != null)
474                this.databaseDescription += " [0x" + ByteUtil.toString(this.keyPrefix) + "]";
475            break;
476        }
477        case KV_BDB:
478        {
479            final BerkeleyKVDatabase bdb = new BerkeleyKVDatabase();
480            bdb.setDirectory(this.bdbDirectory);
481            bdb.setDatabaseName(this.bdbDatabaseName);
482//            if (this.readOnly)
483//                bdb.setDatabaseConfig(bdb.getDatabaseConfig().setReadOnly(true));
484            bdb.start();
485            this.kvdb = bdb;
486            this.databaseDescription = "BerkeleyDB " + this.bdbDirectory.getName();
487            break;
488        }
489        case KV_XML:
490            this.kvdb = new XMLKVDatabase(this.xmlFile);
491            this.databaseDescription = "XML DB " + this.xmlFile.getName();
492            break;
493        case KV_MYSQL:
494        {
495            try {
496                Class.forName(MYSQL_DRIVER_CLASS_NAME);
497            } catch (RuntimeException e) {
498                throw e;
499            } catch (Exception e) {
500                throw new RuntimeException("can't load MySQL driver class `" + MYSQL_DRIVER_CLASS_NAME + "'", e);
501            }
502            final MySQLKVDatabase mysql = new MySQLKVDatabase();
503            mysql.setDataSource(new DriverManagerDataSource(this.jdbcUrl));
504            this.kvdb = mysql;
505            this.databaseDescription = "MySQL";
506            break;
507        }
508        case KV_LEVELDB:
509        {
510            final LevelDBKVDatabase leveldb = new LevelDBKVDatabase();
511            leveldb.setDirectory(this.leveldbDirectory);
512            leveldb.start();
513            this.kvdb = leveldb;
514            this.databaseDescription = "LevelDB " + this.leveldbDirectory.getName();
515            break;
516        }
517        case KV_RAFT:
518        {
519
520            // Setup network
521            final TCPNetwork network = new TCPNetwork();
522            try {
523                network.setListenAddress(this.raftAddress != null ?
524                  new InetSocketAddress(InetAddress.getByName(this.raftAddress), this.raftPort) :
525                  new InetSocketAddress(this.raftPort));
526            } catch (UnknownHostException e) {
527                throw new RuntimeException("can't resolve address `" + this.raftAddress + "'", e);
528            }
529
530            // Set up Raft DB
531            if (this.raftNewCluster && this.raftAddress == null)
532                throw new RuntimeException("`--raft-new-cluster' requires specifying `--raft-address'");
533            final RaftKVDatabase raft = new RaftKVDatabase();
534            raft.setLogDirectory(this.raftDirectory);
535            raft.setNetwork(network);
536            raft.setIdentity(this.raftIdentity);
537            if (this.raftMinElectionTimeout != -1)
538                raft.setMinElectionTimeout(this.raftMinElectionTimeout);
539            if (this.raftMaxElectionTimeout != -1)
540                raft.setMaxElectionTimeout(this.raftMaxElectionTimeout);
541            if (this.raftHeartbeatTimeout != -1)
542                raft.setHeartbeatTimeout(this.raftHeartbeatTimeout);
543            try {
544                raft.start();
545            } catch (Exception e) {
546                throw new RuntimeException("error starting Raft database", e);
547            }
548
549            // Initialize new cluster
550            if (this.raftNewCluster) {
551                assert this.raftAddress != null;
552
553                // Set up local address
554                final String nodeAddress = this.raftAddress
555                  + (this.raftPort != TCPNetwork.DEFAULT_TCP_PORT ? ":" + this.raftPort : "");
556
557                // Add local node to new cluster
558                try {
559                    final RaftKVTransaction tx = raft.createTransaction();
560                    boolean success = false;
561                    try {
562                        tx.configChange(this.raftIdentity, nodeAddress);
563                        tx.commit();
564                        success = true;
565                    } finally {
566                        if (!success)
567                            tx.rollback();
568                    }
569                } catch (Exception e) {
570                    throw new RuntimeException("error initializing new Raft cluster", e);
571                }
572            }
573
574            this.kvdb = raft;
575            this.databaseDescription = "Raft " + this.raftDirectory.getName();
576            break;
577        }
578        default:
579            throw new RuntimeException("internal error");
580        }
581        this.log.debug("using database: " + this.databaseDescription);
582
583        // Construct core Database
584        final Database db = new Database(this.kvdb);
585
586        // Register custom field types
587        if (this.fieldTypeClasses != null)
588            db.getFieldTypeRegistry().addClasses(this.fieldTypeClasses);
589
590        // Done
591        return db;
592    }

626    protected void shutdownKVDatabase() {
627        if (this.kvdb != null) {
628            switch (this.kvType) {
629            case KV_FDB:
630                ((FoundationKVDatabase)this.kvdb).stop();
631                break;
632            case KV_BDB:
633                ((BerkeleyKVDatabase)this.kvdb).stop();
634                break;
635            case KV_LEVELDB:
636                ((LevelDBKVDatabase)this.kvdb).stop();
637                break;
638            case KV_RAFT:
639                ((RaftKVDatabase)this.kvdb).stop();
640                break;
641            default:
642                break;
643            }
644            this.kvdb = null;
645        }
646    }

650    /**
651     * Output usage message flag listing.
652     *
653     * @param subclassOpts array containing flag and description pairs
654     */
655    protected void outputFlags(String[][] subclassOpts) {
656        final String[][] baseOpts = new String[][] {
657            { "--classpath, -cp path",          "Append to the classpath (useful with `java -jar ...')" },
658            { "--fdb file",                     "Use FoundationDB with specified cluster file" },
