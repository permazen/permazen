006package org.jsimpledb.util;
007
008import java.util.Comparator;
009import java.util.Iterator;
010import java.util.NavigableSet;
011import java.util.NoSuchElementException;
012
013/**
014 * Support superclass for {@link NavigableSet} implementations for which calculating {@link #size size()} requires
015 * an iteration through all of the set's elements to count them.
016 *
017 * <p>
018 * For a read-only implementation, subclasses should implement {@link #comparator comparator()}, {@link #contains contains()},
019 * {@link #iterator iterator()}, and {@link #createSubSet createSubSet()} to handle reversed and restricted range sub-sets.
020 * </p>
021 *
022 * <p>
023 * For a mutable implementation, subclasses should also implement {@link #add add()}, {@link #remove remove()},
024 * {@link #clear clear()}, and make the {@link #iterator iterator()} mutable.
025 * </p>
026 *
027 * <p>
028 * All overridden methods must be aware of the {@linkplain #bounds range restriction bounds}, if any.
029 * </p>
030 *
031 * @param <E> element type
032 */
033public abstract class AbstractNavigableSet<E> extends AbstractIterationSet<E> implements NavigableSet<E> {
034
035    /**
036     * Element range bounds associated with this instance.
037     */
038    protected final Bounds<E> bounds;
039
040    /**
041     * Convenience constructor for the case where there are no lower or upper bounds.
042     */
043    protected AbstractNavigableSet() {
044        this(new Bounds<E>());
045    }
046
047    /**
048     * Primary constructor.
049     *
050     * @param bounds range restriction
051     * @throws IllegalArgumentException if {@code bounds} is null
052     */
053    protected AbstractNavigableSet(Bounds<E> bounds) {
054        if (bounds == null)
055            throw new IllegalArgumentException("null bounds");
056        this.bounds = bounds;
057    }
058
059    @Override
060    public E first() {
061        return this.iterator().next();
062    }
063
064    @Override
065    public E last() {
066        return this.descendingIterator().next();
067    }
068
069    @Override
070    public E pollFirst() {
071        final Iterator<E> i = this.iterator();
072        if (!i.hasNext())
073            return null;
074        final E value = i.next();
075        i.remove();
076        return value;
077    }
078
079    @Override
080    public E pollLast() {
081        return this.descendingSet().pollFirst();
082    }
083
084    @Override
085    public Iterator<E> descendingIterator() {
086        return this.descendingSet().iterator();
087    }
088
089    @Override
090    public E lower(E elem) {
091        return this.searchBelow(elem, false);
092    }
093
094    @Override
095    public E floor(E elem) {
096        return this.searchBelow(elem, true);
097    }
098
099    @Override
100    public E ceiling(E elem) {
101        return this.searchAbove(elem, true);
102    }
103
104    @Override
105    public E higher(E elem) {
106        return this.searchAbove(elem, false);
107    }
108
109    @Override
110    public NavigableSet<E> headSet(E newMaxElement) {
111        return this.headSet(newMaxElement, false);
112    }
113
114    @Override
115    public NavigableSet<E> tailSet(E newMinElement) {
116        return this.tailSet(newMinElement, true);
117    }
118
119    @Override
120    public NavigableSet<E> subSet(E newMinElement, E newMaxElement) {
121        return this.subSet(newMinElement, true, newMaxElement, false);
122    }
123
124    @Override
125    public NavigableSet<E> descendingSet() {
126        return this.createSubSet(true, this.bounds.reverse());
127    }
128
129    @Override
130    public NavigableSet<E> headSet(E newMaxElement, boolean inclusive) {
131        final Bounds<E> newBounds = this.bounds.withUpperBound(newMaxElement, BoundType.of(inclusive));
132        if (!this.bounds.isWithinBounds(this.comparator(), newBounds))
133            throw new IllegalArgumentException("upper bound " + newMaxElement + " is out of bounds: " + this.bounds);
134        return this.createSubSet(false, newBounds);
135    }
136
137    @Override
138    public NavigableSet<E> tailSet(E newMinElement, boolean inclusive) {
139        final Bounds<E> newBounds = this.bounds.withLowerBound(newMinElement, BoundType.of(inclusive));
140        if (!this.bounds.isWithinBounds(this.comparator(), newBounds))
141            throw new IllegalArgumentException("lower bound " + newMinElement + " is out of bounds: " + this.bounds);
142        return this.createSubSet(false, newBounds);
143    }
144
145    @Override
146    public NavigableSet<E> subSet(E newMinElement, boolean minInclusive, E newMaxElement, boolean maxInclusive) {
147        final Bounds<E> newBounds = new Bounds<>(newMinElement,
148          BoundType.of(minInclusive), newMaxElement, BoundType.of(maxInclusive));
149        if (!this.bounds.isWithinBounds(this.comparator(), newBounds))
150            throw new IllegalArgumentException("new bound(s) " + newBounds + " are out of bounds: " + this.bounds);
151        return this.createSubSet(false, newBounds);
152    }
153
154    /**
155     * Search for a lower element. Used to implement {@link #floor floor()} and {@link #lower lower()}.
156     *
157     * <p>
158     * The implementation in {@link AbstractNavigableSet} checks the bounds then returns the first element from a head set.
159     * </p>
160     *
161     * @param elem upper limit for search
162     * @param inclusive true if {@code elem} itself is a candidate
163     * @return highest element below {@code elem}, or null if not found
164     */
165    protected E searchBelow(E elem, boolean inclusive) {
166        if (!this.isWithinLowerBound(elem))
167            return null;
168        final NavigableSet<E> subSet = this.isWithinUpperBound(elem) ? this.headSet(elem, inclusive) : this;
169        try {
170            return subSet.last();
171        } catch (NoSuchElementException e) {
172            return null;
173        }
174    }
175
176    /**
177     * Search for a higher element. Used to implement {@link #ceiling ceiling()} and {@link #higher higher()}.
178     *
179     * <p>
180     * The implementation in {@link AbstractNavigableSet} checks the bounds then returns the first element from a tail set.
181     * </p>
182     *
183     * @param elem lower limit for search
184     * @param inclusive true if {@code elem} itself is a candidate
185     * @return lowest element above {@code elem}, or null if not found
186     */
187    protected E searchAbove(E elem, boolean inclusive) {
188        if (!this.isWithinUpperBound(elem))
189            return null;
190        final NavigableSet<E> subSet = this.isWithinLowerBound(elem) ? this.tailSet(elem, inclusive) : this;
191        try {
192            return subSet.first();
193        } catch (NoSuchElementException e) {
194            return null;
195        }
196    }
197
198    /**
199     * Get a non-null {@link Comparator} that sorts consistently with, and optionally reversed from, this instance.
200     *
201     * @param reversed whether to return a reversed {@link Comparator}
202     * @return a non-null {@link Comparator}
203     */
204    protected Comparator<? super E> getComparator(boolean reversed) {
205        return NavigableSets.getComparator(this.comparator(), reversed);
206    }
207
208    /**
209     * Create a (possibly reversed) view of this instance with (possibly) tighter lower and/or upper bounds.
210     * The {@code newBounds} are consistent with the new ordering (i.e., reversed relative to this instance's ordering if
211     * {@code reverse} is true) and have already been range-checked against {@linkplain #bounds this instance's current bounds}.
212     *
213     * @param reverse whether the new set's ordering should be reversed relative to this instance's ordering
214     * @param newBounds new bounds
215     * @return restricted and/or reversed view of this instance
216     * @throws IllegalArgumentException if {@code newBounds} is null
217     * @throws IllegalArgumentException if a bound in {@code newBounds} is null and this set does not permit null elements
218     */
219    protected abstract NavigableSet<E> createSubSet(boolean reverse, Bounds<E> newBounds);
220
221    /**
222     * Determine if the given element is within this instance's lower bound (if any).
223     *
224     * <p>
225     * The implementation in {@link AbstractNavigableSet} returns {@code this.bounds.isWithinLowerBound(this.comparator(), elem)}.
226     * </p>
227     *
228     * @param elem set element
229     * @return true if {@code elem} is within this instance's lower bound, or this instance has no lower bound
230     */
231    protected boolean isWithinLowerBound(E elem) {
232        return this.bounds.isWithinLowerBound(this.comparator(), elem);
233    }
234
235    /**
236     * Determine if the given element is within this instance's upper bound (if any).
237     *
238     * <p>
239     * The implementation in {@link AbstractNavigableSet} returns {@code this.bounds.isWithinUpperBound(this.comparator(), elem)}.
240     * </p>
241     *
242     * @param elem set element
243     * @return true if {@code elem} is within this instance's upper bound, or this instance has no upper bound
244     */
245    protected boolean isWithinUpperBound(E elem) {
246        return this.bounds.isWithinUpperBound(this.comparator(), elem);
247    }
248}
