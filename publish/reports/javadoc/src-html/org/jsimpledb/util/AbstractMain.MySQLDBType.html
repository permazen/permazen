085    /**
086     * Parse command line options.
087     *
088     * @param params command line parameters
089     * @return -1 to proceed, otherwise process exit value
090     */
091    public int parseOptions(ArrayDeque<String> params) {
092
093        // Parse options
094        final ArrayList<DBType<?>> dbTypes = new ArrayList<>();
095        String raftFallbackUnavailableMergeClassName = OverwriteMergeStrategy.class.getName();
096        String raftFallbackRejoinMergeClassName = NullMergeStrategy.class.getName();
097        final LinkedHashSet<String> modelPackages = new LinkedHashSet<>();
098        final LinkedHashSet<String> typePackages = new LinkedHashSet<>();
099        while (!params.isEmpty() && params.peekFirst().startsWith("-")) {
100            final String option = params.removeFirst();
101            if (option.equals("-h") || option.equals("--help")) {
102                this.usageMessage();
103                return 0;
104            } else if (option.equals("-ro") || option.equals("--read-only"))
105                this.readOnly = true;
106            else if (option.equals("-cp") || option.equals("--classpath")) {
107                if (params.isEmpty())
108                    this.usageError();
109                if (!this.appendClasspath(params.removeFirst()))
110                    return 1;
111            } else if (option.equals("--verbose"))
112                this.verbose = true;
113            else if (option.equals("-v") || option.equals("--schema-version")) {
114                if (params.isEmpty())
115                    this.usageError();
116                final String vstring = params.removeFirst();
117                try {
118                    this.schemaVersion = Integer.parseInt(vstring);
119                    if (this.schemaVersion < 0)
120                        throw new IllegalArgumentException("schema version is negative");
121                } catch (Exception e) {
122                    System.err.println(this.getName() + ": invalid schema version `" + vstring + "': " + e.getMessage());
123                    return 1;
124                }
125            } else if (option.equals("--model-pkg")) {
126                if (params.isEmpty())
127                    this.usageError();
128                modelPackages.add(params.removeFirst());
129            } else if (option.equals("--type-pkg")) {
130                if (params.isEmpty())
131                    this.usageError();
132                typePackages.add(params.removeFirst());
133            } else if (option.equals("-p") || option.equals("--pkg")) {
134                if (params.isEmpty())
135                    this.usageError();
136                final String packageName = params.removeFirst();
137                modelPackages.add(packageName);
138                typePackages.add(packageName);
139            } else if (option.equals("--new-schema")) {
140                this.allowNewSchema = true;
141                this.allowAutoDemo = false;
142            } else if (option.equals("--mem"))
143                dbTypes.add(new MemoryDBType());
144            else if (option.equals("--fdb-prefix")) {
145                if (params.isEmpty())
146                    this.usageError();
147                if (this.foundationDBType == null) {
148                    System.err.println(this.getName() + ": `--fdb' must appear before `" + option + "'");
149                    return 1;
150                }
151                final String value = params.removeFirst();
152                byte[] prefix;
153                try {
154                    prefix = ByteUtil.parse(value);
155                } catch (IllegalArgumentException e) {
156                    prefix = value.getBytes(Charset.forName("UTF-8"));
157                }
158                if (prefix.length > 0)
159                    this.allowAutoDemo = false;
160                this.foundationDBType.setPrefix(prefix);
161            } else if (option.equals("--fdb")) {
162                if (params.isEmpty())
163                    this.usageError();
164                final String clusterFile = params.removeFirst();
165                if (!new File(clusterFile).exists()) {
166                    System.err.println(this.getName() + ": file `" + clusterFile + "' does not exist");
167                    return 1;
168                }
169                this.foundationDBType = new FoundationDBType(clusterFile);
170                dbTypes.add(this.foundationDBType);
171            } else if (option.equals("--xml")) {
172                if (params.isEmpty())
173                    this.usageError();
174                dbTypes.add(new XMLDBType(new File(params.removeFirst())));
175            } else if (option.equals("--bdb")) {
176                if (params.isEmpty())
177                    this.usageError();
178                final File dir = new File(params.removeFirst());
179                if (!this.createDirectory(dir))
180                    return 1;
181                this.berkeleyDBType = new BerkeleyDBType(dir);
182                dbTypes.add(this.berkeleyDBType);
183            } else if (option.equals("--bdb-database")) {
184                if (params.isEmpty())
185                    this.usageError();
186                if (this.berkeleyDBType == null) {
187                    System.err.println(this.getName() + ": `--bdb' must appear before `" + option + "'");
188                    return 1;
189                }
190                this.berkeleyDBType.setDatabaseName(params.removeFirst());
191            } else if (option.equals("--mysql")) {
192                if (params.isEmpty())
193                    this.usageError();
194                dbTypes.add(new MySQLDBType(params.removeFirst()));
195            } else if (option.equals("--leveldb")) {
196                if (params.isEmpty())
197                    this.usageError();
198                final File dir = new File(params.removeFirst());
199                if (!this.createDirectory(dir))
200                    return 1;
201                dbTypes.add(new LevelDBType(dir));
202            } else if (option.equals("--rocksdb")) {
203                if (params.isEmpty())
204                    this.usageError();
205                final File dir = new File(params.removeFirst());
206                if (!this.createDirectory(dir))
207                    return 1;
208                dbTypes.add(new RocksDBType(dir));
209            } else if (option.equals("--arraydb")) {
210                if (params.isEmpty())
211                    this.usageError();
212                final File dir = new File(params.removeFirst());
213                if (!this.createDirectory(dir))
214                    return 1;
215                dbTypes.add(new ArrayDBType(dir));
216            } else if (option.equals("--raft") || option.equals("--raft-dir")) {            // --raft-dir is backward compat.
217                if (params.isEmpty())
218                    this.usageError();
219                final File dir = new File(params.removeFirst());
220                if (!this.createDirectory(dir))
221                    return 1;
222                this.raftDBType = new RaftDBType(dir);
223                dbTypes.add(this.raftDBType);
224            } else if (option.matches("--raft-((min|max)-election|heartbeat)-timeout")) {
225                if (params.isEmpty())
226                    this.usageError();
227                if (this.raftDBType == null) {
228                    System.err.println(this.getName() + ": `--raft' must appear before `" + option + "'");
229                    return 1;
230                }
231                final String tstring = params.removeFirst();
232                final int timeout;
233                try {
234                    timeout = Integer.parseInt(tstring);
235                } catch (Exception e) {
236                    System.err.println(this.getName() + ": invalid timeout value `" + tstring + "': " + e.getMessage());
237                    return 1;
238                }
239                if (option.equals("--raft-min-election-timeout"))
240                    this.raftDBType.setMinElectionTimeout(timeout);
241                else if (option.equals("--raft-max-election-timeout"))
242                    this.raftDBType.setMaxElectionTimeout(timeout);
243                else if (option.equals("--raft-heartbeat-timeout"))
244                    this.raftDBType.setHeartbeatTimeout(timeout);
245                else
246                    throw new RuntimeException("internal error");
247            } else if (option.equals("--raft-identity")) {
248                if (params.isEmpty())
249                    this.usageError();
250                if (this.raftDBType == null) {
251                    System.err.println(this.getName() + ": `--raft' must appear before `" + option + "'");
252                    return 1;
253                }
254                this.raftDBType.setIdentity(params.removeFirst());
255            } else if (option.equals("--raft-address")) {
256                if (params.isEmpty())
257                    this.usageError();
258                if (this.raftDBType == null) {
259                    System.err.println(this.getName() + ": `--raft' must appear before `" + option + "'");
260                    return 1;
261                }
262                final String address = params.removeFirst();
263                this.raftDBType.setAddress(TCPNetwork.parseAddressPart(address));
264                this.raftDBType.setPort(TCPNetwork.parsePortPart(address, this.raftDBType.getPort()));
265            } else if (option.equals("--raft-port")) {
266                if (params.isEmpty())
267                    this.usageError();
268                if (this.raftDBType == null) {
269                    System.err.println(this.getName() + ": `--raft' must appear before `" + option + "'");
270                    return 1;
271                }
272                final String portString = params.removeFirst();
273                final int port = TCPNetwork.parsePortPart("x:" + portString, -1);
274                if (port == -1) {
275                    System.err.println(this.getName() + ": invalid TCP port `" + portString + "'");
276                    return 1;
277                }
278                this.raftDBType.setPort(port);
279            } else if (option.equals("--raft-fallback")) {
280                if (params.isEmpty())
281                    this.usageError();
282                if (this.fallbackDBType != null) {
283                    System.err.println(this.getName() + ": duplicate `" + option + "' flag");
284                    return 1;
285                }
286                final File stateFile = new File(params.removeFirst());
287                if (stateFile.exists() && !stateFile.isFile()) {
288                    System.err.println(this.getName() + ": file `" + stateFile + "' is not a regular file");
289                    return 1;
290                }
291                this.fallbackDBType = new FallbackDBType(stateFile);
292                dbTypes.add(this.fallbackDBType);
293            } else if (option.matches("--raft-fallback-(check-(interval|timeout)|min-(un)?available)")) {
294                if (params.isEmpty())
295                    this.usageError();
296                if (this.fallbackDBType == null) {
297                    System.err.println(this.getName() + ": `--raft-fallback' must appear before `" + option + "'");
298                    return 1;
299                }
300                final String tstring = params.removeFirst();
301                final int millis;
302                try {
303                    millis = Integer.parseInt(tstring);
304                } catch (Exception e) {
305                    System.err.println(this.getName() + ": invalid milliseconds value `" + tstring + "': " + e.getMessage());
306                    return 1;
307                }
308                if (option.equals("--raft-fallback-check-interval"))
309                    this.fallbackDBType.getFallbackTarget().setCheckInterval(millis);
310                else if (option.equals("--raft-fallback-check-timeout"))
311                    this.fallbackDBType.getFallbackTarget().setTransactionTimeout(millis);
312                else if (option.equals("--raft-fallback-min-available"))
313                    this.fallbackDBType.getFallbackTarget().setMinAvailableTime(millis);
314                else if (option.equals("--raft-fallback-min-unavailable"))
315                    this.fallbackDBType.getFallbackTarget().setMinUnavailableTime(millis);
316                else
317                    throw new RuntimeException("internal error");
318            } else if (option.equals("--raft-fallback-unavailable-merge")) {
319                if (params.isEmpty())
320                    this.usageError();
321                if (this.fallbackDBType == null) {
322                    System.err.println(this.getName() + ": `--raft-fallback' must appear before `" + option + "'");
323                    return 1;
324                }
325                raftFallbackUnavailableMergeClassName = params.removeFirst();
326            } else if (option.equals("--raft-fallback-rejoin-merge")) {
327                if (params.isEmpty())
328                    this.usageError();
329                if (this.fallbackDBType == null) {
330                    System.err.println(this.getName() + ": `--raft-fallback' must appear before `" + option + "'");
331                    return 1;
332                }
333                raftFallbackRejoinMergeClassName = params.removeFirst();
334            } else if (option.equals("--"))
335                break;
336            else if (!this.parseOption(option, params)) {
337                System.err.println(this.getName() + ": unknown option `" + option + "'");
338                this.usageError();
339                return 1;
340            }
341        }
342
343        // Additional logic post-processing of options
344        if (!modelPackages.isEmpty() || !typePackages.isEmpty())
345            this.allowAutoDemo = false;
346
347        // Pull out standalone k/v type for Raft fallback
348        if (this.fallbackDBType != null) {
349            final int fallbackIndex = dbTypes.indexOf(this.fallbackDBType);
350            if (fallbackIndex == dbTypes.size() - 1) {
351                System.err.println(this.getName() + ": Raft fallback mode requires an additional peristent store"
352                  + " to be used for standalone mode, specified after `--raft-fallback'; use one of `--arraydb', etc.");
353                return 1;
354            }
355            final DBType<?> standaloneDBType = dbTypes.remove(fallbackIndex + 1);
356            if (!standaloneDBType.canBeFallbackStandalone()) {
357                System.err.println(this.getName() + ": incompatible key/value database `" + standaloneDBType.getDescription()
358                  + "' for Raft fallback standalone mode");
359                return 1;
360            }
361            this.fallbackDBType.setStandaloneDBType(standaloneDBType);
362        }
363
364        // Pull out local store k/v type for Raft
365        if (this.raftDBType != null) {
366            final int raftIndex = dbTypes.indexOf(this.raftDBType);
367            if (raftIndex == dbTypes.size() - 1) {
368                System.err.println(this.getName() + ": Raft raft requires an additional peristent store"
369                  + " to be used for private local storage, specified after `--raft'; use one of `--arraydb', etc.");
370                return 1;
371            }
372            final DBType<?> localStorageDBType = dbTypes.remove(raftIndex + 1);
373            if (!localStorageDBType.canBeRaftLocalStorage()) {
374                System.err.println(this.getName() + ": incompatible key/value database `" + localStorageDBType.getDescription()
375                  + "' for Raft local storage");
376                return 1;
377            }
378            this.raftDBType.setLocalStorageDBType(localStorageDBType);
379        }
380
381        // Pull out Raft k/v type for Raft fallback
382        if (this.fallbackDBType != null) {
383            if (this.raftDBType == null) {
384                System.err.println(this.getName() + ": Raft fallback mode requires a configured Raft database; use `--raft', etc.");
385                return 1;
386            }
387            this.fallbackDBType.setRaftDBType(this.raftDBType);
388            dbTypes.remove(this.raftDBType);
389        }
390
391        // Resolve fallback merge strategy class names (do this here so we benefit from any `--classpath' additions)
392        if (this.fallbackDBType != null) {
393            final String[] classNames = new String[] { raftFallbackUnavailableMergeClassName, raftFallbackRejoinMergeClassName };
394            final MergeStrategy[] mergeStrategies = new MergeStrategy[2];
395            for (int i = 0; i < 2; i++) {
396                try {
397                    mergeStrategies[i] = (MergeStrategy)this.loadClass(classNames[i]).newInstance();
398                } catch (Exception e) {
399                    System.err.println(this.getName() + ": invalid Raft fallback merge strategy `"
400                      + classNames[i] + "': " + e.getMessage());
401                    this.usageError();
402                    return 1;
403                }
404            }
405            this.fallbackDBType.getFallbackTarget().setUnavailableMergeStrategy(mergeStrategies[0]);
406            this.fallbackDBType.getFallbackTarget().setRejoinMergeStrategy(mergeStrategies[1]);
407        }
408
409        // Check database choice(s)
410        switch (dbTypes.size()) {
411        case 0:
412            if (this.allowAutoDemo && DEMO_XML_FILE.exists() && DEMO_SUBDIR.exists()) {
413
414                // Configure database
415                System.err.println(this.getName() + ": auto-configuring use of demo database `" + DEMO_XML_FILE + "'");
416                if (dbTypes.isEmpty())
417                    dbTypes.add(new XMLDBType(DEMO_XML_FILE));
418
419                // Add demo subdirectory to class path
420                this.appendClasspath(DEMO_SUBDIR.toString());
421
422                // Scan classes
423                this.scanModelClasses("org.jsimpledb.demo");
424            } else {
425                System.err.println(this.getName() + ": no key/value store specified; use one of `--arraydb', etc.");
426                this.usageError();
427                return 1;
428            }
429            break;
430        case 1:
431            break;
432        default:
433            System.err.println(this.getName() + ": more than one key/value store was specified");
434            this.usageError();
435            return 1;
436        }
437        this.dbType = dbTypes.get(0);
438
439        // Scan for model and type classes
440        final LinkedHashSet<String> emptyPackages = new LinkedHashSet<>();
441        emptyPackages.addAll(modelPackages);
442        emptyPackages.addAll(typePackages);
443        for (String packageName : modelPackages) {
444            if (this.scanModelClasses(packageName) > 0)
445                emptyPackages.remove(packageName);
446        }
447        for (String packageName : typePackages) {
448            if (this.scanTypeClasses(packageName) > 0)
449                emptyPackages.remove(packageName);
450        }
451
452        // Warn if we didn't find anything
453        for (String packageName : emptyPackages) {
454            final boolean isModel = modelPackages.contains(packageName);
455            final boolean isType = typePackages.contains(packageName);
456            if (isModel && isType)
457                this.log.warn("no Java model or custom FieldType classes found under package `" + packageName + "'");
458            else if (isModel)
459                this.log.warn("no Java model classes found under package `" + packageName + "'");
460            else
461                this.log.warn("no custom FieldType classes found under package `" + packageName + "'");
462        }
463
464        // Done
465        return -1;
466    }

502    /**
503     * Append path(s) to the classpath.
504     *
505     * @param path classpath path component
506     * @return true if successful, false if an error occured
507     */
508    protected boolean appendClasspath(String path) {
509        this.log.trace("adding classpath `" + path + "' to system classpath");
510        try {
511
512            // Get URLClassLoader.addURL() method and make accessible
513            final Method addURLMethod = URLClassLoader.class.getDeclaredMethod("addURL", URL.class);
514            addURLMethod.setAccessible(true);
515
516            // Split path and add components
517            for (String file : path.split(System.getProperty("path.separator", ":"))) {
518                if (file.length() == 0)
519                    continue;
520                addURLMethod.invoke(ClassLoader.getSystemClassLoader(), new Object[] { new File(file).toURI().toURL() });
521                this.log.trace("added path component `" + file + "' to system classpath");
522            }
523            return true;
524        } catch (Exception e) {
525            this.log.error("can't append `" + path + " to classpath: " + e, e);
526            return false;
527        }
528    }

530    private int scanModelClasses(String pkgname) {
531        if (this.schemaClasses == null)
532            this.schemaClasses = new HashSet<>();
533        int count = 0;
534        for (String className : new JSimpleDBClassScanner().scanForClasses(pkgname.split("[\\s,]"))) {
535            this.log.debug("loading Java model class " + className);
536            this.schemaClasses.add(this.loadClass(className));
537            count++;
538        }
539        return count;
540    }

542    private int scanTypeClasses(String pkgname) {
543
544        // Check types of annotated classes as we scan them
545        final Function<Class<?>, Class<? extends FieldType<?>>> checkFunction
546          = new Function<Class<?>, Class<? extends FieldType<?>>>() {
547            @Override
548            @SuppressWarnings("unchecked")
549            public Class<? extends FieldType<?>> apply(Class<?> type) {
550                try {
551                    return (Class<? extends FieldType<?>>)type.asSubclass(FieldType.class);
552                } catch (ClassCastException e) {
553                    throw new IllegalArgumentException("invalid @" + JFieldType.class.getSimpleName() + " annotation on "
554                      + type + ": type is not a subclass of " + FieldType.class);
555                }
556            }
557        };
558
559        // Scan classes
560        if (this.fieldTypeClasses == null)
561            this.fieldTypeClasses = new HashSet<>();
562        int count = 0;
563        for (String className : new JSimpleDBFieldTypeScanner().scanForClasses(pkgname.split("[\\s,]"))) {
564            this.log.debug("loading custom FieldType class " + className);
565            this.fieldTypeClasses.add(checkFunction.apply(this.loadClass(className)));
566            count++;
567        }
568        return count;
569    }

571    private boolean createDirectory(File dir) {
572        if (!dir.exists() && !dir.mkdirs()) {
573            System.err.println(this.getName() + ": could not create directory `" + dir + "'");
574            return false;
575        }
576        if (!dir.isDirectory()) {
577            System.err.println(this.getName() + ": file `" + dir + "' is not a directory");
578            return false;
579        }
580        return true;
581    }

583    /**
584     * Load a class.
585     *
586     * @param className class name
587     * @return class with name {@code className}
588     * @throws RuntimeException if load fails
589     */
590    protected Class<?> loadClass(String className) {
591        try {
592            return Class.forName(className, false, Thread.currentThread().getContextClassLoader());
593        } catch (ClassNotFoundException e) {
594            throw new RuntimeException("failed to load class `" + className + "'", e);
595        }
596    }

598    /**
599     * Start the {@link Database} based on the configured {@link KVDatabase} and {@link #fieldTypeClasses} and return it.
600     *
601     * @return initialized database
602     */
603    protected Database startupKVDatabase() {
604
605        // Create and start up database
606        this.kvdb = this.dbType.createKVDatabase();
607        this.databaseDescription = this.dbType.getDescription();
608        AbstractMain.startKVDatabase(this.dbType, this.kvdb);
609        this.log.debug("using database: " + this.databaseDescription);
610
611        // Construct core Database
612        final Database db = new Database(this.kvdb);
613
614        // Register custom field types
615        if (this.fieldTypeClasses != null)
616            db.getFieldTypeRegistry().addClasses(this.fieldTypeClasses);
617
618        // Done
619        return db;
620    }
