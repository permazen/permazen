006package org.jsimpledb.util;
007
008import java.util.AbstractSet;
009import java.util.Iterator;
010
011import org.jsimpledb.core.ObjId;
012
013/**
014 * A set of {@link ObjId}s.
015 *
016 * <p>
017 * This implementation is space optimized for the 64-bits of information contained in an {@link ObjId}.
018 * Instances do not accept null values and are not thread safe.
019 * </p>
020 */
021public class ObjIdSet extends AbstractSet<ObjId> implements Cloneable {
022
023    private /*final*/ ObjIdMap<?> map;
024
025// Constructors
026
027    /**
028     * Constructs an empty instance.
029     */
030    public ObjIdSet() {
031        this(0);
032    }
033
034    /**
035     * Constructs an instance with the given initial capacity.
036     *
037     * @param capacity initial capacity
038     * @throws IllegalArgumentException if {@code capacity} is negative
039     */
040    public ObjIdSet(int capacity) {
041        this(new ObjIdMap<Void>(capacity, false));
042    }
043
044    /**
045     * Constructs an instance initialized with the given ID's.
046     *
047     * @param ids initial contents for this instance
048     * @throws IllegalArgumentException if {@code ids} is null
049     * @throws NullPointerException if any ID in {@code ids} is null
050     */
051    public ObjIdSet(Iterable<? extends ObjId> ids) {
052        this(0);
053        if (ids == null)
054            throw new IllegalArgumentException("null ids");
055        for (ObjId id : ids)
056            this.add(id);
057    }
058
059    // Internal constructor
060    ObjIdSet(ObjIdMap<?> map) {
061        this.map = map;
062    }
063
064// Methods
065
066    @Override
067    public Iterator<ObjId> iterator() {
068        return new Iterator<ObjId>() {
069
070            private final ObjIdMap<?>.EntrySetIterator entryIterator = ObjIdSet.this.map.new EntrySetIterator();
071
072            @Override
073            public boolean hasNext() {
074                return this.entryIterator.hasNext();
075            }
076
077            @Override
078            public ObjId next() {
079                return this.entryIterator.next().getKey();
080            }
081
082            @Override
083            public void remove() {
084                this.entryIterator.remove();
085            }
086        };
087    }
088
089    @Override
090    public int size() {
091        return this.map.size();
092    }
093
094    @Override
095    public boolean isEmpty() {
096        return this.map.isEmpty();
097    }
098
099    @Override
100    public boolean contains(Object obj) {
101        return this.map.containsKey(obj);
102    }
103
104    @Override
105    public boolean add(ObjId id) {
106        if (this.map.containsKey(id))
107            return false;
108        this.map.put(id, null);
109        return true;
110    }
111
112    @Override
113    public boolean remove(Object obj) {
114        if (!this.map.containsKey(obj))
115            return false;
116        this.map.remove(obj);
117        return true;
118    }
119
120    @Override
121    public void clear() {
122        this.map.clear();
123    }
124
125    /**
126     * Produce a debug dump of this instance.
127     */
128    String debugDump() {
129        return this.map.debugDump();
130    }
131
132// Object
133
134    // This works because ObjId.hashCode() == ObjId.asLong().hashCode()
135    @Override
136    public int hashCode() {
137        final long[] keyArray = this.map.getKeys();
138        int hash = 0;
139        for (int i = 0; i < keyArray.length; i++) {
140            final long key = keyArray[i];
141            if (key != 0)
142                hash += (int)(key >>> 32) ^ (int)key;
143        }
144        return hash;
145    }
146
147// Cloneable
148
149    @Override
150    public ObjIdSet clone() {
151        final ObjIdSet clone;
152        try {
153            clone = (ObjIdSet)super.clone();
154        } catch (CloneNotSupportedException e) {
155            throw new RuntimeException(e);
156        }
157        clone.map = clone.map.clone();
158        return clone;
159    }
160}
