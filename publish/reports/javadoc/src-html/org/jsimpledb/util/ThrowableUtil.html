006package org.jsimpledb.util;
007
008import com.google.common.base.Preconditions;
009
010/**
011 * Utility methods dealing with {@link Throwable}s.
012 */
013public final class ThrowableUtil {
014
015    private ThrowableUtil() {
016    }
017
018    /**
019     * Mask a checked exception and throw it as an unchecked exception.
020     *
021     * @param t any exception
022     * @param <T> unchecked exception type
023     * @return never
024     * @throws T always
025     */
026    @SuppressWarnings("unchecked")
027    public static <T extends Throwable> T maskException(Throwable t) throws T {
028        throw (T)t;
029    }
030
031    /**
032     * Prepend stack frames from the current thread onto the given exception's stack frames.
033     *
034     * <p>
035     * This is used to re-throw an exception created in another thread without losing the stack frame information
036     * associated with the current thread.
037     *
038     * @param t exception from another thread
039     * @throws IllegalArgumentException if {@code t} is null
040     */
041    public static void prependCurrentStackTrace(Throwable t) {
042        Preconditions.checkArgument(t != null, "null t");
043        final StackTraceElement[] innerFrames = t.getStackTrace();
044        final StackTraceElement[] outerFrames = new Throwable().getStackTrace();
045        final StackTraceElement[] frames = new StackTraceElement[innerFrames.length + Math.max(outerFrames.length, 0)];
046        System.arraycopy(innerFrames, 0, frames, 0, innerFrames.length);
047        for (int i = 0; i < outerFrames.length; i++)
048            frames[innerFrames.length + i] = outerFrames[i];
049        t.setStackTrace(frames);
050    }
051}
