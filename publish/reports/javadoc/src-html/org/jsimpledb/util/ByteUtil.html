006package org.jsimpledb.util;
007
008import com.google.common.base.Converter;
009
010import java.util.Comparator;
011
012/**
013 * Byte manipulation utilities.
014 */
015public final class ByteUtil {
016
017    /**
018     * An empty byte array. This is the minimum value according to {@link #COMPARATOR}.
019     */
020    public static final byte[] EMPTY = new byte[0];
021
022    /**
023     * {@link Comparator} that compares two byte arrays lexicographically using unsigned values.
024     */
025    public static final Comparator<byte[]> COMPARATOR = new Comparator<byte[]>() {
026        @Override
027        public int compare(byte[] b1, byte[] b2) {
028            return ByteUtil.compare(b1, b2);
029        }
030    };
031
032    /**
033     * A {@link Converter} that converts between {@code byte[]} arrays and hexadecimal {@link String}s.
034     */
035    public static final Converter<byte[], String> STRING_CONVERTER = new Converter<byte[], String>() {
036
037        @Override
038        public String doForward(byte[] b) {
039            return b != null ? ByteUtil.toString(b) : null;
040        }
041
042        @Override
043        public byte[] doBackward(String s) {
044            return s != null ? ByteUtil.parse(s) : null;
045        }
046    };
047
048    private ByteUtil() {
049    }
050
051    /**
052     * Compare two byte arrays lexicographically using unsigned values.
053     *
054     * @param b1 first byte array
055     * @param b2 second byte array
056     * @return -1 if {@code b1 < b2}, 1 if {@code b1 > b2}, or zero if {@code b1 = b2}
057     * @throws NullPointerException if {@code b1} or {@code b2} is null
058     */
059    public static int compare(byte[] b1, byte[] b2) {
060        if (b1 == b2)
061            return 0;
062        final int sharedLength = Math.min(b1.length, b2.length);
063        for (int i = 0; i < sharedLength; i++) {
064            final int v1 = b1[i] & 0xff;
065            final int v2 = b2[i] & 0xff;
066            if (v1 < v2)
067                return -1;
068            if (v1 > v2)
069                return 1;
070        }
071        if (b1.length < b2.length)
072            return -1;
073        if (b1.length > b2.length)
074            return 1;
075        return 0;
076    }
077
078    /**
079     * Determine the smaller of two byte arrays when compared lexicographically using unsigned values.
080     *
081     * @param b1 first byte array
082     * @param b2 second byte array
083     * @return {@code b1} if {@code b1 <= b2}, otherwise {@code b2}
084     * @throws NullPointerException if {@code b1} or {@code b2} is null
085     */
086    public static byte[] min(byte[] b1, byte[] b2) {
087        return ByteUtil.compare(b1, b2) <= 0 ? b1 : b2;
088    }
089
090    /**
091     * Determine the larger of two byte arrays when compared lexicographically using unsigned values.
092     *
093     * @param b1 first byte array
094     * @param b2 second byte array
095     * @return {@code b1} if {@code b1 >= b2}, otherwise {@code b2}
096     * @throws NullPointerException if {@code b1} or {@code b2} is null
097     */
098    public static byte[] max(byte[] b1, byte[] b2) {
099        return ByteUtil.compare(b1, b2) >= 0 ? b1 : b2;
100    }
101
102    /**
103     * Determine if the first of two {@code byte[]} arrays is a prefix of the second.
104     *
105     * @param prefix prefix to check
106     * @param value value to check for having {@code prefix} as a prefix
107     * @return true if {@code prefix} is a prefix of {@code value}
108     * @throws NullPointerException if {@code prefix} or {@code value} is null
109     */
110    public static boolean isPrefixOf(byte[] prefix, byte[] value) {
111        if (prefix.length > value.length)
112            return false;
113        for (int i = 0; i < prefix.length; i++) {
114            if (value[i] != prefix[i])
115                return false;
116        }
117        return true;
118    }
119
120    /**
121     * Get the next key greater than the given key in unsigned lexicographic ordering.
122     * This creates a new key simply by appending a {@code 0x00} byte to the data
123     * contained in the given key.
124     *
125     * @param key previous key
126     * @return next key after {@code key}
127     * @throws NullPointerException if {@code key} is null
128     */
129    public static byte[] getNextKey(byte[] key) {
130        final byte[] nextKey = new byte[key.length + 1];
131        System.arraycopy(key, 0, nextKey, 0, key.length);
132        return nextKey;
133    }
134
135    /**
136     * Get the first key that would be greater than the given key in unsigned lexicographic
137     * ordering <i>and</i> that does not have the given key as a prefix.
138     *
139     * @param prefix lower bound prefix key
140     * @return next key not having {@code prefix} as a prefix
141     * @throws IllegalArgumentException if {@code prefix} has zero length
142     * @throws IllegalArgumentException if {@code prefix} contains only {@code 0xff} bytes
143     * @throws NullPointerException if {@code prefix} is null
144     */
145    public static byte[] getKeyAfterPrefix(byte[] prefix) {
146        int len = prefix.length;
147        if (len == 0)
148            throw new IllegalArgumentException("empty prefix");
149        while (len > 0 && prefix[len - 1] == (byte)0xff)
150            len--;
151        if (len <= 0)
152            throw new IllegalArgumentException("prefix contains only 0xff bytes");
153        final byte[] buf = new byte[len];
154        System.arraycopy(prefix, 0, buf, 0, len);
155        buf[len - 1]++;
156        return buf;
157    }
158
159    /**
160     * Convert a byte array into a string of hex digits, or {@code "null"} if {@code buf} is null.
161     *
162     * @param buf bytes
163     * @return string encoding of {@code buf}
164     * @see #parse parse()
165     */
166    public static String toString(byte[] buf) {
167        if (buf == null)
168            return "null";
169        final char[] result = new char[buf.length * 2];
170        int off = 0;
171        for (int i = 0; i < buf.length; i++) {
172            int value = buf[i];
173            result[off++] = Character.forDigit((value >> 4) & 0x0f, 16);
174            result[off++] = Character.forDigit(value & 0x0f, 16);
175        }
176        return new String(result);
177    }
178
179    /**
180     * Decode a hexadecimal {@link String} into a {@code byte[]} array. The string must have an even
181     * number of digits and contain no other characters (e.g., whitespace).
182     *
183     * @param text string previously encoded by {@link #toString(byte[])}
184     * @return {@code byte[]} decoding of {@code text}
185     * @throws IllegalArgumentException if any non-hexadecimal characters are found or the number of characters is odd
186     * @throws NullPointerException if {@code text} is null
187     * @see #toString(byte[]) toString()
188     */
189    public static byte[] parse(String text) {
190        if ((text.length() & 1) != 0)
191            throw new IllegalArgumentException("byte array has an odd number of digits");
192        final byte[] array = new byte[text.length() / 2];
193        int pos = 0;
194        for (int i = 0; pos < text.length(); i++) {
195            final int nib1 = Character.digit(text.charAt(pos++), 16);
196            if (nib1 == -1)
197                throw new IllegalArgumentException("invalid hex digit `" + text.charAt(pos - 1) + "'");
198            final int nib2 = Character.digit(text.charAt(pos++), 16);
199            if (nib2 == -1)
200                throw new IllegalArgumentException("invalid hex digit `" + text.charAt(pos - 1) + "'");
201            array[i] = (byte)((nib1 << 4) | nib2);
202        }
203        return array;
204    }
205
206    /**
207     * Read an {@code int} as four big-endian bytes.
208     *
209     * @param reader input
210     * @return decoded integer
211     * @throws IndexOutOfBoundsException if less than four bytes remain in {@code reader}
212     * @throws NullPointerException if {@code reader} is null
213     * @see #writeInt writeInt()
214     */
215    public static int readInt(ByteReader reader) {
216        return (reader.readByte() << 24) | (reader.readByte() << 16) | (reader.readByte() << 8) | reader.readByte();
217    }
218
219    /**
220     * Write an {@code int} as four big-endian bytes.
221     *
222     * @param writer byte destination
223     * @param value value to write
224     * @see #readInt readInt()
225     * @throws NullPointerException if {@code writer} is null
226     */
227    public static void writeInt(ByteWriter writer, int value) {
228        writer.writeByte(value >> 24);
229        writer.writeByte(value >> 16);
230        writer.writeByte(value >> 8);
231        writer.writeByte(value);
232    }
233
234    /**
235     * Read a {@code long} as eight big-endian bytes.
236     *
237     * @param reader input
238     * @return decoded long
239     * @throws IndexOutOfBoundsException if less than eight bytes remain in {@code reader}
240     * @see #writeLong writeLong()
241     */
242    public static long readLong(ByteReader reader) {
243        return
244            ((long)reader.readByte() << 56) | ((long)reader.readByte() << 48)
245          | ((long)reader.readByte() << 40) | ((long)reader.readByte() << 32)
246          | ((long)reader.readByte() << 24) | ((long)reader.readByte() << 16)
247          | ((long)reader.readByte() <<  8) |  (long)reader.readByte();
248    }
249
250    /**
251     * Write a {@code long} as eight big-endian bytes.
252     *
253     * @param writer byte destination
254     * @param value value to write
255     * @see #readLong readLong()
256     * @throws NullPointerException if {@code writer} is null
257     */
258    public static void writeLong(ByteWriter writer, long value) {
259        writer.writeByte((int)(value >> 56));
260        writer.writeByte((int)(value >> 48));
261        writer.writeByte((int)(value >> 40));
262        writer.writeByte((int)(value >> 32));
263        writer.writeByte((int)(value >> 24));
264        writer.writeByte((int)(value >> 16));
265        writer.writeByte((int)(value >> 8));
266        writer.writeByte((int)value);
267    }
268}
