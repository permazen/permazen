006package org.jsimpledb.util;
007
008import java.io.EOFException;
009import java.io.IOException;
010import java.io.InputStream;
011import java.io.OutputStream;
012import java.nio.ByteBuffer;
013
014/**
015 * Encodes unsigned (i.e., non-negative) {@code int} values to/from binary, preserving sort order, and such
016 * that the length of the encoding is optimized for values near zero and encoded values never begin with {@code 0xff}.
017 */
018public final class UnsignedIntEncoder {
019
020    /**
021     * Maximum possible length of an encoded value.
022     */
023    public static final int MAX_ENCODED_LENGTH = 5;
024
025    /**
026     * Minimum value that triggers a multi-byte encoding.
027     */
028    public static final int MIN_MULTI_BYTE_VALUE = 0xfb;                       // values 0xfb ... 0xfe prefix multi-byte values
029
030    private UnsignedIntEncoder() {
031    }
032
033    /**
034     * Encode the given value.
035     *
036     * @param value value to encode
037     * @return encoded value
038     * @throws IllegalArgumentException if {@code value} is negative
039     */
040    public static byte[] encode(int value) {
041        final ByteWriter writer = new ByteWriter(UnsignedIntEncoder.encodeLength(value));
042        UnsignedIntEncoder.write(writer, value);
043        return writer.getBytes();
044    }
045
046    /**
047     * Decode the given value.
048     *
049     * @param data encoded value
050     * @return decoded value
051     * @throws IllegalArgumentException if {@code bytes} contains an invalid encoding, or extra trailing garbage
052     */
053    public static int decode(byte[] data) {
054        final ByteReader reader = new ByteReader(data);
055        final int value = UnsignedIntEncoder.read(reader);
056        if (reader.remain() > 0)
057            throw new IllegalArgumentException("encoded value contains extra trailing garbage");
058        return value;
059    }
060
061    /**
062     * Encode the given value to the output.
063     *
064     * @param writer destination for the encoded value
065     * @param value value to encode
066     * @throws IllegalArgumentException if {@code value} is negative
067     */
068    public static void write(ByteWriter writer, int value) {
069        writer.makeRoom(MAX_ENCODED_LENGTH);
070        writer.len += UnsignedIntEncoder.encode(value, writer.buf, writer.len);
071    }
072
073    /**
074     * Encode the given value and write it to the given {@link OutputStream}.
075     *
076     * @param out destination for the encoded value
077     * @param value value to encode
078     * @throws IOException if an I/O error occurs
079     * @throws NullPointerException if {@code out} is null
080     */
081    public static void write(OutputStream out, int value) throws IOException {
082        final byte[] array = new byte[UnsignedIntEncoder.MAX_ENCODED_LENGTH];
083        final int nbytes = UnsignedIntEncoder.encode(value, array, 0);
084        out.write(array, 0, nbytes);
085    }
086
087    /**
088     * Encode the given value and write it to the given {@link ByteBuffer}.
089     *
090     * @param buf destination for the encoded value
091     * @param value value to encode
092     * @throws java.nio.BufferOverflowException if {@code buf} overflows
093     * @throws java.nio.ReadOnlyBufferException if {@code buf} is read-only
094     * @throws NullPointerException if {@code buf} is null
095     */
096    public static void write(ByteBuffer buf, int value) {
097        final byte[] array = new byte[UnsignedIntEncoder.MAX_ENCODED_LENGTH];
098        final int nbytes = UnsignedIntEncoder.encode(value, array, 0);
099        buf.put(array, 0, nbytes);
100    }
101
102    /**
103     * Read and decode a value from the input.
104     *
105     * @param reader input holding an encoded value
106     * @return the decoded value, always non-negative
107     * @throws IllegalArgumentException if the encoded value is truncated
108     * @throws IllegalArgumentException if an invalid encoding is encountered
109     */
110    public static int read(ByteReader reader) {
111        try {
112            final int first = reader.readByte();
113            int value;
114            switch (first) {
115            case 0xfb:
116                value = reader.readByte();
117                break;
118            case 0xfc:
119                value = (reader.readByte() << 8) | reader.readByte();
120                break;
121            case 0xfd:
122                value = (reader.readByte() << 16) | (reader.readByte() << 8) | reader.readByte();
123                break;
124            case 0xfe:
125                value = (reader.readByte() << 24) | (reader.readByte() << 16) | (reader.readByte() << 8) | reader.readByte();
126                if (value + MIN_MULTI_BYTE_VALUE < 0)
127                    throw new IllegalArgumentException("invalid unsigned int encoding with high bit set");
128                break;
129            case 0xff:
130                throw new IllegalArgumentException("invalid unsigned int encoding starting with 0xff");
131            default:
132                return first;
133            }
134            return value + MIN_MULTI_BYTE_VALUE;
135        } catch (IndexOutOfBoundsException e) {
136            throw new IllegalArgumentException("encoded value is truncated", e);
137        }
138    }
139
140    /**
141     * Read and decode a value from the given {@link InputStream}.
142     *
143     * @param input input source for the encoded value
144     * @return the decoded value
145     * @throws IOException if an I/O error occurs
146     * @throws EOFException if an unexpected EOF is encountered
147     * @throws IllegalArgumentException if an invalid encoding is encountered
148     * @throws NullPointerException if {@code input} is null
149     */
150    public static int read(InputStream input) throws IOException {
151        final int first = input.read();
152        if (first == -1)
153            throw new EOFException();
154        final byte[] array = new byte[UnsignedIntEncoder.decodeLength(first)];
155        array[0] = (byte)first;
156        for (int i = 1; i < array.length; i++) {
157            final int next = input.read();
158            if (next == -1)
159                throw new EOFException();
160            array[i] = (byte)next;
161        }
162        return UnsignedIntEncoder.read(new ByteReader(array));
163    }
164
165    /**
166     * Read and decode a value from the given {@link ByteBuffer}.
167     *
168     * @param buf input source for the encoded value
169     * @return the decoded value
170     * @throws java.nio.BufferUnderflowException if {@code buf} underflows
171     * @throws IllegalArgumentException if an invalid encoding is encountered
172     * @throws NullPointerException if {@code buf} is null
173     */
174    public static int read(ByteBuffer buf) {
175        final byte first = buf.get();
176        final byte[] array = new byte[UnsignedIntEncoder.decodeLength(first)];
177        array[0] = first;
178        if (array.length > 1)
179            buf.get(array, 1, array.length - 1);
180        return UnsignedIntEncoder.read(new ByteReader(array));
181    }
182
183    /**
184     * Skip a value from the input.
185     *
186     * @param reader input holding an encoded value
187     */
188    public static void skip(ByteReader reader) {
189        final int first = reader.readByte();
190        if (first >= MIN_MULTI_BYTE_VALUE)
191            reader.skip(first - MIN_MULTI_BYTE_VALUE + 1);
192    }
193
194    /**
195     * Determine the length (in bytes) of an encoded value based on the first byte.
196     *
197     * @param first first byte of encoded value (in lower eight bits; other bits are ignored)
198     * @return the length of the encoded value (including {@code first})
199     * @throws IllegalArgumentException if the lower eight bits of {@code first} equal {@code 0xff}
200     */
201    public static int decodeLength(int first) {
202        first &= 0xff;
203        if (first == 0xff)
204            throw new IllegalArgumentException("invalid unsigned int encoding starting with 0xff");
205        return first < MIN_MULTI_BYTE_VALUE ? 1 : first - MIN_MULTI_BYTE_VALUE + 2;
206    }
207
208    /**
209     * Determine the length (in bytes) of the encoded value.
210     *
211     * @param value value to encode
212     * @return the length of the encoded value, a value between one and {@link #MAX_ENCODED_LENGTH}
213     * @throws IllegalArgumentException if {@code value} is negative
214     */
215    public static int encodeLength(int value) {
216        if (value < 0)
217            throw new IllegalArgumentException("value < 0");
218        value -= MIN_MULTI_BYTE_VALUE;
219        if (value < 0)
220            return 1;
221        int length = 2;
222        while ((value >>= 8) != 0)
223            length++;
224        return length;
225    }
226
227    /**
228     * Encode the given value and write the encoded bytes into the given buffer.
229     *
230     * @param value value to encode
231     * @param buf output buffer
232     * @param off starting offset into output buffer
233     * @return the number of encoded bytes written
234     * @throws IllegalArgumentException if {@code value} is negative
235     * @throws ArrayIndexOutOfBoundsException if {@code off} is negative or the encoded value exceeds the given buffer
236     */
237    private static int encode(int value, byte[] buf, int off) {
238        if (value < 0)
239            throw new IllegalArgumentException("value < 0");
240        if (value < MIN_MULTI_BYTE_VALUE) {
241            buf[off] = (byte)value;
242            return 1;
243        }
244        value -= MIN_MULTI_BYTE_VALUE;
245        int len = 1;
246        int mask = 0xff000000;
247        boolean encoding = false;
248        for (int shift = 24; shift != 0; shift -= 8, mask >>= 8) {
249            if (encoding || (value & mask) != 0L) {
250                buf[off + len++] = (byte)(value >> shift);
251                encoding = true;
252            }
253        }
254        buf[off + len++] = (byte)value;
255        buf[off] = (byte)(MIN_MULTI_BYTE_VALUE + len - 2);
256        return len;
257    }
258
259    /**
260     * Test routine.
261     *
262     * @param args command line arguments
263     */
264    public static void main(String[] args) {
265        for (String arg : args) {
266            byte[] bytes = null;
267            try {
268                bytes = ByteUtil.parse(arg);
269            } catch (IllegalArgumentException e) {
270                if (arg.startsWith("0x"))
271                    bytes = ByteUtil.parse(arg.substring(2));
272            }
273            if (bytes != null) {
274                System.out.println("Decoding bytes: " + ByteUtil.toString(bytes));
275                try {
276                    final long value = UnsignedIntEncoder.decode(bytes);
277                    System.out.println("0x" + ByteUtil.toString(bytes)
278                      + " decodes to " + value + " (" + String.format("0x%016x", value) + ")");
279                } catch (IndexOutOfBoundsException e) {
280                    System.out.println("Error decoding " + ByteUtil.toString(bytes) + ": " + e);
281                } catch (IllegalArgumentException e) {
282                    System.out.println("Error decoding " + ByteUtil.toString(bytes) + ": " + e);
283                }
284            }
285            Integer value = null;
286            try {
287                value = Integer.parseInt(arg);
288            } catch (IllegalArgumentException e) {
289                // ignore
290            }
291            if (value != null && value.intValue() >= 0) {
292                System.out.println("Encoding value " + value);
293                final ByteWriter writer = new ByteWriter();
294                UnsignedIntEncoder.write(writer, value);
295                System.out.println(value + " (" + String.format("0x%016x", value)
296                  + ") encodes to " + ByteUtil.toString(writer.getBytes()));
297            }
298        }
299    }
