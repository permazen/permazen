006package org.jsimpledb.util;
007
008import com.google.common.collect.Iterators;
009import com.google.common.collect.PeekingIterator;
010
011import java.util.Comparator;
012import java.util.Iterator;
013
014/**
015 * {@link Iterator} implementation that provides a read-only view of an inner {@link Iterator}
016 * in which consecutive duplicate values are eliminated.
017 */
018public class UniqueIterator<E> implements Iterator<E> {
019
020    private final PeekingIterator<E> iterator;
021    private final Comparator<? super E> comparator;
022
023    /**
024     * Constructor. Object equality using {@link Object#equals} is used to detect duplicates.
025     *
026     * @param iterator wrapped {@link Iterator}
027     */
028    public UniqueIterator(Iterator<? extends E> iterator) {
029        this.iterator = Iterators.peekingIterator(iterator);
030        this.comparator = null;
031    }
032
033    /**
034     * Constructor. A zero result from the provided {@link Comparator} is used to detect duplicates.
035     *
036     * @param iterator wrapped {@link Iterator}
037     * @param comparator used to compare consecutive values
038     * @throws IllegalArgumentException if {@code comparator} is null
039     */
040    public UniqueIterator(Iterator<? extends E> iterator, Comparator<? super E> comparator) {
041        if (comparator == null)
042            throw new IllegalArgumentException("null comparator");
043        this.iterator = Iterators.peekingIterator(iterator);
044        this.comparator = comparator;
045    }
046
047    @Override
048    public boolean hasNext() {
049        return this.iterator.hasNext();
050    }
051
052    @Override
053    public E next() {
054        final E next = this.iterator.next();
055        while (this.iterator.hasNext()) {
056            final E peek = this.iterator.peek();
057            if (this.comparator != null ? this.comparator.compare(next, peek) != 0 :
058              next != null ? !next.equals(peek) : peek != null)
059                break;
060            this.iterator.next();
061        }
062        return next;
063    }
064
065    @Override
066    public void remove() {
067        throw new UnsupportedOperationException();
068    }
069}
