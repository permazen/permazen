006package org.jsimpledb.util;
007
008import java.util.Arrays;
009
010import javax.xml.namespace.QName;
011import javax.xml.stream.XMLStreamConstants;
012import javax.xml.stream.XMLStreamException;
013import javax.xml.stream.XMLStreamReader;
014import javax.xml.stream.XMLStreamWriter;
015
016/**
017 * Support superclass for classes that serialize and deserialize via XML.
018 */
019public abstract class AbstractXMLStreaming {
020
021    protected AbstractXMLStreaming() {
022    }
023
024    /**
025     * Scan forward until we see an opening or closing tag.
026     * If opening tag is seen, it must match one of {@code names} and then we return true, if not or if {@code names}
027     * is empty throw an exception; if a closing tag is seen, return false if {@code closingOK}, else throw exception.
028     *
029     * @param reader XML input
030     * @param closingOK true if a closing tag is OK, otherwise false
031     * @param names expected opening tag, or null if we expected a closing tag
032     * @return true if matching opening tag seen, false otherwise
033     * @throws XMLStreamException if something unexpected is encountered
034     */
035    protected boolean expect(XMLStreamReader reader, boolean closingOK, QName... names) throws XMLStreamException {
036        while (true) {
037            if (!reader.hasNext())
038                throw new XMLStreamException("unexpected end of input", reader.getLocation());
039            final int eventType = reader.next();
040            if (eventType == XMLStreamConstants.START_ELEMENT)
041                break;
042            if (eventType == XMLStreamConstants.END_ELEMENT) {
043                if (!closingOK) {
044                    throw new XMLStreamException("expected " + this.description(names) + " but found closing <"
045                      + reader.getName() + "> tag instead", reader.getLocation());
046                }
047                return false;
048            }
049        }
050        if (names.length == 0) {
051            throw new XMLStreamException("expected closing tag but found opening <"
052              + reader.getName() + "> tag instead", reader.getLocation());
053        }
054        if (!Arrays.asList(names).contains(reader.getName())) {
055            throw new XMLStreamException("expected " + this.description(names)
056              + " but found <" + reader.getName() + "> instead", reader.getLocation());
057        }
058        return true;
059    }
060
061    /**
062     * Scan forward expecting to see a closing tag.
063     *
064     * @param reader XML input
065     * @throws XMLStreamException if something other than a closing tag is encountered
066     */
067    protected void expectClose(XMLStreamReader reader) throws XMLStreamException {
068        this.expect(reader, true);
069    }
070
071    private String description(QName[] names) {
072        switch (names.length) {
073        case 0:
074            return "closing tag";
075        case 1:
076            return "opening <" + names[0] + "> tag";
077        default:
078            final StringBuilder buf = new StringBuilder();
079            for (QName name : names) {
080                if (buf.length() == 0)
081                    buf.append("one of ");
082                else
083                    buf.append(", ");
084                buf.append('<').append(name).append('>');
085            }
086            return buf.toString();
087        }
088    }
089
090    /**
091     * Write out a simple XML element containing the given content.
092     *
093     * @param writer XML output
094     * @param element element name
095     * @param content simple content
096     * @throws XMLStreamException if error occurs writing output
097     */
098    protected void writeElement(XMLStreamWriter writer, QName element, String content) throws XMLStreamException {
099        writer.writeStartElement(element.getNamespaceURI(), element.getLocalPart());
100        writer.writeCharacters(content);
101        writer.writeEndElement();
102    }
103
104    /**
105     * Get an attribute from the current element.
106     *
107     * @param reader XML input
108     * @param name attribute name
109     * @param required whether attribute must be present
110     * @return attribute value, or null if not {@code required} and no attribute is present
111     * @throws IllegalStateException if the current event is not a start element event
112     * @throws XMLStreamException if {@code required} is true and no such attribute is found
113     */
114    protected String getAttr(XMLStreamReader reader, QName name, boolean required) throws XMLStreamException {
115        final String value = reader.getAttributeValue(name.getNamespaceURI(), name.getLocalPart());
116        if (value == null && required) {
117            throw new XMLStreamException("<" + reader.getName().getLocalPart() + "> element is missing required \""
118              + name + "\" attribute", reader.getLocation());
119        }
120        return value;
121    }
122
123    /**
124     * Get a requried attribute from the current element. Equivalent to: {@code getAttr(reader, name, true)}.
125     *
126     * @param reader XML input
127     * @param name attribute name
128     * @return attribute value
129     * @throws IllegalStateException if the current event is not a start element event
130     * @throws XMLStreamException if no such attribute is found
131     */
132    protected String getAttr(XMLStreamReader reader, QName name) throws XMLStreamException {
133        return this.getAttr(reader, name, true);
134    }
135
136    /**
137     * Get an attribute from the current element and parse as a decimal integer value.
138     *
139     * @param reader XML input
140     * @param name attribute name
141     * @param required whether attribute must be present
142     * @return attribute value, or null if not {@code required} and no attribute is present
143     * @throws IllegalStateException if the current event is not a start element event
144     * @throws XMLStreamException if {@code required} is true and no such attribute is found
145     * @throws XMLStreamException if attribute is not an integer value
146     */
147    protected Integer getIntAttr(XMLStreamReader reader, QName name, boolean required) throws XMLStreamException {
148        final String text = this.getAttr(reader, name, required);
149        if (text == null)
150            return null;
151        try {
152            return Integer.parseInt(text, 10);
153        } catch (NumberFormatException e) {
154            throw new XMLStreamException("<" + reader.getName().getLocalPart() + "> element attribute \""
155              + name + "\" value `" + text + "' is invalid: not a valid value", reader.getLocation(), e);
156        }
157    }
158
159    /**
160     * Get an attribute from the current element and parse as a decimal long value.
161     *
162     * @param reader XML input
163     * @param name attribute name
164     * @param required whether attribute must be present
165     * @return attribute value, or null if not {@code required} and no attribute is present
166     * @throws IllegalStateException if the current event is not a start element event
167     * @throws XMLStreamException if {@code required} is true and no such attribute is found
168     * @throws XMLStreamException if attribute is not an integer value
169     */
170    protected Long getLongAttr(XMLStreamReader reader, QName name, boolean required) throws XMLStreamException {
171        final String text = this.getAttr(reader, name, required);
172        if (text == null)
173            return null;
174        try {
175            return Long.parseLong(text, 10);
176        } catch (NumberFormatException e) {
177            throw new XMLStreamException("<" + reader.getName().getLocalPart() + "> element attribute \""
178              + name + "\" value `" + text + "' is invalid: not a valid long value", reader.getLocation(), e);
179        }
180    }
181
182    /**
183     * Get a requried integer attribute from the current element. Equivalent to: {@code getIntAttr(reader, name, true)}.
184     *
185     * @param reader XML input
186     * @param name attribute name
187     * @return attribute value
188     * @throws IllegalStateException if the current event is not a start element event
189     * @throws XMLStreamException if no such attribute is found
190     * @throws XMLStreamException if attribute is not an integer value
191     */
192    protected int getIntAttr(XMLStreamReader reader, QName name) throws XMLStreamException {
193        return this.getIntAttr(reader, name, true);
194    }
195
196    /**
197     * Get an attribute from the current element and parse as a boolean value.
198     *
199     * @param reader XML input
200     * @param name attribute name
201     * @param required whether attribute must be present
202     * @return attribute value, or null if not {@code required} and no attribute is present
203     * @throws IllegalStateException if the current event is not a start element event
204     * @throws XMLStreamException if {@code required} is true and no such attribute is found
205     * @throws XMLStreamException if attribute is not {@code "true"} or {@code "false"}
206     */
207    protected Boolean getBooleanAttr(XMLStreamReader reader, QName name, boolean required) throws XMLStreamException {
208        final String text = this.getAttr(reader, name, required);
209        if (text == null)
210            return null;
211        switch (text) {
212        case "true":
213            return true;
214        case "false":
215            return false;
216        default:
217            throw new XMLStreamException("<" + reader.getName().getLocalPart() + "> element attribute \""
218              + name + "\" value `" + text + "' is invalid: expected either \"true\" or \"false\"", reader.getLocation());
219        }
220    }
221
222    /**
223     * Get a requried boolean attribute from the current element. Equivalent to: {@code getBooleanAttr(reader, name, true)}.
224     *
225     * @param reader XML input
226     * @param name attribute name
227     * @return attribute value
228     * @throws IllegalStateException if the current event is not a start element event
229     * @throws XMLStreamException if no such attribute is found
230     * @throws XMLStreamException if attribute is not {@code "true"} or {@code "false"}
231     */
232    protected boolean getBooleanAttr(XMLStreamReader reader, QName name) throws XMLStreamException {
233        return this.getBooleanAttr(reader, name, true);
234    }
235}
