006package org.jsimpledb.util;
007
008import java.util.Arrays;
009import java.util.LinkedHashMap;
010import java.util.Map;
011import java.util.regex.Matcher;
012
013/**
014 * Used to report difference(s) between two hierarchical structures.
015 *
016 * <p>
017 * Each individual difference is a {@link String} (map key) and has optional further detail contained in another
018 * {@link Diffs} instance (map value, or null for none). Keys must not be null.
019 * </p>
020 */
021@SuppressWarnings("serial")
022public class Diffs extends LinkedHashMap<String, Diffs> {
023
024    /**
025     * Default constructor.
026     */
027    public Diffs() {
028    }
029
030    /**
031     * Copy constructor. Performs a shallow copy.
032     *
033     * @param other instance to copy
034     */
035    public Diffs(Diffs other) {
036        super(other);
037    }
038
039    /**
040     * Add a difference without any sub-differences.
041     *
042     * @param description description of the difference
043     * @throws IllegalArgumentException if {@code description} is null
044     */
045    public void add(String description) {
046        this.add(description, null);
047    }
048
049    /**
050     * Add a difference with optional sub-differences.
051     *
052     * @param description description of the difference
053     * @param diffs sub-differences, or null for none
054     * @throws IllegalArgumentException if {@code description} is null
055     */
056    public void add(String description, Diffs diffs) {
057        if (description == null)
058            throw new IllegalArgumentException("null description");
059        this.put(description, diffs);
060    }
061
062    /**
063     * Format this instance as a {@link String}.
064     */
065    @Override
066    public String toString() {
067        if (this.isEmpty())
068            return "no differences";
069        final StringBuilder buf = new StringBuilder();
070        this.format(buf, 0);
071        return buf.toString().trim();
072    }
073
074    private void format(StringBuilder buf, int depth) {
075        final String indent = this.getIndent(depth);
076        for (Map.Entry<String, Diffs> entry : this.entrySet()) {
077            final String description = entry.getKey();
078            buf.append(String.format("%s%s%n", indent,
079              description.trim().replaceAll("(\\r?\\n)", "$1" + Matcher.quoteReplacement(indent))));
080            final Diffs child = entry.getValue();
081            if (child != null)
082                child.format(buf, depth + 1);
083        }
084    }
085
086    private String getIndent(int depth) {
087        final char[] indent = new char[depth * 2];
088        Arrays.fill(indent, ' ');
089        return new String(indent) + "-> ";
090    }
091}
