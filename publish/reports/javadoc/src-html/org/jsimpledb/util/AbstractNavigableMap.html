006package org.jsimpledb.util;
007
008import java.util.AbstractMap;
009import java.util.Comparator;
010import java.util.Iterator;
011import java.util.Map;
012import java.util.NavigableMap;
013import java.util.NavigableSet;
014import java.util.NoSuchElementException;
015
016/**
017 * Support superclass for {@link NavigableMap} implementations.
018 *
019 * <p>
020 * For a read-only implementation, subclasses should implement {@link #comparator comparator()}, {@link #get get()},
021 * {@link #entrySet entrySet()}, {@link #navigableKeySet navigableKeySet()}, and {@link #createSubMap createSubMap()}
022 * to handle reversed and restricted range sub-maps.
023 * </p>
024 *
025 * <p>
026 * For a mutable implementation, subclasses should also implement {@link #put put()}, {@link #remove remove()},
027 * {@link #clear clear()}, and make the {@link #keySet keySet()} and {@link #entrySet entrySet()} iterators mutable.
028 * </p>
029 *
030 * <p>
031 * All overridden methods must be aware of the {@linkplain #bounds range restriction bounds}, if any.
032 * </p>
033 *
034 * @param <K> key type
035 * @param <V> value type
036 */
037public abstract class AbstractNavigableMap<K, V> extends AbstractMap<K, V> implements NavigableMap<K, V> {
038
039    /**
040     * Key range bounds associated with this instance.
041     */
042    protected final Bounds<K> bounds;
043
044    /**
045     * Convenience constructor for the case where there are no lower or upper key bounds.
046     */
047    protected AbstractNavigableMap() {
048        this(new Bounds<K>());
049    }
050
051    /**
052     * Primary constructor.
053     *
054     * @param bounds key range restriction
055     * @throws IllegalArgumentException if {@code bounds} is null
056     */
057    protected AbstractNavigableMap(Bounds<K> bounds) {
058        if (bounds == null)
059            throw new IllegalArgumentException("null bounds");
060        this.bounds = bounds;
061    }
062
063    @Override
064    public boolean isEmpty() {
065        return this.navigableKeySet().isEmpty();
066    }
067
068    @Override
069    public boolean containsKey(Object obj) {
070        return this.navigableKeySet().contains(obj);
071    }
072
073    @Override
074    public K firstKey() {
075        return this.navigableKeySet().iterator().next();
076    }
077
078    @Override
079    public K lastKey() {
080        return this.descendingMap().firstKey();
081    }
082
083    @Override
084    public NavigableSet<K> keySet() {
085        return this.navigableKeySet();
086    }
087
088    @Override
089    public Map.Entry<K, V> lowerEntry(K maxKey) {
090        return this.searchBelow(maxKey, false);
091    }
092
093    @Override
094    public Map.Entry<K, V> floorEntry(K maxKey) {
095        return this.searchBelow(maxKey, true);
096    }
097
098    @Override
099    public Map.Entry<K, V> ceilingEntry(K minKey) {
100        return this.searchAbove(minKey, true);
101    }
102
103    @Override
104    public Map.Entry<K, V> higherEntry(K minKey) {
105        return this.searchAbove(minKey, false);
106    }
107
108    @Override
109    public K lowerKey(K maxKey) {
110        final Map.Entry<K, V> entry = this.lowerEntry(maxKey);
111        return entry != null ? entry.getKey() : null;
112    }
113
114    @Override
115    public K floorKey(K maxKey) {
116        final Map.Entry<K, V> entry = this.floorEntry(maxKey);
117        return entry != null ? entry.getKey() : null;
118    }
119
120    @Override
121    public K ceilingKey(K minKey) {
122        final Map.Entry<K, V> entry = this.ceilingEntry(minKey);
123        return entry != null ? entry.getKey() : null;
124    }
125
126    @Override
127    public K higherKey(K minKey) {
128        final Map.Entry<K, V> entry = this.higherEntry(minKey);
129        return entry != null ? entry.getKey() : null;
130    }
131
132    @Override
133    public Map.Entry<K, V> firstEntry() {
134        try {
135            return this.entrySet().iterator().next();
136        } catch (NoSuchElementException e) {
137            return null;
138        }
139    }
140
141    @Override
142    public Map.Entry<K, V> lastEntry() {
143        return this.descendingMap().firstEntry();
144    }
145
146    @Override
147    public Map.Entry<K, V> pollFirstEntry() {
148        final Iterator<Map.Entry<K, V>> i = this.entrySet().iterator();
149        if (!i.hasNext())
150            return null;
151        final Map.Entry<K, V> entry = i.next();
152        i.remove();
153        return entry;
154    }
155
156    @Override
157    public Map.Entry<K, V> pollLastEntry() {
158        return this.descendingMap().pollFirstEntry();
159    }
160
161    @Override
162    public NavigableMap<K, V> descendingMap() {
163        return this.createSubMap(true, this.bounds.reverse());
164    }
165
166    @Override
167    public NavigableSet<K> descendingKeySet() {
168        return this.navigableKeySet().descendingSet();
169    }
170
171    @Override
172    public NavigableMap<K, V> subMap(K minKey, K maxKey) {
173        return this.subMap(minKey, true, maxKey, false);
174    }
175
176    @Override
177    public NavigableMap<K, V> headMap(K maxKey) {
178        return this.headMap(maxKey, false);
179    }
180
181    @Override
182    public NavigableMap<K, V> tailMap(K minKey) {
183        return this.tailMap(minKey, true);
184    }
185
186    @Override
187    public NavigableMap<K, V> headMap(K newMaxKey, boolean inclusive) {
188        final Bounds<K> newBounds = this.bounds.withUpperBound(newMaxKey, BoundType.of(inclusive));
189        if (!this.bounds.isWithinBounds(this.comparator(), newBounds))
190            throw new IllegalArgumentException("upper bound " + newMaxKey + " is out of bounds: " + this.bounds);
191        return this.createSubMap(false, newBounds);
192    }
193
194    @Override
195    public NavigableMap<K, V> tailMap(K newMinKey, boolean inclusive) {
196        final Bounds<K> newBounds = this.bounds.withLowerBound(newMinKey, BoundType.of(inclusive));
197        if (!this.bounds.isWithinBounds(this.comparator(), newBounds))
198            throw new IllegalArgumentException("lower bound " + newMinKey + " is out of bounds: " + this.bounds);
199        return this.createSubMap(false, newBounds);
200    }
201
202    @Override
203    public NavigableMap<K, V> subMap(K newMinKey, boolean minInclusive, K newMaxKey, boolean maxInclusive) {
204        final Bounds<K> newBounds = new Bounds<>(newMinKey, BoundType.of(minInclusive), newMaxKey, BoundType.of(maxInclusive));
205        if (!this.bounds.isWithinBounds(this.comparator(), newBounds))
206            throw new IllegalArgumentException("new bound(s) " + newBounds + " are out of bounds: " + this.bounds);
207        return this.createSubMap(false, newBounds);
208    }
209
210    /**
211     * Search for a lower element. Used to implement {@link #floorEntry floorEntry()} and {@link #lowerEntry lowerEntry()}.
212     *
213     * <p>
214     * The implementation in {@link AbstractNavigableMap} checks the bounds then returns the first entry from a head map.
215     * </p>
216     *
217     * @param maxKey upper limit for search
218     * @param inclusive true if {@code maxKey} itself is a candidate
219     * @return highest element below {@code maxKey}, or null if not found
220     */
221    protected Map.Entry<K, V> searchBelow(K maxKey, boolean inclusive) {
222        if (!this.isWithinLowerBound(maxKey))
223            return null;
224        final NavigableMap<K, V> subMap = this.isWithinUpperBound(maxKey) ? this.headMap(maxKey, inclusive) : this;
225        try {
226            return subMap.lastEntry();
227        } catch (NoSuchElementException e) {
228            return null;
229        }
230    }
231
232    /**
233     * Search for a higher element. Used to implement {@link #ceilingEntry ceilingEntry()} and {@link #higherEntry higherEntry()}.
234     *
235     * <p>
236     * The implementation in {@link AbstractNavigableMap} checks the bounds then returns the first entry from a tail map.
237     * </p>
238     *
239     * @param minKey lower limit for search
240     * @param inclusive true if {@code minKey} itself is a candidate
241     * @return lowest element above {@code minKey}, or null if not found
242     */
243    protected Map.Entry<K, V> searchAbove(K minKey, boolean inclusive) {
244        if (!this.isWithinUpperBound(minKey))
245            return null;
246        final NavigableMap<K, V> subMap = this.isWithinLowerBound(minKey) ? this.tailMap(minKey, inclusive) : this;
247        try {
248            return subMap.firstEntry();
249        } catch (NoSuchElementException e) {
250            return null;
251        }
252    }
253
254    /**
255     * Get a non-null {@link Comparator} that sorts consistently with, and optionally reversed from, this instance.
256     *
257     * @param reversed whether to return a reversed {@link Comparator}
258     * @return a non-null {@link Comparator}
259     */
260    protected Comparator<? super K> getComparator(boolean reversed) {
261        return NavigableSets.getComparator(this.comparator(), reversed);
262    }
263
264    /**
265     * Create a (possibly reversed) view of this instance with (possibly) tighter lower and/or upper bounds.
266     * The {@code newBounds} are consistent with the new ordering (i.e., reversed relative to this instance's ordering if
267     * {@code reverse} is true) and have already been range-checked against {@linkplain #bounds this instance's current bounds}.
268     *
269     * @param reverse whether the new map's ordering should be reversed relative to this instance's ordering
270     * @param newBounds new bounds
271     * @return restricted and/or reversed view of this instance
272     * @throws IllegalArgumentException if {@code newBounds} is null
273     * @throws IllegalArgumentException if a bound in {@code newBounds} is null and this set does not permit null elements
274     */
275    protected abstract NavigableMap<K, V> createSubMap(boolean reverse, Bounds<K> newBounds);
276
277    /**
278     * Determine if the given element is within this instance's lower bound (if any).
279     *
280     * <p>
281     * The implementation in {@link AbstractNavigableMap} returns {@code this.bounds.isWithinLowerBound(this.comparator(), elem)}.
282     * </p>
283     *
284     * @param key map key
285     * @return true if {@code elem} is within this instance's lower bound, or this instance has no lower bound
286     */
287    protected boolean isWithinLowerBound(K key) {
288        return this.bounds.isWithinLowerBound(this.comparator(), key);
289    }
290
291    /**
292     * Determine if the given element is within this instance's upper bound (if any).
293     *
294     * <p>
295     * The implementation in {@link AbstractNavigableMap} returns {@code this.bounds.isWithinUpperBound(this.comparator(), elem)}.
296     * </p>
297     *
298     * @param key map key
299     * @return true if {@code elem} is within this instance's upper bound, or this instance has no upper bound
300     */
301    protected boolean isWithinUpperBound(K key) {
302        return this.bounds.isWithinUpperBound(this.comparator(), key);
303    }
304}
