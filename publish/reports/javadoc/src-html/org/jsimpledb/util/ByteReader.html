006package org.jsimpledb.util;
007
008/**
009 * Reads bytes from a buffer.
010 */
011public class ByteReader {
012
013    final byte[] buf;
014    final int max;
015    int off;
016
017    /**
018     * Constructor. The provided array is read from directly; no copy is made.
019     *
020     * @param buf array to read from
021     * @throws NullPointerException if {@code buf} is null
022     */
023    public ByteReader(byte[] buf) {
024        this.buf = buf;
025        this.max = buf.length;
026        this.off = 0;
027    }
028
029    /**
030     * Constructor. The provided array is read from directly; no copy is made.
031     *
032     * @param buf array to read from
033     * @param off starting offset into {@code buf}
034     * @throws IndexOutOfBoundsException if {@code off} is out of bounds
035     * @throws NullPointerException if {@code buf} is null
036     */
037    public ByteReader(byte[] buf, int off) {
038        this(buf, off, buf.length - off);
039    }
040
041    /**
042     * Constructor. The provided array is read from directly; no copy is made.
043     *
044     * @param buf array to read from
045     * @param off offset into {@code buf}
046     * @param len number of bytes to read
047     * @throws IndexOutOfBoundsException if {@code off} or {@code len} are out of bounds
048     * @throws NullPointerException if {@code buf} is null
049     */
050    public ByteReader(byte[] buf, int off, int len) {
051        if (off < 0 || len < 0 || off > buf.length || off + len < 0 || off + len > buf.length)
052            throw new IndexOutOfBoundsException("buf.length = " + buf.length + ", off = " + off + ", len = " + len);
053        this.buf = buf;
054        this.max = off + len;
055        this.off = off;
056    }
057
058    /**
059     * Constructor. Takes a snapshot of the given writer's entire content.
060     *
061     * @param writer {@code ByteWriter} to read data from
062     * @throws NullPointerException if {@code writer} is null
063     */
064    public ByteReader(ByteWriter writer) {
065        this(writer.buf, 0, writer.len);
066    }
067
068    /**
069     * Constructor. Takes a snapshot of the given writer's content starting at the specified position.
070     *
071     * @param writer {@code ByteWriter} to read data from
072     * @param mark position previously returned by {@code ByteWriter#mark}
073     * @throws IndexOutOfBoundsException if {@code mark} is out of bounds
074     * @throws NullPointerException if {@code writer} is null
075     */
076    public ByteReader(ByteWriter writer, int mark) {
077        this(writer.buf, mark, writer.len - mark);
078    }
079
080    /**
081     * Peek at next byte, if any.
082     *
083     * @return next byte (0-255)
084     * @throws IndexOutOfBoundsException if there are no more bytes
085     */
086    public int peek() {
087        if (this.off == this.max)
088            throw new IndexOutOfBoundsException();
089        return this.buf[this.off] & 0xff;
090    }
091
092    /**
093     * Read the next byte.
094     *
095     * @return next byte (0-255)
096     * @throws IndexOutOfBoundsException if there are no more bytes
097     */
098    public int readByte() {
099        if (this.off == this.max)
100            throw new IndexOutOfBoundsException();
101        return this.buf[this.off++] & 0xff;
102    }
103
104    /**
105     * Unread the previously read byte. Equivalent to {@code unread(1)}.
106     *
107     * @throws IndexOutOfBoundsException if there are no more bytes to unread
108     */
109    public void unread() {
110        if (this.off == 0)
111            throw new IndexOutOfBoundsException();
112        this.off--;
113    }
114
115    /**
116     * Unread the given number of previously read bytes.
117     *
118     * @param len the number of bytes to unread
119     * @throws IndexOutOfBoundsException if there are no more bytes to unread
120     */
121    public void unread(int len) {
122        if (this.off - len < 0)
123            throw new IndexOutOfBoundsException();
124        this.off -= len;
125    }
126
127    /**
128     * Read the specified number of bytes.
129     *
130     * @param len number of bytes to read
131     * @return bytes read
132     * @throws IndexOutOfBoundsException if there are not enough bytes
133     * @throws IllegalArgumentException if {@code len} is negative
134     */
135    public byte[] readBytes(int len) {
136        if (len < 0)
137            throw new IllegalArgumentException("len < 0");
138        if (this.off + len > this.max)
139            throw new IndexOutOfBoundsException();
140        final byte[] result = new byte[len];
141        System.arraycopy(this.buf, this.off, result, 0, len);
142        this.off += len;
143        return result;
144    }
145
146    /**
147     * Get the number of bytes remaining.
148     *
149     * @return bytes remaining
150     */
151    public int remain() {
152        return this.max - this.off;
153    }
154
155    /**
156     * Skip over bytes.
157     *
158     * @param num the number of bytes to skip
159     * @throws IndexOutOfBoundsException if {@code num} is negative
160     * @throws IndexOutOfBoundsException if less than {@code num} bytes remain
161     */
162    public void skip(int num) {
163        if (num < 0 || this.off + num > this.max)
164            throw new IndexOutOfBoundsException();
165        this.off += num;
166    }
167
168    /**
169     * Get current offset into buffer.
170     *
171     * @return current offset
172     */
173    public int getOffset() {
174        return this.off;
175    }
176
177    /**
178     * Get maximum offset into buffer.
179     *
180     * @return maximum offset
181     */
182    public int getMax() {
183        return this.max;
184    }
185
186    /**
187     * Copy a range of bytes from the buffer. Does not change the read position.
188     *
189     * @param off offset into buffer
190     * @param len number of bytes
191     * @return copy of the specified byte range
192     * @throws IndexOutOfBoundsException if {@code off} and/or {@code len} is out of bounds
193     */
194    public byte[] getBytes(int off, int len) {
195        final byte[] data = new byte[len];
196        System.arraycopy(this.buf, off, data, 0, len);
197        return data;
198    }
199
200    /**
201     * Copy a range of bytes from the given offset to the end of the buffer. Does not change the read position.
202     *
203     * @param off offset into buffer
204     * @return copy of the specified byte range
205     * @throws IndexOutOfBoundsException if {@code off} is out of bounds
206     */
207    public byte[] getBytes(int off) {
208        return this.getBytes(off, this.max - off);
209    }
210
211    /**
212     * Copy all the of bytes in the buffer. Does not change the read position.
213     *
214     * @return copy of the entire buffer
215     */
216    public byte[] getBytes() {
217        return this.max == this.buf.length ? this.buf.clone() : this.getBytes(0);
218    }
219
220    /**
221     * Mark current read position.
222     *
223     * @return the current offset
224     */
225    public int mark() {
226        return this.off;
227    }
228
229    /**
230     * Reset read position to a previously marked position.
231     *
232     * @param mark value previously returned by {@link #mark}
233     * @throws IndexOutOfBoundsException if {@code mark} is out of bounds
234     */
235    public void reset(int mark) {
236        if (mark < 0 || mark > this.max)
237            throw new IndexOutOfBoundsException();
238        this.off = mark;
239    }
240}
