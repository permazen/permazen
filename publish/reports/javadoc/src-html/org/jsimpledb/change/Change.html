006package org.jsimpledb.change;
007
008import org.jsimpledb.JObject;
009import org.jsimpledb.JTransaction;
010
011/**
012 * Object change notification.
013 *
014 * @param <T> the type of the object that changed
015 */
016public abstract class Change<T> {
017
018    private final T jobj;
019
020    /**
021     * Constructor.
022     *
023     * @param jobj Java model object that changed
024     * @throws IllegalArgumentException if {@code jobj} is null
025     */
026    protected Change(T jobj) {
027        if (jobj == null)
028            throw new IllegalArgumentException("null jobj");
029        this.jobj = jobj;
030    }
031
032    /**
033     * Get the Java model object containing the field that changed.
034     *
035     * <p>
036     * Although not declared as such to allow flexibility in Java model types, the returned object
037     * will always be a {@link JObject} instance.
038     * </p>
039     *
040     * @return the changed object
041     */
042    public T getObject() {
043        return this.jobj;
044    }
045
046    /**
047     * Get the Java model object containing the field that changed.
048     *
049     * <p>
050     * This is a convenience method, equivalent to:
051     * <blockquote><code>
052     * (JObject)getObject()
053     * </code></blockquote>
054     *
055     * @return the changed object as a {@link JObject}
056     */
057    public JObject getJObject() {
058        return (JObject)this.jobj;
059    }
060
061    /**
062     * Apply visitor pattern. Invokes the method of {@code target} corresponding to this instance's type.
063     *
064     * @param target visitor pattern target
065     * @param <R> visitor return type
066     * @return value returned by the selected method of {@code target}
067     */
068    public abstract <R> R visit(ChangeSwitch<R> target);
069
070    /**
071     * Apply this change to the given object in the given transaction.
072     *
073     * @param jobj the target object to which to apply this change
074     * @param jtx the transaction in which to apply this change
075     * @throws NullPointerException if {@code jtx} or {@code jobj} is null
076     * @throws org.jsimpledb.core.DeletedObjectException if {@code jobj} does not exist in {@code jtx}
077     * @throws org.jsimpledb.core.UnknownFieldException  if {@code jobj} has a schema version that
078     *  does not contain the affected field, or in which the affected field has a different type
079     * @throws RuntimeException if there is some other incompatibility between this change and the target object,
080     *  for example, setting a list element at an index that is out of bounds
081     * @throws org.jsimpledb.kv.StaleTransactionException if {@code jtx} is no longer usable
082     */
083    public abstract void apply(JTransaction jtx, JObject jobj);
084
085    /**
086     * Apply this change to the object associated with this instance in the given transaction.
087     *
088     * <p>
089     * This is a convenience method, equivalent to:
090     * <blockquote><code>
091     * apply(jtx, this.getJObject());
092     * </code></blockquote>
093     *
094     * @param jtx transaction in which to apply this change
095     * @throws IllegalArgumentException if {@code jtx} is null
096     */
097    public void apply(JTransaction jtx) {
098        if (jtx == null)
099            throw new IllegalArgumentException("null jtx");
100        this.apply(jtx, this.getJObject());
101    }
102
103    /**
104     * Apply this change to the transaction associated with the current thread.
105     *
106     * <p>
107     * This is a convenience method, equivalent to:
108     * <blockquote><code>
109     * apply(JTransaction.getCurrent())
110     * </code></blockquote>
111     *
112     * @throws IllegalStateException if there is no {@link JTransaction} associated with the current thread
113     */
114    public void apply() {
115        this.apply(JTransaction.getCurrent());
116    }
117
118    /**
119     * Apply this change to the specified object.
120     *
121     * <p>
122     * This is a convenience method, equivalent to:
123     * <blockquote><code>
124     * apply(obj.getTransaction(), jobj);
125     * </code></blockquote>
126     *
127     * @param jobj object to which to apply this change
128     * @throws IllegalStateException if there is no {@link JTransaction} associated with {@code jobj}
129     * @throws IllegalArgumentException if {@code jobj} is null
130     */
131    public void apply(JObject jobj) {
132        if (jobj == null)
133            throw new IllegalArgumentException("null jobj");
134        this.apply(jobj.getTransaction(), jobj);
135    }
136
137// Object
138
139    @Override
140    public boolean equals(Object obj) {
141        if (obj == this)
142            return true;
143        if (obj == null || obj.getClass() != this.getClass())
144            return false;
145        final Change<?> that = (Change<?>)obj;
146        return this.jobj.equals(that.jobj);
147    }
148
149    @Override
150    public int hashCode() {
151        return this.jobj.hashCode();
152    }
153}
