006package org.jsimpledb.change;
007
008import org.jsimpledb.CopyState;
009import org.jsimpledb.JObject;
010import org.jsimpledb.JTransaction;
011
012/**
013 * Creates a new {@link Change} object based on an existing one where the {@link JObject}s referred to by the
014 * new {@link Change} are copies in a different transaction of the originals. This is useful to allow database
015 * change information to be accessed after the transaction in which the change occured has completed.
016 *
017 * <p>
018 * Each instance has an internal {@link CopyState} used to avoid redundant copies, accessible via {@link #getCopyState}.
019 * </p>
020 */
021public class ChangeCopier implements ChangeSwitch<Change<?>> {
022
023    protected final JTransaction dest;
024    protected final CopyState copyState = new CopyState();
025
026    /**
027     * Primary constructor.
028     *
029     * @param dest destination transaction for copied {@link JObject}s
030     * @throws IllegalArgumentException if {@code dest} is null
031     */
032    public ChangeCopier(JTransaction dest) {
033        if (dest == null)
034            throw new IllegalArgumentException("null dest");
035        this.dest = dest;
036    }
037
038    /**
039     * "Snapshot" constructor for when the destination transaction is the "snapshot" transaction of the transaction
040     * associated with the current thread.
041     *
042     * <p>
043     * This is a convenience constructor, equivalent to:
044     * <blockquote><code>
045     * ChangeCopier(JTransaction.getCurrent().getSnapshotTransaction())
046     * </code></blockquote>
047     *
048     * @throws IllegalStateException if this is not a snapshot instance and there is no {@link JTransaction}
049     *  associated with the current thread
050     */
051    public ChangeCopier() {
052        this(JTransaction.getCurrent().getSnapshotTransaction());
053    }
054
055    /**
056     * Get the destination transaction configured in this instance.
057     *
058     * @return destination transaction
059     */
060    public JTransaction getDestinationTransaction() {
061        return this.dest;
062    }
063
064    /**
065     * Get the {@link CopyState} used by this instance.
066     *
067     * @return associated copy state
068     */
069    public CopyState getCopyState() {
070        return this.copyState;
071    }
072
073    @Override
074    public <T> ObjectCreate<T> caseObjectCreate(ObjectCreate<T> change) {
075        return new ObjectCreate<T>(this.copyIfReference(change.getObject()));
076    }
077
078    @Override
079    public <T> ObjectDelete<T> caseObjectDelete(ObjectDelete<T> change) {
080        return new ObjectDelete<T>(this.copyIfReference(change.getObject()));
081    }
082
083    @Override
084    public <T, E> ListFieldAdd<T, E> caseListFieldAdd(ListFieldAdd<T, E> change) {
085        return new ListFieldAdd<T, E>(this.copyIfReference(change.getObject()), change.getStorageId(), change.getFieldName(),
086          change.getIndex(), this.copyIfReference(change.getElement()));
087    }
088
089    @Override
090    public <T> ListFieldClear<T> caseListFieldClear(ListFieldClear<T> change) {
091        return new ListFieldClear<T>(this.copyIfReference(change.getObject()), change.getStorageId(), change.getFieldName());
092    }
093
094    @Override
095    public <T, E> ListFieldRemove<T, E> caseListFieldRemove(ListFieldRemove<T, E> change) {
096        return new ListFieldRemove<T, E>(this.copyIfReference(change.getObject()), change.getStorageId(), change.getFieldName(),
097          change.getIndex(), this.copyIfReference(change.getElement()));
098    }
099
100    @Override
101    public <T, E> ListFieldReplace<T, E> caseListFieldReplace(ListFieldReplace<T, E> change) {
102        return new ListFieldReplace<T, E>(this.copyIfReference(change.getObject()), change.getStorageId(), change.getFieldName(),
103          change.getIndex(), this.copyIfReference(change.getOldValue()), this.copyIfReference(change.getNewValue()));
104    }
105
106    @Override
107    public <T, K, V> MapFieldAdd<T, K, V> caseMapFieldAdd(MapFieldAdd<T, K, V> change) {
108        return new MapFieldAdd<T, K, V>(this.copyIfReference(change.getObject()), change.getStorageId(), change.getFieldName(),
109          this.copyIfReference(change.getKey()), this.copyIfReference(change.getValue()));
110    }
111
112    @Override
113    public <T> MapFieldClear<T> caseMapFieldClear(MapFieldClear<T> change) {
114        return new MapFieldClear<T>(this.copyIfReference(change.getObject()), change.getStorageId(), change.getFieldName());
115    }
116
117    @Override
118    public <T, K, V> MapFieldRemove<T, K, V> caseMapFieldRemove(MapFieldRemove<T, K, V> change) {
119        return new MapFieldRemove<T, K, V>(this.copyIfReference(change.getObject()), change.getStorageId(), change.getFieldName(),
120          this.copyIfReference(change.getKey()), this.copyIfReference(change.getValue()));
121    }
122
123    @Override
124    public <T, K, V> MapFieldReplace<T, K, V> caseMapFieldReplace(MapFieldReplace<T, K, V> change) {
125        return new MapFieldReplace<T, K, V>(this.copyIfReference(change.getObject()), change.getStorageId(), change.getFieldName(),
126          this.copyIfReference(change.getKey()), this.copyIfReference(change.getOldValue()),
127          this.copyIfReference(change.getNewValue()));
128    }
129
130    @Override
131    public <T, E> SetFieldAdd<T, E> caseSetFieldAdd(SetFieldAdd<T, E> change) {
132        return new SetFieldAdd<T, E>(this.copyIfReference(change.getObject()), change.getStorageId(), change.getFieldName(),
133          this.copyIfReference(change.getElement()));
134    }
135
136    @Override
137    public <T> SetFieldClear<T> caseSetFieldClear(SetFieldClear<T> change) {
138        return new SetFieldClear<T>(this.copyIfReference(change.getObject()), change.getStorageId(), change.getFieldName());
139    }
140
141    @Override
142    public <T, E> SetFieldRemove<T, E> caseSetFieldRemove(SetFieldRemove<T, E> change) {
143        return new SetFieldRemove<T, E>(this.copyIfReference(change.getObject()), change.getStorageId(), change.getFieldName(),
144          this.copyIfReference(change.getElement()));
145    }
146
147    @Override
148    public <T, V> SimpleFieldChange<T, V> caseSimpleFieldChange(SimpleFieldChange<T, V> change) {
149        return new SimpleFieldChange<T, V>(this.copyIfReference(change.getObject()), change.getStorageId(), change.getFieldName(),
150          this.copyIfReference(change.getOldValue()), this.copyIfReference(change.getNewValue()));
151    }
152
153    @SuppressWarnings("unchecked")
154    private <T> T copyIfReference(T obj) {
155        return obj instanceof JObject ? (T)this.copy((JObject)obj) : obj;
156    }
157
158    /**
159     * Copy the given {@link JObject} into the destination transaction.
160     *
161     * <p>
162     * The implementation in {@link ChangeCopier} invokes {@code jobj.copyTo(this.dest, null, this.getCopyState())} unless
163     * {@code jobj} does not exist, in which case it is not copied (but the
164     * {@linkplain JTransaction#getJObject(ObjId) corresponding} {@link JObject} is still returned).
165     * Subclasses may override to copy additional objects referenced by {@code jobj} as needed.
166     * </p>
167     *
168     * @param jobj original object
169     * @return copied object in {@link #dest}
170     * @throws IllegalArgumentException if {@code jobj} is null
171     */
172    @SuppressWarnings("unchecked")
173    protected JObject copy(JObject jobj) {
174        if (jobj == null)
175            throw new IllegalArgumentException("null jobj");
176        return !jobj.exists() ? this.dest.getJObject(jobj) : jobj.copyTo(this.dest, null, this.getCopyState());
177    }
178}
