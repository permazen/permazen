006package org.jsimpledb.change;
007
008import com.google.common.base.Preconditions;
009
010import org.jsimpledb.CopyState;
011import org.jsimpledb.JObject;
012import org.jsimpledb.JTransaction;
013
014/**
015 * Creates a new {@link Change} object based on an existing one where the {@link JObject}s referred to by the
016 * new {@link Change} are copies in a different transaction of the originals. This is useful to allow database
017 * change information to be accessed after the transaction in which the change occured has completed.
018 *
019 * <p>
020 * Each instance has an internal {@link CopyState} used to avoid redundant copies, accessible via {@link #getCopyState}.
021 * </p>
022 */
023public class ChangeCopier implements ChangeSwitch<Change<?>> {
024
025    protected final JTransaction dest;
026    protected final CopyState copyState = new CopyState();
027
028    /**
029     * Primary constructor.
030     *
031     * @param dest destination transaction for copied {@link JObject}s
032     * @throws IllegalArgumentException if {@code dest} is null
033     */
034    public ChangeCopier(JTransaction dest) {
035        Preconditions.checkArgument(dest != null, "null dest");
036        this.dest = dest;
037    }
038
039    /**
040     * "Snapshot" constructor for when the destination transaction is the "snapshot" transaction of the transaction
041     * associated with the current thread.
042     *
043     * <p>
044     * This is a convenience constructor, equivalent to:
045     * <blockquote><code>
046     * ChangeCopier(JTransaction.getCurrent().getSnapshotTransaction())
047     * </code></blockquote>
048     *
049     * @throws IllegalStateException if this is not a snapshot instance and there is no {@link JTransaction}
050     *  associated with the current thread
051     */
052    public ChangeCopier() {
053        this(JTransaction.getCurrent().getSnapshotTransaction());
054    }
055
056    /**
057     * Get the destination transaction configured in this instance.
058     *
059     * @return destination transaction
060     */
061    public JTransaction getDestinationTransaction() {
062        return this.dest;
063    }
064
065    /**
066     * Get the {@link CopyState} used by this instance.
067     *
068     * @return associated copy state
069     */
070    public CopyState getCopyState() {
071        return this.copyState;
072    }
073
074    @Override
075    public <T> ObjectCreate<T> caseObjectCreate(ObjectCreate<T> change) {
076        return new ObjectCreate<T>(this.copyIfReference(change.getObject()));
077    }
078
079    @Override
080    public <T> ObjectDelete<T> caseObjectDelete(ObjectDelete<T> change) {
081        return new ObjectDelete<T>(this.copyIfReference(change.getObject()));
082    }
083
084    @Override
085    public <T, E> ListFieldAdd<T, E> caseListFieldAdd(ListFieldAdd<T, E> change) {
086        return new ListFieldAdd<T, E>(this.copyIfReference(change.getObject()), change.getStorageId(), change.getFieldName(),
087          change.getIndex(), this.copyIfReference(change.getElement()));
088    }
089
090    @Override
091    public <T> ListFieldClear<T> caseListFieldClear(ListFieldClear<T> change) {
092        return new ListFieldClear<T>(this.copyIfReference(change.getObject()), change.getStorageId(), change.getFieldName());
093    }
094
095    @Override
096    public <T, E> ListFieldRemove<T, E> caseListFieldRemove(ListFieldRemove<T, E> change) {
097        return new ListFieldRemove<T, E>(this.copyIfReference(change.getObject()), change.getStorageId(), change.getFieldName(),
098          change.getIndex(), this.copyIfReference(change.getElement()));
099    }
100
101    @Override
102    public <T, E> ListFieldReplace<T, E> caseListFieldReplace(ListFieldReplace<T, E> change) {
103        return new ListFieldReplace<T, E>(this.copyIfReference(change.getObject()), change.getStorageId(), change.getFieldName(),
104          change.getIndex(), this.copyIfReference(change.getOldValue()), this.copyIfReference(change.getNewValue()));
105    }
106
107    @Override
108    public <T, K, V> MapFieldAdd<T, K, V> caseMapFieldAdd(MapFieldAdd<T, K, V> change) {
109        return new MapFieldAdd<T, K, V>(this.copyIfReference(change.getObject()), change.getStorageId(), change.getFieldName(),
110          this.copyIfReference(change.getKey()), this.copyIfReference(change.getValue()));
111    }
112
113    @Override
114    public <T> MapFieldClear<T> caseMapFieldClear(MapFieldClear<T> change) {
115        return new MapFieldClear<T>(this.copyIfReference(change.getObject()), change.getStorageId(), change.getFieldName());
116    }
117
118    @Override
119    public <T, K, V> MapFieldRemove<T, K, V> caseMapFieldRemove(MapFieldRemove<T, K, V> change) {
120        return new MapFieldRemove<T, K, V>(this.copyIfReference(change.getObject()), change.getStorageId(), change.getFieldName(),
121          this.copyIfReference(change.getKey()), this.copyIfReference(change.getValue()));
122    }
123
124    @Override
125    public <T, K, V> MapFieldReplace<T, K, V> caseMapFieldReplace(MapFieldReplace<T, K, V> change) {
126        return new MapFieldReplace<T, K, V>(this.copyIfReference(change.getObject()), change.getStorageId(), change.getFieldName(),
127          this.copyIfReference(change.getKey()), this.copyIfReference(change.getOldValue()),
128          this.copyIfReference(change.getNewValue()));
129    }
130
131    @Override
132    public <T, E> SetFieldAdd<T, E> caseSetFieldAdd(SetFieldAdd<T, E> change) {
133        return new SetFieldAdd<T, E>(this.copyIfReference(change.getObject()), change.getStorageId(), change.getFieldName(),
134          this.copyIfReference(change.getElement()));
135    }
136
137    @Override
138    public <T> SetFieldClear<T> caseSetFieldClear(SetFieldClear<T> change) {
139        return new SetFieldClear<T>(this.copyIfReference(change.getObject()), change.getStorageId(), change.getFieldName());
140    }
141
142    @Override
143    public <T, E> SetFieldRemove<T, E> caseSetFieldRemove(SetFieldRemove<T, E> change) {
144        return new SetFieldRemove<T, E>(this.copyIfReference(change.getObject()), change.getStorageId(), change.getFieldName(),
145          this.copyIfReference(change.getElement()));
146    }
147
148    @Override
149    public <T, V> SimpleFieldChange<T, V> caseSimpleFieldChange(SimpleFieldChange<T, V> change) {
150        return new SimpleFieldChange<T, V>(this.copyIfReference(change.getObject()), change.getStorageId(), change.getFieldName(),
151          this.copyIfReference(change.getOldValue()), this.copyIfReference(change.getNewValue()));
152    }
153
154    @SuppressWarnings("unchecked")
155    private <T> T copyIfReference(T obj) {
156        return obj instanceof JObject ? (T)this.copy((JObject)obj) : obj;
157    }
158
159    /**
160     * Copy the given {@link JObject} into the destination transaction.
161     *
162     * <p>
163     * The implementation in {@link ChangeCopier} invokes {@code jobj.copyTo(this.dest, null, this.getCopyState())} unless
164     * {@code jobj} does not exist, in which case it is not copied (but the
165     * {@linkplain JTransaction#getJObject(ObjId) corresponding} {@link JObject} is still returned).
166     * Subclasses may override to copy additional objects referenced by {@code jobj} as needed.
167     * </p>
168     *
169     * @param jobj original object
170     * @return copied object in {@link #dest}
171     * @throws IllegalArgumentException if {@code jobj} is null
172     */
173    @SuppressWarnings("unchecked")
174    protected JObject copy(JObject jobj) {
175        Preconditions.checkArgument(jobj != null, "null jobj");
176        return !jobj.exists() ? this.dest.getJObject(jobj) : jobj.copyTo(this.dest, null, this.getCopyState());
177    }
178}
