006package org.jsimpledb.change;
007
008import org.jsimpledb.JObject;
009import org.jsimpledb.JTransaction;
010
011/**
012 * Notification object that gets passed to {@link org.jsimpledb.annotation.OnChange &#64;OnChange}-annotated methods
013 * when an element is removed from a list field.
014 *
015 * @param <T> the type of the object containing the changed field
016 * @param <E> the type of the changed list's elements
017 */
018public class ListFieldRemove<T, E> extends ListFieldChange<T> {
019
020    private final int index;
021    private final E element;
022
023    /**
024     * Constructor.
025     *
026     * @param jobj Java object containing the list field that changed
027     * @param storageId the storage ID of the affected field
028     * @param fieldName the name of the field that changed
029     * @param index index at which the removal occurred
030     * @param element the element that was removed
031     * @throws IllegalArgumentException if {@code jobj} or {@code fieldName} is null
032     */
033    public ListFieldRemove(T jobj, int storageId, String fieldName, int index, E element) {
034        super(jobj, storageId, fieldName);
035        this.index = index;
036        this.element = element;
037    }
038
039    @Override
040    public <R> R visit(ChangeSwitch<R> target) {
041        return target.caseListFieldRemove(this);
042    }
043
044    @Override
045    public void apply(JTransaction jtx, JObject jobj) {
046        jtx.readListField(jobj, this.getStorageId(), false).remove(this.index);
047    }
048
049    /**
050     * Get the list index from which the element was removed.
051     *
052     * @return the index at which the value was removed
053     */
054    public int getIndex() {
055        return this.index;
056    }
057
058    /**
059     * Get the element that was removed from the list.
060     *
061     * @return the value removed from the list
062     */
063    public E getElement() {
064        return this.element;
065    }
066
067// Object
068
069    @Override
070    public boolean equals(Object obj) {
071        if (obj == this)
072            return true;
073        if (!super.equals(obj))
074            return false;
075        final ListFieldRemove<?, ?> that = (ListFieldRemove<?, ?>)obj;
076        return this.index == that.index && (this.element != null ? this.element.equals(that.element) : that.element == null);
077    }
078
079    @Override
080    public int hashCode() {
081        return super.hashCode() ^ this.index ^ (this.element != null ? this.element.hashCode() : 0);
082    }
083
084    @Override
085    public String toString() {
086        return "ListFieldRemove[object=" + this.getObject() + ",field=\"" + this.getFieldName() + "\",index="
087          + this.index + ",element=" + this.element + "]";
088    }
089}
