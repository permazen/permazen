006package org.jsimpledb.change;

008/**
009 * Adpater class for the {@link ChangeSwitch} interface.
010 *
011 * @param <R> method return type
012 */
013public class ChangeAdapter<R> implements ChangeSwitch<R> {
014
015    /**
016     * Handle a {@link ObjectCreate} event.
017     *
018     * <p>
019     * The implementation in {@link ChangeAdapter} delegates to {@link #caseChange caseChange()}.
020     * </p>
021     */
022    @Override
023    public <T> R caseObjectCreate(ObjectCreate<T> change) {
024        return this.caseChange(change);
025    }
026
027    /**
