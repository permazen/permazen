006package org.jsimpledb.change;
007
008/**
009 * Notification object that gets passed to {@link org.jsimpledb.annotation.OnChange &#64;OnChange}-annotated methods
010 * when a map field changes.
011 *
012 * @param <T> the type of the object containing the changed field
013 */
014public abstract class MapFieldChange<T> extends FieldChange<T> {
015
016    /**
017     * Constructor.
018     *
019     * @param jobj Java object containing the map field that changed
020     * @param storageId the storage ID of the affected field
021     * @param fieldName the name of the field that changed
022     * @throws IllegalArgumentException if {@code jobj} or {@code fieldName} is null
023     */
024    protected MapFieldChange(T jobj, int storageId, String fieldName) {
025        super(jobj, storageId, fieldName);
026    }
027}
