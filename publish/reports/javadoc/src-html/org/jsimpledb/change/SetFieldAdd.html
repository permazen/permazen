006package org.jsimpledb.change;
007
008import java.util.Set;
009
010import org.jsimpledb.JObject;
011import org.jsimpledb.JTransaction;
012
013/**
014 * Notification object that gets passed to {@link org.jsimpledb.annotation.OnChange &#64;OnChange}-annotated methods
015 * when an element is added to a set field.
016 *
017 * @param <T> the type of the object containing the changed field
018 * @param <E> the type of the changed set's elements
019 */
020public class SetFieldAdd<T, E> extends SetFieldChange<T> {
021
022    private final E element;
023
024    /**
025     * Constructor.
026     *
027     * @param jobj Java object containing the set field that changed
028     * @param storageId the storage ID of the affected field
029     * @param fieldName the name of the field that changed
030     * @param element the element that was added
031     * @throws IllegalArgumentException if {@code jobj} or {@code fieldName} is null
032     */
033    public SetFieldAdd(T jobj, int storageId, String fieldName, E element) {
034        super(jobj, storageId, fieldName);
035        this.element = element;
036    }
037
038    @Override
039    public <R> R visit(ChangeSwitch<R> target) {
040        return target.caseSetFieldAdd(this);
041    }
042
043    @Override
044    @SuppressWarnings("unchecked")
045    public void apply(JTransaction jtx, JObject jobj) {
046        ((Set<E>)jtx.readSetField(jobj, this.getStorageId(), false)).add(this.element);
047    }
048
049    /**
050     * Get the element that was added to the set.
051     *
052     * @return the newly added value
053     */
054    public E getElement() {
055        return this.element;
056    }
057
058// Object
059
060    @Override
061    public boolean equals(Object obj) {
062        if (obj == this)
063            return true;
064        if (!super.equals(obj))
065            return false;
066        final SetFieldAdd<?, ?> that = (SetFieldAdd<?, ?>)obj;
067        return this.element != null ? this.element.equals(that.element) : that.element == null;
068    }
069
070    @Override
071    public int hashCode() {
072        return super.hashCode() ^ (this.element != null ? this.element.hashCode() : 0);
073    }
074
075    @Override
076    public String toString() {
077        return "SetFieldAdd[object=" + this.getObject() + ",field=\"" + this.getFieldName() + "\",element=" + this.element + "]";
078    }
079}
