006package org.jsimpledb.change;
007
008import java.util.Map;
009
010import org.jsimpledb.JObject;
011import org.jsimpledb.JTransaction;
012
013/**
014 * Notification object that gets passed to {@link org.jsimpledb.annotation.OnChange &#64;OnChange}-annotated methods
015 * when a new key/value pair is added to a map field.
016 *
017 * @param <T> the type of the object containing the changed field
018 * @param <K> the type of the changed map's key
019 * @param <V> the type of the changed map's value
020 */
021public class MapFieldAdd<T, K, V> extends MapFieldChange<T> {
022
023    private final K key;
024    private final V value;
025
026    /**
027     * Constructor.
028     *
029     * @param jobj Java object containing the map field that changed
030     * @param storageId the storage ID of the affected field
031     * @param fieldName the name of the field that changed
032     * @param key the key of the new key/value pair
033     * @param value the value of the new key/value pair
034     * @throws IllegalArgumentException if {@code jobj} or {@code fieldName} is null
035     */
036    public MapFieldAdd(T jobj, int storageId, String fieldName, K key, V value) {
037        super(jobj, storageId, fieldName);
038        this.key = key;
039        this.value = value;
040    }
041
042    @Override
043    public <R> R visit(ChangeSwitch<R> target) {
044        return target.caseMapFieldAdd(this);
045    }
046
047    @Override
048    @SuppressWarnings("unchecked")
049    public void apply(JTransaction jtx, JObject jobj) {
050        ((Map<K, V>)jtx.readMapField(jobj, this.getStorageId(), false)).put(this.key, this.value);
051    }
052
053    /**
054     * Get the key of the new key/value pair that was added.
055     *
056     * @return the key of the newly added key/value pair
057     */
058    public K getKey() {
059        return this.key;
060    }
061
062    /**
063     * Get the value of the new key/value pair that was added.
064     *
065     * @return the value of the newly added key/value pair
066     */
067    public V getValue() {
068        return this.value;
069    }
070
071// Object
072
073    @Override
074    public boolean equals(Object obj) {
075        if (obj == this)
076            return true;
077        if (!super.equals(obj))
078            return false;
079        final MapFieldAdd<?, ?, ?> that = (MapFieldAdd<?, ?, ?>)obj;
080        return (this.key != null ? this.key.equals(that.key) : that.key == null)
081          && (this.value != null ? this.value.equals(that.value) : that.value == null);
082    }
083
084    @Override
085    public int hashCode() {
086        return super.hashCode() ^ (this.key != null ? this.key.hashCode() : 0) ^ (this.value != null ? this.value.hashCode() : 0);
087    }
088
089    @Override
090    public String toString() {
091        return "MapFieldAdd[object=" + this.getObject() + ",field=\"" + this.getFieldName() + "\",key="
092          + this.key + ",value=" + this.value + "]";
093    }
094}
