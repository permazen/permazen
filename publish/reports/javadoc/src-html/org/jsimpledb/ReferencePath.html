021/**
022 * Reference paths.
023 *
024 * <p>
025 * A Reference path consists of:
026 * <ul>
027 *  <li>A {@linkplain #getStartType starting Java object type},</li>
028 *  <li>A final {@linkplain #getTargetType target object type} and {@linkplain #getTargetField target field}
029 *      within that type, and</li>
030 *  <li>An intermediate chain of zero or more {@linkplain #getReferenceFields reference fields} through which an
031 *      instance of the target type can be reached from an instance of the starting type.</li>
032 * </ul>
033 *
034 * <p>
035 * Reference paths can be described by the combination of (a) the starting Java object type, and (b) the path of reference
036 * fields and final target field in a {@link String} form consisting of field names separated by period ({@code "."}) characters.
037 * All of the fields except the last must be reference fields.
038 * </p>
039 *
040 * <p>
041 * For example, path {@code "parent.age"} starting from object type {@code Person} might refer to the age of
042 * the parent of a {@code Person}.
043 * </p>
044 *
045 * <p>
046 * When a complex field appears in a reference path, both the name of the complex field and the specific sub-field
047 * being traversed should appear, e.g., {@code "mymap.key.somefield"}. For set and list fields, the (only) sub-field is
048 * named {@code "element"}, while for map fields the sub-fields are named {@code "key"} and {@code "value"}.
049 * </p>
050 *
051 * <p>
052 * <b>Fields of Sub-Types</b>
053 * </p>
054 *
055 * <p>
056 * In some cases, a field may not exist in a Java object type, but it does exist in a some sub-type of that type. For example:
057 * </p>
058 *
059 * <pre>
060 * &#64;JSimpleClass(storageId = 10)
061 * public class Person {
062 *
063 *     &#64;JSimpleSetField(storageId = 11)
064 *     public abstract Set&lt;Person&gt; <b>getFriends</b>();
065 *
066 *     &#64;OnChange("friends.element.<b>name</b>")
067 *     private void friendNameChanged(SimpleFieldChange&lt;NamedPerson, String&gt; change) {
068 *         // ... do whatever
069 *     }
070 * }
071 *
072 * &#64;JSimpleClass(storageId = 20)
073 * public class NamedPerson extends Person {
074 *
075 *     &#64;JSimpleField(storageId = 21)
076 *     public abstract String <b>getName</b>();
077 *     public abstract void setName(String name);
078 * }
079 * </pre>
080 * Here the path {@code "friends.element.name"} is technically incorrect because {@code "friends.element"}
081 * has type {@code Person}, and {@code "name"} is a field of {@code NamedPerson}, not {@code Person}. However, this will
082 * still work as long as there is no ambiguity, i.e., in this example, there are no other sub-types of {@code Person}
083 * with a field named {@code "name"}. Note also in the example above the
084 * {@link org.jsimpledb.change.SimpleFieldChange} parameter to the method
085 * {@code friendNameChanged()} necessarily has generic type {@code NamedPerson}, not {@code Person}.
086 *
087 * <p>
088 * In cases where multiple sub-types of a common super-type type have fields with the same name but different storage IDs,
089 * the storage ID may be explicitly specified as a suffix, for example, {@code "name#123"}.
090 * </p>
091 *
092 * <p>
093 * Reference paths are created via {@link JSimpleDB#parseReferencePath JSimpleDB.parseReferencePath()}.
094 * </p>
095 *
096 * @see JSimpleDB#parseReferencePath JSimpleDB.parseReferencePath()
097 */
098public class ReferencePath {
099
100    final Class<?> startType;
101    final Class<?> targetType;
102    final JFieldInfo targetFieldInfo;
103    final JComplexFieldInfo targetSuperFieldInfo;
104    final ArrayList<JReferenceFieldInfo> referenceFieldInfos = new ArrayList<>();
105    final String path;
106
107    private final Logger log = LoggerFactory.getLogger(this.getClass());
108
109    /**
110     * Constructor.
111     *
112     * @param jdb {@link JSimpleDB} against which to resolve object and field names
113     * @param startType starting Java type for the path
114     * @param path dot-separated path of zero or more reference fields, followed by a target field
115     * @param lastIsSubField true if the last field can be a complex sub-field but not a complex field, false for the reverse,
116     *  or null for don't care
117     * @throws IllegalArgumentException if {@code jdb}, {@code startType}, or {@code path} is null
118     * @throws IllegalArgumentException if {@code path} is invalid
119     */
120    ReferencePath(JSimpleDB jdb, Class<?> startType, String path, Boolean lastIsSubField) {
121
122        // Sanity check
123        if (jdb == null)
124            throw new IllegalArgumentException("null jdb");
125        if (startType == null)
126            throw new IllegalArgumentException("null startType");
127        if (path == null)
128            throw new IllegalArgumentException("null path");
129        final String errorPrefix = "invalid path `" + path + "': ";
130        if (path.length() == 0)
131            throw new IllegalArgumentException(errorPrefix + "path is empty");
132        this.startType = startType;
133        this.path = path;
134
135        // Split the path into field names
136        final ArrayDeque<String> fieldNames = new ArrayDeque<>();
137        while (true) {
138            final int dot = path.indexOf('.');
139            if (dot == -1) {
140                if (path.length() == 0)
141                    throw new IllegalArgumentException(errorPrefix + "ends in `.'");
142                fieldNames.add(path);
143                break;
144            }
145            if (dot == 0)
146                throw new IllegalArgumentException(errorPrefix + "contains an empty path component");
147            fieldNames.add(path.substring(0, dot));
148            path = path.substring(dot + 1);
149        }
150
151        // Initialize loop state
152        Class<?> currentType = this.startType;
153        JFieldInfo fieldInfo = null;
154        JComplexFieldInfo superFieldInfo = null;
155        if (this.log.isTraceEnabled()) {
156            this.log.trace("RefPath: START: startType=" + this.startType + " path=" + fieldNames
157              + " lastIsSubField=" + lastIsSubField);
158        }
159
160        // Parse field names
161        final ArrayList<Integer> storageIds = new ArrayList<>();
162        assert !fieldNames.isEmpty();
163        while (!fieldNames.isEmpty()) {
164            final String fieldName = fieldNames.removeFirst();
165            String description = "field `" + fieldName + "' in type " + currentType;
166
167            // Get explicit storage ID, if any
168            final int hash = fieldName.indexOf('#');
169            int explicitStorageId = 0;
170            final String searchName;
171            if (hash != -1) {
172                try {
173                    explicitStorageId = Integer.parseInt(fieldName.substring(hash + 1));
174                } catch (NumberFormatException e) {
175                    throw new IllegalArgumentException(errorPrefix + "invalid field name `" + fieldName + "'");
176                }
177                searchName = fieldName.substring(0, hash);
178            } else
179                searchName = fieldName;
180
181            // Logging
182            if (this.log.isTraceEnabled())
183                this.log.trace("RefPath: [" + searchName + "] currentType=" + currentType + " storageId=" + explicitStorageId);
184
185            // Find all JFields matching 'fieldName' in some JClass whose type matches 'typeToken'
186            final HashMap<JClass<?>, JField> matchingFields = new HashMap<>();
187            for (JClass<?> jclass : jdb.jclasses.values()) {
188                if (!currentType.isAssignableFrom(jclass.type))
189                    continue;
190                final JField jfield = jclass.jfieldsByName.get(searchName);
191                if (jfield == null)
192                    continue;
193                if (explicitStorageId != 0 && jfield.storageId != explicitStorageId)
194                    continue;
195                matchingFields.put(jclass, jfield);
196            }
197
198            // Logging
199            if (this.log.isTraceEnabled())
200                this.log.trace("RefPath: matching fields: " + matchingFields);
201
202            // Check matching fields and verify they all have the same storage ID
203            if (matchingFields.isEmpty()) {
204                throw new IllegalArgumentException(errorPrefix + "there is no field named `" + searchName + "'"
205                  + (explicitStorageId != 0 ? " with storage ID " + explicitStorageId : "")
206                  + " in (any sub-type of) " + currentType);
207            }
208            final int fieldStorageId = matchingFields.values().iterator().next().storageId;
209            for (JField jfield : matchingFields.values()) {
210                if (jfield.storageId != fieldStorageId) {
211                    throw new IllegalArgumentException(errorPrefix + "there are multiple, non-equal fields named `"
212                      + fieldName + "' in sub-types of type " + currentType);
213                }
214            }
215            fieldInfo = jdb.getJFieldInfo(fieldStorageId, JFieldInfo.class);
216
217            // Get common supertype of all types containing the field
218            currentType = Util.findLowestCommonAncestorOfClasses(
219              Iterables.transform(matchingFields.keySet(), new JClassTypeFunction())).getRawType();
220
221            // Logging
222            if (this.log.isTraceEnabled())
223                this.log.trace("RefPath: updated currentType=" + currentType + " fieldType=" + fieldInfo.getTypeToken(currentType));
224
225            // Handle complex fields
226            superFieldInfo = null;
227            if (fieldInfo instanceof JComplexFieldInfo) {
228                final JComplexFieldInfo complexFieldInfo = (JComplexFieldInfo)fieldInfo;
229
230                // Last field?
231                if (fieldNames.isEmpty()) {
232                    if (lastIsSubField != null && lastIsSubField) {
233                        throw new IllegalArgumentException(errorPrefix + "path may not end on complex " + description
234                          + "; a sub-field of must be specified");
235                    }
236                    break;
237                }
238                superFieldInfo = complexFieldInfo;
239
240                // Get sub-field
241                final String subFieldName = fieldNames.removeFirst();
242                description = "sub-field `" + subFieldName + "' of complex " + description;
243                try {
244                    fieldInfo = complexFieldInfo.getSubFieldInfo(subFieldName);
245                } catch (IllegalArgumentException e) {
246                    throw new IllegalArgumentException(errorPrefix + "invalid " + description + ": " + e.getMessage(), e);
247                }
248
249                // Update matching fields to match sub-field
250                for (Map.Entry<JClass<?>, JField> entry : matchingFields.entrySet())
251                    entry.setValue(((JComplexField)entry.getValue()).getSubField(subFieldName));
252
253                // Logging
254                if (this.log.isTraceEnabled())
255                    this.log.trace("RefPath: [" + searchName + "." + subFieldName + "] new matching fields=" + matchingFields);
256            }
257
258            // Last field?
259            if (fieldNames.isEmpty()) {
260                if (superFieldInfo != null && Boolean.FALSE.equals(lastIsSubField)) {
261                    throw new IllegalArgumentException(errorPrefix + "path may not end on " + description
262                      + "; specify the complex field itself instead");
263                }
264                break;
265            }
266
267            // Field is not last, so it must be a reference field
268            if (!(fieldInfo instanceof JReferenceFieldInfo))
269                throw new IllegalArgumentException(errorPrefix + description + " is not a reference field");
270            final JReferenceFieldInfo referenceFieldInfo = (JReferenceFieldInfo)fieldInfo;
271
272            // Add reference field to the reference field list
273            this.referenceFieldInfos.add(referenceFieldInfo);
274
275            // Advance through the reference, using the narrowest type possible
276            currentType = Util.findLowestCommonAncestor(Iterables.transform(
277               Iterables.transform(matchingFields.values(), new CastFunction<JReferenceField>(JReferenceField.class)),
278              new JReferenceFieldTypeFunction())).getRawType();
279
280            // Logging
281            if (this.log.isTraceEnabled())
282                this.log.trace("RefPath: step through reference, new currentType=" + currentType);
283        }
284
285        // Done
286        this.targetType = currentType;
287        this.targetFieldInfo = fieldInfo;
288        this.targetSuperFieldInfo = superFieldInfo;
289
290        // Logging
291        if (this.log.isTraceEnabled()) {
292            this.log.trace("RefPath: DONE: targetType=" + this.targetType + " targetFieldInfo=" + this.targetFieldInfo
293              + " targetSuperFieldInfo=" + this.targetSuperFieldInfo + " targetFieldType=" + this.getTargetFieldType()
294              + " references=" + this.referenceFieldInfos);
295        }
296    }
297
298    /**
299     * Get the Java type of the object at which this path starts.
300     *
301     * <p>
302     * If there are zero {@linkplain #getReferenceFields reference fields} in this path, then this will
303     * equal the {@linkplain #getTargetType target object type}, or possibly a super-type if the target
304     * field exists only in a sub-type.
305     * </p>
306     *
307     * @return the Java type at which this reference path starts
308     */
309    public Class<?> getStartType() {
310        return this.startType;
311    }
312
313    /**
314     * Get the Java type of the object at which this path ends.
315     *
316     * <p>
317     * If there are zero {@linkplain #getReferenceFields reference fields} in this path, then this will
318     * equal the Java type of the {@linkplain #getStartType starting object type}, or possibly a sub-type
319     * if the target field exists only in a sub-type.
320     * </p>
321     *
322     * @return the Java type at which this reference path ends
323     */
324    public Class<?> getTargetType() {
325        return this.targetType;
326    }
327
328    /**
329     * Get the Java type corresponding to the field at which this path ends.
330     *
331     * @return the type of the field at which this reference path ends
332     */
333    public TypeToken<?> getTargetFieldType() {
334        return this.targetFieldInfo.getTypeToken(this.targetType);
335    }
336
337    /**
338     * Get the storage ID associated with the target field in the {@linkplain #getTargetType target object type}.
339     *
340     * <p>
341     * This is just the storage ID of the last field in the path.
342     * </p>
343     *
344     * @return the storage ID of the field at which this reference path ends
345     */
346    public int getTargetField() {
347        return this.targetFieldInfo.storageId;
348    }
349
350    /**
351     * Get the storage ID associated with the complex field containing the {@linkplain #getTargetField target field}
352     * a sub-field, in the case that the target field is a sub-field of a complex field.
353     *
354     * @return target field's complex super-field storage ID, or zero if the target field is not a complex sub-field
355     */
356    public int getTargetSuperField() {
357        return this.targetSuperFieldInfo != null ? this.targetSuperFieldInfo.storageId : 0;
358    }
359
360    /**
361     * Get the storage IDs of the reference fields in this path.
362     *
363     * <p>
364     * The path may be empty, i.e., zero references are traversed in the path.
365     * </p>
366     *
367     * <p>
368     * Otherwise, the first field is a field in the {@linkplain #getStartType starting object type} and
369     * the last field is field in some object type that refers to the {@linkplain #getTargetType target object type}.
370     * </p>
371     *
372     * @return zero or more reference field storage IDs
373     */
374    public int[] getReferenceFields() {
375        final int[] storageIds = new int[this.referenceFieldInfos.size()];
376        for (int i = 0; i < storageIds.length; i++)
377            storageIds[i] = this.referenceFieldInfos.get(i).storageId;
378        return storageIds;
379    }
380
381    /**
382     * Get the {@link String} form of the path associated with this instance.
383     */
384    @Override
385    public String toString() {
386        return this.path;
387    }
388
389// Functions
390
391    private static class JReferenceFieldTypeFunction implements Function<JReferenceField, TypeToken<?>> {
392
393        @Override
394        public TypeToken<?> apply(JReferenceField jfield) {
395            return jfield.typeToken;
396        }
397    }
398
399    private static class JClassTypeFunction implements Function<JClass<?>, Class<?>> {
400
401        @Override
402        public Class<?> apply(JClass<?> jclass) {
403            return jclass.type;
404        }
405    }
