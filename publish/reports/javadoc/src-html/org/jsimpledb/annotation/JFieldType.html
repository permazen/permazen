006package org.jsimpledb.annotation;
007
008import java.lang.annotation.Documented;
009import java.lang.annotation.ElementType;
010import java.lang.annotation.Retention;
011import java.lang.annotation.RetentionPolicy;
012import java.lang.annotation.Target;
013
014/**
015 * Java annotation for classes that define custom {@link org.jsimpledb.core.FieldType}s
016 * for a {@link org.jsimpledb.JSimpleDB}'s underlying {@link org.jsimpledb.core.Database}.
017 * These classes will be automatically instantiated and registered with the
018 * {@linkplain org.jsimpledb.core.Database#getFieldTypeRegistry associated}
019 * {@link org.jsimpledb.core.FieldTypeRegistry}.
020 *
021 * <p>
022 * Annotated classes must extend {@link org.jsimpledb.core.FieldType} and have a zero-parameter constructor.
023 * </p>
024 *
025 * <p>
026 * Note that once a certain encoding has been used for a given type name in a database, the encoding should not
027 * be changed without creating a new type (and type name), or changing the
028 * {@linkplain org.jsimpledb.core.FieldType#getEncodingSignature encoding signature}.
029 * Otherwise, {@link org.jsimpledb.core.InconsistentDatabaseException}s can result when the new type unexpectedly
030 * encounters the old encoding or vice-versa.
031 * </p>
032 */
033@Retention(RetentionPolicy.RUNTIME)
034@Target(ElementType.TYPE)
035@Documented
036public @interface JFieldType {
037}
