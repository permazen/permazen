006package org.jsimpledb.annotation;
007
008import java.lang.annotation.Documented;
009import java.lang.annotation.ElementType;
010import java.lang.annotation.Retention;
011import java.lang.annotation.RetentionPolicy;
012import java.lang.annotation.Target;
013
014/**
015 * Java annotation for Java classes that are {@link org.jsimpledb.JSimpleDB} object model types.
016 *
017 * <p>
018 * Typically such classes are declared {@code abstract} and contain {@code abstract} Java bean getter methods
019 * with {@link JField &#64;JField}, {@link JSetField &#64;JSetField}, etc. annotations that
020 * define the fields of the object type.
021 * </p>
022 *
023 * <p>
024 * The following annotations are supported on the methods of a {@link JSimpleClass &#64;JSimpleClass}-annotated class:
025 * <ul>
026 *  <li>{@link JField &#64;JField} - defines a simple value or reference field
027 *  <li>{@link JSetField &#64;JSetField} - defines a {@link java.util.NavigableSet} field
028 *  <li>{@link JListField &#64;JListField} - defines a {@link java.util.List} field
029 *  <li>{@link JMapField &#64;JMapField} - defines a {@link java.util.NavigableMap} field
030 *  <li>{@link OnChange &#64;OnChange} - annotates a method to be invoked when some field (possibly seen through
031 *      a path of references) changes
032 *  <li>{@link OnCreate &#64;OnCreate} - annotates a method to be invoked just after object creation
033 *  <li>{@link OnDelete &#64;OnDelete} - annotates a method to be invoked just prior to object deletion
034 *  <li>{@link OnVersionChange &#64;OnVersionChange} - annotates a method to be invoked when the object's schema version changes
035 *  <li>{@link Validate &#64;Validate} - annotates a method to be invoked whenever the object is (re)validated
036 * </ul>
037 *
038 * <p>
039 * The above annotations may be present on any overridden supertype method, including methods in superclasses and superinterfaces
040 * (whether or not annotated with {@link JSimpleClass &#64;JSimpleClass}).
041 * </p>
042 *
043 * <p>
044 * The subclass of the annotated class that is generated by {@link org.jsimpledb.JSimpleDB} will implement
045 * {@link org.jsimpledb.JObject}, so the (abstract) annotated class may be declared that way.
046 * The annotated class must have a zero-parameter constructor.
047 * </p>
048 */
049@Retention(RetentionPolicy.RUNTIME)
050@Target(ElementType.TYPE)
051@Documented
052public @interface JSimpleClass {
053
054    /**
055     * The name of this object type.
056     *
057     * <p>
058     * If equal to the empty string (default value),
059     * the {@linkplain Class#getSimpleName simple name} of the annotated Java class is used.
060     * </p>
061     *
062     * @return object type name
063     */
064    String name() default "";
065
066    /**
067     * Storage ID for this object type. Value should be positive;
068     * if zero, the configured {@link org.jsimpledb.StorageIdGenerator} will be consulted to auto-generate a value.
069     *
070     * @return object type storage ID
071     * @see org.jsimpledb.StorageIdGenerator#generateClassStorageId StorageIdGenerator.generateClassStorageId()
072     */
073    int storageId() default 0;
074
075    /**
076     * Composite indexes (i.e., indexes on two or more fields) defined on this type.
077     *
078     * @return object type composite indexes
079     */
080    JCompositeIndex[] compositeIndexes() default {};
081
082    /**
083     * Whether to automatically generate database fields from un-annotated abstract Java bean methods.
084     *
085     * <p>
086     * If true, database fields corresponding to all <b>abstract</b> bean property getter methods will
087     * be auto-generated even if there is no {@link JField &#64;JField}, {@link JSetField &#64;JSetField},
088     * {@link JListField &#64;JListField}, or {@link JMapField &#64;JMapField} annotation. Note <i>this includes
089     * superclass and interface methods</i>.
090     * </p>
091     *
092     * <p>
093     * Getter methods with return type assignable to {@link java.util.Set}, {@link java.util.List}, and {@link java.util.Map}
094     * will cause the corresponding collection fields to be created; other getter/setter method pairs will cause
095     * the corresponding simple fields to be generated. Auto-generation of storage ID's is performed by the
096     * configured {@link org.jsimpledb.StorageIdGenerator}.
097     * </p>
098     *
099     * @return whether to auto-generate fields from abstract methods
100     * @see org.jsimpledb.JSimpleDBFactory#setStorageIdGenerator JSimpleDBFactory.setStorageIdGenerator()
101     */
102    boolean autogenFields() default true;
103}
