006package org.jsimpledb.annotation;
007
008import java.lang.annotation.Documented;
009import java.lang.annotation.ElementType;
010import java.lang.annotation.Retention;
011import java.lang.annotation.RetentionPolicy;
012import java.lang.annotation.Target;
013
014/**
015 * Annotation for methods that are to be invoked whenever an object's schema version has just changed.
016 *
017 * <p><b>Method Parameters</b></p>
018 *
019 * <p>
020 * The annotated method must be an instance method (i.e., not static), return void, and
021 * take one, two, or all three of the following parameters in order:
022 * <ol>
023 * <li>{@code int oldVersion} - previous schema version; should be present only if {@link #oldVersion} is zero</li>
024 * <li>{@code int newVersion} - new schema version; should be present only if {@link #newVersion} is zero</li>
025 * <li>{@code Map<Integer, Object> oldValues} <i>or</i> {@code Map<String, Object> oldValues} - immutable map containing
026 *      all field values from the previous version of the object, indexed by either storage ID or field name.</li>
027 * </ol>
028 *
029 * <p>
030 * If a class has multiple {@link OnVersionChange &#64;OnVersionChange}-annotated methods, methods with more specific
031 * constraint(s) (i.e., non-zero {@link #oldVersion} and/or {@link #newVersion}) will be invoked first.
032 * </p>
033 *
034 * <p><b>Incompatible Schema Changes</b></p>
035 *
036 * <p>
037 * JSimpleDB supports arbitrary Java model schema changes across schema versions, including adding and removing Java types.
038 * As a result, it's possible for the previous version of an object to contain reference fields whose Java types no longer exist
039 * in the current Java model. Specifically, this can happen in two ways:
040 * <ul>
041 * <li>A reference field refers to an object type that no longer exists; or</li>
042 * <li>An {@link Enum} field refers to an {@link Enum} type that no longer exists, or whose constants have changed</li>
043 * </ul>
044 *
045 * <p>
046 * In these cases, the old field's value cannot be represented in {@code oldValues} using the original Java types.
047 * Instead, more generic types are used:
048 * <ul>
049 * <li>For a reference field whose type no longer exists, the referenced object will be an {@link org.jsimpledb.UntypedJObject}.
050 *      Note that the fields in the {@link org.jsimpledb.UntypedJObject} may still be accessed by invoking the
051 *      {@link org.jsimpledb.JTransaction} field access methods with {@code upgradeVersion} set to false (otherwise,
052 *      a {@link org.jsimpledb.core.TypeNotInSchemaVersionException} is thrown).
053 * <li>For {@link Enum} fields whose type no longer exists or has modified constants, the old value
054 *      will be represented as an {@link org.jsimpledb.core.EnumValue} object.</li>
055 * </ul>
056 *
057 * <p>
058 * In addition to Java types disappearing, it's also possible that the type of a reference field is narrower in the current
059 * Java code than it was in the previous Java code. If an object held a reference in such a field to another object outside
060 * the new, narrower type, then upgrading the object without change would represent a violation of Java type safety.
061 * Therefore, when any object is upgraded, all references that would otherwise be illegal are cleared (in the manner of
062 * {@link org.jsimpledb.core.DeleteAction#UNREFERENCE}); use {@code oldValues} to access the previous field value(s) if needed.
063 * </p>
064 */
065@Retention(RetentionPolicy.RUNTIME)
066@Target(ElementType.METHOD)
067@Documented
068public @interface OnVersionChange {
069
070    /**
071     * Required old schema version.
072     *
073     * <p>
074     * If this property is set to a positive value, only version changes
075     * for which the previous schema version equals the specified version will result in notification,
076     * and the annotated method must have the corresponding parameter omitted. Otherwise notifications
077     * are delivered for any previous schema version and the {@code oldVersion} method parameter is required.
078     * </p>
079     *
080     * <p>
081     * Negative values are not allowed.
082     * </p>
083     *
084     * @return old schema version, or zero for no restriction
085     */
086    int oldVersion() default 0;
087
088    /**
089     * Required new schema version.
090     *
091     * <p>
092     * If this property is set to a positive value, only version changes
093     * for which the new schema version equals the specified version will result in notification,
094     * and the annotated method must have the corresponding parameter omitted. Otherwise notifications
095     * are delivered for any new schema version and the {@code newVersion} method parameter is required.
096     * </p>
097     *
098     * <p>
099     * Negative values are not allowed.
100     * </p>
101     *
102     * @return new schema version, or zero for no restriction
103     */
104    int newVersion() default 0;
105}
