006package org.jsimpledb.annotation;
007
008import java.lang.annotation.Documented;
009import java.lang.annotation.Retention;
010import java.lang.annotation.RetentionPolicy;
011import java.lang.annotation.Target;
012
013/**
014 * Java annotation defining a composite index on a {@link JSimpleClass}-annotated class.
015 *
016 * <p>
017 * A composite index is an index on two or more fields (to define a single-field index,
018 * just set {@link JField#indexed} to true). All fields indexed in a composite index
019 * must be (a) simple and (b) not a sub-field of a complex field.
020 * </p>
021 *
022 * @see JSimpleClass
023 */
024@Retention(RetentionPolicy.RUNTIME)
025@Target({})
026@Documented
027public @interface JCompositeIndex {
028
029    /**
030     * The name of this composite index. Must be globally unique.
031     *
032     * @return the index name
033     */
034    String name();
035
036    /**
037     * The storage ID for this composite index. Value should be positive; if zero, the configured
038     * {@link org.jsimpledb.StorageIdGenerator} will be consulted to auto-generate a value.
039     *
040     * @return the index storage ID
041     * @see org.jsimpledb.StorageIdGenerator#generateCompositeIndexStorageId StorageIdGenerator.generateCompositeIndexStorageId()
042     */
043    int storageId() default 0;
044
045    /**
046     * The names of the indexed fields, in the desired order. At least two fields must be specified.
047     *
048     * @return the names of the indexed fields
049     */
050    String[] fields();
051}
