006package org.jsimpledb.annotation;
007
008import java.lang.annotation.Documented;
009import java.lang.annotation.ElementType;
010import java.lang.annotation.Retention;
011import java.lang.annotation.RetentionPolicy;
012import java.lang.annotation.Target;
013
014/**
015 * Annotation for methods that are to be invoked whenever a target field in some target object changes during a transaction,
016 * where the target object containing the changed field is found at the end of a path of references
017 * starting from the object to be notified.
018 * See {@link org.jsimpledb.ReferencePath} for more information about reference paths.
019 *
020 * <p><b>Method Parameter Types</b></p>
021 *
022 * <p>
023 * In all cases the annotated method must return void and take a single parameter, which is compatible with one or more
024 * of the {@link org.jsimpledb.change.FieldChange} sub-types appropriate for the field being watched.
025 * The method may have any level of access, including {@code private}.
026 * The method parameter type can be used to restrict which notifications are delivered. For example, an annotated method
027 * taking a {@link org.jsimpledb.change.SetFieldChange} will receive notifications about all changes to a set field,
028 * while a method taking a {@link org.jsimpledb.change.SetFieldAdd} will receive notification only when an element
029 * is added to the set.
030 * </p>
031 *
032 * <p>
033 * Multiple reference paths may be specified; if so, all of the specified paths are monitored together, and they all
034 * must emit {@link org.jsimpledb.change.FieldChange}s compatible with the method's parameter type. Therefore, when
035 * mutiple fields are monitored, the method's parameter type may need to be widened (either in raw type, generic type
036 * parameters, or both).
037 * </p>
038 *
039 * <p>
040 * As a special case, if zero fields are specified, then "wildcard" monitoring of every field in the local object occurs.
041 * However in this case, only fields that emit changes compatible with the method's parameter type will be monitored.
042 * So for example, a method taking a {@link org.jsimpledb.change.SetFieldChange} would receive notifications about
043 * changes to all {@code Set} fields in the class, but not any other fields.
044 * </p>
045 *
046 * <p><b>Instance vs. Static Methods</b></p>
047 *
048 * <p>
049 * If the method is an instance method, then {@link #startType} must be left unset; if the instance is a static
050 * method, then {@link #startType} may be explicitly set, or if left unset it defaults to the class containing
051 * the annotated method.
052 * </p>
053 *
054 * <p>
055 * For an instance method, the method will be invoked on <i>each object</i> for which the changed field is
056 * found at the end of the specified reference path <i>starting from that object</i>.
057 * </p>
058 *
059 * <p>
060 * If the annotated method is a static method, the method is invoked <i>once</i> if any instance exists for which the
061 * changed field is found at the end of the specified reference path, no matter how many such instances there are.
062 * Otherwise the behavior is the same.
063 * </p>
064 *
065 * <p><b>Notification Delivery</b></p>
066 *
067 * <p>
068 * Notifications are delivered synchronously within the thread the made the change, after the change is made and just
069 * prior to returning to the original caller.
070 * Additional changes made within an {@link OnChange &#64;OnChange} handler that themselves result in notifications
071 * are also handled prior to returning to the original caller. Therefore, infinite loops are possible if an
072 * {@link OnChange &#64;OnChange} handler method modifies the field it's monitoring (directly, or indirectly via
073 * other {@link OnChange &#64;OnChange} handler methods).
074 * </p>
075 *
076 * <p>
077 * {@link OnChange &#64;OnChange} functions within a single transaction; it does not notify about changes that
078 * may have occurred in a different transaction.
079 * </p>
080 *
081 * <p><b>Other Notes</b></p>
082 *
083 * <p>
084 * No notifications are delivered for "changes" that do not actually change anything (e.g., setting a simple field to
085 * the value already contained in that field, or adding an element to a set which is already contained in the set).
086 * </p>
087 *
088 * <p>
089 * For any given field change and path, only one notification will be delivered per recipient object, even if the changed field
090 * is seen through the path in multiple ways (e.g., via reference path {@code "mylist.element.myfield"} where the changed object
091 * containing {@code myfield} appears multiple times in {@code mylist}).
092 * </p>
093 *
094 * <p>
095 * See {@link org.jsimpledb.core.Transaction#addSimpleFieldChangeListener Transaction.addSimpleFieldChangeListener()}
096 * for further information on other special corner cases.
097 * </p>
098 *
099 * @see org.jsimpledb.ReferencePath
100 * @see org.jsimpledb.change
101 */
102@Retention(RetentionPolicy.RUNTIME)
103@Target(ElementType.METHOD)
104@Documented
105public @interface OnChange {
106
107    /**
108     * Specifies the path to the target field to watch for changes.
109     * See {@link org.jsimpledb.ReferencePath} for information on the proper syntax.
110     *
111     * <p>
112     * Multiple paths may be specified; if so, each path is handled as a separate independent listener registration,
113     * and the method's parameter type must be compatible with at least one of the {@link org.jsimpledb.change.FieldChange}
114     * sub-types emitted by each field. If zero paths are specified, every field in the class (including superclasses)
115     * that emits {@link org.jsimpledb.change.FieldChange}s compatible with the method parameter will be monitored for changes.
116     * </p>
117     *
118     * @return reference path leading to the changed field
119     * @see org.jsimpledb.ReferencePath
120     */
121    String[] value() default { };
122
123    /**
124     * Specifies the starting type for the {@link org.jsimpledb.ReferencePath} specified by {@link #value}.
125     *
126     * <p>
127     * This property must be left unset for instance methods. For static methods, if this property is left unset,
128     * then then class containing the annotated method is assumed.
129     * </p>
130     *
131     * @return Java type at which the reference path starts
132     * @see org.jsimpledb.ReferencePath
133     */
134    Class<?> startType() default void.class;
135
136    /**
137     * Determines whether this annotation should also be enabled for
138     * {@linkplain org.jsimpledb.SnapshotJTransaction snapshot transaction} objects.
139     * If unset, notifications will only be delivered to non-snapshot (i.e., normal) database instances.
140     *
141     * @return whether enabled for snapshot transactions
142     * @see org.jsimpledb.SnapshotJTransaction
143     */
144    boolean snapshotTransactions() default false;
145}
