006package org.jsimpledb.annotation;
007
008import java.lang.annotation.Documented;
009import java.lang.annotation.ElementType;
010import java.lang.annotation.Retention;
011import java.lang.annotation.RetentionPolicy;
012import java.lang.annotation.Target;
013
014import org.jsimpledb.core.DeleteAction;
015
016/**
017 * Java annotation for creating simple fields, including reference fields that refer to other Java model object types.
018 *
019 * <p>
020 * This annotation is used in two scenarios:
021 * <ul>
022 *  <li>To describe a <b>simple</b> database field by annotating the corresponding abstract Java bean property `getter' method</li>
023 *  <li>To describe the <b>sub-field(s)</b> of a <b>complex</b> database field (i.e., set, list, or map),
024 *      that is, the collection element or map key and value</li>
025 * </ul>
026 *
027 * <p>
028 * Note that this annotation can be applied to superclass and interface methods to have the corresponding
029 * field defined in all sub-types.
030 * </p>
031 *
032 * <p>
033 * This annotation is not required when auto-generation of properties is enabled; see {@link JSimpleClass#autogenFields}.
034 * </p>
035 *
036 * <p><b>Non-Reference Fields</b></p>
037 *
038 * <p>
039 * If the field is not a reference field, the property type is inferred from the type of the annotated method or,
040 * in the case of complex sub-fields, the generic type of the collection class. The name of the property type
041 * must be registered in the {@link org.jsimpledb.core.FieldTypeRegistry} (perhaps via {@link JFieldType &#64;JFieldType}),
042 * and the corresponding {@link org.jsimpledb.core.FieldType} is then used to encode/decode field values.
043 * The type name may also be specified explicitly by {@link #name}.
044 * </p>
045 *
046 * <p>
047 * Simple fields may be {@link #indexed}; see {@link org.jsimpledb.index} for information on querying indexes.
048 * </p>
049 *
050 * <p><b>Reference Fields</b></p>
051 *
052 * <p>
053 * If the type of the field is (assignable to) a {@link JSimpleClass &#64;JsimpleClass}-annotated Java model object type,
054 * then the field is a reference field.
055 * </p>
056 *
057 * <p>
058 * Reference fields have configurable behavior when the containing object or the referred-to object is deleted;
059 * see {@link #onDelete} and {@link #cascadeDelete}.
060 * </p>
061 *
062 * <p>
063 * Reference fields are always indexed; the value of {@link #indexed} is ignored.
064 * </p>
065 *
066 * <p><b>Uniqueness Constraints</b></p>
067 *
068 * <p>
069 * Fields that are not complex sub-fields may be marked as {@link #unique} to impose a uniqueness constraint on the value.
070 * Fields with uniqueness constraints must be indexed. Uniqueness constraints are handled at the JSimpleDB layer and function as
071 * an implicit validation constraint. In other words, the constraint is verified when the validation queue is processed
072 * and is affected by the transaction's configured {@link org.jsimpledb.ValidationMode}.
073 * </p>
074 *
075 * <p>
076 * Optionally, specific field values may be marked as excluded from the uniqueness constraint via {@link #uniqueExclude}.
077 * If so, the specified values may appear in multiple objects without violating the constraint. Because null values
078 * are not allowed in annotations, {@link #uniqueExcludeNull} is provided to exclude the null value.
079 * </p>
080 *
081 * <p>
082 * In {@link org.jsimpledb.ValidationMode#AUTOMATIC}, any upgraded {@link org.jsimpledb.JObject}s are automatically
083 * added to the validation queue, so a uniqueness constraint added in a new schema version will be automatically verified
084 * when any object is upgraded.
085 * However, simpley adding, removing, or changing uniqueness constraints on an existing field doesn't force a schema
086 * version change. Therefore, existing database objects that were previously valid could be in violation of a uniqueness
087 * constraint that is added without one. To avoid this possibililty, change the schema version number as well.
088 * </p>
089 */
090@Retention(RetentionPolicy.RUNTIME)
091@Target(ElementType.METHOD)
092@Documented
093public @interface JField {
094
095    /**
096     * The name of this field.
097     *
098     * <p>
099     * If equal to the empty string (default value), the name is inferred from the name of the annotated Java bean getter method.
100     * </p>
101     *
102     * <p>
103     * For sub-fields of complex fields, this property must be left unset.
104     * </p>
105     *
106     * @return the name of the field
107     */
108    String name() default "";
109
110    /**
111     * Optional override for the type of this field.
112     *
113     * <p>
114     * If set, this must equal the name of a type registered in the {@link org.jsimpledb.core.FieldTypeRegistry}
115     * associated with the {@link org.jsimpledb.core.Database} instance, and the annotated method's return type must match the
116     * {@link org.jsimpledb.core.FieldType}'s {@linkplain org.jsimpledb.core.FieldType#getTypeToken supported Java type}.
117     *
118     * <p>
119     * If equal to the empty string (default value), then the Java type is inferred from the return type of the getter method
120     * and the {@link org.jsimpledb.core.FieldType} is found via
121     * {@link org.jsimpledb.core.FieldTypeRegistry#getFieldType(com.google.common.reflect.TypeToken)
122     * FieldTypeRegistry.getFieldType()}.
123     * </p>
124     *
125     * <p>
126     * For reference fields (i.e., methods with return value equal to a {@link JSimpleClass &#64;JSimpleClass}-annotated class),
127     * this property must be left unset.
128     * </p>
129     *
130     * <p>
131     * For sub-fields of complex fields, this property can be used to force a primitive sub-field type instead of a
132     * primitive wrapper type. In that case, the complex field will disallow null values. For example:
133     * <pre>
134     *  &#64;JSetField(element = &#64;JField(<b>type = "float"</b>)) // nulls will be disallowed
135     *  public abstract List&lt;<b>Float</b>&gt; getScores();
136     * </pre>
137     *
138     * @return the name of the field's type
139     */
140    String type() default "";
141
142    /**
143     * Storage ID for this field. Value should be positive and unique within the contained class.
144     * If zero, the configured {@link org.jsimpledb.StorageIdGenerator} will be consulted to auto-generate a value
145     * unless {@link JSimpleClass#autogenFields} is false (in which case an error occurs).
146     *
147     * @see org.jsimpledb.StorageIdGenerator#generateFieldStorageId StorageIdGenerator.generateFieldStorageId()
148     * @see org.jsimpledb.StorageIdGenerator#generateSetElementStorageId StorageIdGenerator.generateSetElementStorageId()
149     * @see org.jsimpledb.StorageIdGenerator#generateListElementStorageId StorageIdGenerator.generateListElementStorageId()
150     * @see org.jsimpledb.StorageIdGenerator#generateMapKeyStorageId StorageIdGenerator.generateMapKeyStorageId()
151     * @see org.jsimpledb.StorageIdGenerator#generateMapValueStorageId StorageIdGenerator.generateMapValueStorageId()
152     *
153     * @return the field's storage ID
154     */
155    int storageId() default 0;
156
157    /**
158     * Whether this field is indexed or not.
159     *
160     * <p>
161     * Setting this property to true creates a simple index on this field. To have this field participate in
162     * a composite index on multiple fields, use {@link JSimpleClass#compositeIndexes}.
163     * </p>
164     *
165     * <p>
166     * Note: reference fields are always indexed (for reference fields, this property is ignored).
167     * </p>
168     *
169     * @return whether the field is indexed
170     */
171    boolean indexed() default false;
172
173    /**
174     * For reference fields, configure the behavior when the referred-to object is
175     * {@linkplain org.jsimpledb.JObject#delete deleted}.
176     *
177     * <p>
178     * For non-reference fields this property must be equal to its default value.
179     * </p>
180     *
181     * @return desired behavior when a referenced object is deleted
182     * @see #cascadeDelete
183     * @see org.jsimpledb.JObject#delete
184     */
185    DeleteAction onDelete() default DeleteAction.EXCEPTION;
186
187    /**
188     * For reference fields, configure cascading behavior when the containing object is
189     * {@linkplain org.jsimpledb.JObject#delete deleted}. If set to true, the referred-to object
190     * is automatically deleted as well.
191     *
192     * <p>
193     * For non-reference fields this property must be equal to its default value.
194     * </p>
195     *
196     * @return whether deletion should cascade to the referred-to object
197     * @see #onDelete
198     * @see org.jsimpledb.JObject#delete
199     */
200    boolean cascadeDelete() default false;
201
202    /**
203     * Require this field's value to be unique among all database objects.
204     *
205     * <p>
206     * This property must be false for sub-fields of complex fields.
207     * </p>
208     *
209     * @return whether the field's value should be unique
210     * @see #uniqueExclude
211     * @see #uniqueExcludeNull
212     */
213    boolean unique() default false;
214
215    /**
216     * Specify field value(s) which are excluded from the uniqueness constraint.
217     *
218     * <p>
219     * The specified values must be valid {@link String} encodings of the associated field. For example:
220     * <pre>
221     *  &#64;JField(indexed = true, unique = true, uniqueExclude = { "Infinity", "-Infinity" })
222     *  public abstract float getPriority();
223     * </pre>
224     *
225     * <p>
226     * This property must be left empty when {@link #unique} is false.
227     * </p>
228     *
229     * @return values to exclude from the uniqueness constraint
230     * @see #unique
231     * @see #uniqueExcludeNull
232     */
233    String[] uniqueExclude() default {};
234
235    /**
236     * Specify that the null value is excluded from the uniqueness constraint.
237     *
238     * <p>
239     * This property must be left false when {@link #unique} is false or the field has primitive type.
240     * </p>
241     *
242     * @return whether null should be excluded from the uniqueness constraint
243     * @see #unique
244     * @see #uniqueExclude
245     */
246    boolean uniqueExcludeNull() default false;
247}
