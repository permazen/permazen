006package org.jsimpledb.core;
007
008import java.util.Set;
009
010/**
011 * Thrown when attempting to set a reference field to a disallowed object type.
012 *
013 * @see ReferenceField#getObjectTypes
014 */
015@SuppressWarnings("serial")
016public class InvalidReferenceException extends IllegalArgumentException {
017
018    private final ObjId id;
019    private final Set<Integer> objectTypes;
020
021    /**
022     * Constructor.
023     *
024     * @param id offending object's ID
025     * @param objectTypes allowed object types
026     */
027    public InvalidReferenceException(ObjId id, Set<Integer> objectTypes) {
028        super("illegal reference " + id + " with object type #" + id.getStorageId() + "; allowed object types are " + objectTypes);
029        this.id = id;
030        this.objectTypes = objectTypes;
031    }
032
033    /**
034     * Constructor.
035     *
036     * @param field disallowing field
037     * @param id offending object's ID
038     */
039    public InvalidReferenceException(ReferenceField field, ObjId id) {
040        this(id, field.getObjectTypes());
041    }
042
043    /**
044     * Get the {@link ObjId} of the object that could not be assigned to the reference field due to disallowed object type.
045     *
046     * @return the object having disallowed type
047     */
048    public ObjId getObjId() {
049        return this.id;
050    }
051
052    /**
053     * Get the set of object types that are allowed to be referenced by the reference field.
054     *
055     * @return set of allowed object type storage IDs
056     */
057    public Set<Integer> getObjectTypes() {
058        return this.objectTypes;
059    }
060}
