006package org.jsimpledb.core;
007
008import java.util.NavigableSet;
009
010/**
011 * Listener interface for notifications of a change in value of a {@link ListField}.
012 *
013 * @see Transaction#addListFieldChangeListener Transaction.addListFieldChangeListener()
014 */
015public interface ListFieldChangeListener {
016
017    /**
018     * Receive notification of the addition (or insertion) of an element to a {@link ListField}.
019     *
020     * <p>
021     * Notifications are only delivered when {@code referrers} is non-empty.
022     * </p>
023     *
024     * <p>
025     * Notifications are delivered in the same thread that made the change, before the outermost mutation operation returns.
026     * </p>
027     *
028     * @param tx associated transaction
029     * @param id the ID of the affected object (i.e., the object containing the field that changed)
030     * @param field the field that changed
031     * @param path path of reference fields (represented by storage IDs) that lead to {@code id}
032     * @param referrers all objects that (indirectly) refer to the affected object via {@code path}
033     * @param index the position in the list at which {@code value} was added or inserted
034     * @param value the value added to the list
035     * @param <E> Java type for {@code field}'s elements
036     */
037    <E> void onListFieldAdd(Transaction tx, ObjId id, ListField<E> field,
038      int[] path, NavigableSet<ObjId> referrers, int index, E value);
039
040    /**
041     * Receive notification of a removal of an element from a {@link ListField}.
042     *
043     * <p>
044     * Notifications are only delivered when {@code referrers} is non-empty.
045     * </p>
046     *
047     * <p>
048     * Notifications are delivered in the same thread that made the change, before the outermost mutation operation returns.
049     * </p>
050     *
051     * @param tx associated transaction
052     * @param id the ID of the affected object (i.e., the object containing the field that changed)
053     * @param field the field that changed
054     * @param path path of reference fields (represented by storage IDs) that lead to {@code id}
055     * @param referrers all objects that (indirectly) refer to the affected object via {@code path}
056     * @param index the position in the list at which {@code value} was removed
057     * @param value the value removed from the list
058     * @param <E> Java type for {@code field}'s elements
059     */
060    <E> void onListFieldRemove(Transaction tx, ObjId id, ListField<E> field,
061      int[] path, NavigableSet<ObjId> referrers, int index, E value);
062
063    /**
064     * Receive notification of the change in element at a specific position in a {@link ListField}.
065     *
066     * <p>
067     * Notifications are only delivered when {@code referrers} is non-empty.
068     * </p>
069     *
070     * <p>
071     * Notifications are delivered in the same thread that made the change, before the outermost mutation operation returns.
072     * </p>
073     *
074     * @param tx associated transaction
075     * @param id the ID of the affected object (i.e., the object containing the field that changed)
076     * @param field the field that changed
077     * @param path path of reference fields (represented by storage IDs) that lead to {@code id}
078     * @param referrers all objects that (indirectly) refer to the affected object via {@code path}
079     * @param index the position in the list at which the change occurred
080     * @param oldValue the old value at list position {@code index}
081     * @param newValue the new value at list position {@code index}
082     * @param <E> Java type for {@code field}'s elements
083     */
084    <E> void onListFieldReplace(Transaction tx, ObjId id, ListField<E> field,
085      int[] path, NavigableSet<ObjId> referrers, int index, E oldValue, E newValue);
086
087    /**
088     * Receive notification of the clearing of a {@link ListField}.
089     *
090     * <p>
091     * This method is only used when the whole list is cleared; when a sub-range of a list is cleared, individual
092     * {@linkplain #onListFieldRemove onListFieldRemove()} invocations for each removed element are made instead.
093     * </p>
094     *
095     * <p>
096     * Notifications are only delivered when {@code referrers} is non-empty.
097     * </p>
098     *
099     * <p>
100     * Notifications are delivered in the same thread that made the change, before the outermost mutation operation returns.
101     * </p>
102     *
103     * @param tx associated transaction
104     * @param id the ID of the affected object (i.e., the object containing the field that was cleared)
105     * @param field the field that changed
106     * @param path path of reference fields (represented by storage IDs) that lead to {@code id}
107     * @param referrers all objects that (indirectly) refer to the affected object via {@code path}
108     */
109    void onListFieldClear(Transaction tx, ObjId id, ListField<?> field, int[] path, NavigableSet<ObjId> referrers);
110}
