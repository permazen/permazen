006package org.jsimpledb.core;
007
008/**
009 * Thrown by {@link Database#createTransaction} when the given schema cannot be matched with the
010 * schema of the same version already recorded in the database, or cannot be added to the database
011 * because it is incompatible with one or more prior schema versions.
012 */
013@SuppressWarnings("serial")
014public class SchemaMismatchException extends InvalidSchemaException {
015
016    SchemaMismatchException() {
017    }
018
019    public SchemaMismatchException(String message) {
020        super(message);
021    }
022
023    public SchemaMismatchException(Throwable cause) {
024        super(cause);
025    }
026
027    public SchemaMismatchException(String message, Throwable cause) {
028        super(message, cause);
029    }
030}
