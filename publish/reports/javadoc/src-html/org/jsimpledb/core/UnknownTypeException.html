006package org.jsimpledb.core;
007
008/**
009 * Thrown when attempting to access an unknown type.
010 */
011@SuppressWarnings("serial")
012public class UnknownTypeException extends DatabaseException {
013
014    private final int storageId;
015    private final int schemaVersion;
016
017    /**
018     * Constructor.
019     *
020     * @param storageId unknown type storage ID
021     * @param schemaVersion schema version in which type was not found, or zero if not version specific
022     */
023    public UnknownTypeException(int storageId, int schemaVersion) {
024        this(storageId, schemaVersion, "no object type with storage ID " + storageId + " exists"
025          + (schemaVersion != 0 ? " in schema version " + schemaVersion : ""));
026    }
027
028    /**
029     * Constructor.
030     *
031     * @param storageId unknown type storage ID
032     * @param schemaVersion schema version in which type was not found, or zero if not version specific
033     * @param message exception message
034     */
035    public UnknownTypeException(int storageId, int schemaVersion, String message) {
036        super(message);
037        this.storageId = storageId;
038        this.schemaVersion = schemaVersion;
039    }
040
041    /**
042     * Get the storage ID that was not recognized.
043     *
044     * @return unrecognized object type storage ID
045     */
046    public int getStorageId() {
047        return this.storageId;
048    }
049
050    /**
051     * Get the schema version in which the type was not found.
052     * This may return zero if a query was not specific to a single schema version.
053     *
054     * @return schema version not having the object type
055     */
056    public int getSchemaVersion() {
057        return this.schemaVersion;
058    }
059}
