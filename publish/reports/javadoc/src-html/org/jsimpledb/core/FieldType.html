006package org.jsimpledb.core;
007
008import com.google.common.reflect.TypeToken;
009
010import java.util.Comparator;
011
012import org.jsimpledb.parse.ParseContext;
013import org.jsimpledb.parse.ParseUtil;
014import org.jsimpledb.util.ByteReader;
015import org.jsimpledb.util.ByteWriter;
016
017/**
018 * Defines the encoding, ordering, and range of possible values for a {@link SimpleField}.
019 *
020 * <p>
021 * A {@link FieldType} maps between instances of its supported Java type and the self-delimited {@code byte[]} encoding of
022 * those instances used in the database. The {@code byte[]} encoding also implicitly defines database the sort order
023 * (via unsigned lexicographical ordering). This ordering is also reflected via {@link #compare compare()}.
024 * </p>
025 *
026 * <p>
027 * A {@link FieldType} also defines a mapping between Java instances and {@link String} values.
028 * There are two separate {@link String} forms, a regular form and a self-delimiting form.
029 * </p>
030 *
031 * <p>
032 * {@link FieldType}s have these requirements and properties:
033 * <ul>
034 *  <li>They have a unique {@linkplain #getName name}; typically the same as their {@linkplain #getTypeToken supported type}.</li>
035 *  <li>All possible values can be represented in Java as an instance of the associated Java type (possibly including null).</li>
036 *  <li>Instances {@linkplain #compare totally order} their Java values. If the associated Java type implements {@link Comparable},
037 *      then the two orderings do not necessarily have to match, but they should if possible.</li>
038 *  <li>All possible values can be encoded/decoded into a self-delimiting binary string (i.e., {@code byte[]} array)
039 *      without losing information, and these binary strings, when sorted lexicographically using unsigned comparison,
040 *      sort consistently with the total ordering of the corresponding Java values defined by {@link #compare compare()}.</li>
041 *  <li>All possible values can be encoded/decoded to/from {@link String}s without losing information,
042 *      with both a {@linkplain #toString(Object) regular string form} for non-null values and a
043 *      {@linkplain #toParseableString self-delimiting string form} for any value (these may be the same).</li>
044 *  <li>{@code null} may or may not be a supported value; if so, it must be handled by {@link #compare compare()} (typically
045 *      null values sort last) and have binary and string encodings just like any other value.</li>
046 *  <li>There is a {@linkplain #getDefaultValue default value}. For types that support null, the default value must be null.</li>
047 *  <li>An optional {@linkplain #getEncodingSignature encoding signature} protects against incompatible encodings
048 *      when a {@link FieldType}'s binary or string encoding changes without changing the {@linkplain #getName name}.</li>
049 * </ul>
050 *
051 * <p>
052 * Two {@link FieldType} instances should be equal according to {@link #equals equals()} if only if they behave identically
053 * with respect to all of the above.
054 * </p>
055 *
056 * <p>
057 * A {@link FieldTypeRegistry} contains a registry of {@link FieldType}s indexed by name.
058 * </p>
059 *
060 * @param <T> The associated Java type
061 * @see FieldTypeRegistry
062 */
063public abstract class FieldType<T> implements Comparator<T> {
064
065    /**
066     * The regular expression that {@link FieldType} names must match. This pattern is the same as is required
067     * for Java identifiers, except that the following additional characters are allowed after the first character:
068     * dot (`.') and dash (`-'), and lastly up to 255 pairs of square brackets (`[]') to indicate an array type.
069     */
070    public static final String NAME_PATTERN = "\\p{javaJavaIdentifierStart}[-.\\p{javaJavaIdentifierPart}]*(\\[\\]){0,255}";
071
072    /**
073     * Type name for reference types.
074     */
075    public static final String REFERENCE_TYPE_NAME = "reference";
076
077    final String name;
078    final TypeToken<T> typeToken;
079    final long signature;
080
081// Constructors
082
083    /**
084     * Constructor.
085     *
086     * @param name the name of this type
087     * @param typeToken Java type for the field's values
088     * @param signature binary encoding signature
089     * @throws IllegalArgumentException if any parameter is null
090     * @throws IllegalArgumentException if {@code name} is invalid
091     */
092    protected FieldType(String name, TypeToken<T> typeToken, long signature) {
093        if (name == null)
094            throw new IllegalArgumentException("null name");
095        if (!name.matches(FieldType.NAME_PATTERN))
096            throw new IllegalArgumentException("invalid type name `" + name + "'");
097        if (typeToken == null)
098            throw new IllegalArgumentException("null typeToken");
099        this.name = name;
100        this.typeToken = typeToken;
101        this.signature = signature;
102    }
103
104    /**
105     * Constructor taking a {@link Class} object.
106     * The {@linkplain #getName name} of this instance will be the {@linkplain Class#getName name} of the given class.
107     *
108     * @param type Java type for the field's values
109     * @param signature binary encoding signature
110     * @throws NullPointerException if {@code type} is null
111     */
112    protected FieldType(Class<T> type, long signature) {
113        this(type.getName(), TypeToken.of(type), signature);
114    }
115
116// Public methods
117
118    /**
119     * Get the name of this type. {@link FieldType} names must be unique in the registry.
120     *
121     * @return this type's name
122     */
123    public String getName() {
124        return this.name;
125    }
126
127    /**
128     * Get the Java type corresponding to this type's values.
129     *
130     * @return this type's Java type
131     */
132    public TypeToken<T> getTypeToken() {
133        return this.typeToken;
134    }
135
136    /**
137     * Get the binary encoding signature of this type.
138     *
139     * <p>
140     * The binary encoding signature is analogous to the {@code serialVersionUID} used by Java serialization.
141     * It represents a specific binary and/or {@link String} encoding for Java values. In the case that a {@link FieldType}
142     * implementation changes its binary encoding, but not it's name, it <b>must</b> use a new, different binary encoding
143     * signature to eliminate the possibility of mixing incompatible encodings in software vs. persistent storage.
144     * Typically a value of zero is used until if/when such a change occurs.
145     * </p>
146     *
147     * <p>
148     * Note that another option when encodings change is simply to change the {@linkplain #getName name} of the type
149     * (encoding signatures are scoped to a single {@link FieldType} name).
150     * </p>
151     *
152     * @return binary encoding signature
153     */
154    public long getEncodingSignature() {
155        return this.signature;
156    }
157
158    /**
159     * Read a value from the given input.
160     *
161     * @param input byte input
162     * @return field value (possibly null)
163     * @throws IllegalArgumentException if invalid input is encountered
164     * @throws IndexOutOfBoundsException if input is truncated
165     */
166    public abstract T read(ByteReader input);
167
168    /**
169     * Write a value to the given output.
170     *
171     * @param writer byte output
172     * @param value value to write (possibly null)
173     * @throws IllegalArgumentException if {@code value} is null and this type does not support null
174     */
175    public abstract void write(ByteWriter writer, T value);
176
177    /**
178     * Get the default value for this field type encoded as a {@code byte[]} array.
179     *
180     * @return encoded default value
181     */
182    public abstract byte[] getDefaultValue();
183
184    /**
185     * Read and discard a value from the given input.
186     *
187     * @param reader byte input
188     * @throws IllegalArgumentException if invalid input is encountered
189     * @throws IndexOutOfBoundsException if input is truncated
190     */
191    public abstract void skip(ByteReader reader);
192
193    /**
194     * Encode a non-null value as a {@link String} for later decoding by {@link #fromString fromString()}.
195     *
196     * <p>
197     * Each of the characters in the returned {@link String} must be one of the valid XML characters
198     * (tab, newline, carriage return, <code>&#92;u0020 - &#92;ud7ff</code>, and <code>&#92;ue000 - &#92;fffdf</code>).
199     * </p>
200     *
201     * <p>
202     * The implementation in {@link FieldType} checks that {@code value} is not null, then delegates to {@link #toParseableString}.
203     * Subclasses that override this method should also override {@link #fromString fromString()}.
204     * </p>
205     *
206     * @param value actual value, never null
207     * @return string encoding of {@code value}
208     * @throws IllegalArgumentException if {@code value} is null
209     * @see <a href="http://www.w3.org/TR/REC-xml/#charsets">The XML 1.0 Specification</a>
210     */
211    public String toString(T value) {
212        if (value == null)
213            throw new IllegalArgumentException("null value");
214        return this.toParseableString(value);
215    }
216
217    /**
218     * Parse a non-null value previously encoded by {@link #toString(Object) toString(T)}.
219     *
220     * <p>
221     * The implementation in {@link FieldType} creates a new {@link ParseContext} based on {@code string},
222     * delegates to {@link #toParseableString} to parse it, and verifies that all of {@code string} was consumed
223     * during the parse. Subclasses that override this method should also override {@link #toString(Object) toString(T)}.
224     * </p>
225     *
226     * @param string non-null value previously encoded as a {@link String} by {@link #toString(Object) toString(T)}
227     * @return actual value
228     * @throws IllegalArgumentException if the input is invalid
229     */
230    public T fromString(String string) {
231        final ParseContext ctx = new ParseContext(string);
232        final T value = this.fromParseableString(ctx);
233        if (!ctx.isEOF()) {
234            throw new IllegalArgumentException("found trailing garbage starting with `"
235              + ParseUtil.truncate(ctx.getInput(), 20) + "'");
236        }
237        return value;
238    }
239
240    /**
241     * Encode a possibly null value as a {@link String} for later decoding by {@link #fromParseableString fromParseableString()}.
242     * The string value must be <i>self-delimiting</i>, i.e., decodable even when followed by arbitrary additional characters,
243     * and must not start with whitespace or closing square bracket ({@code "]"}).
244     *
245     * <p>
246     * In addition, each of the characters in the returned {@link String} must be one of the valid XML characters
247     * (tab, newline, carriage return, <code>&#92;u0020 - &#92;ud7ff</code>, and <code>&#92;ue000 - &#92;fffdf</code>).
248     * </p>
249     *
250     * @param value actual value (possibly null)
251     * @return string encoding of {@code value}
252     * @throws IllegalArgumentException if {@code value} is null and this type does not support null
253     * @see <a href="http://www.w3.org/TR/REC-xml/#charsets">The XML 1.0 Specification</a>
254     */
255    public abstract String toParseableString(T value);
256
257    /**
258     * Parse a value previously encoded by {@link #toParseableString toParseableString()} as a self-delimited {@link String}
259     * and positioned at the start of the given parsing context.
260     *
261     * @param context parse context starting with a string previously encoded via {@link #toParseableString toParseableString()}
262     * @return actual value (possibly null)
263     * @throws IllegalArgumentException if the input is invalid
264     */
265    public abstract T fromParseableString(ParseContext context);
266
267    /**
268     * Verify the given object is a valid instance of this {@link FieldType}'s Java type and cast it to that type.
269     *
270     * <p>
271     * Note that this method must throw {@link IllegalArgumentException}, not {@link ClassCastException}
272     * or {@link NullPointerException}, if {@code obj} does not have the correct type, or is an illegal null value.
273     * </p>
274     *
275     * <p>
276     * This method is allowed to perform widening conversions of the object that lose no information, e.g.,
277     * from {@link Integer} to {@link Long}.
278     * </p>
279     *
280     * <p>
281     * The implementation in {@link FieldType} simply casts the value using this instance's raw Java type.
282     * Subclasses should override this method to implement any other restrictions, e.g., disallowing null values.
283     * </p>
284     *
285     * @param obj object to validate
286     * @return {@code obj} cast to this field's type
287     * @throws IllegalArgumentException if {@code obj} in not of type T
288     * @throws IllegalArgumentException if {@code obj} is null and this type does not support null values
289     * @throws IllegalArgumentException if {@code obj} is in any other way not supported by this {@link FieldType}
290     */
291    @SuppressWarnings("unchecked")
292    public T validate(Object obj) {
293        try {
294            return (T)this.typeToken.getRawType().cast(obj);
295        } catch (ClassCastException e) {
296            throw new IllegalArgumentException(this + " does not support values of type " + obj.getClass().getName());
297        }
298    }
299
300    /**
301     * Compare two values. This method must provide a total ordering of all supported Java values.
302     * If null is a supported Java value, then this method must accept it without throwing an exception
303     * (note, this is a stronger requirement than {@link Comparator} requires). By convention, null usually sorts last.
304     *
305     * @throws IllegalArgumentException if {@code value1} or {@code value2} is null and this type does not support null
306     */
307    @Override
308    public abstract int compare(T value1, T value2);
309
310    /**
311     * Determine whether any of this field type's encoded values start with a {@code 0x00} byte.
312     * Certain optimizations are possible when this is not the case. It is safe for this method to always return true.
313     *
314     * <p>
315     * Note: changing the return value of this method usually means changing the binary encoding, resulting in
316     * an incompatible type.
317     * </p>
318     *
319     * <p>
320     * The implementation in {@link FieldType} returns {@code true}.
321     * </p>
322     *
323     * @return true if an encoded value starting with {@code 0x00} exists
324     */
325    public boolean hasPrefix0x00() {
326        return true;
327    }
328
329    /**
330     * Determine whether any of this field type's encoded values start with a {@code 0xff} byte.
331     * Certain optimizations are possible when this is not the case. It is safe for this method to always return true.
332     *
333     * <p>
334     * Note: changing the return value of this method usually means changing the binary encoding, resulting in
335     * an incompatible type.
336     * </p>
337     *
338     * <p>
339     * The implementation in {@link FieldType} returns {@code true}.
340     * </p>
341     *
342     * @return true if an encoded value starting with {@code 0xff} exists
343     */
344    public boolean hasPrefix0xff() {
345        return true;
346    }
347
348    /**
349     * Convenience method for generic type binding.
350     *
351     * @param obj object to validate
352     * @throws IllegalArgumentException if {@code obj} in not of type T
353     * @throws IllegalArgumentException if {@code obj} is null and this type does not support null values
354     * @throws IllegalArgumentException if {@code obj} is in any other way not supported by this {@link FieldType}
355     */
356    void validateAndWrite(ByteWriter writer, Object obj) {
357        this.write(writer, this.validate(obj));
358    }
359
360// Object
361
362    @Override
363    public String toString() {
364        return "field type `" + this.name + "'";
365    }
366
367    @Override
368    public int hashCode() {
369        return this.name.hashCode() ^ this.typeToken.hashCode();
370    }
371
372    @Override
373    public boolean equals(Object obj) {
374        if (obj == this)
375            return true;
376        if (obj == null || obj.getClass() != this.getClass())
377            return false;
378        final FieldType<?> that = (FieldType<?>)obj;
379        return this.name.equals(that.name) && this.typeToken.equals(that.typeToken);
380    }
381}
