006package org.jsimpledb.core;
007
008import java.util.NavigableMap;
009import java.util.NavigableSet;
010
011import org.jsimpledb.index.Index;
012import org.jsimpledb.kv.KeyFilter;
013import org.jsimpledb.tuple.Tuple2;
014
015/**
016 * Core API {@link Index} implementation representing a index on a single field.
017 *
018 * <p>
019 * Instances are immutable.
020 * </p>
021 *
022 * @param <V> index value type
023 * @param <T> index target type
024 */
025public class CoreIndex<V, T> extends AbstractCoreIndex implements Index<V, T> {
026
027// Constructors
028
029    CoreIndex(Transaction tx, IndexView<V, T> indexView) {
030        super(tx, 2, indexView);
031    }
032
033// Methods
034
035    @Override
036    public CoreIndex<V, T> filter(int index, KeyFilter filter) {
037        return new CoreIndex<V, T>(this.tx, this.getIndexView().filter(index, filter));
038    }
039
040    @SuppressWarnings("unchecked")
041    IndexView<V, T> getIndexView() {
042        return (IndexView<V, T>)this.indexView;
043    }
044
045// Index
046
047    @Override
048    public NavigableSet<Tuple2<V, T>> asSet() {
049
050        // Get index view
051        final IndexView<V, T> iv = this.getIndexView();
052
053        // Create tuple field type
054        final Tuple2FieldType<V, T> tupleFieldType = new Tuple2FieldType<V, T>(iv.getValueType(), iv.getTargetType());
055
056        // Build set and apply filtering
057        IndexSet<Tuple2<V, T>> indexSet = new IndexSet<Tuple2<V, T>>(this.tx, tupleFieldType, iv.prefixMode, iv.prefix);
058        if (iv.hasFilters())
059            indexSet = indexSet.filterKeys(new IndexKeyFilter(this.tx, iv, 2));
060
061        // Done
062        return indexSet;
063    }
064
065    @Override
066    public NavigableMap<V, NavigableSet<T>> asMap() {
067
068        // Get index view
069        final IndexView<V, T> iv = this.getIndexView();
070
071        // Build map and apply filtering
072        IndexMap<V, NavigableSet<T>> indexMap = new IndexMap.OfValues<V, T>(this.tx, iv);
073        if (this.indexView.hasFilters())
074            indexMap = indexMap.filterKeys(new IndexKeyFilter(this.tx, iv, 1));
075
076        // Done
077        return indexMap;
078    }
079}
