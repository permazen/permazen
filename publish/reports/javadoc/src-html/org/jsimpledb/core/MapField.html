006package org.jsimpledb.core;
007
008import com.google.common.collect.Maps;
009import com.google.common.reflect.TypeParameter;
010import com.google.common.reflect.TypeToken;
011
012import java.util.ArrayList;
013import java.util.Iterator;
014import java.util.List;
015import java.util.Map;
016import java.util.NavigableMap;
017import java.util.SortedSet;
018import java.util.TreeMap;
019
020import org.jsimpledb.util.ByteReader;
021import org.jsimpledb.util.ByteWriter;
022
023/**
024 * Map field.
025 *
026 * @param <K> Map key type
027 * @param <V> Map value type
028 */
029public class MapField<K, V> extends ComplexField<NavigableMap<K, V>> {
030
031    public static final String KEY_FIELD_NAME = "key";
032    public static final String VALUE_FIELD_NAME = "value";
033
034    final SimpleField<K> keyField;
035    final SimpleField<V> valueField;
036
037    /**
038     * Constructor.
039     *
040     * @param objType the object type that contains this field
041     * @param name the name of the field
042     * @param storageId field content storage ID
043     * @param schema schema version
044     * @param keyField this field's key sub-field
045     * @param valueField this field's value sub-field
046     * @throws IllegalArgumentException if any parameter is null
047     * @throws IllegalArgumentException if {@code storageId} is non-positive
048     */
049    @SuppressWarnings("serial")
050    MapField(String name, int storageId, Schema schema, SimpleField<K> keyField, SimpleField<V> valueField) {
051        super(name, storageId, schema, new TypeToken<NavigableMap<K, V>>() { }
052          .where(new TypeParameter<K>() { }, keyField.typeToken.wrap())
053          .where(new TypeParameter<V>() { }, valueField.typeToken.wrap()));
054        this.keyField = keyField;
055        this.valueField = valueField;
056        assert this.keyField.parent == null;
057        assert this.valueField.parent == null;
058        this.keyField.parent = this;
059        this.valueField.parent = this;
060    }
061
062// Public methods
063
064    /**
065     * Get the key field.
066     *
067     * @return map key field
068     */
069    public SimpleField<K> getKeyField() {
070        return this.keyField;
071    }
072
073    /**
074     * Get the value field.
075     *
076     * @return map value field
077     */
078    public SimpleField<V> getValueField() {
079        return this.valueField;
080    }
081
082    @Override
083    public List<SimpleField<?>> getSubFields() {
084        final ArrayList<SimpleField<?>> list = new ArrayList<>(2);
085        list.add(this.keyField);
086        list.add(this.valueField);
087        return list;
088    }
089
090    @Override
091    @SuppressWarnings("unchecked")
092    public NavigableMap<K, V> getValue(Transaction tx, ObjId id) {
093        if (tx == null)
094            throw new IllegalArgumentException("null tx");
095        return (NavigableMap<K, V>)tx.readMapField(id, this.storageId, false);
096    }
097
098    @Override
099    public boolean hasDefaultValue(Transaction tx, ObjId id) {
100        return this.getValue(tx, id).isEmpty();
101    }
102
103    @Override
104    public String toString() {
105        return "map field `" + this.name + "' containing key "
106          + this.keyField.fieldType + " and value " + this.valueField.fieldType;
107    }
108
109    @Override
110    public <R> R visit(FieldSwitch<R> target) {
111        return target.caseMapField(this);
112    }
113
114// Non-public methods
115
116    @Override
117    @SuppressWarnings("unchecked")
118    <F> Iterable<F> iterateSubField(Transaction tx, ObjId id, SimpleField<F> subField) {
119        if (subField == this.keyField)
120            return (Iterable<F>)this.getValue(tx, id).keySet();
121        if (subField == this.valueField)
122            return (Iterable<F>)this.getValue(tx, id).values();
123        throw new IllegalArgumentException("unknown sub-field");
124    }
125
126    @Override
127    NavigableMap<K, V> getValueInternal(Transaction tx, ObjId id) {
128        return new JSMap<K, V>(tx, this, id);
129    }
130
131    @Override
132    NavigableMap<K, V> getValueReadOnlyCopy(Transaction tx, ObjId id) {
133        return Maps.unmodifiableNavigableMap(new TreeMap<K, V>(this.getValueInternal(tx, id)));
134    }
135
136    @Override
137    MapFieldStorageInfo<K, V> toStorageInfo() {
138        return new MapFieldStorageInfo<K, V>(this);
139    }
140
141    @Override
142    void copy(ObjId srcId, ObjId dstId, Transaction srcTx, Transaction dstTx) {
143        final FieldType<K> keyFieldType = this.keyField.fieldType;
144        final NavigableMap<K, V> src = this.getValue(srcTx, srcId);
145        final NavigableMap<K, V> dst = this.getValue(dstTx, dstId);
146        final Iterator<Map.Entry<K, V>> si = src.entrySet().iterator();
147        final Iterator<Map.Entry<K, V>> di = dst.entrySet().iterator();
148        if (!si.hasNext()) {
149            dst.clear();
150            return;
151        }
152        if (!di.hasNext()) {
153            dst.putAll(src);
154            return;
155        }
156        Map.Entry<K, V> s = si.next();
157        Map.Entry<K, V> d = di.next();
158        while (true) {
159            final int diff = keyFieldType.compare(s.getKey(), d.getKey());
160            boolean sadvance = true;
161            boolean dadvance = true;
162            if (diff < 0) {
163                dst.put(s.getKey(), s.getValue());
164                dadvance = false;
165            } else if (diff > 0) {
166                di.remove();
167                sadvance = false;
168            } else
169                d.setValue(s.getValue());
170            if (sadvance) {
171                if (!si.hasNext()) {
172                    dst.tailMap(s.getKey(), false).clear();
173                    return;
174                }
175                s = si.next();
176            }
177            if (dadvance) {
178                if (!di.hasNext()) {
179                    dst.putAll(src.tailMap(s.getKey(), true));
180                    return;
181                }
182                d = di.next();
183            }
184        }
185    }
186
187    @Override
188    void buildIndexEntry(ObjId id, SimpleField<?> subField, ByteReader reader, byte[] value, ByteWriter writer) {
189        if (subField == this.keyField) {
190            writer.write(reader);
191            id.writeTo(writer);
192        } else if (subField == this.valueField) {
193            writer.write(value);
194            id.writeTo(writer);
195            writer.write(reader);
196        } else
197            throw new RuntimeException("internal error");
198    }
199
200    @Override
201    void unreferenceRemovedObjectTypes(Transaction tx, ObjId id, ReferenceField subField, SortedSet<Integer> removedStorageIds) {
202        assert subField == this.keyField || subField == this.valueField;
203        for (Iterator<Map.Entry<K, V>> i = this.getValueInternal(tx, id).entrySet().iterator(); i.hasNext(); ) {
204            final Map.Entry<K, V> entry = i.next();
205            final ObjId ref = subField == this.keyField ? (ObjId)entry.getKey() : (ObjId)entry.getValue();
206            if (ref != null && removedStorageIds.contains(ref.getStorageId()))
207                i.remove();
208        }
209    }
210}
