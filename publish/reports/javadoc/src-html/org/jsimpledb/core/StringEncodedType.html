006package org.jsimpledb.core;
007
008import com.google.common.base.Converter;
009import com.google.common.reflect.TypeToken;
010
011/**
012 * A {@link FieldType} implementation for any Java type that can be encoded uniquely as a {@link String}.
013 * A {@link Converter} is used to convert between native and {@link String} forms.
014 *
015 * <p>
016 * This class provides a convenient way to implement custom {@link FieldType}s.
017 * Null values are supported and null is the default value. This type will sort instances according to
018 * the lexicographical sort order of their {@link String} encodings; null will sort last.
019 * </p>
020 *
021 * @param <T> The associated Java type
022 */
023public class StringEncodedType<T> extends NullSafeType<T> {
024
025    /**
026     * Convenience constructor. Uses the name of the {@code type} as this {@link FieldType}'s type name.
027     *
028     * @param type represented Java type
029     * @param signature binary encoding signature (in this case, {@link String} encoding signature)
030     * @param converter converts between native form and {@link String} form
031     * @throws IllegalArgumentException if {@code converter} does not convert null to null
032     * @throws IllegalArgumentException if any parameter is null
033     */
034    public StringEncodedType(Class<T> type, long signature, Converter<T, String> converter) {
035        this(type.getName(), type, signature, converter);
036    }
037
038    /**
039     * Primary constructor.
040     *
041     * @param name the name for this {@link FieldType}
042     * @param type represented Java type
043     * @param signature binary encoding signature (in this case, {@link String} encoding signature)
044     * @param converter converts between native form and {@link String} form
045     * @throws IllegalArgumentException if {@code converter} does not convert null to null
046     * @throws IllegalArgumentException if any parameter is null
047     */
048    public StringEncodedType(String name, Class<T> type, long signature, Converter<T, String> converter) {
049        super(new StringConvertedType<T>(name, TypeToken.of(type), signature, converter));
050    }
051}
