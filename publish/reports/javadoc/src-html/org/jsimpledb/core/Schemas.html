009
010import java.util.Arrays;
011import java.util.Collections;

019/**
020 * Contains the set of all {@link Schema}s of objects visible in a {@link Transaction}.
021 * All database objects have an implicit schema version number, which corresponds to one of the {@link Schema}s
022 * contained here and describes the object's fields.
023 *
024 * <p>
025 * Associated with a {@link Transaction} is a {@linkplain Transaction#getSchema specific schema version}
026 * that the transaction uses to access objects.
027 * </p>
028 */
029public class Schemas {
030
031    final TreeMap<Integer, Schema> versions = new TreeMap<>();
032    final TreeMap<Integer, StorageInfo> storageInfos = new TreeMap<>();
033    final TreeMap<Integer, TreeSet<Integer>> indexedFieldToContainingTypesMap = new TreeMap<>();
034    final TreeSet<Integer> objTypeStorageIds = new TreeSet<>();
035
036    Schemas(SortedMap<Integer, Schema> versions) {
037        this.initialize(versions);
038    }
039
040    private void initialize(SortedMap<Integer, Schema> versions) {
041
042        // Sanity check
043        if (versions == null)
044            throw new IllegalArgumentException("null versions");
045
046        // Verify Versions have the right version numbers
047        for (Map.Entry<Integer, Schema> entry : versions.entrySet()) {
048            final int versionNumber = entry.getKey();
049            final Schema schema = entry.getValue();
050            if (schema == null)
051                throw new IllegalArgumentException("null schema");
052            if (schema.versionNumber != versionNumber) {
053                throw new InvalidSchemaException("schema version at index "
054                  + versionNumber + " has version " + schema.versionNumber);
055            }
056        }
057
058        // Reset state
059        this.versions.clear();
060        this.storageInfos.clear();
061        this.indexedFieldToContainingTypesMap.clear();
062        this.objTypeStorageIds.clear();
063
064        // Copy versions
065        this.versions.putAll(versions);
066
067        // Verify all schema versions use storage IDs in a compatible way
068        final HashMap<StorageInfo, Integer> versionMap = new HashMap<>();
069        for (Schema version : this.versions.values()) {
070            for (Map.Entry<Integer, StorageInfo> entry : version.storageInfoMap.entrySet()) {
071                final int storageId = entry.getKey();
072                final StorageInfo current = entry.getValue();
073                final StorageInfo previous = this.storageInfos.put(storageId, current);
074                if (previous != null && !previous.equals(current)) {
075                    throw new InvalidSchemaException("incompatible use of storage ID " + storageId + " for both "
076                      + previous + " in schema version " + versionMap.get(previous) + " and "
077                      + current + " in schema version " + version.versionNumber);
078                }
079                versionMap.put(current, version.versionNumber);
080            }
081        }
082
083        // Gather all object type storage IDs
084        for (Schema version : this.versions.values()) {
085            for (ObjType objType : version.objTypeMap.values())
086                objTypeStorageIds.add(objType.storageId);
087        }
088
089        // Calculate, for each simple field, the storage ID's of all types in which, for some schema version,
090        // the field exists and is indexed.
091        for (Schema version : this.versions.values()) {
092            for (ObjType objType : version.objTypeMap.values()) {
093                for (SimpleField<?> field : Iterables.filter(objType.getFieldsAndSubFields(), SimpleField.class)) {
094                    if (!field.indexed)
095                        continue;
096                    TreeSet<Integer> containingTypes = this.indexedFieldToContainingTypesMap.get(field.storageId);
097                    if (containingTypes == null) {
098                        containingTypes = new TreeSet<>();
099                        this.indexedFieldToContainingTypesMap.put(field.storageId, containingTypes);
100                    }
101                    containingTypes.add(objType.storageId);
102                }
103            }
104        }
105    }
106
107    /**
108     * Verify the given storage ID has the specified type of {@link StorageInfo} and find it.
109     *
110     * @param storageId schema object storage ID
111     * @param expectedType expected {@link StorageInfo} type
112     * @return the actual {@link StorageInfo} instance found
113     * @throws UnknownFieldException if type doesn't match and {@code expectedType} is a {@link FieldStorageInfo} sub-type
114     * @throws UnknownIndexException if type doesn't match and {@code expectedType} is {@link CompositeIndexStorageInfo}
115     * @throws UnknownTypeException if type doesn't match and {@code expectedType} is {@link ObjectStorageInfo}
116     */
117    <T extends StorageInfo> T verifyStorageInfo(int storageId, Class<T> expectedType) {
118        final StorageInfo storageInfo = this.storageInfos.get(storageId);
119        if (storageInfo != null && expectedType.isInstance(storageInfo))
120            return expectedType.cast(storageInfo);
121        String message = "no " + this.getDescription(expectedType) + " with storage ID " + storageId + " exists";
122        if (storageInfo != null)
123            message += " (found " + storageInfo + " instead)";
124        if (FieldStorageInfo.class.isAssignableFrom(expectedType))
125            throw new UnknownFieldException(storageId, message);
126        if (ObjTypeStorageInfo.class.isAssignableFrom(expectedType))
127            throw new UnknownTypeException(storageId, 0, message);
128        if (CompositeIndexStorageInfo.class.isAssignableFrom(expectedType))
129            throw new UnknownIndexException(storageId, message);
130        throw new IllegalArgumentException(message);                        // should never get here
131    }
132
133    boolean deleteVersion(int version) {
134        final TreeMap<Integer, Schema> newVersions = new TreeMap<>(this.versions);
135        if (newVersions.remove(version) == null)
136            return false;
137        this.initialize(newVersions);
138        return true;
139    }
140
141    private String getDescription(Class<? extends StorageInfo> type) {
142        if (FieldStorageInfo.class.isAssignableFrom(type))
143            return type.getSimpleName().replaceAll("^(.*)Field.*$", "$1").toLowerCase() + " field";
144        if (ObjTypeStorageInfo.class.isAssignableFrom(type))
145            return "object type";
146        if (CompositeIndexStorageInfo.class.isAssignableFrom(type))
147            return "composite index";
148        return type.getSimpleName();    // ???
149    }
150
151// Accessors
152
153    /**
154     * Get all schema versions.
155     *
156     * @return unmodifiable map of {@link Schema}s indexed by version number
157     */
158    public SortedMap<Integer, Schema> getVersions() {
159        return Collections.unmodifiableSortedMap(this.versions);
160    }
161
162    /**
163     * Get the {@link Version} corresponding to the given version number.
164     *
165     * @param versionNumber schema version number
166     * @return schema with version number {@code versionNumber}
167     * @throws IllegalArgumentException if {@code versionNumber} is not known
168     */
169    public Schema getVersion(int versionNumber) {
170        final Schema schema = this.versions.get(versionNumber);
171        if (schema == null)
172            throw new IllegalArgumentException("unknown version " + versionNumber);
173        return schema;
174    }
175
176    // See if we match encoded schemas read from KV transaction
177    boolean isSameVersions(SortedMap<Integer, byte[]> bytesList) {
178        if (bytesList.size() != this.versions.size())
179            return false;
180        final Iterator<Map.Entry<Integer, byte[]>> i1 = bytesList.entrySet().iterator();
181        final Iterator<Map.Entry<Integer, Schema>> i2 = this.versions.entrySet().iterator();
182        while (i1.hasNext() || i2.hasNext()) {
183            if (!i1.hasNext() || !i2.hasNext())
184                return false;
185            final Map.Entry<Integer, byte[]> entry1 = i1.next();
186            final Map.Entry<Integer, Schema> entry2 = i2.next();
187            if (entry1.getKey() != entry2.getKey() || !Arrays.equals(entry1.getValue(), entry2.getValue().encodedXML))
188                return false;
189        }
190        return true;
191    }
