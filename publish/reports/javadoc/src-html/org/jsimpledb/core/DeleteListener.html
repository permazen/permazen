006package org.jsimpledb.core;
007
008/**
009 * Listener interface for notifications that an object is about to be deleted.
010 *
011 * @see Transaction#addDeleteListener Transaction.addDeleteListener()
012 */
013public interface DeleteListener {
014
015    /**
016     * Receive notification of an object being deleted.
017     *
018     * <p>
019     * Notifications are delivered in the same thread that is deleting object, before the delete actually occurs.
020     * At most one notification will be delivered for any object, even if {@link Transaction#delete Transaction.delete()}
021     * is invoked reentrantly from within this listener.
022     * </p>
023     *
024     * @param tx associated transaction
025     * @param id the ID of the object about to be deleted
026     */
027    void onDelete(Transaction tx, ObjId id);
028}
