006package org.jsimpledb.core;
007
008import com.google.common.collect.Sets;
009import com.google.common.reflect.TypeParameter;
010import com.google.common.reflect.TypeToken;
011
012import java.util.Iterator;
013import java.util.NavigableSet;
014import java.util.TreeSet;
015
016import org.jsimpledb.util.ByteReader;
017import org.jsimpledb.util.ByteWriter;
018
019/**
020 * Set field.
021 *
022 * @param <E> Java type for the set elements
023 */
024public class SetField<E> extends CollectionField<NavigableSet<E>, E> {
025
026    /**
027     * Constructor.
028     *
029     * @param name the name of the field
030     * @param storageId field content storage ID
031     * @param schema schema version
032     * @param elementField this field's element sub-field
033     * @throws IllegalArgumentException if any parameter is null
034     * @throws IllegalArgumentException if {@code storageId} is non-positive
035     */
036    @SuppressWarnings("serial")
037    SetField(String name, int storageId, Schema schema, SimpleField<E> elementField) {
038        super(name, storageId, schema, new TypeToken<NavigableSet<E>>() { }
039          .where(new TypeParameter<E>() { }, elementField.typeToken.wrap()), elementField);
040    }
041
042// Public methods
043
044    @Override
045    @SuppressWarnings("unchecked")
046    public NavigableSet<E> getValue(Transaction tx, ObjId id) {
047        if (tx == null)
048            throw new IllegalArgumentException("null tx");
049        return (NavigableSet<E>)tx.readSetField(id, this.storageId, false);
050    }
051
052    @Override
053    public <R> R visit(FieldSwitch<R> target) {
054        return target.caseSetField(this);
055    }
056
057    @Override
058    public String toString() {
059        return "set field `" + this.name + "' containing " + this.elementField;
060    }
061
062// Non-public methods
063
064    @Override
065    NavigableSet<E> getValueInternal(Transaction tx, ObjId id) {
066        return new JSSet<E>(tx, this, id);
067    }
068
069    @Override
070    NavigableSet<E> getValueReadOnlyCopy(Transaction tx, ObjId id) {
071        return Sets.unmodifiableNavigableSet(new TreeSet<E>(this.getValueInternal(tx, id)));
072    }
073
074    @Override
075    SetFieldStorageInfo<E> toStorageInfo() {
076        return new SetFieldStorageInfo<E>(this);
077    }
078
079    @Override
080    void copy(ObjId srcId, ObjId dstId, Transaction srcTx, Transaction dstTx) {
081        final FieldType<E> fieldType = this.elementField.fieldType;
082        final NavigableSet<E> src = this.getValue(srcTx, srcId);
083        final NavigableSet<E> dst = this.getValue(dstTx, dstId);
084        final Iterator<E> si = src.iterator();
085        final Iterator<E> di = dst.iterator();
086        if (!si.hasNext()) {
087            dst.clear();
088            return;
089        }
090        if (!di.hasNext()) {
091            dst.addAll(src);
092            return;
093        }
094        E s = si.next();
095        E d = di.next();
096        while (true) {
097            final int diff = fieldType.compare(s, d);
098            boolean sadvance = true;
099            boolean dadvance = true;
100            if (diff < 0) {
101                dst.add(s);
102                dadvance = false;
103            } else if (diff > 0) {
104                di.remove();
105                sadvance = false;
106            }
107            if (sadvance) {
108                if (!si.hasNext()) {
109                    dst.tailSet(s, false).clear();
110                    return;
111                }
112                s = si.next();
113            }
114            if (dadvance) {
115                if (!di.hasNext()) {
116                    dst.addAll(src.tailSet(s, true));
117                    return;
118                }
119                d = di.next();
120            }
121        }
122    }
123
124    @Override
125    void buildIndexEntry(ObjId id, SimpleField<?> subField, ByteReader reader, byte[] value, ByteWriter writer) {
126        assert subField == this.elementField;
127        writer.write(reader);
128        id.writeTo(writer);
129    }
130}
