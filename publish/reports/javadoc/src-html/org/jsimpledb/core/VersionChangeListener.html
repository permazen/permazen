006package org.jsimpledb.core;
007
008import java.util.Map;
009
010/**
011 * Listener interface for notifications that an object's schema version has been changed to match the
012 * current transaction.
013 *
014 * @see Transaction#addVersionChangeListener Transaction.addVersionChangeListener()
015 */
016public interface VersionChangeListener {
017
018    /**
019     * Receive notification of an object schema version change.
020     *
021     * <p>
022     * Notifications are delivered in the same thread that first reads the object, before the operation
023     * that triggered the schema version change returns.
024     * </p>
025     *
026     * @param tx associated transaction
027     * @param id the ID of the updated object
028     * @param oldVersion previous object schema version
029     * @param newVersion new object schema version
030     * @param oldFieldValues the values of all fields in the old schema version keyed by storage ID
031     */
032    void onVersionChange(Transaction tx, ObjId id, int oldVersion, int newVersion, Map<Integer, Object> oldFieldValues);
033}
