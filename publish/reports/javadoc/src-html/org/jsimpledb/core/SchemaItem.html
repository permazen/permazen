006package org.jsimpledb.core;
007
008/**
009 * Superclass for the {@link ObjType} and {@link Field} classes which make up a {@link Schema} version.
010 *
011 * <p>
012 * Instances have a {@linkplain #getStorageId storage ID} which must be unique across the {@link Schema} version.
013 * Instances also have a {@linkplain #getName name} which must be a {@linkplain #NAME_PATTERN valid Java identifier}.
014 * Instances are also associated with a {@linkplain #getSchema specific} {@link Schema}.
015 * </p>
016 */
017public abstract class SchemaItem {
018
019    /**
020     * The regular expression that all {@link SchemaItem} names must match. This pattern is the same as is required
021     * for Java identifiers.
022     */
023    public static final String NAME_PATTERN = "\\p{javaJavaIdentifierStart}\\p{javaJavaIdentifierPart}*";
024
025    final String name;
026    final int storageId;
027    final Schema schema;
028
029    SchemaItem(String name, int storageId, Schema schema) {
030        if (name == null)
031            throw new IllegalArgumentException("invalid null name");
032        if (!name.matches(NAME_PATTERN))
033            throw new IllegalArgumentException("invalid name `" + name + "'");
034        if (storageId <= 0)
035            throw new IllegalArgumentException("invalid storageId " + storageId);
036        if (schema == null)
037            throw new IllegalArgumentException("null schema");
038        this.name = name;
039        this.storageId = storageId;
040        this.schema = schema;
041    }
042
043    /**
044     * Get the associated with this instance.
045     *
046     * @return name of this object type or field, never null
047     */
048    public String getName() {
049        return this.name;
050    }
051
052    /**
053     * Get the storage ID associated with this instance.
054     *
055     * @return storage ID, always greater than zero
056     */
057    public int getStorageId() {
058        return this.storageId;
059    }
060
061    /**
062     * Get the schema version with which this instance is associated.
063     *
064     * @return associated schema version
065     */
066    public Schema getSchema() {
067        return this.schema;
068    }
069
070    /**
071     * Create corresponding {@link StorageInfo} object.
072     */
073    abstract StorageInfo toStorageInfo();
074
075    @Override
076    public abstract String toString();
077
078    static Class<? extends FieldStorageInfo> infoTypeFor(Class<? extends Field<?>> type) {
079        if (ReferenceField.class.isAssignableFrom(type))
080            return ReferenceFieldStorageInfo.class;
081        if (SimpleField.class.isAssignableFrom(type))
082            return SimpleFieldStorageInfo.class;
083        if (MapField.class.isAssignableFrom(type))
084            return MapFieldStorageInfo.class;
085        if (ListField.class.isAssignableFrom(type))
086            return ListFieldStorageInfo.class;
087        if (SetField.class.isAssignableFrom(type))
088            return SetFieldStorageInfo.class;
089        throw new IllegalArgumentException("no StorageInfo type known for " + type);
090    }
091}
