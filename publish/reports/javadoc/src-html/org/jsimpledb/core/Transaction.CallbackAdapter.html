035/**
036 * A JSimpleDB {@link Database} transaction.
037 *
038 * <p>
039 * Note: this is the lower level, core API for {@link org.jsimpledb.JSimpleDB}. In most cases this API
040 * will only be used indirectly through the higher level {@link org.jsimpledb.JSimpleDB}, {@link org.jsimpledb.JTransaction},
041 * and {@link org.jsimpledb.JObject} APIs.
042 * </p>
043 *
044 * <p>
045 * Methods in this class can be divided into the following categories:
046 * </p>
047 *
048 * <p>
049 * <b>Transaction Meta-Data</b>
050 * <ul>
051 *  <li>{@link #getDatabase getDatabase()} - Get the associated {@link Database}</li>
052 *  <li>{@link #getSchema getSchema()} - Get the database {@link Schema}, as seen by this transaction</li>
053 *  <li>{@link #getSchemaVersion() getSchemaVersion()} - Get the {@link SchemaVersion} that will be used by this transaction</li>
054 * </ul>
055 * </p>
056 *
057 * <p>
058 * <b>Transaction Lifecycle</b>
059 * <ul>
060 *  <li>{@link #commit commit()} - Commit transaction</li>
061 *  <li>{@link #rollback rollback()} - Roll back transaction</li>
062 *  <li>{@link #isValid isValid()} - Test transaction validity</li>
063 *  <li>{@link #setTimeout setTimeout()} - Set transaction timeout</li>
064 *  <li>{@link #setReadOnly setReadOnly()} - Set transaction to read-only</li>
065 *  <li>{@link #setRollbackOnly setRollbackOnly()} - Set transaction for rollack only</li>
066 *  <li>{@link #addCallback addCallback()} - Register a {@link Callback} on transaction completion</li>
067 *  <li>{@link #createSnapshotTransaction createSnapshotTransaction()} - Create a empty, in-memory copy of this transaction</li>
068 * </ul>
069 * </p>
070 *
071 * <p>
072 * <b>Object Lifecycle</b>
073 * <ul>
074 *  <li>{@link #create(int) create()} - Create a database object</li>
075 *  <li>{@link #delete delete()} - Delete a database object</li>
076 *  <li>{@link #exists exists()} - Test whether a database object exists</li>
077 *  <li>{@link #copyTo copyTo()} - "Snapshot" an object by copying it into a different transaction</li>
078 *  <li>{@link #addCreateListener addCreateListener()} - Register a {@link CreateListener} for notifications about new objects</li>
079 *  <li>{@link #removeCreateListener removeCreateListener()} - Unregister a {@link CreateListener}</li>
080 *  <li>{@link #addDeleteListener addDeleteListener()} - Register a {@link DeleteListener} for notifications
081 *      about object deletions</li>
082 *  <li>{@link #removeDeleteListener removeDeleteListener()} - Unregister a {@link DeleteListener}</li>
083 * </ul>
084 * </p>
085 *
086 * <p>
087 * <b>Object Versioning</b>
088 * <ul>
089 *  <li>{@link #getSchemaVersion(ObjId) getSchemaVersion()} - Inspect an object's schema version</li>
090 *  <li>{@link #updateSchemaVersion updateSchemaVersion()} - Update an object's schema version to match this transaction</li>
091 *  <li>{@link #queryVersion queryVersion()} - Locate objects by schema version</li>
092 *  <li>{@link #addVersionChangeListener addVersionChangeListener()} - Register a {@link VersionChangeListener}
093 *      for notifications about object version changes</li>
094 *  <li>{@link #removeVersionChangeListener removeVersionChangeListener()} - Unregister a {@link VersionChangeListener}</li>
095 * </ul>
096 *
097 * <p>
098 * <b>Object and Field Access</b>
099 * <ul>
100 *  <li>{@link #getAll getAll()} - Get all objects of a specific type</li>
101 *  <li>{@link #readSimpleField readSimpleField()} - Read the value of a {@link SimpleField} in an object</li>
102 *  <li>{@link #writeSimpleField writeSimpleField()} - Write the value of a {@link SimpleField} in an object</li>
103 *  <li>{@link #readCounterField readCounterField()} - Read the value of a {@link CounterField} in an object</li>
104 *  <li>{@link #writeCounterField writeCounterField()} - Write the value of a {@link CounterField} in an object</li>

116 * <b>Field Change Notifications</b>
117 * <ul>
118 *  <li>{@link #addSimpleFieldChangeListener addSimpleFieldChangeListener()} - Register a {@link SimpleFieldChangeListener} for

121 *      notifications of changes in a {@link SetField}, as seen through a path of object references</li>

125 *      notifications of changes in a {@link MapField}, as seen through a path of object references</li>

128 *  <li>{@link #removeSetFieldChangeListener removeSetFieldChangeListener()} - Unregister a previously registered

130 *  <li>{@link #removeListFieldChangeListener removeListFieldChangeListener()} - Unregister a previously registered

132 *  <li>{@link #removeMapFieldChangeListener removeMapFieldChangeListener()} - Unregister a previously registered
133 *      {@link MapFieldChangeListener}</li>
134 * </ul>
135 * </p>
136 *
137 * <p>
138 * <b>Reference Path Queries</b>
139 * <ul>
140 *  <li>{@link #invertReferencePath invertReferencePath()} - Find all objects that refer to any element in a given set
141 *      of objects through a specified reference path</li>
142 *  </li>
143 * </ul>
144 * </p>
145 *
146 * <p>
147 * <b>Index Queries</b>
148 * <ul>
149 *  <li>{@link #querySimpleField querySimpleField()} - Query the index associated with a {@link SimpleField}
150 *      to identify all values and all objects having those values</li>
151 *  <li>{@link #querySetField querySetField()} - Query the index associated with a {@link SetField}
152 *      to identify all set elements and all objects having those elements in the set</li>
153 *  <li>{@link #queryListField queryListField()} - Query the index associated with a {@link ListField}
154 *      to identify all list elements and all objects having those elements in the list</li>
155 *  <li>{@link #queryListFieldEntries queryListFieldEntries()} - Query the index associated with a {@link ListField}
156 *      to identify all list elements and all objects with those elements in the list, as well as the associated list indicies</li>
157 *  <li>{@link #queryMapFieldKey queryMapFieldKey()} - Query the index associated with a {@link MapField}
158 *      to identify all map keys and all objects having those keys in the map</li>
159 *  <li>{@link #queryMapFieldKeyEntries queryMapFieldKeyEntries()} - Query the index associated with a {@link MapField}
160 *      to identify all map keys and all objects having those keys in the map, as well as the associated map values</li>

163 *  <li>{@link #queryMapFieldValueEntries queryMapFieldValueEntries()} - Query the index associated with a {@link MapField}

166 * </p>

168 * <p>
169 * All methods returning a set of values return a {@link NavigableSet}.

256    /**
257     * Commit this transaction.
258     *
259     * @throws StaleTransactionException if this transaction is no longer usable
260     * @throws RetryTransactionException from {@link KVTransaction#commit KVTransaction.commit()}
261     * @throws RollbackOnlyTransactionException if this instance has been {@linkplain #setRollbackOnly marked} rollback only;
262     *  this instance will be automatically rolled back
263     */
264    public synchronized void commit() {
265
266        // Sanity check
267        if (this.stale)
268            throw new StaleTransactionException(this);
269
270        // Rollback only?
271        if (this.rollbackOnly) {
272            this.log.info("commit() invoked on transaction " + this + " marked rollback-only, rolling back");
273            this.rollback();
274            throw new RollbackOnlyTransactionException(this);
275        }
276        this.stale = true;
277
278        // Do before completion callbacks
279        this.log.debug("commit() invoked on" + (this.readOnly ? " read-only" : "") + " transaction " + this);
280        Callback failedCallback = null;
281        try {
282            for (Callback callback : this.callbacks) {
283                failedCallback = callback;
284                callback.beforeCommit(this.readOnly);
285            }
286            failedCallback = null;
287        } finally {
288            if (failedCallback != null) {
289                this.log.warn("error invoking beforeCommit() method on transaction "
290                  + this + " callback " + failedCallback + ", rolling back");
291            }
292            this.triggerBeforeCompletion();
293            if (failedCallback != null) {
294                try {
295                    try {
296                        this.kvt.rollback();
297                    } catch (KVTransactionException e) {
298                        // ignore
299                    }
300                } finally {
301                    this.triggerAfterCompletion(false);
302                }
303            }
304        }
305
306        // Commit KVTransaction and trigger after completion callbacks
307        failedCallback = null;
308        try {
309            this.kvt.commit();
310            for (Callback callback : this.callbacks) {
311                failedCallback = callback;
312                callback.afterCommit();
313            }
314            failedCallback = null;
315        } finally {
316            if (failedCallback != null)
317                this.log.warn("error invoking afterCommit() method on transaction " + this + " callback " + failedCallback);
318            this.triggerAfterCompletion(true);
319        }
320    }
321
322    /**
323     * Roll back this transaction.
324     *
325     * @throws StaleTransactionException if this transaction is no longer usable
326     */
327    public synchronized void rollback() {
328
329        // Sanity check
330        if (this.stale)
331            throw new StaleTransactionException(this);
332        this.stale = true;
333        this.log.debug("rollback() invoked on" + (this.readOnly ? " read-only" : "") + " transaction " + this);
334
335        // Do before completion callbacks
336        this.triggerBeforeCompletion();
337
338        // Roll back KVTransaction and trigger after completion callbacks
339        try {
340            this.kvt.rollback();
341        } finally {
342            this.triggerAfterCompletion(false);
343        }
344    }
345
346    private /*synchronized*/ void triggerBeforeCompletion() {
347        for (Callback callback : this.callbacks) {
348            try {
349                callback.beforeCompletion();
350            } catch (Throwable t) {
351                this.log.error("error invoking beforeCompletion() method on transaction callback " + callback, t);
352            }
353        }
354    }
355
356    private /*synchronized*/ void triggerAfterCompletion(boolean committed) {
357        for (Callback callback : this.callbacks) {
358            try {
359                callback.afterCompletion(committed);
360            } catch (Throwable t) {
361                this.log.error("error invoking afterCompletion() method on transaction callback " + callback, t);
362            }
363        }
364    }

503    public synchronized boolean create(ObjId id, int versionNumber) {
504
505        // Sanity check
506        if (id == null)
507            throw new IllegalArgumentException("null id");
508        if (this.stale)
509            throw new StaleTransactionException(this);
510
511        // Does object already exist?
512        if (this.exists(id))
513            return false;
514
515        // Initialize object
516        this.initialize(id, this.schema.getVersion(versionNumber).getSchemaItem(id.getStorageId(), ObjType.class));
517
518        // Done
519        return true;
520    }

594    private /*synchronized*/ ObjId generateIdValidated(int storageId) {
595
596        // Create a new, unique key
597        final ByteWriter keyWriter = new ByteWriter();
598        for (int attempts = 0; attempts < MAX_GENERATED_KEY_ATTEMPTS; attempts++) {
599            final ObjId id = new ObjId(storageId);
600            id.writeTo(keyWriter);
601            if (this.kvt.get(keyWriter.getBytes()) == null)
602                return id;
603            keyWriter.reset(0);
604        }
605
606        // Give up
607        throw new DatabaseException("could not find a new, unused object ID after "
608          + MAX_GENERATED_KEY_ATTEMPTS + " attempts; is our source of randomness truly random?");
609    }

611    private synchronized void initialize(ObjId id, ObjType objType) {
612
613        // Sanity check
614        if (this.stale)
615            throw new StaleTransactionException(this);
616        if (this.readOnly)
617            throw new ReadOnlyTransactionException(this);
618
619        // Write object meta-data
620        ObjInfo.write(this, id, objType.version.versionNumber, false);
621
622        // Write object version index entry
623        this.kvt.put(Database.buildVersionIndexKey(id, objType.version.versionNumber), ByteUtil.EMPTY);
624
625        // Initialize counters to zero
626        if (!objType.counterFields.isEmpty()) {
627            final CountingKVStore ckv = this.getCountingKVTransaction();
628            for (CounterField field : objType.counterFields.values())
629                ckv.put(field.buildKey(id), ckv.encodeCounter(0));
630        }
631
632        // Write simple field index entries
633        for (SimpleField<?> field : objType.simpleFields.values()) {
634            if (field.indexed)
635                this.kvt.put(field.buildIndexKey(id, null), ByteUtil.EMPTY);
636        }
637
638        // Notify listeners
639        for (CreateListener listener : this.createListeners.toArray(new CreateListener[this.createListeners.size()]))
640            listener.onCreate(this, id);
641    }

643    /**
644     * Delete an object. Does nothing if object does not exist (e.g., has already been deleted).
645     *
646     * <p>
647     * This method does <i>not</i> change the object's schema version if it has a different version that this transaction.
648     * </p>
649     *
650     * @param id object ID of the object to delete
651     * @return true if object was found and deleted, false if object was not found
652     * @throws ReferencedObjectException if the object is referenced by some other object
653     *  through a reference field set to {@link DeleteAction#EXCEPTION}
654     * @throws IllegalArgumentException if {@code id} is null
655     * @throws ReadOnlyTransactionException if this transaction has been {@linkplain #setReadOnly set read-only}
656     * @throws StaleTransactionException if this transaction is no longer usable
657     */
658    public synchronized boolean delete(ObjId id) {
659
660        // Sanity check
661        if (id == null)
662            throw new IllegalArgumentException("null id");
663        if (this.stale)
664            throw new StaleTransactionException(this);
665        if (this.readOnly)
666            throw new ReadOnlyTransactionException(this);
667
668        // Handle recurive DeleteAction.DELETE without hogging Java stack
669        final ArrayList<ObjId> deletables = new ArrayList<>(1);
670        deletables.add(id);
671        boolean found = false;
672        while (true) {
673            final int size = deletables.size();
674            if (size == 0)
675                break;
676            id = deletables.remove(size - 1);
677            found |= this.doDelete(id, deletables);
678        }
679        return found;
680    }
681
682    private synchronized boolean doDelete(final ObjId id, List<ObjId> deletables) {
683
684        // Loop here to handle any mutations within delete notification listener callbacks
685        ObjInfo info;
686        while (true) {
687
688            // Get object info
689            try {
690                info = this.getObjectInfo(id, false);
691            } catch (DeletedObjectException e) {                    // possibly due to a cycle of DeleteAction.DELETE references
692                return false;
693            }
694
695            // Determine if any EXCEPTION reference fields refer to the object (from some other object); if so, throw exception
696            for (Map.Entry<ReferenceFieldStorageInfo, DeleteAction> entry : this.schema.referenceFieldOnDeletes.entrySet()) {
697                final ReferenceFieldStorageInfo fieldInfo = entry.getKey();
698                final DeleteAction onDelete = entry.getValue();
699                if (onDelete != null) {                             // field's onDelete is consistent across schema versions
700                    if (onDelete == DeleteAction.EXCEPTION)
701                        this.scanDeleteException(fieldInfo, id, this.findReferrers(fieldInfo, id));
702                } else {                                            // field's onDelete differs across schema versions
703                    for (SchemaVersion schemaVersion : this.schema.versions.values()) {
704                        final ReferenceField field = schemaVersion.referenceFields.get(fieldInfo.storageId);
705                        if (field != null && field.onDelete == DeleteAction.EXCEPTION)
706                            this.scanDeleteException(fieldInfo, id, this.findReferrers(fieldInfo, id, schemaVersion.versionNumber));
707                    }
708                }
709            }
710
711            // Need to issue delete notifications?
712            if (info.isDeleteNotified() || this.deleteListeners.isEmpty())
713                break;
714
715            // Issue delete notifications and retry
716            ObjInfo.write(this, id, info.getVersionNumber(), true);
717            for (DeleteListener listener : this.deleteListeners.toArray(new DeleteListener[this.deleteListeners.size()]))
718                listener.onDelete(this, id);
719        }
720
721        // Actually delete the object
722        this.deleteObject(info);
723
724        // Find all UNREFERENCE references and unreference them
725        for (Map.Entry<ReferenceFieldStorageInfo, DeleteAction> entry : this.schema.referenceFieldOnDeletes.entrySet()) {
726            final ReferenceFieldStorageInfo fieldInfo = entry.getKey();
727            final DeleteAction onDelete = entry.getValue();
728            if (onDelete != null) {                             // field's onDelete is consistent across schema versions
729                if (onDelete == DeleteAction.UNREFERENCE)
730                    this.scanDeleteUnreference(fieldInfo, id, this.findReferrers(fieldInfo, id));
731            } else {                                            // field's onDelete differs across schema versions
732                for (SchemaVersion schemaVersion : this.schema.versions.values()) {
733                    final ReferenceField field = schemaVersion.referenceFields.get(fieldInfo.storageId);
734                    if (field != null && field.onDelete == DeleteAction.UNREFERENCE)
735                        this.scanDeleteUnreference(fieldInfo, id, this.findReferrers(fieldInfo, id, schemaVersion.versionNumber));
736                }
737            }
738        }
739
740        // Find all DELETE references and mark the containing object for deletion (caller will call us back to actually delete)
741        for (Map.Entry<ReferenceFieldStorageInfo, DeleteAction> entry : this.schema.referenceFieldOnDeletes.entrySet()) {
742            final ReferenceFieldStorageInfo fieldInfo = entry.getKey();
743            final DeleteAction onDelete = entry.getValue();
744            if (onDelete != null) {                             // field's onDelete is consistent across schema versions
745                if (onDelete == DeleteAction.DELETE)
746                    this.scanDeleteDelete(fieldInfo, this.findReferrers(fieldInfo, id), deletables);
747            } else {                                            // field's onDelete differs across schema versions
748                for (SchemaVersion schemaVersion : this.schema.versions.values()) {
749                    final ReferenceField field = schemaVersion.referenceFields.get(fieldInfo.storageId);
750                    if (field != null && field.onDelete == DeleteAction.DELETE) {
751                        this.scanDeleteDelete(fieldInfo,
752                          this.findReferrers(fieldInfo, id, schemaVersion.versionNumber), deletables);
753                    }
754                }
755            }
756        }
757
758        // Done
759        return true;
760    }

763    private void scanDeleteException(ReferenceFieldStorageInfo fieldInfo, ObjId id, NavigableSet<ObjId> referrers) {
764        for (ObjId referrer : referrers) {

770    // Scan referrers for DeleteAction.UNREFERENCE
771    private void scanDeleteUnreference(ReferenceFieldStorageInfo fieldInfo, ObjId id, NavigableSet<ObjId> referrers) {
772        if (fieldInfo.isSubField()) {
773            final ComplexFieldStorageInfo superFieldInfo = this.schema.verifyStorageInfo(
774              fieldInfo.superFieldStorageId, ComplexFieldStorageInfo.class);
775            superFieldInfo.unreferenceAll(this, fieldInfo.storageId, id, referrers);
776        } else {
777            final int storageId = fieldInfo.storageId;
778            for (ObjId referrer : referrers)
779                this.writeSimpleField(referrer, storageId, null, false);
780        }
781    }

785        for (ObjId referrer : referrers)        // avoid list.addAll() to avoid invoking size() on 'referrers'
786            deletables.add(referrer);

789    // Delete all of an object's data
790    private void deleteObject(ObjInfo info) {
791
792        // Delete object's simple field index entries
793        final ObjId id = info.getId();
794        final ObjType type = info.getObjType();
795        for (SimpleField<?> field : type.simpleFields.values()) {
796            if (field.indexed)
797                this.kvt.remove(field.buildIndexKey(id, this.kvt.get(field.buildKey(id))));
798        }
799
800        // Delete object's complex field index entries
801        for (ComplexField<?> field : type.complexFields.values()) {
802            for (SimpleField<?> subField : field.getSubFields()) {
803                if (subField.indexed)
804                    field.removeIndexEntries(this, id, subField);
805            }
806        }
807
808        // Delete object meta-data and all field content
809        final byte[] minKey = info.getId().getBytes();
810        final byte[] maxKey = ByteUtil.getKeyAfterPrefix(minKey);
811        this.kvt.removeRange(minKey, maxKey);
812
813        // Delete object schema version entry
814        this.kvt.remove(Database.buildVersionIndexKey(id, info.getVersionNumber()));
815    }

838    /**
839     * "Snapshot" the specified object by copying it into a different transaction. This method copies the object
840     * and all of its fields; any other objects it references are not copied. If the object already exists in {@code dest},
841     * the old copy will be overwritten. The copy is done by copying key/value pairs directly for efficiency.
842     *
843     * <p>
844     * Note: if two threads attempt to copy objects between the same two transactions in opposite directions,
845     * deadlock may result.
846     * </p>
847     *
848     * <p>
849     * If {@code dest} is this instance, no changes are made and false is returned.
850     * </p>
851     *
852     * @param id object ID of the object to copy
853     * @param dest destination for the copy
854     * @return false if object existed in {@code dest} and was overwritten, true if object did not exist in {@code dest}
855     * @throws DeletedObjectException if no object with ID equal to {@code id} is found in this transaction
856     * @throws IllegalArgumentException if {@code id} is null
857     * @throws SchemaMismatchException if the schema version corresponding to the object's version is not identical
858     *  in both transactions
859     */
860    public synchronized boolean copyTo(ObjId id, Transaction dest) {
861
862        // Sanity check
863        if (this == dest)
864            return false;
865
866        // Do the copy while both transactions are locked
867        synchronized (dest) {
868
869            // Get object info
870            final ObjInfo info = this.getObjectInfo(id, false);
871            final int objectVersion = info.getVersionNumber();
872
873            // Verify object's schema version in dest exists and is identical
874            final SchemaVersion destSchema;
875            try {
876                destSchema = dest.schema.getVersion(objectVersion);
877            } catch (IllegalArgumentException e) {
878                throw new SchemaMismatchException("destination transaction has no schema version " + objectVersion);
879            }
880            if (!Arrays.equals(this.version.encodedXML, dest.version.encodedXML))
881                throw new SchemaMismatchException("destination transaction schema version " + objectVersion + " does not match");
882
883            // Does object already exists in dest? If so delete it first
884            final byte[] minKey = id.getBytes();
885            final boolean existed = dest.kvt.get(minKey) != null;
886            if (existed)
887                dest.deleteObject(info);
888
889            // Copy object meta-data and all simple field and counter content
890            final byte[] maxKey = ByteUtil.getKeyAfterPrefix(minKey);
891            for (Iterator<KVPair> i = this.kvt.getRange(minKey, maxKey, false); i.hasNext(); ) {
892                final KVPair kv = i.next();
893                dest.kvt.put(kv.getKey(), kv.getValue());
894            }
895
896            // Add object schema version index entry
897            dest.kvt.put(Database.buildVersionIndexKey(id, objectVersion), ByteUtil.EMPTY);
898
899            // Copy object's simple field index entries
900            final ObjType type = info.getObjType();
901            for (SimpleField<?> field : type.simpleFields.values()) {
902                if (field.indexed) {
903                    final byte[] fieldValue = dest.kvt.get(field.buildKey(id));    // might be null (if field has default value)
904                    final byte[] indexKey = field.buildIndexKey(id, fieldValue);
905                    dest.kvt.put(indexKey, ByteUtil.EMPTY);
906                }
907            }
908
909            // Copy object's complex field index entries
910            for (ComplexField<?> field : type.complexFields.values()) {
911                for (SimpleField<?> subField : field.getSubFields()) {
912                    if (subField.indexed)
913                        field.addIndexEntries(dest, id, subField);
914                }
915            }
916
917            // Done
918            return !existed;
919        }
920    }

1042    /**
1043     * Update object to the current schema version.
1044     *
1045     * @param id object id
1046     * @param info original object info
1047     * @throws ReadOnlyTransactionException if this transaction has been {@linkplain #setReadOnly set read-only}
1048     * @throws InvalidObjectVersionException if the schema version is invalid
1049     * @throws IllegalArgumentException if {@code id} is null
1050     */
1051    private synchronized void updateVersion(ObjId id, ObjInfo info) {
1052
1053        // Get version numbers
1054        final int oldVersion = info.getVersionNumber();
1055        final int newVersion = this.version.versionNumber;
1056
1057        // Sanity check
1058        if (newVersion == oldVersion)
1059            throw new IllegalArgumentException("object already at version " + newVersion);
1060        if (this.readOnly)
1061            throw new ReadOnlyTransactionException(this);
1062
1063        // Get old and new types
1064        final ObjType oldType = info.getObjType();
1065        final ObjType newType;
1066        try {
1067            newType = this.version.getSchemaItem(id.getStorageId(), ObjType.class);
1068        } catch (IllegalArgumentException e) {
1069            throw new InconsistentDatabaseException("object " + id + " has an unrecognized storage ID", e);
1070        }
1071
1072        // Gather removed fields' values here for user migration
1073        final TreeMap<Integer, Object> oldFieldValues = new TreeMap<>();
1074
1075    //////// Update counter fields
1076
1077        // Get counter field storage IDs (old or new)
1078        final TreeSet<Integer> counterFieldStorageIds = new TreeSet<>();
1079        counterFieldStorageIds.addAll(oldType.counterFields.keySet());
1080        counterFieldStorageIds.addAll(newType.counterFields.keySet());
1081
1082        // Update counter fields
1083        for (int storageId : counterFieldStorageIds) {
1084
1085            // Get old and new counter fields having this storage ID
1086            final CounterField oldField = oldType.counterFields.get(storageId);
1087            final CounterField newField = newType.counterFields.get(storageId);
1088
1089            // Save old field values for version change notification
1090            final byte[] key = Field.buildKey(id, storageId);
1091            final byte[] oldValue = this.kvt.get(key);
1092            if (oldField != null)
1093                oldFieldValues.put(storageId, oldValue != null ? this.getCountingKVTransaction().decodeCounter(oldValue) : 0);
1094
1095            // Remove old value if field has disappeared, otherwise leave alone
1096            if (oldValue != null && oldField != null && newField == null)
1097                this.kvt.remove(key);
1098        }
1099
1100    //////// Update simple fields and corresponding index entries
1101
1102        // Get simple field storage IDs (old or new)
1103        final TreeSet<Integer> simpleFieldStorageIds = new TreeSet<>();
1104        simpleFieldStorageIds.addAll(oldType.simpleFields.keySet());
1105        simpleFieldStorageIds.addAll(newType.simpleFields.keySet());
1106
1107        // Update simple fields
1108        for (int storageId : simpleFieldStorageIds) {
1109
1110            // Get old and new simple fields having this storage ID
1111            final SimpleField<?> oldField = oldType.simpleFields.get(storageId);
1112            final SimpleField<?> newField = newType.simpleFields.get(storageId);
1113
1114            // Save old field values for version change notification
1115            final byte[] key = Field.buildKey(id, storageId);
1116            final byte[] oldValue = this.kvt.get(key);
1117            if (oldField != null) {
1118                oldFieldValues.put(storageId,
1119                  oldField.fieldType.read(new ByteReader(oldValue != null ? oldValue : oldField.fieldType.getDefaultValue())));
1120            }
1121
1122            // Preserve the value if we can, otherwise, discard the old value (if any) leaving new value as new field default
1123            boolean skipIndexUpdate = false;
1124            if (oldField != null && newField != null
1125              && newField.isSchemaChangeCompatible(oldField))                               // value is compatible so just leave it
1126                skipIndexUpdate = newField.indexed == oldField.indexed;                     // index need not change either
1127            else if (oldValue != null)
1128                this.kvt.remove(key);                                                       // discard old value
1129
1130            // Remove old index entry
1131            if (oldField != null && oldField.indexed && !skipIndexUpdate)
1132                this.kvt.remove(oldField.buildIndexKey(id, oldValue));
1133
1134            // Add new index entry
1135            if (newField != null && newField.indexed && !skipIndexUpdate)
1136                this.kvt.put(newField.buildIndexKey(id, null), ByteUtil.EMPTY);
1137        }
1138
1139    //////// Update complex fields and corresponding index entries
1140
1141        // Keep track of what to delete when we've done the migration notification
1142        final ArrayList<ComplexField<?>> cleanupList = new ArrayList<>();
1143
1144        // Get complex field storage IDs (old or new)
1145        final TreeSet<Integer> complexFieldStorageIds = new TreeSet<>();
1146        complexFieldStorageIds.addAll(oldType.complexFields.keySet());
1147        complexFieldStorageIds.addAll(newType.complexFields.keySet());
1148
1149        // Notes:
1150        //
1151        // - The only "migration" we support is adding/removing sub-field indexes; all other changes are equivalent to delete+add
1152        // - New complex fields do not need to be explicitly initialized because their initial state is to have zero KV pairs
1153        //
1154        for (int storageId : complexFieldStorageIds) {
1155
1156            // Get old and new complex fields having this storage ID
1157            final ComplexField<?> oldField = oldType.complexFields.get(storageId);
1158            final ComplexField<?> newField = newType.complexFields.get(storageId);
1159
1160            // If there is no old field, new field is new and so it is already initialized (empty)
1161            if (oldField == null)
1162                continue;
1163
1164            // Save old field's value
1165            oldFieldValues.put(storageId, oldField.getValue(this, id));
1166
1167            // Determine if the fields are compatible and field content can be preserved
1168            final boolean compatible = newField != null && newField.isSchemaChangeCompatible(oldField);
1169
1170            // If fields are not compatible, delete old field when done, otherwise check if index(s) should be added/removed
1171            if (!compatible)
1172                cleanupList.add(oldField);
1173            else
1174                newField.updateSubFieldIndexes(this, oldField, id);
1175        }
1176
1177        // Update object version
1178        ObjInfo.write(this, id, newVersion, info.isDeleteNotified());
1179
1180        // Update object version index entry
1181        this.kvt.remove(Database.buildVersionIndexKey(id, oldVersion));
1182        this.kvt.put(Database.buildVersionIndexKey(id, newVersion), ByteUtil.EMPTY);
1183
1184        // Notify about version update
1185        try {
1186            for (VersionChangeListener listener : this.versionChangeListeners)
1187                listener.onVersionChange(this, id, oldVersion, newVersion, oldFieldValues);
1188        } finally {
1189            for (ComplexField<?> oldField : cleanupList)
1190                oldField.deleteContent(this, id);
1191        }
1192    }

1338    private synchronized void doWriteSimpleField(ObjId id, int storageId, final Object newObj, boolean updateVersion) {
1339
1340        // Get object info
1341        final ObjInfo info = this.getObjectInfo(id, updateVersion);
1342
1343        // Find field
1344        final SimpleField<?> field = info.getObjType().simpleFields.get(storageId);
1345        if (field == null)
1346            throw new UnknownFieldException(info.getObjType(), storageId, "simple field");
1347
1348        // Get new value
1349        final byte[] key = field.buildKey(id);
1350        final byte[] newValue = field.encode(newObj);
1351
1352        // Before setting the new value, read the old value if one of the following is true:
1353        //  - The field is being monitored -> we need to detect "changes" that don't actuallly change anything
1354        //  - The field is indexed -> we need the old value so we can remove the old index entry
1355        // If neither of the above is true, then there's no need to read the old value.
1356        byte[] oldValue = null;
1357        if (field.indexed || this.hasFieldMonitor(field)) {
1358
1359            // Get old value
1360            oldValue = this.kvt.get(key);
1361
1362            // Compare new to old value
1363            if (oldValue != null ? newValue != null && Arrays.equals(oldValue, newValue) : newValue == null)
1364                return;
1365        }
1366
1367        // Update value
1368        if (newValue != null)
1369            this.kvt.put(key, newValue);
1370        else
1371            this.kvt.remove(key);
1372
1373        // Update index
1374        if (field.indexed) {
1375            this.kvt.remove(field.buildIndexKey(id, oldValue));
1376            this.kvt.put(field.buildIndexKey(id, newValue), ByteUtil.EMPTY);
1377        }
1378
1379        // Notify monitors
1380        final Object oldObj = field.fieldType.read(new ByteReader(oldValue != null ? oldValue : field.fieldType.getDefaultValue()));
1381        this.addFieldChangeNotification(new SimpleFieldChangeNotifier(field, id) {
1382            @Override
1383            public void notify(Transaction tx, SimpleFieldChangeListener listener, int[] path, NavigableSet<ObjId> referrers) {
1384                listener.onSimpleFieldChange(tx, this.id, this.storageId, path, referrers, oldObj, newObj);
1385            }
1386        });
1387    }

1690     *
1691     * <p>
1692     * When the specified field is changed, a listener notification will be delivered for each referring object that
1693     * refers to the object containing the changed field through the specified path of references. Notifications are
1694     * delivered at the end of the mutation operation just prior to returning to the caller. If the listener method
1695     * performs additional mutation(s) which are themselves being listened for, those notifications will also be delivered
1696     * prior to the returning to the original caller. Therefore, care must be taken to avoid change notification dependency
1697     * loops when listeners can themselves mutate fields, to avoid infinite loops.
1698     * </p>
1699     *
1700     * <p>
1701     * A referring object may refer to the changed object through more than one actual path of references matching {@code path};
1702     * if so, it will still appear only once in the {@link NavigableSet} provided to the listener (this is of course required
1703     * by set semantics).
1704     * </p>
1705     *
1706     * <p>
1707     * Although the reference back-tracking algorithm does consolidate multiple paths between the same two objects,
1708     * be careful to avoid an explosion of notifications when objects in the {@code path} have a high degree of fan-in.
1709     * </p>
1710     *
1711     * <p>
1712     * When a {@link ReferenceField} in {@code path} also happens to be the field being changed, then there is ambiguity
1713     * about how the set of referring objects is calculated: does it use the field's value before or after the change?
1714     * This API guarantees that the answer is "after the change"; however, if another listener on the same field is
1715     * invoked before {@code listener} and mutates any reference field(s) in {@code path}, then whether that additional
1716     * change is also be included in the calculation is undefined.
1717     * </p>
1718     *
1719     * <p>
1720     * Therefore, for consistency, avoid changing any {@link ReferenceField} from within a listener callback when that
1721     * field is also in some other listener's reference path, and both listeners are watching the same field.
1722     * </p>
1723     *

1725     * @param path path of reference fields (represented by storage IDs) through which to monitor field

1727     * @throws UnknownFieldException if no {@link SimpleField} corresponding to {@code storageId} exists

1935    /**
1936     * Perform some action and, when entirely done (including re-entrant invocation), issue pending notifications to monitors.
1937     *
1938     * @param id object containing the mutated field; will be validated
1939     * @param mutation change to apply
1940     * @throws ReadOnlyTransactionException if this transaction has been {@linkplain #setReadOnly set read-only}
1941     * @throws StaleTransactionException if this transaction is no longer usable
1942     * @throws IllegalArgumentException if {@code id} is null
1943     */
1944    synchronized <V> V mutateAndNotify(ObjId id, Mutation<V> mutation) {
1945
1946        // Check that this transaction is still valid and the given object still exists.
1947        if (id == null)
1948            throw new IllegalArgumentException("null id");
1949        if (this.stale)
1950            throw new StaleTransactionException(this);
1951        if (this.readOnly)
1952            throw new ReadOnlyTransactionException(this);
1953        if (this.kvt.get(id.getBytes()) == null)
1954            throw new DeletedObjectException(id);
1955
1956        // If re-entrant invocation, we're already set up
1957        if (this.pendingNotifications.get() != null)
1958            return mutation.mutate();
1959
1960        // Set up pending report list, perform mutation, and then issue reports
1961        this.pendingNotifications.set(new TreeMap<Integer, ArrayList<FieldChangeNotifier>>());
1962        try {
1963            return mutation.mutate();
1964        } finally {
1965            try {
1966                final TreeMap<Integer, ArrayList<FieldChangeNotifier>> pendingNotificationMap = this.pendingNotifications.get();
1967                while (!pendingNotificationMap.isEmpty()) {
1968
1969                    // Get the next field with pending notifications
1970                    final Map.Entry<Integer, ArrayList<FieldChangeNotifier>> entry = pendingNotificationMap.pollFirstEntry();
1971                    final int storageId = entry.getKey();
1972
1973                    // For all pending notifications, back-track references and notify all field monitors for the field
1974                    for (FieldChangeNotifier notifier : entry.getValue()) {
1975                        assert notifier.getStorageId() == storageId;
1976                        final ArrayList<FieldMonitor> monitorList = new ArrayList<>(this.getMonitorsForField(storageId, false));
1977                        if (!monitorList.isEmpty())
1978                            this.notifyFieldMonitors(notifier, NavigableSets.singleton(notifier.getId()), monitorList, 0);
1979                    }
1980                }
1981            } finally {
1982                this.pendingNotifications.remove();
1983            }
1984        }
1985    }

1987    // Recursively back-track references along monitor paths and notify monitors when we reach the end (i.e., beginning)
1988    private void notifyFieldMonitors(FieldChangeNotifier notifier,
1989      NavigableSet<ObjId> objects, ArrayList<FieldMonitor> monitorList, int step) {
1990
1991        // Find the monitors for whom we have completed all the steps in their (inverse) path,
1992        // and group the remaining monitors by their next inverted reference path step.
1993        final HashMap<Integer, ArrayList<FieldMonitor>> remainingMonitorsMap = new HashMap<>();
1994        for (FieldMonitor monitor : monitorList) {
1995
1996            // Issue notification callback if we have back-tracked through the whole path
1997            if (monitor.path.length == step) {
1998                notifier.notify(this, monitor.listener, monitor.path, objects);
1999                continue;
2000            }
2001
2002            // Sort the unfinished monitors by their next (i.e., previous) reference field
2003            final int storageId = monitor.path[monitor.path.length - step - 1];
2004            ArrayList<FieldMonitor> remainingMonitors = remainingMonitorsMap.get(storageId);
2005            if (remainingMonitors == null) {
2006                remainingMonitors = new ArrayList<FieldMonitor>();
2007                remainingMonitorsMap.put(storageId, remainingMonitors);
2008            }
2009            remainingMonitors.add(monitor);
2010        }
2011
2012        // Invert references for each group of remaining monitors and recurse
2013        for (Map.Entry<Integer, ArrayList<FieldMonitor>> entry : remainingMonitorsMap.entrySet()) {
2014            final int storageId = entry.getKey();
2015            final boolean hasComplexIndex = this.schema.hasComplexIndex.contains(storageId);
2016
2017            // Gather all objects that refer to any object in our current "objects" set
2018            final ArrayList<NavigableSet<ObjId>> refsList = new ArrayList<>();
2019            for (ObjId object : objects) {
2020                final NavigableSet<ObjId> refs = this.queryIndex(storageId,
2021                  FieldType.OBJ_ID, FieldType.OBJ_ID, hasComplexIndex).get(object);
2022                if (refs != null)
2023                    refsList.add(refs);
2024            }
2025
2026            // Recurse on the union of those objects
2027            if (!refsList.isEmpty())
2028                this.notifyFieldMonitors(notifier, NavigableSets.union(refsList), entry.getValue(), step + 1);
2029        }
2030    }

2035     * Find all objects that refer to any object in the given target set through the specified path of references.
2036     *
2037     * @param path path of one or more reference fields (represented by storage IDs) through which to reach the target objects
2038     * @param targetObjects target objects
2039     * @return set of objects that refer to the {@code targetObjects} via {@code path}
2040     * @throws UnknownFieldException if {@code path} contains a storage ID that does not correspond to a {@link ReferenceField}
2041     * @throws IllegalArgumentException if {@code targetObjects} or {@code path} is null
2042     * @throws IllegalArgumentException if {@code path} is empty
2043     * @throws StaleTransactionException if this transaction is no longer usable
2044     */
2045    public NavigableSet<ObjId> invertReferencePath(int[] path, Iterable<ObjId> targetObjects) {
2046
2047        // Sanity check
2048        if (targetObjects == null)
2049            throw new IllegalArgumentException("null targetObjects");
2050        if (path == null)
2051            throw new IllegalArgumentException("null path");
2052        if (path.length == 0)
2053            throw new IllegalArgumentException("empty path");
2054
2055        // Verify all fields in the path are reference fields, and reverse their order
2056        final ArrayList<ReferenceFieldStorageInfo> fieldInfos = new ArrayList<>(path.length);
2057        for (int i = path.length - 1; i >= 0; i--)
2058            fieldInfos.add(this.schema.verifyStorageInfo(path[i], ReferenceFieldStorageInfo.class));
2059
2060        // Now invert references in reverse order
2061        NavigableSet<ObjId> result = null;
2062        for (ReferenceFieldStorageInfo fieldInfo : fieldInfos) {
2063
2064            // Gather all objects that refer to any object in our current target objects set
2065            final ArrayList<NavigableSet<ObjId>> refsList = new ArrayList<>();
2066            for (ObjId id : targetObjects) {
2067                final NavigableSet<ObjId> refs = this.queryIndex(fieldInfo, FieldType.OBJ_ID, fieldInfo.hasComplexIndex).get(id);
2068                if (refs != null)
2069                    refsList.add(refs);
2070            }
2071            if (refsList.isEmpty())
2072                return NavigableSets.empty();
2073
2074            // Recurse on the union of those objects
2075            targetObjects = result = NavigableSets.union(refsList);
2076        }
2077
2078        // Done
2079        return result;
2080    }

2298          FieldType.OBJ_ID, fieldInfo.hasComplexIndex).get(target);
2299        if (referringObjects == null)
2300            return NavigableSets.<ObjId>empty(FieldType.OBJ_ID);

2303        final NavigableSet<ObjId> versionObjects = this.queryVersion().get(versionNumber);
