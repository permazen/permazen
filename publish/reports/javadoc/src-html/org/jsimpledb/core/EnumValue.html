006package org.jsimpledb.core;
007
008/**
009 * Holds a non-null {@link Enum} value without actually referencing any Java {@link Enum} type.
010 * Instead, instances hold a name and ordinal value.
011 *
012 * <p>
013 * Instances are immutable.
014 * </p>
015 *
016 * @see EnumFieldType
017 * @see org.jsimpledb.EnumConverter
018 */
019public class EnumValue {
020
021    private final String name;
022    private final int ordinal;
023
024    /**
025     * Constructor taking name and ordinal value.
026     *
027     * @param name enum name
028     * @param ordinal enum ordinal value
029     * @throws IllegalArgumentException if {@code ordinal} is negative
030     * @throws IllegalArgumentException if {@code name} is null
031     */
032    public EnumValue(String name, int ordinal) {
033        if (name == null)
034            throw new IllegalArgumentException("null name");
035        if (ordinal < 0)
036            throw new IllegalArgumentException("invalid negative ordinal " + ordinal);
037        this.name = name;
038        this.ordinal = ordinal;
039    }
040
041    /**
042     * Constructor taking an {@link Enum} value.
043     *
044     * @param value enum value
045     * @throws IllegalArgumentException if {@code value} is null
046     */
047    public EnumValue(Enum<?> value) {
048        if (value == null)
049            throw new IllegalArgumentException("null value");
050        this.name = value.name();
051        this.ordinal = value.ordinal();
052    }
053
054    /**
055     * Get the enum name.
056     *
057     * @return enum value name
058     */
059    public String getName() {
060        return this.name;
061    }
062
063    /**
064     * Get the enum ordinal value.
065     *
066     * @return enum value ordinal
067     */
068    public int getOrdinal() {
069        return this.ordinal;
070    }
071
072    /**
073     * Find the instance in the given {@link Enum} type that matches this instance in both name and ordinal value, if any.
074     *
075     * <p>
076     * Note: to match only by name, just use {@link Enum#valueOf Enum.valueOf()}.
077     * </p>
078     *
079     * @param type {@link Enum} type
080     * @param <T> enum type
081     * @return matching instance of type {@code type}, or null if none exists
082     * @throws IllegalArgumentException if {@code type} is null
083     */
084    public <T extends Enum<T>> T find(Class<T> type) {
085        if (type == null)
086            throw new IllegalArgumentException("null type");
087        final T value;
088        try {
089            value = Enum.valueOf(type, this.name);
090        } catch (IllegalArgumentException e) {
091            return null;
092        }
093        return value.ordinal() == this.ordinal ? value : null;
094    }
095
096// Object
097
098    @Override
099    public String toString() {
100        return this.name + "#" + this.ordinal;
101    }
102
103    @Override
104    public int hashCode() {
105        return this.name.hashCode() ^ this.ordinal;
106    }
107
108    @Override
109    public boolean equals(Object obj) {
110        if (obj == this)
111            return true;
112        if (obj == null || obj.getClass() != this.getClass())
113            return false;
114        final EnumValue that = (EnumValue)obj;
115        return this.name.equals(that.name) && this.ordinal == that.ordinal;
116    }
117}
