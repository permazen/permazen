037/**
038 * Provides an object database abstraction on top of a key/value database.
039 *
040 * <p>
041 * Includes support for:
042 * <ul>
043 *  <li>Objects and fields defined by a {@link SchemaModel}, with positive schema verification</li>
044 *  <li>Simple values fields containing any atomic type, reference or custom {@link FieldType}</li>
045 *  <li>Complex fields of type {@link java.util.List}, {@link java.util.NavigableSet}, and {@link java.util.NavigableMap}</li>
046 *  <li>Invertable reference fields with strong referential integrity and configurable delete cascading</li>
047 *  <li>Configurable indexing of any simple field or complex sub-field</li>
048 *  <li>Composite indexes on multiple simple fields</li>
049 *  <li>Notification of object creation and deletion</li>
050 *  <li>Notification of object field changes, as seen through an arbitrary path of references</li>
051 *  <li>Automatic schema tracking and object versioning with schema change notification support</li>
052 * </ul>
053 *
054 * <p>
055 * See {@link Transaction} for further details on the above functionality.
056 * </p>
057 *
058 * <p>
059 * This class defines an abstraction layer that usually sits below a {@link org.jsimpledb.JSimpleDB} but is completely
060 * independent of {@link org.jsimpledb.JSimpleDB} and can be used on its own.
061 * Compared to {@link org.jsimpledb.JSimpleDB}, a {@link Database} has these differences:
062 * <ul>
063 *  <li>A {@link SchemaModel} must be explicitly provided to define the schema in use, whereas when using a
064 *      {@link org.jsimpledb.JSimpleDB} the schema is derived automatically from annotated Java model classes.</li>
065 *  <li>Object references are represented by {@link ObjId}s instead of Java objects, and there is no notion of object sub-type.
066 *      However, reference fields may be configured with a restricted set of referrable types.</li>
067 *  <li>All object types and fields are referenced by explicit storage ID.</li>
068 *  <li>Enum values are represented by {@link EnumValue} objects.</li>
069 *  <li>There is no automatic validation support.</li>
070 * </ul>
071 *
072 * @see Transaction
073 * @see org.jsimpledb
074 */
075public class Database {

182    /**
183     * Create a new {@link Transaction} on this database and use the specified schema version to access objects and fields.
184     *
185     * <p>
186     * <b>Schema Versions</b>
187     * </p>
188     *
189     * <p>
190     * Within each {@link Database} is stored a record of all schema versions previously used with the database.
191     * When creating a new transaction, the caller provides an expected schema version and corresponding {@link SchemaModel}.
192     * Both of these are optional: a schema version of zero means "use the highest version recorded in the
193     * database", and a null {@link SchemaModel} means "use the {@link SchemaModel} already recorded in the database under
194     * {@code version}".
195     * </p>
196     *
197     * <p>
198     * When this method is invoked, the following checks are applied:
199     * <ul>
200     *  <li>If a schema with version number {@code version != 0} is recorded in the database, and {@code schemaModel} is null or
201     *      matches it, then this method succeeds, and the {@link Transaction} will use that schema.</li>
202     *  <li>If a schema with version number {@code version} (or the highest numbered schema if {@code version == 0})
203     *      is recorded in the database, and {@code schemaModel} is not null and does not match it, then this method fails
204     *      and throws a {@link SchemaMismatchException}.</li>
205     *  <li>If {@code allowNewSchema} is false, and no schema with version number {@code version != 0} has yet been
206     *      recorded in the database, then this method fails and throws a {@link SchemaMismatchException}.</li>
207     *  <li>If {@code allowNewSchema} is true, and no schema with version number {@code version != 0} has yet been
208     *      recorded in the database, then if {@code schemaModel} is null a {@link SchemaMismatchException} is thrown;
209     *      otherwise {@code schemaModel} is checked for compabitility with the schemas previously recorded in the database;
210     *      if compatible, this method succeeds, {@code schema} is recorded in the database under the new version number
211     *      {@code version}, and the {@link Transaction} will use schema version {@code version};
212     *      otherwise a {@link SchemaMismatchException} is thrown.</li>
213     *  <li>If the database is uninitialized and {@code version == 0} or {@code schemaModel} is null,
214     *      a {@link SchemaMismatchException} is thrown.</li>
215     * </ul>
216     *
217     * <p>
218     * For two schemas to "match", they must be identical in all respects, except that object, field, and index names may differ.
219     * In the core API, objects and fields are identified by storage ID, not name.
220     * </p>
221     *
222     * <p>
223     * Schemas must also be compatible with all other schemas previously recorded in the database.
224     * Basically this means storage IDs must be used consistently from a structural point of view:
225     * <ul>
226     *  <li>Once a storage ID is assigned, it cannot be re-assigned to a different type of item (object or field).</li>
227     *  <li>Fields must have a consistent type and structural parent (object type or complex field).</li>
228     * </ul>
229     *
230     * <p>
231     * However, object types and fields may be added or removed across schema versions, field indexing may change,
232     * and reference field {@link DeleteAction}s may change.
233     * </p>
234     *
235     * <p>
236     * <b>Object Versions</b>
237     * </p>
238     *
239     * <p>
240     * Each object in a {@link Database} contains an internal version number that indicates its current schema version;
241     * this in turn dictates what fields that object contains.
242     * </p>
243     *
244     * <p>
245     * When an object is accessed during a {@link Transaction}, the object's version is compared to the {@code version} associated
246     * with that {@link Transaction}. If the versions are the same, no version change occurs and fields are accessed normally.
247     * </p>
248     *
249     * <p>
250     * If the object has a version {@code oldVersion} different from {@code version}, then depending on which {@link Transaction}
251     * method is invoked, the object version may be automatically updated to {@code version}. This will cause fields to be added
252     * or removed, as follows:
253     * <ul>
254     *  <li>Fields that are common to both schema versions remain unchanged (necessarily such fields have the same storage ID,
255     *      type, and structural parent).</li>
256     *  <li>Fields that exist in {@code oldVersion} but not in {@code version} are removed.</li>
257     *  <li>Fields that exist in {@code version} but not in {@code oldVersion} are initialized to their default values.</li>
258     *  <li>All {@link VersionChangeListener}s registered with the {@link Transaction} are notified.</li>
259     * </ul>
260     *
261     * <p>
262     * Note that compatibility between schema versions does not depend on the field name, nor does it depend on whether the field
263     * is indexed, or its {@link DeleteAction} (for reference fields). A field's index may be added or removed between schema
264     * versions without losing information, however, querying a field's index will only return those objects whose schema
265     * version corresponds to a schema in which the field is indexed. Similarly, the {@link DeleteAction} taken when a
266     * referenced object is deleted depends on the {@link DeleteAction} configured in the schema version of the object
267     * containing the reference.
268     * </p>
269     *
270     * <p>
271     * Note that an object's current schema version can go up as well as down, may change non-consecutively, and in fact
272     * nothing requires schema version numbers to be consecutive.
273     * </p>
274     *
275     * @param schemaModel schema to use with the new transaction, or null to use the schema already recorded in the database
276     * @param version the schema version number corresponding to {@code schemaModel}, or zero to use the highest recorded version
277     * @param allowNewSchema whether creating a new schema version is allowed
278     * @return newly created transaction
279     * @throws IllegalArgumentException if {@code version} is less than zero
280     * @throws InvalidSchemaException if {@code schemaModel} is invalid (i.e., does not pass validation checks)
281     * @throws SchemaMismatchException if {@code schemaModel} does not match schema version {@code version}
282     *  as recorded in the database
283     * @throws SchemaMismatchException if schema version {@code version} is not recorded in the database
284     *  and {@code allowNewSchema} is false
285     * @throws SchemaMismatchException if schema version {@code version} is not recorded in the database,
286     *  {@code allowNewSchema} is true, but {@code schemaModel} is incompatible with one or more other schemas
287     *  already recorded in the database (i.e., the same storage ID is used inconsistently between schema versions)
288     * @throws SchemaMismatchException
289     *  if the database is uninitialized and {@code version == 0} or {@code schemaModel} is null
290     * @throws InconsistentDatabaseException if inconsistent or invalid schema information is detected in the database
291     * @throws InconsistentDatabaseException if an uninitialized database is encountered but the database is not empty
292     * @throws IllegalStateException if no underlying {@link KVDatabase} has been configured for this instance
293     */
294    public Transaction createTransaction(final SchemaModel schemaModel, int version, final boolean allowNewSchema) {
295
296        // Sanity check
297        if (version < 0)
298            throw new IllegalArgumentException("invalid schema version: " + version);
299
300        // Validate schema
301        if (schemaModel != null)
302            schemaModel.validate();
303
304        // Open KV transaction
305        final KVTransaction kvt = this.kvdb.createTransaction();
306        boolean success = false;
307        if (this.log.isTraceEnabled()) {
308            this.log.trace("creating transaction using "
309              + (version != 0 ? "schema version " + version : "highest recorded schema version"));
310        }
311        try {
312
313            // Get iterator over meta-data key/value pairs
314            final byte[] metaDataPrefix = new byte[] { METADATA_PREFIX };
315            final Iterator<KVPair> metaDataIterator = kvt.getRange(metaDataPrefix,
316              ByteUtil.getKeyAfterPrefix(metaDataPrefix), false);
317
318            // Get format version; it should be first; if not found, database is uninitialized (and should be empty)
319            byte[] formatVersionBytes = null;
320            if (metaDataIterator.hasNext()) {
321                final KVPair pair = metaDataIterator.next();
322                if (!Arrays.equals(pair.getKey(), FORMAT_VERSION_KEY)) {
323                    throw new InconsistentDatabaseException("database is uninitialized but contains unrecognized garbage (key "
324                      + ByteUtil.toString(pair.getKey()) + ")");
325                }
326                formatVersionBytes = pair.getValue();
327            }
328
329            // Get database format object; check for an uninitialized database
330            final boolean uninitialized = formatVersionBytes == null;
331            final int formatVersion;
332            if (uninitialized) {
333
334                // Sanity checks
335                if (kvt.getAtLeast(new byte[0]) != null)
336                    throw new InconsistentDatabaseException("database is uninitialized but contains unrecognized garbage");
337                if (kvt.getAtMost(new byte[] { (byte)0xff }) != null)
338                    throw new InconsistentDatabaseException("inconsistent results from getAtLeast() and getAtMost()");
339                if (kvt.getRange(new byte[0], new byte[] { (byte)0xff }, false).hasNext())
340                    throw new InconsistentDatabaseException("inconsistent results from getAtLeast() and getRange()");
341                this.checkAddNewSchema(schemaModel, version, allowNewSchema);
342
343                // Initialize database
344                formatVersion = CURRENT_FORMAT_VERSION;
345                this.log.info("detected an uninitialized database; initializing with format version " + formatVersion);
346                final ByteWriter writer = new ByteWriter();
347                UnsignedIntEncoder.write(writer, CURRENT_FORMAT_VERSION);
348                kvt.put(FORMAT_VERSION_KEY.clone(), writer.getBytes());
349
350                // Sanity check again
351                formatVersionBytes = kvt.get(FORMAT_VERSION_KEY.clone());
352                if (formatVersionBytes == null || ByteUtil.compare(formatVersionBytes, writer.getBytes()) != 0)
353                    throw new InconsistentDatabaseException("database failed basic read/write test");
354                final KVPair lower = kvt.getAtLeast(new byte[0]);
355                if (lower == null || !lower.equals(new KVPair(FORMAT_VERSION_KEY, writer.getBytes())))
356                    throw new InconsistentDatabaseException("database failed basic read/write test");
357                final KVPair upper = kvt.getAtMost(new byte[] { (byte)0xff });
358                if (upper == null || !upper.equals(new KVPair(FORMAT_VERSION_KEY, writer.getBytes())))
359                    throw new InconsistentDatabaseException("database failed basic read/write test");
360            } else {
361                try {
362                    formatVersion = UnsignedIntEncoder.decode(formatVersionBytes);
363                } catch (IllegalArgumentException e) {
364                    throw new InconsistentDatabaseException("database contains invalid encoded format version "
365                      + ByteUtil.toString(formatVersionBytes) + " under key " + ByteUtil.toString(FORMAT_VERSION_KEY));
366                }
367            }
368            final boolean compressedSchemaXML;
369            switch (formatVersion) {
370            case FORMAT_VERSION_1:
371            case FORMAT_VERSION_2:
372                compressedSchemaXML = formatVersion >= FORMAT_VERSION_2;
373                break;
374            default:
375                throw new InconsistentDatabaseException("database contains unrecognized format version "
376                  + formatVersion + " under key " + ByteUtil.toString(FORMAT_VERSION_KEY));
377            }
378
379            // There should not be any other meta data prior to recorded schemas
380            if (metaDataIterator.hasNext()) {
381                final KVPair pair = metaDataIterator.next();
382                if (ByteUtil.compare(pair.getKey(), SCHEMA_KEY_PREFIX) < 0) {
383                    throw new InconsistentDatabaseException("database contains unrecognized garbage at key "
384                      + ByteUtil.toString(pair.getKey()));
385                }
386            }
387
388            // Check schema
389            Schemas schemas = null;
390            boolean firstAttempt = true;
391            while (true) {
392
393                // Get iterator over schema key/value pairs
394                final Iterator<KVPair> schemaIterator = kvt.getRange(SCHEMA_KEY_PREFIX.clone(),
395                  ByteUtil.getKeyAfterPrefix(SCHEMA_KEY_PREFIX), false);
396
397                // Read recorded database schema versions - should immediately follow FORMAT_VERSION_KEY
398                final TreeMap<Integer, byte[]> bytesMap = new TreeMap<>();
399                while (schemaIterator.hasNext()) {
400                    final KVPair pair = schemaIterator.next();
401
402                    // Sanity check
403                    if (ByteUtil.compare(pair.getKey(), SCHEMA_KEY_PREFIX) < 0) {
404                        throw new InconsistentDatabaseException("database contains unrecognized garbage key "
405                          + ByteUtil.toString(pair.getKey()));
406                    }
407
408                    // Stop at end of recorded schemas
409                    if (!ByteUtil.isPrefixOf(SCHEMA_KEY_PREFIX, pair.getKey()))
410                        break;
411
412                    // Decode schema version and get XML
413                    final int vers = UnsignedIntEncoder.read(new ByteReader(pair.getKey(), SCHEMA_KEY_PREFIX.length));
414                    if (vers == 0)
415                        throw new InconsistentDatabaseException("database contains an invalid schema version zero");
416                    bytesMap.put(vers, pair.getValue());
417                }
418
419                // Read and decode database schemas, avoiding rebuild if possible
420                schemas = this.lastSchemas;
421                if (schemas != null && !schemas.isSameVersions(bytesMap))
422                    schemas = null;
423                if (schemas == null) {
424                    try {
425                        schemas = this.buildSchemas(bytesMap, compressedSchemaXML);
426                    } catch (IllegalArgumentException e) {
427                        if (firstAttempt)
428                            throw new InconsistentDatabaseException("database contains invalid schema information", e);
429                        else
430                            throw new InvalidSchemaException("schema is not valid: " + e.getMessage(), e);
431                    }
432                }
433
434                // If no version specified, assume the highest recorded version
435                if (version == 0 && !bytesMap.isEmpty())
436                    version = bytesMap.lastKey();
437
438                // If transaction schema was not found in the database, add it and retry
439                if (!bytesMap.containsKey(version)) {
440
441                    // Log it
442                    if (bytesMap.isEmpty()) {
443                        if (!uninitialized)
444                            throw new InconsistentDatabaseException("database is initialized but contains zero schema versions");
445                    } else {
446                        this.log.info("schema version " + version
447                          + " not found in database; known versions are " + bytesMap.keySet());
448                    }
449
450                    // Check whether we can add a new schema version
451                    this.checkAddNewSchema(schemaModel, version, allowNewSchema);
452
453                    // Record new schema in database
454                    this.log.info("recording new schema version " + version + " into database");
455                    this.writeSchema(kvt, version, schemaModel, compressedSchemaXML);
456
457                    // Try again
458                    schemas = null;
459                    firstAttempt = false;
460                    continue;
461                }
462
463                // Compare transaction schema with the schema of the same version found in the database
464                if (this.log.isTraceEnabled())
465                    this.log.trace("found schema version " + version + " in database; known versions are " + bytesMap.keySet());
466                final SchemaModel dbSchemaModel = schemas.getVersion(version).getSchemaModel();
467                if (schemaModel != null) {
468                    if (!schemaModel.isCompatibleWith(dbSchemaModel)) {
469                        final Diffs diffs = schemaModel.differencesFrom(dbSchemaModel);
470                        this.log.error("schema mismatch:\n=== Database schema ===\n{}\n=== Provided schema ===\n{}"
471                          + "\n=== Differences ===\n{}", dbSchemaModel, schemaModel, diffs);
472                        throw new IllegalArgumentException("the provided transaction schema does not match the schema with version "
473                          + version + " that is already recorded in the database:\n" + diffs);
474                    } else if (this.log.isTraceEnabled() && !schemaModel.equals(dbSchemaModel)) {
475                        final Diffs diffs = schemaModel.differencesFrom(dbSchemaModel);
476                        this.log.trace("the provided schema differs from, but is compatible with, the database schema:\n{}", diffs);
477                    }
478                }
479                break;
480            }
481
482            // Save schema for next time
483            this.lastSchemas = schemas;
484
485            // Create transaction
486            final Transaction tx = new Transaction(this, kvt, schemas, version);
487            success = true;
488            return tx;
489        } finally {
490            if (!success) {
491                try {
492                    kvt.rollback();
493                } catch (KVTransactionException e) {
494                    // ignore
495                }
496            }
497        }
498    }

500    /**
501     * Validate a {@link SchemaModel}.
502     *
503     * <p>
504     * This method only performs "static" checks; it does not access the database and therefore
505     * does not validate the schema against existing schema versions previously recorded.
506     * It does however use the {@link FieldTypeRegistry} associated with this instance to look up field types.
507     * </p>
508     *
509     * <p>
510     * To validate a schema against the database contents as well, simply attempt to create a transaction
511     * via {@link #createTransaction createTransaction()}.
512     * </p>
513     *
514     * @param schemaModel schema to validate
515     * @throws InvalidSchemaException if {@code schemaModel} is invalid
516     * @throws IllegalArgumentException if {@code schemaModel} is null
517     */
518    public void validateSchema(SchemaModel schemaModel) {
519
520        // Sanity check
521        if (schemaModel == null)
522            throw new IllegalArgumentException("null schemaModel");
523
524        // Validate
525        schemaModel.validate();
526        try {
527            new Schema(1, new byte[0], schemaModel, this.fieldTypeRegistry);
528        } catch (IllegalArgumentException e) {
529            throw new InvalidSchemaException("invalid schema: " + e.getMessage(), e);
530        }
531    }
532
533    private void checkAddNewSchema(SchemaModel schemaModel, int version, boolean allowNewSchema) {
534        if (version == 0)
535            throw new SchemaMismatchException("database is uninitialized and no schema version was provided");
536        if (schemaModel == null) {
537            throw new SchemaMismatchException("schema version " + version
538              + " was not found in database, and no schema model was provided");
539        }
540        if (!allowNewSchema) {
541            throw new SchemaMismatchException("schema version " + version
542              + " was not found in database, and recording a new schema version is disabled in this transaction");
543        }
544    }

546    void copyMetaData(Transaction src, KVStore dst) {
547        for (Iterator<KVPair> i = src.kvt.getRange(new byte[] { METADATA_PREFIX }, VERSION_INDEX_PREFIX.clone(), false);
548          i.hasNext(); ) {
549            final KVPair pair = i.next();
550            dst.put(pair.getKey(), pair.getValue());
551        }
552    }

554    void reset(SnapshotTransaction tx) {
555        tx.kvt.removeRange(VERSION_INDEX_PREFIX.clone(), null);
556    }

558    static byte[] buildVersionIndexKey(ObjId id, int version) {
559        final ByteWriter writer = new ByteWriter(VERSION_INDEX_PREFIX.length + 1 + ObjId.NUM_BYTES);
560        writer.write(VERSION_INDEX_PREFIX);
561        UnsignedIntEncoder.write(writer, version);
562        id.writeTo(writer);
563        return writer.getBytes();
564    }

571    /**
572     * Build {@link Schemas} object from a schema version XMLs.
573     *
574     * @throws InconsistentDatabaseException if any recorded schema version is invalid
575     */
576    private Schemas buildSchemas(SortedMap<Integer, byte[]> bytesMap, boolean compressed) {
577        final TreeMap<Integer, Schema> versionMap = new TreeMap<>();
578        for (Map.Entry<Integer, byte[]> entry : bytesMap.entrySet()) {
579            final int version = entry.getKey();
580            final byte[] bytes = entry.getValue();
581            final SchemaModel schemaModel;
582            try {
583                schemaModel = this.decodeSchema(bytes, compressed);
584            } catch (InvalidSchemaException e) {
585                throw new InconsistentDatabaseException("found invalid schema version " + version + " recorded in database", e);
586            }
587            if (this.log.isTraceEnabled())
588                this.log.trace("read schema version {} from database:\n{}", version, schemaModel);
589            versionMap.put(version, new Schema(version, bytes, schemaModel, this.fieldTypeRegistry));
590        }
591        return new Schemas(versionMap);
592    }
593
594    /**
595     * Decode and validate schema XML.
596     *
597     * @throws InvalidSchemaException if schema is invalid
598     */
599    private SchemaModel decodeSchema(byte[] value, boolean compressed) {
600
601        // Decompress
602        if (compressed) {
603            try {
604                final Inflater inflater = new Inflater(true);
605                inflater.setDictionary(SCHEMA_XML_COMPRESSION_DICTIONARY);
606                inflater.setInput(Bytes.concat(value, new byte[1]));
607                final ByteArrayOutputStream decompressed = new ByteArrayOutputStream();
608                final byte[] temp = new byte[100];
609                int r;
610                while ((r = inflater.inflate(temp)) != 0)
611                    decompressed.write(temp, 0, r);
612                if (!inflater.finished())
613                    throw new RuntimeException("internal error: inflater did not finish");
614                inflater.end();
615                value = decompressed.toByteArray();
616            } catch (DataFormatException e) {
617                throw new InvalidSchemaException("error in compressed data", e);
618            }
619        }
620
621        // Decode XML
622        try {
623            return SchemaModel.fromXML(new ByteArrayInputStream(value));
624        } catch (IOException e) {
625            throw new RuntimeException("unexpected exception", e);
626        }
627    }
628
629    /**
630     * Record the given schema into the database.
631     */
632    private void writeSchema(KVTransaction kvt, int version, SchemaModel schemaModel, boolean compress) {
633
634        // Encode as XML
635        final ByteArrayOutputStream buf = new ByteArrayOutputStream();
636        try {
637            schemaModel.toXML(buf, false);
638        } catch (IOException e) {
639            throw new RuntimeException("unexpected exception", e);
640        }
641        byte[] value = buf.toByteArray();
642
643        // Compress
644        if (compress) {
645            final Deflater deflater = new Deflater(Deflater.BEST_COMPRESSION, true);
646            deflater.setDictionary(SCHEMA_XML_COMPRESSION_DICTIONARY);
647            deflater.setInput(value);
648            deflater.finish();
649            final ByteArrayOutputStream compressed = new ByteArrayOutputStream();
650            final byte[] temp = new byte[100];
651            int r;
652            while ((r = deflater.deflate(temp)) != 0)
653                compressed.write(temp, 0, r);
654            if (!deflater.finished())
655                throw new RuntimeException("internal error: deflater did not finish");
656            deflater.end();
657            value = compressed.toByteArray();
658        }
659
660        // Write schema
661        kvt.put(this.getSchemaKey(version), value);
662    }
663
664    /**
665     * Delete a schema version. Caller must verify no objects exist.
666     */
667    void deleteSchema(KVTransaction kvt, int version) {
668        kvt.remove(this.getSchemaKey(version));
669    }
670
671    private byte[] getSchemaKey(int version) {
672        final ByteWriter writer = new ByteWriter();
673        writer.write(SCHEMA_KEY_PREFIX);
674        UnsignedIntEncoder.write(writer, version);
675        return writer.getBytes();
676    }
