006package org.jsimpledb.core;
007
008import com.google.common.reflect.TypeParameter;
009import com.google.common.reflect.TypeToken;
010
011import java.util.ArrayList;
012import java.util.Collections;
013import java.util.List;
014
015import org.jsimpledb.util.ByteReader;
016import org.jsimpledb.util.ByteWriter;
017
018/**
019 * List field.
020 *
021 * <p>
022 * JSimpleDB list fields have performance characteristics similar to {@link ArrayList}.
023 * </p>
024 *
025 * @param <E> Java type for the list elements
026 */
027public class ListField<E> extends CollectionField<List<E>, E> {
028
029    /**
030     * Constructor.
031     *
032     * @param name the name of the field
033     * @param storageId field content storage ID
034     * @param schema schema version
035     * @param elementField this field's element sub-field
036     * @throws IllegalArgumentException if any parameter is null
037     * @throws IllegalArgumentException if {@code storageId} is non-positive
038     */
039    @SuppressWarnings("serial")
040    ListField(String name, int storageId, Schema schema, SimpleField<E> elementField) {
041        super(name, storageId, schema, new TypeToken<List<E>>() { }
042          .where(new TypeParameter<E>() { }, elementField.typeToken.wrap()), elementField);
043    }
044
045// Public methods
046
047    @Override
048    @SuppressWarnings("unchecked")
049    public List<E> getValue(Transaction tx, ObjId id) {
050        if (tx == null)
051            throw new IllegalArgumentException("null tx");
052        return (List<E>)tx.readListField(id, this.storageId, false);
053    }
054
055    @Override
056    public <R> R visit(FieldSwitch<R> target) {
057        return target.caseListField(this);
058    }
059
060    @Override
061    public String toString() {
062        return "list field `" + this.name + "' containing " + this.elementField;
063    }
064
065// Non-public methods
066
067    @Override
068    List<E> getValueInternal(Transaction tx, ObjId id) {
069        return new JSList<E>(tx, this, id);
070    }
071
072    @Override
073    List<E> getValueReadOnlyCopy(Transaction tx, ObjId id) {
074        return Collections.unmodifiableList(new ArrayList<E>(this.getValueInternal(tx, id)));
075    }
076
077    @Override
078    ListFieldStorageInfo<E> toStorageInfo() {
079        return new ListFieldStorageInfo<E>(this);
080    }
081
082    @Override
083    void copy(ObjId srcId, ObjId dstId, Transaction srcTx, Transaction dstTx) {
084        final List<E> srcList = this.getValue(srcTx, srcId);
085        final List<E> dstList = this.getValue(dstTx, dstId);
086        final int ssize = srcList.size();
087        final int dsize = dstList.size();
088        final int min = Math.min(ssize, dsize);
089        for (int i = 0; i < min; i++)
090            dstList.set(i, srcList.get(i));
091        if (ssize < dsize)
092            dstList.subList(ssize, dsize).clear();
093        else if (dsize < ssize)
094            dstList.addAll(srcList.subList(dsize, ssize));
095    }
096
097    @Override
098    void buildIndexEntry(ObjId id, SimpleField<?> subField, ByteReader reader, byte[] value, ByteWriter writer) {
099        assert subField == this.elementField;
100        writer.write(value);
101        id.writeTo(writer);
102        writer.write(reader);
103    }
104}
