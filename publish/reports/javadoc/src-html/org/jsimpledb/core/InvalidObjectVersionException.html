006package org.jsimpledb.core;
007
008/**
009 * Thrown when an object is found to have an invalid schema version.
010 * This indicates database or recorded schema corruption.
011 */
012@SuppressWarnings("serial")
013public class InvalidObjectVersionException extends InconsistentDatabaseException {
014
015    private final ObjId id;
016    private final int version;
017
018    InvalidObjectVersionException(ObjId id, int version) {
019        super("object " + id + " contains invalid schema version number "
020          + version + " which is not recorded in the database");
021        this.id = id;
022        this.version = version;
023    }
024
025    /**
026     * Get the ID of the object containing the invalid version number.
027     *
028     * @return object ID
029     */
030    public ObjId getId() {
031        return this.id;
032    }
033
034    /**
035     * Get the invalid version number found.
036     *
037     * @return invalid version number
038     */
039    public int getInvalidVersion() {
040        return this.version;
041    }
042}
