040/**
041 * A JSimpleDB {@link Database} transaction.
042 *
043 * <p>
044 * Note: this is the lower level, core API for {@link org.jsimpledb.JSimpleDB}. In most cases this API
045 * will only be used indirectly through the higher level {@link org.jsimpledb.JSimpleDB}, {@link org.jsimpledb.JTransaction},
046 * and {@link org.jsimpledb.JObject} APIs.
047 * </p>
048 *
049 * <p>
050 * Methods in this class can be divided into the following categories:
051 * </p>
052 *
053 * <p>
054 * <b>Transaction Meta-Data</b>
055 * <ul>
056 *  <li>{@link #getDatabase getDatabase()} - Get the associated {@link Database}</li>
057 *  <li>{@link #getSchemas getSchemas()} - Get the database {@link Schemas}, as seen by this transaction</li>
058 *  <li>{@link #getSchema() getSchema()} - Get the {@link Schema} that will be used by this transaction</li>
059 *  <li>{@link #deleteSchemaVersion deleteSchemaVersion()} - Delete a schema version that is no longer being used</li>
060 * </ul>
061 *
062 * <p>
063 * <b>Transaction Lifecycle</b>
064 * <ul>
065 *  <li>{@link #commit commit()} - Commit transaction</li>
066 *  <li>{@link #rollback rollback()} - Roll back transaction</li>
067 *  <li>{@link #isValid isValid()} - Test transaction validity</li>
068 *  <li>{@link #setTimeout setTimeout()} - Set transaction timeout</li>
069 *  <li>{@link #setReadOnly setReadOnly()} - Set transaction to read-only</li>
070 *  <li>{@link #setRollbackOnly setRollbackOnly()} - Set transaction for rollack only</li>
071 *  <li>{@link #addCallback addCallback()} - Register a {@link Callback} on transaction completion</li>
072 *  <li>{@link #createSnapshotTransaction createSnapshotTransaction()} - Create a empty, in-memory copy of this transaction</li>
073 * </ul>
074 *
075 * <p>
076 * <b>Object Lifecycle</b>
077 * <ul>
078 *  <li>{@link #create(int) create()} - Create a database object</li>
079 *  <li>{@link #delete delete()} - Delete a database object</li>
080 *  <li>{@link #exists exists()} - Test whether a database object exists</li>
081 *  <li>{@link #copy copy()} - Copy an object's fields onto another object in a (possibly) different transaction</li>
082 *  <li>{@link #addCreateListener addCreateListener()} - Register a {@link CreateListener} for notifications about new objects</li>
083 *  <li>{@link #removeCreateListener removeCreateListener()} - Unregister a {@link CreateListener}</li>
084 *  <li>{@link #addDeleteListener addDeleteListener()} - Register a {@link DeleteListener} for notifications
085 *      about object deletions</li>
086 *  <li>{@link #removeDeleteListener removeDeleteListener()} - Unregister a {@link DeleteListener}</li>
087 * </ul>
088 *
089 * <p>
090 * <b>Object Versioning</b>
091 * <ul>
092 *  <li>{@link #getSchemaVersion(ObjId) getSchemaVersion()} - Inspect an object's schema version</li>
093 *  <li>{@link #updateSchemaVersion updateSchemaVersion()} - Update an object's schema version to match this transaction</li>
094 *  <li>{@link #queryVersion queryVersion()} - Locate objects by schema version</li>
095 *  <li>{@link #addVersionChangeListener addVersionChangeListener()} - Register a {@link VersionChangeListener}
096 *      for notifications about object version changes</li>
097 *  <li>{@link #removeVersionChangeListener removeVersionChangeListener()} - Unregister a {@link VersionChangeListener}</li>
098 * </ul>
099 *
100 * <p>
101 * <b>Object and Field Access</b>
102 * <ul>
103 *  <li>{@link #getAll getAll(int)} - Get all objects, or all objects of a specific type</li>
104 *  <li>{@link #readSimpleField readSimpleField()} - Read the value of a {@link SimpleField} in an object</li>
105 *  <li>{@link #writeSimpleField writeSimpleField()} - Write the value of a {@link SimpleField} in an object</li>
106 *  <li>{@link #readCounterField readCounterField()} - Read the value of a {@link CounterField} in an object</li>
107 *  <li>{@link #writeCounterField writeCounterField()} - Write the value of a {@link CounterField} in an object</li>
108 *  <li>{@link #adjustCounterField adjustCounterField()} - Adjust the value of a {@link CounterField} in an object</li>
109 *  <li>{@link #readSetField readSetField()} - Access a {@link SetField} in an object as a {@link NavigableSet}</li>
110 *  <li>{@link #readListField readListField()} - Access a {@link ListField} in an object as a {@link List}</li>
111 *  <li>{@link #readMapField readMapField()} - Access a {@link MapField} in an object as a {@link NavigableMap}</li>
112 *  <li>{@link #getKey getKey(ObjId)} - Get the {@link org.jsimpledb.kv.KVDatabase} key corresponding to an object</li>
113 *  <li>{@link #getKey getKey(ObjId, int)} - Get the {@link org.jsimpledb.kv.KVDatabase}
114 *      key corresponding to a field in an object</li>
115 * </ul>
116 *
117 * <p>
118 * <b>Field Change Notifications</b>
119 * <ul>
120 *  <li>{@link #addSimpleFieldChangeListener addSimpleFieldChangeListener()} - Register a {@link SimpleFieldChangeListener} for
121 *      notifications of changes in a {@link SimpleField}, as seen through a path of object references</li>
122 *  <li>{@link #addSetFieldChangeListener addSetFieldChangeListener()} - Register a {@link SetFieldChangeListener} for
123 *      notifications of changes in a {@link SetField}, as seen through a path of object references</li>
124 *  <li>{@link #addListFieldChangeListener addListFieldChangeListener()} - Register a {@link ListFieldChangeListener} for

126 *  <li>{@link #addMapFieldChangeListener addMapFieldChangeListener()} - Register a {@link MapFieldChangeListener} for
127 *      notifications of changes in a {@link MapField}, as seen through a path of object references</li>

130 *  <li>{@link #removeSetFieldChangeListener removeSetFieldChangeListener()} - Unregister a previously registered
131 *      {@link SetFieldChangeListener}</li>
132 *  <li>{@link #removeListFieldChangeListener removeListFieldChangeListener()} - Unregister a previously registered
133 *      {@link ListFieldChangeListener}</li>
134 *  <li>{@link #removeMapFieldChangeListener removeMapFieldChangeListener()} - Unregister a previously registered
135 *      {@link MapFieldChangeListener}</li>
136 * </ul>
137 *
138 * <p>
139 * <b>Reference Inversion</b>
140 * <ul>
141 *  <li>{@link #invertReferencePath invertReferencePath()} - Find all objects that refer to any element in a given set
142 *      of objects through a specified reference path</li>
143 * </ul>
144 *
145 * <p>
146 * <b>Index Queries</b>
147 * <ul>
148 *  <li>{@link #queryIndex queryIndex()} - Query the index associated with a {@link SimpleField}
149 *      to identify all values and all objects having those values</li>
150 *  <li>{@link #queryListElementIndex queryListElementIndex()} - Query the index associated with a {@link ListField}
151 *      to identify all list elements, all objects having those elements in the list, and thier corresponding indicies</li>
152 *  <li>{@link #queryMapValueIndex queryMapValueIndex()} - Query the index associated with a {@link MapField}
153 *      to identify all map values, all objects having those values in the map, and the corresponding keys</li>
154 *  <li>{@link #queryCompositeIndex queryCompositeIndex()} - Query any composite index</li>
155 *  <li>{@link #queryCompositeIndex2 queryCompositeIndex2()} - Query a composite index on two fields</li>
156 *  <li>{@link #queryCompositeIndex3 queryCompositeIndex3()} - Query a composite index on three fields</li>
157 *  <li>{@link #queryCompositeIndex3 queryCompositeIndex4()} - Query a composite index on four fields</li>

234    /**
235     * Delete a schema version that is no longer being used. There must be no objects with the given version
236     * in the database, and {@code version} must not be the version being used by this transaction.
237     *
238     * @param version schema version to remove
239     * @return true if schema version was found and removed, false if schema version does not exist in database
240     * @throws InvalidSchemaException if one or more objects with schema version {@code version} still exist
241     * @throws InvalidSchemaException if {@code version} is equal to this transaction's version
242     * @throws IllegalArgumentException if {@code version} is zero or negative
243     * @throws StaleTransactionException if this transaction is no longer usable
244     */
245    public synchronized boolean deleteSchemaVersion(int version) {
246
247        // Sanity check
248        if (version <= 0)
249            throw new IllegalArgumentException("invalid schema version " + version);
250        if (version == this.schema.getVersionNumber())
251            throw new InvalidSchemaException("version " + version + " is this transaction's version");
252        if (this.stale)
253            throw new StaleTransactionException(this);
254        if (this.queryVersion().asMap().containsKey(version))
255            throw new InvalidSchemaException("one or more version " + version + " objects still exist in database");
256
257        // Delete schema version
258        if (!this.schemas.deleteVersion(version))
259            return false;
260        this.db.deleteSchema(this.kvt, version);
261        return true;
262    }

280    /**
281     * Commit this transaction.
282     *
283     * @throws StaleTransactionException if this transaction is no longer usable
284     * @throws org.jsimpledb.kv.RetryTransactionException from {@link KVTransaction#commit KVTransaction.commit()}
285     * @throws RollbackOnlyTransactionException if this instance has been {@linkplain #setRollbackOnly marked} rollback only;
286     *  this instance will be automatically rolled back
287     */
288    public synchronized void commit() {
289
290        // Sanity check
291        if (this.stale)
292            throw new StaleTransactionException(this);
293
294        // Rollback only?
295        if (this.rollbackOnly) {
296            this.log.debug("commit() invoked on transaction " + this + " marked rollback-only, rolling back");
297            this.rollback();
298            throw new RollbackOnlyTransactionException(this);
299        }
300        this.stale = true;
301
302        // Do before completion callbacks
303        if (this.log.isTraceEnabled())
304            this.log.trace("commit() invoked on" + (this.readOnly ? " read-only" : "") + " transaction " + this);
305        Callback failedCallback = null;
306        try {
307            for (Callback callback : this.callbacks) {
308                failedCallback = callback;
309                callback.beforeCommit(this.readOnly);
310            }
311            failedCallback = null;
312        } finally {
313            if (failedCallback != null) {
314                this.log.warn("error invoking beforeCommit() method on transaction "
315                  + this + " callback " + failedCallback + ", rolling back");
316            }
317            this.triggerBeforeCompletion();
318            if (failedCallback != null) {
319                try {
320                    try {
321                        this.kvt.rollback();
322                    } catch (KVTransactionException e) {
323                        // ignore
324                    }
325                } finally {
326                    this.triggerAfterCompletion(false);
327                }
328            }
329        }
330
331        // Commit KVTransaction and trigger after completion callbacks
332        failedCallback = null;
333        try {
334            this.kvt.commit();
335            for (Callback callback : this.callbacks) {
336                failedCallback = callback;
337                callback.afterCommit();
338            }
339            failedCallback = null;
340        } finally {
341            if (failedCallback != null)
342                this.log.warn("error invoking afterCommit() method on transaction " + this + " callback " + failedCallback);
343            this.triggerAfterCompletion(true);
344        }
345    }

347    /**
348     * Roll back this transaction.
349     *
350     * @throws StaleTransactionException if this transaction is no longer usable
351     */
352    public synchronized void rollback() {
353
354        // Sanity check
355        if (this.stale)
356            throw new StaleTransactionException(this);
357        this.stale = true;
358        if (this.log.isTraceEnabled())
359            this.log.trace("rollback() invoked on" + (this.readOnly ? " read-only" : "") + " transaction " + this);
360
361        // Do before completion callbacks
362        this.triggerBeforeCompletion();
363
364        // Roll back KVTransaction and trigger after completion callbacks
365        try {
366            this.kvt.rollback();
367        } finally {
368            this.triggerAfterCompletion(false);
369        }
370    }
371
372    private /*synchronized*/ void triggerBeforeCompletion() {
373        for (Callback callback : this.callbacks) {
374            try {
375                callback.beforeCompletion();
376            } catch (Throwable t) {
377                this.log.error("error invoking beforeCompletion() method on transaction callback " + callback, t);
378            }
379        }
380    }
381
382    private /*synchronized*/ void triggerAfterCompletion(boolean committed) {
383        for (Callback callback : this.callbacks) {
384            try {
385                callback.afterCompletion(committed);
386            } catch (Throwable t) {
387                this.log.error("error invoking afterCompletion() method on transaction callback " + callback, t);
388            }
389        }
390    }

534
535        // Sanity check
536        if (id == null)
537            throw new IllegalArgumentException("null id");
538        if (this.stale)
539            throw new StaleTransactionException(this);
540
541        // Does object already exist?
542        if (this.exists(id))
543            return false;
544
545        // Initialize object
546        this.initialize(id, this.schemas.getVersion(versionNumber).getObjType(id.getStorageId()));
547
548        // Done
549        return true;

622    private /*synchronized*/ ObjId generateIdValidated(int storageId) {
623
624        // Create a new, unique key
625        final ByteWriter keyWriter = new ByteWriter();
626        for (int attempts = 0; attempts < MAX_GENERATED_KEY_ATTEMPTS; attempts++) {
627            final ObjId id = new ObjId(storageId);
628            id.writeTo(keyWriter);
629            if (this.kvt.get(keyWriter.getBytes()) == null)
630                return id;
631            keyWriter.reset(0);
632        }
633
634        // Give up
635        throw new DatabaseException("could not find a new, unused object ID after "
636          + MAX_GENERATED_KEY_ATTEMPTS + " attempts; is our source of randomness truly random?");
637    }

674    /**
675     * Delete an object. Does nothing if object does not exist (e.g., has already been deleted).
676     *
677     * <p>
678     * This method does <i>not</i> change the object's schema version if it has a different version that this transaction.
679     * </p>
680     *
681     * <p><b>Secondary Deletions</b></p>
682     *
683     * <p>
684     * Deleting an object can trigger additional secondary deletions. Specifically,
685     * (a) if the object contains reference fields with {@linkplain ReferenceField#cascadeDelete delete cascade} enabled,
686     * any objects referred to through those fields will also be deleted, and (b) if the object is referred to by any other objects
687     * through fields configured for {@link DeleteAction#DELETE}, those referring objects will be deleted.
688     * </p>
689     *
690     * <p>
691     * In any case, deletions occur one at a time, and only when an object is actually deleted are any associated secondary
692     * deletions added to an internal deletion queue. However, the order in which objects on this deletion queue are
693     * processed is unspecified. For an example of where this ordering matters, consider an object {@code A} referring to objects
694     * {@code B} and {@code C} with delete cascading references, where B also refers to C with a {@link DeleteAction#EXCEPTION}
695     * reference. Then if {@code A} is deleted, it's indeterminate whether a {@link ReferencedObjectException} will be thrown,
696     * as that depends on whether {@code B} or {@code C} is deleted first (with the answer being, respectively, no and yes).
697     * </p>
698     *
699     * @param id object ID of the object to delete
700     * @return true if object was found and deleted, false if object was not found,
701     *  or if {@code id} specifies an unknown object type
702     * @throws ReferencedObjectException if the object is referenced by some other object
703     *  through a reference field configured for {@link DeleteAction#EXCEPTION}
704     * @throws IllegalArgumentException if {@code id} is null
705     * @throws ReadOnlyTransactionException if this transaction has been {@linkplain #setReadOnly set read-only}
706     * @throws StaleTransactionException if this transaction is no longer usable
707     */
708    public synchronized boolean delete(ObjId id) {
709
710        // Sanity check
711        if (id == null)
712            throw new IllegalArgumentException("null id");
713        if (this.stale)
714            throw new StaleTransactionException(this);
715        if (this.readOnly)
716            throw new ReadOnlyTransactionException(this);
717
718        // Get object info
719        try {
720            this.getObjectInfo(id, false);
721        } catch (DeletedObjectException | UnknownTypeException e) {
722            return false;
723        }
724
725        // Handle delete cascade and recurive DeleteAction.DELETE without hogging Java stack
726        final ObjIdSet deletables = new ObjIdSet();
727        deletables.add(id);
728        boolean found = false;
729        while (!deletables.isEmpty())
730            found |= this.doDelete(deletables.iterator().next(), deletables);
731
732        // Done
733        return found;
734    }

736    private synchronized boolean doDelete(final ObjId id, ObjIdSet deletables) {
737
738        // Loop here to handle any mutations within delete notification listener callbacks
739        ObjInfo info;
740        while (true) {
741
742            // Get object info
743            try {
744                info = this.getObjectInfo(id, false);
745            } catch (DeletedObjectException e) {                    // possibly due to a cycle of DeleteAction.DELETE references
746                deletables.remove(id);
747                return false;
748            } catch (UnknownTypeException e) {
749                throw new InconsistentDatabaseException("encountered reference with unknown type during delete cascade: " + id, e);
750            }
751
752            // Determine if any EXCEPTION reference fields refer to the object (from some other object); if so, throw exception
753            for (ReferenceFieldStorageInfo fieldInfo :
754              Iterables.filter(this.schemas.storageInfos.values(), ReferenceFieldStorageInfo.class)) {
755                for (ObjId referrer : this.findReferrers(id, DeleteAction.EXCEPTION, fieldInfo.storageId)) {
756                    if (!referrer.equals(id))
757                        throw new ReferencedObjectException(id, referrer, fieldInfo.storageId);
758                }
759            }
760
761            // Do we need to issue delete notifications for the object being deleted?
762            if (info.isDeleteNotified() || this.deleteListeners.isEmpty())
763                break;
764
765            // Issue delete notifications and retry
766            ObjInfo.write(this, id, info.getVersion(), true);
767            for (DeleteListener listener : this.deleteListeners.toArray(new DeleteListener[this.deleteListeners.size()]))
768                listener.onDelete(this, id);
769        }
770
771        // Find all objects referred to by a reference field with cascadeDelete = true and add them to deletables
772        for (ReferenceField field : Iterables.filter(
773          Iterables.filter(info.getObjType().referenceFields.values(), ReferenceField.class), new HasCascadeDelete())) {
774            final Iterable<ObjId> refs = field.parent != null ?
775              field.parent.iterateSubField(this, id, field) : Collections.singleton(field.getValue(this, id));
776            for (ObjId ref : refs) {
777                if (ref != null)
778                    deletables.add(ref);
779            }
780        }
781
782        // Actually delete the object
783        this.deleteObjectData(info);
784        deletables.remove(id);
785
786        // Find all UNREFERENCE references and unreference them
787        for (ReferenceFieldStorageInfo fieldInfo :
788          Iterables.filter(this.schemas.storageInfos.values(), ReferenceFieldStorageInfo.class)) {
789            final NavigableSet<ObjId> referrers = this.findReferrers(id, DeleteAction.UNREFERENCE, fieldInfo.storageId);
790            if (fieldInfo.isSubField()) {
791                final ComplexFieldStorageInfo<?> superFieldInfo = this.schemas.verifyStorageInfo(
792                  fieldInfo.superFieldStorageId, ComplexFieldStorageInfo.class);
793                superFieldInfo.unreferenceAll(this, fieldInfo.storageId, id, referrers);
794            } else {
795                final int storageId = fieldInfo.storageId;
796                for (ObjId referrer : referrers)
797                    this.writeSimpleField(referrer, storageId, null, false);
798            }
799        }
800
801        // Find all DELETE references and mark the containing object for deletion (caller will call us back to actually delete)
802        for (ObjId referrer : this.findReferrers(id, DeleteAction.DELETE, -1))
803            deletables.add(referrer);
804
805        // Done
806        return true;
807    }

809    // Delete all of an object's data
810    private void deleteObjectData(ObjInfo info) {
811
812        // Delete object's simple field index entries
813        final ObjId id = info.getId();
814        final ObjType type = info.getObjType();
815        for (SimpleField<?> field : type.simpleFields.values()) {
816            if (field.indexed)
817                this.kvt.remove(Transaction.buildSimpleIndexEntry(field, id, this.kvt.get(field.buildKey(id))));
818        }
819
820        // Delete object's composite index entries
821        for (CompositeIndex index : type.compositeIndexes.values())
822            this.kvt.remove(this.buildCompositeIndexEntry(id, index));
823
824        // Delete object's complex field index entries
825        for (ComplexField<?> field : type.complexFields.values())
826            field.removeIndexEntries(this, id);
827
828        // Delete object meta-data and all field content
829        final byte[] minKey = info.getId().getBytes();
830        final byte[] maxKey = ByteUtil.getKeyAfterPrefix(minKey);
831        this.kvt.removeRange(minKey, maxKey);
832
833        // Delete object schema version entry
834        this.kvt.remove(Database.buildVersionIndexKey(id, info.getVersion()));
835    }

859    /**
860     * Copy all of an object's fields onto a target object in a (possibly) different transaction, replacing any previous values.
861     *
862     * <p>
863     * If necessary, {@code source} object is first upgraded to {@linkplain #getSchema() the schema version associated with
864     * this transaction}, which must be identical in this transaction and {@code dest}.
865     * </p>
866     *
867     * <p>
868     * Only the object's fields are copied; any other objects they reference are not copied. If the {@code target} object
869     * does not exist in {@code dest}, it will be created first (and {@link CreateListener}s notified); if {@code target}
870     * does exist in {@code dest}, it's schema version will be upgraded if necessary to match {@code source} (and any registered
871     * {@link VersionChangeListener}s notified).
872     * (Meaningful) changes to {@code target}'s fields generate change listener notifications.
873     * </p>
874     *
875     * <p>
876     * Note: if two threads attempt to copy objects between the same two transactions at the same time but in opposite directions,
877     * deadlock could result.
878     * </p>
879     *
880     * <p>
881     * If {@code dest} is this instance, and {@code source} equals {@code target}, no changes are made and false is returned.
882     * </p>
883     *
884     * @param source object ID of the source object in this transaction
885     * @param target object ID of the target object in {@code dest}
886     * @param dest destination transaction containing {@code target} (possibly same as this transaction)
887     * @param updateVersion true to first automatically update the object's schema version, false to not change it
888     * @return false if object already existed in {@code dest}, true if {@code target} did not exist in {@code dest}
889     * @throws DeletedObjectException if no object with ID equal to {@code id} is found in this transaction
890     * @throws UnknownTypeException if {@code source} or {@code target} specifies an unknown object type
891     * @throws IllegalArgumentException if {@code source} and {@code target} specify different object types
892     * @throws IllegalArgumentException if any parameter is null
893     * @throws IllegalArgumentException if any {@code source} and {@code target} have different object types
894     * @throws ReadOnlyTransactionException if {@code dest} has been {@linkplain #setReadOnly set read-only}
895     * @throws StaleTransactionException if this transaction or {@code dest} is no longer usable
896     * @throws SchemaMismatchException if the schema version associated with {@code source} differs between
897     *  this transaction and {@code dest}
898     * @throws TypeNotInSchemaVersionException {@code updateVersion} is true and the object could not be updated because
899     *   the object's type does not exist in the schema version associated with this transaction
900     */
901    public synchronized boolean copy(ObjId source, final ObjId target, final Transaction dest, final boolean updateVersion) {
902
903        // Sanity check
904        if (source == null)
905            throw new IllegalArgumentException("null source");
906        if (target == null)
907            throw new IllegalArgumentException("null target");
908        if (dest == null)
909            throw new IllegalArgumentException("null dest");
910        if (this.stale)
911            throw new StaleTransactionException(this);
912
913        // Do nothing if nothing to do
914        if (this == dest && source.equals(target))
915            return false;
916
917        // Get object info
918        final ObjInfo srcInfo = this.getObjectInfo(source, updateVersion);
919
920        // Do the copy while both transactions are locked
921        synchronized (dest) {
922
923            // Sanity check
924            if (dest.stale)
925                throw new StaleTransactionException(dest);
926            if (dest.readOnly)
927                throw new ReadOnlyTransactionException(dest);
928
929            // Copy fields
930            return dest.mutateAndNotify(new Mutation<Boolean>() {
931                @Override
932                public Boolean mutate() {
933                    return Transaction.doCopyFields(srcInfo, target, Transaction.this, dest, updateVersion);
934                }
935            });
936        }
937    }

939    // This method assumes both transactions are locked
940    private static boolean doCopyFields(ObjInfo srcInfo, ObjId dstId, Transaction srcTx, Transaction dstTx, boolean updateVersion) {
941
942        // Sanity check
943        final ObjId srcId = srcInfo.getId();
944        if (srcId.equals(dstId) && srcTx == dstTx)
945            throw new RuntimeException("internal error");
946
947        // Verify objects have the same type
948        if (srcId.getStorageId() != dstId.getStorageId()) {
949            throw new IllegalArgumentException("can't copy " + srcId + " to " + dstId
950              + " due to non-equal object types (" + srcId.getStorageId() + " != " + dstId.getStorageId() + ")");
951        }
952
953        // Upgrade source object if necessary
954        if (updateVersion && srcInfo.getVersion() != srcTx.schema.versionNumber) {
955            srcTx.updateVersion(srcInfo, srcTx.schema);
956            srcInfo = new ObjInfo(srcTx, srcId);
957        }
958
959        // Find and verify source object's schema version in destination transaction
960        final int objectVersion = srcInfo.getVersion();
961        final Schema dstSchema;
962        try {
963            dstSchema = dstTx.schemas.getVersion(objectVersion);
964        } catch (IllegalArgumentException e) {
965            throw new SchemaMismatchException("destination transaction has no schema version " + objectVersion);
966        }
967        if (!Arrays.equals(srcTx.schema.encodedXML, dstTx.schema.encodedXML))
968            throw new SchemaMismatchException("destination transaction schema version " + objectVersion + " does not match");
969
970        // Create destination object if it doesn't already exist
971        final boolean existed = !dstTx.create(dstId, objectVersion);
972        final ObjInfo dstInfo = dstTx.getObjectInfo(dstId, false);
973        final boolean needUpgrade = dstInfo.getVersion() != objectVersion;      // can only happen if object already existed
974
975        // Do field-by-field copy if there are change or version listeners, otherwise do fast copy of key/value pairs
976        final ObjType type = srcInfo.getObjType();
977        if (dstTx.hasFieldMonitor(type) || (needUpgrade && !dstTx.versionChangeListeners.isEmpty())) {
978
979            // Upgrade object first
980            if (needUpgrade)
981                dstTx.updateVersion(dstInfo, dstSchema);
982
983            // Copy fields
984            for (Field<?> field : type.fields.values())
985                field.copy(srcId, dstId, srcTx, dstTx);
986        } else {
987
988            // Delete pre-existing object's field content, if any
989            dstTx.deleteObjectData(dstInfo);
990
991            // Add schema version index entry
992            dstTx.kvt.put(Database.buildVersionIndexKey(dstId, objectVersion), ByteUtil.EMPTY);
993
994            // Copy object meta-data and all field content in one key range sweep
995            final byte[] srcMinKey = srcId.getBytes();
996            final byte[] srcMaxKey = ByteUtil.getKeyAfterPrefix(srcMinKey);
997            final ByteWriter dstWriter = new ByteWriter();
998            dstWriter.write(dstId.getBytes());
999            final int dstMark = dstWriter.mark();
1000            for (Iterator<KVPair> i = srcTx.kvt.getRange(srcMinKey, srcMaxKey, false); i.hasNext(); ) {
1001                final KVPair kv = i.next();
1002                final ByteReader srcReader = new ByteReader(kv.getKey());
1003                srcReader.skip(srcMinKey.length);
1004                dstWriter.reset(dstMark);
1005                dstWriter.write(srcReader);
1006                dstTx.kvt.put(dstWriter.getBytes(), kv.getValue());
1007            }
1008
1009            // Create object's simple field index entries
1010            for (SimpleField<?> field : type.simpleFields.values()) {
1011                if (field.indexed) {
1012                    final byte[] fieldValue = dstTx.kvt.get(field.buildKey(dstId));     // can be null (if field has default value)
1013                    final byte[] indexKey = Transaction.buildSimpleIndexEntry(field, dstId, fieldValue);
1014                    dstTx.kvt.put(indexKey, ByteUtil.EMPTY);
1015                }
1016            }
1017
1018            // Create object's composite index entries
1019            for (CompositeIndex index : type.compositeIndexes.values())
1020                dstTx.kvt.put(Transaction.buildCompositeIndexEntry(dstTx, dstId, index), ByteUtil.EMPTY);
1021
1022            // Create object's complex field index entries
1023            for (ComplexField<?> field : type.complexFields.values()) {
1024                for (SimpleField<?> subField : field.getSubFields()) {
1025                    if (subField.indexed)
1026                        field.addIndexEntries(dstTx, dstId, subField);
1027                }
1028            }
1029        }
1030
1031        // Done
1032        return !existed;
1033    }

1165    /**
1166     * Update object to the current schema version.
1167     *
1168     * @param info original object info
1169     * @param targetVersion version to change to
1170     */
1171    private synchronized void updateVersion(final ObjInfo info, final Schema targetVersion) {
1172
1173        // Get version numbers
1174        final ObjId id = info.getId();
1175        final int oldVersion = info.getVersion();
1176        final int newVersion = targetVersion.versionNumber;
1177
1178        // Sanity check
1179        assert this.schemas.getVersion(targetVersion.versionNumber) == targetVersion;
1180        if (newVersion == oldVersion)
1181            throw new IllegalArgumentException("object already at version " + newVersion);
1182        if (this.readOnly)
1183            throw new ReadOnlyTransactionException(this);
1184
1185        // Get old and new types
1186        final ObjType oldType = info.getObjType();
1187        final ObjType newType;
1188        try {
1189            newType = targetVersion.getObjType(id.getStorageId());
1190        } catch (UnknownTypeException e) {
1191            throw (TypeNotInSchemaVersionException)new TypeNotInSchemaVersionException(id, newVersion).initCause(e);
1192        }
1193
1194        // Gather removed fields' values here for user migration if any VersionChangeListeners are registered
1195        final TreeMap<Integer, Object> oldValueMap = !this.versionChangeListeners.isEmpty() ?
1196          new TreeMap<Integer, Object>() : null;
1197
1198    //////// Remove the index entries corresponding to removed composite indexes
1199
1200        // Get composite index storage IDs (old or new)
1201        final TreeSet<Integer> compositeIndexStorageIds = new TreeSet<>();
1202        compositeIndexStorageIds.addAll(oldType.compositeIndexes.keySet());
1203        compositeIndexStorageIds.addAll(newType.compositeIndexes.keySet());
1204
1205        // Remove index entries for composite indexes that are going away
1206        for (CompositeIndex index : oldType.compositeIndexes.values()) {
1207            if (!newType.compositeIndexes.containsKey(index.storageId))
1208                this.kvt.remove(this.buildCompositeIndexEntry(id, index));
1209        }
1210
1211    //////// Update counter fields
1212
1213        // Get counter field storage IDs (old or new)
1214        final TreeSet<Integer> counterFieldStorageIds = new TreeSet<>();
1215        counterFieldStorageIds.addAll(oldType.counterFields.keySet());
1216        counterFieldStorageIds.addAll(newType.counterFields.keySet());
1217
1218        // Update counter fields
1219        for (int storageId : counterFieldStorageIds) {
1220
1221            // Get old and new counter fields having this storage ID
1222            final CounterField oldField = oldType.counterFields.get(storageId);
1223            final CounterField newField = newType.counterFields.get(storageId);
1224
1225            // Save old field values for version change notification
1226            final byte[] key = Field.buildKey(id, storageId);
1227            if (oldField != null && oldValueMap != null) {
1228                final byte[] oldValue = this.kvt.get(key);
1229                final long value = oldValue != null ? this.kvt.decodeCounter(oldValue) : 0;
1230                oldValueMap.put(storageId, value);
1231            }
1232
1233            // Remove old value if field has disappeared, otherwise leave alone
1234            if (newField == null)
1235                this.kvt.remove(key);
1236        }
1237
1238    //////// Update simple fields and corresponding index entries
1239
1240        // Get simple field storage IDs (old or new)
1241        final TreeSet<Integer> simpleFieldStorageIds = new TreeSet<>();
1242        simpleFieldStorageIds.addAll(oldType.simpleFields.keySet());
1243        simpleFieldStorageIds.addAll(newType.simpleFields.keySet());
1244
1245        // Update simple fields
1246        for (int storageId : simpleFieldStorageIds) {
1247
1248            // Get old and new simple fields having this storage ID
1249            final SimpleField<?> oldField = oldType.simpleFields.get(storageId);
1250            final SimpleField<?> newField = newType.simpleFields.get(storageId);
1251
1252            // Save old field values for version change notification
1253            final byte[] key = Field.buildKey(id, storageId);
1254            final byte[] oldValue = oldField != null ? this.kvt.get(key) : null;
1255            if (oldField != null && oldValueMap != null) {
1256                final byte[] bytes = oldValue != null ? oldValue : oldField.fieldType.getDefaultValue();
1257                final Object value = oldField.fieldType.read(new ByteReader(bytes));
1258                oldValueMap.put(storageId, value);
1259            }
1260
1261            // If field is being removed, discard the old value
1262            if (newField == null && oldValue != null)
1263                this.kvt.remove(key);
1264
1265            // Remove old index entry if index removed in new version
1266            if (oldField != null && oldField.indexed && (newField == null || !newField.indexed))
1267                this.kvt.remove(Transaction.buildSimpleIndexEntry(oldField, id, oldValue));
1268
1269            // Add new index entry if index added in new version
1270            if (newField != null && newField.indexed && (oldField == null || !oldField.indexed))
1271                this.kvt.put(Transaction.buildSimpleIndexEntry(newField, id, oldValue), ByteUtil.EMPTY);
1272        }
1273
1274    //////// Add composite index entries for newly added composite indexes
1275
1276        // Add index entries for composite indexes that are newly added
1277        for (CompositeIndex index : newType.compositeIndexes.values()) {
1278            if (!oldType.compositeIndexes.containsKey(index.storageId))
1279                this.kvt.put(this.buildCompositeIndexEntry(id, index), ByteUtil.EMPTY);
1280        }
1281
1282    //////// Update complex fields and corresponding index entries
1283
1284        // Get complex field storage IDs (old or new)
1285        final TreeSet<Integer> complexFieldStorageIds = new TreeSet<>();
1286        complexFieldStorageIds.addAll(oldType.complexFields.keySet());
1287        complexFieldStorageIds.addAll(newType.complexFields.keySet());
1288
1289        // Notes:
1290        //
1291        // - The only changes we support are sub-field changes that don't affect the corresponding StorageInfo's
1292        // - New complex fields do not need to be explicitly initialized because their initial state is to have zero KV pairs
1293        //
1294        for (int storageId : complexFieldStorageIds) {
1295
1296            // Get old and new complex fields having this storage ID
1297            final ComplexField<?> oldField = oldType.complexFields.get(storageId);
1298            final ComplexField<?> newField = newType.complexFields.get(storageId);
1299
1300            // If there is no old field, new field and any associated indexes are already initialized (i.e., they're empty)
1301            if (oldField == null)
1302                continue;
1303
1304            // Save old field's value
1305            if (oldField != null && oldValueMap != null)
1306                oldValueMap.put(storageId, oldField.getValueReadOnlyCopy(this, id));
1307
1308            // If field is being removed, delete old field content, otherwise check if index entries should be added/removed
1309            if (newField == null) {
1310                oldField.removeIndexEntries(this, id);
1311                oldField.deleteContent(this, id);
1312            } else
1313                newField.updateSubFieldIndexes(this, oldField, id);
1314        }
1315
1316    //////// Remove references that are no longer valid
1317
1318        // Unreference any reference fields that refer to no-longer-valid object types
1319        for (int storageId : simpleFieldStorageIds) {
1320
1321            // Get storage IDs that are no longer allowed, if any
1322            final TreeSet<Integer> removedObjectTypes = this.findRemovedObjectTypes(info, targetVersion, storageId);
1323            if (removedObjectTypes == null)
1324                continue;
1325
1326            // Unreference field if needed
1327            final ReferenceField oldField = (ReferenceField)oldType.simpleFields.get(storageId);
1328            final ObjId ref = oldField.getValue(this, id);
1329            if (ref != null && removedObjectTypes.contains(ref.getStorageId()))
1330                this.writeSimpleField(id, oldField.storageId, null, false);
1331        }
1332
1333        // Unreference any reference sub-fields of complex fields that refer to no-longer-valid object types
1334        for (int storageId : complexFieldStorageIds) {
1335
1336            // Get old and new complex fields having this storage ID
1337            final ComplexField<?> oldField = oldType.complexFields.get(storageId);
1338            final ComplexField<?> newField = newType.complexFields.get(storageId);
1339            if (oldField == null || newField == null)
1340                continue;
1341            final List<? extends SimpleField<?>> oldSubFields = oldField.getSubFields();
1342            final List<? extends SimpleField<?>> newSubFields = newField.getSubFields();
1343            final int numSubFields = oldSubFields.size();
1344            assert numSubFields == newSubFields.size();
1345
1346            // Iterate over subfields
1347            for (int i = 0; i < numSubFields; i++) {
1348                final SimpleField<?> oldSubField = oldSubFields.get(i);
1349                final SimpleField<?> newSubField = newSubFields.get(i);
1350                assert oldSubField.storageId == newSubField.storageId;
1351                final int subStorageId = oldSubField.storageId;
1352
1353                // Get storage IDs that are no longer allowed, if any
1354                final TreeSet<Integer> removedObjectTypes = this.findRemovedObjectTypes(info, targetVersion, subStorageId);
1355                if (removedObjectTypes == null)
1356                    continue;
1357
1358                // Unreference sub-fields as needed
1359                oldField.unreferenceRemovedObjectTypes(this, id, (ReferenceField)oldSubField, removedObjectTypes);
1360            }
1361        }
1362
1363    //////// Update object version and corresponding index entry
1364
1365        // Update object version
1366        ObjInfo.write(this, id, newVersion, info.isDeleteNotified());
1367
1368        // Update object version index entry
1369        this.kvt.remove(Database.buildVersionIndexKey(id, oldVersion));
1370        this.kvt.put(Database.buildVersionIndexKey(id, newVersion), ByteUtil.EMPTY);
1371
1372    //////// Notify listeners
1373
1374        // Lock down old field values map
1375        final NavigableMap<Integer, Object> readOnlyOldValuesMap = oldValueMap != null ?
1376          Maps.unmodifiableNavigableMap(oldValueMap) : null;
1377
1378        // Notify about version update
1379        for (VersionChangeListener listener : this.versionChangeListeners)
1380            listener.onVersionChange(this, id, oldVersion, newVersion, readOnlyOldValuesMap);
1381    }

1383    /**
1384     * Find storage ID's which are no longer allowed by a reference field when upgrading to the specified
1385     * schema version and therefore need to be scrubbed during the upgrade.
1386     */
1387    private TreeSet<Integer> findRemovedObjectTypes(ObjInfo info, Schema newVersion, int storageId) {
1388
1389        // Get old schema version
1390        final Schema oldVersion = info.getSchema();
1391
1392        // Get old and new object types
1393        final ObjType oldType = info.getObjType();
1394        final ObjType newType = newVersion.getObjType(info.getId().getStorageId());
1395
1396        // Get old and new reference fields having specified storage ID
1397        final ReferenceField oldField = oldType.referenceFields.get(storageId);
1398        final ReferenceField newField = newType.referenceFields.get(storageId);
1399        if (oldField == null || newField == null)
1400            return null;
1401
1402        // Check allowed storage IDs
1403        final SortedSet<Integer> newObjectTypes = newField.getObjectTypes();
1404        if (newObjectTypes == null)
1405            return null;                                                // new field can refer to any type in any schema version
1406        SortedSet<Integer> oldObjectTypes = oldField.getObjectTypes();
1407        if (oldObjectTypes == null)
1408            oldObjectTypes = this.schemas.objTypeStorageIds;            // old field can refer to any type in any schema version
1409
1410        // Identify storage IDs which are were allowed by old field but are no longer allowed by new field
1411        final TreeSet<Integer> removedObjectTypes = new TreeSet<>(oldObjectTypes);
1412        removedObjectTypes.removeAll(newObjectTypes);
1413        return !removedObjectTypes.isEmpty() ? removedObjectTypes : null;
1414    }

1590    private synchronized void doWriteSimpleField(ObjId id, int storageId, final Object newObj, boolean updateVersion) {
1591
1592        // Get object info
1593        final ObjInfo info = this.getObjectInfo(id, updateVersion);
1594
1595        // Find field
1596        final SimpleField<?> field = info.getObjType().simpleFields.get(storageId);
1597        if (field == null)
1598            throw new UnknownFieldException(info.getObjType(), storageId, "simple field");
1599
1600        // Get new value
1601        final byte[] key = field.buildKey(id);
1602        final byte[] newValue = field.encode(newObj);
1603
1604        // Before setting the new value, read the old value if one of the following is true:
1605        //  - The field is being monitored -> we need to filter out "changes" that don't actuallly change anything
1606        //  - The field is indexed -> we need the old value so we can remove the old index entry
1607        // If neither of the above is true, then there's no need to read the old value.
1608        byte[] oldValue = null;
1609        if (field.indexed || field.compositeIndexMap != null || this.hasFieldMonitor(id, field)) {
1610
1611            // Get old value
1612            oldValue = this.kvt.get(key);
1613
1614            // Compare new to old value
1615            if (oldValue != null ? newValue != null && Arrays.equals(oldValue, newValue) : newValue == null)
1616                return;
1617        }
1618
1619        // Update value
1620        if (newValue != null)
1621            this.kvt.put(key, newValue);
1622        else
1623            this.kvt.remove(key);
1624
1625        // Update simple index, if any
1626        if (field.indexed) {
1627            this.kvt.remove(Transaction.buildSimpleIndexEntry(field, id, oldValue));
1628            this.kvt.put(Transaction.buildSimpleIndexEntry(field, id, newValue), ByteUtil.EMPTY);
1629        }
1630
1631        // Update affected composite indexes, if any
1632        if (field.compositeIndexMap != null) {
1633            for (Map.Entry<CompositeIndex, Integer> entry : field.compositeIndexMap.entrySet()) {
1634                final CompositeIndex index = entry.getKey();
1635                final int fieldIndexOffset = entry.getValue();
1636
1637                // Build old composite index entry
1638                final ByteWriter oldWriter = new ByteWriter();
1639                UnsignedIntEncoder.write(oldWriter, index.storageId);
1640                int fieldStart = -1;
1641                int fieldEnd = -1;
1642                for (SimpleField<?> otherField : index.fields) {
1643                    final byte[] otherValue;
1644                    if (otherField == field) {
1645                        fieldStart = oldWriter.getLength();
1646                        otherValue = oldValue;
1647                    } else
1648                        otherValue = this.kvt.get(otherField.buildKey(id));         // can be null (if field has default value)
1649                    oldWriter.write(otherValue != null ? otherValue : otherField.fieldType.getDefaultValue());
1650                    if (otherField == field)
1651                        fieldEnd = oldWriter.getLength();
1652                }
1653                assert fieldStart != -1;
1654                assert fieldEnd != -1;
1655                id.writeTo(oldWriter);
1656
1657                // Remove old composite index entry
1658                final byte[] oldIndexEntry = oldWriter.getBytes();
1659                this.kvt.remove(oldIndexEntry);
1660
1661                // Patch in new field value to create new composite index entry
1662                final ByteWriter newWriter = new ByteWriter(oldIndexEntry.length);
1663                newWriter.write(oldIndexEntry, 0, fieldStart);
1664                newWriter.write(newValue != null ? newValue : field.fieldType.getDefaultValue());
1665                newWriter.write(oldIndexEntry, fieldEnd, oldIndexEntry.length - fieldEnd);
1666
1667                // Add new composite index entry
1668                this.kvt.put(newWriter.getBytes(), ByteUtil.EMPTY);
1669            }
1670        }
1671
1672        // Notify monitors
1673        final Object oldObj = field.fieldType.read(new ByteReader(oldValue != null ? oldValue : field.fieldType.getDefaultValue()));
1674        this.addFieldChangeNotification(new SimpleFieldChangeNotifier(field, id) {
1675            @Override
1676            @SuppressWarnings("unchecked")
1677            public void notify(Transaction tx, SimpleFieldChangeListener listener, int[] path, NavigableSet<ObjId> referrers) {
1678                listener.onSimpleFieldChange(tx, this.id, (SimpleField<Object>)field, path, referrers, oldObj, newObj);
1679            }
1680        });
1681    }

2037    /**
2038     * Monitor for changes within this transaction of the value of the given field, as seen through a path of references.
2039     *
2040     * <p>
2041     * When the specified field is changed, a listener notification will be delivered for each referring object that
2042     * refers to the object containing the changed field through the specified path of references. Notifications are
2043     * delivered at the end of the mutation operation just prior to returning to the caller. If the listener method
2044     * performs additional mutation(s) which are themselves being listened for, those notifications will also be delivered
2045     * prior to the returning to the original caller. Therefore, care must be taken to avoid change notification dependency
2046     * loops when listeners can themselves mutate fields, to avoid infinite loops.
2047     * </p>
2048     *
2049     * <p>
2050     * If a non-null {@code types} is provided, then only objects whose types have one of the specified storage ID's
2051     * will trigger notifications to {@code listener}.
2052     * </p>
2053     *
2054     * <p>
2055     * A referring object may refer to the changed object through more than one actual path of references matching {@code path};
2056     * if so, it will still appear only once in the {@link NavigableSet} provided to the listener (this is of course required
2057     * by set semantics).
2058     * </p>
2059     *
2060     * <p>
2061     * Although the reference back-tracking algorithm does consolidate multiple paths between the same two objects,
2062     * be careful to avoid an explosion of notifications when objects in the {@code path} have a high degree of fan-in.
2063     * </p>
2064     *
2065     * <p>
2066     * When a {@link ReferenceField} in {@code path} also happens to be the field being changed, then there is ambiguity
2067     * about how the set of referring objects is calculated: does it use the field's value before or after the change?
2068     * This API guarantees that the answer is "after the change"; however, if another listener on the same field is
2069     * invoked before {@code listener} and mutates any reference field(s) in {@code path}, then whether that additional
2070     * change is also be included in the calculation is undefined.
2071     * </p>
2072     *
2073     * <p>
2074     * Therefore, for consistency, avoid changing any {@link ReferenceField} from within a listener callback when that
2075     * field is also in some other listener's reference path, and both listeners are watching the same field.
2076     * </p>
2077     *
2078     * @param storageId storage ID of the field to monitor
2079     * @param path path of reference fields (represented by storage IDs) through which to monitor field
2080     * @param types set of allowed storage IDs for the changed object, or null for no restriction
2081     * @param listener callback for notifications on changes in value
2082     * @throws UnknownFieldException if no {@link SimpleField} corresponding to {@code storageId} exists
2083     * @throws UnknownFieldException if {@code path} contains a storage ID that does not correspond to a {@link ReferenceField}
2084     * @throws IllegalArgumentException if {@code storageId} refers to a sub-field of a complex field
2085     * @throws IllegalArgumentException if {@code path} or {@code listener} is null
2086     * @throws StaleTransactionException if this transaction is no longer usable
2087     */

2089      SimpleFieldChangeListener listener) {
2090        this.validateChangeListener(SimpleField.class, storageId, path, listener);
2091        this.getMonitorsForField(storageId, true).add(new FieldMonitor(storageId, path, types, listener));

2339    private synchronized <V> V mutateAndNotify(Mutation<V> mutation) {
2340
2341        // Validate transaction
2342        if (this.stale)
2343            throw new StaleTransactionException(this);
2344        if (this.readOnly)
2345            throw new ReadOnlyTransactionException(this);
2346
2347        // If re-entrant invocation, we're already set up
2348        if (this.pendingNotifications.get() != null)
2349            return mutation.mutate();
2350
2351        // Set up pending report list, perform mutation, and then issue reports
2352        this.pendingNotifications.set(new TreeMap<Integer, ArrayList<FieldChangeNotifier>>());
2353        try {
2354            return mutation.mutate();
2355        } finally {
2356            try {
2357                final TreeMap<Integer, ArrayList<FieldChangeNotifier>> pendingNotificationMap = this.pendingNotifications.get();
2358                while (!pendingNotificationMap.isEmpty()) {
2359
2360                    // Get the next field with pending notifications
2361                    final Map.Entry<Integer, ArrayList<FieldChangeNotifier>> entry = pendingNotificationMap.pollFirstEntry();
2362                    final int storageId = entry.getKey();
2363
2364                    // For all pending notifications, back-track references and notify all field monitors for the field
2365                    for (FieldChangeNotifier notifier : entry.getValue()) {
2366                        assert notifier.getStorageId() == storageId;
2367                        final ArrayList<FieldMonitor> monitorList = new ArrayList<>(this.getMonitorsForField(storageId, false));
2368                        if (!monitorList.isEmpty())
2369                            this.notifyFieldMonitors(notifier, NavigableSets.singleton(notifier.getId()), monitorList, 0);
2370                    }
2371                }
2372            } finally {
2373                this.pendingNotifications.remove();
2374            }
2375        }
2376    }

2378    // Recursively back-track references along monitor paths and notify monitors when we reach the end (i.e., beginning)
2379    private void notifyFieldMonitors(FieldChangeNotifier notifier,
2380      NavigableSet<ObjId> objects, ArrayList<FieldMonitor> monitorList, int step) {
2381
2382        // Find the monitors for whom we have completed all the steps in their (inverse) path,
2383        // and group the remaining monitors by their next inverted reference path step.
2384        final HashMap<Integer, ArrayList<FieldMonitor>> remainingMonitorsMap = new HashMap<>();
2385        for (FieldMonitor monitor : monitorList) {
2386
2387            // On the first step, apply the monitor's type filter, if any
2388            if (step == 0 && monitor.types != null && !monitor.types.contains(notifier.getId().getBytes()))
2389                continue;
2390
2391            // Issue notification callback if we have back-tracked through the whole path
2392            if (monitor.path.length == step) {
2393                notifier.notify(this, monitor.listener, monitor.path, objects);
2394                continue;
2395            }
2396
2397            // Sort the unfinished monitors by their next (i.e., previous) reference field
2398            final int storageId = monitor.path[monitor.path.length - step - 1];
2399            ArrayList<FieldMonitor> remainingMonitors = remainingMonitorsMap.get(storageId);
2400            if (remainingMonitors == null) {
2401                remainingMonitors = new ArrayList<FieldMonitor>();
2402                remainingMonitorsMap.put(storageId, remainingMonitors);
2403            }
2404            remainingMonitors.add(monitor);
2405        }
2406
2407        // Invert references for each group of remaining monitors and recurse
2408        for (Map.Entry<Integer, ArrayList<FieldMonitor>> entry : remainingMonitorsMap.entrySet()) {
2409            final int storageId = entry.getKey();
2410
2411            // Gather all objects that refer to any object in our current "objects" set
2412            final ArrayList<NavigableSet<ObjId>> refsList = new ArrayList<>();
2413            for (ObjId object : objects) {
2414                final NavigableSet<ObjId> refs = this.queryReferences(storageId).get(object);
2415                if (refs != null)
2416                    refsList.add(refs);
2417            }
2418
2419            // Recurse on the union of those objects
2420            if (!refsList.isEmpty())
2421                this.notifyFieldMonitors(notifier, NavigableSets.union(refsList), entry.getValue(), step + 1);
2422        }
2423    }

2427    /**
2428     * Find all objects that refer to any object in the given target set through the specified path of references.
2429     *
2430     * @param path path of one or more reference fields (represented by storage IDs) through which to reach the target objects
2431     * @param targetObjects target objects
2432     * @return set of objects that refer to the {@code targetObjects} via {@code path}
2433     * @throws UnknownFieldException if {@code path} contains a storage ID that does not correspond to a {@link ReferenceField}
2434     * @throws IllegalArgumentException if {@code targetObjects} or {@code path} is null
2435     * @throws IllegalArgumentException if {@code path} is empty
2436     * @throws StaleTransactionException if this transaction is no longer usable
2437     */
2438    public NavigableSet<ObjId> invertReferencePath(int[] path, Iterable<ObjId> targetObjects) {
2439
2440        // Sanity check
2441        if (targetObjects == null)
2442            throw new IllegalArgumentException("null targetObjects");
2443        if (path == null)
2444            throw new IllegalArgumentException("null path");
2445        if (path.length == 0)
2446            throw new IllegalArgumentException("empty path");
2447
2448        // Verify all fields in the path are reference fields
2449        for (int storageId : path)
2450            this.schemas.verifyStorageInfo(storageId, ReferenceFieldStorageInfo.class);
2451
2452        // Invert references in reverse order
2453        NavigableSet<ObjId> result = null;
2454        for (int i = path.length - 1; i >= 0; i--) {
2455            final int storageId = path[i];
2456
2457            // Gather all objects that refer to any object in our current target objects set
2458            final ArrayList<NavigableSet<ObjId>> refsList = new ArrayList<>();
2459            for (ObjId id : targetObjects) {
2460                final NavigableSet<ObjId> refs = this.queryReferences(storageId).get(id);
2461                if (refs != null)
2462                    refsList.add(refs);
2463            }
2464            if (refsList.isEmpty())
2465                return NavigableSets.empty();
2466
2467            // Recurse on the union of those objects
2468            targetObjects = result = NavigableSets.union(refsList);
2469        }
2470
2471        // Done
2472        return result;
2473    }

2647    /**
2648     * Find all objects that refer to the given target object through the/any reference field with the specified
2649     * {@link DeleteAction}.
2650     *
2651     * <p>
2652     * Because different schema versions can have different {@link DeleteAction}'s configured for the
2653     * same field, we have to iterate through each schema version separately.
2654     * </p>
2655     *
2656     * @param target referred-to object
2657     * @param onDelete {@link DeleteAction} to match
2658     * @param fieldStorageId reference field storage ID, or -1 to match any reference field
2659     */
2660    private NavigableSet<ObjId> findReferrers(ObjId target, DeleteAction onDelete, int fieldStorageId) {
2661        final ArrayList<NavigableSet<ObjId>> refSets = new ArrayList<>();
2662        for (Schema schemaVersion : this.schemas.versions.values()) {
2663
2664            // Check whether any object of this version exist; if not, skip
2665            final NavigableSet<ObjId> versionObjects = this.queryVersion().asMap().get(schemaVersion.versionNumber);
2666            if (versionObjects == null)
2667                continue;
2668
2669            // Find all reference fields with storage ID matching fieldStorageId (if not -1) and check them.
2670            // Do this separately for each such field in each object type because the fields may have different DeleteAction's.
2671            for (ObjType objType : schemaVersion.objTypeMap.values()) {
2672                for (ReferenceField field : Iterables.filter(objType.getFieldsAndSubFields(), ReferenceField.class)) {
2673
2674                    // Check delete action and field
2675                    if (field.onDelete != onDelete || (fieldStorageId != -1 && field.storageId != fieldStorageId))
2676                        continue;
2677
2678                    // Query index on this field, restricting to those only references coming from objType objects
2679                    final NavigableSet<ObjId> refs = this.queryIndex(field.storageId)
2680                      .filter(1, new KeyRanges(ObjId.getKeyRange(objType.storageId))).asMap().get(target);
2681                    if (refs == null)
2682                        continue;
2683
2684                    // Restrict further to the specific schema version
2685                    refSets.add(NavigableSets.intersection(versionObjects, refs));
2686                }
2687            }
2688        }
2689        return !refSets.isEmpty() ? NavigableSets.union(refSets) : NavigableSets.empty(FieldTypeRegistry.OBJ_ID);
2690    }
