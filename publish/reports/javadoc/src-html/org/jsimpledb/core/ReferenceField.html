006package org.jsimpledb.core;
007
008import java.util.Set;
009import java.util.SortedSet;
010
011/**
012 * A field that references another {@link Database} object.
013 *
014 * <p>
015 * Null values sort last.
016 * </p>
017 *
018 * <p>
019 * Reference fields are always indexed.
020 * </p>
021 */
022public class ReferenceField extends SimpleField<ObjId> {
023
024    final DeleteAction onDelete;
025    final boolean cascadeDelete;
026
027    /**
028     * Constructor.
029     *
030     * @param name the name of the field
031     * @param storageId field storage ID
032     * @param schema schema version
033     * @param onDelete deletion behavior
034     * @param cascadeDelete whether to cascade deletes
035     * @param objectTypes allowed object type storage IDs, or null for no restriction
036     * @throws IllegalArgumentException if any parameter is null
037     * @throws IllegalArgumentException if {@code name} is invalid
038     * @throws IllegalArgumentException if {@code storageId} is invalid
039     */
040    ReferenceField(String name, int storageId, Schema schema,
041      DeleteAction onDelete, boolean cascadeDelete, Set<Integer> objectTypes) {
042        super(name, storageId, schema, new ReferenceFieldType(objectTypes), true);
043        if (onDelete == null)
044            throw new IllegalArgumentException("null onDelete");
045        this.onDelete = onDelete;
046        this.cascadeDelete = cascadeDelete;
047    }
048
049// Public methods
050
051    /**
052     * Get the desired behavior when an object referred to by this field is deleted.
053     *
054     * @return desired behavior when a referenced object is deleted
055     */
056    public DeleteAction getOnDelete() {
057        return this.onDelete;
058    }
059
060    /**
061     * Determine whether the referred-to object should be deleted when an object containing this field is deleted.
062     *
063     * @return whether deletion should cascade to the referred-to object
064     */
065    public boolean isCascadeDelete() {
066        return this.cascadeDelete;
067    }
068
069    /**
070     * Get the object types this field is allowed to reference, if so restricted.
071     *
072     * @return storage IDs of allowed object types, or null if there is no restriction
073     */
074    public SortedSet<Integer> getObjectTypes() {
075        return ((ReferenceFieldType)this.getFieldType()).getObjectTypes();
076    }
077
078    @Override
079    public <R> R visit(FieldSwitch<R> target) {
080        return target.caseReferenceField(this);
081    }
082
083    @Override
084    public String toString() {
085        return "reference field `" + this.name + "'";
086    }
087
088// Non-public methods
089
090    @Override
091    ReferenceFieldStorageInfo toStorageInfo() {
092        return new ReferenceFieldStorageInfo(this, this.parent != null ? this.parent.storageId : 0);
093    }
094}
