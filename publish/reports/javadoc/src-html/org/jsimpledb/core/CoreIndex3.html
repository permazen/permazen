008package org.jsimpledb.core;
009
010import java.util.NavigableSet;
011
012import org.jsimpledb.index.Index;
013import org.jsimpledb.index.Index2;
014import org.jsimpledb.index.Index3;
015import org.jsimpledb.kv.KeyFilter;
016import org.jsimpledb.tuple.Tuple2;
017import org.jsimpledb.tuple.Tuple3;
018import org.jsimpledb.tuple.Tuple4;
019
020/**
021 * Core API {@link Index} implementation representing a composite index on three fields.
022 *
023 * <p>
024 * Instances are immutable.
025 * </p>
026 *
027 * @param <V1> first index value type
028 * @param <V2> second index value type
029 * @param <V3> third index value type
030 * @param <T> index target type
031 */
032public class CoreIndex3<V1, V2, V3, T> extends AbstractCoreIndex implements Index3<V1, V2, V3, T> {
033
034// Constructors
035
036    CoreIndex3(Transaction tx, Index3View<V1, V2, V3, T> indexView) {
037        super(tx, 4, indexView);
038    }
039
040// Methods
041
042    @Override
043    public CoreIndex3<V1, V2, V3, T> filter(int index, KeyFilter filter) {
044        return new CoreIndex3<V1, V2, V3, T>(this.tx, this.getIndex3View().filter(index, filter));
045    }
046
047    @SuppressWarnings("unchecked")
048    Index3View<V1, V2, V3, T> getIndex3View() {
049        return (Index3View<V1, V2, V3, T>)this.indexView;
050    }
051
052// Index3
053
054    @Override
055    public IndexSet<Tuple4<V1, V2, V3, T>> asSet() {
056
057        // Get index view
058        final Index3View<V1, V2, V3, T> iv = this.getIndex3View();
059
060        // Create field type for Tuple4<V1, V2, V3, T>
061        final Tuple4FieldType<V1, V2, V3, T> fieldType = new Tuple4FieldType<V1, V2, V3, T>(
062          iv.getValue1Type(), iv.getValue2Type(), iv.getValue3Type(), iv.getTargetType());
063
064        // Build set and apply filtering
065        IndexSet<Tuple4<V1, V2, V3, T>> indexSet = new IndexSet<Tuple4<V1, V2, V3, T>>(this.tx,
066          fieldType, iv.prefixMode, iv.prefix);
067        if (iv.hasFilters())
068            indexSet = indexSet.filterKeys(new IndexKeyFilter(this.tx, iv, 4));
069
070        // Done
071        return indexSet;
072    }
073
074    @Override
075    public IndexMap<Tuple3<V1, V2, V3>, NavigableSet<T>> asMap() {
076
077        // Get index view
078        final Index3View<V1, V2, V3, T> iv = this.getIndex3View();
079
080        // Create new IndexView
081        final IndexView<Tuple3<V1, V2, V3>, T> tupleIV = iv.asTuple3IndexView();
082
083        // Build map and apply filtering
084        IndexMap<Tuple3<V1, V2, V3>, NavigableSet<T>> indexMap = new IndexMap.OfValues<Tuple3<V1, V2, V3>, T>(this.tx, tupleIV);
085        if (tupleIV.hasFilters())
086            indexMap = indexMap.filterKeys(new IndexKeyFilter(this.tx, tupleIV, 1));
087
088        // Done
089        return indexMap;
090    }
091
092    @Override
093    public IndexMap<Tuple2<V1, V2>, Index<V3, T>> asMapOfIndex() {
094
095        // Get index view
096        final Index3View<V1, V2, V3, T> iv = this.getIndex3View();
097
098        // Create new IndexView
099        final Index2View<Tuple2<V1, V2>, V3, T> tupleIV = iv.asTuple2Index2View();
100
101        // Build map and apply filtering
102        IndexMap<Tuple2<V1, V2>, Index<V3, T>> indexMap = new IndexMap.OfIndex<Tuple2<V1, V2>, V3, T>(this.tx, tupleIV);
103        if (iv.hasFilters())
104            indexMap = indexMap.filterKeys(new IndexKeyFilter(this.tx, tupleIV, 1));
105
106        // Done
107        return indexMap;
108    }
109
110    @Override
111    public IndexMap<V1, Index2<V2, V3, T>> asMapOfIndex2() {
112
113        // Get index view
114        final Index3View<V1, V2, V3, T> iv = this.getIndex3View();
115
116        // Build map and apply filtering
117        IndexMap<V1, Index2<V2, V3, T>> indexMap = new IndexMap.OfIndex2<V1, V2, V3, T>(this.tx, iv);
118        if (iv.hasFilters())
119            indexMap = indexMap.filterKeys(new IndexKeyFilter(this.tx, iv, 1));
120
121        // Done
122        return indexMap;
123    }
124
125    @Override
126    public CoreIndex2<V1, V2, V3> asIndex2() {
127        return new CoreIndex2<V1, V2, V3>(this.tx, this.getIndex3View().asIndex2View());
128    }
129
130    @Override
131    public CoreIndex<V1, V2> asIndex() {
132        return new CoreIndex<V1, V2>(this.tx, this.getIndex3View().asIndex2View().asIndexView());
133    }
134}
