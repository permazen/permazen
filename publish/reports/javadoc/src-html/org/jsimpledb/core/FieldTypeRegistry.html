027/**
028 * A registry of {@link FieldType}s.
029 *
030 * <p>
031 * All {@link FieldType}s in a {@link FieldTypeRegistry} are registered under a unique type name.
032 * However, multiple registered {@link FieldType}s may support the same {@linkplain FieldType#getTypeToken Java type}.
033 * </p>
034 *
035 * <p><b>Arrays</b></p>
036 *
037 * <p>
038 * Array types will be automatically created and registered on demand, assuming the base element
039 * type is already registered.
040 * </p>
041 *
042 * <p>
043 * Arrays are passed by value: i.e., the entire array is copied. Therefore, changes to array elements after
044 * getting or setting a field have no effect on the field's value.
045 * </p>
046 *
047 * <p>
048 * Arrays of references are not currently supported.
049 * </p>
050 *
051 * <p>
052 * The following types are automatically registered (or generated on demand):
053 * <ul>
054 *  <li>Primitive types ({@code boolean}, {@code int}, etc.)</li>
055 *  <li>Primitive wrapper types ({@link Boolean}, {@link Integer}, etc.)</li>
056 *  <li>{@link String}</li>
057 *  <li>Array types</li>
058 *  <li>{@link Enum} types</li>
059 *  <li>{@link java.util.Date}</li>
060 *  <li>{@link java.util.UUID}</li>
061 *  <li>{@link java.net.URI}</li>
062 *  <li>{@link java.io.File}</li>
063 *  <li>{@link java.util.regex.Pattern}</li>
064 * </ul>
065 */
066public class FieldTypeRegistry {

232    /**
233     * Add multiple user-defined {@link FieldType} to this registry, using newly created instances of the named classes.
234     *
235     * @param classNames names of classes that implement {@link FieldType}
236     * @throws IllegalArgumentException if {@code classNames} is null
237     * @throws IllegalArgumentException if {@code classNames} contains a null class or a class with invalid annotation(s)
238     * @throws IllegalArgumentException if {@code classNames} contains an invalid {@link FieldType} class
239     * @throws RuntimeException if instantiation of a class fails
240     */
241    public void addNamedClasses(Iterable<String> classNames) {
242        if (classNames == null)
243            throw new IllegalArgumentException("null classNames");
244        this.addClasses(Iterables.transform(classNames, new Function<String, Class<? extends FieldType<?>>>() {
245            @Override
246            @SuppressWarnings("unchecked")
247            public Class<? extends FieldType<?>> apply(String name) {
248                final Class<?> c;
249                try {
250                    c = Class.forName(name, false, Thread.currentThread().getContextClassLoader());
251                } catch (RuntimeException e) {
252                    throw e;
253                } catch (Exception e) {
254                    throw new RuntimeException(e);
255                }
256                try {
257                    return (Class<? extends FieldType<?>>)(Object)c.asSubclass(FieldType.class);
258                } catch (ClassCastException e) {
259                    throw new IllegalArgumentException("class `" + name + "' does not implement " + FieldType.class, e);
260                }
261            }
262        }));
263    }
264
265    /**
266     * Add multiple user-defined {@link FieldType} to this registry, using newly created instances of the specified classes.
267     *
268     * @param classes implementations of types to add
269     * @throws IllegalArgumentException if {@code classes} is null
270     * @throws IllegalArgumentException if {@code classes} contains a null class or a class with invalid annotation(s)
271     * @throws IllegalArgumentException if {@code classes} contains an invalid {@link FieldType} class
272     */
273    public void addClasses(Iterable<? extends Class<? extends FieldType<?>>> classes) {
274        if (classes == null)
275            throw new IllegalArgumentException("null classes");
276        for (Class<? extends FieldType<?>> type : classes)
277            this.addClass(type);
278    }

280    /**
281     * Add a user-defined {@link FieldType} to this registry, using a newly created instance of the specified class.
282     *
283     * @param typeClass implementation of type to add
284     * @throws IllegalArgumentException if {@code typeClass} is null
285     * @throws IllegalArgumentException if {@code typeClass} cannot be instantiated
286     * @throws IllegalArgumentException if the {@linkplain FieldType#getName type name} conflicts with an existing type
287     * @throws IllegalArgumentException if the {@linkplain FieldType#getName type name} ends with {@code []} (array name)
288     */
289    public void addClass(Class<? extends FieldType<?>> typeClass) {
290
291        // Sanity check
292        if (typeClass == null)
293            throw new IllegalArgumentException("null typeClass");
294
295        // Instantiate class
296        final FieldType<?> fieldType;
297        try {
298            fieldType = (FieldType<?>)typeClass.newInstance();
299        } catch (Exception e) {
300            throw new IllegalArgumentException("can't instantiate " + typeClass, e);
301        }
302
303        // Register it
304        this.add(fieldType);
305    }
306
307    /**
308     * Add a user-defined {@link FieldType} to the registry.
309     *
310     * @param type type to add
311     * @return true if type was added, false if type was already registered
312     * @throws IllegalArgumentException if {@code type} is null
313     * @throws IllegalArgumentException if {@code type} has a name that conflicts with an existing, but different, type
314     * @throws IllegalArgumentException if {@code type} has a name that ends with {@code []} (array name)
315     */
316    public synchronized boolean add(FieldType<?> type) {
317        if (type == null)
318            throw new IllegalArgumentException("null type");
319        if (type.name.endsWith(ArrayType.ARRAY_SUFFIX))
320            throw new IllegalArgumentException("illegal array type name `" + type.name + "'");
321        final FieldType<?> other = this.typesByName.get(type.name);
322        if (other != null) {
323            if (other.equals(type))
324                return false;
325            throw new IllegalArgumentException("type name `" + type.name + "' conflicts with existing type " + other);
326        }
327        this.typesByName.put(type.name, type);
328        final TypeToken<?> typeToken = type.getTypeToken();
329        if (!this.typesByType.containsKey(typeToken))
330            this.typesByType.put(typeToken, Lists.newArrayList(Collections.<FieldType<?>>singleton(type)));
331        else
332            this.typesByType.get(typeToken).add(type);
333        return true;
334    }

336    /**
337     * Get the {@link FieldType} with the given name in this registry.
338     *
339     * @param name type name
340     * @return type with name {@code name}, or null if not found
341     * @throws IllegalArgumentException if {@code name} is null
342     */
343    public synchronized FieldType<?> getFieldType(final String name) {
344
345        // Sanity check
346        if (name == null)
347            throw new IllegalArgumentException("null name");
348
349        // Handle array types
350        if (name.endsWith(ArrayType.ARRAY_SUFFIX)) {
351            final FieldType<?> elementType = this.getFieldType(name.substring(0, name.length() - ArrayType.ARRAY_SUFFIX.length()));
352            return this.getArrayType(elementType);
353        }
354
355        // Handle non-array type
356        return this.typesByName.get(name);
357    }

359    /**
360     * Get the array {@link FieldType} with the given element type.
361     *
362     * @param elementType array element type
363     * @param <E> array element type
364     * @throws IllegalArgumentException if {@code elementType} is null
365     * @throws IllegalArgumentException if the resulting array type has too many dimensions
366     * @return array type with element type {@code elementType}
367     */
368    @SuppressWarnings("unchecked")
369    public <E> FieldType<E[]> getArrayType(final FieldType<E> elementType) {
370
371        // Sanity check
372        if (elementType == null)
373            throw new IllegalArgumentException("null elementType");
374
375        // Create array type
376        final Primitive<?> primitive = Primitive.get(elementType.typeToken.getRawType());
377        final ArrayType<?, ?> notNullType = primitive == null ? this.createObjectArrayType(elementType) :
378          primitive.visit(new PrimitiveSwitch<ArrayType<?, ?>>() {
379            @Override
380            public ArrayType<?, ?> caseVoid() {
381                throw new IllegalArgumentException("cannot create array of type `" + elementType.name + "'");
382            }
383            @Override
384            public ArrayType<?, ?> caseBoolean() {
385                return new BooleanArrayType();
386            }
387            @Override
388            public ArrayType<?, ?> caseByte() {
389                return new ByteArrayType();
390            }
391            @Override
392            public ArrayType<?, ?> caseCharacter() {
393                return new CharacterArrayType();
394            }
395            @Override
396            public ArrayType<?, ?> caseShort() {
397                return new ShortArrayType();
398            }
399            @Override
400            public ArrayType<?, ?> caseInteger() {
401                return new IntegerArrayType();
402            }
403            @Override
404            public ArrayType<?, ?> caseFloat() {
405                return new FloatArrayType();
406            }
407            @Override
408            public ArrayType<?, ?> caseLong() {
409                return new LongArrayType();
410            }
411            @Override
412            public ArrayType<?, ?> caseDouble() {
413                return new DoubleArrayType();
414            }
415        });
416        return (FieldType<E[]>)this.createNullSafeType(notNullType);
417    }

419    /**
420     * Get all {@link FieldType}s in this registry that supports values of the given Java type, which must
421     * exactly match the {@link FieldType}'s {@linkplain FieldType#getTypeToken supported Java type}.
422     *
423     * @param typeToken field value type
424     * @param <T> field value type
425     * @return unmodifiable list of {@link FieldType}s supporting Java values of type {@code typeToken}
426     * @throws IllegalArgumentException if {@code typeToken} is null
427     */
428    @SuppressWarnings("unchecked")
429    public synchronized <T> List<FieldType<T>> getFieldTypes(TypeToken<T> typeToken) {
430
431        // Sanity check
432        if (typeToken == null)
433            throw new IllegalArgumentException("null typeToken");
434
435        // Handle array types
436        final TypeToken<?> elementTypeToken = typeToken.getComponentType();
437        if (elementTypeToken != null) {
438            final ArrayList<FieldType<T>> fieldTypes = new ArrayList<FieldType<T>>();
439            for (FieldType<?> elementFieldType : this.getFieldTypes(elementTypeToken))
440                fieldTypes.add((FieldType<T>)this.getArrayType(this.getFieldType(elementTypeToken)));
441            return Collections.unmodifiableList(fieldTypes);
442        }
443
444        // Handle non-array types
445        final ArrayList<FieldType<T>> fieldTypes = (ArrayList<FieldType<T>>)(Object)this.typesByType.get(typeToken);
446        return fieldTypes != null ? Collections.unmodifiableList(fieldTypes) : Collections.<FieldType<T>>emptyList();
447    }
448
449    /**
450     * Get the unique {@link FieldType} in this registry that supports values of the given Java type, which must
451     * exactly match the {@link FieldType}'s {@linkplain FieldType#getTypeToken supported Java type}.
452     * There must be exactly one such {@link FieldType}, otherwise an {@link IllegalArgumentException} is thrown.
453     *
454     * @param typeToken field value type
455     * @param <T> field value type
456     * @return {@link FieldType} supporting Java values of type {@code typeToken}
457     * @throws IllegalArgumentException if {@code typeToken} is null
458     * @throws IllegalArgumentException if no registered {@link FieldType}s supports {@code typeToken}
459     * @throws IllegalArgumentException if more than one registered {@link FieldType} supports {@code typeToken}
460     */
461    public <T> FieldType<T> getFieldType(TypeToken<T> typeToken) {
462        final List<FieldType<T>> fieldTypes = this.getFieldTypes(typeToken);
463        switch (fieldTypes.size()) {
464        case 0:
465            throw new IllegalArgumentException("no registered types support values of type " + typeToken);
466        case 1:
467            return fieldTypes.get(0);
468        default:
469            throw new IllegalArgumentException("multiple registered types support values of type " + typeToken + ": " + fieldTypes);
470        }
471    }
