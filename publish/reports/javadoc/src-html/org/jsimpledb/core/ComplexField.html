006package org.jsimpledb.core;
007
008import com.google.common.reflect.TypeToken;
009
010import java.util.Iterator;
011import java.util.List;
012import java.util.SortedSet;
013
014import org.jsimpledb.kv.KVPair;
015import org.jsimpledb.util.ByteReader;
016import org.jsimpledb.util.ByteUtil;
017import org.jsimpledb.util.ByteWriter;
018import org.jsimpledb.util.UnsignedIntEncoder;
019
020/**
021 * A complex {@link Field}, such as a collection or map field.
022 *
023 * @param <T> Java type for the field's values
024 */
025public abstract class ComplexField<T> extends Field<T> {
026
027    private final int storageIdLength;
028
029    /**
030     * Constructor.
031     *
032     * @param name the name of the field
033     * @param storageId field content storage ID
034     * @param schema schema version
035     * @param typeToken Java type for the field's values
036     * @throws IllegalArgumentException if any parameter is null
037     * @throws IllegalArgumentException if {@code name} is invalid
038     * @throws IllegalArgumentException if {@code storageId} is non-positive
039     */
040    ComplexField(String name, int storageId, Schema schema, TypeToken<T> typeToken) {
041        super(name, storageId, schema, typeToken);
042        this.storageIdLength = UnsignedIntEncoder.encodeLength(storageId);
043    }
044
045// Public methods
046
047    /**
048     * Get the sub-field(s) associated with this instance, ordered according to their meaning.
049     *
050     * @return unmodifiable list of simple fields
051     */
052    public abstract List<? extends SimpleField<?>> getSubFields();
053
054// Non-public methods
055
056    /**
057     * Get the Java collection object representing the value of this instance in the given object.
058     * This method does not need to do any validity checking of its parameters.
059     */
060    abstract T getValueInternal(Transaction tx, ObjId id);
061
062    /**
063     * Copy the Java collection object representing the value of this instance into memory and return a read-only view.
064     * This method does not need to do any validity checking of its parameters.
065     */
066    abstract T getValueReadOnlyCopy(Transaction tx, ObjId id);
067
068    /**
069     * Iterate all values in the specified subfield.
070     *
071     * @throws IllegalArgumentException if {@code subField} is not a sub-field of this instance
072     */
073    abstract <F> Iterable<F> iterateSubField(Transaction tx, ObjId id, SimpleField<F> subField);
074
075    @Override
076    abstract ComplexFieldStorageInfo<T> toStorageInfo();
077
078    /**
079     * Delete all content (but not index entries) for the given object.
080     *
081     * @param tx transaction
082     * @param id object id
083     */
084    void deleteContent(Transaction tx, ObjId id) {
085        final byte[] minKey = this.buildKey(id);
086        final byte[] maxKey = ByteUtil.getKeyAfterPrefix(minKey);
087        this.deleteContent(tx, minKey, maxKey);
088    }
089
090    /**
091     * Delete all content (but not index entries) for the given object in the given key range.
092     *
093     * @param tx transaction
094     * @param id object id
095     * @see #removeIndexEntries(Transaction, ObjId)
096     */
097    void deleteContent(Transaction tx, byte[] minKey, byte[] maxKey) {
098        tx.kvt.removeRange(minKey, maxKey);
099    }
100
101    /**
102     * Add an index entry corresponding to the given sub-field and content key/value pair.
103     *
104     * @param tx transaction
105     * @param id object id
106     * @param subField indexed sub-field
107     * @param contentKey the content key
108     * @param contentValue the value associated with the content key, or null if not needed
109     */
110    void addIndexEntry(Transaction tx, ObjId id, SimpleField<?> subField, byte[] contentKey, byte[] contentValue) {
111        tx.kvt.put(this.buildIndexEntry(id, subField, contentKey, contentValue), ByteUtil.EMPTY);
112    }
113
114    /**
115     * Remove an index entry corresponding to the given sub-field and content key/value pair.
116     *
117     * @param tx transaction
118     * @param id object id
119     * @param subField indexed sub-field
120     * @param contentKey the content key
121     * @param contentValue the value associated with the content key, or null if not needed
122     */
123    void removeIndexEntry(Transaction tx, ObjId id, SimpleField<?> subField, byte[] contentKey, byte[] contentValue) {
124        tx.kvt.remove(this.buildIndexEntry(id, subField, contentKey, contentValue));
125    }
126
127    private byte[] buildIndexEntry(ObjId id, SimpleField<?> subField, byte[] contentKey, byte[] contentValue) {
128        final ByteReader contentKeyReader = new ByteReader(contentKey);
129        contentKeyReader.skip(ObjId.NUM_BYTES + this.storageIdLength);                  // skip to content
130        final ByteWriter writer = new ByteWriter();
131        UnsignedIntEncoder.write(writer, subField.storageId);
132        this.buildIndexEntry(id, subField, contentKeyReader, contentValue, writer);
133        return writer.getBytes();
134    }
135
136    /**
137     * Build an index key for the given object, sub-field, and content key/value pair.
138     *
139     * @param id object id
140     * @param subField indexed sub-field
141     * @param reader reader of content key, positioned just after the object ID and the storage ID for this field
142     * @param value the value associated with the content key, or null if not needed
143     * @param writer writer for the index entry, with the sub-field's storage ID already written
144     */
145    abstract void buildIndexEntry(ObjId id, SimpleField<?> subField, ByteReader reader, byte[] value, ByteWriter writer);
146
147    /**
148     * Add or remove index entries for the given object as appropriate after a schema version change
149     * which changed only whether some or all sub-field(s) are indexed.
150     *
151     * @param kvt KV store
152     * @param oldField compatible field in older schema
153     * @param id object id
154     */
155    void updateSubFieldIndexes(Transaction tx, ComplexField<?> oldField, ObjId id) {
156        final Iterator<? extends SimpleField<?>> oldSubFields = oldField.getSubFields().iterator();
157        final Iterator<? extends SimpleField<?>> newSubFields = this.getSubFields().iterator();
158        while (oldSubFields.hasNext() || newSubFields.hasNext()) {
159            final SimpleField<?> oldSubField = oldSubFields.next();
160            final SimpleField<?> newSubField = newSubFields.next();
161            if (!oldSubField.indexed && newSubField.indexed)
162                this.addIndexEntries(tx, id, newSubField);
163            else if (oldSubField.indexed && !newSubField.indexed)
164                oldField.removeIndexEntries(tx, id, oldSubField);
165        }
166    }
167
168    /**
169     * Add all index entries for the given object and sub-field.
170     *
171     * @param tx transaction
172     * @param id object id
173     * @param subField sub-field of this field
174     */
175    void addIndexEntries(Transaction tx, ObjId id, SimpleField<?> subField) {
176        if (!subField.indexed)
177            throw new IllegalArgumentException(this + " is not indexed");
178        final byte[] prefix = this.buildKey(id);
179        final byte[] prefixEnd = ByteUtil.getKeyAfterPrefix(prefix);
180        for (Iterator<KVPair> i = tx.kvt.getRange(prefix, prefixEnd, false); i.hasNext(); ) {
181            final KVPair pair = i.next();
182            this.addIndexEntry(tx, id, subField, pair.getKey(), pair.getValue());
183        }
184    }
185
186    /**
187     * Remove all index entries for the given object.
188     *
189     * @param tx transaction
190     * @param id object id
191     */
192    void removeIndexEntries(Transaction tx, ObjId id) {
193        for (SimpleField<?> subField : this.getSubFields()) {
194            if (subField.indexed)
195                this.removeIndexEntries(tx, id, subField);
196        }
197    }
198
199    /**
200     * Remove all index entries for the given object and sub-field.
201     *
202     * @param tx transaction
203     * @param id object id
204     * @param subField sub-field of this field
205     */
206    void removeIndexEntries(Transaction tx, ObjId id, SimpleField<?> subField) {
207        final byte[] prefix = this.buildKey(id);
208        this.removeIndexEntries(tx, id, subField, prefix, ByteUtil.getKeyAfterPrefix(prefix));
209    }
210
211    /**
212     * Remove index entries for the given object and sub-field, restricted to the given key range.
213     *
214     * @param tx transaction
215     * @param id object id
216     * @param subField sub-field of this field
217     */
218    void removeIndexEntries(Transaction tx, ObjId id, SimpleField<?> subField, byte[] minKey, byte[] maxKey) {
219        if (!subField.indexed)
220            throw new IllegalArgumentException(this + " is not indexed");
221        for (Iterator<KVPair> i = tx.kvt.getRange(minKey, maxKey, false); i.hasNext(); ) {
222            final KVPair pair = i.next();
223            this.removeIndexEntry(tx, id, subField, pair.getKey(), pair.getValue());
224        }
225    }
226
227    /**
228     * Remove all field entries in which the specified reference sub-field refers to an object
229     * type that is in the specified set of newly disallowed object types.
230     */
231    abstract void unreferenceRemovedObjectTypes(Transaction tx,
232      ObjId id, ReferenceField subField, SortedSet<Integer> removedStorageIds);
233}
