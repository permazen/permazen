006package org.jsimpledb.core;
007
008import java.security.SecureRandom;
009import java.util.regex.Pattern;
010
011import org.jsimpledb.kv.KeyRange;
012import org.jsimpledb.util.ByteReader;
013import org.jsimpledb.util.ByteUtil;
014import org.jsimpledb.util.ByteWriter;
015import org.jsimpledb.util.UnsignedIntEncoder;
016
017/**
018 * Object IDs. Instances identify individual {@link Database} objects.
019 */
020public class ObjId implements Comparable<ObjId> {
021
022    /**
023     * The number of bytes in the binary encoding of an {@link ObjId}.
024     */
025    public static final int NUM_BYTES = 8;
026
027    /**
028     * Regular expression that matches the string encoding of an {@link ObjId}.
029     */
030    public static final Pattern PATTERN = Pattern.compile("\\p{XDigit}{" + (NUM_BYTES * 2) + "}");
031
032    private static final ThreadLocal<SecureRandom> RANDOM = new ThreadLocal<SecureRandom>() {
033        @Override
034        protected SecureRandom initialValue() {
035            return new SecureRandom();
036        }
037    };
038
039    private final long value;
040
041// Constructors
042
043    /**
044     * Create a new, random instance with the given storage ID.
045     *
046     * @param storageId storage ID, must be greater than zero
047     * @throws IllegalArgumentException if {@code storageId} is zero or negative
048     */
049    public ObjId(int storageId) {
050        this(ObjId.buildRandom(storageId));
051    }
052
053    /**
054     * Constructor that parses a string previously returned by {@link #toString}.
055     *
056     * @param string string encoding of an instance
057     * @throws IllegalArgumentException if {@code string} is invalid
058     */
059    public ObjId(String string) {
060        this(ObjId.parseString(string));
061    }
062
063    /**
064     * Constructor that reads an encoded instance from the given {@link ByteReader}.
065     *
066     * @param reader input for binary encoding of an instance
067     * @throws IllegalArgumentException if {@code reader} contains invalid data
068     */
069    public ObjId(ByteReader reader) {
070        if (reader == null)
071            throw new IllegalArgumentException("null reader");
072        this.value = ByteUtil.readLong(reader);
073        this.validateStorageId();
074    }
075
076    /**
077     * Constructor using a long value previously returned by {@link #asLong}.
078     *
079     * @param value long encoding of an instance
080     * @throws IllegalArgumentException if {@code value} is invalid
081     */
082    public ObjId(long value) {
083        this.value = value;
084        this.validateStorageId();
085    }
086
087    private void validateStorageId() {
088        final int storageId;
089        try {
090            storageId = this.getStorageId();
091        } catch (IllegalArgumentException e) {
092            throw new IllegalArgumentException("invalid object ID", e);
093        }
094        if (storageId <= 0)
095            throw new IllegalArgumentException("invalid object ID containing storage ID " + storageId);
096    }
097
098// Methods
099
100    /**
101     * Get the storage ID associated with this instance. This is the storage ID of the associated object type.
102     *
103     * @return object type storage ID
104     */
105    public int getStorageId() {
106        return UnsignedIntEncoder.read(new ByteReader(this.getBytes()));
107    }
108
109    /**
110     * Get the binary encoding of this instance.
111     *
112     * @return binary encoding
113     */
114    public byte[] getBytes() {
115        final ByteWriter writer = new ByteWriter(NUM_BYTES);
116        this.writeTo(writer);
117        return writer.getBytes();
118    }
119
120    /**
121     * Get this instance encoded as a {@code long} value.
122     *
123     * @return long encoding
124     */
125    public long asLong() {
126        return this.value;
127    }
128
129    /**
130     * Write the binary encoding of this instance to the given output.
131     *
132     * @param writer destination
133     */
134    public void writeTo(ByteWriter writer) {
135        ByteUtil.writeLong(writer, this.value);
136    }
137
138    /**
139     * Get the smallest (i.e., first) instance having the given storage ID.
140     *
141     * @param storageId storage ID, must be greater than zero
142     * @return smallest instance with storage ID {@code storageId} (inclusive)
143     * @throws IllegalArgumentException if {@code storageId} is zero or negative
144     */
145    public static ObjId getMin(int storageId) {
146        return ObjId.getFill(storageId, 0x00);
147    }
148
149    /**
150     * Get the largest (i.e., last) instance having the given storage ID.
151     *
152     * @param storageId storage ID, must be greater than zero
153     * @return largest instance with storage ID {@code storageId} (inclusive)
154     * @throws IllegalArgumentException if {@code storageId} is zero or negative
155     */
156    public static ObjId getMax(int storageId) {
157        return ObjId.getFill(storageId, 0xff);
158    }
159
160    /**
161     * Get the {@link KeyRange} containing all object IDs with the given storage ID.
162     *
163     * @param storageId storage ID, must be greater than zero
164     * @return {@link KeyRange} containing all object IDs having the specified type
165     * @throws IllegalArgumentException if {@code storageId} is zero or negative
166     */
167    public static KeyRange getKeyRange(int storageId) {
168        if (storageId <= 0)
169            throw new IllegalArgumentException("invalid storageId " + storageId);
170        final ByteWriter writer = new ByteWriter(NUM_BYTES);
171        UnsignedIntEncoder.write(writer, storageId);
172        return KeyRange.forPrefix(writer.getBytes());
173    }
174
175    private static ObjId getFill(int storageId, int value) {
176        if (storageId <= 0)
177            throw new IllegalArgumentException("invalid storage ID " + storageId);
178        final ByteWriter writer = new ByteWriter(NUM_BYTES);
179        UnsignedIntEncoder.write(writer, storageId);
180        for (int remain = NUM_BYTES - writer.getLength(); remain > 0; remain--)
181            writer.writeByte(value);
182        return new ObjId(new ByteReader(writer));
183    }
184
185// Object
186
187    /**
188     * Encode this instance as a string.
189     */
190    @Override
191    public String toString() {
192        final ByteWriter writer = new ByteWriter(NUM_BYTES);
193        this.writeTo(writer);
194        final byte[] buf = writer.getBytes();
195        final char[] result = new char[NUM_BYTES * 2];
196        int off = 0;
197        for (int i = 0; i < buf.length; i++) {
198            int b = buf[i];
199            result[off++] = Character.forDigit((b >> 4) & 0x0f, 16);
200            result[off++] = Character.forDigit(b & 0x0f, 16);
201        }
202        return new String(result);
203    }
204
205    @Override
206    public boolean equals(Object obj) {
207        if (obj == this)
208            return true;
209        if (obj == null || obj.getClass() != this.getClass())
210            return false;
211        final ObjId that = (ObjId)obj;
212        return this.value == that.value;
213    }
214
215    /**
216     * Returns the hash code value for this instance.
217     *
218     * <p>
219     * The hash code of an {@link ObjId} is defined as the hash code of its {@link #asLong} value,
220     * which is {@linkplain Long#hashCode defined} as the exclusive-or of the upper and lower 32 bits.
221     * </p>
222     */
223    @Override
224    public int hashCode() {
225        return (int)(this.value >> 32) ^ (int)this.value;
226    }
227
228// Comparable
229
230    @Override
231    public int compareTo(ObjId that) {
232        return ByteUtil.compare(this.getBytes(), that.getBytes());
233    }
234
235// Internal methods
236
237    private static ByteReader buildRandom(int storageId) {
238        if (storageId <= 0)
239            throw new IllegalArgumentException("invalid storage ID " + storageId);
240        final ByteWriter writer = new ByteWriter(NUM_BYTES);
241        UnsignedIntEncoder.write(writer, storageId);
242        final byte[] randomPart = new byte[NUM_BYTES - writer.getLength()];
243        ObjId.RANDOM.get().nextBytes(randomPart);
244        writer.write(randomPart);
245        return new ByteReader(writer);
246    }
247
248    private static ByteReader parseString(String string) {
249        if (string == null)
250            throw new IllegalArgumentException("null string");
251        if (string.length() != NUM_BYTES * 2)
252            throw new IllegalArgumentException("invalid object ID `" + string + "'");
253        final byte[] buf = new byte[NUM_BYTES];
254        int off = 0;
255        for (int i = 0; i < buf.length; i++) {
256            final int digit1 = Character.digit(string.charAt(off++), 16);
257            final int digit2 = Character.digit(string.charAt(off++), 16);
258            if (digit1 == -1 || digit2 == -1)
259                throw new IllegalArgumentException("invalid object ID `" + string + "'");
260            buf[i] = (byte)((digit1 << 4) | digit2);
261        }
262        return new ByteReader(buf);
263    }
264}
