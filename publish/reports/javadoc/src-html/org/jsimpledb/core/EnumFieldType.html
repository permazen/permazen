006package org.jsimpledb.core;
007
008import com.google.common.base.Function;
009import com.google.common.collect.Lists;
010
011import java.util.Collection;
012import java.util.LinkedHashMap;
013import java.util.List;
014import java.util.Map;
015
016import org.dellroad.stuff.java.EnumUtil;
017
018/**
019 * The {@link FieldType} for {@link EnumField}s.
020 *
021 * <p>
022 * Each {@link EnumFieldType} instance has an ordered list of {@link String} identifiers; two {@link EnumFieldType}
023 * instances are not compatible unless they have identical identifier lists. The identifiers must be valid Java identifiers.
024 * </p>
025 *
026 * <p>
027 * Null values are supported by this class.
028 * </p>
029 */
030public class EnumFieldType extends NullSafeType<EnumValue> {
031
032    public static final String IDENT_PATTERN = "\\p{javaJavaIdentifierStart}\\p{javaJavaIdentifierPart}*";
033
034    /**
035     * Constructor to use when there is an associated {@link Enum} type;
036     * {@link #getEnumType} will return {@code enumType}.
037     *
038     * @param enumType Java {@link Enum} type from which to derive the ordered identifier list
039     * @param <T> enum type
040     * @throws NullPointerException if {@code enumType} is null
041     */
042    public <T extends Enum<T>> EnumFieldType(Class<T> enumType) {
043        super(new EnumType(enumType, enumType.getName(), EnumFieldType.getIdentifiers(enumType)));
044    }
045
046    /**
047     * Constructor to use when there is no corresponding {@link Enum} type;
048     * {@link #getEnumType} will return null.
049     *
050     * @param name name of this type
051     * @param idents ordered list of identifiers
052     * @throws IllegalArgumentException if {@code name} is null or invalid
053     * @throws IllegalArgumentException if {@code idents} is null or contains a duplicate or invalid identifier
054     */
055    public EnumFieldType(String name, List<String> idents) {
056        super(new EnumType(null, name, idents));
057    }
058
059    /**
060     * Get the identifiers associated with this instance in ordinal order.
061     *
062     * @return unmodifiable, ordinally ordered list of identifiers
063     */
064    public List<String> getIdentifiers() {
065        return ((EnumType)this.inner).getIdentifiers();
066    }
067
068    /**
069     * Get the {@link Enum} type associated with this instance, if known.
070     *
071     * @return associated {@link Enum} type, or null if actual {@link Enum} type is unknown
072     */
073    public Class<? extends Enum<?>> getEnumType() {
074        return ((EnumType)this.inner).getEnumType();
075    }
076
077    /**
078     * Create an {@link EnumFieldType} instance suitable for use with the given {@link Enum} type.
079     *
080     * @param enumType Java {@link Enum} type from which to derive the ordered identifier list
081     * @return an {@link EnumFieldType} based on {@code enumType}
082     * @throws ClassCastException if {@code enumType} does not subclass {@link Enum}
083     * @throws NullPointerException if {@code enumType} is null
084     */
085    public static EnumFieldType create(Class<?> enumType) {
086        return EnumFieldType.doCreate(enumType);
087    }
088
089    // This method exists solely to bind the generic type parameters
090    @SuppressWarnings("unchecked")
091    private static <T extends Enum<T>> EnumFieldType doCreate(Class<?> enumType) {
092        return new EnumFieldType((Class<T>)enumType.asSubclass(Enum.class));
093    }
094
095    /**
096     * Create an {@link EnumFieldType} instance suitable for use with the given identifier list.
097     *
098     * <p>
099     * If {@code name} is the name of an {@link Enum} class with matching identifiers, then it will
100     * associated with the created instance and returned by its {@link #getEnumType} method.
101     * </p>
102     *
103     * @param name name of the type
104     * @param idents ordered list of identifiers
105     * @return an {@link EnumFieldType} based on the specified {@code idents}
106     * @throws IllegalArgumentException if {@code name} is null or invalid
107     * @throws IllegalArgumentException if {@code idents} is null or contains a duplicate or invalid identifier
108     */
109    public static EnumFieldType create(String name, List<String> idents) {
110        return EnumFieldType.doCreate(name, idents);
111    }
112
113    // This method exists solely to bind the generic type parameters
114    @SuppressWarnings("unchecked")
115    private static <T extends Enum<T>> EnumFieldType doCreate(String name, List<String> idents) {
116        do {
117
118            // Search for Enum type
119            final Class<T> enumType;
120            try {
121                enumType = (Class<T>)Class.forName(name, false,
122                  Thread.currentThread().getContextClassLoader()).asSubclass(Enum.class);
123            } catch (Exception e) {
124                break;
125            }
126
127            // Validate it has the expected identifier list
128            if (!EnumFieldType.getIdentifiers(enumType).equals(idents))
129                break;
130
131            // We're good
132            return new EnumFieldType(enumType);
133        } while (false);
134
135        // Enum type not found or has incompatible identifier list
136        return new EnumFieldType(name, idents);
137    }
138
139    /**
140     * Validate a list of enum identifiers and build a mapping from identifier to corresponding {@link EnumValue}.
141     * The returned mapping will iterate the {@link EnumValue}s in ordinal order.
142     *
143     * @param idents enum identifiers
144     * @return ordinally ordered mapping from identifier to {@link EnumValue}
145     * @throws IllegalArgumentException if {@code idents} is null
146     * @throws IllegalArgumentException if any identifier in {@code idents} is null, duplicate, or not a valid Java identifier
147     */
148    public static Map<String, EnumValue> validateIdentifiers(List<String> idents) {
149        if (idents == null)
150            throw new IllegalArgumentException("null idents");
151        final LinkedHashMap<String, EnumValue> identifierMap = idents instanceof Collection ?
152          new LinkedHashMap<String, EnumValue>(((Collection<?>)idents).size()) : new LinkedHashMap<String, EnumValue>();
153        for (String ident : idents) {
154            final int index = identifierMap.size();
155            if (ident == null)
156                throw new IllegalArgumentException("invalid null enum identifier at index " + index);
157            if (ident.equals("null") || !ident.matches(IDENT_PATTERN))
158                throw new IllegalArgumentException("invalid enum identifier `" + ident + "' at index " + index);
159            final EnumValue otherValue = identifierMap.put(ident, new EnumValue(ident, index));
160            if (otherValue != null) {
161                throw new IllegalArgumentException("invalid duplicate enum identifier `" + ident
162                  + "' at indexes " + otherValue.getOrdinal() + " and " + index);
163            }
164        }
165        return identifierMap;
166    }
167
168    @SuppressWarnings("unchecked")
169    private static <T extends Enum<T>> List<String> getIdentifiers(Class<T> enumType) {
170        return Lists.transform(EnumUtil.getValues(enumType), new Function<T, String>() {
171            @Override
172            public String apply(T value) {
173                return value.name();
174            }
175        });
176    }
177}
