006package org.jsimpledb.core;
007
008import com.google.common.base.Function;
009import com.google.common.base.Predicate;
010import com.google.common.collect.Iterables;
011import com.google.common.collect.Maps;
012import com.google.common.primitives.Ints;
013
014import java.util.ArrayList;
015import java.util.Collections;
016import java.util.HashMap;
017import java.util.Map;
018import java.util.SortedMap;
019import java.util.TreeMap;
020
021import org.jsimpledb.schema.SchemaCompositeIndex;
022import org.jsimpledb.schema.SchemaField;
023import org.jsimpledb.schema.SchemaObjectType;
024
025/**
026 * Represents a {@link Database} object type.
027 */
028public class ObjType extends SchemaItem {
029
030    final FieldTypeRegistry fieldTypeRegistry;
031    final TreeMap<Integer, Field<?>> fields = new TreeMap<>();
032    final TreeMap<String, Field<?>> fieldsByName = new TreeMap<>();
033    final TreeMap<Integer, CompositeIndex> compositeIndexes = new TreeMap<>();
034    final TreeMap<String, CompositeIndex> compositeIndexesByName = new TreeMap<>();
035    final TreeMap<Integer, SimpleField<?>> simpleFields = new TreeMap<>();
036    final TreeMap<Integer, ComplexField<?>> complexFields = new TreeMap<>();
037    final TreeMap<Integer, CounterField> counterFields = new TreeMap<>();
038    final TreeMap<Integer, ReferenceField> referenceFields = new TreeMap<>();           // includes sub-fields too
039
040    /**
041     * Constructor.
042     */
043    ObjType(SchemaObjectType schemaObjectType, Schema schema, FieldTypeRegistry fieldTypeRegistry) {
044        super(schemaObjectType.getName(), schemaObjectType.getStorageId(), schema);
045
046        // Sanity check
047        if (fieldTypeRegistry == null)
048            throw new IllegalArgumentException("null fieldTypeRegistry");
049        this.fieldTypeRegistry = fieldTypeRegistry;
050
051        // Build fields
052        final FieldBuilder fieldBuilder = new FieldBuilder(this.schema, this.fieldTypeRegistry);
053        for (SchemaField schemaField : schemaObjectType.getSchemaFields().values())
054            this.addSchemaItem(fields, fieldsByName, schemaField.visit(fieldBuilder));
055
056        // Build mappings for various field types
057        this.buildMap(this.simpleFields, SimpleField.class);
058        this.buildMap(this.complexFields, ComplexField.class);
059        this.buildMap(this.counterFields, CounterField.class);
060        for (ReferenceField referenceField : Iterables.filter(this.getFieldsAndSubFields(), ReferenceField.class))
061            this.referenceFields.put(referenceField.storageId, referenceField);
062
063        // Build composite indexes
064        for (SchemaCompositeIndex schemaIndex : schemaObjectType.getSchemaCompositeIndexes().values())
065            this.addCompositeIndex(this.schema, schemaIndex);
066
067        // Link simple fields to the composite indexes they include
068        for (SimpleField<?> field : this.simpleFields.values()) {
069            final HashMap<CompositeIndex, Integer> indexMap = new HashMap<>();
070            for (CompositeIndex index : this.compositeIndexes.values()) {
071                final int i = index.fields.indexOf(field);
072                if (i != -1)
073                    indexMap.put(index, i);
074            }
075            assert field.compositeIndexMap == null;
076            if (!indexMap.isEmpty())
077                field.compositeIndexMap = Collections.unmodifiableMap(indexMap);
078        }
079    }
080
081    /**
082     * Get all fields associated with this object type keyed by storage ID. Does not include sub-fields of complex fields.
083     *
084     * @return unmodifiable mapping from {@linkplain Field#getStorageId field storage ID} to field
085     */
086    public SortedMap<Integer, Field<?>> getFields() {
087        return Collections.unmodifiableSortedMap(this.fields);
088    }
089
090    /**
091     * Get the {@link Field} in this instance with the given storage ID.
092     *
093     * @param storageId storage ID
094     * @return the {@link Field} with storage ID {@code storageID}
095     * @throws UnknownFieldException if no {@link Field} with storage ID {@code storageId} exists
096     */
097    public Field<?> getField(int storageId) {
098        final Field<?> field = this.fields.get(storageId);
099        if (field == null)
100            throw new UnknownFieldException(this, storageId, "field");
101        return field;
102    }
103
104    /**
105     * Get all fields associated with this object type keyed by name. Does not include sub-fields of complex fields.
106     *
107     * @return unmodifiable mapping from {@linkplain Field#getName field name} to field
108     */
109    public SortedMap<String, Field<?>> getFieldsByName() {
110        return Collections.unmodifiableSortedMap(this.fieldsByName);
111    }
112
113    /**
114     * Get all composite indexes associated with this object type keyed by storage ID.
115     *
116     * @return unmodifiable mapping from {@linkplain CompositeIndex#getStorageId composite index storage ID} to field
117     */
118    public SortedMap<Integer, CompositeIndex> getCompositeIndexes() {
119        return Collections.unmodifiableSortedMap(this.compositeIndexes);
120    }
121
122    /**
123     * Get the {@link CompositeIndex} associated with this instance with the given storage ID.
124     *
125     * @param storageId storage ID
126     * @return the {@link CompositeIndex} with storage ID {@code storageID}
127     * @throws UnknownIndexException if no {@link CompositeIndex} with storage ID {@code storageId} exists
128     */
129    public CompositeIndex getCompositeIndex(int storageId) {
130        final CompositeIndex index = this.compositeIndexes.get(storageId);
131        if (index == null)
132            throw new UnknownIndexException(storageId, "composite index");
133        return index;
134    }
135
136    /**
137     * Get all composite indexes associated with this object type keyed by name.
138     *
139     * @return unmodifiable mapping from {@linkplain CompositeIndex#getName index name} to index
140     */
141    public SortedMap<String, CompositeIndex> getCompositeIndexesByName() {
142        return Collections.unmodifiableSortedMap(this.compositeIndexesByName);
143    }
144
145    /**
146     * Get all fields, including sub-fields.
147     */
148    Iterable<Field<?>> getFieldsAndSubFields() {
149        Iterable<Field<?>> i = this.fields.values();
150        for (ComplexField<?> field : this.complexFields.values())
151            i = Iterables.concat(i, field.getSubFields());
152        return i;
153    }
154
155    @Override
156    ObjTypeStorageInfo toStorageInfo() {
157        return new ObjTypeStorageInfo(this);
158    }
159
160    @Override
161    public String toString() {
162        return "object type `" + this.name + "' in " + this.schema;
163    }
164
165// Internal methods
166
167    private <T extends Field<?>> void buildMap(TreeMap<Integer, T> map, final Class<? super T> type) {
168        map.putAll(Maps.transformValues(Maps.filterValues(this.fields, new Predicate<Field<?>>() {
169            @Override
170            public boolean apply(Field<?> field) {
171                return type.isInstance(field);
172            }
173        }), new Function<Field<?>, T>() {
174            @Override
175            @SuppressWarnings("unchecked")
176            public T apply(Field<?> field) {
177                return (T)type.cast(field);
178            }
179        }));
180    }
181
182    private <T extends SchemaItem> void addSchemaItem(Map<Integer, T> byStorageId, Map<String, T> byName, T item) {
183        T previous = byStorageId.put(item.storageId, item);
184        if (previous != null) {
185            throw new IllegalArgumentException("duplicate use of storage ID " + item.storageId
186              + " by " + previous + " and " + item + " in " + this);
187        }
188        previous = byName.put(item.name, item);
189        if (previous != null) {
190            throw new IllegalArgumentException("duplicate use of name `" + item.name
191              + "' by " + previous + " and " + item + " in " + this);
192        }
193    }
194
195    private CompositeIndex addCompositeIndex(Schema schema, SchemaCompositeIndex schemaIndex) {
196
197        // Get fields corresponding to specified storage IDs
198        final int[] storageIds = Ints.toArray(schemaIndex.getIndexedFields());
199        if (storageIds.length < 2 || storageIds.length > Database.MAX_INDEXED_FIELDS)
200            throw new IllegalArgumentException("invalid " + schemaIndex + ": can't index " + storageIds.length + " fields");
201        final ArrayList<SimpleField<?>> list = new ArrayList<>(storageIds.length);
202        int count = 0;
203        for (int storageId : storageIds) {
204            final Field<?> field = this.fields.get(storageId);
205            if (!(field instanceof SimpleField)) {
206                throw new IllegalArgumentException("invalid " + schemaIndex
207                  + ": no simple field with storage ID " + storageId + " found");
208            }
209            final SimpleField<?> simpleField = (SimpleField<?>)field;
210            if (simpleField.parent != null) {
211                throw new IllegalArgumentException("invalid " + schemaIndex
212                  + ": simple field with storage ID " + storageId + " is a sub-field of a complex field");
213            }
214            list.add(simpleField);
215        }
216
217        // Create and add index
218        final CompositeIndex index = new CompositeIndex(schemaIndex.getName(), schemaIndex.getStorageId(), schema, this, list);
219        this.addSchemaItem(this.compositeIndexes, this.compositeIndexesByName, index);
220        return index;
221    }
222}
