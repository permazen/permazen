006package org.jsimpledb.core;
007
008import org.jsimpledb.parse.ParseContext;
009import org.jsimpledb.util.ByteReader;
010import org.jsimpledb.util.ByteWriter;
011
012/**
013 * A {@link FieldType} that wraps any other {@link FieldType} not supporting null values and adds support for null values
014 * by pre-pending to the binary encoding a {@code 0x00} for non-null values or a {@code 0xff} byte for null values. Therefore,
015 * null values sort last.
016 *
017 * <p>
018 * The default value becomes null, for which {@code "null"} is the value returned by {@link #toParseableString toParseableString()}.
019 * Therefore, {@code "null"} must not be returned by the wrapped type's {@link #toParseableString toParseableString()}
020 * for any value.
021 * </p>
022 *
023 * <p>
024 * This class will automatically "inline" the {@code 0xff} for null values if the wrapped {@link FieldType}'s
025 * {@link FieldType#hasPrefix0xff} method returns false.
026 * </p>
027 */
028public class NullSafeType<T> extends FieldType<T> {
029
030    /**
031     * Not null sentinel byte value. Used only when inlining is not in effect.
032     */
033    public static final int NOT_NULL_SENTINEL = 0x01;
034
035    /**
036     * Null sentinel byte value.
037     */
038    public static final int NULL_SENTINEL = 0xff;
039
040    private static final byte[] DEFAULT_VALUE = new byte[] { (byte)NULL_SENTINEL };
041
042    /**
043     * The inner {@link FieldType} that this instance wraps.
044     */
045    protected final FieldType<T> inner;
046
047    private final boolean inline;
048
049    /**
050     * Constructor.
051     *
052     * @param name type name
053     * @param inner inner type that is not null safe
054     */
055    public NullSafeType(String name, FieldType<T> inner) {
056        super(name, inner.getTypeToken().wrap(), inner.getEncodingSignature());
057        if (inner instanceof NullSafeType)
058            throw new IllegalArgumentException("inner type is already null-safe");
059        this.inner = inner;
060        this.inline = !inner.hasPrefix0xff();
061    }
062
063    /**
064     * Constructor. Takes type name from {@code inner}; therefore, this instance and {@code inner}
065     * cannot be both registered in a {@link FieldTypeRegistry}.
066     *
067     * @param inner inner type that is not null safe
068     */
069    public NullSafeType(FieldType<T> inner) {
070       this(inner.name, inner);
071    }
072
073    @Override
074    public T read(ByteReader reader) {
075        if (this.inline) {
076            if (reader.peek() == NULL_SENTINEL) {
077                reader.skip(1);
078                return null;
079            }
080            return this.inner.read(reader);
081        } else {
082            switch (reader.readByte()) {
083            case NULL_SENTINEL:
084                return null;
085            case NOT_NULL_SENTINEL:
086                return this.inner.read(reader);
087            default:
088                throw new IllegalArgumentException("invalid encoding of " + this);
089            }
090        }
091    }
092
093    @Override
094    public void write(ByteWriter writer, T value) {
095        if (value == null) {
096            writer.writeByte(NULL_SENTINEL);
097            return;
098        }
099        if (!this.inline)
100            writer.writeByte(NOT_NULL_SENTINEL);
101        this.inner.write(writer, value);
102    }
103
104    @Override
105    public byte[] getDefaultValue() {
106        return DEFAULT_VALUE;
107    }
108
109    @Override
110    public void skip(ByteReader reader) {
111        if (this.inline) {
112            final int prefix = reader.peek();
113            switch (prefix) {
114            case NULL_SENTINEL:
115                reader.skip(1);
116                break;
117            default:
118                this.inner.skip(reader);
119                break;
120            }
121        } else {
122            final int prefix = reader.readByte();
123            switch (prefix) {
124            case NULL_SENTINEL:
125                break;
126            case NOT_NULL_SENTINEL:
127                this.inner.skip(reader);
128                break;
129            default:
130                throw new IllegalArgumentException("invalid encoding of " + this);
131            }
132        }
133    }
134
135    @Override
136    public T fromString(String string) {
137        return this.inner.fromString(string);
138    }
139
140    @Override
141    public String toString(T value) {
142        if (value == null)
143            throw new IllegalArgumentException("null value");
144        return this.inner.toString(value);
145    }
146
147    @Override
148    public T fromParseableString(ParseContext context) {
149        return context.tryPattern("null\\b") != null ? null : this.inner.fromParseableString(context);
150    }
151
152    @Override
153    public String toParseableString(T value) {
154        return value == null ? "null" : this.inner.toParseableString(value);
155    }
156
157    @Override
158    public int compare(T value1, T value2) {
159        if (value1 == null)
160            return value2 == null ? 0 : 1;
161        if (value2 == null)
162            return value1 == null ? 0 : -1;
163        return this.inner.compare(value1, value2);
164    }
165
166    @Override
167    public boolean hasPrefix0xff() {
168        return true;
169    }
170
171    @Override
172    public boolean hasPrefix0x00() {
173        return this.inline && this.inner.hasPrefix0x00();
174    }
175
176// Object
177
178    @Override
179    public int hashCode() {
180        return super.hashCode() ^ this.inner.hashCode();
181    }
182
183    @Override
184    public boolean equals(Object obj) {
185        if (obj == this)
186            return true;
187        if (!super.equals(obj))
188            return false;
189        final NullSafeType<?> that = (NullSafeType<?>)obj;
190        return this.inner.equals(that.inner);
191    }
192}
