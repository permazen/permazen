006package org.jsimpledb.core;
007
008import java.util.List;
009
010/**
011 * A field that contains one entry in an ordered list of unique {@link String} identifiers.
012 *
013 * <p>
014 * Two instances of this class are considered compatible only when their ordered lists of identifiers are identical.
015 * </p>
016 */
017public class EnumField extends SimpleField<EnumValue> {
018
019    /**
020     * Constructor.
021     *
022     * @param name the name of the field
023     * @param storageId field storage ID
024     * @param schema schema version
025     * @param typeName {@link Enum} class type name
026     * @param idents the unique enum identifiers
027     * @param indexed whether this field is indexed
028     * @throws IllegalArgumentException if any parameter is null
029     * @throws IllegalArgumentException if {@code name} is invalid
030     * @throws IllegalArgumentException if {@code storageId} is invalid
031     * @throws IllegalArgumentException if any identifier in {@code idents} is null, duplicate, or not a valid Java identifier
032     */
033    EnumField(String name, int storageId, Schema schema, boolean indexed, String typeName, List<String> idents) {
034        super(name, storageId, schema, EnumFieldType.create(typeName, idents), indexed);
035    }
036
037// Public methods
038
039    /**
040     * Get the {@link EnumFieldType} associated with this instance.
041     */
042    @Override
043    public EnumFieldType getFieldType() {
044        return (EnumFieldType)super.getFieldType();
045    }
046
047    @Override
048    public <R> R visit(FieldSwitch<R> target) {
049        return target.caseEnumField(this);
050    }
051
052    @Override
053    public String toString() {
054        return "enum field `" + this.name + "'";
055    }
056
057// Non-public methods
058
059    @Override
060    EnumFieldStorageInfo toStorageInfo() {
061        return new EnumFieldStorageInfo(this, this.parent != null ? this.parent.storageId : 0);
062    }
063}
