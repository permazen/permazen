006package org.jsimpledb.core;
007
008import java.util.Collections;
009import java.util.Set;
010import java.util.SortedSet;
011import java.util.TreeSet;
012
013import org.jsimpledb.util.ByteWriter;
014
015/**
016 * The {@link FieldType} for {@link ReferenceField}s. Instances support object type restriction.
017 *
018 * <p>
019 * Null values are supported by this class.
020 * </p>
021 */
022public class ReferenceFieldType extends NullSafeType<ObjId> {
023
024    private final TreeSet<Integer> objectTypes;
025
026    /**
027     * Constructor.
028     *
029     * <p>
030     * No restrictions will be placed on encoded references.
031     * </p>
032     */
033    public ReferenceFieldType() {
034        this(null);
035    }
036
037    /**
038     * Constructor.
039     *
040     * @param objectTypes allowed object type storage IDs, or null for no restriction
041     */
042    public ReferenceFieldType(Set<Integer> objectTypes) {
043        super(FieldType.REFERENCE_TYPE_NAME, FieldTypeRegistry.OBJ_ID);
044        this.objectTypes = objectTypes != null ? new TreeSet<>(objectTypes) : null;
045    }
046
047    /**
048     * Get the object types this field type is allowed to reference, if so restricted.
049     *
050     * @return storage IDs of allowed object types, or null if there is no restriction
051     */
052    public SortedSet<Integer> getObjectTypes() {
053        return objectTypes != null ? Collections.unmodifiableSortedSet(this.objectTypes) : null;
054    }
055
056// FieldType
057
058    @Override
059    public void write(ByteWriter writer, ObjId id) {
060        super.write(writer, this.checkAllowed(id));
061    }
062
063    @Override
064    public ObjId validate(Object obj) {
065        return this.checkAllowed(super.validate(obj));
066    }
067
068    /**
069     * Verify the reference value is permitted by this instance.
070     *
071     * @param id reference value
072     * @return validated reference
073     * @throws InvalidReferenceException if {@code id} has a storage ID that is not allowed by this instance
074     */
075    protected ObjId checkAllowed(ObjId id) {
076        if (this.objectTypes != null && id != null && !this.objectTypes.contains(id.getStorageId()))
077            throw new InvalidReferenceException(id, this.objectTypes);
078        return id;
079    }
080
081// Object
082
083    @Override
084    public String toString() {
085        String desc = super.toString();
086        if (this.objectTypes != null)
087            desc += " to " + this.objectTypes;
088        return desc;
089    }
090
091    @Override
092    public boolean equals(Object obj) {
093        if (obj == this)
094            return true;
095        if (!super.equals(obj))
096            return false;
097        final ReferenceFieldType that = (ReferenceFieldType)obj;
098        return this.objectTypes != null ? this.objectTypes.equals(that.objectTypes) : that.objectTypes == null;
099    }
100
101    @Override
102    public int hashCode() {
103        return super.hashCode() ^ (this.objectTypes != null ? this.objectTypes.hashCode() : 0);
104    }
105}
