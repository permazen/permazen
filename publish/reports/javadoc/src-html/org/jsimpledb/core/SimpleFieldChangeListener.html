006package org.jsimpledb.core;
007
008import java.util.NavigableSet;
009
010/**
011 * Listener interface for notifications of a change in value of a {@link SimpleField}.
012 *
013 * @see Transaction#addSimpleFieldChangeListener Transaction.addSimpleFieldChangeListener()
014 */
015public interface SimpleFieldChangeListener {
016
017    /**
018     * Receive notification of a change in the value of a {@link SimpleField} in an object.
019     *
020     * <p>
021     * Notifications are only delivered when the set of referring objects is non-empty.
022     * </p>
023     *
024     * <p>
025     * Notifications are delivered in the same thread that made the change, before the outermost mutation operation returns.
026     * </p>
027     *
028     * @param tx associated transaction
029     * @param id the ID of the affected object (i.e., the object containing the field that changed)
030     * @param field the field that changed
031     * @param path path of reference fields (represented by storage IDs) that lead to {@code id}
032     * @param referrers all objects that (indirectly) refer to the affected object via the {@code path}
033     * @param oldValue the field's previous value
034     * @param newValue the field's new value
035     * @param <V> Java type for {@code field}'s values
036     */
037    <V> void onSimpleFieldChange(Transaction tx, ObjId id, SimpleField<V> field,
038      int[] path, NavigableSet<ObjId> referrers, V oldValue, V newValue);
039}
