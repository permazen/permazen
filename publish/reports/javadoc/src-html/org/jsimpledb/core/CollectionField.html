006package org.jsimpledb.core;
007
008import com.google.common.reflect.TypeToken;
009
010import java.util.Collection;
011import java.util.Collections;
012import java.util.Iterator;
013import java.util.List;
014import java.util.SortedSet;
015
016/**
017 * Superclass of fields with types assignable to {@link Collection}.
018 *
019 * @param <C> Java collection type
020 * @param <E> Java collection element type
021 */
022public abstract class CollectionField<C extends Collection<E>, E> extends ComplexField<C> {
023
024    public static final String ELEMENT_FIELD_NAME = "element";
025
026    final SimpleField<E> elementField;
027
028    /**
029     * Constructor.
030     *
031     * @param name the name of the field
032     * @param storageId field content storage ID
033     * @param typeToken Java type for the field's values
034     * @param schema schema version
035     * @param elementField this field's element sub-field
036     * @throws IllegalArgumentException if any parameter is null
037     * @throws IllegalArgumentException if {@code storageId} is non-positive
038     */
039    CollectionField(String name, int storageId, Schema schema, TypeToken<C> typeToken, SimpleField<E> elementField) {
040        super(name, storageId, schema, typeToken);
041        if (elementField == null)
042            throw new IllegalArgumentException("null elementField");
043        this.elementField = elementField;
044        assert this.elementField.parent == null;
045        this.elementField.parent = this;
046    }
047
048// Public methods
049
050    /**
051     * Get the element field.
052     *
053     * @return collection element field
054     */
055    public SimpleField<E> getElementField() {
056        return this.elementField;
057    }
058
059    @Override
060    public final List<SimpleField<E>> getSubFields() {
061        return Collections.<SimpleField<E>>singletonList(this.elementField);
062    }
063
064    @Override
065    public boolean hasDefaultValue(Transaction tx, ObjId id) {
066        return this.getValue(tx, id).isEmpty();
067    }
068
069    @Override
070    public abstract C getValue(Transaction tx, ObjId id);
071
072// Non-public methods
073
074    @Override
075    @SuppressWarnings("unchecked")
076    <F> Iterable<F> iterateSubField(Transaction tx, ObjId id, SimpleField<F> subField) {
077        if (subField == this.elementField)
078            return (Iterable<F>)this.getValue(tx, id);
079        throw new IllegalArgumentException("unknown sub-field");
080    }
081
082    @Override
083    abstract CollectionFieldStorageInfo<C, E> toStorageInfo();
084
085    @Override
086    void unreferenceRemovedObjectTypes(Transaction tx, ObjId id, ReferenceField subField, SortedSet<Integer> removedStorageIds) {
087        assert subField == this.elementField;
088        for (Iterator<?> i = this.getValueInternal(tx, id).iterator(); i.hasNext(); ) {
089            final ObjId ref = (ObjId)i.next();
090            if (ref != null && removedStorageIds.contains(ref.getStorageId()))
091                i.remove();
092        }
093    }
094}
