008package org.jsimpledb.change;
009
010/**
011 * Notification object that gets passed to {@link org.jsimpledb.annotation.OnChange &#64;OnChange}-annotated methods
012 * when a field changes.
013 *
014 * @param <T> the type of the object containing the changed field
015 */
016public abstract class FieldChange<T> {
017
018    private final T jobj;
019
020    /**
021     * Constructor.
022     *
023     * @param jobj Java object containing the field that changed
024     * @throws IllegalArgumentException if {@code jobj} is null
025     */
026    protected FieldChange(T jobj) {
027        if (jobj == null)
028            throw new IllegalArgumentException("null jobj");
029        this.jobj = jobj;
030    }
031
032    /**
033     * Get the Java model object containing the field that changed.
034     */
035    public T getObject() {
036        return this.jobj;
037    }
038
039    @Override
040    public boolean equals(Object obj) {
041        if (obj == this)
042            return true;
043        if (obj == null || obj.getClass() != this.getClass())
044            return false;
045        final FieldChange<?> that = (FieldChange<?>)obj;
046        return this.jobj.equals(that.jobj);
047    }
048
049    @Override
050    public int hashCode() {
051        return this.jobj.hashCode();
052    }
053}
