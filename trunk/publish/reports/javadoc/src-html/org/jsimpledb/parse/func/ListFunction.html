008package org.jsimpledb.parse.func;
009
010import com.google.common.collect.Lists;
011
012import java.util.Arrays;
013
014import org.jsimpledb.parse.ParseSession;
015import org.jsimpledb.parse.expr.ConstValue;
016import org.jsimpledb.parse.expr.Value;
017
018@Function
019public class ListFunction extends SimpleFunction {
020
021    public ListFunction() {
022        super("list", 0, Integer.MAX_VALUE);
023    }
024
025    @Override
026    public String getHelpSummary() {
027        return "creates a list of items from explicitly provided values";
028    }
029
030    @Override
031    public String getUsage() {
032        return "list(item, ...)";
033    }
034
035    @Override
036    protected Value apply(final ParseSession session, Value[] params) {
037        return new ConstValue(Lists.transform(Arrays.asList(params), new Value.GetFunction(session)));
038    }
039}
