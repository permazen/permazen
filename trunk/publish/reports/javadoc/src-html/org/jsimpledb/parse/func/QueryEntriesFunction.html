008package org.jsimpledb.parse.func;
009
010import java.util.NavigableMap;
011import java.util.NavigableSet;
012
013import org.jsimpledb.JTransaction;
014import org.jsimpledb.core.Field;
015import org.jsimpledb.core.FieldSwitchAdapter;
016import org.jsimpledb.core.ListField;
017import org.jsimpledb.core.ListIndexEntry;
018import org.jsimpledb.core.MapField;
019import org.jsimpledb.core.MapKeyIndexEntry;
020import org.jsimpledb.core.MapValueIndexEntry;
021import org.jsimpledb.core.Transaction;
022import org.jsimpledb.parse.IndexedFieldParser;
023import org.jsimpledb.parse.ParseContext;
024import org.jsimpledb.parse.ParseException;
025import org.jsimpledb.parse.ParseSession;
026import org.jsimpledb.parse.SpaceParser;
027import org.jsimpledb.parse.expr.Value;
028
029@Function
030public class QueryEntriesFunction extends AbstractFunction {
031
032    private final SpaceParser spaceParser = new SpaceParser();
033
034    public QueryEntriesFunction() {
035        super("queryEntries");
036    }
037
038    @Override
039    public String getHelpSummary() {
040        return "queries an indexed list or map field's element, key, or value index entries";
041    }
042
043    @Override
044    public String getUsage() {
045        return "queryEntries(type.field.subfield)";
046    }
047
048    @Override
049    public String getHelpDetail() {
050        return "Queries an indexed list or map field and returns a mapping from list element or map key or value to"
051          + " the set of ListIndexEntry objects (for list fields) or MapKeyIndexEntry or MapValueIndexEntry objects"
052          + " (for map fields) corresponding to objects having that value in the field.";
053    }
054
055    @Override
056    public Object parseParams(final ParseSession session, final ParseContext ctx, final boolean complete) {
057
058        // Parse indexed field parameter
059        if (ctx.tryLiteral(")"))
060            throw new ParseException(ctx, "indexed field parameter required");
061        final IndexedFieldParser.Result result = new IndexedFieldParser().parse(session, ctx, complete);
062
063        // Check field type
064        final Query query = result.getParentField() != null ? result.getParentField().visit(new FieldSwitchAdapter<Query>() {
065            @Override
066            public <E> Query caseListField(ListField<E> field) {
067                return new Query(field.getStorageId(), SubField.LIST_ELEMENT);
068            }
069            @Override
070            public <K, V> Query caseMapField(MapField<K, V> field) {
071                return new Query(field.getStorageId(),
072                  result.getField().getStorageId() == field.getKeyField().getStorageId() ? SubField.MAP_KEY : SubField.MAP_VALUE);
073            }
074            @Override
075            protected <T> Query caseField(Field<T> field) {
076                return null;
077            }
078        }) : null;
079        if (query == null) {
080            throw new ParseException(ctx, "indexed field `" + result.getField().getName()
081              + "' is not a sub-field of a list or map field");
082        }
083
084        // Finish parse
085        ctx.skipWhitespace();
086        if (!ctx.tryLiteral(")"))
087            throw new ParseException(ctx, "expected `)'").addCompletion(") ");
088
089        // Done
090        return query;
091    }
092
093    @Override
094    public Value apply(ParseSession session, Object params) {
095        final Query query = (Query)params;
096        return new Value(null) {
097            @Override
098            public Object get(ParseSession session) {
099                return query.query(session);
100            }
101        };
102    }
103
104// Query
105
106    private static class Query {
107
108        private final int storageId;
109        private final SubField subField;
110
111        Query(int storageId, SubField subField) {
112            this.storageId = storageId;
113            this.subField = subField;
114        }
115
116        public NavigableMap<?, ? extends NavigableSet<?>> query(ParseSession session) {
117            return session.hasJSimpleDB() ?
118              this.subField.query(JTransaction.getCurrent(), this.storageId) :
119              this.subField.query(session.getTransaction(), this.storageId);
120        }
121    }
122
123// Sub-fields with index entries
124
125    private enum SubField {
126        LIST_ELEMENT() {
127
128            @Override
129            NavigableMap<?, NavigableSet<ListIndexEntry>> query(Transaction tx, int storageId) {
130                return tx.queryListFieldEntries(storageId);
131            }
132
133            @Override
134            NavigableMap<?, NavigableSet<org.jsimpledb.ListIndexEntry<?>>> query(JTransaction jtx, int storageId) {
135                return jtx.queryListFieldEntries(storageId, null);
136            }
137        },
138        MAP_KEY() {
139
140            @Override
141            NavigableMap<?, NavigableSet<MapKeyIndexEntry<?>>> query(Transaction tx, int storageId) {
142                return tx.queryMapFieldKeyEntries(storageId);
143            }
144
145            @Override
146            NavigableMap<?, NavigableSet<org.jsimpledb.MapKeyIndexEntry<?, ?>>> query(JTransaction jtx, int storageId) {
147                return jtx.queryMapFieldKeyEntries(storageId, null);
148            }
149        },
150        MAP_VALUE() {
151
152            @Override
153            NavigableMap<?, NavigableSet<MapValueIndexEntry<?>>> query(Transaction tx, int storageId) {
154                return tx.queryMapFieldValueEntries(storageId);
155            }
156
157            @Override
158            NavigableMap<?, NavigableSet<org.jsimpledb.MapValueIndexEntry<?, ?>>> query(JTransaction jtx, int storageId) {
159                return jtx.queryMapFieldValueEntries(storageId, null);
160            }
161        };
162
163        // Core API
164        abstract NavigableMap<?, ? extends NavigableSet<?>> query(Transaction tx, int storageId);
165
166        // JSimpleDB API
167        abstract NavigableMap<?, ? extends NavigableSet<?>> query(JTransaction jtx, int storageId);
168    }
169}
