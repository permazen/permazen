008package org.jsimpledb.kv.mvcc;
009
010import java.util.ArrayList;
011import java.util.Iterator;
012import java.util.Map;
013import java.util.TreeMap;
014
015import org.jsimpledb.kv.KVDatabase;
016import org.jsimpledb.kv.KVStore;
017import org.jsimpledb.kv.KVTransaction;
018import org.jsimpledb.kv.KVTransactionException;
019import org.jsimpledb.kv.RetryTransactionException;
020import org.slf4j.Logger;
021import org.slf4j.LoggerFactory;
022
023/**
024 * Multi-version conccurrency control (MVCC) database using snapshot views of an underlying {@link AtomicKVStore}.
025 *
026 * <p>
027 * This class provides a {@link KVDatabase} view of an underlying {@link AtomicKVStore},
028 * supporting concurrent transactions via MVCC and linearizable ACID semantics.
029 * </p>
030 *
031 * <p>
032 * Instances implement a simple optimistic locking scheme for MVCC using {@link AtomicKVStore#snapshot}. Concurrent transactions
033 * do not contend for any locks until commit time. During each transaction, reads are noted and derive from the snapshot,
034 * while writes are batched up. At commit time, if any other transaction has committed writes since the transaction's
035 * snapshot was created, and any of those writes {@linkplain Reads#isConflict conflict} with any of the committing
036 * transaction's reads, a {@link RetryTransactionException} is thrown. Otherwise, the transaction is committed and its
037 * writes are applied.
038 * </p>
039 *
040 * <p>
041 * Each outstanding transaction's mutations are batched up in memory using a {@link Writes} instance. Therefore, the
042 * transaction load supported by this class is limited to what can fit in memory.
043 * </p>
044 */
045public class SnapshotKVDatabase implements KVDatabase {
046
047// Locking order: (1) SnapshotKVTransaction, (2) SnapshotKVDatabase
048
049    protected final Logger log = LoggerFactory.getLogger(this.getClass());
050
051    private final TreeMap<Long, SnapshotVersion> versionInfoMap = new TreeMap<>();
052
053    private AtomicKVStore kvstore;
054    private long currentVersion;
055
056// Constructors
057
058    /**
059     * Default constructor.
060     *
061     * <p>
062     * The underlying key/value store must still be configured before creating any transactions.
063     */
064    public SnapshotKVDatabase() {
065    }
066
067    /**
068     * Constructor.
069     *
070     * @param kvstore underlying key/value store
071     */
072    public SnapshotKVDatabase(AtomicKVStore kvstore) {
073        this.kvstore = kvstore;
074    }
075
076// Properties
077
078    /**
079     * Get the underlying {@link AtomicKVStore}.
080     *
081     * @return underlying key/value store
082     */
083    protected synchronized AtomicKVStore getKVStore() {
084        return this.kvstore;
085    }
086
087    /**
088     * Configure the underlying {@link AtomicKVStore}.
089     *
090     * <p>
091     * Required property; must be configured before creating any transactions.
092     *
093     * @param kvstore underlying key/value store
094     * @throws IllegalStateException if there are any transactions open
095     */
096    protected synchronized void setKVStore(AtomicKVStore kvstore) {
097        if (!this.versionInfoMap.isEmpty())
098            throw new IllegalStateException("transactions exist");
099        this.kvstore = kvstore;
100    }
101
102// KVDatabase
103
104    /**
105     * Create a new transaction.
106     *
107     * @throws IllegalStateException if no {@link AtomicKVStore} is configured
108     */
109    @Override
110    public synchronized KVTransaction createTransaction() {
111
112        // Sanity check
113        if (this.kvstore == null)
114            throw new IllegalStateException("no KVSTore configured");
115
116        // Get info for the current version
117        final SnapshotVersion versionInfo = this.getCurrentSnapshotVersion();
118
119        // Create the new transaction and associate it with the current version
120        final SnapshotKVTransaction tx = this.createSnapshotKVTransaction(versionInfo);
121        versionInfo.addOpenTransaction(tx);
122        if (this.log.isDebugEnabled())
123            this.log.debug("created new transaction " + tx);
124        if (this.log.isTraceEnabled())
125            this.log.trace("updated current version info: " + versionInfo);
126
127        // Done
128        return tx;
129    }
130
131// Subclass methods
132
133    /**
134     * Instantiate a new {@link SnapshotKVTransaction} instance.
135     *
136     * <p>
137     * The implementation in {@link SnapshotKVDatabase} just invokes the {@link SnapshotKVTransaction}
138     * constructor using {@code this} and {@code versionInfo}. Subclasses may want to override this method
139     * to create a more specific subclass.
140     *
141     * @param versionInfo associated snapshot info
142     * @return new transaction instance
143     * @throws KVTransactionException if an error occurs
144     */
145    protected SnapshotKVTransaction createSnapshotKVTransaction(SnapshotVersion versionInfo) {
146        return new SnapshotKVTransaction(this, versionInfo);
147    }
148
149    /**
150     * Forcibly close all outstanding transactions.
151     *
152     * <p>
153     * Can be used by subclasses during the shutdown sequence to ensure everything is properly cleaned up.
154     * To avoid a possible lock order reversal deadlock, this instance should <b>not</b> be locked when invoking this method.
155     */
156    protected void closeTransactions() {
157
158        // Grab all remaining open transactions
159        final ArrayList<SnapshotKVTransaction> openTransactions = new ArrayList<>();
160        synchronized (this) {
161            for (SnapshotVersion versionInfo : this.versionInfoMap.values())
162                openTransactions.addAll(versionInfo.getOpenTransactions());
163        }
164
165        // Close them (but not while holding my lock, to avoid lock order reversal)
166        for (SnapshotKVTransaction tx : openTransactions) {
167            try {
168                tx.rollback();
169            } catch (Throwable e) {
170                this.log.debug("caught exception closing open transaction during shutdown (ignoring)", e);
171            }
172        }
173    }
174
175    /**
176     * Log specified exception. Used to ensure exceptions are logged at the point they are thrown.
177     *
178     * @param e exception to log
179     * @return e
180     */
181    protected KVTransactionException logException(KVTransactionException e) {
182        if (this.log.isDebugEnabled())
183            this.log.debug("throwing exception for " + e.getTransaction() + ": " + e);
184        return e;
185    }
186
187    /**
188     * Wrap a {@link RuntimeException} as needed.
189     *
190     * <p>
191     * The implementation in {@link SnapshotKVDatabase} just returns {@code e}.
192     *
193     * @param tx transaction in which the exception occurred
194     * @param e original exception
195     * @return wrapped exception, or just {@code e}
196     */
197    protected RuntimeException wrapException(SnapshotKVTransaction tx, RuntimeException e) {
198        return e;
199    }
200
201// Package methods
202
203    /**
204     * Commit a transaction.
205     */
206    synchronized void commit(SnapshotKVTransaction tx) {
207        try {
208            this.doCommit(tx);
209        } finally {
210            this.cleanupTransaction(tx);
211        }
212    }
213
214    /**
215     * Rollback a transaction.
216     */
217    synchronized void rollback(SnapshotKVTransaction tx) {
218        if (this.log.isDebugEnabled())
219            this.log.debug("rolling back transaction " + tx);
220        this.cleanupTransaction(tx);
221    }
222
223// Internal methods
224
225    private synchronized void doCommit(SnapshotKVTransaction tx) {
226
227        // Get current and transaction's version info
228        final SnapshotVersion currentSnapshotVersion = this.getCurrentSnapshotVersion();
229        final SnapshotVersion transactionSnapshotVersion = tx.getSnapshotVersion();
230        final long transactionVersion = transactionSnapshotVersion.getVersion();
231        assert this.currentVersion - transactionVersion >= 0;
232        assert transactionSnapshotVersion.getOpenTransactions().contains(tx);
233
234        // Debug
235        if (this.log.isDebugEnabled()) {
236            this.log.debug("committing transaction " + tx + " based on version "
237              + transactionVersion + " (current version is " + this.currentVersion + ")");
238        }
239
240        // Check whether transaction has been forcibly killed somehow
241        if (!transactionSnapshotVersion.getOpenTransactions().contains(tx))
242            throw this.logException(new RetryTransactionException(tx, "transaction has been forcibly invalidated"));
243
244        // Get transaction reads & writes
245        final Reads transactionReads = tx.getMutableView().getReads();
246        final Writes transactionWrites = tx.getMutableView().getWrites();
247
248        // If the current version has advanced past the transaction's version, check for conflicts from intervening commits
249        for (long version = transactionVersion; version != this.currentVersion; version++) {
250            final SnapshotVersion committedSnapshotVersion = this.versionInfoMap.get(version);
251            final Writes committedWrites = committedSnapshotVersion.getCommittedWrites();
252            final boolean conflict = transactionReads.isConflict(committedWrites);
253            if (this.log.isDebugEnabled()) {
254                this.log.debug("ordering " + tx + " after writes in version " + version + " results in "
255                  + (conflict ? "conflict" : "no conflict"));
256                if (this.log.isTraceEnabled())
257                    this.log.trace("transaction reads: {} committed writes: {}", transactionReads, committedWrites);
258            }
259            if (conflict) {
260                throw this.logException(new RetryTransactionException(tx, "transaction is based on MVCC version "
261                  + transactionSnapshotVersion.getVersion() + " but the transaction committed at MVCC version "
262                  + version + " contains conflicting writes"));
263            }
264        }
265
266        // Atomically apply the transaction's mutations
267        if (this.log.isDebugEnabled())
268            this.log.debug("applying mutations of " + tx + " to SnapshotMVCC database");
269        this.kvstore.mutate(transactionWrites, true);
270
271        // Record transaction's writes for this version
272        currentSnapshotVersion.setCommittedWrites(transactionWrites);
273
274        // Advance to the next MVCC version
275        if (this.log.isDebugEnabled())
276            this.log.debug("updating current version from " + this.currentVersion + " -> " + (this.currentVersion + 1));
277        this.currentVersion++;
278    }
279
280    private void cleanupTransaction(SnapshotKVTransaction tx) {
281
282        // Debug
283        assert Thread.holdsLock(this);
284        if (this.log.isTraceEnabled())
285            this.log.trace("cleaning up transaction " + tx);
286
287        // Remove open transaction from version
288        tx.getSnapshotVersion().removeOpenTransaction(tx);
289
290        // Discard all versions older than all remaining open transactions
291        for (Iterator<Map.Entry<Long, SnapshotVersion>> i = this.versionInfoMap.entrySet().iterator(); i.hasNext(); ) {
292            final SnapshotVersion versionInfo = i.next().getValue();
293            if (!versionInfo.getOpenTransactions().isEmpty())
294                break;
295            if (this.log.isDebugEnabled())
296                this.log.debug("discarding obsolete version " + versionInfo);
297            final KVStore snapshot = versionInfo.getSnapshot();
298            if (snapshot instanceof AutoCloseable) {
299                try {
300                    ((AutoCloseable)snapshot).close();
301                } catch (Exception e) {
302                    // ignore
303                }
304            }
305            i.remove();
306        }
307    }
308
309    // Get SnapshotVersion for the current MVCC version, creating on demand if necessary
310    private SnapshotVersion getCurrentSnapshotVersion() {
311        SnapshotVersion versionInfo = this.versionInfoMap.get(this.currentVersion);
312        if (versionInfo == null) {
313            versionInfo = new SnapshotVersion(this.currentVersion, this.kvstore.snapshot());
314            this.versionInfoMap.put(this.currentVersion, versionInfo);
315            if (this.log.isTraceEnabled())
316                this.log.trace("created new version " + versionInfo);
317        }
318        return versionInfo;
319    }
320}
