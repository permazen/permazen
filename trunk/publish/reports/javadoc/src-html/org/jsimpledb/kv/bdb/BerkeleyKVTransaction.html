036/**
037 * Oracle Berkeley DB Java Edition {@link KVTransaction} implementation.
038 */
039public class BerkeleyKVTransaction extends AbstractCountingKVStore implements KVTransaction, Closeable {
040
041/*
042
043    The Berkeley DB presents a small challenge because all Cursor objects must be closed before
044    committing a transaction. Because these underly the Iterators returned by getRange(), and
045    Iterators do not normally need to be "closed", we have to track the unclosed Cursor objects
046    ourselves to ensure they eventually do get closed.
047
048    The naive approach is to track the unclosed Cursors in an internal set so we can close them
049    prior to transaction commit. However, this presents a memory-leak situation, e.g., in the case
050    of a transaction that creates and then discards zillions of Iterators (and therefore Cursors),
051    because we will still hold references to them, preventing them from being GC'd. The solution is
052    to register a WeakReference to every created Iterator on an internal ReferenceQueue, and poll()
053    this queue at every opportunity. When we get a reference back, close the corresponding Cursor
054    and remove it from the internal set.
055
056*/
057
058// Note: locking order: (1) BerkeleyKVTransaction, (2) BerkeleyKVDatabase
059
060    private static final byte[] MIN_KEY = new byte[0];                      // minimum possible key (inclusive)
061    private static final byte[] MAX_KEY = new byte[] { (byte)0xff };        // maximum possible key (exclusive)
062
063    private final Logger log = LoggerFactory.getLogger(this.getClass());
064    private final BerkeleyKVDatabase store;
065    private final Transaction tx;
066    private final CursorConfig cursorConfig = new CursorConfig().setNonSticky(true);
067    private final HashSet<Cursor> unclosedCursors = new HashSet<>();                        // unclosed Cursors are tracked here
068    private final ReferenceQueue<CursorIterator> iteratorQueue = new ReferenceQueue<>();    // discarded Iterators appear here
069
070    private boolean closed;

077        assert tx != null;
078        this.store = store;
079        this.tx = tx;

103// KVStore

105    @Override
106    public synchronized byte[] get(byte[] key) {
107        if (this.closed)
108            throw new StaleTransactionException(this);
109        this.closeUnreferencedIterators();
110        if (key.length > 0 && key[0] == (byte)0xff)
111            throw new IllegalArgumentException("key starts with 0xff");
112        final DatabaseEntry value = new DatabaseEntry();
113        try {
114            final OperationStatus status = this.store.getDatabase().get(this.tx, new DatabaseEntry(key), value, null);
115            switch (status) {
116            case SUCCESS:
117                return value.getData();
118            case NOTFOUND:
119                return null;
120            default:
121                throw this.weirdStatus(status, "get");
122            }
123        } catch (DatabaseException e) {
124            throw this.wrapException(e);
125        }
126    }
127
128    @Override
129    public KVPair getAtLeast(byte[] minKey) {
130        try (CursorIterator i = this.getRange(minKey, null, false)) {
131            return i.hasNext() ? i.next() : null;
132        }
133    }
134
135    @Override
136    public KVPair getAtMost(byte[] maxKey) {
137        try (CursorIterator i = this.getRange(null, maxKey, true)) {
138            return i.hasNext() ? i.next() : null;
139        }
140    }
141
142    @Override
143    public synchronized CursorIterator getRange(byte[] minKey, byte[] maxKey, boolean reverse) {
144        if (this.closed)
145            throw new StaleTransactionException(this);
146        this.closeUnreferencedIterators();
147        final Cursor cursor;
148        try {
149            cursor = this.store.getDatabase().openCursor(this.tx, this.cursorConfig);
150        } catch (DatabaseException e) {
151            throw this.wrapException(e);
152        }
153        return new CursorIterator(cursor, minKey, maxKey, reverse);
154    }
155
156    @Override
157    public synchronized void put(byte[] key, byte[] value) {
158        if (this.closed)
159            throw new StaleTransactionException(this);
160        this.closeUnreferencedIterators();
161        if (key.length > 0 && key[0] == (byte)0xff)
162            throw new IllegalArgumentException("key starts with 0xff");
163        try {
164            this.store.getDatabase().put(this.tx, new DatabaseEntry(key), new DatabaseEntry(value));
165        } catch (DatabaseException e) {
166            throw this.wrapException(e);
167        }
168    }
169
170    @Override
171    public synchronized void remove(byte[] key) {
172        if (this.closed)
173            throw new StaleTransactionException(this);
174        this.closeUnreferencedIterators();
175        if (key.length > 0 && key[0] == (byte)0xff)
176            throw new IllegalArgumentException("key starts with 0xff");
177        try {
178            this.store.getDatabase().delete(this.tx, new DatabaseEntry(key));
179        } catch (DatabaseException e) {
180            throw this.wrapException(e);
181        }
182    }
183
184    @Override
185    public void removeRange(byte[] minKey, byte[] maxKey) {
186        try (CursorIterator i = this.getRange(minKey, maxKey, false)) {
187            while (i.hasNext()) {
188                i.next();
189                i.remove();
190            }
191        }
192    }
193
194    @Override
195    public synchronized void commit() {
196        if (this.closed)
197            throw new StaleTransactionException(this);
198        this.close();
199        try {
200            this.tx.commit();
201        } catch (DatabaseException e) {
202            throw this.wrapException(e);
203        }
204    }
205
206    @Override
207    public synchronized void rollback() {
208        if (this.closed)
209            throw new StaleTransactionException(this);
210        this.close();
211        try {
212            this.tx.abort();
213        } catch (DatabaseException e) {
214            throw this.wrapException(e);
215        }
216    }
217
218// Closeable
219
220    @Override
221    public synchronized void close() {
222
223        // Already closed?
224        if (this.closed)
225            return;
226        this.closed = true;
227
228        // Close all unclosed cursors
229        for (Cursor cursor : this.unclosedCursors) {
230            try {
231                cursor.close();
232            } catch (Throwable e) {
233                this.log.debug("caught exception closing unreferenced cursor (ignoring)", e);
234            }
235        }
236        this.unclosedCursors.clear();
237
238        // Remove this transction from database
239        this.store.removeTransaction(this);
240    }
241
242    private /*synchronized*/ void closeUnreferencedIterators() {
243        assert Thread.holdsLock(this);
244        for (IteratorReference ref; (ref = (IteratorReference)this.iteratorQueue.poll()) != null; ) {
245            final Cursor cursor = ref.getCursor();
246            try {
247                cursor.close();
248            } catch (Throwable e) {
249                this.log.debug("caught exception closing unreferenced cursor (ignoring)", e);
250            }
251            this.unclosedCursors.remove(cursor);
252        }
253    }
254
255// Object
256
257    @Override
258    protected void finalize() throws Throwable {
259        try {
260            this.close();
261        } finally {
262            super.finalize();
263        }
264    }

268    /**
269     * Wrap the given {@link DatabaseException} in the appropriate {@link KVTransactionException}.
270     *
271     * @param e Berkeley database exception
272     * @return appropriate {@link KVTransactionException} with chained exception {@code e}
273     * @throws NullPointerException if {@code e} is null
274     */
275    public KVTransactionException wrapException(DatabaseException e) {
276        if (e instanceof LockConflictException)
277            return new RetryTransactionException(this, e);
278        return new KVTransactionException(this, e);
279    }

281    private KVTransactionException weirdStatus(OperationStatus status, String methodName) {
282        return new KVTransactionException(BerkeleyKVTransaction.this, "unexpected status " + status + " from " + methodName + "()");
283    }

285// CursorIterator

287    /**
288     * {@link Iterator} implementation used by {@link BerkeleyKVTransaction#getRange BerkeleyKVTransaction.getRange()}.
289     *
290     * <p>
291     * Instances implement {@link Closeable}.
292     * </p>
293     */
294    public final class CursorIterator implements Iterator<KVPair>, Closeable {
295
296        private final Cursor cursor;
297        private final byte[] minKey;
298        private final byte[] maxKey;
299        private final boolean reverse;
300
301        private KVPair nextPair;
302        private byte[] removeKey;
303        private boolean canRemoveWithCursor;
304        private boolean completed;
305        private boolean initialized;
306
307        CursorIterator(Cursor cursor, byte[] minKey, byte[] maxKey, boolean reverse) {
308            assert Thread.holdsLock(BerkeleyKVTransaction.this);
309            assert cursor != null;
310            if (minKey != null && maxKey != null && ByteUtil.compare(minKey, maxKey) > 0)
311                throw new IllegalArgumentException("minKey > maxKey");
312            this.cursor = cursor;
313            this.minKey = minKey != null ? ByteUtil.min(minKey, BerkeleyKVTransaction.MAX_KEY) : BerkeleyKVTransaction.MIN_KEY;
314            this.maxKey = maxKey != null ? ByteUtil.min(maxKey, BerkeleyKVTransaction.MAX_KEY) : BerkeleyKVTransaction.MAX_KEY;
315            this.reverse = reverse;
316
317            // Make sure we eventually close the cursor, prior to closing the transaction
318            new IteratorReference(this, this.cursor, BerkeleyKVTransaction.this.iteratorQueue);
319            BerkeleyKVTransaction.this.unclosedCursors.add(this.cursor);
320        }
321
322    // Iterator
323
324        @Override
325        public synchronized boolean hasNext() {
326            return this.findNext();
327        }
328
329        @Override
330        public synchronized KVPair next() {
331            if (!this.findNext())
332                throw new NoSuchElementException();
333            assert this.nextPair != null;
334            final KVPair result = this.nextPair;
335            this.removeKey = result.getKey().clone();
336            this.canRemoveWithCursor = true;
337            this.nextPair = null;
338            return result;
339        }
340
341        @Override
342        public synchronized void remove() {
343            if (BerkeleyKVTransaction.this.closed)
344                throw new StaleTransactionException(BerkeleyKVTransaction.this);
345            if (this.removeKey == null)
346                throw new IllegalStateException();
347            try {
348                final OperationStatus status = this.canRemoveWithCursor ? this.cursor.delete() :
349                  this.cursor.getDatabase().delete(BerkeleyKVTransaction.this.tx, new DatabaseEntry(this.removeKey));
350                switch (status) {
351                case SUCCESS:
352                case KEYEMPTY:
353                    break;
354                default:
355                    throw BerkeleyKVTransaction.this.weirdStatus(status, "delete");
356                }
357            } catch (DatabaseException e) {
358                throw BerkeleyKVTransaction.this.wrapException(e);
359            }
360            this.removeKey = null;
361        }
362
363    // Closeable
364
365        @Override
366        public void close() {
367            try {
368                this.cursor.close();
369            } catch (Throwable e) {
370                BerkeleyKVTransaction.this.log.debug("caught exception closing iterator cursor (ignoring)", e);
371            }
372        }
373
374    // Object
375
376        @Override
377        protected void finalize() throws Throwable {
378            try {
379                this.close();
380            } finally {
381                super.finalize();
382            }
383        }
384
385    // Internal methods
386
387        @SuppressWarnings("fallthrough")
388        private /*synchronized*/ boolean findNext() {
389            assert Thread.holdsLock(this);
390            if (BerkeleyKVTransaction.this.closed)
391                throw new StaleTransactionException(BerkeleyKVTransaction.this);
392            if (!this.initialized)
393                this.initialize();
394            assert this.initialized;
395            if (this.nextPair != null)
396                return true;
397            if (this.completed)
398                return false;
399            final DatabaseEntry key = new DatabaseEntry();
400            final DatabaseEntry value = new DatabaseEntry();
401            this.canRemoveWithCursor = false;
402            try {
403                final OperationStatus status = this.reverse ?
404                  this.cursor.getPrev(key, value, null) : this.cursor.getNext(key, value, null);
405                switch (status) {
406                case SUCCESS:
407                    final byte[] keyData = key.getData();
408                    if (this.reverse ? ByteUtil.compare(keyData, this.minKey) >= 0 : ByteUtil.compare(keyData, this.maxKey) < 0) {
409                        this.nextPair = new KVPair(keyData, value.getData());
410                        return true;
411                    }
412                    // FALLTHROUGH
413                case NOTFOUND:
414                    this.completed = true;
415                    return false;
416                default:
417                    throw BerkeleyKVTransaction.this.weirdStatus(status, this.reverse ? "getPrev" : "getNext");
418                }
419            } catch (DatabaseException e) {
420                throw BerkeleyKVTransaction.this.wrapException(e);
421            }
422        }
423
424        // We initialize on demand from findNext()
425        @SuppressWarnings("fallthrough")
426        private /*synchronized*/ void initialize() {
427            assert Thread.holdsLock(this);
428            assert !this.initialized;
429            assert !this.completed;
430            assert this.nextPair == null;
431            if (reverse) {
432
433                // Search for maxKey; thereafter, a call to getPrev() will return the first element in the reverse iteration.
434                // We don't care whether maxKey is found or not, we are just positioning the cursor.
435                try {
436                    final OperationStatus status = this.cursor.getSearchKey(
437                      new DatabaseEntry(this.maxKey), new DatabaseEntry(), null);
438                    switch (status) {
439                    case SUCCESS:
440                    case NOTFOUND:
441                        break;
442                    default:
443                        throw BerkeleyKVTransaction.this.weirdStatus(status, "getSearchKey");
444                    }
445                } catch (DatabaseException e) {
446                    throw BerkeleyKVTransaction.this.wrapException(e);
447                }
448            } else if (this.minKey.length > 0) {    // if minKey has zero length, no need to initialize (getNext() will just work)
449                final DatabaseEntry key = new DatabaseEntry(this.minKey);
450                final DatabaseEntry value = new DatabaseEntry();
451                try {
452                    final OperationStatus status = this.cursor.getSearchKeyRange(key, value, null);
453                    switch (status) {
454                    case SUCCESS:
455                        final byte[] keyData = key.getData();
456                        if (ByteUtil.compare(keyData, this.maxKey) < 0) {
457                            this.nextPair = new KVPair(keyData, value.getData());
458                            break;
459                        }
460                        // FALLTHROUGH
461                    case NOTFOUND:
462                        this.completed = true;
463                        break;
464                    default:
465                        throw BerkeleyKVTransaction.this.weirdStatus(status, "getSearchKeyRange");
466                    }
467                } catch (DatabaseException e) {
468                    throw BerkeleyKVTransaction.this.wrapException(e);
469                }
470            }
471            this.initialized = true;
472        }
473    }

479        private final Cursor cursor;
