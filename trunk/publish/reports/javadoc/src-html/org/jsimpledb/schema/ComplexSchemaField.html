008package org.jsimpledb.schema;
009
010import java.util.Map;
011
012import javax.xml.namespace.QName;
013import javax.xml.stream.XMLStreamException;
014import javax.xml.stream.XMLStreamReader;
015import javax.xml.stream.XMLStreamWriter;
016
017import org.jsimpledb.core.InvalidSchemaException;
018
019/**
020 * A complex field in one version of a {@link SchemaObject}.
021 */
022public abstract class ComplexSchemaField extends SchemaField {
023
024    @Override
025    public void validate() {
026        super.validate();
027        if (this.getName() == null || this.getName().length() == 0)
028            throw new InvalidSchemaException(this + " must specify a name");
029        for (Map.Entry<String, SimpleSchemaField> entry : this.getSubFields().entrySet()) {
030            final String subFieldName = entry.getKey();
031            final SimpleSchemaField subField = entry.getValue();
032            if (subField == null)
033                throw new InvalidSchemaException("invalid " + this + ": missing sub-field `" + subFieldName + "'");
034            if (subField.getName() != null)
035                throw new InvalidSchemaException("sub-" + subField + " of " + this + " must not specify a name");
036            subField.validate();
037        }
038    }
039
040    public abstract Map<String, SimpleSchemaField> getSubFields();
041
042    @Override
043    boolean isCompatibleWithInternal(AbstractSchemaItem that0) {
044        final ComplexSchemaField that = (ComplexSchemaField)that0;
045        if (!this.getSubFields().keySet().equals(that.getSubFields().keySet()))
046            return false;
047        for (String subFieldName : this.getSubFields().keySet()) {
048            final SimpleSchemaField thisSubField = this.getSubFields().get(subFieldName);
049            final SimpleSchemaField thatSubField = that.getSubFields().get(subFieldName);
050            if (!thisSubField.isCompatibleWith(thatSubField))
051                return false;
052        }
053        return true;
054    }
055
056    SimpleSchemaField readSubField(XMLStreamReader reader) throws XMLStreamException {
057        this.expect(reader, false, REFERENCE_FIELD_TAG, SIMPLE_FIELD_TAG);
058        SimpleSchemaField field;
059        if (reader.getName().equals(REFERENCE_FIELD_TAG))
060            field = new ReferenceSchemaField();
061        else if (reader.getName().equals(SIMPLE_FIELD_TAG))
062            field = new SimpleSchemaField();
063        else
064            throw new RuntimeException("internal error");
065        field.readXML(reader);
066        return field;
067    }
068
069    @Override
070    void writeXML(XMLStreamWriter writer) throws XMLStreamException {
071        final QName tag = this.getXMLTag();
072        writer.writeStartElement(tag.getNamespaceURI(), tag.getLocalPart());
073        this.writeAttributes(writer);
074        for (SimpleSchemaField subField : this.getSubFields().values())
075            subField.writeXML(writer);
076        writer.writeEndElement();
077    }
078
079    abstract QName getXMLTag();
080
081// Cloneable
082
083    @Override
084    public ComplexSchemaField clone() {
085        return (ComplexSchemaField)super.clone();
086    }
087}
